93  protected override void StartMissionSpecific(Level level)
94  {
95  if (items.Any())
96  {
97 #if DEBUG
98  throw new Exception($"items.Count > 0 ({items.Count})");
99 #else
100  DebugConsole.AddWarning("Item list was not empty at the start of a nest mission. The mission instance may not have been ended correctly on previous rounds.");
101  items.Clear();
102 #endif
103  }
104 
105  if (!IsClient)
106  {
107  //ruin/cave/wreck items are allowed to spawn close to the sub
108  float minDistance = spawnPositionType == Level.PositionType.Ruin || spawnPositionType == Level.PositionType.Cave || spawnPositionType == Level.PositionType.Wreck ?
109  0.0f : Level.Loaded.Size.X * 0.3f;
110 
111  nestPosition = Level.Loaded.GetRandomItemPos(spawnPositionType, 100.0f, minDistance, 30.0f);
112  List<GraphEdge> spawnEdges = new List<GraphEdge>();
113  if (spawnPositionType == Level.PositionType.Cave)
114  {
115  Level.Cave closestCave = null;
116  float closestCaveDist = float.PositiveInfinity;
117  foreach (var cave in Level.Loaded.Caves)
118  {
119  float dist = Vector2.DistanceSquared(nestPosition, cave.Area.Center.ToVector2());
120  if (dist < closestCaveDist)
121  {
122  closestCave = cave;
123  closestCaveDist = dist;
124  }
125  }
126  if (closestCave != null)
127  {
128  closestCave.DisplayOnSonar = true;
129  SpawnNestObjects(level, closestCave);
130 #if SERVER
131  selectedCave = closestCave;
132 #endif
133  }
134  var nearbyCells = Level.Loaded.GetCells(nestPosition, searchDepth: 3);
135  if (nearbyCells.Any())
136  {
137  List<GraphEdge> validEdges = new List<GraphEdge>();
138  foreach (var edge in nearbyCells.SelectMany(c => c.Edges))
139  {
140  if (!edge.NextToCave || !edge.IsSolid) { continue; }
141  if (Level.Loaded.ExtraWalls.Any(w => w.IsPointInside(edge.Center + edge.GetNormal(edge.Cell1 ?? edge.Cell2) * 100.0f))) { continue; }
142  validEdges.Add(edge);
143  }
144 
145  if (validEdges.Any())
146  {
147  spawnEdges.AddRange(validEdges.Where(e => MathUtils.LineSegmentToPointDistanceSquared(e.Point1.ToPoint(), e.Point2.ToPoint(), nestPosition.ToPoint()) < itemSpawnRadius * itemSpawnRadius).Distinct());
148  }
149  //no valid edges found close enough to the nest position, find the closest one
150  if (!spawnEdges.Any())
151  {
152  GraphEdge closestEdge = null;
153  float closestDistSqr = float.PositiveInfinity;
154  foreach (var edge in nearbyCells.SelectMany(c => c.Edges))
155  {
156  if (!edge.NextToCave || !edge.IsSolid) { continue; }
157  float dist = Vector2.DistanceSquared(edge.Center, nestPosition);
158  if (dist < closestDistSqr)
159  {
160  closestEdge = edge;
161  closestDistSqr = dist;
162  }
163  }
164  if (closestEdge != null)
165  {
166  spawnEdges.Add(closestEdge);
167  itemSpawnRadius = Math.Max(itemSpawnRadius, (float)Math.Sqrt(closestDistSqr) * 1.5f);
168  }
169  }
170  }
171  }
172 
173  foreach (var subElement in itemConfig.Elements())
174  {
175  string itemIdentifier = subElement.GetAttributeString("identifier", "");
176  if (!(MapEntityPrefab.Find(null, itemIdentifier) is ItemPrefab itemPrefab))
177  {
178  DebugConsole.ThrowError("Couldn't spawn item for nest mission: item prefab \"" + itemIdentifier + "\" not found");
179  continue;
180  }
181 
182  Vector2 spawnPos = nestPosition;
183  float rotation = 0.0f;
184  if (spawnEdges.Any())
185  {
186  var edge = spawnEdges.GetRandom(Rand.RandSync.ServerAndClient);
187  spawnPos = Vector2.Lerp(edge.Point1, edge.Point2, Rand.Range(0.1f, 0.9f, Rand.RandSync.ServerAndClient));
188  Vector2 normal = Vector2.UnitY;
189  if (edge.Cell1 != null && edge.Cell1.CellType == CellType.Solid)
190  {
191  normal = edge.GetNormal(edge.Cell1);
192  }
193  else if (edge.Cell2 != null && edge.Cell2.CellType == CellType.Solid)
194  {
195  normal = edge.GetNormal(edge.Cell2);
196  }
197  spawnPos += normal * 10.0f;
198  rotation = MathUtils.VectorToAngle(normal) - MathHelper.PiOver2;
199  }
200 
201  var item = new Item(itemPrefab, spawnPos, null);
202  item.body.FarseerBody.BodyType = BodyType.Kinematic;
203  item.body.SetTransformIgnoreContacts(item.body.SimPosition, rotation);
204  item.FindHull();
205  items.Add(item);
206 
207  var statusEffectElement =
208  subElement.GetChildElement("StatusEffectOnApproach")
209  ?? subElement.GetChildElement("statuseffectonapproach");
210  if (statusEffectElement != null)
211  {
212  statusEffectOnApproach.Add(item, StatusEffect.Load(statusEffectElement, Prefab.Identifier.Value));
213  }
214  }
215  }

223  protected override void UpdateMissionSpecific(float deltaTime)
224  {
225  if (IsClient)
226  {
227  foreach (Item item in items)
228  {
229  if (item.ParentInventory != null && item.body != null) { item.body.FarseerBody.BodyType = BodyType.Dynamic; }
230  }
231  return;
232  }
233  switch (State)
234  {
235  case 0:
236  foreach (Item item in items)
237  {
238  if (item.ParentInventory != null && item.body != null) { item.body.FarseerBody.BodyType = BodyType.Dynamic; }
239  if (statusEffectOnApproach.ContainsKey(item))
240  {
241  foreach (Character character in Character.CharacterList)
242  {
243  if (character.IsPlayer && Vector2.DistanceSquared(nestPosition, character.WorldPosition) < approachItemsRadius * approachItemsRadius)
244  {
245  statusEffectOnApproach[item].Apply(statusEffectOnApproach[item].type, 1.0f, item, item);
246  statusEffectOnApproach.Remove(item);
247  break;
248  }
249  }
250  }
251  }
252  if (monsterPrefabs.Any())
253  {
254  foreach (Character character in Character.CharacterList)
255  {
256  if (character.IsPlayer && Vector2.DistanceSquared(nestPosition, character.WorldPosition) < monsterSpawnRadius * monsterSpawnRadius)
257  {
258  foreach (var monster in monsterPrefabs)
259  {
260  int amount = Rand.Range(monster.Item2.X, monster.Item2.Y + 1);
261  for (int i = 0; i < amount; i++)
262  {
263  Character.Create(monster.Item1.Identifier, nestPosition + Rand.Vector(100.0f), ToolBox.RandomSeed(8), createNetworkEvent: true);
264  }
265  }
266  monsterPrefabs.Clear();
267  break;
268  }
269  }
270  }
271 
272  //continue when all items are in the sub or destroyed
273  if (AllItemsDestroyedOrRetrieved()) { State = 1; }
274 
275  break;
276  case 1:
277  if (!Submarine.MainSub.AtEndExit && !Submarine.MainSub.AtStartExit) { return; }
278  State = 2;
279  break;
280  }
281  }

283  private bool AllItemsDestroyedOrRetrieved()
284  {
285  if (requireDelivery)
286  {
287  foreach (Item item in items)
288  {
289  Submarine parentSub = item.CurrentHull?.Submarine ?? item.GetRootInventoryOwner()?.Submarine;
290  if (parentSub?.Info?.Type == SubmarineType.Player) { continue; }
291  return false;
292  }
293  }
294  else
295  {
296  foreach (Item item in items)
297  {
298  if (item.Removed || item.Condition <= 0.0f) { continue; }
299  if (Vector2.Distance(item.WorldPosition, nestPosition) > Math.Max(itemSpawnRadius * 2, 3000.0f)) { continue; }
300  Submarine parentSub = item.CurrentHull?.Submarine ?? item.GetRootInventoryOwner()?.Submarine;
301  if (parentSub?.Info?.Type == SubmarineType.Player) { continue; }
302  return false;
303  }
304  }
305  return true;
306  }
