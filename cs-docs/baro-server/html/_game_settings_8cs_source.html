325  public KeyMapping(IEnumerable<XElement> elements, in KeyMapping? fallback)
326  {
327  var defaultBindings = GetDefault().Bindings;
328  Dictionary<InputType, KeyOrMouse> bindings = fallback?.Bindings?.ToMutable() ?? defaultBindings.ToMutable();
329  foreach (InputType inputType in (InputType[])Enum.GetValues(typeof(InputType)))
330  {
331  if (!bindings.ContainsKey(inputType)) { bindings.Add(inputType, defaultBindings[inputType]); }
332  }
333 
334  foreach (XElement element in elements)
335  {
336  foreach (XAttribute attribute in element.Attributes())
337  {
338  if (Enum.TryParse(attribute.Name.LocalName, out InputType result))
339  {
340  bindings[result] = element.GetAttributeKeyOrMouse(attribute.Name.LocalName, bindings[result]);
341  }
342  }
343  }
344 
345  Bindings = bindings.ToImmutableDictionary();
