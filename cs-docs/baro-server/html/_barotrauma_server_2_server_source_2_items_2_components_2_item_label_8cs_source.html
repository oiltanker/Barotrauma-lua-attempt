61 private IEnumerable<CoroutineStatus> SendStateAfterDelay()
62 {
63 while (sendStateTimer > 0.0f)
64 {
65 sendStateTimer -= CoroutineManager.DeltaTime;
66 yield return CoroutineStatus.Running;
67 }
68
69 if (item.Removed || GameMain.NetworkMember == null)
70 {
71 yield return CoroutineStatus.Success;
72 }
73
74 sendStateCoroutine = null;
75 if (lastSentText != Text) { item.CreateServerEvent(this); }
76 yield return CoroutineStatus.Success;
77 }
