142  public override void Update(float deltaTime, Camera cam)
143  {
144  LevelTrigger.RemoveInActiveTriggerers(PhysicsBody, triggerers);
145 
146  if (triggerOnce)
147  {
148  if (TriggeredOnce) { return; }
149  if (triggerers.Count > 0)
150  {
151  TriggeredOnce = true;
152  IsActive = false;
153  triggerers.Clear();
154  }
155  }
156 
157  TriggerActive = triggerers.Any();
158 
159  if (forceFluctuation && TriggerActive && (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
160  {
161  ForceFluctuationTimer += deltaTime;
162  if (ForceFluctuationTimer >= forceFluctuationInterval)
163  {
164  float v = MathF.Sin(2 * MathF.PI * forceFluctuationFrequency * TimeInLevel);
165  float amount = MathUtils.InverseLerp(-1.0f, 1.0f, v);
166  CurrentForceFluctuation = MathHelper.Lerp(1.0f - forceFluctuationStrength, 1.0f, amount);
167  ForceFluctuationTimer = 0.0f;
168  GameMain.NetworkMember?.CreateEntityEvent(this);
169  }
170  }
171 
172  foreach (Entity triggerer in triggerers)
173  {
174  LevelTrigger.ApplyStatusEffects(statusEffects, item.WorldPosition, triggerer, deltaTime, statusEffectTargets);
175 
176  if (triggerer is IDamageable damageable)
177  {
178  LevelTrigger.ApplyAttacks(attacks, damageable, item.WorldPosition, deltaTime);
179  }
180  else if (triggerer is Submarine submarine)
181  {
182  LevelTrigger.ApplyAttacks(attacks, item.WorldPosition, deltaTime);
183  }
184 
185  if (Force < 0.01f)
186  {
187  // Just ignore very minimal forces
188  continue;
189  }
190  else if (triggerer is Character c)
191  {
192  ApplyForce(c.AnimController.Collider);
193  }
194  else if (triggerer is Submarine s)
195  {
196  ApplyForce(s.SubBody.Body);
197  }
198  else if (triggerer is Item i && i.body != null)
199  {
200  ApplyForce(i.body);
201  }
202  }
203 
204  item.SendSignal(IsActive ? "1" : "0", "state_out");
205  }

207  private void ApplyForce(PhysicsBody body)
208  {
209  Vector2 diff = ConvertUnits.ToDisplayUnits(PhysicsBody.SimPosition - body.SimPosition);
210  if (diff.LengthSquared() < 0.0001f) { return; }
211  float distanceFactor = distanceBasedForce ? LevelTrigger.GetDistanceFactor(body, PhysicsBody, RadiusInDisplayUnits) : 1.0f;
212  if (distanceFactor <= 0.0f) { return; }
213  Vector2 force = distanceFactor * (CurrentForceFluctuation * Force) * Vector2.Normalize(diff);
214  if (force.LengthSquared() < 0.01f) { return; }
215  body.ApplyForce(force);
216  }
