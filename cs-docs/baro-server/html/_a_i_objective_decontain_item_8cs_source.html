70 protected override void Act(float deltaTime)
71 {
72 Item itemToDecontain =
73 targetItem ??
74 sourceContainer.Inventory.FindItem(i => itemIdentifiers.Any(id => i.Prefab.Identifier == id || i.HasTag(id) && !i.IgnoreByAI(character)), recursive: false);
75
76 if (itemToDecontain == null)
77 {
78 Abandon = true;
79 return;
80 }
81 if (itemToDecontain.IgnoreByAI(character))
82 {
83 Abandon = true;
84 return;
85 }
86 if (targetContainer == null)
87 {
88 if (sourceContainer == null)
89 {
90 Abandon = true;
91 return;
92 }
93 if (itemToDecontain.Container != sourceContainer.Item)
94 {
95 itemToDecontain.Drop(character);
96 IsCompleted = true;
97 return;
98 }
99 }
100 else if (targetContainer.Inventory.Contains(itemToDecontain))
101 {
102 IsCompleted = true;
103 return;
104 }
105 if (getItemObjective == null && !itemToDecontain.IsOwnedBy(character))
106 {
107 TryAddSubObjective(ref getItemObjective,
108 constructor: () => new AIObjectiveGetItem(character, targetItem, objectiveManager, Equip)
109 {
110 CannotFindDialogueCondition = AbandonGetItemDialogueCondition,
111 CannotFindDialogueIdentifierOverride = AbandonGetItemDialogueIdentifier,
112 SpeakIfFails = AbandonGetItemDialogueIdentifier != null,
113 TakeWholeStack = this.TakeWholeStack
114 },
115 onAbandon: () => Abandon = true);
116 return;
117 }
118 if (targetContainer != null)
119 {
120 TryAddSubObjective(ref containObjective,
121 constructor: () => new AIObjectiveContainItem(character, itemToDecontain, targetContainer, objectiveManager)
122 {
123 MoveWholeStack = TakeWholeStack,
124 Equip = Equip,
125 RemoveEmpty = false,
126 RemoveExistingWhenNecessary = RemoveExistingWhenNecessary,
127 RemoveExistingPredicate = RemoveExistingPredicate,
128 RemoveMax = RemoveExistingMax,
129 GetItemPriority = GetItemPriority,
130 ignoredContainerIdentifiers = sourceContainer != null ? new Identifier[] { sourceContainer.Item.Prefab.Identifier } : null
131 },
132 onCompleted: () => IsCompleted = true,
133 onAbandon: () => Abandon = true);
134 }
135 else
136 {
137 itemToDecontain.Drop(character);
138 IsCompleted = true;
139 }
140 }
