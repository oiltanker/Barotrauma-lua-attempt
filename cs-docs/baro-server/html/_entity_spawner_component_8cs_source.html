167  private bool CanSpawnMore()
168  {

199  private bool IsInRange(Vector2 worldPos, bool crewArea = false, bool rangePad = false)
200  {
201  Vector2 offset = crewArea ? CrewAreaOffset : SpawnAreaOffset;
202  switch (crewArea ? CrewAreaShape : SpawnAreaShape)
203  {
204  case AreaShape.Circle:
205  Vector2 center = item.WorldPosition + offset;
206  float distance = (crewArea ? CrewAreaRadius : SpawnAreaRadius) + (rangePad ? MaximumAmountRangePadding : 0);
207  return Vector2.DistanceSquared(worldPos, center) < distance * distance;
208 
209  case AreaShape.Rectangle:
210  RectangleF rect = GetAreaRectangle(crewArea ? CrewAreaBounds : SpawnAreaBounds, offset, draw: false);
211  if (rangePad)
212  {
213  rect.Inflate(MaximumAmountRangePadding, MaximumAmountRangePadding);
214  }
215 
216  return rect.Contains(worldPos);
217  }
218 
219  return false;

222  public void Spawn()
223  {
224  if (!CanSpawnMore()) { return; }
225 
226  int minAmount = Math.Min((int)SpawnAmountRange.X, (int)SpawnAmountRange.Y),
227  maxAmount = Math.Max((int)SpawnAmountRange.X, (int)SpawnAmountRange.Y);
228 
229  int amount = Rand.Range(minAmount, maxAmount, Rand.RandSync.Unsynced);
230 
231  Vector2 offset = SpawnAreaOffset;
232 
233  switch (SpawnAreaShape)
234  {
235  case AreaShape.Circle:
236  {
237  var (x, y) = item.WorldPosition + offset;
238 
239  for (int i = 0; i < Math.Max(1, amount); i++)
240  {
241  float angle = Rand.Range(-MathHelper.TwoPi, MathHelper.TwoPi);
242  float distance = Rand.Range(0, SpawnAreaRadius, Rand.RandSync.Unsynced);
243  Vector2 spawnPos = new Vector2(x + distance * (float)Math.Cos(angle), y + distance * (float)Math.Sin(angle));
244 
245  SpawnEntity(spawnPos);
246  }
247  break;
248  }
249  case AreaShape.Rectangle:
250  {
251  RectangleF rect = GetAreaRectangle(SpawnAreaBounds, offset, draw: false);
252 
253  for (int i = 0; i < Math.Max(1, amount); i++)
254  {
255  float minX = Math.Min(rect.Left, rect.Right),
256  maxX = Math.Max(rect.Left, rect.Right),
257  minY = Math.Min(rect.Top, rect.Bottom),
258  maxY = Math.Max(rect.Top, rect.Bottom);
259 
260  Vector2 spawnPos = new Vector2(Rand.Range(minX, maxX, Rand.RandSync.Unsynced), Rand.Range(minY, maxY, Rand.RandSync.Unsynced));
261 
262  SpawnEntity(spawnPos);
263  }
264  break;
265  }
