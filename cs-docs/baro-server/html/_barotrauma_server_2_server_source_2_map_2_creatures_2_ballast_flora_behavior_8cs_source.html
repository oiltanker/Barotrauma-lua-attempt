35 partial void UpdateDamage(float deltaTime)
36 {
37 damageUpdateTimer -= deltaTime;
38 if (damageUpdateTimer > 0.0f) { return; }
39
40 const int maxMessagesPerSecond = 10;
41 int messages = 0;
42 foreach (BallastFloraBranch branch in Branches)
43 {
44 //don't notify about minuscule amounts of damage (<= 1.0f)
45 if (branch.AccumulatedDamage > 1.0f)
46 {
47 CreateNetworkMessage(new BranchDamageEventData(branch));
48 branch.AccumulatedDamage = 0.0f;
49 messages++;
50 //throttle a bit: if a large ballast flora is withering, it can lead to a very large number of events otherwise
51 if (messages > maxMessagesPerSecond) { break; }
52 }
53 }
54 damageUpdateTimer = DamageUpdateInterval;
55 }

57 public void ServerWrite(IWriteMessage msg, IEventData eventData)
58 {
59 msg.Write((byte)eventData.NetworkHeader);
60
61 switch (eventData)
62 {
63 case SpawnEventData _:
64 ServerWriteSpawn(msg);
65 break;
66 case KillEventData _:
67 //do nothing
68 break;
69 case BranchCreateEventData branchCreateEventData:
70 ServerWriteBranchGrowth(msg, branchCreateEventData.NewBranch, branchCreateEventData.Parent.ID);
71 break;
72 case BranchDamageEventData branchDamageEventData:
73 ServerWriteBranchDamage(msg, branchDamageEventData.Branch);
74 break;
75 case InfectEventData infectEventData:
76 ServerWriteInfect(msg, infectEventData.Item.ID, infectEventData.Infect, infectEventData.Infector);
77 break;
78 case BranchRemoveEventData branchRemoveEventData:
79 ServerWriteBranchRemove(msg, branchRemoveEventData.Branch);
80 break;
81 }
82
83 msg.Write(PowerConsumptionTimer);
84 }

93 private void ServerWriteBranchGrowth(IWriteMessage msg, BallastFloraBranch branch, int parentId = -1)
94 {
95 var (x, y) = branch.Position;
96 msg.Write(parentId);
97 msg.Write((int)branch.ID);
98 msg.Write(branch.IsRootGrowth);
99 msg.WriteRangedInteger((byte)branch.Type, 0b0000, 0b1111);
100 msg.WriteRangedInteger((byte)branch.Sides, 0b0000, 0b1111);
101 msg.WriteRangedInteger(branch.FlowerConfig.Serialize(), 0, 0xFFF);
102 msg.WriteRangedInteger(branch.LeafConfig.Serialize(), 0, 0xFFF);
103 msg.Write((ushort)branch.MaxHealth);
104 msg.Write((int)(x / VineTile.Size));
105 msg.Write((int)(y / VineTile.Size));
106 msg.Write(branch.ParentBranch == null ? -1 : Branches.IndexOf(branch.ParentBranch));
107 }
