138  public void DetermineGapPositions(Submarine sub)
139  {
140  GapPositions = GapPosition.None;
141  foreach (Gap gap in Gap.GapList)
142  {
143  if (gap.Submarine != sub || gap.linkedTo.Count != 1) { continue; }
144  if (gap.ConnectedDoor != null && !gap.ConnectedDoor.UseBetweenOutpostModules) { continue; }
145 
146  //ignore gaps that are at a docking port
147  bool portFound = false;
148  foreach (DockingPort port in DockingPort.List)
149  {
150  if (Submarine.RectContains(gap.WorldRect, port.Item.WorldPosition))
151  {
152  portFound = true;
153  break;
154  }
155  }
156  if (portFound) { continue; }
157 
158  GapPositions |= gap.IsHorizontal ?
159  gap.linkedTo[0].WorldPosition.X < gap.WorldPosition.X ? GapPosition.Right : GapPosition.Left :
160  gap.linkedTo[0].WorldPosition.Y < gap.WorldPosition.Y ? GapPosition.Top : GapPosition.Bottom;
161  }
