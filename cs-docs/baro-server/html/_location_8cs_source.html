242  public void GenerateSpecials()
243  {
244  var availableStock = new Dictionary<ItemPrefab, float>();
245  foreach (var stockItem in Stock)
246  {
247  if (stockItem.Quantity < 1) { continue; }
248  float weight = 1.0f;
249  if (stockItem.ItemPrefab.GetPriceInfo(this) is PriceInfo priceInfo)
250  {
251  if (!priceInfo.CanBeSpecial) { continue; }
252  var baseQuantity = priceInfo.MinAvailableAmount > 0 ? priceInfo.MinAvailableAmount : PriceInfo.DefaultAmount;
253  weight += (float)(stockItem.Quantity - baseQuantity) / baseQuantity;
254  if (weight < 0.0f) { continue; }
255  }
256  availableStock.Add(stockItem.ItemPrefab, weight);
257  }
258  DailySpecials.Clear();
259  int extraSpecialSalesCount = Location.GetExtraSpecialSalesCount();
260  for (int i = 0; i < Location.DailySpecialsCount + extraSpecialSalesCount; i++)
261  {
262  if (availableStock.None()) { break; }
263  var item = ToolBox.SelectWeightedRandom(availableStock.Keys.ToList(), availableStock.Values.ToList(), Rand.RandSync.Unsynced);
264  if (item == null) { break; }
265  DailySpecials.Add(item);
266  availableStock.Remove(item);
267  }
268  RequestedGoods.Clear();
269  for (int i = 0; i < Location.RequestedGoodsCount; i++)
270  {
271  var item = ItemPrefab.Prefabs.GetRandom(p =>
272  p.CanBeSold && !RequestedGoods.Contains(p) &&
273  p.GetPriceInfo(this) is PriceInfo pi && pi.CanBeSpecial, Rand.RandSync.Unsynced);
274  if (item == null) { break; }
275  RequestedGoods.Add(item);
276  }
277  Location.StepsSinceSpecialsUpdated = 0;

287  public int GetAdjustedItemBuyPrice(ItemPrefab item, PriceInfo priceInfo = null, bool considerDailySpecials = true)
288  {
289  priceInfo ??= item?.GetPriceInfo(this);
290  if (priceInfo == null) { return 0; }
291  float price = priceInfo.Price;
292  // Adjust by random price modifier
293  price = (100 + PriceModifier) / 100.0f * price;
294  price *= priceInfo.BuyingPriceMultiplier;
295  // Adjust by daily special status
296  if (considerDailySpecials && DailySpecials.Contains(item))
297  {
298  price = Location.DailySpecialPriceModifier * price;
299  }
300  // Adjust by current location reputation
301  if (Location.Reputation.Value > 0.0f)
302  {
303  price = MathHelper.Lerp(1.0f, 1.0f - Location.StoreMaxReputationModifier, Location.Reputation.Value / Location.Reputation.MaxReputation) * price;
304  }
305  else
306  {
307  price = MathHelper.Lerp(1.0f, 1.0f + Location.StoreMaxReputationModifier, Location.Reputation.Value / Location.Reputation.MinReputation) * price;
308  }
309  // Price should never go below 1 mk
310  return Math.Max((int)price, 1);
311  }

315  public int GetAdjustedItemSellPrice(ItemPrefab item, PriceInfo priceInfo = null, bool considerRequestedGoods = true)
316  {
317  priceInfo ??= item?.GetPriceInfo(this);
318  if (priceInfo == null) { return 0; }
319  float price = Location.StoreSellPriceModifier * priceInfo.Price;
320  // Adjust by random price modifier
321  price = (100 - PriceModifier) / 100.0f * price;
322  // Adjust by current store balance

807  static float GetConnectionWeight(Location location, LocationConnection c)
808  {
809  float weight = c.Passed ? 1.0f : 5.0f;
810  Location destination = c.OtherLocation(location);

813  if (destination.MapPosition.X > location.MapPosition.X) { weight *= 2.0f; }
814  int missionCount = location.availableMissions.Count(m => m.Locations.Contains(destination));
815  if (missionCount > 0)
816  {
817  weight /= missionCount * 2;
818  }

820  {
821  weight *= 0.001f;
822  }
823  }
824  return weight;
825  }
