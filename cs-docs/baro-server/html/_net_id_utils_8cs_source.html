14 public static bool IdMoreRecent(ushort newID, ushort oldID)
15 {
16 uint id1 = newID;
17 uint id2 = oldID;
18
19 return
20 (id1 > id2) && (id1 - id2 <= ushort.MaxValue / 2)
21 ||
22 (id2 > id1) && (id2 - id1 > ushort.MaxValue / 2);
23 }
24
28 public static bool IdMoreRecentOrMatches(ushort newId, ushort oldId)
29 => !IdMoreRecent(oldId, newId);

31 public static ushort Difference(ushort id1, ushort id2)
32 {
33 int diff = id2 > id1 ? id2 - id1 : id1 - id2;
34 return (ushort)(diff > ushort.MaxValue / 2 ? ushort.MaxValue - diff : diff);
35 }

37 public static ushort Clamp(ushort id, ushort min, ushort max)
38 {
39 if (IdMoreRecent(min, max))
40 {
41 throw new ArgumentException($"Min cannot be larger than max ({min}, {max})");
42 }
43
44 if (!IdMoreRecent(id, min))
45 {
46 return min;
47 }
48 else if (IdMoreRecent(id, max))
49 {
50 return max;
51 }
52
53 return id;
54 }

59 public static bool IsValidId(ushort currentId, ushort previousId, ushort latestPossibleId)
60 {
61 //cannot be valid if more recent than the latest Id
62 if (IdMoreRecent(currentId, latestPossibleId)) { return false; }
63
64 //normally the id needs to be more recent than the previous id,
65 //but there's a special case when the previous id was 0:
66 // if a client reconnects mid-round and tries to jump from the unitialized state (0) back to some previous high id (> ushort.MaxValue / 2),
67 // this would normally get interpreted as trying to jump backwards, but in this case we'll allow it
68 return IdMoreRecent(currentId, previousId) || (previousId == 0 && currentId > ushort.MaxValue / 2);
69 }
