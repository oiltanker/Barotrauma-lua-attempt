42 public override void Update(float deltaTime, Camera cam)
43 {
44 UpdateOnActiveEffects(deltaTime);
45
46 CurrFlow = 0.0f;
47
48 if (item.CurrentHull == null) { return; }
49
50 if (Voltage < MinVoltage && PowerConsumption > 0)
51 {
52 return;
53 }
54
55 CurrFlow = Math.Min(PowerConsumption > 0 ? Voltage : 1.0f, 1.0f) * generatedAmount * 100.0f;
56 float conditionMult = item.Condition / item.MaxCondition;
57 //100% condition = 100% oxygen
58 //50% condition = 25% oxygen
59 //20% condition = 4%
60 CurrFlow *= conditionMult * conditionMult;
61
62 UpdateVents(CurrFlow, deltaTime);
63 }

88 private void GetVents()
89 {
90 totalHullVolume = 0.0f;
91 ventList ??= new List<(Vent vent, float hullVolume)>();
92 ventList.Clear();
93 foreach (MapEntity entity in item.linkedTo)
94 {
95 if (!(entity is Item linkedItem)) { continue; }
96
97 Vent vent = linkedItem.GetComponent<Vent>();
98 if (vent?.Item.CurrentHull == null) { continue; }
99
100 totalHullVolume += vent.Item.CurrentHull.Volume;
101 ventList.Add((vent, vent.Item.CurrentHull.Volume));
102 }
103
104 for (int i = 0; i < ventList.Count; i++)
105 {
106 Vent vent = ventList[i].vent;
107 foreach (Hull connectedHull in vent.Item.CurrentHull.GetConnectedHulls(includingThis: false, searchDepth: 3, ignoreClosedGaps: true))
108 {
109 //another vent in the connected hull -> don't add it to this vent's total hull volume
110 if (ventList.Any(v => v.vent != vent && v.vent.Item.CurrentHull == connectedHull)) { continue; }
111 totalHullVolume += connectedHull.Volume;
112 ventList[i] = (ventList[i].vent, ventList[i].hullVolume + connectedHull.Volume);
113 }
114 }
115 }
116
117 private void UpdateVents(float deltaOxygen, float deltaTime)
118 {
119 if (ventList == null || ventUpdateTimer < 0.0f)
120 {
121 GetVents();
122 ventUpdateTimer = VentUpdateInterval;
123 }
124 ventUpdateTimer -= deltaTime;
125
126 if (!ventList.Any() || totalHullVolume <= 0.0f) { return; }
127
128 foreach ((Vent vent, float hullVolume) in ventList)
129 {
130 if (vent.Item.CurrentHull == null) { continue; }
131
132 vent.OxygenFlow = deltaOxygen * (hullVolume / totalHullVolume);
133 vent.IsActive = true;
134 }
135 }
