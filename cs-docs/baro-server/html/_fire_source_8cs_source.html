130  public static void UpdateAll(List<FireSource> fireSources, float deltaTime)
131  {
132  for (int i = fireSources.Count - 1; i >= 0; i--)
133  {
134  fireSources[i].Update(deltaTime);
135  }
136 
137  //combine overlapping fires
138  for (int i = fireSources.Count - 1; i >= 0; i--)
139  {
140  for (int j = i - 1; j >= 0; j--)
141  {
142  i = Math.Min(i, fireSources.Count - 1);
143  j = Math.Min(j, i - 1);
144 
145  if (!fireSources[i].CheckOverLap(fireSources[j])) { continue; }
146 
147  float leftEdge = Math.Min(fireSources[i].position.X, fireSources[j].position.X);
148 
149  fireSources[j].size.X =
150  Math.Max(fireSources[i].position.X + fireSources[i].size.X, fireSources[j].position.X + fireSources[j].size.X)
151  - leftEdge;
152 
153  fireSources[j].position.X = leftEdge;
154  fireSources[j].burnDecals.AddRange(fireSources[i].burnDecals);
155  fireSources[j].burnDecals.Sort((d1, d2) => { return Math.Sign(d1.WorldPosition.X - d2.WorldPosition.X); });
156  fireSources[i].Remove();
157  }
158  }

201  public void Update(float deltaTime)
202  {
203  //the firesource will start to shrink if oxygen percentage is below 10
204  float growModifier = Math.Min((hull.OxygenPercentage / 10.0f) - 1.0f, 1.0f);
205 
206  if (DamagesCharacters) { DamageCharacters(deltaTime); }
207  if (DamagesItems) { DamageItems(deltaTime); }
208 
209  if (hull.WaterVolume > 0.0f)
210  {
211  HullWaterExtinguish(deltaTime);
212  if (removed) { return; }
213  }
214 
215  if (!(this is DummyFireSource))
216  {
217  ReduceOxygen(deltaTime);
218  }
219 
220  AdjustXPos(growModifier, deltaTime);
221 
222  size.X += GrowSpeed * growModifier * deltaTime;
223  size.Y = MathHelper.Clamp(size.Y + GrowSpeed * growModifier * deltaTime, 10.0f, 50.0f);
224 
225  if (size.X > 50.0f)
226  {
227  this.position.Y = MathHelper.Lerp(this.position.Y, hull.Rect.Y - hull.Rect.Height + size.Y, deltaTime);
228  }
229 
230  LimitSize();
231 
232  if (size.X > 256.0f && !(this is DummyFireSource))
233  {
234  if (burnDecals.Count == 0)
235  {
236  var newDecal = hull.AddDecal("burnt", WorldPosition + size / 2, 1f, isNetworkEvent: false);
237  if (newDecal != null) { burnDecals.Add(newDecal); }
238  }
239  else if (WorldPosition.X < burnDecals[0].WorldPosition.X - 256.0f)
240  {
241  var newDecal = hull.AddDecal("burnt", WorldPosition, 1f, isNetworkEvent: false);
242  if (newDecal != null) { burnDecals.Insert(0, newDecal); }
243  }
244  else if (WorldPosition.X + size.X > burnDecals[burnDecals.Count - 1].WorldPosition.X + 256.0f)
245  {
246  var newDecal = hull.AddDecal("burnt", WorldPosition + Vector2.UnitX * size.X, 1f, isNetworkEvent: false);
247  if (newDecal != null) { burnDecals.Add(newDecal); }
248  }
249  }
250 
251  foreach (Decal d in burnDecals)
252  {
253  //prevent the decals from fading out as long as the firesource is alive
254  d.ForceRefreshFadeTimer(Math.Min(d.FadeTimer, d.FadeInTime));
255  }
256 
257  UpdateProjSpecific(growModifier);
258 
259 
260  if (size.X < 1.0f && (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
261  {
262  Remove();

288  private void DamageCharacters(float deltaTime)
289  {
290  if (size.X <= 0.0f) { return; }
291 
292  for (int i = 0; i < Character.CharacterList.Count; i++)
293  {
294  Character c = Character.CharacterList[i];
295  if (c.CurrentHull == null || c.IsDead) { continue; }
296 
297  if (!IsInDamageRange(c, DamageRange)) { continue; }
298 
299  //GetApproximateDistance returns float.MaxValue if there's no path through open gaps between the hulls (e.g. if there's a door/wall in between)
300  if (hull.GetApproximateDistance(Position, c.Position, c.CurrentHull, 10000.0f) > size.X + DamageRange)
301  {
302  continue;
303  }
304 
305  float dmg = (float)Math.Sqrt(Math.Min(500, size.X)) * deltaTime / c.AnimController.Limbs.Count(l => !l.IsSevered && !l.Hidden);
306  foreach (Limb limb in c.AnimController.Limbs)
307  {
308  if (limb.IsSevered) { continue; }
309  c.LastDamageSource = null;
310  c.DamageLimb(WorldPosition, limb, AfflictionPrefab.Burn.Instantiate(dmg).ToEnumerable(), 0.0f, false, 0.0f);
311  }
312 #if CLIENT
313  //let clients display the client-side damage immediately, otherwise they may not be able to react to the damage fast enough
314  c.CharacterHealth.DisplayedVitality = c.Vitality;
315 #endif

336  private void DamageItems(float deltaTime)
337  {
338  if (size.X <= 0.0f) { return; }
339 #if CLIENT
340  if (GameMain.Client != null) { return; }
341 #endif
342  foreach (Item item in Item.ItemList)
343  {
344  if (item.CurrentHull != hull || item.FireProof || item.Condition <= 0.0f) { continue; }
345 
346  //don't apply OnFire effects if the item is inside a fireproof container
347  //(or if it's inside a container that's inside a fireproof container, etc)
348  Item container = item.Container;
349  bool fireProof = false;
350  while (container != null)
351  {
352  if (container.FireProof)
353  {
354  fireProof = true;
355  break;
356  }
357  container = container.Container;
358  }
359  if (fireProof) { continue; }
360 
361  float range = (float)Math.Sqrt(size.X) * 10.0f;
362  if (item.Position.X < position.X - range || item.Position.X > position.X + size.X + range) { continue; }
363  if (item.Position.Y < position.Y - size.Y || item.Position.Y > hull.Rect.Y) { continue; }
364 
365  item.ApplyStatusEffects(ActionType.OnFire, deltaTime);
366  if (item.Condition <= 0.0f && GameMain.NetworkMember is { IsServer: true })
367  {
