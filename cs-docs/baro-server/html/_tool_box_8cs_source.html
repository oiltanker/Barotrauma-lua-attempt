88 public static string CorrectFilenameCase(string filename, out bool corrected, string directory = "")
89 {
90 char[] delimiters = { '/', '\\' };
91 string[] subDirs = filename.Split(delimiters);
92 string originalFilename = filename;
93 filename = "";
94 corrected = false;
95
96#if !WINDOWS
97 if (File.Exists(originalFilename))
98 {
99 return originalFilename;
100 }
101#endif
102
103 if (Path.IsPathRooted(originalFilename))
104 {
105 return originalFilename; //assume that rooted paths have correct case since these are generated by the game
106 }
107
108 string startPath = directory ?? "";
109
110 for (int i = 0; i < subDirs.Length; i++)
111 {
112 if (i == subDirs.Length - 1 && string.IsNullOrEmpty(subDirs[i]))
113 {
114 break;
115 }
116
117 string subDir = subDirs[i].TrimEnd();
118 string enumPath = Path.Combine(startPath, filename);
119
120 if (string.IsNullOrWhiteSpace(filename))
121 {
122 enumPath = string.IsNullOrWhiteSpace(startPath) ? "./" : startPath;
123 }
124
125 string[] filePaths = Directory.GetFileSystemEntries(enumPath).Select(Path.GetFileName).ToArray();
126
127 if (filePaths.Any(s => s.Equals(subDir, StringComparison.Ordinal)))
128 {
129 filename += subDir;
130 }
131 else
132 {
133 string[] correctedPaths = filePaths.Where(s => s.Equals(subDir, StringComparison.OrdinalIgnoreCase)).ToArray();
134 if (correctedPaths.Any())
135 {
136 corrected = true;
137 filename += correctedPaths.First();
138 }
139 else
140 {
141 //DebugConsole.ThrowError($"File \"{originalFilename}\" not found!");
142 corrected = false;
143 return originalFilename;
144 }
145 }
146 if (i < subDirs.Length - 1) { filename += "/"; }
147 }
148
149 return filename;
150 }

217 public static Color HSVToRGB(float hue, float saturation, float value)
218 {
219 float c = value * saturation;
220
221 float h = Math.Clamp(hue, 0, 360) / 60f;
222
223 float x = c * (1 - Math.Abs(h % 2 - 1));
224
225 float r = 0,
226 g = 0,
227 b = 0;
228
229 if (0 <= h && h <= 1) { r = c; g = x; b = 0; }
230 else if (1 < h && h <= 2) { r = x; g = c; b = 0; }
231 else if (2 < h && h <= 3) { r = 0; g = c; b = x; }
232 else if (3 < h && h <= 4) { r = 0; g = x; b = c; }
233 else if (4 < h && h <= 5) { r = x; g = 0; b = c; }
234 else if (5 < h && h <= 6) { r = c; g = 0; b = x; }
235
236 float m = value - c;
237
238 return new Color(r + m, g + m, b + m);
239 }

276 // Convert an RGB value into an HLS value.
277 public static Vector3 RgbToHLS(Vector3 color)
278 {
279 double h, l, s;
280
281 double double_r = color.X;
282 double double_g = color.Y;
283 double double_b = color.Z;
284
285 // Get the maximum and minimum RGB components.
286 double max = double_r;
287 if (max < double_g) max = double_g;
288 if (max < double_b) max = double_b;
289
290 double min = double_r;
291 if (min > double_g) min = double_g;
292 if (min > double_b) min = double_b;
293
294 double diff = max - min;
295 l = (max + min) / 2;
296 if (Math.Abs(diff) < 0.00001)
297 {
298 s = 0;
299 h = 0; // H is really undefined.
300 }
301 else
302 {
303 if (l <= 0.5) s = diff / (max + min);
304 else s = diff / (2 - max - min);
305
306 double r_dist = (max - double_r) / diff;
307 double g_dist = (max - double_g) / diff;
308 double b_dist = (max - double_b) / diff;
309
310 if (double_r == max) h = b_dist - g_dist;
311 else if (double_g == max) h = 2 + r_dist - b_dist;
312 else h = 4 + g_dist - r_dist;
313
314 h = h * 60;
315 if (h < 0) h += 360;
316 }
317
318 return new Vector3((float)h, (float)l, (float)s);
319 }

324 public static int LevenshteinDistance(string s, string t)
325 {
326 int n = s.Length;
327 int m = t.Length;
328 int[,] d = new int[n + 1, m + 1];
329
330 if (n == 0 || m == 0) return 0;
331
332 for (int i = 0; i <= n; d[i, 0] = i++) ;
333 for (int j = 0; j <= m; d[0, j] = j++) ;
334
335 for (int i = 1; i <= n; i++)
336 {
337 for (int j = 1; j <= m; j++)
338 {
339 int cost = (t[j - 1] == s[i - 1]) ? 0 : 1;
340
341 d[i, j] = Math.Min(
342 Math.Min(d[i - 1, j] + 1, d[i, j - 1] + 1),
343 d[i - 1, j - 1] + cost);
344 }
345 }
346
347 return d[n, m];
348 }

442 public static T SelectWeightedRandom<T>(IList<T> objects, IList<float> weights, Random random)
443 {
444 if (objects.Count == 0) return default(T);
445
446 if (objects.Count != weights.Count)
447 {
448 DebugConsole.ThrowError("Error in SelectWeightedRandom, number of objects does not match the number of weights.\n" + Environment.StackTrace.CleanupStackTrace());
449 return objects[0];
450 }
451
452 float totalWeight = weights.Sum();
453
454 float randomNum = (float)(random.NextDouble() * totalWeight);
455 for (int i = 0; i < objects.Count; i++)
456 {
457 if (randomNum <= weights[i])
458 {
459 return objects[i];
460 }
461 randomNum -= weights[i];
462 }
463 return default(T);
464 }
