112  public void Update(EnemyAIController enemyAI, float deltaTime)
113  {
114  if (TargetCharacter != null && character.Submarine != TargetCharacter.Submarine ||
115  character.Submarine != null && TargetSubmarine != null && TargetCharacter == null)
116  {
117  DeattachFromBody(reset: true);
118  return;
119  }
120  if (IsAttached)
121  {
122  if (Math.Sign(attachLimb.Dir) != Math.Sign(jointDir))
123  {
124  var attachJoint = AttachJoints[0];
125  if (attachJoint is WeldJoint weldJoint)
126  {
127  weldJoint.LocalAnchorA = new Vector2(-weldJoint.LocalAnchorA.X, weldJoint.LocalAnchorA.Y);
128  weldJoint.ReferenceAngle = -weldJoint.ReferenceAngle;
129  }
130  else if (attachJoint is RevoluteJoint revoluteJoint)
131  {
132  revoluteJoint.LocalAnchorA = new Vector2(-revoluteJoint.LocalAnchorA.X, revoluteJoint.LocalAnchorA.Y);
133  revoluteJoint.ReferenceAngle = -revoluteJoint.ReferenceAngle;
134  }
135  jointDir = attachLimb.Dir;
136  }
137  for (int i = 0; i < AttachJoints.Count; i++)
138  {
139  //something went wrong, limb body is very far from the joint anchor -> deattach
140  if (Vector2.DistanceSquared(AttachJoints[i].WorldAnchorB, AttachJoints[i].BodyA.Position) > 10.0f * 10.0f)
141  {
142 #if DEBUG
143  DebugConsole.Log("Limb body of the character \"" + character.Name + "\" is very far from the attach joint anchor -> deattach");
144 #endif
145  DeattachFromBody(reset: true);
146  return;
147  }
148  }
149  if (TargetCharacter != null)
150  {
151  if (enemyAI.AttackLimb?.attack == null)
152  {
153  DeattachFromBody(reset: true, cooldown: 1);
154  }
155  else
156  {
157  float range = enemyAI.AttackLimb.attack.DamageRange * 2f;
158  if (Vector2.DistanceSquared(TargetCharacter.WorldPosition, enemyAI.AttackLimb.WorldPosition) > range * range)
159  {
160  DeattachFromBody(reset: true, cooldown: 1);
161  }
162  else
163  {
164  TargetCharacter.Latchers.Add(this);
165  }
166  }
167  }
168  }
169 
170  if (attachCooldown > 0)
171  {
172  attachCooldown -= deltaTime;
173  }
174  if (deattachCheckTimer > 0)
175  {
176  deattachCheckTimer -= deltaTime;
177  }
178 
179  if (TargetCharacter != null)
180  {
181  // Own sim pos -> target where we are
182  _attachPos = character.SimPosition;
183  }
184  Vector2 transformedAttachPos = _attachPos;
185  if (character.Submarine == null && TargetSubmarine != null)
186  {
187  transformedAttachPos += ConvertUnits.ToSimUnits(TargetSubmarine.Position);
188  }
189  if (transformedAttachPos != Vector2.Zero)
190  {
191  AttachPos = transformedAttachPos;
192  }
193 
194  switch (enemyAI.State)
195  {
196  case AIState.Idle:
197  if (AttachToWalls && character.Submarine == null && Level.Loaded != null)
198  {
199  if (!IsAttached)
200  {
201  raycastTimer -= deltaTime;
202  //check if there are any walls nearby the character could attach to
203  if (raycastTimer < 0.0f)
204  {
205  _attachPos = Vector2.Zero;
206 
207  var cells = Level.Loaded.GetCells(character.WorldPosition, 1);
208  if (cells.Count > 0)
209  {
210  //ignore walls more than 200 meters away
211  float closestDist = 200.0f * 200.0f;
212  foreach (Voronoi2.VoronoiCell cell in cells)
213  {
214  foreach (Voronoi2.GraphEdge edge in cell.Edges)
215  {
216  if (MathUtils.GetLineIntersection(edge.Point1, edge.Point2, character.WorldPosition, cell.Center, out Vector2 intersection))
217  {
218  Vector2 potentialAttachPos = ConvertUnits.ToSimUnits(intersection);
219  float distSqr = Vector2.DistanceSquared(character.SimPosition, potentialAttachPos);
220  if (distSqr < closestDist)
221  {
222  attachSurfaceNormal = edge.GetNormal(cell);
223  targetBody = cell.Body;
224  _attachPos = potentialAttachPos;
225  closestDist = distSqr;
226  }
227  break;
228  }
229  }
230  }
231  }
232  raycastTimer = RaycastInterval;
233  }
234  }
235  }
236  else
237  {
238  _attachPos = Vector2.Zero;
239  }
240  if (_attachPos == Vector2.Zero || targetBody == null)
241  {
242  DeattachFromBody(reset: false);
243  }
244  else
245  {
246  float squaredDistance = Vector2.DistanceSquared(character.SimPosition, _attachPos);
247  float targetDistance = Math.Max(Math.Max(character.AnimController.Collider.radius, character.AnimController.Collider.width), character.AnimController.Collider.height) * 1.2f;
248  if (squaredDistance < targetDistance * targetDistance)
249  {
250  //close enough to a wall -> attach
251  AttachToBody(_attachPos);
252  enemyAI.SteeringManager.Reset();
253  }
254  else
255  {
256  //move closer to the wall
257  DeattachFromBody(reset: false);
258  enemyAI.SteeringManager.SteeringAvoid(deltaTime, 1.0f, 0.1f);
259  enemyAI.SteeringManager.SteeringSeek(_attachPos);
260  }
261  }
262  break;
263  case AIState.Attack:
264  case AIState.Aggressive:
265  if (enemyAI.IsSteeringThroughGap) { break; }
266  if (_attachPos == Vector2.Zero) { break; }
267  if (!AttachToSub && !AttachToCharacters) { break; }
268  if (enemyAI.AttackLimb == null) { break; }
269  if (targetBody == null) { break; }
270  if (IsAttached && AttachJoints[0].BodyB == targetBody) { break; }
271  Vector2 referencePos = TargetCharacter != null ? TargetCharacter.WorldPosition : ConvertUnits.ToDisplayUnits(transformedAttachPos);
272  if (Vector2.DistanceSquared(referencePos, enemyAI.AttackLimb.WorldPosition) < enemyAI.AttackLimb.attack.DamageRange * enemyAI.AttackLimb.attack.DamageRange)
273  {
274  AttachToBody(transformedAttachPos);
275  }
276  break;
277  default:
278  DeattachFromBody(reset: true);
279  break;
280  }
281 
282  if (IsAttached && targetBody != null && deattachCheckTimer <= 0.0f)
283  {
284  bool deattach = false;
285  if (maxAttachDuration > 0)
286  {
287  deattach = true;
288  attachCooldown = coolDown;
289  }
290  if (!deattach && TargetWall != null && TargetSubmarine != null)
291  {
292  // Deattach if the wall is broken enough where we are attached to
293  int targetSection = TargetWall.FindSectionIndex(attachLimb.WorldPosition, world: true, clamp: true);
294  if (enemyAI.CanPassThroughHole(TargetWall, targetSection))
295  {
296  deattach = true;
297  attachCooldown = coolDown;
298  }
299  if (!deattach)
300  {
301  // Deattach if the velocity is high
302  float velocity = TargetSubmarine.Velocity == Vector2.Zero ? 0.0f : TargetSubmarine.Velocity.Length();
303  deattach = velocity > maxDeattachSpeed;
304  if (!deattach)
305  {
306  if (velocity > minDeattachSpeed)
307  {
308  float velocityFactor = (maxDeattachSpeed - minDeattachSpeed <= 0.0f) ?
309  Math.Sign(Math.Abs(velocity) - minDeattachSpeed) :
310  (Math.Abs(velocity) - minDeattachSpeed) / (maxDeattachSpeed - minDeattachSpeed);
311 
312  if (Rand.Range(0.0f, 1.0f) < velocityFactor)
313  {
314  deattach = true;
315  character.AddDamage(character.WorldPosition, new List<Affliction>() { AfflictionPrefab.InternalDamage.Instantiate(damageOnDetach) }, detachStun, true);
316  attachCooldown = Math.Max(detachStun * 2, coolDown);
317  }
318  }
319  }
320  }
321  deattachCheckTimer = 5.0f;
322  }
323  if (deattach)
324  {
325  DeattachFromBody(reset: true);
326  }
327  }

330  private void AttachToBody(Vector2 attachPos)
331  {
332  if (attachLimb == null) { return; }
333  if (targetBody == null) { return; }
334  if (attachCooldown > 0) { return; }
335  var collider = character.AnimController.Collider;
336  //already attached to something
337  if (AttachJoints.Count > 0)
338  {
339  //already attached to the target body, no need to do anything
340  if (AttachJoints[0].BodyB == targetBody) { return; }
341  DeattachFromBody(reset: false);
342  }
343 
344  jointDir = attachLimb.Dir;
345 
346  Vector2 transformedLocalAttachPos = localAttachPos * attachLimb.Scale * attachLimb.Params.Ragdoll.LimbScale;
347  if (jointDir < 0.0f)
348  {
349  transformedLocalAttachPos.X = -transformedLocalAttachPos.X;
350  }
351 
352  float angle = MathUtils.VectorToAngle(-attachSurfaceNormal) - MathHelper.PiOver2 + attachLimbRotation * attachLimb.Dir;
353  attachLimb.body.SetTransform(attachPos + attachSurfaceNormal * transformedLocalAttachPos.Length(), angle);
354 
355  var limbJoint = new WeldJoint(attachLimb.body.FarseerBody, targetBody,
356  transformedLocalAttachPos, targetBody.GetLocalPoint(attachPos), false)
357  {
358  FrequencyHz = 10.0f,
359  DampingRatio = 0.5f,
360  KinematicBodyB = true,
361  CollideConnected = false,
362  };
363  GameMain.World.Add(limbJoint);
364  AttachJoints.Add(limbJoint);
365 
366  // Limb scale is already taken into account when creating the collider.
367  Vector2 colliderFront = collider.GetLocalFront();
368  if (jointDir < 0.0f)
369  {
370  colliderFront.X = -colliderFront.X;
371  }
372  collider.SetTransform(attachPos + attachSurfaceNormal * colliderFront.Length(), MathUtils.VectorToAngle(-attachSurfaceNormal) - MathHelper.PiOver2);
373 
374  Joint colliderJoint = weld ?
375  new WeldJoint(collider.FarseerBody, targetBody, colliderFront, targetBody.GetLocalPoint(attachPos), false)
376  {
377  FrequencyHz = 10.0f,
378  DampingRatio = 0.5f,
379  KinematicBodyB = true,
380  CollideConnected = false,
381  } :
382  new RevoluteJoint(collider.FarseerBody, targetBody, colliderFront, targetBody.GetLocalPoint(attachPos), false)
383  {
384  MotorEnabled = true,
385  MaxMotorTorque = 0.25f
386  } as Joint;
387 
388  GameMain.World.Add(colliderJoint);
389  AttachJoints.Add(colliderJoint);
390  TargetCharacter?.Latchers.Add(this);
391  if (maxAttachDuration > 0)
392  {
393  deattachCheckTimer = maxAttachDuration;
394  }
395  }
