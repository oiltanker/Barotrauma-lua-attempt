30 foreach (var target in targets)

32 if (target.CharacterHealth == null) { continue; }

35 if (target.CharacterHealth.GetAffliction(Identifier, AllowLimbAfflictions) != null) { return true; }
36 }

38 {
39 LimbType? limbType = target.CharacterHealth.GetAfflictionLimb(affliction)?.type;
40 if (limbType == null) { return false; }

46 }
