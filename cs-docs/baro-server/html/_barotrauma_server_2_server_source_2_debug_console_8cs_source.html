68 public static void UpdateCommandLine(int maxTime)
69 {
70 Stopwatch sw = new Stopwatch();
71 sw.Start();
72
73 int consoleWidth = 0;
74 int consoleHeight = 0;
75
76 if(!Console.IsOutputRedirected)
77 {
78 consoleWidth = Console.WindowWidth;
79 if (consoleWidth < 5) consoleWidth = 5;
80 consoleHeight = Console.WindowHeight;
81 if (consoleHeight < 5) consoleHeight = 5;
82 }
83
84 //dequeue messages
85 lock (queuedMessages)
86 {
87 if (queuedMessages.Count > 0)
88 {
89
90 if (!Console.IsOutputRedirected)
91 {
92 Console.CursorLeft = 0;
93 }
94 while (queuedMessages.Count > 0)
95 {
96 ColoredText msg = queuedMessages.Dequeue();
97 Messages.Add(msg);
98 if (GameSettings.CurrentConfig.SaveDebugConsoleLogs || GameSettings.CurrentConfig.VerboseLogging)
99 {
100 unsavedMessages.Add(msg);
101 if (unsavedMessages.Count >= messagesPerFile)
102 {
103 SaveLogs();
104 unsavedMessages.Clear();
105 }
106 }
107
108 string msgTxt = msg.Text;
109
110 if (msg.IsCommand) commandMemory.Add(msgTxt);
111
112 if(!Console.IsOutputRedirected)
113 {
114 int paddingLen = consoleWidth - (msg.Text.Length % consoleWidth) - 1;
115 msgTxt += new string(' ', paddingLen > 0 ? paddingLen : 0);
116
117 Console.ForegroundColor = XnaToConsoleColor.Convert(msg.Color);
118 }
119 Console.WriteLine(msgTxt);
120
121 if (sw.ElapsedMilliseconds >= maxTime) { break; }
122 }
123 if(!Console.IsOutputRedirected)
124 {
125 RewriteInputToCommandLine(input);
126 }
127 }
128 if (Messages.Count > MaxMessages)
129 {
130 Messages.RemoveRange(0, Messages.Count - MaxMessages);
131 }
132 }
133
134 // No good way to display input when console output is redirected, and can't read from redirected input using KeyAvailable.
135 if(!Console.IsOutputRedirected && !Console.IsInputRedirected)
136 {
137 //read player input
138 bool rewriteInput = false;
139 while (Console.KeyAvailable)
140 {
141 if (sw.ElapsedMilliseconds >= maxTime)
142 {
143 rewriteInput = false;
144 break;
145 }
146 rewriteInput = true;
147 ConsoleKeyInfo key = Console.ReadKey(true);
148 switch (key.Key)
149 {
150 case ConsoleKey.Enter:
151 lock (QueuedCommands)
152 {
153 QueuedCommands.Add(input);
154 }
155 input = "";
156 memoryIndex = -1;
157 break;
158 case ConsoleKey.Backspace:
159 if (input.Length > 0) input = input.Substring(0, input.Length - 1);
160 ResetAutoComplete();
161 memoryIndex = -1;
162 break;
163 case ConsoleKey.LeftArrow:
164 input = AutoComplete(input, -1);
165 break;
166 case ConsoleKey.RightArrow:
167 input = AutoComplete(input, 1);
168 break;
169 case ConsoleKey.UpArrow:
170 memoryIndex--;
171 if (memoryIndex < 0) memoryIndex = commandMemory.Count - 1;
172 if (memoryIndex >= commandMemory.Count) memoryIndex = commandMemory.Count - 1;
173 if (memoryIndex >= 0)
174 {
175 input = commandMemory[memoryIndex];
176 }
177 break;
178 case ConsoleKey.DownArrow:
179 memoryIndex++;
180 if (memoryIndex < 0) memoryIndex = 0;
181 if (memoryIndex >= commandMemory.Count) memoryIndex = 0;
182 if (commandMemory.Count>0)
183 {
184 input = commandMemory[memoryIndex];
185 }
186 break;
187 case ConsoleKey.Tab:
188 if (input.Length > 0)
189 {
190 input = AutoComplete(input, 0);
191 memoryIndex = -1;
192 }
193 break;
194 default:
195 if (key.KeyChar != 0)
196 {
197 input += key.KeyChar;
198 memoryIndex = -1;
199 }
200 ResetAutoComplete();
201 break;
202 }
203 }
204 if (rewriteInput) { RewriteInputToCommandLine(input); }
205 }
206
207 sw.Stop();
208 }

210 private static void WriteAndResetLine(string txt)
211 {
212 int consoleWidth = Console.BufferWidth;
213 int linesWritten = 0;
214 while (true)
215 {
216 if (txt.Length > consoleWidth)
217 {
218 linesWritten++;
219 Console.Write(txt.Substring(0, consoleWidth));
220 txt = txt.Substring(consoleWidth);
221 }
222 else
223 {
224 Console.Write(txt);
225 if (txt.Length == consoleWidth)
226 {
227 Console.Write(' '); Console.CursorLeft--;
228 linesWritten++;
229 }
230 break;
231 }
232 }
233 Console.CursorTop -= linesWritten;
234 }

236 private static void RewriteInputToCommandLine(string input)
237 {
238 if (Console.WindowWidth == 0 || Console.WindowHeight == 0) { return; }
239
240 int consoleWidth = Math.Max(Console.BufferWidth, 5);
241 //int inputLines = Math.Max((int)Math.Ceiling(input.Length / (float)consoleWidth), 1);
242 //int cursorLine = Math.Max((int)Math.Ceiling((input.Length + 1) / (float)consoleWidth), 1);
243
244 try
245 {
246 string tmpInput = input;
247 while (tmpInput.Length >= consoleWidth)
248 {
249 tmpInput = tmpInput.Substring(consoleWidth);
250 }
251 string ln = input.Length > 0 ? AutoComplete(input, 0) : "";
252 while (ln.Length >= consoleWidth)
253 {
254 ln = ln.Substring(consoleWidth);
255 }
256 ln += new string(' ', consoleWidth - (ln.Length % consoleWidth));
257 Console.ForegroundColor = ConsoleColor.DarkGray;
258 Console.CursorLeft = 0;
259 WriteAndResetLine(ln);
260 Console.ForegroundColor = ConsoleColor.White;
261 Console.CursorLeft = 0;
262 WriteAndResetLine(tmpInput);
263 Console.CursorLeft = input.Length % consoleWidth;
264 }
265 catch (Exception e)
266 {
267 string errorMsg = "Failed to write input to command line (window width: " + Console.WindowWidth + ", window height: " + Console.WindowHeight + ")\n"
268 + e.Message + "\n" + e.StackTrace.CleanupStackTrace();
269 GameAnalyticsManager.AddErrorEventOnce("DebugConsole.RewriteInputToCommandLine", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
270 }
