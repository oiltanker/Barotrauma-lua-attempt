361  bool ingredientsAllowStealing = true;
362 
363  if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
364  {
365  fabricatedItem.RequiredItems.ForEach(requiredItem =>
366  {
367  for (int usedPrefabsAmount = 0; usedPrefabsAmount < requiredItem.Amount; usedPrefabsAmount++)
368  {

370  {
371  if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
372 
373  var availableItems = availableIngredients[requiredPrefab.Identifier];
374  var availableItem = availableItems.FirstOrDefault(potentialPrefab =>
375  {
376  return potentialPrefab.ConditionPercentage >= requiredItem.MinCondition * 100.0f &&
377  potentialPrefab.ConditionPercentage <= requiredItem.MaxCondition * 100.0f;
378  });
379 
380  if (availableItem == null) { continue; }
381 
382  ingredientsStolen |= availableItem.StolenDuringRound;
383  if (!availableItem.AllowStealing)
384  {
385  ingredientsAllowStealing = false;
386  }
387 
388  //Leave it behind with reduced condition if it has enough to stay above 0
389  if (requiredItem.UseCondition && availableItem.ConditionPercentage - requiredItem.MinCondition * 100 > 0.0f)
390  {
391  availableItem.Condition -= availableItem.Prefab.Health * requiredItem.MinCondition;
392  continue;
393  }
394  if (availableItem.OwnInventory != null)
395  {
396  foreach (Item containedItem in availableItem.OwnInventory.AllItemsMod)

415  });

420

423  {
424  quality = fabricatedItem.Quality.Value;
425  }

568  private bool CanBeFabricated(FabricationRecipe fabricableItem, IReadOnlyDictionary<Identifier, List<Item>> availableIngredients, Character character)
569  {
570  if (fabricableItem == null) { return false; }
571  if (fabricableItem.RequiresRecipe && (character == null || !character.HasRecipeForItem(fabricableItem.TargetItem.Identifier))) { return false; }
572 
573  if (fabricableItem.RequiredMoney > 0)
574  {
575  switch (GameMain.GameSession?.GameMode)
576  {
577  case MultiPlayerCampaign mpCampaign:
578  {
579  if (!mpCampaign.CanAfford(fabricableItem.RequiredMoney, GetUsingClient())) { return false; }
580 
581  break;
582  }
583  case CampaignMode campaign:
584  {
585  if (campaign.Bank.Balance < fabricableItem.RequiredMoney) { return false; }
586 
587  break;
588  }
589  default:
590  return false;
591  }
592  }
593 
594  if (fabricationLimits.TryGetValue(fabricableItem.RecipeHash, out int amount) && amount <= 0)
595  {
596  return false;
597  }
598 
599  return fabricableItem.RequiredItems.All(requiredItem =>
600  {
601  int availablePrefabsAmount = 0;
602  foreach (ItemPrefab requiredPrefab in requiredItem.ItemPrefabs)
603  {
604  if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
605 
606  var availablePrefabs = availableIngredients[requiredPrefab.Identifier];
607  foreach (Item availablePrefab in availablePrefabs)
608  {
609  if (availablePrefab.ConditionPercentage / 100.0f >= requiredItem.MinCondition &&
610  availablePrefab.ConditionPercentage / 100.0f <= requiredItem.MaxCondition)
611  {
612  availablePrefabsAmount++;
613  }
614 
615  if (availablePrefabsAmount >= requiredItem.Amount)
616  {
617  return true;
618  }
619  }
620  }
621 
622  return false;
623  });
624  }

626  private float GetRequiredTime(FabricationRecipe fabricableItem, Character user)
627  {
628  float degreeOfSuccess = FabricationDegreeOfSuccess(user, fabricableItem.RequiredSkills);
629 
630  float t = degreeOfSuccess < 0.5f ? degreeOfSuccess * degreeOfSuccess : degreeOfSuccess * 2;
631 
632  //fabricating takes 100 times longer if degree of success is close to 0
633  //characters with a higher skill than required can fabricate up to 100% faster
634  return fabricableItem.RequiredTime / FabricationSpeed / MathHelper.Clamp(t, 0.01f, 2.0f);
635  }

637  public float FabricationDegreeOfSuccess(Character character, ImmutableArray<Skill> skills)
638  {
639  if (skills.Length == 0) { return 1.0f; }
640  if (character == null) { return 0.0f; }
641 
642  float skillSum = (from t in skills let characterLevel = character.GetSkillLevel(t.Identifier) select (characterLevel - (t.Level * SkillRequirementMultiplier))).Sum();
643  float average = skillSum / skills.Length;
644 
645  return (average + 100.0f) / 2.0f / 100.0f;
646  }

710  private void MoveIngredientsToInputContainer(FabricationRecipe targetItem)
711  {
712  //required ingredients that are already present in the input container
713  List<Item> usedItems = new List<Item>();
714 
715  targetItem.RequiredItems.ForEach(requiredItem => {
716  for (int i = 0; i < requiredItem.Amount; i++)
717  {
718  foreach (ItemPrefab requiredPrefab in requiredItem.ItemPrefabs)
719  {
720  if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
721 
722  var availablePrefabs = availableIngredients[requiredPrefab.Identifier];
723  var availablePrefab = availablePrefabs.FirstOrDefault(potentialPrefab =>
724  {
725  return !usedItems.Contains(potentialPrefab) &&
726  potentialPrefab.ConditionPercentage >= requiredItem.MinCondition * 100.0f &&
727  potentialPrefab.ConditionPercentage <= requiredItem.MaxCondition * 100.0f;
728  });
729  if (availablePrefab == null) { continue; }
730 
731  availablePrefabs.Remove(availablePrefab);
732 
733  if (availablePrefab.ParentInventory == inputContainer.Inventory)
734  {
735  //already in input container, all good
736  usedItems.Add(availablePrefab);
737  }
738  else //in another inventory, we need to move the item
739  {
