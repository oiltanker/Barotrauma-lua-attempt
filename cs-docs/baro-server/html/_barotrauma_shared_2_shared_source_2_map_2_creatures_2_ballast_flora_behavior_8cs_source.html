505  public void Update(float deltaTime)
506  {
507  if (GameMain.NetworkMember == null || !GameMain.NetworkMember.IsClient)
508  {
509  if (Branches.Count == 0)
510  {
511  Remove();
512  return;
513  }
514  }
515 
516  foreach (BallastFloraBranch branch in Branches)
517  {
518  branch.UpdateScale(deltaTime);
519  branch.UpdatePulse(deltaTime, PulseInflateSpeed, PulseDeflateSpeed, PulseDelay);
520 #if CLIENT
521  branch.UpdateHealth();
522 #endif
523  }
524 
525  UpdateDamage(deltaTime);
526 
527  UpdatePowerDrain(deltaTime);
528 
529  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
530 
531  if (root != null && HealthRegenPerBranch > 0.0f)
532  {
533  float healAmount = Branches.Count(b => !b.IsRoot && !b.IsRootGrowth && !b.DisconnectedFromRoot) * HealthRegenPerBranch;
534 
535  foreach (BallastFloraBranch branch in Branches)
536  {
537  if (branch.Health > branch.MaxHealth * 0.9f || branch.DisconnectedFromRoot) { continue; }
538  float branchHealAmount = (float)(MaxBranchHealthRegenDistance - branch.BranchDepth) / MaxBranchHealthRegenDistance * healAmount;
539  if (branchHealAmount <= 0.0f) { continue; }
540  float prevHealth = branch.Health;
541  branch.Health += branchHealAmount;
542  branch.AccumulatedDamage += (prevHealth - branch.Health);
543  }
544  }
545  StateMachine.Update(deltaTime);
546 
547  if (HasBrokenThrough)
548  {
549  // I wasn't 100% sure what the performance impact on this so I decide to limit it to only check every 5 seconds
550  if (fireCheckCooldown <= 0)
551  {
552  UpdateFireSources();
553  fireCheckCooldown = 5f;
554  }
555  else
556  {
557  fireCheckCooldown -= deltaTime;
558  }
559 
560  foreach (BallastFloraBranch branch in branchesVulnerableToFire)
561  {
562  if (!branch.Removed)
563  {
564  DamageBranch(branch, FireVulnerability * deltaTime, AttackType.Fire, null);
565  }
566  }
567  }
568 
569  UpdateSelfDamage(deltaTime);
570 
571  if (Anger > 1f)
572  {
573  Anger -= deltaTime;
574  }
575 
576  if (toxinsTimer > 0.1f)
577  {
578  toxinsSpawnTimer -= deltaTime;
579  if (!AttackItemPrefab.IsEmpty && toxinsSpawnTimer <= 0.0f)
580  {
581  toxinsSpawnTimer = 1.0f;
582  Dictionary<Hull, List<BallastFloraBranch>> branches = new Dictionary<Hull, List<BallastFloraBranch>>();
583  foreach (BallastFloraBranch branch in Branches)
584  {
585  if (branch.CurrentHull == null || branch.FlowerConfig.Variant < 0 || branch.DisconnectedFromRoot) { continue; }
586 
587  if (branches.TryGetValue(branch.CurrentHull, out List<BallastFloraBranch>? list))
588  {
589  list.Add(branch);
590  }
591  else
592  {
593  branches.Add(branch.CurrentHull, new List<BallastFloraBranch> { branch });
594  }
595  }
596 
597  foreach (Hull hull in branches.Keys)
598  {
599  List<BallastFloraBranch> list = branches[hull];
600  if (!list.Any(HasAcidEmitter))
601  {
602  BallastFloraBranch randomBranch = branches[hull].GetRandomUnsynced();
603  randomBranch.SpawningItem = true;
604 
605  ItemPrefab prefab = ItemPrefab.Find(null, AttackItemPrefab);
606 #warning TODO: Parent needs a nullability sanity check
607  Entity.Spawner?.AddItemToSpawnQueue(prefab, Parent!.Position + Offset + randomBranch.Position, Parent.Submarine, onSpawned: item =>
608  {
609  randomBranch.AttackItem = item;
610  randomBranch.SpawningItem = false;
611  });
612  }

615  }
616  }
617 
618  toxinsTimer -= deltaTime;
619  }
620 
621  if (defenseCooldown >= 0)
622  {
623  defenseCooldown -= deltaTime;
624  }
625 
626  if (toxinsCooldown >= 0)

628  toxinsCooldown -= deltaTime;

634  private readonly List<BallastFloraBranch> toBeRemoved = new List<BallastFloraBranch>();
635  private void UpdateSelfDamage(float deltaTime)
636  {
637  if (selfDamageTimer <= 0)
638  {
639  if (!HasBrokenThrough && !CanGrowMore())
640  {
641  Branches.ForEachMod(branch =>
642  {
643  float maxHealth = branch.IsRoot ? RootHealth : BranchHealth;
644  DamageBranch(branch, Rand.Range(1f, maxHealth), AttackType.Other);
645  });
646  }
647  selfDamageTimer = 1f;
648  }
649  toBeRemoved.Clear();
650  foreach (BallastFloraBranch branch in Branches)
651  {
652  if (branch.ParentBranch != null && (branch.ParentBranch.DisconnectedFromRoot || branch.ParentBranch.Health <= 0.0f))
653  {
654  float speed = MathHelper.Lerp(5.0f, 0.1f, branch.ParentBranch.Health / branch.ParentBranch.MaxHealth);
655  DamageBranch(branch, speed * speed * deltaTime, AttackType.CutFromRoot);
656  }
657  if (branch.Health <= 0.0f)
658  {
659  if (branch.ClaimedItem != null)
660  {
661  RemoveClaim(branch.ClaimedItem);
662  branch.ClaimedItem = null;
663  }
664 
665  branch.RemoveTimer -= deltaTime;
666  if (branch.RemoveTimer <= 0.0f)
667  {
668  toBeRemoved.Add(branch);
669  }
670  }
671  }
672  foreach (BallastFloraBranch branch in toBeRemoved)
673  {
674  RemoveBranch(branch);
675  }
676  selfDamageTimer -= deltaTime;
677  }

679  private void UpdatePowerDrain(float deltaTime)
680  {
681  PowerConsumptionTimer += deltaTime;
682  if (PowerConsumptionTimer > PowerConsumptionDuration)
683  {
684  PowerConsumptionTimer = 0f;
685  }
686 
687  float powerConsumption = MathHelper.Lerp(PowerConsumptionMax, PowerConsumptionMin, PowerConsumptionTimer / PowerConsumptionDuration);
688  float powerDelta = powerConsumption * deltaTime;
689 
690  foreach (PowerTransfer jb in ClaimedJunctionBoxes)
691  {
