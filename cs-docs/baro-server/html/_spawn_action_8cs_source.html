250  public static WayPoint GetSpawnPos(SpawnLocationType spawnLocation, SpawnType? spawnPointType, IEnumerable<Identifier> moduleFlags = null, IEnumerable<Identifier> spawnpointTags = null, bool asFarAsPossibleFromAirlock = false)
251  {
252  List<WayPoint> potentialSpawnPoints = spawnLocation switch
253  {
254  SpawnLocationType.MainSub => WayPoint.WayPointList.FindAll(wp => wp.Submarine == Submarine.MainSub && wp.CurrentHull != null),
255  SpawnLocationType.MainPath => WayPoint.WayPointList.FindAll(wp => wp.Submarine == null),
256  SpawnLocationType.Outpost => WayPoint.WayPointList.FindAll(wp => wp.Submarine?.Info != null && wp.CurrentHull != null && wp.Submarine.Info.IsOutpost),
257  SpawnLocationType.Wreck => WayPoint.WayPointList.FindAll(wp => wp.Submarine?.Info != null && wp.Submarine.Info.IsWreck),
258  SpawnLocationType.Ruin => WayPoint.WayPointList.FindAll(wp => wp.Submarine?.Info != null && wp.Submarine.Info.IsRuin),
259  SpawnLocationType.BeaconStation => WayPoint.WayPointList.FindAll(wp => wp.Submarine?.Info != null && wp.Submarine.Info.IsBeacon),
260  _ => throw new NotImplementedException()
261  };
262 
263  potentialSpawnPoints = potentialSpawnPoints.FindAll(wp => wp.ConnectedDoor == null && wp.Ladders == null && !wp.isObstructed);
264 
265  if (moduleFlags != null && moduleFlags.Any())
266  {
267  List<WayPoint> spawnPoints = potentialSpawnPoints.Where(wp => wp.CurrentHull?.OutpostModuleTags.Any(moduleFlags.Contains) ?? false).ToList();
268  if (spawnPoints.Any())
269  {
270  potentialSpawnPoints = spawnPoints;
271  }
272  }
273 
274  if (spawnpointTags != null && spawnpointTags.Any())
275  {
276  var spawnPoints = potentialSpawnPoints
277  .Where(wp => spawnpointTags.Any(tag => wp.Tags.Contains(tag)))
278  .Where(wp => wp.ConnectedDoor == null && !wp.isObstructed);
279 
280  if (spawnPoints.Any())
281  {
282  potentialSpawnPoints = spawnPoints.ToList();
283  }
284  }
285 
286  if (potentialSpawnPoints.Count == 0)
287  {
288  DebugConsole.ThrowError($"Could not find a spawn point for a SpawnAction (spawn location: {spawnLocation})");
289  return null;
290  }
291 
292  IEnumerable<WayPoint> validSpawnPoints;
293  if (spawnPointType.HasValue)
294  {
295  validSpawnPoints = potentialSpawnPoints.FindAll(wp => spawnPointType.Value.HasFlag(wp.SpawnType));
296  }
297  else
298  {
299  validSpawnPoints = potentialSpawnPoints.FindAll(wp => wp.SpawnType != SpawnType.Path);
300  if (!validSpawnPoints.Any()) { validSpawnPoints = potentialSpawnPoints; }
301  }
302 
303  //don't spawn in an airlock module if there are other options
304  var airlockSpawnPoints = potentialSpawnPoints.Where(wp => wp.CurrentHull?.OutpostModuleTags.Contains("airlock".ToIdentifier()) ?? false);
305  if (airlockSpawnPoints.Count() < validSpawnPoints.Count())
306  {
307  validSpawnPoints = validSpawnPoints.Except(airlockSpawnPoints);
308  }
309 
310  if (!validSpawnPoints.Any())
311  {
312  DebugConsole.ThrowError($"Could not find a spawn point of the correct type for a SpawnAction (spawn location: {spawnLocation}, type: {spawnPointType}, module flags: {((moduleFlags == null || !moduleFlags.Any()) ? "none" : string.Join(", ", moduleFlags))})");
313  return potentialSpawnPoints.GetRandomUnsynced();
314  }
315 
316  //avoid using waypoints if there's any actual spawnpoints available
317  if (validSpawnPoints.Any(wp => wp.SpawnType != SpawnType.Path))
318  {
319  validSpawnPoints = validSpawnPoints.Where(wp => wp.SpawnType != SpawnType.Path);
320  }
321 
322  //if not trying to spawn at a tagged spawnpoint, favor spawnpoints without tags
323  if (spawnpointTags == null || !spawnpointTags.Any())
324  {
325  var spawnPoints = validSpawnPoints.Where(wp => !wp.Tags.Any());
326  if (spawnPoints.Any())
327  {
328  validSpawnPoints = spawnPoints.ToList();
329  }
330  }
331 
332  if (asFarAsPossibleFromAirlock && airlockSpawnPoints.Any())
333  {
334  WayPoint furthestPoint = validSpawnPoints.First();
335  float furthestDist = 0.0f;
336  foreach (WayPoint waypoint in validSpawnPoints)
337  {
338  float dist = Vector2.DistanceSquared(waypoint.WorldPosition, airlockSpawnPoints.First().WorldPosition);
339  if (dist > furthestDist)
340  {
341  furthestDist = dist;
342  furthestPoint = waypoint;
343  }
344  }
345  return furthestPoint;
346  }
347  else
348  {
349  return validSpawnPoints.GetRandomUnsynced();
350  }
351  }
