21 public CharacterAbilityApplyStatusEffects(CharacterAbilityGroup characterAbilityGroup, ContentXElement abilityElement) : base(characterAbilityGroup, abilityElement)

31 protected void ApplyEffectSpecific(Character targetCharacter)
32 {
33 foreach (var statusEffect in statusEffects)
34 {
35 if (statusEffect.HasTargetType(StatusEffect.TargetType.UseTarget))
36 {
37 // currently used to spawn items on the targeted character
38 statusEffect.SetUser(targetCharacter);
39 statusEffect.Apply(ActionType.OnAbility, EffectDeltaTime, targetCharacter, targetCharacter);
40 }
41 else if (statusEffect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
42 {
43 targets.Clear();
44 targets.AddRange(statusEffect.GetNearbyTargets(targetCharacter.WorldPosition, targets));
45 if (!nearbyCharactersAppliesToSelf)
46 {
47 targets.RemoveAll(c => c == Character);
48 }
49 if (!nearbyCharactersAppliesToAllies)
50 {
51 targets.RemoveAll(c => c is Character otherCharacter && HumanAIController.IsFriendly(otherCharacter, Character));
52 }
53 if (!nearbyCharactersAppliesToEnemies)
54 {
55 targets.RemoveAll(c => c is Character otherCharacter && !HumanAIController.IsFriendly(otherCharacter, Character));
56 }
57 statusEffect.SetUser(Character);
58 statusEffect.Apply(ActionType.OnAbility, EffectDeltaTime, targetCharacter, targets);
59 }
60 else if (statusEffect.HasTargetType(StatusEffect.TargetType.Character))
61 {
62 statusEffect.SetUser(Character);
63 statusEffect.Apply(ActionType.OnAbility, EffectDeltaTime, Character, targetCharacter);
64 }
65 else
66 {
67 statusEffect.SetUser(Character);
68 statusEffect.Apply(ActionType.OnAbility, EffectDeltaTime, Character, Character);
69 }
70 }
71 }
72 protected override void ApplyEffect()
73 {
74 if (applyToSelected && Character.SelectedCharacter is Character selectedCharacter)
75 {
76 ApplyEffectSpecific(selectedCharacter);
77 }
78 else
79 {
80 ApplyEffectSpecific(Character);
81 }
82 }
83
84 protected override void ApplyEffect(AbilityObject abilityObject)
85 {
86 if ((abilityObject as IAbilityCharacter)?.Character is Character targetCharacter && !applyToSelf)
87 {
88 ApplyEffectSpecific(targetCharacter);
89 }
90 else
91 {
92 ApplyEffect();
93 }
94 }
