445  void UpdateSineAnim(float deltaTime)
446  {
447  if (CurrentSwimParams == null) { return; }
448  movement = TargetMovement;
449  bool isMoving = movement.LengthSquared() > 0.00001f;
450  var mainLimb = MainLimb;
451  float t = 0.5f;
452  if (isMoving && !SimplePhysicsEnabled && CurrentSwimParams.RotateTowardsMovement)
453  {
454  Vector2 forward = VectorExtensions.Forward(Collider.Rotation + MathHelper.PiOver2);
455  float dot = Vector2.Dot(forward, Vector2.Normalize(movement));
456  if (dot < 0)
457  {
458  // Reduce the linear movement speed when not facing the movement direction
459  t = MathHelper.Clamp((1 + dot) / 10, 0.01f, 0.1f);
460  }
461  }
462  Collider.LinearVelocity = Vector2.Lerp(Collider.LinearVelocity, movement, t);
463  //limbs are disabled when simple physics is enabled, no need to move them
464  if (SimplePhysicsEnabled) { return; }
465  mainLimb.PullJointEnabled = true;
466 
467  if (aiming && movement.Length() <= 0.1f)
468  {
469  Vector2 mousePos = ConvertUnits.ToSimUnits(character.CursorPosition);
470  Vector2 diff = (mousePos - (GetLimb(LimbType.Torso) ?? MainLimb).SimPosition) * Dir;
471  TargetMovement = new Vector2(0.0f, -0.1f);
472  float newRotation = MathUtils.VectorToAngle(diff);
473  Collider.SmoothRotate(newRotation, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
474  }
475 
476  if (!isMoving)
477  {
478  WalkPos = MathHelper.SmoothStep(WalkPos, MathHelper.PiOver2, deltaTime * 5);
479  mainLimb.PullJointWorldAnchorB = Collider.SimPosition;
480  }
481  else
482  {
483  Vector2 transformedMovement = reverse ? -movement : movement;
484  float movementAngle = MathUtils.VectorToAngle(transformedMovement) - MathHelper.PiOver2;
485  float mainLimbAngle = 0;
486  if (mainLimb.type == LimbType.Torso && TorsoAngle.HasValue)
487  {
488  mainLimbAngle = TorsoAngle.Value;
489  }
490  else if (mainLimb.type == LimbType.Head && HeadAngle.HasValue)
491  {
492  mainLimbAngle = HeadAngle.Value;
493  }
494  mainLimbAngle *= Dir;
495  while (mainLimb.Rotation - (movementAngle + mainLimbAngle) > MathHelper.Pi)
496  {
497  movementAngle += MathHelper.TwoPi;
498  }
499  while (mainLimb.Rotation - (movementAngle + mainLimbAngle) < -MathHelper.Pi)
500  {
501  movementAngle -= MathHelper.TwoPi;
502  }
503  if (CurrentSwimParams.RotateTowardsMovement)
504  {
505  Collider.SmoothRotate(movementAngle, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
506  if (TorsoAngle.HasValue)
507  {
508  Limb torso = GetLimb(LimbType.Torso);
509  if (torso != null)
510  {
511  SmoothRotateWithoutWrapping(torso, movementAngle + TorsoAngle.Value * Dir, mainLimb, TorsoTorque);
512  }
513  }
514  if (HeadAngle.HasValue)
515  {
516  Limb head = GetLimb(LimbType.Head);
517  if (head != null)
518  {
519  SmoothRotateWithoutWrapping(head, movementAngle + HeadAngle.Value * Dir, mainLimb, HeadTorque);
520  }
521  }
522  if (TailAngle.HasValue)
523  {
524  bool isAngleApplied = false;
525  foreach (var limb in Limbs)
526  {
527  if (limb.IsSevered) { continue; }
528  if (limb.type != LimbType.Tail) { continue; }
529  if (!limb.Params.ApplyTailAngle) { continue; }
530  RotateTail(limb);

538  void RotateTail(Limb tail)
539  {
540  if (tail == null) { return; }
541  float? mainLimbTargetAngle = null;
542  if (mainLimb.type == LimbType.Torso)
543  {
544  mainLimbTargetAngle = TorsoAngle;
545  }
546  else if (mainLimb.type == LimbType.Head)
547  {
548  mainLimbTargetAngle = HeadAngle;
549  }
550  float torque = TailTorque;
551  float maxMultiplier = CurrentSwimParams.TailTorqueMultiplier;
552  if (mainLimbTargetAngle.HasValue && maxMultiplier > 1)
553  {
554  float diff = Math.Abs(mainLimb.Rotation - tail.Rotation);
555  float offset = Math.Abs(mainLimbTargetAngle.Value - TailAngle.Value);
556  torque *= MathHelper.Lerp(1, maxMultiplier, MathUtils.InverseLerp(0, MathHelper.PiOver2, diff - offset));
557  }
558  SmoothRotateWithoutWrapping(tail, movementAngle + TailAngle.Value * Dir, mainLimb, torque);
559  }
560  }
561  }
562  else
563  {
564  movementAngle = Dir > 0 ? -MathHelper.PiOver2 : MathHelper.PiOver2;
565  if (reverse)

574  {
575  Collider.SmoothRotate(TorsoAngle.Value * Dir, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
576  }

638  }

644  {
645  var limb = Limbs[i];
646  if (limb.IsSevered) { continue; }
647  if (limb.SteerForce <= 0.0f) { continue; }
648  if (!Collider.PhysEnabled) { continue; }
649  Vector2 pullPos = limb.PullJointWorldAnchorA;
