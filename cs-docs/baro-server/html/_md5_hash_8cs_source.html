22 public static void Load()
23 {
24 if (!File.Exists(cachePath)) { return; }
25 var lines = File.ReadAllLines(cachePath);
26 if (Version.TryParse(lines[0], out var cacheVersion) && cacheVersion == GameMain.Version)
27 {
28 for (int i = 1; i < lines.Length; i++)
29 {
30 string[] split = lines[i].Split('|');
31 string path = split[0].CleanUpPathCrossPlatform();
32 Md5Hash hash = Md5Hash.StringAsHash(split[1]);
33 DateTime? dateTime = null;

37 }
38
39 if (File.Exists(path) && dateTime.HasValue && dateTime >= File.GetLastWriteTime(path))
40 {

42 }
43 }

167 public static Md5Hash CalculateForFile(string path, StringHashOptions options)
168 {
169 if (options.HasFlag(StringHashOptions.IgnoreWhitespace) || options.HasFlag(StringHashOptions.IgnoreCase))

181 public static Md5Hash CalculateForString(string str, StringHashOptions options)
182 {
183 if (options.HasFlag(StringHashOptions.IgnoreCase))
184 {
185 str = str.ToLowerInvariant();
186 }
187 if (options.HasFlag(StringHashOptions.IgnoreWhitespace))
188 {
189 str = RemoveWhitespace(str);
190 }
191 byte[] bytes = Encoding.UTF8.GetBytes(str);
192 return CalculateForBytes(bytes);
