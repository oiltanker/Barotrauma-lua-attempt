78 private static List<Identifier> GetCurrentFlags(Character speaker)
79 {
80 var currentFlags = new List<Identifier>();
81 if (Submarine.MainSub != null && Submarine.MainSub.AtDamageDepth) { currentFlags.Add("SubmarineDeep".ToIdentifier()); }
82
83 if (GameMain.GameSession != null && Level.Loaded != null)
84 {
85 if (Level.Loaded.Type == LevelData.LevelType.LocationConnection)
86 {
87 if (Timing.TotalTime < GameMain.GameSession.RoundStartTime + 30.0f) { currentFlags.Add("Initial".ToIdentifier()); }
88 }
89 else if (Level.Loaded.Type == LevelData.LevelType.Outpost)
90 {
91 if (Timing.TotalTime < GameMain.GameSession.RoundStartTime + 120.0f &&
92 speaker?.CurrentHull != null &&
93 (speaker.TeamID == CharacterTeamType.FriendlyNPC || speaker.TeamID == CharacterTeamType.None) &&
94 Character.CharacterList.Any(c => c.TeamID != speaker.TeamID && c.CurrentHull == speaker.CurrentHull))
95 {
96 currentFlags.Add("EnterOutpost".ToIdentifier());
97 }
98 }
99 if (GameMain.GameSession.EventManager.CurrentIntensity <= 0.2f)
100 {
101 currentFlags.Add("Casual".ToIdentifier());
102 }
103
104 if (GameMain.GameSession.IsCurrentLocationRadiated())
105 {
106 currentFlags.Add("InRadiation".ToIdentifier());
107 }
108 }
109
110 if (speaker != null)
111 {
112 if (speaker.AnimController.InWater) { currentFlags.Add("Underwater".ToIdentifier()); }
113 currentFlags.Add((speaker.CurrentHull == null ? "Outside" : "Inside").ToIdentifier());
114
115 if (Character.Controlled != null)
116 {
117 if (Character.Controlled.CharacterHealth.GetAffliction("psychosis") != null)
118 {
119 currentFlags.Add((speaker != Character.Controlled ? "Psychosis" : "PsychosisSelf").ToIdentifier());
120 }
121 }
122
123 var afflictions = speaker.CharacterHealth.GetAllAfflictions();
124 foreach (Affliction affliction in afflictions)
125 {
126 var currentEffect = affliction.GetActiveEffect();
127 if (currentEffect != null && !string.IsNullOrEmpty(currentEffect.DialogFlag.Value) && !currentFlags.Contains(currentEffect.DialogFlag))
128 {
129 currentFlags.Add(currentEffect.DialogFlag);
130 }
131 }
132
133 if (speaker.TeamID == CharacterTeamType.FriendlyNPC && speaker.Submarine != null && speaker.Submarine.Info.IsOutpost)
134 {
135 currentFlags.Add("OutpostNPC".ToIdentifier());
136 }
137 if (speaker.CampaignInteractionType != CampaignMode.InteractionType.None)
138 {
139 currentFlags.Add($"CampaignNPC.{speaker.CampaignInteractionType}".ToIdentifier());
140 }
141 if (GameMain.GameSession?.GameMode is CampaignMode campaignMode &&
142 (campaignMode.Map?.CurrentLocation?.Type?.Identifier == "abandoned"))
143 {
144 if (speaker.TeamID == CharacterTeamType.None)
145 {
146 currentFlags.Add("Bandit".ToIdentifier());
147 }
148 else if (speaker.TeamID == CharacterTeamType.FriendlyNPC)
149 {
150 currentFlags.Add("Hostage".ToIdentifier());
151 }
152 }
153 if (speaker.IsEscorted)
154 {
155 currentFlags.Add("escort".ToIdentifier());
156 }
157 }
158
159 return currentFlags;
160 }

182 {
183 Dictionary<int, Character> assignedSpeakers = new Dictionary<int, Character>();
184 List<(Character speaker, string line)> lines = new List<(Character speaker, string line)>();
185

190 language = TextManager.DefaultLanguage;

202 private static void CreateConversation(
203 List<Character> availableSpeakers,
204 Dictionary<int, Character> assignedSpeakers,
205 NPCConversation baseConversation,
206 IList<(Character speaker, string line)> lineList,
207 IList<NPCConversation> availableConversations,
208 bool ignoreFlags = false)
209 {
210 IList<NPCConversation> conversations = baseConversation == null ? availableConversations : baseConversation.Responses;
211 if (conversations.Count == 0) { return; }
212
213 int conversationIndex = Rand.Int(conversations.Count);
214 NPCConversation selectedConversation = conversations[conversationIndex];
215 if (string.IsNullOrEmpty(selectedConversation.Line)) { return; }
216
217 Character speaker = null;
218 //speaker already assigned for this line
219 if (assignedSpeakers.ContainsKey(selectedConversation.speakerIndex))
220 {
221 //check if the character has all required flags to say the line
222 var characterFlags = GetCurrentFlags(assignedSpeakers[selectedConversation.speakerIndex]);
223 if (selectedConversation.Flags.All(flag => characterFlags.Contains(flag)))
224 {
225 speaker = assignedSpeakers[selectedConversation.speakerIndex];
226 }
227 }
228 if (speaker == null)
229 {
230 var allowedSpeakers = new List<Character>();
231
232 List<NPCConversation> potentialLines = new List<NPCConversation>(conversations);
233
234 //remove lines that are not appropriate for the intensity of the current situation
235 if (GameMain.GameSession?.EventManager != null)
236 {
237 potentialLines.RemoveAll(l =>
238 (l.minIntensity.HasValue && GameMain.GameSession.EventManager.CurrentIntensity < l.minIntensity) ||
239 (l.maxIntensity.HasValue && GameMain.GameSession.EventManager.CurrentIntensity > l.maxIntensity));
240 }
241
242 while (potentialLines.Count > 0)
243 {
244 //select a random line and attempt to find a speaker for it
245 // and if no valid speaker is found, choose another random line
246 selectedConversation = GetRandomConversation(potentialLines, baseConversation == null);
247 if (selectedConversation == null || string.IsNullOrEmpty(selectedConversation.Line)) { return; }
248
249 //speaker already assigned for this line
250 if (assignedSpeakers.ContainsKey(selectedConversation.speakerIndex))
251 {
252 speaker = assignedSpeakers[selectedConversation.speakerIndex];
253 break;
254 }
255
256 foreach (Character potentialSpeaker in availableSpeakers)
257 {
258 if (CheckSpeakerViability(potentialSpeaker, selectedConversation, assignedSpeakers.Values.ToList(), ignoreFlags))
259 {
260 allowedSpeakers.Add(potentialSpeaker);
261 }
262 }
263
264 if (allowedSpeakers.Count == 0 || NextLineFailure(selectedConversation, availableSpeakers, allowedSpeakers, ignoreFlags))
265 {
266 allowedSpeakers.Clear();
267 potentialLines.Remove(selectedConversation);
268 }
269 else
270 {
271 break;
272 }
273 }
274
275 if (allowedSpeakers.Count == 0) { return; }
276 speaker = allowedSpeakers[Rand.Int(allowedSpeakers.Count)];
277 availableSpeakers.Remove(speaker);
278 assignedSpeakers.Add(selectedConversation.speakerIndex, speaker);
279 }
280
281 if (baseConversation == null)
282 {
283 previousConversations.Insert(0, selectedConversation);
284 if (previousConversations.Count > MaxPreviousConversations) previousConversations.RemoveAt(MaxPreviousConversations);
285 }
286 lineList.Add((speaker, selectedConversation.Line));
287 CreateConversation(availableSpeakers, assignedSpeakers, selectedConversation, lineList, availableConversations);
288 }

290 static bool NextLineFailure(NPCConversation selectedConversation, List<Character> availableSpeakers, List<Character> allowedSpeakers, bool ignoreFlags)
291 {
292 if (selectedConversation.requireNextLine)
293 {
294 foreach (NPCConversation nextConversation in selectedConversation.Responses)
295 {
296 foreach (Character potentialNextSpeaker in availableSpeakers)
297 {
298 if (CheckSpeakerViability(potentialNextSpeaker, nextConversation, allowedSpeakers, ignoreFlags))
299 {
300 return false;
301 }
302 }
303 }
304 return true;
305 }
306 return false;
307 }

309 static bool CheckSpeakerViability(Character potentialSpeaker, NPCConversation selectedConversation, List<Character> checkedSpeakers, bool ignoreFlags)
310 {
311 //check if the character has an appropriate job to say the line
312 if ((potentialSpeaker.Info?.Job != null && potentialSpeaker.Info.Job.Prefab.OnlyJobSpecificDialog) || selectedConversation.AllowedJobs.Count > 0)
313 {
314 if (!(potentialSpeaker.Info?.Job?.Prefab is { } speakerJobPrefab)
315 || !selectedConversation.AllowedJobs.Contains(speakerJobPrefab.Identifier)) { return false; }
316 }
317
318 //check if the character has all required flags to say the line
319 if (!ignoreFlags)
320 {
321 var characterFlags = GetCurrentFlags(potentialSpeaker);
322 if (!selectedConversation.Flags.All(flag => characterFlags.Contains(flag))) { return false; }
323 }
324
325 //check if the character is close enough to hear the rest of the speakers
326 if (checkedSpeakers.Any(s => !potentialSpeaker.CanHearCharacter(s))) { return false; }
327
328 //check if the character is close enough to see the rest of the speakers (this should be replaced with a more performant method)
329 if (checkedSpeakers.Any(s => !potentialSpeaker.CanSeeCharacter(s))) { return false; }
330
331 //check if the character has an appropriate personality
332 if (selectedConversation.allowedSpeakerTags.Count > 0)
333 {
334 if (potentialSpeaker.Info?.PersonalityTrait == null) { return false; }
335 if (!selectedConversation.allowedSpeakerTags.Any(t => potentialSpeaker.Info.PersonalityTrait.AllowedDialogTags.Any(t2 => t2 == t))) { return false; }
336 }
337 else
338 {
339 if (potentialSpeaker.Info?.PersonalityTrait != null &&
340 !potentialSpeaker.Info.PersonalityTrait.AllowedDialogTags.Contains("none"))
341 {
342 return false;
343 }
344 }
345 return true;

347 private static NPCConversation GetRandomConversation(List<NPCConversation> conversations, bool avoidPreviouslyUsed)
348 {
349 if (!avoidPreviouslyUsed)
350 {
351 return conversations.Count == 0 ? null : conversations[Rand.Int(conversations.Count)];
352 }
353
354 List<float> probabilities = new List<float>();
355 foreach (NPCConversation conversation in conversations)
356 {
357 probabilities.Add(GetConversationProbability(conversation));
358 }
359 return ToolBox.SelectWeightedRandom(conversations, probabilities, Rand.RandSync.Unsynced);
360 }

362 private static float GetConversationProbability(NPCConversation conversation)
363 {
364 int index = previousConversations.IndexOf(conversation);
365 if (index < 0) return 10.0f;
366
367 return 1.0f - 1.0f / (index + 1);
368 }
