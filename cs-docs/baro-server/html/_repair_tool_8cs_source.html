168  public override bool Use(float deltaTime, Character character = null)
169  {
170  if (character != null)
171  {
172  if (item.RequireAimToUse && !character.IsKeyDown(InputType.Aim)) { return false; }
173  }
174 
175  float degreeOfSuccess = character == null ? 0.5f : DegreeOfSuccess(character);
176 
177  if (Rand.Range(0.0f, 0.5f) > degreeOfSuccess)
178  {
179  ApplyStatusEffects(ActionType.OnFailure, deltaTime, character);
180  return false;
181  }
182 
183  if (UsableIn == UseEnvironment.None)
184  {
185  ApplyStatusEffects(ActionType.OnFailure, deltaTime, character);
186  return false;
187  }
188 
189  if (item.InWater)
190  {
191  if (UsableIn == UseEnvironment.Air)
192  {
193  ApplyStatusEffects(ActionType.OnFailure, deltaTime, character);
194  return false;
195  }
196  }
197  else
198  {
199  if (UsableIn == UseEnvironment.Water)
200  {
201  ApplyStatusEffects(ActionType.OnFailure, deltaTime, character);
202  return false;

206  Vector2 rayStart;
207  Vector2 rayStartWorld;
208  Vector2 sourcePos = character?.AnimController == null ? item.SimPosition : character.AnimController.AimSourceSimPos;
209  Vector2 barrelPos = item.SimPosition + ConvertUnits.ToSimUnits(TransformedBarrelPos);
210  //make sure there's no obstacles between the base of the item (or the shoulder of the character) and the end of the barrel
211  if (Submarine.PickBody(sourcePos, barrelPos, collisionCategory: Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionItemBlocking) == null)
212  {
213  //no obstacles -> we start the raycast at the end of the barrel
214  rayStart = ConvertUnits.ToSimUnits(item.Position + TransformedBarrelPos);
215  rayStartWorld = ConvertUnits.ToSimUnits(item.WorldPosition + TransformedBarrelPos);
216  }
217  else
218  {
219  rayStart = rayStartWorld = Submarine.LastPickedPosition + Submarine.LastPickedNormal * 0.1f;
220  if (item.Submarine != null) { rayStartWorld += item.Submarine.SimPosition; }
221  }
222 
223  //if the calculated barrel pos is in another hull, use the origin of the item to make sure the particles don't end up in an incorrect hull
224  if (item.CurrentHull != null)
225  {
226  var barrelHull = Hull.FindHull(ConvertUnits.ToDisplayUnits(rayStartWorld), item.CurrentHull, useWorldCoordinates: true);
227  if (barrelHull != null && barrelHull != item.CurrentHull)
228  {
229  if (MathUtils.GetLineRectangleIntersection(ConvertUnits.ToDisplayUnits(sourcePos), ConvertUnits.ToDisplayUnits(rayStart), item.CurrentHull.Rect, out Vector2 hullIntersection))
230  {
231  if (!item.CurrentHull.ConnectedGaps.Any(g => g.Open > 0.0f && Submarine.RectContains(g.Rect, hullIntersection)))
232  {
233  Vector2 rayDir = rayStart.NearlyEquals(sourcePos) ? Vector2.Zero : Vector2.Normalize(rayStart - sourcePos);
234  rayStartWorld = ConvertUnits.ToSimUnits(hullIntersection - rayDir * 5.0f);
235  if (item.Submarine != null) { rayStartWorld += item.Submarine.SimPosition; }
236  }
237  }
238  }
239  }
240 
241  float spread = MathHelper.ToRadians(MathHelper.Lerp(UnskilledSpread, Spread, degreeOfSuccess));
242 
243  float angle = MathHelper.ToRadians(BarrelRotation) + spread * Rand.Range(-0.5f, 0.5f);
244  float dir = 1;
245  if (item.body != null)
246  {
247  angle += item.body.Rotation;
248  dir = item.body.Dir;
249  }
250  Vector2 rayEnd = rayStartWorld + ConvertUnits.ToSimUnits(new Vector2((float)Math.Cos(angle), (float)Math.Sin(angle)) * Range * dir);
251 
252  ignoredBodies.Clear();
253  if (character != null)
254  {
255  foreach (Limb limb in character.AnimController.Limbs)
256  {
257  if (Rand.Range(0.0f, 0.5f) > degreeOfSuccess) continue;
258  ignoredBodies.Add(limb.body.FarseerBody);
259  }
260  ignoredBodies.Add(character.AnimController.Collider.FarseerBody);
261  }
262 
263  IsActive = true;
264  activeTimer = 0.1f;
265 
266  debugRayStartPos = ConvertUnits.ToDisplayUnits(rayStartWorld);
267  debugRayEndPos = ConvertUnits.ToDisplayUnits(rayEnd);
268 
269  Submarine parentSub = character?.Submarine ?? item.Submarine;
270  if (parentSub == null)
271  {
272  foreach (Submarine sub in Submarine.Loaded)
273  {
274  Rectangle subBorders = sub.Borders;
275  subBorders.Location += new Point((int)sub.WorldPosition.X, (int)sub.WorldPosition.Y - sub.Borders.Height);
276  if (!MathUtils.CircleIntersectsRectangle(item.WorldPosition, Range * 5.0f, subBorders))
277  {
278  continue;
279  }
280  Repair(rayStartWorld - sub.SimPosition, rayEnd - sub.SimPosition, deltaTime, character, degreeOfSuccess, ignoredBodies);
281  }
282  Repair(rayStartWorld, rayEnd, deltaTime, character, degreeOfSuccess, ignoredBodies);
283  }
284  else
285  {
286  Repair(rayStartWorld - parentSub.SimPosition, rayEnd - parentSub.SimPosition, deltaTime, character, degreeOfSuccess, ignoredBodies);
287  }
288 
289  UseProjSpecific(deltaTime, rayStartWorld);
290 
291  return true;

299  private void Repair(Vector2 rayStart, Vector2 rayEnd, float deltaTime, Character user, float degreeOfSuccess, List<Body> ignoredBodies)
300  {
301  var collisionCategories = Physics.CollisionWall | Physics.CollisionCharacter | Physics.CollisionItem | Physics.CollisionLevel | Physics.CollisionRepair;
302 
303  //if the item can cut off limbs, activate nearby bodies to allow the raycast to hit them
304  if (statusEffectLists != null && statusEffectLists.ContainsKey(ActionType.OnUse))
305  {
306  if (statusEffectLists[ActionType.OnUse].Any(s => s.SeverLimbsProbability > 0.0f))
307  {
308  float rangeSqr = ConvertUnits.ToSimUnits(Range);
309  rangeSqr *= rangeSqr;
310  foreach (Character c in Character.CharacterList)
311  {
312  if (!c.Enabled || !c.AnimController.BodyInRest) { continue; }
313  //do a broad check first
314  if (Math.Abs(c.WorldPosition.X - item.WorldPosition.X) > 1000.0f) { continue; }
315  if (Math.Abs(c.WorldPosition.Y - item.WorldPosition.Y) > 1000.0f) { continue; }
316  foreach (Limb limb in c.AnimController.Limbs)
317  {
318  if (Vector2.DistanceSquared(limb.SimPosition, item.SimPosition) < rangeSqr && Vector2.Dot(rayEnd - rayStart, limb.SimPosition - rayStart) > 0)
319  {
320  c.AnimController.BodyInRest = false;
321  break;
322  }
323  }
324  }
325  }
326  }
327 
328  float lastPickedFraction = 0.0f;
329  if (RepairMultiple)
330  {
331  var bodies = Submarine.PickBodies(rayStart, rayEnd, ignoredBodies, collisionCategories,
332  ignoreSensors: false,
333  customPredicate: (Fixture f) =>
334  {
335  if (f.IsSensor)
336  {
337  if (RepairThroughHoles && f.Body?.UserData is Structure) { return false; }
338  if (f.Body?.UserData is PhysicsBody) { return false; }
339  }
340  if (f.Body?.UserData is Item it && it.GetComponent<Planter>() != null) { return false; }
341  if (f.Body?.UserData as string == "ruinroom") { return false; }
342  if (f.Body?.UserData is VineTile && !(FireDamage > 0)) { return false; }
343  return true;
344  },
345  allowInsideFixture: true);
346 
347  hitBodies.Clear();
348  hitBodies.AddRange(bodies.Distinct());
349 
350  lastPickedFraction = Submarine.LastPickedFraction;
351  Type lastHitType = null;
352  hitCharacters.Clear();
353  foreach (Body body in hitBodies)
354  {
355  Type bodyType = body.UserData?.GetType();
356  if (!RepairThroughWalls && bodyType != null && bodyType != lastHitType)
357  {
358  //stop the ray if it already hit a door/wall and is now about to hit some other type of entity
359  if (lastHitType == typeof(Item) || lastHitType == typeof(Structure)) { break; }
360  }
361 
362  Character hitCharacter = null;
363  if (body.UserData is Limb limb)
364  {
365  hitCharacter = limb.character;
366  }
367  else if (body.UserData is Character character)
368  {
369  hitCharacter = character;
370  }
371  //only do damage once to each character even if they ray hit multiple limbs
372  if (hitCharacter != null)
373  {
374  if (hitCharacters.Contains(hitCharacter)) { continue; }
375  hitCharacters.Add(hitCharacter);
376  }
377 
378  //if repairing through walls is not allowed and the next wall is more than 100 pixels away from the previous one, stop here
379  //(= repairing multiple overlapping walls is allowed as long as the edges of the walls are less than MaxOverlappingWallDist pixels apart)
380  float thisBodyFraction = Submarine.LastPickedBodyDist(body);
381  if (!RepairThroughWalls && lastHitType == typeof(Structure) && Range * (thisBodyFraction - lastPickedFraction) > MaxOverlappingWallDist)
382  {
383  break;
384  }
385  pickedPosition = rayStart + (rayEnd - rayStart) * thisBodyFraction;
386  if (FixBody(user, deltaTime, degreeOfSuccess, body))
387  {
388  lastPickedFraction = thisBodyFraction;
389  if (bodyType != null) { lastHitType = bodyType; }
390  }
391  }
392  }
393  else
394  {
395  var pickedBody = Submarine.PickBody(rayStart, rayEnd,
396  ignoredBodies, collisionCategories,
397  ignoreSensors: false,
398  customPredicate: (Fixture f) =>
399  {
400  if (f.IsSensor)
401  {
402  if (RepairThroughHoles && f.Body?.UserData is Structure) { return false; }
403  if (f.Body?.UserData is PhysicsBody) { return false; }
404  }
405  if (f.Body?.UserData as string == "ruinroom") { return false; }
406  if (f.Body?.UserData is VineTile && !(FireDamage > 0)) { return false; }
407 
408  if (f.Body?.UserData is Item targetItem)
409  {
410  if (!HitItems) { return false; }
411  if (HitBrokenDoors)
412  {
413  if (targetItem.GetComponent<Door>() == null && targetItem.Condition <= 0) { return false; }
414  }
415  else
416  {
417  if (targetItem.Condition <= 0) { return false; }
418  }
419  }
420  return f.Body?.UserData != null;
421  },
422  allowInsideFixture: true);
423  pickedPosition = Submarine.LastPickedPosition;
424  FixBody(user, deltaTime, degreeOfSuccess, pickedBody);
425  lastPickedFraction = Submarine.LastPickedFraction;
426  }
427 
428  if (ExtinguishAmount > 0.0f && item.CurrentHull != null)
429  {
430  fireSourcesInRange.Clear();
431  //step along the ray in 10% intervals, collecting all fire sources in the range
432  for (float x = 0.0f; x <= lastPickedFraction; x += 0.1f)
433  {
434  Vector2 displayPos = ConvertUnits.ToDisplayUnits(rayStart + (rayEnd - rayStart) * x);
435  if (item.CurrentHull.Submarine != null) { displayPos += item.CurrentHull.Submarine.Position; }
436 
437  Hull hull = Hull.FindHull(displayPos, item.CurrentHull);
438  if (hull == null) continue;
439  foreach (FireSource fs in hull.FireSources)
440  {
441  if (fs.IsInDamageRange(displayPos, 100.0f) && !fireSourcesInRange.Contains(fs))
442  {
443  fireSourcesInRange.Add(fs);
444  }
445  }
446  foreach (FireSource fs in hull.FakeFireSources)
447  {
448  if (fs.IsInDamageRange(displayPos, 100.0f) && !fireSourcesInRange.Contains(fs))
449  {
450  fireSourcesInRange.Add(fs);
451  }
452  }
453  }
454 
455  foreach (FireSource fs in fireSourcesInRange)
456  {
457  fs.Extinguish(deltaTime, ExtinguishAmount);
458 #if SERVER
459  if (!(fs is DummyFireSource))
460  {
461  GameMain.Server.KarmaManager.OnExtinguishingFire(user, deltaTime);
462  }
463 #endif
464  }
465  }
466 
467  if (WaterAmount > 0.0f && item.Submarine != null)
468  {
469  Vector2 pos = ConvertUnits.ToDisplayUnits(rayStart + item.Submarine.SimPosition);
470 
471  // Could probably be done much efficiently here
472  foreach (Item it in Item.ItemList)
473  {
474  if (it.Submarine == item.Submarine && it.GetComponent<Planter>() is { } planter)
475  {
476  if (it.GetComponent<Holdable>() is { } holdable && holdable.Attachable && !holdable.Attached) { continue; }
477 
478  Rectangle collisionRect = it.WorldRect;
479  collisionRect.Y -= collisionRect.Height;
480  if (collisionRect.Left < pos.X && collisionRect.Right > pos.X && collisionRect.Bottom < pos.Y)
481  {
482  Body collision = Submarine.PickBody(rayStart, it.SimPosition, ignoredBodies, collisionCategories);
483  if (collision == null)
484  {
485  for (var i = 0; i < planter.GrowableSeeds.Length; i++)
486  {
487  Growable seed = planter.GrowableSeeds[i];
488  if (seed == null || seed.Decayed) { continue; }
489 
490  seed.Health += WaterAmount * deltaTime;
491 
492 #if CLIENT
493  float barOffset = 10f * GUI.Scale;
494  Vector2 offset = planter.PlantSlots.ContainsKey(i) ? planter.PlantSlots[i].Offset : Vector2.Zero;
495  user?.UpdateHUDProgressBar(planter, planter.Item.DrawPosition + new Vector2(barOffset, 0) + offset, seed.Health / seed.MaxHealth, GUIStyle.Blue, GUIStyle.Blue, "progressbar.watering");
496 #endif
497  }
498  }
499  }
500  }
501  }
502  }
503 
504  if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
505  {
506  if (Rand.Range(0.0f, 1.0f) < FireProbability * deltaTime)
507  {
508  Vector2 displayPos = ConvertUnits.ToDisplayUnits(rayStart + (rayEnd - rayStart) * lastPickedFraction * 0.9f);
509  if (item.CurrentHull.Submarine != null) { displayPos += item.CurrentHull.Submarine.Position; }
510  new FireSource(displayPos);
511  }
512  }

515  private bool FixBody(Character user, float deltaTime, float degreeOfSuccess, Body targetBody)
516  {
517  if (targetBody?.UserData == null) { return false; }
518 
519  if (targetBody.UserData is Structure targetStructure)
520  {
521  if (targetStructure.IsPlatform) { return false; }
522  int sectionIndex = targetStructure.FindSectionIndex(ConvertUnits.ToDisplayUnits(pickedPosition));
523  if (sectionIndex < 0) { return false; }
524 
525  if (!fixableEntities.Contains("structure") && !fixableEntities.Contains(targetStructure.Prefab.Identifier)) { return true; }
526 
527  ApplyStatusEffectsOnTarget(user, deltaTime, ActionType.OnUse, structure: targetStructure);
528  FixStructureProjSpecific(user, deltaTime, targetStructure, sectionIndex);
529 
530  float structureFixAmount = StructureFixAmount;
531  if (structureFixAmount >= 0f)
532  {
533  structureFixAmount *= 1 + user.GetStatValue(StatTypes.RepairToolStructureRepairMultiplier);
534  structureFixAmount *= 1 + item.GetQualityModifier(Quality.StatType.RepairToolStructureRepairMultiplier);
535  }
536  else
537  {
538  structureFixAmount *= 1 + user.GetStatValue(StatTypes.RepairToolStructureDamageMultiplier);
539  structureFixAmount *= 1 + item.GetQualityModifier(Quality.StatType.RepairToolStructureDamageMultiplier);
540  }
541 
542  targetStructure.AddDamage(sectionIndex, -structureFixAmount * degreeOfSuccess, user);
543 
544  //if the next section is small enough, apply the effect to it as well
545  //(to make it easier to fix a small "left-over" section)
546  for (int i = -1; i < 2; i += 2)
547  {
548  int nextSectionLength = targetStructure.SectionLength(sectionIndex + i);
549  if ((sectionIndex == 1 && i == -1) ||
550  (sectionIndex == targetStructure.SectionCount - 2 && i == 1) ||
551  (nextSectionLength > 0 && nextSectionLength < Structure.WallSectionSize * 0.3f))
552  {
553  //targetStructure.HighLightSection(sectionIndex + i);
554  targetStructure.AddDamage(sectionIndex + i, -structureFixAmount * degreeOfSuccess);
555  }
556  }
557  return true;
558  }
559  else if (targetBody.UserData is Voronoi2.VoronoiCell cell && cell.IsDestructible)
560  {
561  if (Level.Loaded?.ExtraWalls.Find(w => w.Body == cell.Body) is DestructibleLevelWall levelWall)
562  {
563  levelWall.AddDamage(-LevelWallFixAmount * deltaTime, item.WorldPosition);
564  }
565  return true;
566  }
567  else if (targetBody.UserData is LevelObject levelObject && levelObject.Prefab.TakeLevelWallDamage)
568  {
569  levelObject.AddDamage(-LevelWallFixAmount, deltaTime, item);
570  return true;
571  }
572  else if (targetBody.UserData is Character targetCharacter)
573  {
574  if (targetCharacter.Removed) { return false; }
575  targetCharacter.LastDamageSource = item;
576  Limb closestLimb = null;
577  float closestDist = float.MaxValue;
578  foreach (Limb limb in targetCharacter.AnimController.Limbs)
579  {
580  float dist = Vector2.DistanceSquared(item.SimPosition, limb.SimPosition);
581  if (dist < closestDist)
582  {
583  closestLimb = limb;
584  closestDist = dist;
585  }
586  }
587 
588  if (closestLimb != null && !MathUtils.NearlyEqual(TargetForce, 0.0f))
589  {
590  Vector2 dir = closestLimb.WorldPosition - item.WorldPosition;
591  dir = dir.LengthSquared() < 0.0001f ? Vector2.UnitY : Vector2.Normalize(dir);
592  closestLimb.body.ApplyForce(dir * TargetForce, maxVelocity: 10.0f);
593  }
594 
595  ApplyStatusEffectsOnTarget(user, deltaTime, ActionType.OnUse, character: targetCharacter, limb: closestLimb);
596  FixCharacterProjSpecific(user, deltaTime, targetCharacter);
597  return true;
598  }
599  else if (targetBody.UserData is Limb targetLimb)
600  {
601  if (targetLimb.character == null || targetLimb.character.Removed) { return false; }
602 
603  if (!MathUtils.NearlyEqual(TargetForce, 0.0f))
604  {
605  Vector2 dir = targetLimb.WorldPosition - item.WorldPosition;
606  dir = dir.LengthSquared() < 0.0001f ? Vector2.UnitY : Vector2.Normalize(dir);

676  private float sinTime;
677  private float repairTimer;
678  private Gap previousGap;
679  private readonly float repairTimeOut = 5;
680  public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
681  {
682  if (!(objective.OperateTarget is Gap leak))
683  {
684  Reset();
685  return true;
686  }
687  if (leak.Submarine == null || leak.Submarine != character.Submarine)
688  {
689  Reset();
690  return true;
691  }
692  if (leak != previousGap)
693  {
694  Reset();
695  previousGap = leak;
696  }
697  Vector2 fromCharacterToLeak = leak.WorldPosition - character.AnimController.AimSourceWorldPos;
698  float dist = fromCharacterToLeak.Length();
699  float reach = AIObjectiveFixLeak.CalculateReach(this, character);
700 
701  if (dist > reach * 3)
702  {
703  // Too far away -> consider this done and hope the AI is smart enough to move closer
704  Reset();
705  return true;
706  }
707  character.AIController.SteeringManager.Reset();
708  if (!character.AnimController.InWater)
709  {
710  // TODO: use the collider size?
711  if (!character.AnimController.InWater && character.AnimController is HumanoidAnimController humanAnim &&
712  Math.Abs(fromCharacterToLeak.X) < 100.0f && fromCharacterToLeak.Y < 0.0f && fromCharacterToLeak.Y > -150.0f)
713  {
714  humanAnim.Crouching = true;
715  }
716  }
717  if (dist > reach * 0.8f || dist > reach * 0.5f && character.AnimController.Limbs.Any(l => l.InWater))
718  {
719  // Steer closer
720  if (character.AIController.SteeringManager is IndoorsSteeringManager indoorSteering)
721  {
722  // Swimming inside the sub
723  if (indoorSteering.CurrentPath != null && !indoorSteering.IsPathDirty && (indoorSteering.CurrentPath.Unreachable || indoorSteering.CurrentPath.Finished))
724  {
725  Vector2 dir = Vector2.Normalize(fromCharacterToLeak);
726  character.AIController.SteeringManager.SteeringManual(deltaTime, dir);
727  }
728  else
729  {
730  character.AIController.SteeringManager.SteeringSeek(character.GetRelativeSimPosition(leak));
731  }
732  }
733  else
734  {
735  // Swimming outside the sub
736  character.AIController.SteeringManager.SteeringSeek(character.GetRelativeSimPosition(leak));
737  }
738  }
739  else if (dist < reach * 0.25f)
740  {
741  // Too close -> steer away
742  character.AIController.SteeringManager.SteeringManual(deltaTime, Vector2.Normalize(character.SimPosition - leak.SimPosition));
743  }
744  if (dist <= reach)
745  {
746  // In range
747  character.CursorPosition = leak.WorldPosition;
748  if (character.Submarine != null)
749  {
750  character.CursorPosition -= character.Submarine.Position;
751  }
752  character.CursorPosition += VectorExtensions.Forward(Item.body.TransformedRotation + (float)Math.Sin(sinTime) / 2, dist / 2);
753  if (character.AnimController.InWater)
754  {
755  var torso = character.AnimController.GetLimb(LimbType.Torso);
756  // Turn facing the target when not moving (handled in the animcontroller if not moving)
757  Vector2 mousePos = ConvertUnits.ToSimUnits(character.CursorPosition);
758  Vector2 diff = (mousePos - torso.SimPosition) * character.AnimController.Dir;
759  float newRotation = MathUtils.VectorToAngle(diff);
760  character.AnimController.Collider.SmoothRotate(newRotation, 5.0f);
761 
762  if (VectorExtensions.Angle(VectorExtensions.Forward(torso.body.TransformedRotation), fromCharacterToLeak) < MathHelper.PiOver4)
763  {
764  // Swim past
765  Vector2 moveDir = leak.IsHorizontal ? Vector2.UnitY : Vector2.UnitX;
766  moveDir *= character.AnimController.Dir;
767  character.AIController.SteeringManager.SteeringManual(deltaTime, moveDir);
768  }
769  }
770  if (item.RequireAimToUse)
771  {
772  character.SetInput(InputType.Aim, false, true);
773  sinTime += deltaTime * 5;
774  }
775  // Press the trigger only when the tool is approximately facing the target.
776  Vector2 fromItemToLeak = leak.WorldPosition - item.WorldPosition;
777  var angle = VectorExtensions.Angle(VectorExtensions.Forward(item.body.TransformedRotation), fromItemToLeak);
778  if (angle < MathHelper.PiOver4)
779  {
780  if (Submarine.PickBody(item.SimPosition, leak.SimPosition, collisionCategory: Physics.CollisionWall, allowInsideFixture: true)?.UserData is Item i)
781  {
782  var door = i.GetComponent<Door>();
783  // Hit a door, abandon so that we don't weld it shut.
784  return door != null && !door.CanBeTraversed;
785  }
786  // Check that we don't hit any friendlies
787  if (Submarine.PickBodies(item.SimPosition, leak.SimPosition, collisionCategory: Physics.CollisionCharacter).None(hit =>
788  {
789  if (hit.UserData is Character c)
790  {
791  if (c == character) { return false; }
792  return HumanAIController.IsFriendly(character, c);
793  }
794  return false;
795  }))
796  {
797  character.SetInput(InputType.Shoot, false, true);
798  Use(deltaTime, character);
799  repairTimer += deltaTime;
800  if (repairTimer > repairTimeOut)
801  {
802 #if DEBUG
803  DebugConsole.NewMessage($"{character.Name}: timed out while welding a leak in {leak.FlowTargetHull.DisplayName}.", color: Color.Yellow);
804 #endif
805  Reset();
806  return true;
807  }
808  }
809  }
810  }
811  else
812  {
813  // Reset the timer so that we don't time out if the water forces push us away
814  repairTimer = 0;
815  }
816 
817  bool leakFixed = (leak.Open <= 0.0f || leak.Removed) &&
818  (leak.ConnectedWall == null || leak.ConnectedWall.Sections.Average(s => s.damage) < 1);
819 
820  if (leakFixed && leak.FlowTargetHull?.DisplayName != null && character.IsOnPlayerTeam)
821  {
822  if (!leak.FlowTargetHull.ConnectedGaps.Any(g => !g.IsRoomToRoom && g.Open > 0.0f))
823  {
824  character.Speak(TextManager.GetWithVariable("DialogLeaksFixed", "[roomname]", leak.FlowTargetHull.DisplayName, FormatCapitals.Yes).Value, null, 0.0f, "leaksfixed".ToIdentifier(), 10.0f);
825  }
826  else
827  {
828  character.Speak(TextManager.GetWithVariable("DialogLeakFixed", "[roomname]", leak.FlowTargetHull.DisplayName, FormatCapitals.Yes).Value, null, 0.0f, "leakfixed".ToIdentifier(), 10.0f);
829  }
830  }
831 
832  return leakFixed;
833 
834  void Reset()
835  {
836  sinTime = 0;
837  repairTimer = 0;
838  }
839  }

842  private void ApplyStatusEffectsOnTarget(Character user, float deltaTime, ActionType actionType, Item targetItem = null, Character character = null, Limb limb = null, Structure structure = null)
843  {
844  if (statusEffectLists == null) { return; }
845  if (!statusEffectLists.TryGetValue(actionType, out List<StatusEffect> statusEffects)) { return; }
846 
847  foreach (StatusEffect effect in statusEffects)
848  {
849  currentTargets.Clear();
850  effect.SetUser(user);
851  if (effect.HasTargetType(StatusEffect.TargetType.UseTarget))
852  {
853  if (targetItem != null)
854  {
855  currentTargets.AddRange(targetItem.AllPropertyObjects);
856  }
857  if (structure != null)
858  {
859  currentTargets.Add(structure);
860  }
861  if (character != null)
862  {
863  currentTargets.Add(character);
864  }
865  effect.Apply(actionType, deltaTime, item, currentTargets);
866  }
867  else if (effect.HasTargetType(StatusEffect.TargetType.Character))
868  {
869  currentTargets.Add(character);
870  effect.Apply(actionType, deltaTime, item, currentTargets);
871  }
872  else if (effect.HasTargetType(StatusEffect.TargetType.Limb))
873  {
874  currentTargets.Add(limb);
875  effect.Apply(actionType, deltaTime, item, currentTargets);
876  }
877 
878 #if CLIENT
879  if (user == null) { return; }
880  // Hard-coded progress bars for welding doors stuck.
881  // A general purpose system could be better, but it would most likely require changes in the way we define the status effects in xml.
882  foreach (ISerializableEntity target in currentTargets)
883  {
884  if (!(target is Door door)) { continue; }
885  if (!door.CanBeWelded || !door.Item.IsInteractable(user)) { continue; }

887  {
888  Identifier propertyName = effect.propertyNames[i];
889  if (propertyName != "stuck") { continue; }
890  if (door.SerializableProperties == null || !door.SerializableProperties.TryGetValue(propertyName, out SerializableProperty property)) { continue; }
891  object value = property.GetValue(target);
892  if (door.Stuck > 0)
893  {
894  bool isCutting = effect.propertyEffects[i].GetType() == typeof(float) && (float)effect.propertyEffects[i] < 0;
895  var progressBar = user.UpdateHUDProgressBar(door, door.Item.WorldPosition, door.Stuck / 100, Color.DarkGray * 0.5f, Color.White,
896  textTag: isCutting ? "progressbar.cutting" : "progressbar.welding");
897  if (progressBar != null) { progressBar.Size = new Vector2(60.0f, 20.0f); }
898  if (!isCutting) { HintManager.OnWeldingDoor(user, door); }
899  }
900  }
901  }
902 #endif
903  }
904  }
905  }
