365  private void CreateEvents(EventSet eventSet, Random rand)
366  {
367  if (level == null) { return; }
368  if (level.LevelData.HasHuntingGrounds && eventSet.DisableInHuntingGrounds) { return; }
369  DebugConsole.NewMessage($"Loading event set {eventSet.Identifier}", Color.LightBlue, debugOnly: true);
370 
371  int applyCount = 1;
372  List<Func<Level.InterestingPosition, bool>> spawnPosFilter = new List<Func<Level.InterestingPosition, bool>>();
373  if (eventSet.PerRuin)
374  {
375  applyCount = level.Ruins.Count();
376  foreach (var ruin in level.Ruins)
377  {
378  spawnPosFilter.Add(pos => pos.Ruin == ruin);
379  }
380  }
381  else if (eventSet.PerCave)
382  {
383  applyCount = level.Caves.Count();
384  foreach (var cave in level.Caves)
385  {
386  spawnPosFilter.Add(pos => pos.Cave == cave);
387  }
388  }
389  else if (eventSet.PerWreck)
390  {
391  var wrecks = Submarine.Loaded.Where(s => s.Info.IsWreck && (s.WreckAI == null || !s.WreckAI.IsAlive));
392  applyCount = wrecks.Count();
393  foreach (var wreck in wrecks)
394  {
395  spawnPosFilter.Add(pos => pos.Submarine == wreck);
396  }
397  }
398 
399  bool isPrefabSuitable(EventPrefab e)
400  => e.BiomeIdentifier.IsEmpty ||
401  e.BiomeIdentifier == level.LevelData?.Biome?.Identifier;
402 
403  var suitablePrefabSubsets = eventSet.EventPrefabs.Where(
404  e => e.EventPrefabs.Any(isPrefabSuitable)).ToArray();
405 
406  for (int i = 0; i < applyCount; i++)
407  {
408  if (eventSet.ChooseRandom)
409  {
410  if (suitablePrefabSubsets.Any())
411  {
412  var unusedEvents = suitablePrefabSubsets.ToList();
413  for (int j = 0; j < eventSet.EventCount; j++)
414  {
415  if (unusedEvents.All(e => e.EventPrefabs.All(p => CalculateCommonness(p, e.Commonness) <= 0.0f))) { break; }
416  EventSet.SubEventPrefab subEventPrefab = ToolBox.SelectWeightedRandom(unusedEvents, e => e.EventPrefabs.Max(p => CalculateCommonness(p, e.Commonness)), rand);
417  (IEnumerable<EventPrefab> eventPrefabs, float commonness, float probability) = subEventPrefab;
418  if (eventPrefabs != null && rand.NextDouble() <= probability)
419  {
420  var eventPrefab = ToolBox.SelectWeightedRandom(eventPrefabs.Where(isPrefabSuitable), e => e.Commonness, rand);
421 
422  var newEvent = eventPrefab.CreateInstance();
423  if (newEvent == null) { continue; }
424  newEvent.Init(true);
425  if (i < spawnPosFilter.Count) { newEvent.SpawnPosFilter = spawnPosFilter[i]; }
426  DebugConsole.NewMessage($"Initialized event {newEvent}", debugOnly: true);
427  if (!selectedEvents.ContainsKey(eventSet))
428  {
429  selectedEvents.Add(eventSet, new List<Event>());
430  }
431  selectedEvents[eventSet].Add(newEvent);
432  unusedEvents.Remove(subEventPrefab);
433  }
434  }
435  }
436  if (eventSet.ChildSets.Any())
437  {
438  var newEventSet = SelectRandomEvents(eventSet.ChildSets, random: rand);
439  if (newEventSet != null)
440  {
441  CreateEvents(newEventSet, rand);
442  }
443  }
444  }
445  else
446  {
447  foreach ((IEnumerable<EventPrefab> eventPrefabs, float commonness, float probability) in suitablePrefabSubsets)
448  {
449  if (rand.NextDouble() > probability) { continue; }
450 
451  var eventPrefab = ToolBox.SelectWeightedRandom(eventPrefabs.Where(isPrefabSuitable), e => e.Commonness, rand);
452  var newEvent = eventPrefab.CreateInstance();
453  if (newEvent == null) { continue; }
454  newEvent.Init(true);
455  DebugConsole.NewMessage($"Initialized event {newEvent}", debugOnly: true);
456  if (!selectedEvents.ContainsKey(eventSet))
457  {
458  selectedEvents.Add(eventSet, new List<Event>());
459  }
460  selectedEvents[eventSet].Add(newEvent);
461  }
462 
463  Location location = (GameMain.GameSession?.GameMode as CampaignMode)?.Map?.CurrentLocation ?? level?.StartLocation;
464  foreach (EventSet childEventSet in eventSet.ChildSets)
465  {
466  if (!IsValidForLevel(childEventSet, level)) { continue; }
467  if (location != null && !IsValidForLocation(childEventSet, location)) { continue; }
468  CreateEvents(childEventSet, rand);
469  }
470  }
471  }

474  private EventSet SelectRandomEvents(IReadOnlyList<EventSet> eventSets, bool? requireCampaignSet = null, Random random = null)
475  {
476  if (level == null) { return null; }
477  Random rand = random ?? new MTRandom(ToolBox.StringToInt(level.Seed));
478 
479  var allowedEventSets =
480  eventSets.Where(set => IsValidForLevel(set, level));
481 
482  if (requireCampaignSet.HasValue)
483  {
484  if (requireCampaignSet.Value)
485  {
486  if (allowedEventSets.Any(es => es.IsCampaignSet))
487  {
488  allowedEventSets =
489  allowedEventSets.Where(es => es.IsCampaignSet);
490  }
491  else
492  {
493  DebugConsole.AddWarning("No campaign event sets available. Using a non-campaign-specific set instead.");
494  }
495  }
496  else
497  {
498  allowedEventSets =
499  allowedEventSets.Where(es => !es.IsCampaignSet);
500  }
501  }
502 
503  Location location = (GameMain.GameSession?.GameMode as CampaignMode)?.Map?.CurrentLocation ?? level?.StartLocation;
504  if (location != null)
505  {
506  allowedEventSets = allowedEventSets.Where(set => IsValidForLocation(set, location));
507  }
508 
509  float totalCommonness = allowedEventSets.Sum(e => e.GetCommonness(level));
510  float randomNumber = (float)rand.NextDouble();
511  randomNumber *= totalCommonness;
512  foreach (EventSet eventSet in allowedEventSets)
513  {
514  float commonness = eventSet.GetCommonness(level);
515  if (randomNumber <= commonness)
516  {
517  return eventSet;
518  }
519  randomNumber -= commonness;
520  }
521 
522  return null;
523  }

539  private bool CanStartEventSet(EventSet eventSet)
540  {
541  ISpatialEntity refEntity = GetRefEntity();
542  float distFromStart = (float)Math.Sqrt(MathUtils.LineSegmentToPointDistanceSquared(level.StartExitPosition.ToPoint(), level.StartPosition.ToPoint(), refEntity.WorldPosition.ToPoint()));
543  float distFromEnd = (float)Math.Sqrt(MathUtils.LineSegmentToPointDistanceSquared(level.EndExitPosition.ToPoint(), level.EndPosition.ToPoint(), refEntity.WorldPosition.ToPoint()));
544 
545  //don't create new events if within 50 meters of the start/end of the level
546  if (!eventSet.AllowAtStart)
547  {
548  if (distanceTraveled <= 0.0f ||
549  distFromStart * Physics.DisplayToRealWorldRatio < 50.0f ||
550  distFromEnd * Physics.DisplayToRealWorldRatio < 50.0f)
551  {
552  return false;
553  }
554  }
555 
556  if (eventSet.DelayWhenCrewAway)
557  {
558  if ((isCrewAway && crewAwayDuration < settings.FreezeDurationWhenCrewAway) || crewAwayResetTimer > 0.0f)
559  {
560  return false;
561  }
562  }
563 
564  if ((Submarine.MainSub == null || distanceTraveled < eventSet.MinDistanceTraveled) &&
565  roundDuration < eventSet.MinMissionTime)
566  {
567  return false;
568  }
569 
570  if (CurrentIntensity < eventSet.MinIntensity || CurrentIntensity > eventSet.MaxIntensity)
571  {
572  return false;
573  }
574 
575  return true;
576  }

579  public void Update(float deltaTime)
580  {
581  if (!Enabled || level == null) { return; }
582  if (GameMain.GameSession.Campaign?.DisableEvents ?? false) { return; }
583 
584  //clients only calculate the intensity but don't create any events
585  //(the intensity is used for controlling the background music)
586  CalculateCurrentIntensity(deltaTime);
587 
588 #if DEBUG && SERVER
589  if (DateTime.Now > nextIntensityLogTime)
590  {
591  DebugConsole.NewMessage("EventManager intensity: " + (int)Math.Round(currentIntensity * 100) + " %");
592  nextIntensityLogTime = DateTime.Now + new TimeSpan(0, minutes: 1, seconds: 0);
593  }
594 #endif
595 
596  if (isClient) { return; }
597 
598  roundDuration += deltaTime;
599 
600  if (settings == null)
601  {
602  DebugConsole.ThrowError("Event settings not set before updating EventManager. Attempting to select...");
603  SelectSettings();
604  if (settings == null)
605  {
606  DebugConsole.ThrowError("Could not select EventManager settings. Disabling EventManager for the round...");
607 #if SERVER
608  GameMain.Server?.SendChatMessage("Could not select EventManager settings. Disabling EventManager for the round...", Networking.ChatMessageType.Error);
609 #endif
610  Enabled = false;
611  return;
612  }
613  }
614 
615  if (IsCrewAway())
616  {
617  isCrewAway = true;
618  crewAwayResetTimer = CrewAwayResetDelay;
619  crewAwayDuration += deltaTime;
620  }
621  else if (crewAwayResetTimer > 0.0f)
622  {
623  isCrewAway = false;
624  crewAwayResetTimer -= deltaTime;
625  }
626  else
627  {
628  isCrewAway = false;
629  crewAwayDuration = 0.0f;
630  eventThreshold += settings.EventThresholdIncrease * deltaTime;
631  eventThreshold = Math.Min(eventThreshold, 1.0f);
632  eventCoolDown -= deltaTime;
633  }
634 
635  calculateDistanceTraveledTimer -= deltaTime;
636  if (calculateDistanceTraveledTimer <= 0.0f)
637  {
638  distanceTraveled = CalculateDistanceTraveled();
639  calculateDistanceTraveledTimer = CalculateDistanceTraveledInterval;
640  }
641 
642  if (currentIntensity < eventThreshold)
643  {
644  bool recheck = false;
645  do
646  {
647  recheck = false;
648  //activate pending event sets that can be activated
649  for (int i = pendingEventSets.Count - 1; i >= 0; i--)
650  {
651  var eventSet = pendingEventSets[i];
652  if (eventCoolDown > 0.0f && !eventSet.IgnoreCoolDown) { continue; }
653 
654  if (!CanStartEventSet(eventSet)) { continue; }
655 
656  pendingEventSets.RemoveAt(i);
657 
658  if (selectedEvents.ContainsKey(eventSet))
659  {
660  //start events in this set
661  foreach (Event ev in selectedEvents[eventSet])
662  {
663  activeEvents.Add(ev);
664  eventThreshold = settings.DefaultEventThreshold;
665  if (eventSet.TriggerEventCooldown && selectedEvents[eventSet].Any(e => e.Prefab.TriggerEventCooldown))
666  {
667  eventCoolDown = settings.EventCooldown;
668  }
669  }
670  }
671 
672  //add child event sets to pending
673  foreach (EventSet childEventSet in eventSet.ChildSets)
674  {
675  pendingEventSets.Add(childEventSet);
676  recheck = true;
677  }
678  }
679  } while (recheck);
680  }
681 
682  foreach (Event ev in activeEvents)
683  {
684  if (!ev.IsFinished) { ev.Update(deltaTime); }
685  }
686 
687  if (QueuedEvents.Count > 0)
688  {
689  activeEvents.Add(QueuedEvents.Dequeue());
690  }
691  }

693  private void CalculateCurrentIntensity(float deltaTime)
694  {
695  intensityUpdateTimer -= deltaTime;
696  if (intensityUpdateTimer > 0.0f) { return; }
697  intensityUpdateTimer = IntensityUpdateInterval;
698 
699  // crew health --------------------------------------------------------
700 
701  avgCrewHealth = 0.0f;
702  int characterCount = 0;
703  foreach (Character character in Character.CharacterList)
704  {
705  if (character.IsDead || character.TeamID == CharacterTeamType.FriendlyNPC) { continue; }
706  if (character.AIController is HumanAIController || character.IsRemotePlayer)
707  {
708  avgCrewHealth += character.Vitality / character.MaxVitality * (character.IsUnconscious ? 0.5f : 1.0f);
709  characterCount++;
710  }
711  }
712  if (characterCount > 0)
713  {
714  avgCrewHealth /= characterCount;
715  }
716  else
717  {
718  avgCrewHealth = 0.5f;
719  }
720 
721  // enemy amount --------------------------------------------------------
722 
723  enemyDanger = 0.0f;
724  monsterStrength = 0;
725  foreach (Character character in Character.CharacterList)
726  {
727  if (character.IsIncapacitated || !character.Enabled || character.IsPet || character.Params.CompareGroup(CharacterPrefab.HumanSpeciesName)) { continue; }
728 
729  if (!(character.AIController is EnemyAIController enemyAI)) { continue; }
730 
731  if (!enemyAI.AIParams.StayInAbyss)
732  {
733  // Ignore abyss monsters because they can stay active for quite great distances. They'll be taken into account when they target the sub.
734  monsterStrength += enemyAI.CombatStrength;
735  }
736 
737  if (character.CurrentHull?.Submarine?.Info != null &&
738  (character.CurrentHull.Submarine == Submarine.MainSub || Submarine.MainSub.DockedTo.Contains(character.CurrentHull.Submarine)) &&
739  character.CurrentHull.Submarine.Info.Type == SubmarineType.Player)
740  {
741  // Enemy onboard -> Crawler inside the sub adds 0.2 to enemy danger, Mudraptor 0.42
742  enemyDanger += enemyAI.CombatStrength / 500.0f;
743  }
744  else if (enemyAI.SelectedAiTarget?.Entity?.Submarine != null)
745  {
746  // Enemy outside targeting the sub or something in it
747  // -> One Crawler adds 0.02, a Mudraptor 0.042, a Hammerhead 0.1, and a Moloch 0.25.
748  enemyDanger += enemyAI.CombatStrength / 5000.0f;
749  }
750  }
751  // Add a portion of the total strength of active monsters to the enemy danger so that we don't spawn too many monsters around the sub.
752  // On top of the existing value, so if 10 crawlers are targeting the sub simultaneously from outside, the final value would be: 0.02 x 10 + 0.2 = 0.4.
753  // And if they get inside, we add 0.1 per crawler on that.
754  // So, in practice the danger per enemy that is attacking the sub is half of what it would be when the enemy is not targeting the sub.
755  // 10 Crawlers -> +0.2 (0.4 in total if all target the sub from outside).
756  // 5 Mudraptors -> +0.21 (0.42 in total, before they get inside).
757  // 3 Hammerheads -> +0.3 (0.6 in total, if they all target the sub).
758  // 2 Molochs -> +0.5 (1.0 in total, if both target the sub).
759  enemyDanger += monsterStrength / 5000f;
760  enemyDanger = MathHelper.Clamp(enemyDanger, 0.0f, 1.0f);
761 
762  // The definitions above aim for that we never spawn more monsters that the player (and the performance) can handle.
763  // Some examples that result in the max intensity even when the creatures would just idle around.
764  // The values are theoretical, because in practice many of the monsters are targeting the sub, which will double the danger of those monster and effectively halve the max monster count.
765  // In practice we don't use the max intensity. For example on level 50 we use max intensity 50, which would mean that we'd halve the numbers below.
766  // There's no hard cap for the monster count, but if the amount of monsters is higher than this, we don't spawn more monsters from the events:
767  // 50 Crawlers (We shouldn't actually ever spawn that many. 12 is the max per event, but theoretically 25 crawlers would result in max intensity).
768  // 25 Tigerthreshers (Max 9 per event. 12 targeting the sub at the same time results in max intensity).
769  // 10 Hammerheads (Max 3 per event. 5 targeting the sub at the same time results in max intensity).
770  // 4 Molochs (Max 2 per event and 2 targeting the sub at the same time results in max intensity).
771 
772  // hull status (gaps, flooding, fire) --------------------------------------------------------
773 
774  float holeCount = 0.0f;
775  float waterAmount = 0.0f;
776  float dryHullVolume = 0.0f;
777  foreach (Hull hull in Hull.HullList)
778  {
779  if (hull.Submarine == null || hull.Submarine.Info.Type != SubmarineType.Player) { continue; }
780  if (GameMain.GameSession?.GameMode is PvPMode)
781  {
782  if (hull.Submarine.TeamID != CharacterTeamType.Team1 && hull.Submarine.TeamID != CharacterTeamType.Team2) { continue; }
783  }
784  else
785  {
786  if (hull.Submarine.TeamID != CharacterTeamType.Team1) { continue; }
787  }
788  fireAmount += hull.FireSources.Sum(fs => fs.Size.X);
789  if (hull.IsWetRoom) { continue; }
790  foreach (Gap gap in hull.ConnectedGaps)
791  {
792  if (!gap.IsRoomToRoom)
793  {
794  holeCount += gap.Open;
795  }
796  }
797  waterAmount += hull.WaterVolume;
798  dryHullVolume += hull.Volume;
799  }
800  if (dryHullVolume > 0)
801  {
802  floodingAmount = waterAmount / dryHullVolume;
803  }
804 
805  //hull integrity at 0.0 if there are 10 or more wide-open holes
806  avgHullIntegrity = MathHelper.Clamp(1.0f - holeCount / 10.0f, 0.0f, 1.0f);
807 
808  //a fire of any size bumps up the fire amount to 20%
809  //if the total width of the fires is 1000 or more, the fire amount is considered to be at 100%
810  fireAmount = MathHelper.Clamp(fireAmount / 1000.0f, fireAmount > 0.0f ? 0.2f : 0.0f, 1.0f);
811 
812  //flooding less than 10% of the sub is ignored
813  //to prevent ballast tanks from affecting the intensity
814  if (floodingAmount < 0.1f)
815  {
816  floodingAmount = 0.0f;
817  }
818  else
819  {
820  floodingAmount *= 1.5f;
821  }
822 
823  // calculate final intensity --------------------------------------------------------
824 
825  targetIntensity =
826  ((1.0f - avgCrewHealth) + (1.0f - avgHullIntegrity) + floodingAmount) / 3.0f;
827  targetIntensity += fireAmount * 0.5f;
828  targetIntensity += enemyDanger;
829  targetIntensity = MathHelper.Clamp(targetIntensity, 0.0f, 1.0f);
830 
831  if (targetIntensity > currentIntensity)
832  {
833  //25 seconds for intensity to go from 0.0 to 1.0
834  currentIntensity = Math.Min(currentIntensity + 0.04f * IntensityUpdateInterval, targetIntensity);
835  //20 seconds for intensity to go from 0.0 to 1.0
836  musicIntensity = Math.Min(musicIntensity + 0.05f * IntensityUpdateInterval, targetIntensity);
837  }
838  else
839  {
840  //400 seconds for intensity to go from 1.0 to 0.0
841  currentIntensity = Math.Max(currentIntensity - 0.0025f * IntensityUpdateInterval, targetIntensity);
842  //20 seconds for intensity to go from 1.0 to 0.0
843  musicIntensity = Math.Max(musicIntensity - 0.05f * IntensityUpdateInterval, targetIntensity);
844  }
845  }
