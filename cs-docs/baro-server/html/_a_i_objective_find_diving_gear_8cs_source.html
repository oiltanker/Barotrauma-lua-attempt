39 protected override void Act(float deltaTime)
40 {
41 if (character.LockHands)
42 {
43 Abandon = true;
44 return;
45 }
46 targetItem = character.Inventory.FindItemByTag(gearTag, true);
47 if (targetItem == null && gearTag == LIGHT_DIVING_GEAR)
48 {
49 targetItem = character.Inventory.FindItemByTag(HEAVY_DIVING_GEAR, true);
50 }
51 if (targetItem == null || !character.HasEquippedItem(targetItem, slotType: InvSlotType.OuterClothes | InvSlotType.Head | InvSlotType.InnerClothes) && targetItem.ContainedItems.Any(i => i.HasTag(OXYGEN_SOURCE) && i.Condition > 0))
52 {
53 TryAddSubObjective(ref getDivingGear, () =>
54 {
55 if (targetItem == null && character.IsOnPlayerTeam)
56 {
57 character.Speak(TextManager.Get("DialogGetDivingGear").Value, null, 0.0f, "getdivinggear".ToIdentifier(), 30.0f);
58 }
59 return new AIObjectiveGetItem(character, gearTag, objectiveManager, equip: true)
60 {
61 AllowStealing = HumanAIController.NeedsDivingGear(character.CurrentHull, out _),
62 AllowToFindDivingGear = false,
63 AllowDangerousPressure = true,
64 EquipSlotType = InvSlotType.OuterClothes | InvSlotType.Head | InvSlotType.InnerClothes,
65 Wear = true
66 };
67 },
68 onAbandon: () => Abandon = true,
69 onCompleted: () =>
70 {
71 RemoveSubObjective(ref getDivingGear);
72 if (gearTag == HEAVY_DIVING_GEAR && HumanAIController.HasItem(character, LIGHT_DIVING_GEAR, out IEnumerable<Item> masks, requireEquipped: true))
73 {
74 foreach (Item mask in masks)
75 {
76 if (mask != targetItem)
77 {
78 character.Inventory.TryPutItem(mask, character, CharacterInventory.anySlot);
79 }
80 }
81 }
82 });
83 }
84 else
85 {
86 float min = GetMinOxygen(character);
87 if (targetItem.OwnInventory != null && targetItem.OwnInventory.AllItems.None(it => it != null && it.HasTag(OXYGEN_SOURCE) && it.Condition > min))
88 {
89 TryAddSubObjective(ref getOxygen, () =>
90 {
91 if (character.IsOnPlayerTeam)
92 {
93 if (HumanAIController.HasItem(character, OXYGEN_SOURCE, out _, conditionPercentage: min))
94 {
95 character.Speak(TextManager.Get("dialogswappingoxygentank").Value, null, 0, "swappingoxygentank".ToIdentifier(), 30.0f);
96 if (character.Inventory.FindAllItems(i => i.HasTag(OXYGEN_SOURCE) && i.Condition > min).Count == 1)
97 {
98 character.Speak(TextManager.Get("dialoglastoxygentank").Value, null, 0.0f, "dialoglastoxygentank".ToIdentifier(), 30.0f);
99 }
100 }
101 else
102 {
103 character.Speak(TextManager.Get("DialogGetOxygenTank").Value, null, 0, "getoxygentank".ToIdentifier(), 30.0f);
104 }
105 }
106 return new AIObjectiveContainItem(character, OXYGEN_SOURCE, targetItem.GetComponent<ItemContainer>(), objectiveManager, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC)
107 {
108 AllowToFindDivingGear = false,
109 AllowDangerousPressure = true,
110 ConditionLevel = MIN_OXYGEN,
111 RemoveExistingWhenNecessary = true
112 };
113 },
114 onAbandon: () =>
115 {
116 getOxygen = null;
117 int remainingTanks = ReportOxygenTankCount();
118 // Try to seek any oxygen sources, even if they have minimal amount of oxygen.
119 TryAddSubObjective(ref getOxygen, () =>
120 {
121 return new AIObjectiveContainItem(character, OXYGEN_SOURCE, targetItem.GetComponent<ItemContainer>(), objectiveManager, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC)
122 {
123 AllowToFindDivingGear = false,
124 AllowDangerousPressure = true,
125 RemoveExisting = true
126 };
127 },
128 onAbandon: () =>
129 {
130 Abandon = true;
131 if (remainingTanks > 0 && !HumanAIController.HasItem(character, OXYGEN_SOURCE, out _, conditionPercentage: 0.01f))
132 {
133 character.Speak(TextManager.Get("dialogcantfindtoxygen").Value, null, 0, "cantfindoxygen".ToIdentifier(), 30.0f);
134 }
135 },
136 onCompleted: () => RemoveSubObjective(ref getOxygen));
137 },
138 onCompleted: () =>
139 {
140 RemoveSubObjective(ref getOxygen);
141 ReportOxygenTankCount();
142 });
143
144 int ReportOxygenTankCount()
145 {
146 if (character.Submarine != Submarine.MainSub) { return 1; }
147 int remainingOxygenTanks = Submarine.MainSub.GetItems(false).Count(i => i.HasTag(OXYGEN_SOURCE) && i.Condition > 1);
148 if (remainingOxygenTanks == 0)
149 {
150 character.Speak(TextManager.Get("DialogOutOfOxygenTanks").Value, null, 0.0f, "outofoxygentanks".ToIdentifier(), 30.0f);
151 }
152 else if (remainingOxygenTanks < 10)
153 {
154 character.Speak(TextManager.Get("DialogLowOnOxygenTanks").Value, null, 0.0f, "lowonoxygentanks".ToIdentifier(), 30.0f);
155 }
156 return remainingOxygenTanks;
157 }

170 public static float GetMinOxygen(Character character)
171 {
172 // Seek oxygen that has at least 10% condition left, if we are inside a friendly sub.
173 // The margin helps us to survive, because we might need some oxygen before we can find more oxygen.
174 // When we are venturing outside of our sub, let's just suppose that we have enough oxygen with us and optimize it so that we don't keep switching off half used tanks.
175 float min = 0.01f;
176 float minOxygen = character.IsInFriendlySub ? MIN_OXYGEN : min;
177 if (minOxygen > min && character.Inventory.AllItems.Any(i => i.HasTag("oxygensource") && i.ConditionPercentage >= minOxygen))
178 {
179 // There's a valid oxygen tank in the inventory -> no need to swap the tank too early.
180 minOxygen = min;
181 }
182 return minOxygen;
183 }
