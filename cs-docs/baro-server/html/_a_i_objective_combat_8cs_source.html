154 protected override float GetPriority()
155 {
156 if (Enemy == null)
157 {
158 Priority = 0;
159 Abandon = true;
160 return Priority;
161 }
162 if (character.TeamID == CharacterTeamType.FriendlyNPC)
163 {
164 if (Enemy.Submarine == null || (Enemy.Submarine.TeamID != character.TeamID && Enemy.Submarine != character.Submarine))
165 {
166 Priority = 0;
167 Abandon = true;
168 return Priority;
169 }
170 }
171 float damageFactor = MathUtils.InverseLerp(0.0f, 5.0f, character.GetDamageDoneByAttacker(Enemy) / 100.0f);
172 Priority = TargetEliminated ? 0 : Math.Min((95 + damageFactor) * PriorityModifier, 100);
173 if (Priority > 0)
174 {
175 if (EnemyAIController.IsLatchedToSomeoneElse(Enemy, character))
176 {
177 Priority = 0;
178 }
179 }
180 return Priority;

227 protected override void Act(float deltaTime)
228 {
229 if (IsEnemyDisabled)
230 {
231 IsCompleted = true;
232 return;
233 }
234 if (AllowCoolDown)
235 {
236 coolDownTimer -= deltaTime;
237 }
238 if (seekAmmunitionObjective == null && seekWeaponObjective == null)
239 {
240 if (Mode != CombatMode.Retreat && TryArm())
241 {
242 OperateWeapon(deltaTime);
243 }
244 if (HoldPosition)
245 {
246 SteeringManager.Reset();
247 }
248 else if (seekAmmunitionObjective == null && seekWeaponObjective == null)
249 {
250 Move(deltaTime);
251 }
252 switch (Mode)
253 {
254 case CombatMode.Offensive:
255 if (TargetEliminated && objectiveManager.IsCurrentOrder<AIObjectiveFightIntruders>())
256 {
257 character.Speak(TextManager.Get("DialogTargetDown").Value, null, 3.0f, "targetdown".ToIdentifier(), 30.0f);
258 }
259 break;
260 case CombatMode.Arrest:
261 if (HumanAIController.HasItem(Enemy, "handlocker".ToIdentifier(), out _, requireEquipped: true))
262 {
263 IsCompleted = true;
264 }
265 else if (Enemy.IsKnockedDown &&
266 !objectiveManager.IsCurrentObjective<AIObjectiveFightIntruders>() &&
267 !HumanAIController.HasItem(character, "handlocker".ToIdentifier(), out _, requireEquipped: false))
268 {
269 IsCompleted = true;
270 }
271 break;
272 }
273 }

276 private void Move(float deltaTime)
277 {
278 switch (Mode)
279 {
280 case CombatMode.Offensive:
281 case CombatMode.Arrest:
282 Engage(deltaTime);
283 break;
284 case CombatMode.Defensive:
285 if (character.IsOnPlayerTeam && !Enemy.IsPlayer && objectiveManager.IsCurrentOrder<AIObjectiveGoTo>())
286 {
287 if ((character.CurrentHull == null || character.CurrentHull == Enemy.CurrentHull) && sqrDistance < 200 * 200)
288 {
289 Engage(deltaTime);
290 }
291 else
292 {
293 // Keep following the goto target
294 var gotoObjective = objectiveManager.GetOrder<AIObjectiveGoTo>();
295 if (gotoObjective != null)
296 {
297 gotoObjective.ForceAct(deltaTime);
298 if (!character.AnimController.InWater)
299 {
300 HumanAIController.FaceTarget(Enemy);
301 ForceWalk = true;
302 HumanAIController.AutoFaceMovement = false;
303 }
304 }
305 else
306 {
307 SteeringManager.Reset();
308 }
309 }
310 }
311 else
312 {
313 AskHelp();
314 Retreat(deltaTime);
315 }
316 break;
317 case CombatMode.Retreat:
318 AskHelp();
319 Retreat(deltaTime);
320 break;
321 default:
322 throw new NotImplementedException();
323 }

326 private bool TryArm()
327 {
328 if (character.LockHands || Enemy == null)
329 {
330 Weapon = null;
331 return false;
332 }
333 if (checkWeaponsTimer < 0)
334 {
335 checkWeaponsTimer = checkWeaponsInterval;
336 // First go through all weapons and try to reload without seeking ammunition
337 var allWeapons = FindWeaponsFromInventory();
338 while (allWeapons.Any())
339 {
340 Weapon = GetWeapon(allWeapons, out _weaponComponent);
341 if (Weapon == null)
342 {
343 // No weapons
344 break;
345 }
346 if (!character.Inventory.Contains(Weapon) || WeaponComponent == null)
347 {
348 // Not in the inventory anymore or cannot find the weapon component
349 allWeapons.Remove(WeaponComponent);
350 Weapon = null;
351 continue;
352 }
353 if (WeaponComponent.IsLoaded(character))
354 {
355 // All good, the weapon is loaded
356 break;
357 }
358 if (Reload(seekAmmo: false))
359 {
360 // All good, we can use the weapon.
361 break;
362 }
363 else
364 {
365 // No ammo.
366 allWeapons.Remove(WeaponComponent);
367 Weapon = null;
368 }
369 }
370 if (Weapon == null)
371 {
372 // No weapon found with the conditions above. Try again, now let's try to seek ammunition too
373 Weapon = FindWeapon(out _weaponComponent);
374 if (Weapon != null)
375 {
376 if (!CheckWeapon(seekAmmo: true))
377 {
378 if (seekAmmunitionObjective != null)
379 {
380 // No loaded weapon, but we are trying to seek ammunition.
381 return false;
382 }
383 else
384 {
385 Weapon = null;
386 }
387 }
388 }
389 }
390 bool isAllowedToSeekWeapons = character.CurrentHull != null && !IsEnemyCloserThan(300) && character.IsOnPlayerTeam && IsOffensiveOrArrest;
391 if (!isAllowedToSeekWeapons)
392 {
393 if (WeaponComponent == null)
394 {
395 SpeakNoWeapons();
396 Mode = CombatMode.Retreat;
397 }
398 }
399 else if (seekAmmunitionObjective == null && (WeaponComponent == null || WeaponComponent.CombatPriority < goodWeaponPriority))
400 {
401 // Poor weapon equipped -> try to find better.
402 RemoveSubObjective(ref seekAmmunitionObjective);
403 RemoveSubObjective(ref retreatObjective);
404 RemoveSubObjective(ref followTargetObjective);
405 TryAddSubObjective(ref seekWeaponObjective,
406 constructor: () => new AIObjectiveGetItem(character, "weapon".ToIdentifier(), objectiveManager, equip: true, checkInventory: false)
407 {
408 AllowStealing = HumanAIController.IsMentallyUnstable,
409 EvaluateCombatPriority = false, // Use a custom formula instead
410 GetItemPriority = i =>
411 {
412 if (Weapon != null && (i == Weapon || i.Prefab.Identifier == Weapon.Prefab.Identifier)) { return 0; }
413 if (i.IsOwnedBy(character)) { return 0; }
414 var mw = i.GetComponent<MeleeWeapon>();
415 var rw = i.GetComponent<RangedWeapon>();
416 float priority = 0;
417 if (mw != null)
418 {
419 priority = mw.CombatPriority / 100;
420 }
421 else if (rw != null)
422 {
423 priority = rw.CombatPriority / 100;
424 }
425 if (i.HasTag("stunner"))
426 {
427 if (Mode == CombatMode.Arrest)
428 {
429 priority *= 2;
430 }
431 else
432 {
433 priority /= 2;
434 }
435 }
436 return priority;
437 }
438 },
439 onCompleted: () => RemoveSubObjective(ref seekWeaponObjective),
440 onAbandon: () =>
441 {
442 RemoveSubObjective(ref seekWeaponObjective);
443 if (Weapon == null)
444 {
445 SpeakNoWeapons();
446 Mode = CombatMode.Retreat;
447 }
448 else
449 {
450 Mode = CombatMode.Defensive;
451 }
452 });
453 }
454 }
455 else
456 {
457 if (!CheckWeapon(seekAmmo: false))
458 {
459 Weapon = null;
460 }
461 }
462 return Weapon != null;
463
464 bool CheckWeapon(bool seekAmmo)
465 {
466 if (!character.Inventory.Contains(Weapon) || WeaponComponent == null)
467 {
468 // Not in the inventory anymore or cannot find the weapon component
469 return false;
470 }
471 if (!WeaponComponent.IsLoaded(character))
472 {
473 // Try reloading (and seek ammo)
474 if (!Reload(seekAmmo))
475 {
476 return false;
477 }
478 }
479 return true;
480 };

504 private Item GetWeapon(IEnumerable<ItemComponent> weaponList, out ItemComponent weaponComponent)
505 {
506 weaponComponent = null;
507 float bestPriority = 0;
508 float lethalDmg = -1;
509 bool isAllowedToSeekWeapons = !IsEnemyCloserThan(300);
510 bool prioritizeMelee = IsEnemyCloserThan(50) || EnemyAIController.IsLatchedTo(Enemy, character);
511 foreach (var weapon in weaponList)
512 {
513 float priority = weapon.CombatPriority;
514 if (prioritizeMelee)
515 {
516 if (weapon is MeleeWeapon)
517 {
518 priority *= 5;
519 }
520 else
521 {
522 priority /= 2;
523 }
524 }
525 if (!weapon.IsLoaded(character))
526 {
527 if (weapon is RangedWeapon && !isAllowedToSeekWeapons)
528 {
529 // Close to the enemy. Ignore weapons that don't have any ammunition (-> Don't seek ammo).
530 continue;
531 }
532 else
533 {
534 // Halve the priority for weapons that don't have proper ammunition loaded.
535 priority /= 2;
536 }
537 }
538 if (Enemy.IsKnockedDown)
539 {
540 // Enemy is stunned, reduce the priority of stunner weapons.
541 Attack attack = GetAttackDefinition(weapon);
542 if (attack != null)
543 {
544 lethalDmg = attack.GetTotalDamage();
545 float max = lethalDmg + 1;
546 if (weapon.Item.HasTag("stunner"))
547 {
548 priority = max;
549 }
550 else
551 {
552 float stunDmg = ApproximateStunDamage(weapon, attack);
553 float diff = stunDmg - lethalDmg;
554 priority = Math.Clamp(priority - Math.Max(diff * 2, 0), min: 1, max);
555 }
556 }
557 }
558 else if (Mode == CombatMode.Arrest)
559 {
560 // Enemy is not stunned, increase the priority of stunner weapons and decrease the priority of lethal weapons.
561 if (weapon.Item.HasTag("stunner"))
562 {
563 priority *= 2;
564 }
565 else
566 {
567 Attack attack = GetAttackDefinition(weapon);
568 if (attack != null)
569 {
570 lethalDmg = attack.GetTotalDamage();
571 float stunDmg = ApproximateStunDamage(weapon, attack);
572 float diff = stunDmg - lethalDmg;
573 if (diff < 0)
574 {
575 priority /= 2;
576 }
577 }
578 }
579 }
580 else if (weapon is MeleeWeapon && weapon.Item.HasTag("stunner") && !CanMeleeStunnerStun(weapon))
581 {
582 Attack attack = GetAttackDefinition(weapon);
583 priority = attack?.GetTotalDamage() ?? priority / 2;
584 }
585 if (priority > bestPriority)
586 {
587 weaponComponent = weapon;
588 bestPriority = priority;
589 }
590 }
591 if (weaponComponent == null) { return null; }
592 if (bestPriority < 1) { return null; }
593 if (Mode == CombatMode.Arrest)
594 {
595 if (weaponComponent.Item.HasTag("stunner"))
596 {
597 isLethalWeapon = false;
598 }
599 else
600 {
601 if (lethalDmg < 0)
602 {
603 lethalDmg = GetLethalDamage(weaponComponent);
604 }
605 isLethalWeapon = lethalDmg > 1;
606 }
607 if (allowHoldFire && !hasAimed && holdFireTimer <= 0)
608 {
609 holdFireTimer = arrestHoldFireTime * Rand.Range(0.75f, 1.25f);
610 }
611 }
612 return weaponComponent.Item;

614 float ApproximateStunDamage(ItemComponent weapon, Attack attack)
615 {
616 // Try to reduce the priority using the actual damage values and status effects.
617 // This is an approximation, because we can't check the status effect conditions here.
618 // The result might be incorrect if there is a high stun effect that's only applied in certain conditions.
619 var statusEffects = attack.StatusEffects.Where(se => !se.HasConditions && se.type == ActionType.OnUse && se.HasRequiredItems(character));
620 if (weapon.statusEffectLists != null && weapon.statusEffectLists.TryGetValue(ActionType.OnUse, out List<StatusEffect> hitEffects))
621 {
622 statusEffects = statusEffects.Concat(hitEffects);
623 }
624 float afflictionsStun = attack.Afflictions.Keys.Sum(a => a.Identifier == "stun" ? a.Strength : 0);
625 float effectsStun = statusEffects.None() ? 0 : statusEffects.Max(se =>
626 {
627 float stunAmount = 0;
628 var stunAffliction = se.Afflictions.Find(a => a.Identifier == "stun");
629 if (stunAffliction != null)
630 {
631 stunAmount = stunAffliction.Strength;
632 }
633 return stunAmount;
634 });
635 return attack.Stun + afflictionsStun + effectsStun;
636 }

638 bool CanMeleeStunnerStun(ItemComponent weapon)
639 {
640 // If there's an item container that takes a battery,
641 // assume that it's required for the stun effect
642 // as we can't check the status effect conditions here.
643 var mobileBatteryTag = "mobilebattery".ToIdentifier();
644 var containers = weapon.Item.Components.Where(ic =>
645 ic is ItemContainer container &&
646 container.ContainableItemIdentifiers.Contains(mobileBatteryTag));
647 // If there's no such container, assume that the melee weapon can stun without a battery.
648 return containers.None() || containers.Any(container =>
649 (container as ItemContainer)?.Inventory.AllItems.Any(i => i != null && i.HasTag(mobileBatteryTag) && i.Condition > 0.0f) ?? false);
650 }

679 {

723 private bool Equip()
724 {
725 if (character.LockHands) { return false; }
726 if (!WeaponComponent.HasRequiredContainedItems(character, addMessage: false))
727 {
728 return false;
729 }
730 if (!character.HasEquippedItem(Weapon, predicate: IsHandSlotType))
731 {
732 Weapon.TryInteract(character, forceSelectKey: true);
733 var slots = Weapon.AllowedSlots.Where(s => IsHandSlotType(s));
734 if (character.Inventory.TryPutItem(Weapon, character, slots))
735 {
736 SetAimTimer(Rand.Range(0.2f, 0.4f) / AimSpeed);
737 }
738 else
739 {
740 SpeakNoWeapons();
741 Weapon = null;
742 Mode = CombatMode.Retreat;
743 return false;
744 }
745 }
746 return true;

754 private void Retreat(float deltaTime)
755 {
756 RemoveFollowTarget();
757 RemoveSubObjective(ref seekAmmunitionObjective);
758 if (retreatObjective != null && retreatObjective.Target != retreatTarget)
759 {
760 RemoveSubObjective(ref retreatObjective);
761 }
762 if (character.Submarine == null && sqrDistance < MathUtils.Pow2(maxDistance))
763 {
764 // Swim away
765 SteeringManager.Reset();
766 SteeringManager.SteeringManual(deltaTime, Vector2.Normalize(character.WorldPosition - Enemy.WorldPosition));
767 SteeringManager.SteeringAvoid(deltaTime, 5, weight: 2);
768 return;
769 }
770 if (retreatTarget == null || (retreatObjective != null && !retreatObjective.CanBeCompleted))
771 {
772 if (findHullTimer > 0)
773 {
774 findHullTimer -= deltaTime;
775 }
776 else
777 {
778 retreatTarget = findSafety.FindBestHull(HumanAIController.VisibleHulls, allowChangingTheSubmarine: character.TeamID != CharacterTeamType.FriendlyNPC);
779 findHullTimer = findHullInterval * Rand.Range(0.9f, 1.1f);
780 }
781 }
782 if (retreatTarget != null && character.CurrentHull != retreatTarget)
783 {
784 TryAddSubObjective(ref retreatObjective, () => new AIObjectiveGoTo(retreatTarget, character, objectiveManager)
785 {
786 UsePathingOutside = false
787 },
788 onAbandon: () =>
789 {
790 if (Enemy != null && HumanAIController.VisibleHulls.Contains(Enemy.CurrentHull))
791 {
792 // If in the same room with an enemy -> don't try to escape because we'd want to fight it
793 SteeringManager.Reset();
794 RemoveSubObjective(ref retreatObjective);
795 }
796 else
797 {
798 // else abandon and fall back to find safety mode
799 Abandon = true;
800 }
801 },
802 onCompleted: () => RemoveSubObjective(ref retreatObjective));
803 }

806 private void Engage(float deltaTime)
807 {
808 if (WeaponComponent == null)
809 {
810 RemoveFollowTarget();
811 SteeringManager.Reset();
812 return;
813 }
814 if (character.LockHands || Enemy == null)
815 {
816 Mode = CombatMode.Retreat;
817 SteeringManager.Reset();
818 return;
819 }
820 retreatTarget = null;

1019 }

1021 }
1022 if (WeaponComponent.HasRequiredContainedItems(character, addMessage: false))

1025 }

1033 private void Attack(float deltaTime)
1034 {
1035 character.CursorPosition = Enemy.WorldPosition;
1036 if (AimAccuracy < 1)
1037 {
1038 spreadTimer += deltaTime * Rand.Range(0.01f, 1f);
1039 float shake = Rand.Range(0.95f, 1.05f);
1040 float offsetAmount = (1 - AimAccuracy) * Rand.Range(300f, 500f);
1041 float distanceFactor = MathUtils.InverseLerp(0, 1000 * 1000, sqrDistance);
1042 float offset = (float)Math.Sin(spreadTimer * shake) * offsetAmount * distanceFactor;
1043 character.CursorPosition += new Vector2(0, offset);
1044 }
1045 if (character.Submarine != null)
1046 {
1047 character.CursorPosition -= character.Submarine.Position;
1048 }
1049 visibilityCheckTimer -= deltaTime;
1050 if (visibilityCheckTimer <= 0.0f)
1051 {
1052 canSeeTarget = character.CanSeeTarget(Enemy);
1053 visibilityCheckTimer = visibilityCheckInterval;
1054 }
1055 if (!canSeeTarget)
1056 {
1057 SetAimTimer(Rand.Range(0.2f, 0.4f) / AimSpeed);
1058 return;
1059 }
1060 if (Weapon.RequireAimToUse)
1061 {
1062 character.SetInput(InputType.Aim, false, true);
1063 }
1064 hasAimed = true;
1065 if (holdFireTimer > 0)
1066 {
1067 holdFireTimer -= deltaTime;
1068 return;
1069 }
1070 if (aimTimer > 0)
1071 {
1072 aimTimer -= deltaTime;
1073 return;
1074 }
1075 if (reloadTimer > 0) { return; }
1076 if (holdFireCondition != null && holdFireCondition()) { return; }
1077 sqrDistance = Vector2.DistanceSquared(character.WorldPosition, Enemy.WorldPosition);
1078 distanceTimer = distanceCheckInterval;
1079 if (WeaponComponent is MeleeWeapon meleeWeapon)
1080 {
1081 bool closeEnough = true;
1082 float sqrRange = meleeWeapon.Range * meleeWeapon.Range;
1083 if (character.AnimController.InWater)
1084 {
1085 if (sqrDistance > sqrRange)
1086 {
1087 closeEnough = false;
1088 }
1089 }
1090 else
1091 {
1092 // It's possible that the center point of the creature is out of reach, but we could still hit the character.
1093 float xDiff = Math.Abs(Enemy.WorldPosition.X - character.WorldPosition.X);
1094 if (xDiff > meleeWeapon.Range)
1095 {
1096 closeEnough = false;
1097 }
1098 float yDiff = Math.Abs(Enemy.WorldPosition.Y - character.WorldPosition.Y);
1099 if (yDiff > Math.Max(meleeWeapon.Range, 100))
1100 {
1101 closeEnough = false;
1102 }
1103 if (closeEnough && Enemy.WorldPosition.Y < character.WorldPosition.Y && yDiff > 25)
1104 {
1105 // The target is probably knocked down? -> try to reach it by crouching.
1106 HumanAIController.AnimController.Crouching = true;
1107 }
1108 }
1109 if (closeEnough)
1110 {
1111 UseWeapon(deltaTime);
1112 character.AIController.SteeringManager.Reset();
1113 }
1114 else if (!character.IsFacing(Enemy.WorldPosition))
1115 {
1116 // Don't do the facing check if we are close to the target, because it easily causes the character to get stuck here when it flips around.
1117 SetAimTimer(Rand.Range(1f, 1.5f) / AimSpeed);
1118 }
1119 }
1120 else
1121 {
1122 if (WeaponComponent is RepairTool repairTool)
1123 {
1124 if (sqrDistance > repairTool.Range * repairTool.Range) { return; }
1125 }
1126 float aimFactor = MathHelper.PiOver2 * (1 - AimAccuracy);
1127 if (VectorExtensions.Angle(VectorExtensions.Forward(Weapon.body.TransformedRotation), Enemy.Position - Weapon.Position) < MathHelper.PiOver4 + aimFactor)
1128 {
1129 if (myBodies == null)
1130 {
1131 myBodies = character.AnimController.Limbs.Select(l => l.body.FarseerBody);
1132 }
1133 var collisionCategories = Physics.CollisionCharacter | Physics.CollisionWall | Physics.CollisionLevel;
1134 var pickedBody = Submarine.PickBody(Weapon.SimPosition, Enemy.SimPosition, myBodies, collisionCategories, allowInsideFixture: true);
1135 if (pickedBody != null)
1136 {
1137 Character target = null;
1138 if (pickedBody.UserData is Character c)
1139 {
1140 target = c;
1141 }
1142 else if (pickedBody.UserData is Limb limb)
1143 {
1144 target = limb.character;
1145 }
1146 if (target != null && (target == Enemy || !HumanAIController.IsFriendly(target)))
1147 {
1148 UseWeapon(deltaTime);
1149 }
1150 }
1151 }
1152 }
1153 }

1157 // Never allow to attack characters with deadly weapons while trying to arrest.
1158 if (Mode == CombatMode.Arrest && isLethalWeapon) { return; }

1160 if (WeaponComponent is RangedWeapon rangedWeapon)
