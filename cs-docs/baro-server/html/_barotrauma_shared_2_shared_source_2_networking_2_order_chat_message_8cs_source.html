48 public static void WriteOrder(IWriteMessage msg, Order order, Character targetCharacter, bool isNewOrder)
49 {
50 msg.Write(order.Prefab.Identifier);
51 msg.Write(targetCharacter == null ? (UInt16)0 : targetCharacter.ID);
52 msg.Write(order.TargetSpatialEntity is Entity ? (order.TargetEntity as Entity).ID : (UInt16)0);
53
54 // The option of a Dismiss order is written differently so we know what order we target
55 // now that the game supports multiple current orders simultaneously
56 if (!order.IsDismissal)
57 {
58 msg.Write((byte)order.Options.IndexOf(order.Option));
59 }
60 else
61 {
62 if (order.Option != Identifier.Empty)
63 {
64 msg.Write(true);
65 string[] dismissedOrder = order.Option.Value.Split('.');
66 msg.Write((byte)dismissedOrder.Length);
67 if (dismissedOrder.Length > 0)
68 {
69 Identifier dismissedOrderIdentifier = dismissedOrder[0].ToIdentifier();
70 var orderPrefab = OrderPrefab.Prefabs[dismissedOrderIdentifier];
71 msg.Write(dismissedOrderIdentifier);
72 if (dismissedOrder.Length > 1)
73 {
74 Identifier dismissedOrderOption = dismissedOrder[1].ToIdentifier();
75 msg.Write((byte)orderPrefab.Options.IndexOf(dismissedOrderOption));
76 }
77 }
78 }
79 else
80 {
81 // If the order option is not specified for a Dismiss order,
82 // we dismiss all current orders for the character
83 msg.Write(false);
84 }
85 }
86
87 msg.Write((byte)order.ManualPriority);
88 msg.Write((byte)order.TargetType);
89 if (order.TargetType == Order.OrderTargetType.Position && order.TargetSpatialEntity is OrderTarget orderTarget)
90 {
91 msg.Write(true);
92 msg.Write(orderTarget.Position.X);
93 msg.Write(orderTarget.Position.Y);
94 msg.Write(orderTarget.Hull == null ? (UInt16)0 : orderTarget.Hull.ID);
95 }
96 else
97 {
98 msg.Write(false);
99 if (order.TargetType == Order.OrderTargetType.WallSection)
100 {
101 msg.Write((byte)(order.WallSectionIndex ?? 0));
102 }
103 }
104
105 msg.Write(isNewOrder);
106 }

143 public static OrderMessageInfo ReadOrder(IReadMessage msg)
144 {
145 Identifier orderIdentifier = msg.ReadIdentifier();
146 ushort targetCharacterId = msg.ReadUInt16();
147 Character targetCharacter = targetCharacterId != Entity.NullEntityID ? Entity.FindEntityByID(targetCharacterId) as Character : null;
148 ushort targetEntityId = msg.ReadUInt16();
149 Entity targetEntity = targetEntityId != Entity.NullEntityID ? Entity.FindEntityByID(targetEntityId) : null;
150
151 int? optionIndex = null;
152 Identifier orderOption = Identifier.Empty;
153 // The option of a Dismiss order is written differently so we know what order we target
154 // now that the game supports multiple current orders simultaneously
155 if (orderIdentifier != Identifier.Empty)
156 {
157 var orderPrefab = OrderPrefab.Prefabs[orderIdentifier];
158 if (!orderPrefab.IsDismissal)
159 {
160 optionIndex = msg.ReadByte();
161 }
162 // Does the dismiss order have a specified target?
163 else if (msg.ReadBoolean())
164 {
165 int identifierCount = msg.ReadByte();
166 if (identifierCount > 0)
167 {
168 Identifier dismissedOrderIdentifier = msg.ReadIdentifier();
169 OrderPrefab dismissedOrderPrefab = null;
170 if (dismissedOrderIdentifier != Identifier.Empty)
171 {
172 dismissedOrderPrefab = OrderPrefab.Prefabs[dismissedOrderIdentifier];
173 orderOption = dismissedOrderPrefab.Identifier;
174 }
175 if (identifierCount > 1)
176 {
177 int dismissedOrderOptionIndex = msg.ReadByte();
178 if (dismissedOrderPrefab != null)
179 {
180 var options = dismissedOrderPrefab.Options;
181 if (options != null && dismissedOrderOptionIndex >= 0 && dismissedOrderOptionIndex < options.Length)
182 {
183 orderOption = $"{orderOption.Value}.{options[dismissedOrderOptionIndex]}".ToIdentifier();
184 }
185 }
186 }
187 }
188 }
189 }
190 else
191 {
192 optionIndex = msg.ReadByte();
193 }
194
195 int orderPriority = msg.ReadByte();
196 OrderTarget orderTargetPosition = null;
197 Order.OrderTargetType orderTargetType = (Order.OrderTargetType)msg.ReadByte();
198 int wallSectionIndex = 0;
199 if (msg.ReadBoolean())
200 {
201 float x = msg.ReadSingle();
202 float y = msg.ReadSingle();
203 ushort hullId = msg.ReadUInt16();
204 var hull = hullId != Entity.NullEntityID ? Entity.FindEntityByID(hullId) as Hull : null;
205 orderTargetPosition = new OrderTarget(new Vector2(x, y), hull, creatingFromExistingData: true);
206 }
207 else if (orderTargetType == Order.OrderTargetType.WallSection)
208 {
209 wallSectionIndex = msg.ReadByte();
210 }
211
212 bool isNewOrder = msg.ReadBoolean();
213 return new OrderMessageInfo(orderIdentifier, orderOption, optionIndex, targetCharacter,
214 orderTargetType, targetEntity, orderTargetPosition, wallSectionIndex, orderPriority, isNewOrder);
215 }
216 }
