71  public override void Update(float deltaTime)
72  {
73  if (!started) { return; }
74 
75  if (OnOwnerDetermined != null && OwnerConnection != null)
76  {
77  OnOwnerDetermined?.Invoke(OwnerConnection);
78  OnOwnerDetermined = null;
79  }
80 
81  //backwards for loop so we can remove elements while iterating
82  for (int i = connectedClients.Count - 1; i >= 0; i--)
83  {
84  SteamP2PConnection conn = connectedClients[i] as SteamP2PConnection;
85  conn.Decay(deltaTime);
86  if (conn.Timeout < 0.0)
87  {
88  Disconnect(conn, "Timed out");
89  }
90  }
91 
92  try
93  {
94  while (ChildServerRelay.Read(out byte[] incBuf))
95  {
96  IReadMessage inc = new ReadOnlyMessage(incBuf, false, 0, incBuf.Length, OwnerConnection);
97 
98  HandleDataMessage(inc);
99  }
100  }
101 
102  catch (Exception e)
103  {
104  string errorMsg = "Server failed to read an incoming message. {" + e + "}\n" + e.StackTrace.CleanupStackTrace();
105  GameAnalyticsManager.AddErrorEventOnce("SteamP2PServerPeer.Update:ClientReadException" + e.TargetSite.ToString(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
106 #if DEBUG
107  DebugConsole.ThrowError(errorMsg);
108 #else
109  if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError(errorMsg); }
110 #endif
111  }
112 
113  for (int i = 0; i < pendingClients.Count; i++)
114  {
115  PendingClient pendingClient = pendingClients[i];
116  UpdatePendingClient(pendingClient);
117  if (i >= pendingClients.Count || pendingClients[i] != pendingClient) { i--; }
118  }

121  private void HandleDataMessage(IReadMessage inc)
122  {
123  if (!started) { return; }
124 
125  UInt64 senderSteamId = inc.ReadUInt64();
126  UInt64 ownerSteamId = inc.ReadUInt64();
127 
128  PacketHeader packetHeader = (PacketHeader)inc.ReadByte();
129 
130  if (packetHeader.IsServerMessage())
131  {
132  DebugConsole.ThrowError("Got server message from" + senderSteamId.ToString());
133  return;
134  }
135 
136  if (senderSteamId != OwnerSteamID) //sender is remote, handle disconnects and heartbeats
137  {
138  PendingClient pendingClient = pendingClients.Find(c => c.SteamID == senderSteamId);
139  SteamP2PConnection connectedClient = connectedClients.Find(c => c.SteamID == senderSteamId) as SteamP2PConnection;
140 
141  pendingClient?.Heartbeat();
142  connectedClient?.Heartbeat();
143 
144  string banReason;
145  if (serverSettings.BanList.IsBanned(senderSteamId, out banReason) ||
146  serverSettings.BanList.IsBanned(ownerSteamId, out banReason))
147  {
148  if (pendingClient != null)
149  {
150  RemovePendingClient(pendingClient, DisconnectReason.Banned, banReason);
151  }
152  else if (connectedClient != null)
153  {
154  Disconnect(connectedClient, DisconnectReason.Banned.ToString() + "/ "+ banReason);
155  }
156  return;
157  }
158  else if (packetHeader.IsDisconnectMessage())
159  {
160  if (pendingClient != null)
161  {
162  string disconnectMsg = $"ServerMessage.HasDisconnected~[client]={pendingClient.Name}";
163  RemovePendingClient(pendingClient, DisconnectReason.Unknown, disconnectMsg);
164  }
165  else if (connectedClient != null)
166  {
167  string disconnectMsg = $"ServerMessage.HasDisconnected~[client]={connectedClient.Name}";
168  Disconnect(connectedClient, disconnectMsg, false);
169  }
170  return;
171  }
172  else if (packetHeader.IsHeartbeatMessage())
173  {
174  //message exists solely as a heartbeat, ignore its contents
175  return;
176  }
177  else if (packetHeader.IsConnectionInitializationStep())
178  {
179 
180  if (pendingClient != null)
181  {
182  if (ownerSteamId != 0)
183  {
184  pendingClient.Connection.SetOwnerSteamIDIfUnknown(ownerSteamId);
185  }
186  ReadConnectionInitializationStep(pendingClient, new ReadOnlyMessage(inc.Buffer, false, inc.BytePosition, inc.LengthBytes - inc.BytePosition, null));
187  }
188  else
189  {
190  ConnectionInitialization initializationStep = (ConnectionInitialization)inc.ReadByte();
191  if (initializationStep == ConnectionInitialization.ConnectionStarted)
192  {
193  pendingClients.Add(new PendingClient(new SteamP2PConnection("PENDING", senderSteamId)) { SteamID = senderSteamId });
194  }
195  }
196  }
197  else if (connectedClient != null)
198  {
199  UInt16 length = inc.ReadUInt16();
200 
201  IReadMessage msg = new ReadOnlyMessage(inc.Buffer, packetHeader.IsCompressed(), inc.BytePosition, length, connectedClient);
202  OnMessageReceived?.Invoke(connectedClient, msg);
203  }
204  }
205  else //sender is owner
206  {
207  if (OwnerConnection != null) { (OwnerConnection as SteamP2PConnection).Heartbeat(); }
208 
209  if (packetHeader.IsDisconnectMessage())
210  {
211  DebugConsole.ThrowError("Received disconnect message from owner");
212  return;
213  }
214  if (packetHeader.IsServerMessage())
215  {
216  DebugConsole.ThrowError("Received server message from owner");
217  return;
218  }
219  if (packetHeader.IsConnectionInitializationStep())
220  {
221  if (OwnerConnection == null)
222  {
223  string ownerName = inc.ReadString();
224  OwnerConnection = new SteamP2PConnection(ownerName, OwnerSteamID)
225  {
226  Language = GameSettings.CurrentConfig.Language
227  };
228  OwnerConnection.SetOwnerSteamIDIfUnknown(OwnerSteamID);
229 
230  OnInitializationComplete?.Invoke(OwnerConnection);
231  }
232  return;
233  }
234  if (packetHeader.IsHeartbeatMessage())
235  {
236  return;
237  }
238  else
239  {
240  UInt16 length = inc.ReadUInt16();
241 
242  IReadMessage msg = new ReadOnlyMessage(inc.Buffer, packetHeader.IsCompressed(), inc.BytePosition, length, OwnerConnection);
243  OnMessageReceived?.Invoke(OwnerConnection, msg);
244  }
245  }
