131 public int DetermineItemSwapCost(Item item, ItemPrefab? replacement)
132 {
133 if (replacement == null)
134 {
135 replacement = ItemPrefab.Find("", item.Prefab.SwappableItem.ReplacementOnUninstall);
136 if (replacement == null)
137 {
138 DebugConsole.ThrowError("Failed to determine swap cost for item \"{}\". Trying to uninstall the item but no replacement item found.");
139 return 0;
140 }
141 }
142
143 int price = 0;
144 if (replacement == item.Prefab)
145 {
146 if (item.PendingItemSwap != null)
147 {
148 //refund the pending swap
149 price -= item.PendingItemSwap.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
150 //buy back the current item
151 price += item.Prefab.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
152 }
153 }
154 else
155 {
156 price = replacement.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
157 if (item.PendingItemSwap != null)
158 {
159 //refund the pending swap
160 price -= item.PendingItemSwap.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
161 //buy back the current item
162 price += item.Prefab.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
163 }
164 //refund the current item
165 if (replacement != ((MapEntity)item).Prefab)
166 {
167 price -= item.Prefab.SwappableItem.GetPrice(Campaign?.Map?.CurrentLocation);
168 }
169 }
170 return price;
171 }

182 public void PurchaseUpgrade(UpgradePrefab prefab, UpgradeCategory category, bool force = false, Client? client = null)
183 {
184 if (!CanUpgradeSub())
185 {
186 DebugConsole.ThrowError("Cannot upgrade when switching to another submarine.");
187 return;
188 }
189
190 int price = prefab.Price.GetBuyprice(GetUpgradeLevel(prefab, category), Campaign.Map?.CurrentLocation);
191 int currentLevel = GetUpgradeLevel(prefab, category);
192
193 if (currentLevel + 1 > prefab.MaxLevel)
194 {
195 DebugConsole.ThrowError($"Tried to purchase \"{prefab.Name}\" over the max level! ({currentLevel + 1} > {prefab.MaxLevel}). The transaction has been cancelled.");
196 return;
197 }
198
199 if (price < 0)
200 {
201 Location? location = Campaign.Map?.CurrentLocation;
202 LogError($"Upgrade price is less than 0! ({price})",
203 new Dictionary<string, object?>
204 {
205 { "Level", currentLevel },
206 { "Saved Level", GetRealUpgradeLevel(prefab, category) },
207 { "Upgrade", $"{category.Identifier}.{prefab.Identifier}" },
208 { "Location", location?.Type },
209 { "Reputation", $"{location?.Reputation?.Value} / {location?.Reputation?.MaxReputation}" },
210 { "Base Price", prefab.Price.BasePrice }
211 });
212 }
213
214 if (force)
215 {
216 price = 0;
217 }
218
219 if (Campaign.TryPurchase(client, price))
220 {
221 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
222 {
223 // only make the NPC speak if more than 5 minutes have passed since the last purchased service
224 if (lastUpgradeSpeak == DateTime.MinValue || lastUpgradeSpeak.AddMinutes(5) < DateTime.Now)
225 {
226 UpgradeNPCSpeak(TextManager.Get("Dialog.UpgradePurchased").Value, Campaign.IsSinglePlayer);
227 lastUpgradeSpeak = DateTime.Now;
228 }
229 }
230
231 GameAnalyticsManager.AddMoneySpentEvent(price, GameAnalyticsManager.MoneySink.SubmarineUpgrade, prefab.Identifier.Value);
232
233 PurchasedUpgrade? upgrade = FindMatchingUpgrade(prefab, category);
234
235#if CLIENT
236 DebugLog($"CLIENT: Purchased level {GetUpgradeLevel(prefab, category) + 1} {category.Name}.{prefab.Name} for {price}", GUIStyle.Orange);
237#endif
238
239 if (upgrade == null)
240 {
241 PendingUpgrades.Add(new PurchasedUpgrade(prefab, category));
242 }
243 else
244 {
245 upgrade.Level++;
246 }
247#if CLIENT
248 // tell the server that this item is yet to be paid for server side
249 PurchasedUpgrades.Add(new PurchasedUpgrade(prefab, category));
250#endif
251 OnUpgradesChanged?.Invoke();
252 }
253 else
254 {
255 DebugConsole.ThrowError("Tried to purchase an upgrade with insufficient funds, the transaction has not been completed.\n" +
256 $"Upgrade: {prefab.Name}, Cost: {price}, Have: {Campaign.GetWallet(client).Balance}");
257 }

263 public void PurchaseItemSwap(Item itemToRemove, ItemPrefab itemToInstall, bool force = false, Client? client = null)
264 {
265 if (!CanUpgradeSub())
266 {
267 DebugConsole.ThrowError("Cannot swap items when switching to another submarine.");
268 return;
269 }
270 if (itemToRemove == null)
271 {
272 DebugConsole.ThrowError($"Cannot swap null item!");
273 return;
274 }
275 if (itemToRemove.HiddenInGame)
276 {
277 DebugConsole.ThrowError($"Cannot swap item \"{itemToRemove.Name}\" because it's set to be hidden in-game.");
278 return;
279 }
280 if (!itemToRemove.AllowSwapping)
281 {
282 DebugConsole.ThrowError($"Cannot swap item \"{itemToRemove.Name}\" because it's configured to be non-swappable.");
283 return;
284 }
285 if (!UpgradeCategory.Categories.Any(c => c.ItemTags.Any(t => itemToRemove.HasTag(t)) && c.ItemTags.Any(t => itemToInstall.Tags.Contains(t))))
286 {
287 DebugConsole.ThrowError($"Failed to swap item \"{itemToRemove.Name}\" with \"{itemToInstall.Name}\" (not in the same upgrade category).");
288 return;
289 }
290
291 if (((MapEntity)itemToRemove).Prefab == itemToInstall)
292 {
293 DebugConsole.ThrowError($"Failed to swap item \"{itemToRemove.Name}\" (trying to swap with the same item!).");
294 return;
295 }
296 SwappableItem? swappableItem = itemToRemove.Prefab.SwappableItem;
297 if (swappableItem == null)
298 {
299 DebugConsole.ThrowError($"Failed to swap item \"{itemToRemove.Name}\" (not configured as a swappable item).");
300 return;
301 }
302
303 var linkedItems = GetLinkedItemsToSwap(itemToRemove);
304
305 int price = 0;
306 if (!itemToRemove.AvailableSwaps.Contains(itemToInstall))
307 {
308 price = itemToInstall.SwappableItem.GetPrice(Campaign.Map?.CurrentLocation) * linkedItems.Count;
309 }
310
311 if (force)
312 {
313 price = 0;
314 }
315
316 if (Campaign.TryPurchase(client, price))
317 {
318 PurchasedItemSwaps.RemoveAll(p => linkedItems.Contains(p.ItemToRemove));
319 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
320 {
321 // only make the NPC speak if more than 5 minutes have passed since the last purchased service
322 if (lastUpgradeSpeak == DateTime.MinValue || lastUpgradeSpeak.AddMinutes(5) < DateTime.Now)
323 {
324 UpgradeNPCSpeak(TextManager.Get("Dialog.UpgradePurchased").Value, Campaign.IsSinglePlayer);
325 lastUpgradeSpeak = DateTime.Now;
326 }
327 }
328
329 GameAnalyticsManager.AddMoneySpentEvent(price, GameAnalyticsManager.MoneySink.SubmarineWeapon, itemToInstall.Identifier.Value);
330
331 foreach (Item itemToSwap in linkedItems)
332 {
333 itemToSwap.AvailableSwaps.Add(itemToSwap.Prefab);
334 if (itemToInstall != null && !itemToSwap.AvailableSwaps.Contains(itemToInstall))
335 {
336 itemToSwap.PurchasedNewSwap = true;
337 itemToSwap.AvailableSwaps.Add(itemToInstall);
338 }
339
340 if (itemToSwap.Prefab != itemToInstall && itemToInstall != null)
341 {
342 itemToSwap.PendingItemSwap = itemToInstall;
343 PurchasedItemSwaps.Add(new PurchasedItemSwap(itemToSwap, itemToInstall));
344 DebugLog($"CLIENT: Swapped item \"{itemToSwap.Name}\" with \"{itemToInstall.Name}\".", Color.Orange);
345 }
346 else
347 {
348 DebugLog($"CLIENT: Cancelled swapping the item \"{itemToSwap.Name}\" with \"{(itemToSwap.PendingItemSwap?.Name ?? null)}\".", Color.Orange);
349 }
350 }
351
352 OnUpgradesChanged?.Invoke();
353 }
354 else
355 {
356 DebugConsole.ThrowError("Tried to swap an item with insufficient funds, the transaction has not been completed.\n" +
357 $"Item to remove: {itemToRemove.Name}, Item to install: {itemToInstall.Name}, Cost: {price}, Have: {Campaign.GetWallet(client).Balance}");
358 }

364 public void CancelItemSwap(Item itemToRemove, bool force = false)
365 {
366 if (!CanUpgradeSub())
367 {
368 DebugConsole.ThrowError("Cannot swap items when switching to another submarine.");
369 return;
370 }
371
372 if (itemToRemove?.PendingItemSwap == null && (itemToRemove?.Prefab.SwappableItem?.ReplacementOnUninstall.IsEmpty ?? true))
373 {
374 DebugConsole.ThrowError($"Cannot uninstall item \"{itemToRemove?.Name}\" (no replacement item configured).");
375 return;
376 }
377
378 SwappableItem? swappableItem = itemToRemove.Prefab.SwappableItem;
379 if (swappableItem == null)
380 {
381 DebugConsole.ThrowError($"Failed to uninstall item \"{itemToRemove.Name}\" (not configured as a swappable item).");
382 return;
383 }
384
385 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
386 {
387 // only make the NPC speak if more than 5 minutes have passed since the last purchased service
388 if (lastUpgradeSpeak == DateTime.MinValue || lastUpgradeSpeak.AddMinutes(5) < DateTime.Now)
389 {
390 UpgradeNPCSpeak(TextManager.Get("Dialog.UpgradePurchased").Value, Campaign.IsSinglePlayer);
391 lastUpgradeSpeak = DateTime.Now;
392 }
393 }
394
395 var linkedItems = GetLinkedItemsToSwap(itemToRemove);
396
397 foreach (Item itemToCancel in linkedItems)
398 {
399 if (itemToCancel.PendingItemSwap == null)
400 {
401 var replacement = MapEntityPrefab.Find("", swappableItem.ReplacementOnUninstall) as ItemPrefab;
402 if (replacement == null)
403 {
404 DebugConsole.ThrowError($"Failed to uninstall item \"{itemToCancel.Name}\". Could not find the replacement item \"{swappableItem.ReplacementOnUninstall}\".");
405 return;
406 }
407 PurchasedItemSwaps.RemoveAll(p => p.ItemToRemove == itemToCancel);
408 PurchasedItemSwaps.Add(new PurchasedItemSwap(itemToCancel, replacement));
409 DebugLog($"Uninstalled item item \"{itemToCancel.Name}\".", Color.Orange);
410 itemToCancel.PendingItemSwap = replacement;
411 }
412 else
413 {
414 PurchasedItemSwaps.RemoveAll(p => p.ItemToRemove == itemToCancel);
415 DebugLog($"Cancelled swapping the item \"{itemToCancel.Name}\" with \"{itemToCancel.PendingItemSwap.Name}\".", Color.Orange);
416 itemToCancel.PendingItemSwap = null;
417 }
418 }

425 public List<Item> GetLinkedItemsToSwap(Item item)
426 {
427 List<Item> linkedItems = new List<Item>() { item };
428 foreach (MapEntity linkedEntity in item.linkedTo)
429 {
430 foreach (MapEntity secondLinkedEntity in linkedEntity.linkedTo)
431 {
432 if (!(secondLinkedEntity is Item linkedItem) || linkedItem == item) { continue; }
433 if (linkedItem.AllowSwapping &&
434 linkedItem.Prefab.SwappableItem != null && (linkedItem.Prefab.SwappableItem.CanBeBought || item.Prefab.SwappableItem.ReplacementOnUninstall == ((MapEntity)linkedItem).Prefab.Identifier) &&
435 linkedItem.Prefab.SwappableItem.SwapIdentifier.Equals(item.Prefab.SwappableItem.SwapIdentifier, StringComparison.OrdinalIgnoreCase))
436 {
437 linkedItems.Add(linkedItem);
438 }
439 }
440 }
441 return linkedItems;

458 public void ApplyUpgrades()
459 {
460 PurchasedUpgrades.Clear();
461 PurchasedItemSwaps.Clear();
462 if (Submarine.MainSub == null) { return; }
463
464 List<PurchasedUpgrade> pendingUpgrades = PendingUpgrades;
465
466 if (Level.Loaded is { Type: LevelData.LevelType.Outpost })
467 {
468 return;
469 }
470
471 if (GameMain.NetworkMember is { IsClient: true })
472 {
473 if (loadedUpgrades != null)
474 {
475 // client receives pending upgrades from the save file
476 pendingUpgrades = loadedUpgrades;
477 }
478 }
479
480 DebugConsole.Log("Applying upgrades...");
481 foreach (var (prefab, category, level) in pendingUpgrades)
482 {
483 int newLevel = BuyUpgrade(prefab, category, Submarine.MainSub, level);
484 DebugConsole.Log($" - {category.Identifier}.{prefab.Identifier} lvl. {level}, new: ({newLevel})");
485 SetUpgradeLevel(prefab, category, Math.Clamp(GetRealUpgradeLevel(prefab, category) + level, 0, prefab.MaxLevel));
486 }
487
488 PendingUpgrades.Clear();
489 loadedUpgrades?.Clear();
490 loadedUpgrades = null;

525 public void SanityCheckUpgrades(Submarine submarine)
526 {
527 // check walls
528 foreach (Structure wall in submarine.GetWalls(UpgradeAlsoConnectedSubs))
529 {
530 foreach (UpgradeCategory category in UpgradeCategory.Categories)
531 {
532 foreach (UpgradePrefab prefab in UpgradePrefab.Prefabs)
533 {
534 int level = GetRealUpgradeLevel(prefab, category);
535 if (level == 0 || !prefab.IsWallUpgrade) { continue; }
536
537 Upgrade? upgrade = wall.GetUpgrade(prefab.Identifier);
538
539 bool isOverMax = IsOverMaxLevel(level, prefab);
540 if (isOverMax)
541 {
542 SetUpgradeLevel(prefab, category, prefab.MaxLevel);
543 level = prefab.MaxLevel;
544 }
545
546 if (upgrade == null || upgrade.Level != level || isOverMax)
547 {
548 DebugLog($"{((MapEntity)wall).Prefab.Name} has incorrect \"{prefab.Name}\" level! Expected {level} but got {upgrade?.Level ?? 0}. Fixing...");
549 FixUpgradeOnItem(wall, prefab, level);
550 }
551 }
552 }
553 }
554
555 // Check items
556 foreach (Item item in submarine.GetItems(UpgradeAlsoConnectedSubs))
557 {
558 foreach (UpgradeCategory category in UpgradeCategory.Categories)
559 {
560 foreach (UpgradePrefab prefab in UpgradePrefab.Prefabs)
561 {
562 if (!category.CanBeApplied(item, prefab)) { continue; }
563
564 int level = GetRealUpgradeLevel(prefab, category);
565 if (level == 0) { continue; }
566
567 Upgrade? upgrade = item.GetUpgrade(prefab.Identifier);
568 bool isOverMax = IsOverMaxLevel(level, prefab);
569 if (isOverMax)
570 {
571 SetUpgradeLevel(prefab, category, prefab.MaxLevel);
572 level = prefab.MaxLevel;
573 }
574
575 if (upgrade == null || upgrade.Level != level || isOverMax)
576 {
577 DebugLog($"{((MapEntity)item).Prefab.Name} has incorrect \"{prefab.Name}\" level! Expected {level} but got {upgrade?.Level ?? 0}{(isOverMax ? " (Over max level!)" : string.Empty)}. Fixing...");
578 FixUpgradeOnItem(item, prefab, level);
579 }
580 }
581 }

603 private static int BuyUpgrade(UpgradePrefab prefab, UpgradeCategory category, Submarine submarine, int level = 1, Submarine? parentSub = null)
604 {
605 int? newLevel = null;
606 if (category.IsWallUpgrade)
607 {
608 foreach (Structure structure in submarine.GetWalls(UpgradeAlsoConnectedSubs))
609 {
610 Upgrade upgrade = new Upgrade(structure, prefab, level);
611 structure.AddUpgrade(upgrade, createNetworkEvent: false);
612
613 Upgrade? newUpgrade = structure.GetUpgrade(prefab.Identifier);
614 if (newUpgrade != null)
615 {
616 SanityCheck(newUpgrade, structure);
617 newLevel ??= newUpgrade.Level;
618 }
619 }
620 }
621 else
622 {
623 foreach (Item item in submarine.GetItems(UpgradeAlsoConnectedSubs))
624 {
625 if (category.CanBeApplied(item, prefab))
626 {
627 Upgrade upgrade = new Upgrade(item, prefab, level);
628 item.AddUpgrade(upgrade, createNetworkEvent: false);
629
630 Upgrade? newUpgrade = item.GetUpgrade(prefab.Identifier);
631 if (newUpgrade != null)
632 {
633 SanityCheck(newUpgrade, item);
634 newLevel ??= newUpgrade.Level;
635 }
636 }
637 }
638 }
639
640 foreach (Submarine loadedSub in Submarine.Loaded.Where(sub => sub != submarine))
641 {
642 if (loadedSub == parentSub) { continue; }
643 XElement? root = loadedSub.Info?.SubmarineElement;
644 if (root == null) { continue; }
645
646 if (root.Name.ToString().Equals("LinkedSubmarine", StringComparison.OrdinalIgnoreCase))
647 {
648 if (root.Attribute("location") == null) { continue; }
649
650 // Check if this is our linked submarine
651 ushort dockingPortID = (ushort) root.GetAttributeInt("originallinkedto", 0);
652 if (dockingPortID > 0 && submarine.GetItems(true).Any(item => item.ID == dockingPortID))
653 {
654 BuyUpgrade(prefab, category, loadedSub, level, submarine);
655 }
656 }
657 }
658
659 return newLevel ?? -1;

749 private void LoadPendingUpgrades(XElement? element, bool isSingleplayer = true)
750 {
751 if (!(element is { HasElements: true })) { return; }
752
753 List<PurchasedUpgrade> pendingUpgrades = new List<PurchasedUpgrade>();
754
755 // ReSharper disable once LoopCanBeConvertedToQuery
756 foreach (XElement upgrade in element.Elements())
757 {
758 Identifier categoryIdentifier = upgrade.GetAttributeIdentifier("category", Identifier.Empty);
759 UpgradeCategory? category = UpgradeCategory.Find(categoryIdentifier);
760 if (categoryIdentifier.IsEmpty || category == null) { continue; }
761
762 Identifier prefabIdentifier = upgrade.GetAttributeIdentifier("prefab", Identifier.Empty);
763 UpgradePrefab? prefab = UpgradePrefab.Find(prefabIdentifier);
764 if (prefabIdentifier.IsEmpty || prefab == null) { continue; }
765
766 int level = upgrade.GetAttributeInt("level", -1);
767 if (level < 0) { continue; }
768
769 pendingUpgrades.Add(new PurchasedUpgrade(prefab, category, level));
770 }
771
772#if CLIENT
773 if (isSingleplayer)
774 {
775 SetPendingUpgrades(pendingUpgrades);
776 }
777 else
778 {
779 loadedUpgrades = pendingUpgrades;
780 }
781#else
782 SetPendingUpgrades(pendingUpgrades);
783#endif
784 }
