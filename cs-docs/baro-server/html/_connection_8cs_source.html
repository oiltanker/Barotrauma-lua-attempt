238  public void SetWire(int index, Wire wire)
239  {
240  Wire previousWire = wires[index];
241  if (wire != previousWire && previousWire != null)
242  {
243  var otherConnection = previousWire.OtherConnection(this);
244  if (otherConnection != null)
245  {
246  //Change the connection grids or flag them for updating
247  if (IsPower && otherConnection.IsPower && Grid != null)
248  {
249  //Check if both connections belong to a larger grid
250  if (otherConnection.recipients.Count > 1 && recipients.Count > 1)
251  {
252  Powered.ChangedConnections.Add(otherConnection);
253  Powered.ChangedConnections.Add(this);
254  }
255  else if (recipients.Count > 1)
256  {
257  //This wire was the only one at the other grid
258  otherConnection.Grid?.RemoveConnection(otherConnection);
259  otherConnection.Grid = null;
260  }
261  else if (otherConnection.recipients.Count > 1)
262  {
263  Grid?.RemoveConnection(this);
264  Grid = null;
265  }
266  else if (Grid.Connections.Count == 2)
267  {
268  //Delete the grid as these were the only 2 devices
269  Powered.Grids.Remove(Grid.ID);
270  Grid = null;
271  otherConnection.Grid = null;
272  }
273  }
274  otherConnection.recipientsDirty = true;
275  }
276  }
277 
278  wires[index] = wire;
279  recipientsDirty = true;
280  if (wire != null)
281  {
282 
283  ConnectionPanel.DisconnectedWires.Remove(wire);
284  var otherConnection = wire.OtherConnection(this);
285  if (otherConnection != null)
286  {
287  //Set the other connection grid if a grid exists already
288  if (Powered.ValidPowerConnection(this, otherConnection))
289  {
290  if (Grid == null && otherConnection.Grid != null)
291  {
292  otherConnection.Grid.AddConnection(this);
293  Grid = otherConnection.Grid;
294  }
295  else if (Grid != null && otherConnection.Grid == null)
296  {
297  Grid.AddConnection(otherConnection);
298  otherConnection.Grid = Grid;
299  }
300  else
301  {
302  //Flag change so that proper grids can be formed
303  Powered.ChangedConnections.Add(this);
304  Powered.ChangedConnections.Add(otherConnection);
305  }
306  }
307 
308  otherConnection.recipientsDirty = true;
309  }

317  if (wires[i] == null) { continue; }

319  Connection recipient = wires[i].OtherConnection(this);
