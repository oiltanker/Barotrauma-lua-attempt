77
78#if CLIENT
79 if (connector == null)

88 ConnectionPanel = connectionPanel;
89 item = connectionPanel.Item;

91 MaxWires = element.GetAttributeInt("maxwires", DefaultMaxWires);
92 MaxWires = Math.Max(element.Elements().Count(e => e.Name.ToString().Equals("link", StringComparison.OrdinalIgnoreCase)), MaxWires);
93
94 MaxPlayerConnectableWires = element.GetAttributeInt("maxplayerconnectablewires", MaxWires);
95 wires = new Wire[MaxWires];
96
97 IsOutput = element.Name.ToString() == "output";
98 Name = element.GetAttributeString("name", IsOutput ? "output" : "input");
99
100 string displayNameTag = "", fallbackTag = "";
101 //if displayname is not present, attempt to find it from the prefab
102 if (element.GetAttribute("displayname") == null)
103 {
104 foreach (var subElement in item.Prefab.ConfigElement.Elements())
105 {
106 if (!subElement.Name.ToString().Equals("connectionpanel", StringComparison.OrdinalIgnoreCase)) { continue; }
107
108 foreach (XElement connectionElement in subElement.Elements())
109 {
110 string prefabConnectionName = connectionElement.GetAttributeString("name", null);
111 string[] aliases = connectionElement.GetAttributeStringArray("aliases", Array.Empty<string>());
112 if (prefabConnectionName == Name || aliases.Contains(Name))
113 {
114 displayNameTag = connectionElement.GetAttributeString("displayname", "");
115 fallbackTag = connectionElement.GetAttributeString("fallbackdisplayname", "");
116 }
117 }
118 }
119 }
120 else
121 {
122 displayNameTag = element.GetAttributeString("displayname", "");
123 fallbackTag = element.GetAttributeString("fallbackdisplayname", null);
124 }
125
126 if (!string.IsNullOrEmpty(displayNameTag))
127 {
128 //extract the tag parts in case the tags contains variables
129 string tagWithoutVariables = displayNameTag?.Split('~')?.FirstOrDefault();
130 string fallbackTagWithoutVariables = fallbackTag?.Split('~')?.FirstOrDefault();
131 //use displayNameTag if found, otherwise fallBack
132 if (TextManager.ContainsTag(tagWithoutVariables))
133 {
134 DisplayName = TextManager.GetServerMessage(displayNameTag);
135 }
136 else if (TextManager.ContainsTag(fallbackTagWithoutVariables))
137 {
138 DisplayName = TextManager.GetServerMessage(fallbackTag);
139 }
140 }
141
142 if (DisplayName.IsNullOrEmpty())
143 {
144#if DEBUG
145 DebugConsole.ThrowError("Missing display name in connection " + item.Name + ": " + Name);
146#endif
147 DisplayName = Name;
148 }
149
150 IsPower = Name == "power_in" || Name == "power" || Name == "power_out";
151
152
153 wireId = new ushort[MaxWires];
154
155 foreach (var subElement in element.Elements())
156 {
157 switch (subElement.Name.ToString().ToLowerInvariant())
158 {
159 case "link":
160 int index = -1;
161 for (int i = 0; i < MaxWires; i++)
162 {
163 if (wireId[i] < 1) { index = i; }
164 }
165 if (index == -1) { break; }
166
167 int id = subElement.GetAttributeInt("w", 0);
168 if (id < 0) { id = 0; }
169 wireId[index] = idRemap.GetOffsetId(id);
170
171 break;
172 case "statuseffect":
173 Effects ??= new List<StatusEffect>();
174 Effects.Add(StatusEffect.Load(subElement, item.Name + ", connection " + Name));
175 break;
176 }
177 }

238 public void SetWire(int index, Wire wire)
239 {
240 Wire previousWire = wires[index];
241 if (wire != previousWire && previousWire != null)
242 {
243 var otherConnection = previousWire.OtherConnection(this);
244 if (otherConnection != null)
245 {
246 //Change the connection grids or flag them for updating
247 if (IsPower && otherConnection.IsPower && Grid != null)
248 {
249 //Check if both connections belong to a larger grid
250 if (otherConnection.recipients.Count > 1 && recipients.Count > 1)
251 {
252 Powered.ChangedConnections.Add(otherConnection);
253 Powered.ChangedConnections.Add(this);
254 }
255 else if (recipients.Count > 1)
256 {
257 //This wire was the only one at the other grid
258 otherConnection.Grid?.RemoveConnection(otherConnection);
259 otherConnection.Grid = null;
260 }
261 else if (otherConnection.recipients.Count > 1)
262 {
263 Grid?.RemoveConnection(this);
264 Grid = null;
265 }
266 else if (Grid.Connections.Count == 2)
267 {
268 //Delete the grid as these were the only 2 devices
269 Powered.Grids.Remove(Grid.ID);
270 Grid = null;
271 otherConnection.Grid = null;
272 }
273 }
274 otherConnection.recipientsDirty = true;
275 }
276 }
277
278 wires[index] = wire;
279 recipientsDirty = true;
280 if (wire != null)
281 {
282
283 ConnectionPanel.DisconnectedWires.Remove(wire);
284 var otherConnection = wire.OtherConnection(this);
285 if (otherConnection != null)
286 {
287 //Set the other connection grid if a grid exists already
288 if (Powered.ValidPowerConnection(this, otherConnection))
289 {
290 if (Grid == null && otherConnection.Grid != null)
291 {
292 otherConnection.Grid.AddConnection(this);
293 Grid = otherConnection.Grid;
294 }
295 else if (Grid != null && otherConnection.Grid == null)
296 {
297 Grid.AddConnection(otherConnection);
298 otherConnection.Grid = Grid;
299 }
300 else
301 {
302 //Flag change so that proper grids can be formed
303 Powered.ChangedConnections.Add(this);
304 Powered.ChangedConnections.Add(otherConnection);
305 }
306 }
307
308 otherConnection.recipientsDirty = true;
309 }
310 }

313 public void SendSignal(Signal signal)
314 {
315 for (int i = 0; i < MaxWires; i++)
316 {
317 if (wires[i] == null) { continue; }
318
319 Connection recipient = wires[i].OtherConnection(this);
320 if (recipient == null) { continue; }
321 if (recipient.item == this.item || signal.source?.LastSentSignalRecipients.LastOrDefault() == recipient) { continue; }
322
323 signal.source?.LastSentSignalRecipients.Add(recipient);
324
325 Connection connection = recipient;
326
327 object[] obj = new object[] { signal, connection };
328 GameMain.LuaCs.Hook.Call("signalReceived", obj);
329 GameMain.LuaCs.Hook.Call("signalReceived." + recipient.item.Prefab.Identifier, obj);
330
331 foreach (ItemComponent ic in recipient.item.Components)
332 {
333 ic.ReceiveSignal(signal, connection);
334 }
335
336 if (recipient.Effects != null && signal.value != "0")
337 {
338 foreach (StatusEffect effect in recipient.Effects)
339 {
340 recipient.Item.ApplyStatusEffect(effect, ActionType.OnUse, (float)Timing.Step);
341 }
342 }
343 }

367 public void ConnectLinked()
368 {
369 if (wireId == null) return;
370
371 for (int i = 0; i < MaxWires; i++)
372 {
373 if (wireId[i] == 0) { continue; }
374
375 if (!(Entity.FindEntityByID(wireId[i]) is Item wireItem)) { continue; }
376 wires[i] = wireItem.GetComponent<Wire>();
377 recipientsDirty = true;
378
379 if (wires[i] != null)
380 {
381 if (wires[i].Item.body != null) wires[i].Item.body.Enabled = false;
382 wires[i].Connect(this, false, false);
383 wires[i].FixNodeEnds();
384 }
385 }

389 public void Save(XElement parentElement)
390 {
391 XElement newElement = new XElement(IsOutput ? "output" : "input", new XAttribute("name", Name));
392
393 Array.Sort(wires, delegate (Wire wire1, Wire wire2)
394 {
395 if (wire1 == null) return 1;
396 if (wire2 == null) return -1;
397 return wire1.Item.ID.CompareTo(wire2.Item.ID);
398 });
399
400 for (int i = 0; i < MaxWires; i++)
401 {
402 if (wires[i] == null) continue;
403
404 newElement.Add(new XElement("link",
405 new XAttribute("w", wires[i].Item.ID.ToString())));
406 }
407
408 parentElement.Add(newElement);
409 }
