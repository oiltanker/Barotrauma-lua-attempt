32 void ReplaceElement(XElement element, XElement replacement)
33 {
34 List<XElement> elementsToRemove = new List<XElement>();
35 foreach (XAttribute attribute in replacement.Attributes())
36 {
37 ReplaceAttribute(element, attribute);
38 }
39 foreach (XElement replacementSubElement in replacement.Elements())
40 {
41 int index = replacement.Elements().ToList().FindAll(e => e.Name.ToString().Equals(replacementSubElement.Name.ToString(), StringComparison.OrdinalIgnoreCase)).IndexOf(replacementSubElement);
42 System.Diagnostics.Debug.Assert(index > -1);
43
44 int i = 0;
45 bool matchingElementFound = false;
46 foreach (var subElement in element.Elements())
47 {
48 if (!subElement.Name.ToString().Equals(replacementSubElement.Name.ToString(), StringComparison.OrdinalIgnoreCase)) { continue; }
49 if (i == index)
50 {
51 if (!replacementSubElement.HasAttributes && !replacementSubElement.HasElements)
52 {
53 //if the replacement is empty (no attributes or child elements)
54 //remove the element from the variant
55 elementsToRemove.Add(subElement);
56 }
57 else
58 {
59 ReplaceElement(subElement, replacementSubElement);
60 }
61 matchingElementFound = true;
62 break;
63 }
64 i++;
65 }
66 if (!matchingElementFound)
67 {
68 element.Add(replacementSubElement);
69 }
70 }
71 elementsToRemove.ForEach(e => e.Remove());
72 }

74 void ReplaceAttribute(XElement element, XAttribute newAttribute)
75 {
76 XAttribute existingAttribute = element.Attributes().FirstOrDefault(a => a.Name.ToString().Equals(newAttribute.Name.ToString(), StringComparison.OrdinalIgnoreCase));
77 if (existingAttribute == null)
78 {
79 element.Add(newAttribute);
80 return;
81 }
82 float.TryParse(existingAttribute.Value, NumberStyles.Any, CultureInfo.InvariantCulture, out float value);
83 if (newAttribute.Value.StartsWith('*'))
84 {
85 string multiplierStr = newAttribute.Value.Substring(1, newAttribute.Value.Length - 1);
86 float.TryParse(multiplierStr, NumberStyles.Any, CultureInfo.InvariantCulture, out float multiplier);
87 if (multiplierStr.Contains('.') || existingAttribute.Value.Contains('.'))
88 {
89 existingAttribute.Value = (value * multiplier).ToString("G", CultureInfo.InvariantCulture);
90 }
91 else
92 {
93 existingAttribute.Value = ((int)(value * multiplier)).ToString();
94 }
95 }
96 else if (newAttribute.Value.StartsWith('+'))
97 {
98 string additionStr = newAttribute.Value.Substring(1, newAttribute.Value.Length - 1);
99 float.TryParse(additionStr, NumberStyles.Any, CultureInfo.InvariantCulture, out float addition);
100 if (additionStr.Contains('.') || existingAttribute.Value.Contains('.'))
101 {
102 existingAttribute.Value = (value + addition).ToString("G", CultureInfo.InvariantCulture);
103 }
104 else
105 {
106 existingAttribute.Value = ((int)(value + addition)).ToString();
107 }
108 }
109 else
110 {
111 existingAttribute.Value = newAttribute.Value;
112 }
113 }
114
115 return newElement;
116 }
