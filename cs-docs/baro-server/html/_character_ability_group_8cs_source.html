90  private AbilityCondition ConstructCondition(CharacterTalent characterTalent, ContentXElement conditionElement, bool errorMessages = true)
91  {
92  AbilityCondition newCondition = null;
93 
94  Type conditionType;
95  string type = conditionElement.Name.ToString().ToLowerInvariant();
96  try
97  {
98  conditionType = Type.GetType("Barotrauma.Abilities." + type + "", false, true);
99  if (conditionType == null)
100  {
101  if (errorMessages) DebugConsole.ThrowError("Could not find the component \"" + type + "\" (" + characterTalent.DebugIdentifier + ")");
102  return null;
103  }
104  }
105  catch (Exception e)
106  {
107  if (errorMessages) DebugConsole.ThrowError("Could not find the component \"" + type + "\" (" + characterTalent.DebugIdentifier + ")", e);
108  return null;
109  }
110 
111  object[] args = { characterTalent, conditionElement };
112 
113  try
114  {
115  newCondition = (AbilityCondition)Activator.CreateInstance(conditionType, args);
116  }
117  catch (TargetInvocationException e)
118  {
119  DebugConsole.ThrowError("Error while creating an instance of an ability condition of the type " + conditionType + ".", e.InnerException);
120  return null;
121  }
122 
123  if (newCondition == null)
124  {
125  DebugConsole.ThrowError("Error while creating an instance of an ability condition of the type " + conditionType + ", instance was null");
126  return null;
127  }
128 
129  return newCondition;
130  }
