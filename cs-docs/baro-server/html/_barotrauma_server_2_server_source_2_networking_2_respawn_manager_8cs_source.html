23 private IEnumerable<Client> GetClientsToRespawn()
24 {
25 MultiPlayerCampaign campaign = GameMain.GameSession.GameMode as MultiPlayerCampaign;
26 foreach (Client c in networkMember.ConnectedClients)
27 {
28 if (GameMain.LuaCs.Game.overrideRespawnSub)
29 continue;
30
31 if (!c.InGame) { continue; }
32 if (c.SpectateOnly && (GameMain.Server.ServerSettings.AllowSpectating || GameMain.Server.OwnerConnection == c.Connection)) { continue; }
33 if (c.Character != null && !c.Character.IsDead) { continue; }
34
35 //don't allow respawn if the client already has a character (they'll regain control once they're in sync)
36 var matchingData = campaign?.GetClientCharacterData(c);
37 if (matchingData != null && matchingData.HasSpawned &&
38 Character.CharacterList.Any(c => c.Info == matchingData.CharacterInfo && !c.IsDead))
39 {
40 continue;
41 }
42
43 if (UseRespawnPrompt)
44 {
45 if (matchingData != null && matchingData.HasSpawned)
46 {
47 if (!c.WaitForNextRoundRespawn.HasValue || c.WaitForNextRoundRespawn.Value) { continue; }
48 }
49 }
50
51 yield return c;
52 }
53 }

55 private bool IsRespawnPromptPendingForClient(Client c)
56 {
57 if (!UseRespawnPrompt || !(GameMain.GameSession.GameMode is MultiPlayerCampaign campaign)) { return false; }
58
59 if (!c.InGame) { return false; }
60 if (c.SpectateOnly && (GameMain.Server.ServerSettings.AllowSpectating || GameMain.Server.OwnerConnection == c.Connection)) { return false; }
61 if (c.Character != null && !c.Character.IsDead) { return false; }
62
63 var matchingData = campaign.GetClientCharacterData(c);
64 if (matchingData != null && matchingData.HasSpawned)
65 {
66 if (Character.CharacterList.Any(c => c.Info == matchingData.CharacterInfo && !c.IsDead))
67 {
68 return false;
69 }
70 else if (!c.WaitForNextRoundRespawn.HasValue)
71 {
72 return true;
73 }
74 }
75 return false;

78 private List<CharacterInfo> GetBotsToRespawn()
79 {
80 if (GameMain.Server.ServerSettings.BotSpawnMode == BotSpawnMode.Normal)
81 {
82 return Character.CharacterList
83 .FindAll(c => c.TeamID == CharacterTeamType.Team1 && c.AIController != null && c.Info != null && c.IsDead)
84 .Select(c => c.Info)
85 .ToList();
86 }
87
88 int currPlayerCount = GameMain.Server.ConnectedClients.Count(c =>
89 c.InGame &&
90 (!c.SpectateOnly || (!GameMain.Server.ServerSettings.AllowSpectating && GameMain.Server.OwnerConnection != c.Connection)));
91
92 var existingBots = Character.CharacterList
93 .FindAll(c => c.TeamID == CharacterTeamType.Team1 && c.AIController != null && c.Info != null);
94
95 int requiredBots = GameMain.Server.ServerSettings.BotCount - currPlayerCount;
96 requiredBots -= existingBots.Count(b => !b.IsDead);
97
98 List<CharacterInfo> botsToRespawn = new List<CharacterInfo>();
99 for (int i = 0; i < requiredBots; i++)
100 {
101 CharacterInfo botToRespawn = existingBots.Find(b => b.IsDead)?.Info;
102 if (botToRespawn == null)
103 {
104 botToRespawn = new CharacterInfo(CharacterPrefab.HumanSpeciesName);
105 }
106 else
107 {
108 existingBots.Remove(botToRespawn.Character);
109 }
110 botsToRespawn.Add(botToRespawn);
111 }
112 return botsToRespawn;

136 {

285 partial void UpdateTransportingProjSpecific(float deltaTime)
286 {
287 if (!ReturnCountdownStarted)
288 {
289 //if there are no living chracters inside, transporting can be stopped immediately
290 if (CheckShuttleEmpty(deltaTime))
291 {
292 ReturnTime = DateTime.Now;
293 ReturnCountdownStarted = true;
294 }
295 else if (!ShouldStartRespawnCountdown())
296 {
297 //don't start counting down until someone else needs to respawn
298 ReturnTime = DateTime.Now + new TimeSpan(0, 0, 0, 0, milliseconds: (int)(maxTransportTime * 1000));
299 despawnTime = ReturnTime + new TimeSpan(0, 0, seconds: 30);
300 return;
301 }
302 else
303 {
304 ReturnCountdownStarted = true;
305 GameMain.Server.CreateEntityEvent(this);
306 }
307 }
308 else if (CheckShuttleEmpty(deltaTime))
309 {
310 ReturnTime = DateTime.Now;
311 }
312
313 if (DateTime.Now > ReturnTime)
314 {
315 GameServer.Log("The respawn shuttle is leaving.", ServerLog.MessageType.ServerMessage);
316 CurrentState = State.Returning;
317
318 GameMain.Server.CreateEntityEvent(this);
319
320 RespawnCountdownStarted = false;
321 maxTransportTime = GameMain.Server.ServerSettings.MaxTransportTime;

541 public static void ReduceCharacterSkills(CharacterInfo characterInfo)
542 {
543 if (characterInfo?.Job == null) { return; }
544 foreach (Skill skill in characterInfo.Job.GetSkills())
545 {
546 var skillPrefab = characterInfo.Job.Prefab.Skills.Find(s => skill.Identifier == s.Identifier);
547 if (skillPrefab == null) { continue; }
548 skill.Level = MathHelper.Lerp(skill.Level, skillPrefab.LevelRange.Start, SkillReductionOnCampaignMidroundRespawn);

552 public void ServerEventWrite(IWriteMessage msg, Client c, NetEntityEvent.IData extraData = null)
553 {
554 msg.WriteRangedInteger((int)CurrentState, 0, Enum.GetNames(typeof(State)).Length);
555
556 switch (CurrentState)
557 {
558 case State.Transporting:
559 msg.Write(ReturnCountdownStarted);
560 msg.Write(GameMain.Server.ServerSettings.MaxTransportTime);
561 msg.Write((float)(ReturnTime - DateTime.Now).TotalSeconds);
562 break;
563 case State.Waiting:
564 MultiPlayerCampaign campaign = GameMain.GameSession.GameMode as MultiPlayerCampaign;
565 var matchingData = campaign?.GetClientCharacterData(c);
566 bool forceSpawnInMainSub = matchingData != null && !matchingData.HasSpawned;
567 msg.Write((ushort)pendingRespawnCount);
568 msg.Write((ushort)requiredRespawnCount);
569 msg.Write(IsRespawnPromptPendingForClient(c));
570 msg.Write(RespawnCountdownStarted);
571 msg.Write(forceSpawnInMainSub);
572 msg.Write((float)(RespawnTime - DateTime.Now).TotalSeconds);
573 break;
574 case State.Returning:
575 break;
576 }
577
578 msg.WritePadBits();
579 }
