51 private void LoadMonsters(XElement monsterElement, MonsterSet set)
52 {
53 Identifier speciesName = monsterElement.GetAttributeIdentifier("character", Identifier.Empty);
54 int defaultCount = monsterElement.GetAttributeInt("count", -1);
55 if (defaultCount < 0)
56 {
57 defaultCount = monsterElement.GetAttributeInt("amount", 1);
58 }
59 int min = Math.Min(monsterElement.GetAttributeInt("min", defaultCount), 255);
60 int max = Math.Min(Math.Max(min, monsterElement.GetAttributeInt("max", defaultCount)), 255);
61 var characterPrefab = CharacterPrefab.FindBySpeciesName(speciesName);
62 if (characterPrefab != null)
63 {
64 set.MonsterPrefabs.Add((characterPrefab, new Point(min, max)));
65 }
66 else
67 {
68 DebugConsole.ThrowError($"Error in beacon mission \"{Prefab.Identifier}\". Could not find a character prefab with the name \"{speciesName}\".");

92 protected override void UpdateMissionSpecific(float deltaTime)
93 {
94 if (IsClient) { return; }
95 if (!swarmSpawned && level.CheckBeaconActive())
96 {
97 List<Submarine> connectedSubs = level.BeaconStation.GetConnectedSubs();
98 foreach (Item item in Item.ItemList)
99 {
100 if (!connectedSubs.Contains(item.Submarine)) { continue; }
101 if (item.GetComponent<PowerTransfer>() != null ||
102 item.GetComponent<PowerContainer>() != null ||
103 item.GetComponent<Reactor>() != null)
104 {
105 item.InvulnerableToDamage = true;
106 }
107 }
108
109 State = 1;
110
111 Vector2 spawnPos = level.BeaconStation.WorldPosition;
112 spawnPos.Y += level.BeaconStation.GetDockedBorders().Height * 1.5f;
113
114 var availablePositions = Level.Loaded.PositionsOfInterest.FindAll(p =>
115 p.PositionType == Level.PositionType.MainPath ||
116 p.PositionType == Level.PositionType.SidePath);
117 availablePositions.RemoveAll(p => Level.Loaded.ExtraWalls.Any(w => w.IsPointInside(p.Position.ToVector2())));
118 availablePositions.RemoveAll(p => Submarine.FindContaining(p.Position.ToVector2()) != null);
119
120 if (availablePositions.Any())
121 {
122 Level.InterestingPosition? closestPos = null;
123 float closestDist = float.PositiveInfinity;
124 foreach (var pos in availablePositions)
125 {
126 float dist = Vector2.DistanceSquared(pos.Position.ToVector2(), level.BeaconStation.WorldPosition);
127 if (dist < closestDist)
128 {
129 closestDist = dist;
130 closestPos = pos;
131 }
132 }
133 if (closestPos.HasValue)
134 {
135 spawnPos = closestPos.Value.Position.ToVector2();
136 }
137 }
138
139 var monsterSet = ToolBox.SelectWeightedRandom(monsterSets, m => m.Commonness, Rand.RandSync.Unsynced);
140 foreach ((CharacterPrefab monsterSpecies, Point monsterCountRange) in monsterSet.MonsterPrefabs)
141 {
142 int amount = Rand.Range(monsterCountRange.X, monsterCountRange.Y + 1);
143 for (int i = 0; i < amount; i++)
144 {
145 CoroutineManager.Invoke(() =>
146 {
147 //round ended before the coroutine finished
148 if (GameMain.GameSession == null || Level.Loaded == null) { return; }
149 Entity.Spawner.AddCharacterToSpawnQueue(monsterSpecies.Identifier, spawnPos);
150 }, Rand.Range(0f, amount));
151 }
152 }
153
154 swarmSpawned = true;
155 }
156#if DEBUG || UNSTABLE
157 if (State == 1 && !level.CheckBeaconActive())
158 {
159 DebugConsole.ThrowError("Beacon became inactive!");
160 State = 2;
161 }
162#endif
163 }
