260 public override void Update(float deltaTime, Camera cam)
261 {
262 if (item.AiTarget != null)
263 {
264 UpdateAITarget(item.AiTarget);
265 }
266 UpdateOnActiveEffects(deltaTime);
267
268#if CLIENT
269 Light.ParentSub = item.Submarine;
270#endif
271 if (item.Container != null)
272 {
273 SetLightSourceState(false, 0.0f);
274 return;
275 }
276
277 SetLightSourceTransformProjSpecific();
278
279 PhysicsBody body = ParentBody ?? item.body;
280 if (body != null && !body.Enabled)
281 {
282 SetLightSourceState(false, 0.0f);
283 return;
284 }
285
286 //currPowerConsumption = powerConsumption;
287 if (Rand.Range(0.0f, 1.0f) < 0.05f && Voltage < Rand.Range(0.0f, MinVoltage))
288 {
289#if CLIENT
290 if (Voltage > 0.1f)
291 {
292 SoundPlayer.PlaySound("zap", item.WorldPosition, hullGuess: item.CurrentHull);
293 }
294#endif
295 lightBrightness = 0.0f;
296 }
297 else
298 {
299 lightBrightness = MathHelper.Lerp(lightBrightness, powerConsumption <= 0.0f ? 1.0f : Math.Min(Voltage, 1.0f), 0.1f);
300 }
301
302 SetLightSourceState(true, lightBrightness);

317 public override void ReceiveSignal(Signal signal, Connection connection)
318 {
319 switch (connection.Name)
320 {
321 case "toggle":
322 if (signal.value != "0")
323 {
324 if (!IgnoreContinuousToggle || lastToggleSignalTime < Timing.TotalTime - 0.1)
325 {
326 IsOn = !IsOn;
327 }
328 lastToggleSignalTime = Timing.TotalTime;
329 }
330 break;
331 case "set_state":
332 IsOn = signal.value != "0";
333 break;
334 case "set_color":
335 if (signal.value != prevColorSignal)
336 {
337 LightColor = XMLExtensions.ParseColor(signal.value, false);
338#if CLIENT
339 SetLightSourceState(Light.Enabled, currentBrightness);
340#endif
341 prevColorSignal = signal.value;
342 }
343 break;
344 }
345 }
