94 public override void Update(float deltaTime, Camera cam)
95 {
96 pumpSpeedLockTimer -= deltaTime;
97 isActiveLockTimer -= deltaTime;
98
99 if (!IsActive) { return; }
100
101 currFlow = 0.0f;
102
103 if (TargetLevel != null)
104 {
105 float hullPercentage = 0.0f;
106 if (item.CurrentHull != null)
107 {
108 float hullWaterVolume = item.CurrentHull.WaterVolume;
109 float totalHullVolume = item.CurrentHull.Volume;
110 foreach (var linked in item.CurrentHull.linkedTo)
111 {
112 if ((linked is Hull linkedHull))
113 {
114 hullWaterVolume += linkedHull.WaterVolume;
115 totalHullVolume += linkedHull.Volume;
116 }
117 }
118 hullPercentage = hullWaterVolume / totalHullVolume * 100.0f;
119 }
120 FlowPercentage = ((float)TargetLevel - hullPercentage) * 10.0f;
121 }
122
123 if (!HasPower) { return; }
124
125 UpdateProjSpecific(deltaTime);
126
127 ApplyStatusEffects(ActionType.OnActive, deltaTime, null);
128
129 if (item.CurrentHull == null) { return; }
130
131 float powerFactor = Math.Min(currPowerConsumption <= 0.0f || MinVoltage <= 0.0f ? 1.0f : Voltage, 1.0f);
132
133 currFlow = flowPercentage / 100.0f * maxFlow * powerFactor;
134
135 if (item.GetComponent<Repairable>() is Repairable repairable && repairable.IsTinkering)
136 {
137 currFlow *= 1f + repairable.TinkeringStrength * TinkeringSpeedIncrease;
138 }
139
140 //less effective when in a bad condition
141 currFlow *= MathHelper.Lerp(0.5f, 1.0f, item.Condition / item.MaxCondition);
142
143 item.CurrentHull.WaterVolume += currFlow * deltaTime * Timing.FixedUpdateRate;
144 if (item.CurrentHull.WaterVolume > item.CurrentHull.Volume) { item.CurrentHull.Pressure += 30.0f * deltaTime; }
145
146 }

148 public void InfectBallast(Identifier identifier, bool allowMultiplePerShip = false)
149 {
150 Hull hull = item.CurrentHull;
151 if (hull == null) { return; }
152
153 if (!allowMultiplePerShip)
154 {
155 // if the ship is already infected then do nothing
156 if (Hull.HullList.Where(h => h.Submarine == hull.Submarine).Any(h => h.BallastFlora != null)) { return; }
157 }
158
159 if (hull.BallastFlora != null) { return; }
160
161 var ballastFloraPrefab = BallastFloraPrefab.Find(identifier);
162 if (ballastFloraPrefab == null)
163 {
164 DebugConsole.ThrowError($"Failed to infect a ballast pump (could not find a ballast flora prefab with the identifier \"{identifier}\").\n" + Environment.StackTrace);
165 return;
166 }
167
168 Vector2 offset = item.WorldPosition - hull.WorldPosition;
169 hull.BallastFlora = new BallastFloraBehavior(hull, ballastFloraPrefab, offset, firstGrowth: true);
170
171#if SERVER
172 hull.BallastFlora.CreateNetworkMessage(new BallastFloraBehavior.SpawnEventData());
173#endif
174 }

196 public override void ReceiveSignal(Signal signal, Connection connection)
197 {
198 if (Hijacked) { return; }
199
200 if (connection.Name == "toggle")
201 {
202 IsActive = !IsActive;
203 isActiveLockTimer = 0.1f;
204 }
205 else if (connection.Name == "set_active")
206 {
207 IsActive = signal.value != "0";
208 isActiveLockTimer = 0.1f;
209 }
210 else if (connection.Name == "set_speed")
211 {
212 if (float.TryParse(signal.value, NumberStyles.Any, CultureInfo.InvariantCulture, out float tempSpeed))
213 {
214 flowPercentage = MathHelper.Clamp(tempSpeed, -100.0f, 100.0f);
215 TargetLevel = null;
216 pumpSpeedLockTimer = 0.1f;
217 }
218 }
219 else if (connection.Name == "set_targetlevel")
220 {
221 if (float.TryParse(signal.value, NumberStyles.Any, CultureInfo.InvariantCulture, out float tempTarget))
222 {
223 TargetLevel = MathUtils.InverseLerp(-100.0f, 100.0f, tempTarget) * 100.0f;
224 pumpSpeedLockTimer = 0.1f;
225 }
226 }

229 public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
230 {
231#if CLIENT
232 if (GameMain.Client != null) { return false; }
233#endif
234
235 switch (objective.Option.Value.ToLowerInvariant())
236 {
237 case "pumpout":
238#if SERVER
239 if (objective.Override || !IsActive || FlowPercentage > -100.0f)
240 {
241 item.CreateServerEvent(this);
242 }
243#endif
244 IsActive = true;
245 FlowPercentage = -100.0f;
246 break;
247 case "pumpin":
248#if SERVER
249 if (objective.Override || !IsActive || FlowPercentage < 100.0f)
250 {
251 item.CreateServerEvent(this);
252 }
253#endif
254 IsActive = true;
255 FlowPercentage = 100.0f;
256 break;
257 case "stoppumping":
258#if SERVER
259 if (objective.Override || FlowPercentage > 0.0f)
260 {
261 item.CreateServerEvent(this);
262 }
263#endif
264 IsActive = false;
265 FlowPercentage = 0.0f;
266 break;
267 }
268 return true;
269 }
