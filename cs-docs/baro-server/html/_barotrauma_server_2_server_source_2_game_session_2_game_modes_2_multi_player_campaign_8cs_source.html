214  foreach (Client c in GameMain.Server.ConnectedClients)
215  {
216  //ignore if the character is controlling a monster

220  c.Character = null;

223  // or the previously saved info if not (e.g. if the client has been spectating or died)
224  var characterInfo = c.Character?.Info;
225  var matchingCharacterData = characterData.Find(d => d.MatchesClient(c));
226  if (matchingCharacterData != null)
227  {
228  //hasn't spawned this round -> don't touch the data
229  if (!matchingCharacterData.HasSpawned) { continue; }
230  characterInfo ??= matchingCharacterData.CharacterInfo;
231  }
232  if (characterInfo == null) { continue; }
233  //reduce skills if the character has died
234  if (characterInfo.CauseOfDeath != null && characterInfo.CauseOfDeath.Type != CauseOfDeathType.Disconnected)
235  {
236  RespawnManager.ReduceCharacterSkills(characterInfo);
237  characterInfo.RemoveSavedStatValuesOnDeath();
238  characterInfo.CauseOfDeath = null;
239  }
240  c.CharacterInfo = characterInfo;
241  SetClientCharacterData(c);
242  }
243 
244  //refresh the character data of clients who aren't in the server anymore
245  List<CharacterCampaignData> prevCharacterData = new List<CharacterCampaignData>(characterData);
246  foreach (CharacterCampaignData data in prevCharacterData)
247  {
248  if (data.HasSpawned && !GameMain.Server.ConnectedClients.Any(c => data.MatchesClient(c)))
249  {
250  var character = Character.CharacterList.Find(c => c.Info == data.CharacterInfo && !c.IsHusk);
251  if (character != null && (!character.IsDead || character.CauseOfDeath?.Type == CauseOfDeathType.Disconnected))

305  protected override IEnumerable<CoroutineStatus> DoLevelTransition(TransitionType transitionType, LevelData newLevel, Submarine leavingSub, bool mirror, List<TraitorMissionResult> traitorResults)
306  {
307  lastUpdateID++;
308 
309  switch (transitionType)
310  {
311  case TransitionType.None:
312  throw new InvalidOperationException("Level transition failed (no transitions available).");
313  case TransitionType.ReturnToPreviousLocation:
314  //deselect destination on map
315  map.SelectLocation(-1);
316  break;
317  case TransitionType.ProgressToNextLocation:
318  Map.MoveToNextLocation();
319  TotalPassedLevels++;
320  break;
321  case TransitionType.End:
322  EndCampaign();
323  IsFirstRound = true;
324  break;
325  case TransitionType.ProgressToNextEmptyLocation:
326  TotalPassedLevels++;
327  break;
328  }
329 
330  Map.ProgressWorld(transitionType, (float)(Timing.TotalTime - GameMain.GameSession.RoundStartTime));
331 
332  bool success = GameMain.Server.ConnectedClients.Any(c => c.InGame && c.Character != null && !c.Character.IsDead);
333  if (success)
334  {
335  foreach (Client c in GameMain.Server.ConnectedClients)
336  {
337  if (c.Character?.HasAbilityFlag(AbilityFlags.RetainExperienceForNewCharacter) ?? false)
338  {
339  (GameMain.GameSession?.GameMode as MultiPlayerCampaign)?.SaveExperiencePoints(c);
340  }
341  }
342  }
343 
344  GameMain.GameSession.EndRound("", traitorResults, transitionType);
345 
346  //--------------------------------------
347 
348  if (success)
349  {
350  SavePlayers();
351 
352  yield return CoroutineStatus.Running;
353 
354  if (leavingSub != Submarine.MainSub && !leavingSub.DockedTo.Contains(Submarine.MainSub))
355  {
356  Submarine.MainSub = leavingSub;
357  GameMain.GameSession.Submarine = leavingSub;
358  GameMain.GameSession.SubmarineInfo = leavingSub.Info;
359  leavingSub.Info.FilePath = System.IO.Path.Combine(SaveUtil.TempPath, leavingSub.Info.Name + ".sub");
360  var subsToLeaveBehind = GetSubsToLeaveBehind(leavingSub);
361  GameMain.GameSession.OwnedSubmarines.Add(leavingSub.Info);
362  foreach (Submarine sub in subsToLeaveBehind)
363  {
364  GameMain.GameSession.OwnedSubmarines.RemoveAll(s => s != leavingSub.Info && s.Name == sub.Info.Name);
365  MapEntity.mapEntityList.RemoveAll(e => e.Submarine == sub && e is LinkedSubmarine);
366  LinkedSubmarine.CreateDummy(leavingSub, sub);
367  }
368  }
369  NextLevel = newLevel;
370  GameMain.GameSession.SubmarineInfo = new SubmarineInfo(GameMain.GameSession.Submarine);
371 
372  if (PendingSubmarineSwitch != null)
373  {
374  SubmarineInfo previousSub = GameMain.GameSession.SubmarineInfo;
375  GameMain.GameSession.SubmarineInfo = PendingSubmarineSwitch;
376 
377  for (int i = 0; i < GameMain.GameSession.OwnedSubmarines.Count; i++)
378  {
379  if (GameMain.GameSession.OwnedSubmarines[i].Name == previousSub.Name)
380  {
381  GameMain.GameSession.OwnedSubmarines[i] = previousSub;
382  break;
383  }
384  }
385  }
386 
387  SaveUtil.SaveGame(GameMain.GameSession.SavePath);
388  PendingSubmarineSwitch = null;
389  }
390  else
391  {
392  PendingSubmarineSwitch = null;
393  GameMain.Server.EndGame(TransitionType.None, wasSaved: false);
394  LoadCampaign(GameMain.GameSession.SavePath);
395  LastSaveID++;
396  LastUpdateID++;
397  yield return CoroutineStatus.Success;
398  }
399 
400  CrewManager?.ClearCurrentOrders();
401 
402  //--------------------------------------
403 
404  GameMain.Server.EndGame(transitionType, wasSaved: true);
405 
406  ForceMapUI = false;
407 
408  NextLevel = newLevel;
409  MirrorLevel = mirror;
410 
411  //give clients time to play the end cinematic before starting the next round
412  if (transitionType == TransitionType.End)
413  {
414  yield return new WaitForSeconds(EndCinematicDuration);
415  }
416  else
417  {
418  yield return new WaitForSeconds(EndTransitionDuration * 0.5f);
419  }
420 
421  GameMain.Server.StartGame();
422 
423  yield return CoroutineStatus.Success;
424  }
