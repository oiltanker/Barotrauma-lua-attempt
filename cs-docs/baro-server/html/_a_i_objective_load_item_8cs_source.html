66  private ImmutableHashSet<Identifier> GetValidContainableItemIdentifiers()
67  {
68  if (AllValidContainableItemIdentifiers.TryGetValue(Container.Prefab, out var existingIdentifiers))
69  {
70  return existingIdentifiers;
71  }
72  // Status effects are often used to alter item condition so using the Containable Item Identifiers directly can lead to unwanted results
73  // For example, placing welding fuel tanks inside oxygen tank shelves
74  bool useDefaultContainableItemIdentifiers = true;
75  var potentialContainablePrefabs = MapEntityPrefab.List
76  .Where(mep => mep is ItemPrefab ip && ItemContainer.ContainableItemIdentifiers.Any(i => i == ip.Identifier || ip.Tags.Contains(i)))
77  .Cast<ItemPrefab>();
78  var validContainableItemIdentifiers = new HashSet<Identifier>();
79  foreach (var component in Container.Components)
80  {
81  if (CheckComponent() == CheckStatus.Finished)
82  {
83  break;
84  }
85  CheckStatus CheckComponent()
86  {
87  if (component.statusEffectLists != null)
88  {
89  foreach (var (_, statusEffects) in component.statusEffectLists)
90  {
91  if (CheckStatusEffects(statusEffects) == CheckStatus.Finished)
92  {
93  return CheckStatus.Finished;
94  }
95  }
96  }
97  if (component is ItemContainer itemContainer && itemContainer.ContainableItems != null)
98  {
99  foreach (var item in itemContainer.ContainableItems)
100  {
101  if (CheckStatusEffects(item.statusEffects) == CheckStatus.Finished)
102  {
103  return CheckStatus.Finished;
104  }
105  }
106  }
107  return CheckStatus.Unfinished;
108  CheckStatus CheckStatusEffects(IEnumerable<StatusEffect> statusEffects)
109  {
110  if (statusEffects == null) { return CheckStatus.Unfinished; }
111  foreach (var statusEffect in statusEffects)
112  {
113  if ((statusEffect.TargetIdentifiers == null || statusEffect.TargetIdentifiers.None()) && !statusEffect.HasConditions) { continue; }
114  switch (TargetItemCondition)
115  {
116  case AIObjectiveLoadItems.ItemCondition.Empty:
117  if (!statusEffect.ReducesItemCondition()) { continue; }
118  break;
119  case AIObjectiveLoadItems.ItemCondition.Full:
120  if (!statusEffect.IncreasesItemCondition()) { continue; }
121  break;
122  default:
123  continue;
124  }
125  useDefaultContainableItemIdentifiers = false;
126  if (statusEffect.TargetIdentifiers != null)
127  {
128  foreach (Identifier target in statusEffect.TargetIdentifiers)
129  {
130  foreach (var prefab in potentialContainablePrefabs)
131  {
132  if (CheckPrefab(prefab, () => prefab.Tags.Contains(target)) == CheckStatus.Finished) { return CheckStatus.Finished; }
133  }
134  }
135  }
136  foreach (var prefab in potentialContainablePrefabs)
137  {
138  if (CheckPrefab(prefab, () => statusEffect.MatchesTagConditionals(prefab)) == CheckStatus.Finished) { return CheckStatus.Finished; }
139  }
140  CheckStatus CheckPrefab(ItemPrefab prefab, Func<bool> isValid)
141  {
142  if (validContainableItemIdentifiers.Contains(prefab.Identifier)) { return CheckStatus.Unfinished; }
143  if (!isValid()) { return CheckStatus.Unfinished; }
144  validContainableItemIdentifiers.Add(prefab.Identifier);
145  if (potentialContainablePrefabs.Any(p => !validContainableItemIdentifiers.Contains(p.Identifier))) { return CheckStatus.Unfinished; }
146  return CheckStatus.Finished;
147  }
148  }
149  return CheckStatus.Unfinished;
150  }
151  }
152  }
153  var identifiers = useDefaultContainableItemIdentifiers ?
154  potentialContainablePrefabs.Select(p => p.Identifier).ToImmutableHashSet() :
155  validContainableItemIdentifiers.ToImmutableHashSet();
156  AllValidContainableItemIdentifiers.Add(Container.Prefab, identifiers);
157  return identifiers;
158  }

160  protected override float GetPriority()
161  {
162  if (!IsAllowed)
163  {
164  Priority = 0;
165  Abandon = true;
166  return Priority;
167  }
168  else if (!AIObjectiveLoadItems.IsValidTarget(Container, character, targetCondition: TargetItemCondition))
169  {
170  // Reduce priority to 0 if the this isn't a valid container right now
171  Priority = 0;
172  }
173  else if (targetItem == null)
174  {
175  Priority = 0;
176  }
177  else
178  {
179  float dist = 0.0f;
180  if (character.CurrentHull != targetItem.CurrentHull)
181  {
182  AddDistance(character.WorldPosition, targetItem.WorldPosition);
183  }
184  if (targetItem.CurrentHull != Container.CurrentHull)
185  {
186  AddDistance(targetItem.WorldPosition, Container.WorldPosition);
187  }
188  void AddDistance(Vector2 startPos, Vector2 targetPos)
189  {
190  float yDist = Math.Abs(startPos.Y - targetPos.Y);
191  // If we're on the same level with the target, we'll disregard the vertical distance
192  if (yDist > 100) { dist += yDist * 5; }
193  dist += Math.Abs(character.WorldPosition.X - targetPos.X);
194  }
195  float distanceFactor = dist > 0.0f ? MathHelper.Lerp(0.9f, 0, MathUtils.InverseLerp(0, 5000, dist)) : 0.9f;
196  bool hasContainable = character.HasItem(targetItem);
197  float devotion = (CumulatedDevotion + (hasContainable ? 100 - MaxDevotion : 0)) / 100;
198  float max = AIObjectiveManager.LowestOrderPriority - (hasContainable ? 1 : 2);
199  Priority = MathHelper.Lerp(0, max, MathHelper.Clamp(devotion + (distanceFactor * PriorityModifier), 0, 1));
200  if (decontainObjective != null && targetItem.Container != Container)
201  {
202  if (!IsValidContainable(targetItem))
203  {
204  // Target is not valid anymore, abandon the objective
205  decontainObjective.Abandon = true;
206  }
207  else if (!ItemContainer.Inventory.CanBePut(targetItem) && ItemContainer.Inventory.AllItems.None(i => AIObjectiveLoadItems.ItemMatchesTargetCondition(i, TargetItemCondition)))
208  {
209  // The container is full and there's no item that should be removed, abandon the objective
210  decontainObjective.Abandon = true;
211  }
212  }
213  if (ItemContainer.Inventory.IsFull())
214  {
215  // Prioritize containers that still have empty space by lowering the priority of objectives with a full target container
216  Priority /= 4;
217  }
218  }
219  return Priority;
220  }

222  public override void Update(float deltaTime)
223  {
224  base.Update(deltaTime);
225  if (targetItem == null)
226  {
227  if (character.FindItem(ref itemIndex, out Item item, identifiers: ValidContainableItemIdentifiers, ignoreBroken: false, customPredicate: IsValidContainable, customPriorityFunction: GetPriority))
228  {
229  if (item == null)
230  {
231  // No possible containables found, abandon the objective
232  Abandon = true;
233  }
234  targetItem = item;
235  }
236  float GetPriority(Item item)
237  {
238  try
239  {
240  // Prefer items closer to full condition when target condition is Empty, and vice versa
241  float conditionBasedPriority = TargetItemCondition switch
242  {
243  AIObjectiveLoadItems.ItemCondition.Full => MathUtils.InverseLerp(100.0f, 0.0f, item.ConditionPercentage),
244  AIObjectiveLoadItems.ItemCondition.Empty => MathUtils.InverseLerp(0.0f, 100.0f, item.ConditionPercentage),
245  _ => throw new NotImplementedException()
246  };
247  // Prefer items that have the same identifier as one of the already contained items
248  return ItemContainer.ContainsItemsWithSameIdentifier(item) ? conditionBasedPriority : conditionBasedPriority / 2;
249  }
250  catch (NotImplementedException)
251  {
252 #if DEBUG
253  DebugConsole.ShowError($"Unexpected target condition \"{TargetItemCondition}\" in local function GetConditionBasedProperty");
254 #endif
255  return 0.0f;
256  }
257  }

299  private bool IsValidContainable(Item item)
300  {
301  if (item == null) { return false; }
302  if (item.Removed) { return false; }
303  if (!ValidContainableItemIdentifiers.Contains(item.Prefab.Identifier)) { return false; }
304  if (ignoredItems.Contains(item)) { return false; }
305  if ((item.SpawnedInCurrentOutpost && !item.AllowStealing) == character.IsOnPlayerTeam) { return false; }
306  var rootInventoryOwner = item.GetRootInventoryOwner();
307  if (rootInventoryOwner is Character owner && owner != character) { return false; }
308  if (rootInventoryOwner is Item parentItem)
309  {
310  if (parentItem.HasTag("donttakeitems")) { return false; }
311  }
312  if (!item.HasAccess(character)) { return false; }
313  if (!character.HasItem(item) && !CanEquip(item)) { return false; }
314  if (!ItemContainer.CanBeContained(item)) { return false; }
315  if (AIObjectiveLoadItems.ItemMatchesTargetCondition(item, TargetItemCondition)) { return false; }
316  if (TargetItemCondition == AIObjectiveLoadItems.ItemCondition.Full)
317  {
318  // Ignore items that have had their condition increase recently
319  if (TargetItemCondition == AIObjectiveLoadItems.ItemCondition.Full && item.ConditionIncreasedRecently) { return false; }
320  // Ignore items inside their (condition-restricted) primary containers
321  if (item.ParentInventory is ItemInventory itemInventory && item.IsContainerPreferred(itemInventory.Container, out bool _, out bool isSecondary, requireConditionRestriction: true) && !isSecondary) { return false; }
322  }
323  // Ignore items inside another valid container
324  if (AIObjectiveLoadItems.IsValidTarget(item.Container, character, TargetContainerTags)) { return false; }
325  return true;
326  }
