22 public float GetPositionUpdateInterval(Client recipient)
23 {
24 if (!Enabled) { return 1000.0f; }
25
26 Vector2 comparePosition = recipient.SpectatePos == null ? recipient.Character.WorldPosition : recipient.SpectatePos.Value;
27
28 float distance = Vector2.Distance(comparePosition, WorldPosition);
29 if (recipient.Character?.ViewTarget != null)
30 {
31 distance = Math.Min(distance, Vector2.Distance(recipient.Character.ViewTarget.WorldPosition, WorldPosition));
32 }
33
34 float priority = 1.0f - MathUtils.InverseLerp(
35 NetConfig.HighPrioCharacterPositionUpdateDistance,
36 NetConfig.LowPrioCharacterPositionUpdateDistance,
37 distance);
38
39 float interval = MathHelper.Lerp(
40 NetConfig.LowPrioCharacterPositionUpdateInterval,
41 NetConfig.HighPrioCharacterPositionUpdateInterval,
42 priority);
43
44 if (IsDead)
45 {
46 interval = Math.Max(interval * 2, 0.1f);
47 }
48
49 return interval;
50 }

52 partial void UpdateNetInput()
53 {
54 if (!(this is AICharacter) || IsRemotePlayer)
55 {
56 if (!CanMove)
57 {
58 AnimController.Frozen = false;
59 if (memInput.Count > 0)
60 {
61 prevDequeuedInput = dequeuedInput;
62 dequeuedInput = memInput[memInput.Count - 1].states & InputNetFlags.Ragdoll;
63 memInput.RemoveAt(memInput.Count - 1);
64 }
65 }
66 else if (memInput.Count == 0)
67 {
68 AnimController.Frozen = true;
69 if (Timing.TotalTime > LastInputTime + 0.5)
70 {
71 //no inputs have been received in 0.5 seconds, reset input
72 //(if there's a temporary network hiccup that prevents us from receiving inputs, we assume the inputs haven't changed,
73 //but if it takes too long, for example due to a client crashing/disconnecting, we don't want to keep the character
74 //firing a welding tool or whatever else they were doing until the kill disconnect timer kicks in)
75 prevDequeuedInput = dequeuedInput =
76 dequeuedInput.HasFlag(InputNetFlags.FacingLeft) ? InputNetFlags.FacingLeft : InputNetFlags.None;
77 }
78 }
79 else
80 {
81 AnimController.Frozen = false;
82 prevDequeuedInput = dequeuedInput;
83
84 LastProcessedID = memInput[memInput.Count - 1].networkUpdateID;
85 dequeuedInput = memInput[memInput.Count - 1].states;
86
87 double aimAngle = ((double)memInput[memInput.Count - 1].intAim / 65535.0) * 2.0 * Math.PI;
88 cursorPosition = AimRefPosition + new Vector2((float)Math.Cos(aimAngle), (float)Math.Sin(aimAngle)) * 500.0f;
89
90 //reset focus when attempting to use/select something
91 if (memInput[memInput.Count - 1].states.HasFlag(InputNetFlags.Use) ||
92 memInput[memInput.Count - 1].states.HasFlag(InputNetFlags.Select) ||
93 memInput[memInput.Count - 1].states.HasFlag(InputNetFlags.Deselect) ||
94 memInput[memInput.Count - 1].states.HasFlag(InputNetFlags.Health) ||
95 memInput[memInput.Count - 1].states.HasFlag(InputNetFlags.Grab))
96 {
97 focusedItem = null;
98 FocusedCharacter = null;
99 }
100 var closestEntity = FindEntityByID(memInput[memInput.Count - 1].interact);
101 if (closestEntity is Item item)
102 {
103 if (CanInteractWith(item))
104 {
105 focusedItem = item;
106 FocusedCharacter = null;
107 }
108 else
109 {
110 //failed to interact with the item
111 // -> correct the position and the state of the Holdable component (in case the item was deattached client-side)
112 item.PositionUpdateInterval = 0.0f;
113 var holdable = item.GetComponent<Items.Components.Holdable>();
114 holdable?.Item?.CreateServerEvent(holdable);
115 }
116 }
117 else if (closestEntity is Character character)
118 {
119 if (CanInteractWith(character, maxDist: 250.0f))
120 {
121 FocusedCharacter = character;
122 focusedItem = null;
123 }
124 }
125
126 memInput.RemoveAt(memInput.Count - 1);
127
128 TransformCursorPos();
129
130 if ((dequeuedInput == InputNetFlags.None || dequeuedInput == InputNetFlags.FacingLeft) && Math.Abs(AnimController.Collider.LinearVelocity.X) < 0.005f && Math.Abs(AnimController.Collider.LinearVelocity.Y) < 0.2f)
131 {
132 while (memInput.Count > 5 && memInput[memInput.Count - 1].states == dequeuedInput)
133 {
134 //remove inputs where the player is not moving at all
135 //helps the server catch up, shouldn't affect final position
136 LastProcessedID = memInput[memInput.Count - 1].networkUpdateID;
137 memInput.RemoveAt(memInput.Count - 1);
138 }
139 }
140 }
141 }
142 AnimController.Frozen = false;
143
144 if (networkUpdateSent)
145 {
146 foreach (Key key in keys)
147 {
148 key.DequeueHit();
149 key.DequeueHeld();
150 }
151
152 networkUpdateSent = false;
153 }

156 public void ServerReadInput(IReadMessage msg, Client c)
157 {
158 if (c.Character != this)
159 {
160#if DEBUG
161 DebugConsole.Log("Received a character update message from a client who's not controlling the character");
162#endif
163 return;
164 }
165
166 UInt16 networkUpdateID = msg.ReadUInt16();
167 byte inputCount = msg.ReadByte();
168
169 if (AllowInput) { Enabled = true; }
170
171 for (int i = 0; i < inputCount; i++)
172 {
173 InputNetFlags newInput = (InputNetFlags)msg.ReadRangedInteger(0, (int)InputNetFlags.MaxVal);
174 UInt16 newAim = 0;
175 UInt16 newInteract = 0;
176
177 if (newInput != InputNetFlags.None && newInput != InputNetFlags.FacingLeft)
178 {
179 c.KickAFKTimer = 0.0f;
180 }
181 else if (AnimController.Dir < 0.0f != newInput.HasFlag(InputNetFlags.FacingLeft))
182 {
183 //character changed the direction they're facing
184 c.KickAFKTimer = 0.0f;
185 }
186
187 newAim = msg.ReadUInt16();
188 if (newInput.HasFlag(InputNetFlags.Select) ||
189 newInput.HasFlag(InputNetFlags.Deselect) ||
190 newInput.HasFlag(InputNetFlags.Use) ||
191 newInput.HasFlag(InputNetFlags.Health) ||
192 newInput.HasFlag(InputNetFlags.Grab))
193 {
194 newInteract = msg.ReadUInt16();
195 }
196
197 if (NetIdUtils.IdMoreRecent((ushort)(networkUpdateID - i), LastNetworkUpdateID) && (i < 60))
198 {
199 if ((i > 0 && memInput[i - 1].intAim != newAim))
200 {
201 c.KickAFKTimer = 0.0f;
202 }
203 NetInputMem newMem = new NetInputMem
204 {
205 states = newInput,
206 intAim = newAim,
207 interact = newInteract,
208 networkUpdateID = (ushort)(networkUpdateID - i)
209 };
210 memInput.Insert(i, newMem);
211 LastInputTime = Timing.TotalTime;
212 }
213 }
214
215 if (NetIdUtils.IdMoreRecent(networkUpdateID, LastNetworkUpdateID))
216 {
217 LastNetworkUpdateID = networkUpdateID;
218 }
219 else if (NetIdUtils.Difference(networkUpdateID, LastNetworkUpdateID) > 500)
220 {
221#if DEBUG || UNSTABLE
222 DebugConsole.AddWarning($"Large disrepancy between a client character's network update ID server-side and client-side (client: {networkUpdateID}, server: {LastNetworkUpdateID}). Resetting the ID.");
223#endif
224 LastNetworkUpdateID = networkUpdateID;
225 }
226 if (memInput.Count > 60)
227 {
228 //deleting inputs from the queue here means the server is way behind and data needs to be dropped
229 //we'll make the server drop down to 30 inputs for good measure
230 memInput.RemoveRange(30, memInput.Count - 30);
231 }
232 }

234 public virtual void ServerEventRead(IReadMessage msg, Client c)
235 {
236 EventType eventType = (EventType)msg.ReadRangedInteger((int)EventType.MinValue, (int)EventType.MaxValue);
237 switch (eventType)
238 {
239 case EventType.InventoryState:
240 Inventory.ServerEventRead(msg, c);
241 break;
242 case EventType.Treatment:
243 bool doingCPR = msg.ReadBoolean();
244 if (c.Character != this)
245 {
246#if DEBUG
247 DebugConsole.Log("Received a character update message from a client who's not controlling the character");
248#endif
249 return;
250 }
251
252 AnimController.Anim = doingCPR ? AnimController.Animation.CPR : AnimController.Animation.None;
253 break;
254 case EventType.Status:
255 if (c.Character != this)
256 {
257#if DEBUG
258 DebugConsole.Log("Received a character update message from a client who's not controlling the character");
259#endif
260 return;
261 }
262
263 if (IsIncapacitated)
264 {
265 var causeOfDeath = CharacterHealth.GetCauseOfDeath();
266 Kill(causeOfDeath.type, causeOfDeath.affliction);
267 }
268 break;
269 case EventType.UpdateTalents:
270 if (c.Character != this)
271 {
272#if DEBUG
273 DebugConsole.Log("Received a character update message from a client who's not controlling the character");
274#endif
275 return;
276 }
277
278 // get the full list of talents from the player, only give the ones
279 // that are not already given (or otherwise not viable)
280 ushort talentCount = msg.ReadUInt16();
281 List<Identifier> talentSelection = new List<Identifier>();
282 for (int i = 0; i < talentCount; i++)
283 {
284 UInt32 talentIdentifier = msg.ReadUInt32();
285 var prefab = TalentPrefab.TalentPrefabs.Find(p => p.UintIdentifier == talentIdentifier);
286 if (prefab == null) { continue; }
287
288 if (TalentTree.IsViableTalentForCharacter(this, prefab.Identifier, talentSelection))
289 {
290 GiveTalent(prefab.Identifier);
291 talentSelection.Add(prefab.Identifier);
292 }
293 }
294 if (talentSelection.Count != talentCount)
295 {
296 DebugConsole.AddWarning($"Failed to unlock talents: the amount of unlocked talents doesn't match (client: {talentCount}, server: {talentSelection.Count})");
297 }
298 break;
299 }

303 {
304 msg.Write(ID);
305
306 IWriteMessage tempBuffer = new WriteOnlyMessage();
307
308 if (this == c.Character)
309 {
310 tempBuffer.Write(true);
311 if (LastNetworkUpdateID < memInput.Count + 1)
312 {
313 tempBuffer.Write((UInt16)0);
314 }
315 else
316 {
317 tempBuffer.Write((UInt16)(LastNetworkUpdateID - memInput.Count - 1));
318 }
319 }
320 else
321 {
322 tempBuffer.Write(false);
323
324 bool aiming = false;
325 bool use = false;
326 bool attack = false;
327 bool shoot = false;
328
329 if (IsRemotePlayer)
330 {
331 aiming = dequeuedInput.HasFlag(InputNetFlags.Aim);
332 use = dequeuedInput.HasFlag(InputNetFlags.Use);
333 attack = dequeuedInput.HasFlag(InputNetFlags.Attack);
334 shoot = dequeuedInput.HasFlag(InputNetFlags.Shoot);
335 }
336 else if (keys != null)
337 {
338 aiming = keys[(int)InputType.Aim].GetHeldQueue;
339 use = keys[(int)InputType.Use].GetHeldQueue;
340 attack = keys[(int)InputType.Attack].GetHeldQueue;
341 shoot = keys[(int)InputType.Shoot].GetHeldQueue;
342 networkUpdateSent = true;
343 }

345 tempBuffer.Write(aiming);

347 tempBuffer.Write(use);

350 tempBuffer.Write(((HumanoidAnimController)AnimController).Crouching);
351 }
352 tempBuffer.Write(attack);
353
354 Vector2 relativeCursorPos = cursorPosition - AimRefPosition;
355 tempBuffer.Write((UInt16)(65535.0 * Math.Atan2(relativeCursorPos.Y, relativeCursorPos.X) / (2.0 * Math.PI)));
356
357 tempBuffer.Write(IsRagdolled || Stun > 0.0f || IsDead || IsIncapacitated);
358
359 tempBuffer.Write(AnimController.Dir > 0.0f);

362 if (SelectedCharacter != null || SelectedConstruction != null)
363 {
364 tempBuffer.Write(true);
365 tempBuffer.Write(SelectedCharacter != null ? SelectedCharacter.ID : NullEntityID);
366 tempBuffer.Write(SelectedConstruction != null ? SelectedConstruction.ID : NullEntityID);
367 if (SelectedCharacter != null)
368 {
369 tempBuffer.Write(AnimController.Anim == AnimController.Animation.CPR);
370 }
371 }
372 else
373 {
374 tempBuffer.Write(false);
375 }
376
377 tempBuffer.Write(SimPosition.X);
378 tempBuffer.Write(SimPosition.Y);
379 float MaxVel = NetConfig.MaxPhysicsBodyVelocity;
380 AnimController.Collider.LinearVelocity = new Vector2(
381 MathHelper.Clamp(AnimController.Collider.LinearVelocity.X, -MaxVel, MaxVel),
382 MathHelper.Clamp(AnimController.Collider.LinearVelocity.Y, -MaxVel, MaxVel));
383 tempBuffer.WriteRangedSingle(AnimController.Collider.LinearVelocity.X, -MaxVel, MaxVel, 12);
384 tempBuffer.WriteRangedSingle(AnimController.Collider.LinearVelocity.Y, -MaxVel, MaxVel, 12);
385
386 bool fixedRotation = AnimController.Collider.FarseerBody.FixedRotation || !AnimController.Collider.PhysEnabled;
387 tempBuffer.Write(fixedRotation);
388 if (!fixedRotation)
389 {
390 tempBuffer.Write(AnimController.Collider.Rotation);
391 float MaxAngularVel = NetConfig.MaxPhysicsBodyAngularVelocity;
392 AnimController.Collider.AngularVelocity = NetConfig.Quantize(AnimController.Collider.AngularVelocity, -MaxAngularVel, MaxAngularVel, 8);
393 tempBuffer.WriteRangedSingle(MathHelper.Clamp(AnimController.Collider.AngularVelocity, -MaxAngularVel, MaxAngularVel), -MaxAngularVel, MaxAngularVel, 8);
394 }
395
396 bool writeStatus = healthUpdateTimer <= 0.0f;
397 tempBuffer.Write(writeStatus);
398 if (writeStatus)
399 {
400 WriteStatus(tempBuffer);
401 AIController?.ServerWrite(tempBuffer);
402 HealthUpdatePending = false;
403 }
404
405 tempBuffer.WritePadBits();
406
407 msg.WriteVariableUInt32((uint)tempBuffer.LengthBytes);
408 msg.Write(tempBuffer.Buffer, 0, tempBuffer.LengthBytes);
409 }

411 public virtual void ServerEventWrite(IWriteMessage msg, Client c, NetEntityEvent.IData extraData = null)
412 {
413 if (!(extraData is IEventData eventData)) { throw new Exception($"Malformed character event: expected {nameof(Character)}.{nameof(IEventData)}, got {extraData?.GetType().Name ?? "[NULL]"}"); }
414
415 msg.WriteRangedInteger((int)eventData.EventType, (int)EventType.MinValue, (int)EventType.MaxValue);
416 switch (eventData)
417 {
418 case InventoryStateEventData _:
419 msg.Write(GameMain.Server.EntityEventManager.Events.Last()?.ID ?? (ushort)0);
420 Inventory.ServerEventWrite(msg, c);
421 break;
422 case ControlEventData controlEventData:
423 Client owner = controlEventData.Owner;
424 msg.Write(owner == c && owner.Character == this);
425 msg.Write(owner != null && owner.Character == this && GameMain.Server.ConnectedClients.Contains(owner) ? owner.ID : (byte)0);
426 break;
427 case CharacterStatusEventData _:
428 WriteStatus(msg);
429 break;
430 case UpdateSkillsEventData _:
431 if (Info?.Job == null)
432 {
433 msg.Write((byte)0);
434 }
435 else
436 {
437 var skills = Info.Job.GetSkills();
438 msg.Write((byte)skills.Count());
439 foreach (Skill skill in skills)
440 {
441 msg.Write(skill.Identifier);
442 msg.Write(skill.Level);
443 }
444 }
445 break;
446 case IAttackEventData attackEventData:
447 {
448 int attackLimbIndex = Removed ? -1 : Array.IndexOf(AnimController.Limbs, attackEventData.AttackLimb);
449 ushort targetEntityId = 0;
450 int targetLimbIndex = -1;
451 if (attackEventData.TargetEntity is Entity { Removed: false } targetEntity)
452 {
453 targetEntityId = targetEntity.ID;
454 if (targetEntity is Character { AnimController: { Limbs: var targetLimbsArray } })
455 {
456 targetLimbIndex = targetLimbsArray.IndexOf(attackEventData.TargetLimb);
457 }
458 }
459 msg.Write((byte)(attackLimbIndex < 0 ? 255 : attackLimbIndex));
460 msg.Write((ushort)targetEntityId);
461 msg.Write((byte)(targetLimbIndex < 0 ? 255 : targetLimbIndex));
462 msg.Write(attackEventData.TargetSimPos.X);
463 msg.Write(attackEventData.TargetSimPos.Y);
464 }
465 break;
466 case AssignCampaignInteractionEventData _:
467 msg.Write((byte)CampaignInteractionType);
468 msg.Write(RequireConsciousnessForCustomInteract);
469 break;
470 case ObjectiveManagerStateEventData objectiveManagerStateEventData:
471 AIObjectiveManager.ObjectiveType type = objectiveManagerStateEventData.ObjectiveType;
472 msg.WriteRangedInteger((int)type, (int)AIObjectiveManager.ObjectiveType.MinValue, (int)AIObjectiveManager.ObjectiveType.MaxValue);
473 if (!(AIController is HumanAIController controller))
474 {
475 msg.Write(false);
476 break;
477 }
478 if (type == AIObjectiveManager.ObjectiveType.Order)
479 {
480 var currentOrderInfo = controller.ObjectiveManager.GetCurrentOrderInfo();
481 bool validOrder = currentOrderInfo != null;
482 msg.Write(validOrder);
483 if (!validOrder) { break; }
484 var orderPrefab = currentOrderInfo.Prefab;
485 msg.Write(orderPrefab.UintIdentifier);
486 if (!orderPrefab.HasOptions) { break; }
487 int optionIndex = orderPrefab.AllOptions.IndexOf(currentOrderInfo.Option);
488 if (optionIndex == -1)
489 {
490 DebugConsole.AddWarning($"Error while writing order data. Order option \"{currentOrderInfo.Option}\" not found in the order prefab \"{orderPrefab.Name}\".");
491 }
492 msg.WriteRangedInteger(optionIndex, -1, orderPrefab.AllOptions.Length);
493 }
494 else if (type == AIObjectiveManager.ObjectiveType.Objective)
495 {
496 var objective = controller.ObjectiveManager.CurrentObjective;
497 bool validObjective = objective?.Identifier is { IsEmpty: false };
498 msg.Write(validObjective);
499 if (!validObjective) { break; }
500 msg.Write(objective.Identifier);
501 msg.Write(objective.Option);
502 UInt16 targetEntityId = 0;
503 if (objective is AIObjectiveOperateItem operateObjective && operateObjective.OperateTarget != null)
504 {
505 targetEntityId = operateObjective.OperateTarget.ID;
506 }
507 msg.Write(targetEntityId);
508 }
509 break;
510 case TeamChangeEventData _:
511 msg.Write((byte)TeamID);
512 break;
513 case AddToCrewEventData addToCrewEventData:
514 msg.Write((byte)addToCrewEventData.TeamType); // team id
515 ushort[] inventoryItemIDs = addToCrewEventData.InventoryItems.Select(item => item.ID).ToArray();
516 msg.Write((ushort)inventoryItemIDs.Length);
517 for (int i = 0; i < inventoryItemIDs.Length; i++)
518 {
519 msg.Write(inventoryItemIDs[i]);
520 }
521 break;
522 case UpdateExperienceEventData _:
523 msg.Write(Info.ExperiencePoints);
524 break;
525 case UpdateTalentsEventData _:
526 msg.Write((ushort)characterTalents.Count);
527 foreach (var unlockedTalent in characterTalents)
528 {
529 msg.Write(unlockedTalent.AddedThisRound);
530 msg.Write(unlockedTalent.Prefab.UintIdentifier);
531 }
532 break;
533 case UpdateMoneyEventData _:
534 msg.Write(GameMain.GameSession.Campaign.GetWallet(c).Balance);
535 break;
536 case UpdatePermanentStatsEventData updatePermanentStatsEventData:
537 StatTypes statType = updatePermanentStatsEventData.StatType;
538 if (Info == null)
539 {
540 msg.Write((byte)0);
541 msg.Write((byte)0);
542 }
543 else if (!Info.SavedStatValues.ContainsKey(statType))
544 {
545 msg.Write((byte)0);
546 msg.Write((byte)statType);
547 }
548 else
549 {
550 msg.Write((byte)Info.SavedStatValues[statType].Count);
551 msg.Write((byte)statType);
552 foreach (var savedStatValue in Info.SavedStatValues[statType])
553 {
554 msg.Write(savedStatValue.StatIdentifier);
555 msg.Write(savedStatValue.StatValue);
556 msg.Write(savedStatValue.RemoveOnDeath);
557 }
558 }
559 break;
560 default:
561 throw new Exception($"Malformed character event: did not expect {eventData.GetType().Name}");
562 }

566 private void WriteStatus(IWriteMessage msg)
567 {
568 msg.Write(IsDead);

578 {
579 CharacterHealth.ServerWrite(msg);
580 }
581 if (AnimController?.LimbJoints == null)
582 {
583 //0 limbs severed
584 msg.Write((byte)0);
585 }
586 else
587 {
588 severedJointIndices.Clear();
589 for (int i = 0; i < AnimController.LimbJoints.Length; i++)
590 {
591 if (AnimController.LimbJoints[i] != null && AnimController.LimbJoints[i].IsSevered)
592 {
593 severedJointIndices.Add(i);
594 }
595 }
596 msg.Write((byte)severedJointIndices.Count);
597 foreach (int jointIndex in severedJointIndices)
598 {
599 msg.Write((byte)jointIndex);
600 }
601 }

604 public void WriteSpawnData(IWriteMessage msg, UInt16 entityId, bool restrictMessageSize)
605 {
606 if (GameMain.Server == null) { return; }
607
608 int initialMsgLength = msg.LengthBytes;
609
610 msg.Write(Info == null);
611 msg.Write(entityId);
612 msg.Write(SpeciesName);
613 msg.Write(Seed);
614
615 if (Removed)
616 {
617 msg.Write(0.0f);
618 msg.Write(0.0f);
619 }
620 else
621 {
622 msg.Write(WorldPosition.X);
623 msg.Write(WorldPosition.Y);
624 }
625
626 msg.Write(Enabled);
627
628 //character with no characterinfo (e.g. some monster)
629 if (Info == null)
630 {
631 TryWriteStatus(msg);
632 return;
633 }
634
635 Client ownerClient = GameMain.Server.ConnectedClients.Find(c => c.Character == this);
636 if (ownerClient != null)
637 {
638 msg.Write(true);
639 msg.Write(ownerClient.ID);
640 }
641 else if (GameMain.Server.Character == this)
642 {
643 msg.Write(true);
644 msg.Write((byte)0);
645 }
646 else
647 {
648 msg.Write(false);
649 }
650 msg.Write(Wallet.Balance);
651 msg.WriteRangedInteger(Wallet.RewardDistribution, 0, 100);
652 msg.Write((byte)TeamID);
653 msg.Write(this is AICharacter);
654 msg.Write(info.SpeciesName);
655 int msgLengthBeforeInfo = msg.LengthBytes;
656 info.ServerWrite(msg);
657 int infoLength = msg.LengthBytes - msgLengthBeforeInfo;
658
659 msg.Write((byte)CampaignInteractionType);
660 if (CampaignInteractionType == CampaignMode.InteractionType.Store)
661 {
662 msg.Write(MerchantIdentifier);
663 }
664
665 int msgLengthBeforeOrders = msg.LengthBytes;
666 // Current orders
667 msg.Write((byte)info.CurrentOrders.Count(o => o != null));
668 foreach (var orderInfo in info.CurrentOrders)
669 {
670 if (orderInfo == null) { continue; }
671 msg.Write(orderInfo.Prefab.UintIdentifier);
672 msg.Write(orderInfo.TargetEntity == null ? (UInt16)0 : orderInfo.TargetEntity.ID);
673 var hasOrderGiver = orderInfo.OrderGiver != null;
674 msg.Write(hasOrderGiver);
675 if (hasOrderGiver) { msg.Write(orderInfo.OrderGiver.ID); }
676 msg.Write((byte)(orderInfo.Option == Identifier.Empty ? 0 : orderInfo.Prefab.Options.IndexOf(orderInfo.Option)));
677 msg.Write((byte)orderInfo.ManualPriority);
678 var hasTargetPosition = orderInfo.TargetPosition != null;
679 msg.Write(hasTargetPosition);
680 if (hasTargetPosition)
681 {
682 msg.Write(orderInfo.TargetPosition.Position.X);
683 msg.Write(orderInfo.TargetPosition.Position.Y);
684 msg.Write(orderInfo.TargetPosition.Hull == null ? (UInt16)0 : orderInfo.TargetPosition.Hull.ID);
685 }
686 }
687 int ordersLength = msg.LengthBytes - msgLengthBeforeOrders;
688
689 if (msg.LengthBytes - initialMsgLength >= 255 && restrictMessageSize)
690 {
691 string errorMsg = $"Error when writing character spawn data: data exceeded 255 bytes (info: {infoLength}, orders: {ordersLength}, total: {msg.LengthBytes - initialMsgLength})";
692 DebugConsole.ThrowError(errorMsg);
693 GameAnalyticsManager.AddErrorEventOnce("Character.WriteSpawnData:TooMuchData", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
694 }

698 void TryWriteStatus(IWriteMessage msg)
699 {
700 int msgLengthBeforeStatus = msg.LengthBytes - initialMsgLength;
701
702 var tempBuffer = new ReadWriteMessage();
703 WriteStatus(tempBuffer);
704 if (msgLengthBeforeStatus + tempBuffer.LengthBytes >= 255 && restrictMessageSize && GameMain.LuaCs.Networking.restrictMessageSize)
705 {
706 msg.Write(false);
707 if (msgLengthBeforeStatus < 255)
708 {
709 string errorMsg = $"Error when writing character spawn data: status data caused the length of the message to exceed 255 bytes ({msgLengthBeforeStatus} + {tempBuffer.LengthBytes})";
710 DebugConsole.ThrowError(errorMsg);
711 GameAnalyticsManager.AddErrorEventOnce("Character.WriteSpawnData:TooMuchDataForStatus", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
712 }
713 }
714 else
