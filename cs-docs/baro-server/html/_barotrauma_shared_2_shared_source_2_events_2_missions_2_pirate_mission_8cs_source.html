131  private float GetDifficultyModifiedValue(float preferredDifficulty, float levelDifficulty, float randomnessModifier, Random rand)
132  {
133  return Math.Abs(levelDifficulty - preferredDifficulty + MathHelper.Lerp(-randomnessModifier, randomnessModifier, (float)rand.NextDouble()));
134  }
135  private int GetDifficultyModifiedAmount(int minAmount, int maxAmount, float levelDifficulty, Random rand)
136  {
137  return Math.Max((int)Math.Round(minAmount + (maxAmount - minAmount) * (levelDifficulty + MathHelper.Lerp(-RandomnessModifier, RandomnessModifier, (float)rand.NextDouble())) / MaxDifficulty), minAmount);
138  }

140  private XElement GetRandomDifficultyModifiedElement(XElement parentElement, float levelDifficulty, float randomnessModifier)
141  {
142  Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
143  // look for the element that is closest to our difficulty, with some randomness

158  private void CreateMissionPositions(out Vector2 preferredSpawnPos)
159  {
160  Vector2 patrolPos = enemySub.WorldPosition;
161  Point subSize = enemySub.GetDockedBorders().Size;
162 
163  if (!Level.Loaded.TryGetInterestingPosition(true, Level.PositionType.MainPath, Level.Loaded.Size.X * 0.3f, out preferredSpawnPos))
164  {
165  DebugConsole.ThrowError("Could not spawn pirate submarine in an interesting location! " + this);
166  }
167  if (!Level.Loaded.TryGetInterestingPositionAwayFromPoint(true, Level.PositionType.MainPath, Level.Loaded.Size.X * 0.3f, out patrolPos, preferredSpawnPos, minDistFromPoint: 10000f))
168  {
169  DebugConsole.ThrowError("Could not give pirate submarine an interesting location to patrol to! " + this);
170  }
171 
172  patrolPos = enemySub.FindSpawnPos(patrolPos, subSize);
173 
174  patrolPositions.Add(patrolPos);
175  patrolPositions.Add(preferredSpawnPos);
176 
177  if (!IsClient)
178  {
179  PathFinder pathFinder = new PathFinder(WayPoint.WayPointList, false);
180  var path = pathFinder.FindPath(ConvertUnits.ToSimUnits(patrolPos), ConvertUnits.ToSimUnits(preferredSpawnPos));
181  if (!path.Unreachable)
182  {
183  var validNodes = path.Nodes.FindAll(n => !Level.Loaded.ExtraWalls.Any(w => w.Cells.Any(c => c.IsPointInside(n.WorldPosition))));
184  if (validNodes.Any())
185  {
186  preferredSpawnPos = validNodes.GetRandomUnsynced().WorldPosition; // spawn the sub in a random point in the path if possible
187  }
188  }
189 
190  int graceDistance = 500; // the sub still spawns awkwardly close to walls, so this helps. could also be given as a parameter instead
191  preferredSpawnPos = enemySub.FindSpawnPos(preferredSpawnPos, new Point(subSize.X + graceDistance, subSize.Y + graceDistance));
192  }

346  protected override void UpdateMissionSpecific(float deltaTime)
347  {
348  if (state >= 2) { return; }
349 
350  float sqrSonarRange = MathUtils.Pow2(Sonar.DefaultSonarRange);
351  outsideOfSonarRange = Vector2.DistanceSquared(enemySub.WorldPosition, Submarine.MainSub.WorldPosition) > sqrSonarRange;
352  if (CheckWinState())
353  {
354  State = 2;
355  }
356  else
357  {
358  switch (State)
359  {
360  case 0:
361  for (int i = patrolPositions.Count - 1; i >= 0; i--)
362  {
363  if (Vector2.DistanceSquared(patrolPositions[i], Submarine.MainSub.WorldPosition) < sqrSonarRange)
364  {
365  patrolPositions.RemoveAt(i);
366  }
367  }
368  if (!outsideOfSonarRange || patrolPositions.None())
369  {
370  State = 1;
371  }
372  break;
373  case 1:
374  if (outsideOfSonarRange)
375  {
376  if (lastSighting.HasValue && Vector2.DistanceSquared(lastSighting.Value, Submarine.MainSub.WorldPosition) < sqrSonarRange)
377  {
378  lastSighting = null;
379  }
380  pirateSightingUpdateTimer -= deltaTime;
381  if (pirateSightingUpdateTimer < 0)
382  {
383  pirateSightingUpdateTimer = pirateSightingUpdateFrequency;
384  lastSighting = enemySub.WorldPosition;
385  }
386  }
387  else
388  {
389  lastSighting = enemySub.WorldPosition;
390  pirateSightingUpdateTimer = 0;
391  }
392  break;
393  }
394  }
395  }
