26  public void ServerEventWrite(IWriteMessage msg, Client c, NetEntityEvent.IData extraData = null)
27  {
28  Exception error(string reason)
29  {
30  string errorMsg = $"Failed to write a network event for the item \"{Name}\" - {reason}";
31  GameAnalyticsManager.AddErrorEventOnce($"Item.ServerWrite:{Name}", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
32  return new Exception(errorMsg);
33  }
34 
35  if (extraData is null) { throw error("event data was null"); }
36  if (!(extraData is IEventData itemEventData)) { throw error($"event data was of the wrong type (\"{extraData.GetType().Name}\")"); }
37 
38  msg.WriteRangedInteger((int)itemEventData.EventType, (int)EventType.MinValue, (int)EventType.MaxValue);
39  switch (itemEventData)
40  {
41  case ComponentStateEventData componentStateEventData:
42  int componentIndex = components.IndexOf(componentStateEventData.Component);
43  if (componentIndex < 0)
44  {
45  throw error($"component index out of range ({componentIndex})");
46  }
47  if (!(components[componentIndex] is IServerSerializable serializableComponent))
48  {
49  throw error($"component \"{components[componentIndex]}\" is not server serializable");
50  }
51  msg.WriteRangedInteger(componentIndex, 0, components.Count - 1);
52  serializableComponent.ServerEventWrite(msg, c, extraData);
53  break;
54  case InventoryStateEventData inventoryStateEventData:
55  int containerIndex = components.IndexOf(inventoryStateEventData.Component);
56  if (containerIndex < 0)
57  {
58  throw error($"container index out of range ({containerIndex})");
59  }
60  if (!(components[containerIndex] is ItemContainer itemContainer))
61  {
62  throw error("component \"" + components[containerIndex] + "\" is not server serializable");
63  }
64  msg.WriteRangedInteger(containerIndex, 0, components.Count - 1);
65  msg.Write(GameMain.Server.EntityEventManager.Events.Last()?.ID ?? (ushort)0);
66  itemContainer.Inventory.ServerEventWrite(msg, c);
67  break;
68  case ItemStatusEventData _:
69  msg.Write(condition);
70  break;
71  case AssignCampaignInteractionEventData _:
72  msg.Write((byte)CampaignInteractionType);
73  break;
74  case ApplyStatusEffectEventData applyStatusEffectEventData:
75  {
76  ActionType actionType = applyStatusEffectEventData.ActionType;
77  ItemComponent targetComponent = applyStatusEffectEventData.TargetItemComponent;
78  Limb targetLimb = applyStatusEffectEventData.TargetLimb;
79  Vector2? worldPosition = applyStatusEffectEventData.WorldPosition;
80 
81  Character targetCharacter = applyStatusEffectEventData.TargetCharacter;
82  if (targetCharacter != null && targetCharacter.Removed) { targetCharacter = null; }
83  byte targetLimbIndex = targetLimb != null && targetCharacter != null ? (byte)Array.IndexOf(targetCharacter.AnimController.Limbs, targetLimb) : (byte)255;
84 
85  msg.WriteRangedInteger((int)actionType, 0, Enum.GetValues(typeof(ActionType)).Length - 1);
86  msg.Write((byte)(targetComponent == null ? 255 : components.IndexOf(targetComponent)));
87  msg.Write(applyStatusEffectEventData.TargetCharacter?.ID ?? (ushort)0);
88  msg.Write(targetLimbIndex);
89  msg.Write(applyStatusEffectEventData.UseTarget?.ID ?? (ushort)0);
90  msg.Write(worldPosition.HasValue);
91  if (worldPosition.HasValue)
92  {
93  msg.Write(worldPosition.Value.X);
94  msg.Write(worldPosition.Value.Y);
95  }
96  }
97  break;
98  case ChangePropertyEventData changePropertyEventData:
99  try
100  {
101  WritePropertyChange(msg, changePropertyEventData, inGameEditableOnly: !GameMain.NetworkMember.IsServer);
102  }
103  catch (Exception e)
104  {
105  throw new Exception(
106  $"Failed to write a ChangeProperty network event for the item \"{Name}\" ({e.Message})");
107  }
108  break;
109  case UpgradeEventData upgradeEventData:
110  var upgrade = upgradeEventData.Upgrade;
111  var upgradeTargets = upgrade.TargetComponents;
112  msg.Write(upgrade.Identifier);
113  msg.Write((byte)upgrade.Level);
114  msg.Write((byte)upgradeTargets.Count);
115  foreach (var (_, value) in upgrade.TargetComponents)
116  {
117  msg.Write((byte)value.Length);
118  foreach (var propertyReference in value)
119  {
120  object originalValue = propertyReference.OriginalValue;
121  msg.Write((float)(originalValue ?? -1));
122  }
123  }
124  break;
125  default:
126  throw error($"Unsupported event type {itemEventData.GetType().Name}");
127  }

130  public void ServerEventRead(IReadMessage msg, Client c)
131  {
132  EventType eventType =
133  (EventType)msg.ReadRangedInteger((int)EventType.MinValue, (int)EventType.MaxValue);
134 
135  c.KickAFKTimer = 0.0f;
136 
137  switch (eventType)
138  {
139  case EventType.ComponentState:
140  int componentIndex = msg.ReadRangedInteger(0, components.Count - 1);
141  (components[componentIndex] as IClientSerializable).ServerEventRead(msg, c);
142  break;
143  case EventType.InventoryState:
144  int containerIndex = msg.ReadRangedInteger(0, components.Count - 1);
145  (components[containerIndex] as ItemContainer).Inventory.ServerEventRead(msg, c);
146  break;
147  case EventType.Treatment:
148  if (c.Character == null || !c.Character.CanInteractWith(this)) return;
149 
150  UInt16 characterID = msg.ReadUInt16();
151  byte limbIndex = msg.ReadByte();
152 
153  Character targetCharacter = FindEntityByID(characterID) as Character;
154  if (targetCharacter == null) break;
155  if (targetCharacter != c.Character && c.Character.SelectedCharacter != targetCharacter) break;
156 
157  Limb targetLimb = limbIndex < targetCharacter.AnimController.Limbs.Length ? targetCharacter.AnimController.Limbs[limbIndex] : null;
158 
159  if (ContainedItems == null || ContainedItems.All(i => i == null))
160  {
161  GameServer.Log(GameServer.CharacterLogName(c.Character) + " used item " + Name, ServerLog.MessageType.ItemInteraction);
162  }
163  else
164  {
165  GameServer.Log(
166  GameServer.CharacterLogName(c.Character) + " used item " + Name + " (contained items: " + string.Join(", ", ContainedItems.Select(i => i.Name)) + ")",
167  ServerLog.MessageType.ItemInteraction);
168  }
169 
170  ApplyTreatment(c.Character, targetCharacter, targetLimb);
171 
172  break;
173  case EventType.ChangeProperty:
174  ReadPropertyChange(msg, inGameEditableOnly: GameMain.NetworkMember.IsServer, sender: c);
175  break;
176  case EventType.Combine:
177  UInt16 combineTargetID = msg.ReadUInt16();
178  Item combineTarget = FindEntityByID(combineTargetID) as Item;
179  if (combineTarget == null || !c.Character.CanInteractWith(this) || !c.Character.CanInteractWith(combineTarget))
180  {
181  return;
182  }
183  Combine(combineTarget, c.Character);
184  break;
185  }

276  partial void UpdateNetPosition(float deltaTime)
277  {
278  if (parentInventory != null || body == null || !body.Enabled || Removed)
279  {
280  PositionUpdateInterval = float.PositiveInfinity;
281  return;
282  }
283 
284  //gradually increase the interval of position updates
285  PositionUpdateInterval += deltaTime;
286 
287  float maxInterval = 30.0f;
288 
289  float velSqr = body.LinearVelocity.LengthSquared();
290  if (velSqr > 10.0f * 10.0f)
291  {
292  //over 10 m/s (projectile, thrown item or similar) -> send updates very frequently
293  maxInterval = 0.1f;
294  }
295  else if (velSqr > 1.0f)
296  {
297  //over 1 m/s
298  maxInterval = 0.25f;
299  }
300  else if (velSqr > 0.05f * 0.05f)
301  {
302  //over 0.05 m/s
303  maxInterval = 1.0f;
304  }
305 
306  PositionUpdateInterval = Math.Min(PositionUpdateInterval, maxInterval);
307  }

309  public float GetPositionUpdateInterval(Client recipient)
310  {
311  if (PositionUpdateInterval == float.PositiveInfinity || body == null || parentInventory != null)
312  {
313  return float.PositiveInfinity;
314  }
315 
316  if (recipient.Character == null || recipient.Character.IsDead)
317  {
318  //less frequent updates for clients who aren't controlling a character (max 2 updates/sec)
319  return Math.Max(PositionUpdateInterval, 0.5f);
320  }
321  else
322  {
323  float distSqr = Vector2.DistanceSquared(recipient.Character.WorldPosition, WorldPosition);
324  if (distSqr > 20000.0f * 20000.0f)
325  {
326  //don't send position updates at all if >20 000 units away
327  return float.PositiveInfinity;
328  }
329  else if (distSqr > 10000.0f * 10000.0f)
330  {
331  //drop the update rate to 10% if too far to see the item
332  return PositionUpdateInterval * 10;
333  }
334  else if (distSqr > 1000.0f * 1000.0f)
335  {
336  //halve the update rate if the client is far away (but still close enough to possibly see the item)
337  return PositionUpdateInterval * 2;
338  }
339  return PositionUpdateInterval;
340  }
