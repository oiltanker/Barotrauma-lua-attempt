160  public override void Update(float deltaTime, Camera cam)
161  {
162  if (source == null || target == null || target.Removed ||
163  (source is Entity sourceEntity && sourceEntity.Removed))
164  {
165  ResetSource();
166  target = null;
167  IsActive = false;
168  return;
169  }
170 
171  if (Snapped)
172  {
173  snapTimer += deltaTime;
174  if (snapTimer >= SnapAnimDuration)
175  {
176  IsActive = false;
177  }
178  return;
179  }
180 
181  Vector2 diff = target.WorldPosition - source.WorldPosition;
182  float lengthSqr = diff.LengthSquared();
183  if (lengthSqr > MaxLength * MaxLength)
184  {
185  Snap();
186  return;
187  }
188 
189  if (MaxAngle < 180 && lengthSqr > 2500)
190  {
191  if (launchDir == null)
192  {
193  launchDir = diff;
194  }
195  float angle = MathHelper.ToDegrees(VectorExtensions.Angle(launchDir.Value, diff));
196  if (angle > MaxAngle)
197  {
198  Snap();
199  return;
200  }
201  }
202 
203 #if CLIENT
204  item.ResetCachedVisibleSize();
205 #endif
206  var projectile = target.GetComponent<Projectile>();
207  if (projectile == null) { return; }
208 
209  if (SnapOnCollision)
210  {
211  raycastTimer += deltaTime;
212  if (raycastTimer > RayCastInterval)
213  {
214  if (Submarine.PickBody(ConvertUnits.ToSimUnits(source.WorldPosition), ConvertUnits.ToSimUnits(target.WorldPosition),
215  collisionCategory: Physics.CollisionLevel | Physics.CollisionWall,
216  customPredicate: (Fixture f) =>
217  {
218  foreach (Body body in projectile.Hits)
219  {
220  Submarine alreadyHitSub = null;
221  if (body.UserData is Structure hitStructure)
222  {
223  alreadyHitSub = hitStructure.Submarine;
224  }
225  else if (body.UserData is Submarine hitSub)
226  {
227  alreadyHitSub = hitSub;
228  }
229  if (alreadyHitSub != null)
230  {
231  if (f.Body?.UserData is MapEntity me && me.Submarine == alreadyHitSub) { return false; }
232  if (f.Body?.UserData as Submarine == alreadyHitSub) { return false; }
233  }
234  }
235  Submarine targetSub = projectile.StickTarget?.UserData as Submarine ?? target.Submarine;
236 
237  if (f.Body?.UserData is MapEntity mapEntity && mapEntity.Submarine != null)
238  {
239  if (mapEntity.Submarine == targetSub || mapEntity.Submarine == source.Submarine)
240  {
241  return false;
242  }
243  }
244  else if (f.Body?.UserData is Submarine sub)
245  {
246  if (sub == targetSub || sub == source.Submarine)
247  {
248  return false;
249  }
250  }
251  return true;
252  }) != null)
253  {
254  Snap();
255  return;
256  }
257  raycastTimer = 0.0f;
258  }
259  }

261  Vector2 forceDir = diff;
262  float distance = diff.Length();
263  if (distance > 0.001f)
264  {
265  forceDir = Vector2.Normalize(forceDir);
266  }
267 
268  if (Math.Abs(ProjectilePullForce) > 0.001f)
269  {
270  projectile.Item?.body?.ApplyForce(-forceDir * ProjectilePullForce);
271  }
272 
273  if (projectile.StickTarget != null)
274  {
275  float targetMass = float.MaxValue;
276  Character targetCharacter = null;
277  if (projectile.StickTarget.UserData is Limb targetLimb)
278  {
279  targetCharacter = targetLimb.character;
280  targetMass = targetLimb.ragdoll.Mass;
281  }
282  else if (projectile.StickTarget.UserData is Character character)
283  {
284  targetCharacter = character;
285  targetMass = character.Mass;
286  }
287  else if (projectile.StickTarget.UserData is Item item)
288  {
289  targetMass = projectile.StickTarget.Mass;
290  }
291  if (projectile.StickTarget.BodyType != BodyType.Dynamic)
292  {
293  targetMass = float.MaxValue;
294  }
295  var user = item.GetComponent<Projectile>()?.User;
296  if (targetMass > TargetMinMass)
297  {
298  if (Math.Abs(SourcePullForce) > 0.001f)
299  {
300  var sourceBody = GetBodyToPull(source);
301  if (sourceBody != null)
302  {
303  var targetBody = GetBodyToPull(target);
304  if (targetBody != null && !(targetBody.UserData is Character))
305  {
306  sourceBody.ApplyForce(targetBody.LinearVelocity * sourceBody.Mass);
307  }
308  float forceMultiplier = 1;
309  if (user != null)
310  {
311  user.AnimController.Hang();
312  if (user.InWater)
313  {
314  if (user.IsRagdolled)
315  {
316  forceMultiplier = 0;
317  }
318  }
319  else
320  {
321  forceMultiplier = user.IsRagdolled ? 0.1f : 0.4f;
322  // Prevents too easy smashing to the walls
323  forceDir.X /= 4;
324  // Prevents rubberbanding up and down
325  if (forceDir.Y < 0)
326  {
327  forceDir.Y = 0;
328  }
329  }
330  if (targetCharacter != null)
331  {
332  var myCollider = user.AnimController.Collider;
333  var targetCollider = targetCharacter.AnimController.Collider;
334  if (myCollider.LinearVelocity != Vector2.Zero && targetCollider.LinearVelocity != Vector2.Zero)
335  {
336  if (Vector2.Dot(Vector2.Normalize(myCollider.LinearVelocity), Vector2.Normalize(targetCollider.LinearVelocity)) < 0)
337  {
338  myCollider.ApplyForce(targetCollider.LinearVelocity * targetCollider.Mass);
339  }
340  }
341  }
342  }
343  float force = LerpForces ? MathHelper.Lerp(0, SourcePullForce, MathUtils.InverseLerp(0, MaxLength / 2, distance)) * forceMultiplier : SourcePullForce * forceMultiplier;
344  sourceBody.ApplyForce(forceDir * force);
345  }
346  }
347  }
348  if (Math.Abs(TargetPullForce) > 0.001f)
349  {
350  var targetBody = GetBodyToPull(target);
351  if (user != null && targetCharacter != null && !user.AnimController.InWater)
352  {
353  // Prevents rubberbanding horizontally when dragging a corpse.
354  if ((forceDir.X < 0) != (user.AnimController.Dir < 0))
355  {
356  forceDir.X = Math.Clamp(forceDir.X, -0.1f, 0.1f);
357  }
358  }
359  float force = LerpForces ? MathHelper.Lerp(0, TargetPullForce, MathUtils.InverseLerp(0, MaxLength / 3, distance - 50)) : TargetPullForce;
360  targetBody?.ApplyForce(-forceDir * force);
361  var targetRagdoll = targetCharacter?.AnimController;
362  if (targetRagdoll != null && (targetRagdoll.InWater || targetRagdoll.OnGround))
363  {
364  targetRagdoll.Collider.ApplyForce(-forceDir * force * 3);
365  }
366  }

383  private PhysicsBody GetBodyToPull(ISpatialEntity target)
384  {
385  if (target is Item targetItem)
386  {
387  if (targetItem.ParentInventory is CharacterInventory characterInventory &&
388  characterInventory.Owner is Character ownerCharacter)
389  {
390  if (ownerCharacter.Removed) { return null; }
391  return ownerCharacter.AnimController.Collider;
392  }
393  var projectile = targetItem.GetComponent<Projectile>();
394  if (projectile != null && projectile.StickTarget != null)
395  {
396  if (projectile.StickTarget.UserData is Structure structure)
397  {
398  return structure.Submarine?.PhysicsBody;
399  }
400  else if (projectile.StickTarget.UserData is Submarine sub)
401  {
402  return sub.PhysicsBody;
403  }
404  else if (projectile.StickTarget.UserData is Item item)
405  {
406  return item.body;
407  }
408  else if (projectile.StickTarget.UserData is Limb limb)
409  {
410  return limb.body;
411  }
412  return null;
413  }
414  if (targetItem.body != null) { return targetItem.body; }
415  }
416  else if (target is Limb targetLimb)
417  {
418  return targetLimb.body;
419  }
