17 GitRevision = attrs.FirstOrDefault(a => a.Key == "GitRevision")?.Value;

21 ProjectDir = attrs.FirstOrDefault(a => a.Key == "ProjectDir")?.Value;

23 string[] dirSplit = ProjectDir.Split('/', '\\');
24 ProjectDir = string.Join(ProjectDir.Contains('/') ? '/' : '\\', dirSplit.Take(dirSplit.Length - 2));
