40  protected override float GetPriority()
41  {
42  bool isOrder = objectiveManager.IsOrder(this);
43  if (!IsAllowed || character.LockHands)
44  {
45  Priority = 0;
46  Abandon = !isOrder;
47  return Priority;
48  }
49  if (!isOrder && component.Item.ConditionPercentage <= 0)
50  {
51  Priority = 0;
52  }
53  else
54  {
55  if (isOrder)
56  {
57  Priority = objectiveManager.GetOrderPriority(this);
58  }
59  ItemComponent target = GetTarget();
60  Item targetItem = target?.Item;
61  if (targetItem == null)
62  {
63 #if DEBUG
64  DebugConsole.ThrowError("Item or component of AI Objective Operate item was null. This shouldn't happen.");
65 #endif
66  Abandon = true;
67  Priority = 0;
68  return Priority;
69  }
70  else if (targetItem.IsClaimedByBallastFlora)
71  {
72  Priority = 0;
73  return Priority;
74  }
75  var reactor = component?.Item.GetComponent<Reactor>();
76  if (reactor != null)
77  {
78  if (!isOrder)
79  {
80  if (reactor.LastUserWasPlayer && character.TeamID != CharacterTeamType.FriendlyNPC)
81  {
82  // The reactor was previously operated by a player -> ignore.
83  Priority = 0;
84  return Priority;
85  }
86  }
87  switch (Option.Value.ToLowerInvariant())
88  {
89  case "shutdown":
90  if (!reactor.PowerOn)
91  {
92  Priority = 0;
93  return Priority;
94  }
95  break;
96  case "powerup":
97  // Check that we don't already have another order that is targeting the same item.
98  // Without this the autonomous objective will tell the bot to turn the reactor on again.
99  if (IsAnotherOrderTargetingSameItem(objectiveManager.ForcedOrder) || objectiveManager.CurrentOrders.Any(o => IsAnotherOrderTargetingSameItem(o.Objective)))
100  {
101  Priority = 0;
102  return Priority;
103  }
104  bool IsAnotherOrderTargetingSameItem(AIObjective objective)
105  {
106  return objective is AIObjectiveOperateItem operateObjective && operateObjective != this && operateObjective.GetTarget() == target && operateObjective.Option != Option;
107  }
108  break;
109  }
110  }
111  else if (!isOrder)
112  {
113  var steering = component?.Item.GetComponent<Steering>();
114  if (steering != null && (steering.AutoPilot || HumanAIController.IsTrueForAnyCrewMember(c => c != HumanAIController && c.Character.IsCaptain)))
115  {
116  // Ignore if already set to autopilot or if there's a captain onboard
117  Priority = 0;
118  return Priority;
119  }
120  }
121  if (targetItem.CurrentHull == null ||
122  targetItem.Submarine != character.Submarine && !isOrder ||
123  targetItem.CurrentHull.FireSources.Any() ||
124  HumanAIController.IsItemOperatedByAnother(target, out _) ||
125  Character.CharacterList.Any(c => c.CurrentHull == targetItem.CurrentHull && !HumanAIController.IsFriendly(c) && HumanAIController.IsActive(c))
126  || component.Item.IgnoreByAI(character) || useController && controller.Item.IgnoreByAI(character))
127  {
128  Priority = 0;
129  }
130  else
131  {
132  if (isOrder)
133  {
134  float max = objectiveManager.GetOrderPriority(this);
135  float value = CumulatedDevotion + (max * PriorityModifier);
136  Priority = MathHelper.Clamp(value, 0, max);
137  }
138  else
139  {
140  float value = CumulatedDevotion + (AIObjectiveManager.LowestOrderPriority * PriorityModifier);
141  float max = AIObjectiveManager.LowestOrderPriority - 1;
142  if (reactor != null && reactor.PowerOn && reactor.FissionRate > 1 && reactor.AutoTemp && Option == "powerup")
143  {
144  // Already on, no need to operate.
145  value = 0;
146  }
147  Priority = MathHelper.Clamp(value, 0, max);
148  }
149  }
150  }
151  return Priority;
152  }

177  protected override void Act(float deltaTime)
178  {
179  if (character.LockHands)
180  {
181  Abandon = true;
182  return;
183  }
184  ItemComponent target = GetTarget();
185  if (useController && controller == null)
186  {
187  if (character.IsOnPlayerTeam)
188  {
189  character.Speak(TextManager.GetWithVariable("DialogCantFindController", "[item]", component.Item.Name).Value, delay: 2.0f, identifier: "cantfindcontroller".ToIdentifier(), minDurationBetweenSimilar: 30.0f);
190  }
191  Abandon = true;
192  return;
193  }
194  if (operateTarget != null)
195  {
196  if (HumanAIController.IsTrueForAnyCrewMember(other => other != HumanAIController && other.Character.IsBot && other.ObjectiveManager.GetActiveObjective() is AIObjectiveOperateItem operateObjective && operateObjective.operateTarget == operateTarget))
197  {
198  // Another crew member is already targeting this entity (leak).
199  Abandon = true;
200  return;
201  }
202  }
203  if (target.CanBeSelected)
204  {
205  if (!character.IsClimbing && character.CanInteractWith(target.Item, out _, checkLinked: false))
206  {
207  HumanAIController.FaceTarget(target.Item);
208  if (character.SelectedConstruction != target.Item)
209  {
210  target.Item.TryInteract(character, forceSelectKey: true);
211  }
212  if (component.AIOperate(deltaTime, character, this))
213  {
214  isDoneOperating = completionCondition == null || completionCondition();
215  }
216  }
217  else
218  {
219  TryAddSubObjective(ref goToObjective, () => new AIObjectiveGoTo(target.Item, character, objectiveManager, closeEnough: 50)
220  {
221  TargetName = target.Item.Name,
222  endNodeFilter = node => node.Waypoint.Ladders == null
223  },
224  onAbandon: () => Abandon = true,
225  onCompleted: () => RemoveSubObjective(ref goToObjective));
226  }
227  }
228  else
229  {
230  if (component.Item.GetComponent<Pickable>() == null)
231  {
232  //controller/target can't be selected and the item cannot be picked -> objective can't be completed
233  Abandon = true;
234  return;
235  }
236  else if (!character.Inventory.Contains(component.Item))
237  {
238  TryAddSubObjective(ref getItemObjective, () => new AIObjectiveGetItem(character, component.Item, objectiveManager, equip: true),
239  onAbandon: () => Abandon = true,
240  onCompleted: () => RemoveSubObjective(ref getItemObjective));
241  }
242  else
243  {
244  if (requireEquip && !character.HasEquippedItem(component.Item))
245  {
246  //the item has to be equipped before using it if it's holdable
247  var holdable = component.Item.GetComponent<Holdable>();
248  if (holdable == null)
249  {
250 #if DEBUG
251  DebugConsole.ThrowError($"{character.Name}: AIObjectiveOperateItem failed - equipping item " + component.Item + " is required but the item has no Holdable component");
252 #endif
253  return;
254  }
255  for (int i = 0; i < character.Inventory.Capacity; i++)
256  {
257  if (character.Inventory.SlotTypes[i] == InvSlotType.Any || !holdable.AllowedSlots.Any(s => s.HasFlag(character.Inventory.SlotTypes[i])))
258  {
259  continue;
260  }
261  //equip slot already taken
262  var existingItem = character.Inventory.GetItemAt(i);
263  if (existingItem != null)
264  {
265  //try to put the item in an Any slot, and drop it if that fails
266  if (!existingItem.AllowedSlots.Contains(InvSlotType.Any) ||
267  !character.Inventory.TryPutItem(existingItem, character, new List<InvSlotType>() { InvSlotType.Any }))
268  {
269  existingItem.Drop(character);
270  }
271  }
272  if (character.Inventory.TryPutItem(component.Item, i, true, false, character))
273  {
274  component.Item.Equip(character);
275  break;
276  }
277  }
278  return;
279  }
280  if (component.AIOperate(deltaTime, character, this))
281  {
282  isDoneOperating = completionCondition == null || completionCondition();
283  }
284  }
285  }
