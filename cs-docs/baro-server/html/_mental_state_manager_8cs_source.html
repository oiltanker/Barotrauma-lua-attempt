50  private void UpdateMentalState()
51  {
52  MentalType newMentalType = GetMentalType(character.CharacterHealth.GetAffliction("psychosis"));
53  bool createdCombat = false;
54 
55  switch (newMentalType)
56  {
57  case MentalType.Normal:
58  case MentalType.Confused:
59  // remove combat if we became normal again
60  mentalBehaviorTimer = 0f;
61  break;
62  case MentalType.Afraid:
63  case MentalType.Desperate:
64  case MentalType.Berserk:
65  // berserk is not removed unless we drop to normal behavior again
66  if (CurrentMentalType == MentalType.Berserk)
67  {
68  newMentalType = MentalType.Berserk;
69  }
70  // give players a full interval to react to mental changes
71  if (newMentalType == CurrentMentalType)
72  {
73  createdCombat = CreateCombatBehavior(CurrentMentalType);
74  }
75  break;
76  }
77 
78  if (!createdCombat)
79  {
80  CreateDialogueBehavior(newMentalType);
81  }
82 
83  if (newMentalType != MentalType.Berserk)
84  {
85  character.TryRemoveTeamChange(MentalTeamChange);
86  }
87 
88  CurrentMentalType = newMentalType;
89  }

104  private MentalType GetMentalType(Affliction affliction)
105  {
106  if (affliction == null)
107  {
108  return MentalType.Normal;
109  }
110  // test this later
111  int psychosisIndex = (int)(affliction.Strength / (affliction.Prefab.MaxStrength / MentalTypeCount) * Rand.Range(1f, 1.2f));
112  psychosisIndex = Math.Clamp(psychosisIndex, 0, 4);
113  MentalType mentalType = psychosisIndex switch
114  {
115  0 => MentalType.Normal,
116  1 => MentalType.Confused,
117  2 => MentalType.Afraid,
118  3 => MentalType.Desperate,
119  4 => MentalType.Berserk,
120  _ => throw new ArgumentOutOfRangeException(psychosisIndex.ToString()),
121  };
122  return mentalType;
123  }

125  public bool CreateCombatBehavior(MentalType mentalType)
126  {
127  Character mentalAttackTarget = Character.CharacterList.Where(
128  possibleTarget => HumanAIController.IsActive(possibleTarget) &&
129  (possibleTarget.TeamID != character.TeamID || mentalType == MentalType.Berserk) &&
130  humanAIController.VisibleHulls.Contains(possibleTarget.CurrentHull) &&
131  possibleTarget != character).GetRandomUnsynced();
132 
133  if (mentalAttackTarget == null)
134  {
135  return false;
136  }
137 
138  var combatMode = AIObjectiveCombat.CombatMode.None;
139  bool holdFire = mentalType == MentalType.Afraid && character.IsSecurity;
140  switch (mentalType)
141  {
142  case MentalType.Afraid:
143  combatMode = character.IsSecurity ? AIObjectiveCombat.CombatMode.Arrest : AIObjectiveCombat.CombatMode.Retreat;
144  break;
145  case MentalType.Desperate:
146  // might be unnecessary to explicitly declare as arrest against non-humans
147  combatMode = character.IsSecurity && mentalAttackTarget.IsHuman ? AIObjectiveCombat.CombatMode.Arrest : AIObjectiveCombat.CombatMode.Defensive;
148  break;
149  case MentalType.Berserk:
150  combatMode = AIObjectiveCombat.CombatMode.Offensive;
151  break;
152  }
153 
154  // using this as an explicit time-out for the behavior. it's possible it will never run out because of the manager being disabled, but combat objective has failsafes for that
155  mentalBehaviorTimer = MentalBehaviorInterval;
156  humanAIController.AddCombatObjective(combatMode, mentalAttackTarget, allowHoldFire: holdFire, abortCondition: obj => mentalBehaviorTimer <= 0f);
157  Identifier textIdentifier = $"dialogmentalstatereaction{combatMode}".ToIdentifier();
158  character.Speak(TextManager.Get(textIdentifier).Value, delay: Rand.Range(0.5f, 1.0f), identifier: textIdentifier, minDurationBetweenSimilar: 25f);
159 
160  if (mentalType == MentalType.Berserk && !character.HasTeamChange(MentalTeamChange))
161  {
162  // TODO: could this be handled in the switch block above?
163  character.TryAddNewTeamChange(MentalTeamChange, new ActiveTeamChange(CharacterTeamType.None, ActiveTeamChange.TeamChangePriorities.Absolute, aggressiveBehavior: true));
164  }
165 
166  return true;
167  }
