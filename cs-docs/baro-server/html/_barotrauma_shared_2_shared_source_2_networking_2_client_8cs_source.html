264 public static void ReadPermissions(IReadMessage inc, out ClientPermissions permissions, out List<DebugConsole.Command> permittedCommands)
265 {
266 int permissionsInt = inc.ReadRangedInteger(0, (int)ClientPermissions.All);
267 permissions = ClientPermissions.None;
268 permittedCommands = new List<DebugConsole.Command>();
269 try
270 {
271 permissions = (ClientPermissions)permissionsInt;
272 }
273 catch (InvalidCastException)
274 {
275 return;
276 }
277 if (permissions.HasFlag(ClientPermissions.ConsoleCommands))
278 {
279 UInt16 commandCount = inc.ReadUInt16();
280 for (int i = 0; i < commandCount; i++)
281 {
282 string commandName = inc.ReadString();
283 var consoleCommand = DebugConsole.Commands.Find(c => c.names.Contains(commandName));
284 if (consoleCommand != null)
285 {
286 permittedCommands.Add(consoleCommand);
287 }
288 }
289 }
290 }
