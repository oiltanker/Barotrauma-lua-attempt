52 public override void Update(float deltaTime)
53 {
54 if (isFinished) { return; }
55
56 if (GameMain.GameSession.GameMode is CampaignMode campaign)
57 {
58 MissionPrefab prefab = null;
59 var unlockLocation = FindUnlockLocation();
60 if (unlockLocation == null && CreateLocationIfNotFound)
61 {
62 //find an empty location at least 3 steps away, further on the map
63 var emptyLocation = FindUnlockLocationRecursive(campaign.Map.CurrentLocation, Math.Max(MinLocationDistance, 3), "none", true, new HashSet<Location>());
64 if (emptyLocation != null)
65 {
66 emptyLocation.ChangeType(Barotrauma.LocationType.Prefabs[LocationType]);
67 unlockLocation = emptyLocation;
68 }
69 }
70
71 if (unlockLocation != null)
72 {
73 if (!MissionIdentifier.IsEmpty)
74 {
75 prefab = unlockLocation.UnlockMissionByIdentifier(MissionIdentifier);
76 }
77 else if (!MissionTag.IsEmpty)
78 {
79 prefab = unlockLocation.UnlockMissionByTag(MissionTag);
80 }
81 if (campaign is MultiPlayerCampaign mpCampaign)
82 {
83 mpCampaign.LastUpdateID++;
84 }
85 if (prefab != null)
86 {
87 DebugConsole.NewMessage($"Unlocked mission \"{prefab.Name}\" in the location \"{unlockLocation.Name}\".");
88 #if CLIENT
89 new GUIMessageBox(string.Empty, TextManager.GetWithVariable("missionunlocked", "[missionname]", prefab.Name),
90 Array.Empty<LocalizedString>(), type: GUIMessageBox.Type.InGame, icon: prefab.Icon, relativeSize: new Vector2(0.3f, 0.15f), minSize: new Point(512, 128))
91 {
92 IconColor = prefab.IconColor
93 };
94 #else
95 NotifyMissionUnlock(prefab);
96 #endif
97 }
98 }
99 else
100 {
101 DebugConsole.AddWarning($"Failed to find a suitable location to unlock a mission in (LocationType: {LocationType}, MinLocationDistance: {MinLocationDistance}, UnlockFurtherOnMap: {UnlockFurtherOnMap})");
102 }
103 }
104 isFinished = true;
105 }

107 private Location FindUnlockLocation()
108 {
109 var campaign = GameMain.GameSession.GameMode as CampaignMode;
110 if (string.IsNullOrEmpty(LocationType) && MinLocationDistance <= 1)
111 {
112 return campaign.Map.CurrentLocation;
113 }
114
115 return FindUnlockLocationRecursive(campaign.Map.CurrentLocation, 0, LocationType, UnlockFurtherOnMap, new HashSet<Location>());
116 }
117
118 private Location FindUnlockLocationRecursive(Location currLocation, int currDistance, string locationType, bool unlockFurtherOnMap, HashSet<Location> checkedLocations)
119 {
120 var campaign = GameMain.GameSession.GameMode as CampaignMode;
121 if (currLocation.Type.Identifier == locationType && currDistance >= MinLocationDistance &&
122 (!unlockFurtherOnMap || currLocation.MapPosition.X > campaign.Map.CurrentLocation.MapPosition.X))
123 {
124 return currLocation;
125 }
126 checkedLocations.Add(currLocation);
127 foreach (LocationConnection connection in currLocation.Connections)
128 {
129 var otherLocation = connection.OtherLocation(currLocation);
130 if (checkedLocations.Contains(otherLocation)) { continue; }
131 var unlockLocation = FindUnlockLocationRecursive(otherLocation, ++currDistance, locationType, unlockFurtherOnMap, checkedLocations);
132 if (unlockLocation != null) { return unlockLocation; }
133 }
134 return null;
135 }
