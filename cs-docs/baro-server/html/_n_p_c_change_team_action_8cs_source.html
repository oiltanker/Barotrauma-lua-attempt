25 public override void Update(float deltaTime)
26 {
27 if (isFinished) { return; }
28
29 affectedNpcs = ParentEvent.GetTargets(NPCTag).Where(c => c is Character).Select(c => c as Character).ToList();
30 foreach (var npc in affectedNpcs)
31 {
32 // characters will still remain on friendlyNPC team for rest of the tick
33 npc.SetOriginalTeam(TeamTag);
34
35 if (AddToCrew && (TeamTag == CharacterTeamType.Team1 || TeamTag == CharacterTeamType.Team2))
36 {
37 npc.Info.StartItemsGiven = true;
38
39 GameMain.GameSession.CrewManager.AddCharacter(npc);
40 foreach (Item item in npc.Inventory.AllItems)
41 {
42 item.AllowStealing = true;
43 var wifiComponent = item.GetComponent<Items.Components.WifiComponent>();
44 if (wifiComponent != null)
45 {
46 wifiComponent.TeamID = TeamTag;
47 }
48 var idCard = item.GetComponent<Items.Components.IdCard>();
49 if (idCard != null)
50 {
51 idCard.TeamID = TeamTag;
52 idCard.SubmarineSpecificID = 0;
53 }
54 }
55
56 WayPoint subWaypoint =
57 WayPoint.WayPointList.Find(wp => wp.Submarine == Submarine.MainSub && wp.SpawnType == SpawnType.Human && wp.AssignedJob == npc.Info.Job?.Prefab) ??
58 WayPoint.WayPointList.Find(wp => wp.Submarine == Submarine.MainSub && wp.SpawnType == SpawnType.Human);
59 if (subWaypoint != null)
60 {
61 npc.GiveIdCardTags(subWaypoint, createNetworkEvent: true);
62 }
63#if SERVER
64 GameMain.NetworkMember.CreateEntityEvent(npc, new Character.AddToCrewEventData(TeamTag, npc.Inventory.AllItems));
65#endif
66 }
67 }
68 isFinished = true;
69 }
