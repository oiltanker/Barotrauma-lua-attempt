185  public override float GetCurrentPowerConsumption(Connection connection = null)
186  {
187  if (connection == powerIn)
188  {
189  //Don't draw power if fully charged
190  if (charge >= capacity)
191  {
192  charge = capacity;
193  return 0;
194  }
195  else
196  {
197  if (item.Condition <= 0.0f) { return 0.0f; }
198 
199  float missingCharge = capacity - charge;
200  float targetRechargeSpeed = rechargeSpeed;
201 
202  if (ExponentialRechargeSpeed)
203  {
204  targetRechargeSpeed = MathF.Pow(rechargeSpeed / maxRechargeSpeed, 2) * maxRechargeSpeed;
205  }
206  //For the last kwMin scale the recharge rate linearly to prevent overcharging and to have a smooth cutoff
207  if (missingCharge < 1.0f)
208  {
209  targetRechargeSpeed *= missingCharge;
210  }
211 
212  return MathHelper.Clamp(targetRechargeSpeed, 0, MaxRechargeSpeed);
213  }
214  }
215  else
216  {
217  CurrPowerOutput = 0;
218  return charge > 0 ? -1 : 0;
219  }
220  }

231  public override PowerRange MinMaxPowerOut(Connection connection, float load = 0)
232  {
233  if (connection == powerOut)

294  public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
295  {
296  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return false; }
297 
298  if (objective.Override)
299  {
300  HasBeenTuned = false;
301  }
302  if (HasBeenTuned) { return true; }
303 
304  float targetRatio = objective.Option.IsEmpty || objective.Option == "charge" ? aiRechargeTargetRatio : -1;
305  if (targetRatio > 0 || float.TryParse(objective.Option.Value, out targetRatio))
306  {
307  if (Math.Abs(rechargeSpeed - maxRechargeSpeed * targetRatio) > 0.05f)
308  {
309 #if SERVER
310  item.CreateServerEvent(this);
311 #endif
312  RechargeSpeed = maxRechargeSpeed * targetRatio;
313 #if CLIENT
314  if (rechargeSpeedSlider != null)
315  {
316  rechargeSpeedSlider.BarScroll = RechargeSpeed / Math.Max(maxRechargeSpeed, 1.0f);
317  }
318 #endif
319  if (character.IsOnPlayerTeam)
320  {
321  character.Speak(TextManager.GetWithVariables("DialogChargeBatteries",
322  ("[itemname]", item.Name, FormatCapitals.Yes),
323  ("[rate]", ((int)(rechargeSpeed / maxRechargeSpeed * 100.0f)).ToString(), FormatCapitals.No)).Value,
324  null, 1.0f, "chargebattery".ToIdentifier(), 10.0f);
325  }
326  }
327  }
328  else
329  {
330  if (rechargeSpeed > 0.0f)
331  {
332 #if SERVER
333  item.CreateServerEvent(this);
334 #endif
335  RechargeSpeed = 0.0f;
336 #if CLIENT
337  if (rechargeSpeedSlider != null)
338  {
339  rechargeSpeedSlider.BarScroll = RechargeSpeed / Math.Max(maxRechargeSpeed, 1.0f);
340  }
341 #endif
342  if (character.IsOnPlayerTeam)
343  {
344  character.Speak(TextManager.GetWithVariables("DialogStopChargingBatteries",
345  ("[itemname]", item.Name, FormatCapitals.Yes),
346  ("[rate]", ((int)(rechargeSpeed / maxRechargeSpeed * 100.0f)).ToString(), FormatCapitals.No)).Value,
347  null, 1.0f, "chargebattery".ToIdentifier(), 10.0f);
348  }
349  }
350  }
351 
352  return true;
