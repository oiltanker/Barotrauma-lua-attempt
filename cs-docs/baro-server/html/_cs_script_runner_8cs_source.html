51  public object Run(string code)
52  {
53  object scriptResilt = null;
54 
55  try
56  {
57  code = ToOneTimeScript(code);
58  var syntaxTree = SyntaxFactory.ParseSyntaxTree(code, ParseOptions);
59  var compilation = CSharpCompilation.Create(NET_ONE_TIME_SCRIPT_ASSEMBLY, new[] { AssemblyInfoSyntaxTree(NET_ONE_TIME_SCRIPT_ASSEMBLY), syntaxTree }, defaultReferences, compileOptions);
60 
61  Assembly assembly = null;
62  using (var mem = new MemoryStream())
63  {
64  var result = compilation.Emit(mem);
65  if (!result.Success)
66  {
67  IEnumerable<Diagnostic> failures = result.Diagnostics.Where(d => d.IsWarningAsError || d.Severity == DiagnosticSeverity.Error);
68 
69  string errStr = "Script compilation errors:";
70  var lineErr = new SortedDictionary<int, (string, string)>();
71  foreach (Diagnostic diagnostic in failures)
72  {
73  var line = syntaxTree.GetLineSpan(diagnostic.Location.SourceSpan).StartLinePosition.Line;
74  lineErr[line] = (diagnostic.Id, diagnostic.ToString());
75  }
76  var lines = code.Split('\n');
77  for (var i = 1; i < lines.Length - 1; i++)
78  {
79  errStr += $"\n{i} >> {lines[i]}";
80  if (lineErr.ContainsKey(i)) errStr += $" <=== {lineErr[i].Item1}";
81  }
82  errStr += "\n";
83  foreach ((var idx, (var id, var err)) in lineErr)
84  {
85  errStr += $"\n{idx}: {err}";
86  }
87  LuaCsSetup.PrintCsError(errStr);
88  }
89  else
90  {
91  mem.Seek(0, SeekOrigin.Begin);
92  var errStr = CsScriptFilter.FilterOneTimeMetadata(new PEReader(mem).GetMetadataReader());
93  if (errStr == null)
94  {
95  mem.Seek(0, SeekOrigin.Begin);
96  assembly = LoadFromStream(mem);
97  var runner = assembly.CreateInstance("NetOneTimeScript.NetOneTimeScriptRunner");
98  if (runner != null)
99  {
100  var method = runner.GetType().GetMethod("Run", BindingFlags.Public | BindingFlags.Instance);
101  if (method != null)
102  {
103  scriptResilt = method.Invoke(runner, null);
104  foreach (var type in assembly.GetTypes())
105  {
106  UserData.UnregisterType(type, true);
107  }
108  }
109  else LuaCsSetup.PrintCsError("Script Error - no run method detected");
110  }
111  else LuaCsSetup.PrintCsError("Script Error - no runner class detected");
112  }
113  else LuaCsSetup.PrintCsError(errStr);
114  }
115  }
116  Unload();
117  }
118  catch (Exception ex)
119  {
120  LuaCsSetup.PrintCsError("Error running script:\n" + ex.Message + "\n" + ex.StackTrace);
121  }
122 
123  return scriptResilt;
124  }
