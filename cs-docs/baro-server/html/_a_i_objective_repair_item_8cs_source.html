32 : base(character, objectiveManager, priorityModifier)

38 protected override float GetPriority()
39 {
40 if (!IsAllowed || Item.IgnoreByAI(character))
41 {
42 Priority = 0;
43 Abandon = true;
44 if (IsRepairing())
45 {
46 Item.Repairables.ForEach(r => r.StopRepairing(character));
47 }
48 return Priority;
49 }
50 if (HumanAIController.IsItemRepairedByAnother(Item, out _))
51 {
52 Priority = 0;
53 IsCompleted = true;
54 }
55 else if (Item.IsClaimedByBallastFlora)
56 {
57 Priority = 0;
58 }
59 else
60 {
61 float distanceFactor = 1;
62 if (!isPriority && Item.CurrentHull != character.CurrentHull)
63 {
64 float yDist = Math.Abs(character.WorldPosition.Y - Item.WorldPosition.Y);
65 yDist = yDist > 100 ? yDist * 5 : 0;
66 float dist = Math.Abs(character.WorldPosition.X - Item.WorldPosition.X) + yDist;
67 distanceFactor = MathHelper.Lerp(1, 0.25f, MathUtils.InverseLerp(0, 4000, dist));
68 }
69 float requiredSuccessFactor = objectiveManager.HasOrder<AIObjectiveRepairItems>() ? 0 : AIObjectiveRepairItems.RequiredSuccessFactor;
70 float severity = isPriority ? 1 : AIObjectiveRepairItems.GetTargetPriority(Item, character, requiredSuccessFactor) / 100;
71 bool isSelected = IsRepairing();
72 float selectedBonus = isSelected ? 100 - MaxDevotion : 0;
73 float devotion = (CumulatedDevotion + selectedBonus) / 100;
74 float reduction = isPriority ? 1 : isSelected ? 2 : 3;
75 float max = AIObjectiveManager.LowestOrderPriority - reduction;
76 float highestWeight = -1;
77 foreach (Identifier tag in Item.Prefab.Tags)
78 {
79 if (JobPrefab.ItemRepairPriorities.TryGetValue(tag, out float weight) && weight > highestWeight)
80 {
81 highestWeight = weight;
82 }
83 }
84 if (highestWeight == -1)
85 {
86 // Predefined weight not found.
87 highestWeight = 1;
88 }
89 Priority = MathHelper.Lerp(0, max, MathHelper.Clamp(devotion + (severity * distanceFactor * highestWeight * PriorityModifier), 0, 1));
90 }
91 return Priority;
92 }

94 protected override bool CheckObjectiveSpecific()

96 IsCompleted = Item.IsFullCondition;
97 if (character.IsOnPlayerTeam && IsCompleted && IsRepairing())

104 protected override void Act(float deltaTime)
105 {
106 // Only continue when the get item sub objectives have been completed.
107 if (subObjectives.Any()) { return; }
108 foreach (Repairable repairable in Item.Repairables)
109 {
110 if (!repairable.HasRequiredItems(character, false))
111 {
112 //make sure we have all the items required to fix the target item
113 foreach (var kvp in repairable.requiredItems)
114 {
115 foreach (RelatedItem requiredItem in kvp.Value)
116 {
117 var getItemObjective = new AIObjectiveGetItem(character, requiredItem.Identifiers, objectiveManager, equip: true)
118 {
119 AllowVariants = requiredItem.AllowVariants
120 };
121 if (objectiveManager.IsCurrentOrder<AIObjectiveRepairItems>())
122 {
123 if (character.IsOnPlayerTeam)
124 {
125 getItemObjective.Abandoned += () => character.Speak(TextManager.Get("dialogcannotfindrequireditemtorepair").Value, null, 0.0f, "dialogcannotfindrequireditemtorepair".ToIdentifier(), 10.0f);
126 }
127 }
128 subObjectives.Add(getItemObjective);
129 }
130 }
131 return;
132 }
133 }
134 if (repairTool == null)
135 {
136 FindRepairTool();
137 }
138 if (repairTool != null)
139 {
140 if (repairTool.Item.OwnInventory == null)
141 {
142#if DEBUG
143 DebugConsole.ThrowError($"{character.Name}: AIObjectiveRepairItem failed - the item \"" + repairTool + "\" has no proper inventory");
144#endif
145 Abandon = true;
146 return;
147 }
148 RelatedItem item = null;
149 Item fuel = null;
150 foreach (RelatedItem requiredItem in repairTool.requiredItems[RelatedItem.RelationType.Contained])
151 {
152 item = requiredItem;
153 fuel = repairTool.Item.OwnInventory.AllItems.FirstOrDefault(it => it.Condition > 0.0f && requiredItem.MatchesItem(it));
154 if (fuel != null) { break; }
155 }
156 if (fuel == null)
157 {
158 RemoveSubObjective(ref goToObjective);
159 TryAddSubObjective(ref refuelObjective, () => new AIObjectiveContainItem(character, item.Identifiers, repairTool.Item.GetComponent<ItemContainer>(), objectiveManager, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC)
160 {
161 RemoveExisting = true
162 },
163 onCompleted: () => RemoveSubObjective(ref refuelObjective),
164 onAbandon: () => Abandon = true);
165 return;
166 }
167 }
168 if (!character.IsClimbing && character.CanInteractWith(Item, out _, checkLinked: false))
169 {
170 waitTimer += deltaTime;
171 if (waitTimer < WaitTimeBeforeRepair) { return; }
172
173 HumanAIController.FaceTarget(Item);
174 if (repairTool != null)
175 {
176 OperateRepairTool(deltaTime);
177 }
178 foreach (Repairable repairable in Item.Repairables)
179 {
180 if (repairable.CurrentFixer != null && repairable.CurrentFixer != character)
181 {
182 // Someone else is repairing the target. Abandon the objective if the other is better at this than us.
183 Abandon = repairable.CurrentFixer.IsPlayer || repairable.DegreeOfSuccess(character) < repairable.DegreeOfSuccess(repairable.CurrentFixer);
184 }
185 if (!Abandon)
186 {
187 if (character.SelectedConstruction != Item)
188 {
189 if (Item.TryInteract(character, ignoreRequiredItems: true, forceSelectKey: true) ||
190 Item.TryInteract(character, ignoreRequiredItems: true, forceUseKey: true))
191 {
192 character.SelectedConstruction = Item;
193 }
194 else
195 {
196 Abandon = true;
197 }
198 }
199 if (previousCondition == -1)
200 {
201 previousCondition = Item.Condition;
202 }
203 else if (Item.Condition < previousCondition)
204 {
205 // If the current condition is less than the previous condition, we can't complete the task, so let's abandon it. The item is probably deteriorating at a greater speed than we can repair it.
206 Abandon = true;
207 }
208 }
209 if (Abandon)
210 {
211 if (character.IsOnPlayerTeam && IsRepairing())
212 {
213 character.Speak(TextManager.GetWithVariable("DialogCannotRepair", "[itemname]", Item.Name, FormatCapitals.Yes).Value, null, 0.0f, "cannotrepair".ToIdentifier(), 10.0f);
214 }
215 repairable.StopRepairing(character);
216 }
217 else if (repairable.CurrentFixer != character)
218 {
219 repairable.StartRepairing(character, Repairable.FixActions.Repair);
220 }
221 break;
222 }
223 }
224 else
225 {
226 waitTimer = 0.0f;
227 RemoveSubObjective(ref refuelObjective);
228 // If cannot reach the item, approach it.
229 TryAddSubObjective(ref goToObjective,
230 constructor: () =>
231 {
232 previousCondition = -1;
233 var objective = new AIObjectiveGoTo(Item, character, objectiveManager)
234 {
235 // Don't stop in ladders, because we can't interact with other items while holding the ladders.
236 endNodeFilter = node => node.Waypoint.Ladders == null,
237 TargetName = Item.Name
238 };
239 if (repairTool != null)
240 {
241 objective.CloseEnough = repairTool.Range * 0.75f;
242 }
243 return objective;
244 },
245 onAbandon: () =>
246 {
247 Abandon = true;
248 if (character.IsOnPlayerTeam && IsRepairing())
249 {
250 character.Speak(TextManager.GetWithVariable("DialogCannotRepair", "[itemname]", Item.Name, FormatCapitals.Yes).Value, null, 0.0f, "cannotrepair".ToIdentifier(), 10.0f);
251 }
252 });
253 }
254 }

256 private void FindRepairTool()
257 {
258 foreach (Repairable repairable in Item.Repairables)
259 {
260 foreach (var kvp in repairable.requiredItems)
261 {
262 foreach (RelatedItem requiredItem in kvp.Value)
263 {
264 foreach (var item in character.Inventory.AllItems)
265 {
266 if (requiredItem.MatchesItem(item))
267 {
268 repairTool = item.GetComponent<RepairTool>();
269 }
270 }
271 }
272 }
273 }

276 private void OperateRepairTool(float deltaTime)
277 {
278 character.CursorPosition = Item.WorldPosition;
279 if (character.Submarine != null)
280 {
281 character.CursorPosition -= character.Submarine.Position;
282 }
283 if (repairTool.Item.RequireAimToUse)
284 {
285 character.SetInput(InputType.Aim, false, true);
286 }
287 Vector2 fromToolToTarget = Item.Position - repairTool.Item.Position;
288 if (fromToolToTarget.LengthSquared() < MathUtils.Pow(repairTool.Range / 2, 2))
289 {
290 // Too close -> steer away
291 character.AIController.SteeringManager.SteeringManual(deltaTime, Vector2.Normalize(character.SimPosition - Item.SimPosition) / 2);
292 }
293 else
294 {
295 character.AIController.SteeringManager.Reset();
296 }
297 if (VectorExtensions.Angle(VectorExtensions.Forward(repairTool.Item.body.TransformedRotation), fromToolToTarget) < MathHelper.PiOver4)
298 {
299 repairTool.Use(deltaTime, character);
300 }
