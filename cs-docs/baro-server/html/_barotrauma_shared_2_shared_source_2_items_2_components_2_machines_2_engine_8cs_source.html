98  public override void Update(float deltaTime, Camera cam)
99  {
100  UpdateOnActiveEffects(deltaTime);
101 
102  UpdateAnimation(deltaTime);
103 
104  controlLockTimer -= deltaTime;
105 
106  if (powerConsumption == 0.0f)
107  {
108  prevVoltage = 1;
109  hasPower = true;
110  }
111  else
112  {
113  hasPower = Voltage > MinVoltage;
114  }
115 
116 
117  Force = MathHelper.Lerp(force, (Voltage < MinVoltage) ? 0.0f : targetForce, deltaTime * 10.0f);
118  if (Math.Abs(Force) > 1.0f)
119  {
120  float voltageFactor = MinVoltage <= 0.0f ? 1.0f : Math.Min(Voltage, 1.0f);
121  float currForce = force * voltageFactor;
122  float condition = item.Condition / item.MaxCondition;
123  // Broken engine makes more noise.
124  float noise = Math.Abs(currForce) * MathHelper.Lerp(1.5f, 1f, condition);
125  UpdateAITargets(noise);
126  //arbitrary multiplier that was added to changes in submarine mass without having to readjust all engines
127  float forceMultiplier = 0.1f;
128  if (User != null)
129  {
130  forceMultiplier *= MathHelper.Lerp(0.5f, 2.0f, (float)Math.Sqrt(User.GetSkillLevel("helm") / 100));
131  }
132  currForce *= maxForce * forceMultiplier;
133  if (item.GetComponent<Repairable>() is Repairable repairable && repairable.IsTinkering)
134  {
135  currForce *= 1f + repairable.TinkeringStrength * TinkeringForceIncrease;
136  }
137 
138  //less effective when in a bad condition
139  currForce *= MathHelper.Lerp(0.5f, 2.0f, condition);
140  if (item.Submarine.FlippedX) { currForce *= -1; }
141  Vector2 forceVector = new Vector2(currForce, 0);
142  item.Submarine.ApplyForce(forceVector * deltaTime * Timing.FixedUpdateRate);
143  UpdatePropellerDamage(deltaTime);
144 #if CLIENT
145  float particleInterval = 1.0f / particlesPerSec;
146  particleTimer += deltaTime;
147  while (particleTimer > particleInterval)
148  {
149  Vector2 particleVel = -forceVector.ClampLength(5000.0f) / 5.0f;
150  GameMain.ParticleManager.CreateParticle("bubbles", item.WorldPosition + PropellerPos * item.Scale,
151  particleVel * Rand.Range(0.8f, 1.1f),
152  0.0f, item.CurrentHull);
153  particleTimer -= particleInterval;
154  }
155 #endif
156  }
157  }

199  private void UpdatePropellerDamage(float deltaTime)
200  {
201  if (DisablePropellerDamage) { return; }
202 
203  damageTimer += deltaTime;
204  if (damageTimer < 0.5f) { return; }
205  damageTimer = 0.1f;
206 
207  if (propellerDamage == null) { return; }
208 
209  float scaledDamageRange = propellerDamage.DamageRange * item.Scale;
210 
211  Vector2 propellerWorldPos = item.WorldPosition + PropellerPos * item.Scale;
212  float broadRange = Math.Max(scaledDamageRange * 2, 500);
213  foreach (Character character in Character.CharacterList)
214  {
215  if (!character.Enabled || character.Removed) { continue; }
216  if (Math.Abs(character.WorldPosition.X - propellerWorldPos.X) > broadRange) { continue; }
217  if (Math.Abs(character.WorldPosition.Y - propellerWorldPos.Y) > broadRange) { continue; }
218 
219  foreach (Limb limb in character.AnimController.Limbs)
220  {
221  if (limb.IsSevered || !limb.body.Enabled) { continue; }
222  float distSqr = Vector2.DistanceSquared(limb.WorldPosition, propellerWorldPos);
223  if (distSqr > scaledDamageRange * scaledDamageRange) { continue; }
224  character.LastDamageSource = item;
225  propellerDamage.DoDamage(null, character, propellerWorldPos, 1.0f, true);
226  break;
227  }
228  }
