68 public override void Close(string msg = null)
69 {
70 if (netServer == null) { return; }
71
72 for (int i = pendingClients.Count - 1; i >= 0; i--)
73 {
74 RemovePendingClient(pendingClients[i], DisconnectReason.ServerShutdown, msg);
75 }
76
77 for (int i = connectedClients.Count - 1; i >= 0; i--)
78 {
79 Disconnect(connectedClients[i], msg ?? DisconnectReason.ServerShutdown.ToString());
80 }
81
82 netServer.Shutdown(msg ?? DisconnectReason.ServerShutdown.ToString());
83
84 pendingClients.Clear();
85 connectedClients.Clear();
86
87 netServer = null;
88
89 Steamworks.SteamServer.OnValidateAuthTicketResponse -= OnAuthChange;
90
91 OnShutdown?.Invoke();
92 }

94 public override void Update(float deltaTime)
95 {
96 if (netServer == null) { return; }
97
98 if (OnOwnerDetermined != null && OwnerConnection != null)
99 {
100 OnOwnerDetermined?.Invoke(OwnerConnection);
101 OnOwnerDetermined = null;
102 }
103
104 netServer.ReadMessages(incomingLidgrenMessages);
105
106 //process incoming connections first
107 foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType == NetIncomingMessageType.ConnectionApproval))
108 {
109 HandleConnection(inc);
110 }
111
112 try
113 {
114 //after processing connections, go ahead with the rest of the messages
115 foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType != NetIncomingMessageType.ConnectionApproval))
116 {
117 switch (inc.MessageType)
118 {
119 case NetIncomingMessageType.Data:
120 HandleDataMessage(inc);
121 break;
122 case NetIncomingMessageType.StatusChanged:
123 HandleStatusChanged(inc);
124 break;
125 }
126 }
127 }
128
129 catch (Exception e)
130 {
131 string errorMsg = "Server failed to read an incoming message. {" + e + "}\n" + e.StackTrace.CleanupStackTrace();
132 GameAnalyticsManager.AddErrorEventOnce("LidgrenServerPeer.Update:ClientReadException" + e.TargetSite.ToString(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
133#if DEBUG
134 DebugConsole.ThrowError(errorMsg);
135#else
136 if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError(errorMsg); }
137#endif
138 }
139
140 for (int i = 0; i < pendingClients.Count; i++)
141 {
142 PendingClient pendingClient = pendingClients[i];
143
144 var connection = pendingClient.Connection as LidgrenConnection;
145 if (connection.NetConnection.Status == NetConnectionStatus.InitiatedConnect ||
146 connection.NetConnection.Status == NetConnectionStatus.ReceivedInitiation ||
147 connection.NetConnection.Status == NetConnectionStatus.RespondedAwaitingApproval ||
148 connection.NetConnection.Status == NetConnectionStatus.RespondedConnect)
149 {
150 continue;
151 }
152 UpdatePendingClient(pendingClient);
153 if (i >= pendingClients.Count || pendingClients[i] != pendingClient) { i--; }
154 }
155
156 incomingLidgrenMessages.Clear();
157 }

181 private void HandleConnection(NetIncomingMessage inc)
182 {
183 if (netServer == null) { return; }
184
185 var skipDeny = false;
186 {
187 var result = GameMain.LuaCs.Hook.Call<bool?>("lidgren.handleConnection", inc);
188 if (result != null) {
189 if (result.Value) skipDeny = true;
190 else return;
191 }
192 }
193
194 if (!skipDeny && connectedClients.Count >= serverSettings.MaxPlayers)
195 {
196 inc.SenderConnection.Deny(DisconnectReason.ServerFull.ToString());
197 return;
198 }
199
200 if (serverSettings.BanList.IsBanned(inc.SenderConnection.RemoteEndPoint.Address, 0, 0, out string banReason))
201 {
202 //IP banned: deny immediately
203 inc.SenderConnection.Deny(DisconnectReason.Banned.ToString() + "/ " + banReason);
204 return;
205 }
206
207 PendingClient pendingClient = pendingClients.Find(c => c.Connection is LidgrenConnection l && l.NetConnection == inc.SenderConnection);
208
209 if (pendingClient == null)
210 {
211 pendingClient = new PendingClient(new LidgrenConnection("PENDING", inc.SenderConnection, 0));
212 pendingClients.Add(pendingClient);
213 }
214
215 inc.SenderConnection.Approve();
216 }

218 private void HandleDataMessage(NetIncomingMessage inc)
219 {
220 if (netServer == null) { return; }
221
222 PendingClient pendingClient = pendingClients.Find(c => (c.Connection is LidgrenConnection l) && l.NetConnection == inc.SenderConnection);
223
224 PacketHeader packetHeader = (PacketHeader)inc.ReadByte();
225
226 if (packetHeader.IsConnectionInitializationStep() && pendingClient != null)
227 {
228 ReadConnectionInitializationStep(pendingClient, new ReadWriteMessage(inc.Data, (int)inc.Position, inc.LengthBits, false));
229 }
230 else if (!packetHeader.IsConnectionInitializationStep())
231 {
232 LidgrenConnection conn = connectedClients.Find(c => (c is LidgrenConnection l) && l.NetConnection == inc.SenderConnection) as LidgrenConnection;
233 if (conn == null)
234 {
235 if (pendingClient != null)
236 {
237 RemovePendingClient(pendingClient, DisconnectReason.AuthenticationRequired, "Received data message from unauthenticated client");
238 }
239 else if (inc.SenderConnection.Status != NetConnectionStatus.Disconnected &&
240 inc.SenderConnection.Status != NetConnectionStatus.Disconnecting)
241 {
242 inc.SenderConnection.Disconnect(DisconnectReason.AuthenticationRequired.ToString() + "/ Received data message from unauthenticated client");
243 }
244 return;
245 }
246 if (pendingClient != null) { pendingClients.Remove(pendingClient); }
247 if (serverSettings.BanList.IsBanned(conn.IPEndPoint.Address, conn.SteamID, conn.OwnerSteamID, out string banReason))
248 {
249 Disconnect(conn, DisconnectReason.Banned.ToString() + "/ " + banReason);
250 return;
251 }
252 UInt16 length = inc.ReadUInt16();
253
254 //DebugConsole.NewMessage(isCompressed + " " + isConnectionInitializationStep + " " + (int)incByte + " " + length);
255
256 IReadMessage msg = new ReadOnlyMessage(inc.Data, packetHeader.IsCompressed(), inc.PositionInBytes, length, conn);
257 OnMessageReceived?.Invoke(conn, msg);
258 }
259 }

261 private void HandleStatusChanged(NetIncomingMessage inc)
262 {
263 if (netServer == null) { return; }
264
265 switch (inc.SenderConnection.Status)
266 {
267 case NetConnectionStatus.Disconnected:
268 string disconnectMsg;
269 LidgrenConnection conn = connectedClients.Select(c => c as LidgrenConnection).FirstOrDefault(c => c.NetConnection == inc.SenderConnection);
270 if (conn != null)
271 {
272 if (conn == OwnerConnection)
273 {
274 DebugConsole.NewMessage("Owner disconnected: closing the server...");
275 GameServer.Log("Owner disconnected: closing the server...", ServerLog.MessageType.ServerMessage);
276 Close(DisconnectReason.ServerShutdown.ToString() + "/ Owner disconnected");
277 }
278 else
279 {
280 disconnectMsg = $"ServerMessage.HasDisconnected~[client]={conn.Name}";
281 Disconnect(conn, disconnectMsg);
282 }
283 }
284 else
285 {
286 PendingClient pendingClient = pendingClients.Find(c => (c.Connection is LidgrenConnection l) && l.NetConnection == inc.SenderConnection);
287 if (pendingClient != null)
288 {
289 RemovePendingClient(pendingClient, DisconnectReason.Unknown, $"ServerMessage.HasDisconnected~[client]={pendingClient.Name}");
290 }
291 }
292 break;
293 }

301 private void OnAuthChange(Steamworks.SteamId steamID, Steamworks.SteamId ownerID, Steamworks.AuthResponse status)
302 {
303 if (netServer == null) { return; }
304
305 PendingClient pendingClient = pendingClients.Find(c => c.SteamID == steamID);
306 DebugConsole.Log(steamID + " validation: " + status+", "+(pendingClient!=null));
307
308 if (pendingClient == null)
309 {
310 if (status != Steamworks.AuthResponse.OK)
311 {
312 LidgrenConnection connection = connectedClients.Find(c => c.SteamID == steamID) as LidgrenConnection;
313 if (connection != null)
314 {
315 Disconnect(connection, DisconnectReason.SteamAuthenticationFailed.ToString() + "/ Steam authentication status changed: " + status.ToString());
316 }
317 }
318 return;
319 }
320
321 LidgrenConnection pendingConnection = pendingClient.Connection as LidgrenConnection;
322 string banReason;
323 if (serverSettings.BanList.IsBanned(pendingConnection.NetConnection.RemoteEndPoint.Address, steamID, ownerID, out banReason))
324 {
325 RemovePendingClient(pendingClient, DisconnectReason.Banned, banReason);
326 return;
327 }
328
329 if (status == Steamworks.AuthResponse.OK)
330 {
331 pendingClient.OwnerSteamID = ownerID;
332 pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
333 pendingClient.UpdateTime = Timing.TotalTime;
334 }
335 else
336 {
337 RemovePendingClient(pendingClient, DisconnectReason.SteamAuthenticationFailed, "Steam authentication failed: " + status.ToString());
338 return;
339 }

342 public override void Send(IWriteMessage msg, NetworkConnection conn, DeliveryMethod deliveryMethod, bool compressPastThreshold = true)
343 {
344 if (netServer == null) { return; }
345
346 if (!(conn is LidgrenConnection lidgrenConn)) return;
347 if (!connectedClients.Contains(lidgrenConn))
348 {
349 DebugConsole.ThrowError("Tried to send message to unauthenticated connection: " + lidgrenConn.IPString);
350 return;
351 }
352
353 NetDeliveryMethod lidgrenDeliveryMethod = NetDeliveryMethod.Unreliable;
354 switch (deliveryMethod)
355 {
356 case DeliveryMethod.Unreliable:
357 lidgrenDeliveryMethod = NetDeliveryMethod.Unreliable;
358 break;
359 case DeliveryMethod.Reliable:
360 lidgrenDeliveryMethod = NetDeliveryMethod.ReliableUnordered;
361 break;
362 case DeliveryMethod.ReliableOrdered:
363 lidgrenDeliveryMethod = NetDeliveryMethod.ReliableOrdered;
364 break;
365 }
366
367#if DEBUG
368 netPeerConfiguration.SimulatedDuplicatesChance = GameMain.Server.SimulatedDuplicatesChance;
369 netPeerConfiguration.SimulatedMinimumLatency = GameMain.Server.SimulatedMinimumLatency;
370 netPeerConfiguration.SimulatedRandomLatency = GameMain.Server.SimulatedRandomLatency;
371 netPeerConfiguration.SimulatedLoss = GameMain.Server.SimulatedLoss;
372#endif
373
374 NetOutgoingMessage lidgrenMsg = netServer.CreateMessage();
375 byte[] msgData = new byte[msg.LengthBytes];
376 msg.PrepareForSending(ref msgData, compressPastThreshold, out bool isCompressed, out int length);
377 lidgrenMsg.Write((byte)(isCompressed ? PacketHeader.IsCompressed : PacketHeader.None));
378 lidgrenMsg.Write((UInt16)length);
379 lidgrenMsg.Write(msgData, 0, length);
380
381 NetSendResult result = netServer.SendMessage(lidgrenMsg, lidgrenConn.NetConnection, lidgrenDeliveryMethod);
382 if (result != NetSendResult.Sent && result != NetSendResult.Queued)
383 {
384 DebugConsole.NewMessage("Failed to send message to "+conn.Name+": " + result.ToString(), Microsoft.Xna.Framework.Color.Yellow);
385 }
386 }

441 protected override void ProcessAuthTicket(string name, int ownKey, ulong steamId, PendingClient pendingClient, byte[] ticket)
442 {
443 if (pendingClient.SteamID == null)
444 {
445 bool requireSteamAuth = GameSettings.CurrentConfig.RequireSteamAuthentication;
446#if DEBUG
447 requireSteamAuth = false;
448#endif
449
450 //steam auth cannot be done (SteamManager not initialized or no ticket given),
451 //but it's not required either -> let the client join without auth
452 if ((!Steam.SteamManager.IsInitialized || (ticket?.Length ?? 0) == 0) &&
453 !requireSteamAuth)
454 {
455 pendingClient.Connection.Name = name;
456 pendingClient.Name = name;
457 pendingClient.OwnerKey = ownKey;
458 pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
459 }
460 else
461 {
462 Steamworks.BeginAuthResult authSessionStartState = Steam.SteamManager.StartAuthSession(ticket, steamId);
463 if (authSessionStartState != Steamworks.BeginAuthResult.OK)
464 {
465 if (requireSteamAuth)
466 {
467 RemovePendingClient(pendingClient, DisconnectReason.SteamAuthenticationFailed, "Steam auth session failed to start: " + authSessionStartState.ToString());
468 return;
469 }
470 else
471 {
472 steamId = 0;
473 pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
474 }
475 }
476 pendingClient.SteamID = steamId;
477 pendingClient.Connection.Name = name;
478 pendingClient.Name = name;
479 pendingClient.OwnerKey = ownKey;
480 pendingClient.AuthSessionStarted = true;
481 }
482 }
483 else
484 {
485 if (pendingClient.SteamID != steamId)
486 {
487 RemovePendingClient(pendingClient, DisconnectReason.SteamAuthenticationFailed, "SteamID mismatch");
488 return;
489 }
490 }
491 }
