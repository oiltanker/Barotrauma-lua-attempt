94  public override void Update(float deltaTime)
95  {
96  if (netServer == null) { return; }
97 
98  if (OnOwnerDetermined != null && OwnerConnection != null)
99  {
100  OnOwnerDetermined?.Invoke(OwnerConnection);
101  OnOwnerDetermined = null;
102  }
103 
104  netServer.ReadMessages(incomingLidgrenMessages);
105 
106  //process incoming connections first
107  foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType == NetIncomingMessageType.ConnectionApproval))
108  {
109  HandleConnection(inc);
110  }
111 
112  try
113  {
114  //after processing connections, go ahead with the rest of the messages
115  foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType != NetIncomingMessageType.ConnectionApproval))
116  {
117  switch (inc.MessageType)
118  {
119  case NetIncomingMessageType.Data:
120  HandleDataMessage(inc);
121  break;
122  case NetIncomingMessageType.StatusChanged:
123  HandleStatusChanged(inc);
124  break;
125  }
126  }
127  }
128 
129  catch (Exception e)
130  {
131  string errorMsg = "Server failed to read an incoming message. {" + e + "}\n" + e.StackTrace.CleanupStackTrace();
132  GameAnalyticsManager.AddErrorEventOnce("LidgrenServerPeer.Update:ClientReadException" + e.TargetSite.ToString(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
133 #if DEBUG
134  DebugConsole.ThrowError(errorMsg);
135 #else
136  if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError(errorMsg); }
137 #endif
138  }
139 
140  for (int i = 0; i < pendingClients.Count; i++)
141  {
142  PendingClient pendingClient = pendingClients[i];
143 
144  var connection = pendingClient.Connection as LidgrenConnection;
145  if (connection.NetConnection.Status == NetConnectionStatus.InitiatedConnect ||
146  connection.NetConnection.Status == NetConnectionStatus.ReceivedInitiation ||
147  connection.NetConnection.Status == NetConnectionStatus.RespondedAwaitingApproval ||
148  connection.NetConnection.Status == NetConnectionStatus.RespondedConnect)
149  {
150  continue;
151  }
152  UpdatePendingClient(pendingClient);
153  if (i >= pendingClients.Count || pendingClients[i] != pendingClient) { i--; }
154  }
155 
156  incomingLidgrenMessages.Clear();
157  }

218  private void HandleDataMessage(NetIncomingMessage inc)
219  {
220  if (netServer == null) { return; }
221 
222  PendingClient pendingClient = pendingClients.Find(c => (c.Connection is LidgrenConnection l) && l.NetConnection == inc.SenderConnection);
223 
224  PacketHeader packetHeader = (PacketHeader)inc.ReadByte();
225 
226  if (packetHeader.IsConnectionInitializationStep() && pendingClient != null)
227  {
228  ReadConnectionInitializationStep(pendingClient, new ReadWriteMessage(inc.Data, (int)inc.Position, inc.LengthBits, false));
229  }
230  else if (!packetHeader.IsConnectionInitializationStep())
231  {
232  LidgrenConnection conn = connectedClients.Find(c => (c is LidgrenConnection l) && l.NetConnection == inc.SenderConnection) as LidgrenConnection;
233  if (conn == null)
234  {
235  if (pendingClient != null)
236  {
237  RemovePendingClient(pendingClient, DisconnectReason.AuthenticationRequired, "Received data message from unauthenticated client");
238  }
239  else if (inc.SenderConnection.Status != NetConnectionStatus.Disconnected &&
240  inc.SenderConnection.Status != NetConnectionStatus.Disconnecting)
241  {
242  inc.SenderConnection.Disconnect(DisconnectReason.AuthenticationRequired.ToString() + "/ Received data message from unauthenticated client");
243  }
244  return;
245  }
246  if (pendingClient != null) { pendingClients.Remove(pendingClient); }
247  if (serverSettings.BanList.IsBanned(conn.IPEndPoint.Address, conn.SteamID, conn.OwnerSteamID, out string banReason))
248  {
249  Disconnect(conn, DisconnectReason.Banned.ToString() + "/ " + banReason);
250  return;
251  }
252  UInt16 length = inc.ReadUInt16();
253 
254  //DebugConsole.NewMessage(isCompressed + " " + isConnectionInitializationStep + " " + (int)incByte + " " + length);
255 
256  IReadMessage msg = new ReadOnlyMessage(inc.Data, packetHeader.IsCompressed(), inc.PositionInBytes, length, conn);
257  OnMessageReceived?.Invoke(conn, msg);
258  }
259  }

287  if (pendingClient != null)

342  public override void Send(IWriteMessage msg, NetworkConnection conn, DeliveryMethod deliveryMethod, bool compressPastThreshold = true)
343  {
344  if (netServer == null) { return; }
345 
346  if (!(conn is LidgrenConnection lidgrenConn)) return;
347  if (!connectedClients.Contains(lidgrenConn))
348  {
349  DebugConsole.ThrowError("Tried to send message to unauthenticated connection: " + lidgrenConn.IPString);
350  return;
351  }
352 
353  NetDeliveryMethod lidgrenDeliveryMethod = NetDeliveryMethod.Unreliable;
354  switch (deliveryMethod)
355  {
356  case DeliveryMethod.Unreliable:
357  lidgrenDeliveryMethod = NetDeliveryMethod.Unreliable;
358  break;
359  case DeliveryMethod.Reliable:
360  lidgrenDeliveryMethod = NetDeliveryMethod.ReliableUnordered;
361  break;
362  case DeliveryMethod.ReliableOrdered:
363  lidgrenDeliveryMethod = NetDeliveryMethod.ReliableOrdered;
364  break;
365  }
366 
367 #if DEBUG
368  netPeerConfiguration.SimulatedDuplicatesChance = GameMain.Server.SimulatedDuplicatesChance;
369  netPeerConfiguration.SimulatedMinimumLatency = GameMain.Server.SimulatedMinimumLatency;
370  netPeerConfiguration.SimulatedRandomLatency = GameMain.Server.SimulatedRandomLatency;
371  netPeerConfiguration.SimulatedLoss = GameMain.Server.SimulatedLoss;
372 #endif
373 
374  NetOutgoingMessage lidgrenMsg = netServer.CreateMessage();
375  byte[] msgData = new byte[msg.LengthBytes];
376  msg.PrepareForSending(ref msgData, compressPastThreshold, out bool isCompressed, out int length);
377  lidgrenMsg.Write((byte)(isCompressed ? PacketHeader.IsCompressed : PacketHeader.None));
378  lidgrenMsg.Write((UInt16)length);
379  lidgrenMsg.Write(msgData, 0, length);
380 
381  NetSendResult result = netServer.SendMessage(lidgrenMsg, lidgrenConn.NetConnection, lidgrenDeliveryMethod);
382  if (result != NetSendResult.Sent && result != NetSendResult.Queued)
383  {
384  DebugConsole.NewMessage("Failed to send message to "+conn.Name+": " + result.ToString(), Microsoft.Xna.Framework.Color.Yellow);
385  }

442  {
