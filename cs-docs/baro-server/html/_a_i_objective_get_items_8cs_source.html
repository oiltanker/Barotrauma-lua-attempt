53 int count = gearTags.Count(t => t == tag);
54 AIObjectiveGetItem? getItem = null;
55 TryAddSubObjective(ref getItem, () =>
56 new AIObjectiveGetItem(character, tag, objectiveManager, Equip, CheckInventory && count <= 1)
57 {

59 Wear = Wear,

62 ignoredIdentifiersOrTags = ignoredTags,
63 CheckPathForEachItem = CheckPathForEachItem,
64 RequireLoaded = RequireLoaded,
65 ItemCount = count,
66 SpeakIfFails = RequireAllItems
67 },
68 onCompleted: () =>
69 {

71 if (item?.IsOwnedBy(character) != null)
72 {
73 achievedItems.Add(item);
74 }
75 },
76 onAbandon: () =>
77 {
78 var item = getItem?.TargetItem;
79 if (item != null)
80 {
81 achievedItems.Remove(item);
82 }

87 }
88 });
