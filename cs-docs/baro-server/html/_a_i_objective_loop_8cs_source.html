54  {
55  base.Update(deltaTime);
56  if (IgnoreListClearInterval > 0)
57  {
58  if (ignoreListTimer > IgnoreListClearInterval)

75  if (targetUpdateTimer <= 0)

77  UpdateTargets();
78  }
79  else
80  {

84  {

98  }
99  }

101  }

119  protected override float GetPriority()
120  {
121  if (!IsAllowed)
122  {
123  Priority = 0;
124  return Priority;
125  }
126  if (character.LockHands)
127  {
128  Priority = 0;
129  }
130  else
131  {
132  // Allow the target value to be more than 100.
133  float targetValue = TargetEvaluation();
134  if (InverseTargetEvaluation)
135  {
136  targetValue = 100 - targetValue;
137  }
138  var currentSubObjective = CurrentSubObjective;
139  if (currentSubObjective != null && currentSubObjective.Priority > targetValue)
140  {
141  // If the priority is higher than the target value, let's just use it.
142  // The priority calculation is more precise, but it takes into account things like distances,
143  // so it's better not to use it if it's lower than the rougher targetValue.
144  targetValue = currentSubObjective.Priority;
145  }
146  // If the target value is less than 1% of the max value, let's just treat it as zero.
147  if (targetValue < 1)
148  {
149  Priority = 0;
150  }
151  else
152  {
153  if (objectiveManager.IsOrder(this))
154  {
155  Priority = ForceOrderPriority ? objectiveManager.GetOrderPriority(this) : targetValue;
156  }
157  else
158  {
159  float max = AIObjectiveManager.LowestOrderPriority - 1;
160  float value = MathHelper.Clamp((CumulatedDevotion + (targetValue * PriorityModifier)) / 100, 0, 1);
161  Priority = MathHelper.Lerp(0, max, value);
162  }
163  }
164  }
165  return Priority;
166  }

176  protected virtual void FindTargets()
177  {
178  foreach (T target in GetList())
179  {
180  // The bots always find targets when the objective is an order.
181  if (!objectiveManager.IsOrder(this))
182  {
183  // Battery or pump states cannot currently be reported (not implemented) and therefore we must ignore them -> the bots always know if they require attention.
184  bool ignore = this is AIObjectiveChargeBatteries || this is AIObjectivePumpWater;
185  if (!ignore && !ReportedTargets.Contains(target)) { continue; }
186  }
187  if (!Filter(target)) { continue; }
188  if (!ignoreList.Contains(target))
189  {
190  Targets.Add(target);
191  }
192  }
193  }

195  protected virtual void CreateObjectives()
196  {
197  foreach (T target in Targets)
198  {
199  if (ignoreList.Contains(target)) { continue; }
200  if (!Objectives.TryGetValue(target, out AIObjective objective))
201  {
202  objective = ObjectiveConstructor(target);
203  Objectives.Add(target, objective);
204  if (!subObjectives.Contains(objective))
205  {
206  subObjectives.Add(objective);
207  }
208  objective.Completed += () =>
209  {
210  Objectives.Remove(target);
211  OnObjectiveCompleted(objective, target);
212  };
213  objective.Abandoned += () =>
214  {
215  Objectives.Remove(target);
216  ignoreList.Add(target);
217  targetUpdateTimer = Math.Min(0.1f, targetUpdateTimer);
218  };
219  }
220  }
