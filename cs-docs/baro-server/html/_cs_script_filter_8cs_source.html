40 public static bool IsTypeAllowed(string name)
41 {
42 var matchPermitted = typesPermitted.Where(s => name.StartsWith(s));
43 var longestPemitted = matchPermitted.Count() > 0 ? matchPermitted.Max(s => s.Length) : 0;
44 var matchProhibited = typesProhibited.Where(s => name.StartsWith(s));
45 var longestProhibited = matchProhibited.Count() > 0 ? matchProhibited.Max(s => s.Length) : 0;
46 if (longestPemitted == 0 || longestPemitted < longestProhibited) return false;
47 else return true;
48 }

50 public static string FilterSyntaxTree(CSharpSyntaxTree tree)
51 {
52 if (tree == null) throw new ArgumentNullException("Syntax tree must not be null.");
53
54 { // Disallow top-level statements
55 var nodeCheck = tree.GetRoot().DescendantNodes();
56
57 var tlStatements = nodeCheck.Where(n => n is GlobalStatementSyntax).ToList();
58 if (tlStatements.Count > 0)
59 {
60 string errStr = "Compilation Error:";
61 foreach (var tls in tlStatements) tls.GetDiagnostics().ToList().ForEach(d => errStr += $"\n {d.ToString()}");
62 return errStr;
63 }
64 }
65
66 return null;
67 }

69 private static string ResolveTypeRef(MetadataReader reader, TypeReferenceHandle t)
70 {
71 var tRef = reader.GetTypeReference(t);
72
73 var typeName = $"{reader.GetString(tRef.Name)}";
74 EntityHandle handle = tRef.ResolutionScope;
75 TypeReference tr = tRef;
76 while (!handle.IsNil && handle.Kind == HandleKind.TypeReference)
77 {
78 tr = reader.GetTypeReference((TypeReferenceHandle)handle);
79 handle = tr.ResolutionScope;
80 typeName = $"{reader.GetString(tr.Name)}.{typeName}";
81 }
82 typeName = $"{reader.GetString(tr.Namespace)}.{typeName}";
83
84 return typeName;
85 }

87 public static string FilterMetadata(MetadataReader reader)
88 {
89 if (reader == null) throw new ArgumentNullException("Metadata Reader must not be null.");
90
91 var conflictingTypes = new List<string>();
92 reader.TypeReferences.ToList().ForEach(t =>
93 {
94 var typeName = ResolveTypeRef(reader, t);
95
96 if (!IsTypeAllowed(typeName)) conflictingTypes.Add(typeName);
97 });
98
99 if (conflictingTypes.Count > 0)
100 {
101 string errStr = "Metadata Error:";
102 conflictingTypes.ForEach(t => errStr += $"\n Usage of type '{t}' in mods is prohibited.");
103 return errStr;
104 }
105
106 return null;
107 }

117 public static string FilterOneTimeMetadata(MetadataReader reader)
118 {
119 var errStr = FilterMetadata(reader);
120 if (errStr != null) return errStr;
121
122 TypeDefinition? runDef = null;
123 reader.TypeDefinitions.Select(t => reader.GetTypeDefinition(t)).ToList().ForEach(t =>
124 {
125 var typeName = $"{reader.GetString(t.Name)}";
126 if (typeName == "NetOneTimeScriptRunner") runDef = t;
127 while (t.IsNested)
128 {
129 t = reader.GetTypeDefinition(t.GetDeclaringType());
130 typeName = $"{reader.GetString(t.Name)}.{typeName}";
131 }
132 typeName = $"{reader.GetString(t.Namespace)}.{typeName}";
133 if (!permitedDefinitions.Contains(typeName)) errStr = "Malformed assembly";
134 });
135 if (errStr != null) return errStr;
136
137 if (runDef == null) return "runner class not detected";
138 else
139 {
140 var methods = runDef.Value.GetMethods();
141 if (methods.Count > 2) return "malformed runner class";
142
143 methods.Select(m => reader.GetMethodDefinition(m)).ToList().ForEach(m => {
144 if (!permitedMethods.Contains(reader.GetString(m.Name))) errStr = "malformed runner class";
145 });
146 if (errStr != null) return errStr;
147 }
148
149 return null;
150 }
