144 public bool CanReceive(WifiComponent sender)
145 {
146 if (sender == null || sender.channel != channel) { return false; }
147 if (sender.TeamID != TeamID && !AllowCrossTeamCommunication) { return false; }
148
149 //if the component is not linked to chat and has nothing connected to the output, sending a signal to it does nothing
150 // = no point in receiving
151 if (!LinkToChat)
152 {
153 if (signalOutConnection == null || !signalOutConnection.Wires.Any(w => w != null))
154 {
155 return false;
156 }
157 }
158
159 if (Vector2.DistanceSquared(item.WorldPosition, sender.item.WorldPosition) > sender.range * sender.range) { return false; }

161 return HasRequiredContainedItems(user: null, addMessage: false);

206 public void TransmitSignal(Signal signal, bool sentFromChat)
207 {
208 var should = GameMain.LuaCs.Hook.Call<bool?>("wifiSignalTransmitted", this, signal, sentFromChat);
209
210 if (should != null && should.Value)
211 return;
212
213 if (sentFromChat)
214 {
215 item.LastSentSignalRecipients.Clear();
216 }
217
218 bool chatMsgSent = false;
219
220 var receivers = GetReceiversInRange();
221 foreach (WifiComponent wifiComp in receivers)
222 {
223 if (sentFromChat && !wifiComp.LinkToChat) { continue; }
224
225 //signal strength diminishes by distance
226 float sentSignalStrength = signal.strength *
227 MathHelper.Clamp(1.0f - (Vector2.Distance(item.WorldPosition, wifiComp.item.WorldPosition) / wifiComp.range), 0.0f, 1.0f);
228 Signal s = new Signal(signal.value, signal.stepsTaken + 1, sender: signal.sender, source: signal.source,
229 power: 0.0f, strength: sentSignalStrength);
230
231 if (wifiComp.signalOutConnection != null)
232 {
233 if (signal.source != null && wifiComp.signalInConnection != null)
234 {
235 if (signal.source.LastSentSignalRecipients.Contains(wifiComp.signalInConnection))
236 {
237 //signal already passed through this wifi component -> stop here to prevent an infinite loop
238 continue;
239 }
240 else
241 {
242 signal.source.LastSentSignalRecipients.Add(wifiComp.signalInConnection);
243 }
244 }
245 wifiComp.item.SendSignal(s, wifiComp.signalOutConnection);
246 }
247
248 if (signal.source != null)
249 {
250 foreach (Connection receiver in wifiComp.item.LastSentSignalRecipients)
251 {
252 if (!signal.source.LastSentSignalRecipients.Contains(receiver))
253 {
254 signal.source.LastSentSignalRecipients.Add(receiver);
255 }
256 }
257 }
258
259 if (DiscardDuplicateChatMessages && signal.value == prevSignal) { continue; }
260
261 //create a chat message
262 if (LinkToChat && wifiComp.LinkToChat && chatMsgCooldown <= 0.0f && !sentFromChat)
263 {
264 if (wifiComp.item.ParentInventory != null &&
265 wifiComp.item.ParentInventory.Owner != null)
266 {
267 string chatMsg = signal.value;
268 if (sentSignalStrength <= 1.0f)
269 {
270 chatMsg = ChatMessage.ApplyDistanceEffect(chatMsg, 1.0f - sentSignalStrength);
271 }
272 if (chatMsg.Length > ChatMessage.MaxLength) { chatMsg = chatMsg.Substring(0, ChatMessage.MaxLength); }
273 if (string.IsNullOrEmpty(chatMsg)) { continue; }
274
275#if CLIENT
276 if (wifiComp.item.ParentInventory.Owner == Character.Controlled)
277 {
278 if (GameMain.Client == null)
279 {
280 GameMain.GameSession?.CrewManager?.AddSinglePlayerChatMessage(signal.source?.Name ?? "", signal.value, ChatMessageType.Radio, sender: null);
281 }
282 }
283#elif SERVER
284 if (GameMain.Server != null)
285 {
286 Client recipientClient = GameMain.Server.ConnectedClients.Find(c => c.Character == wifiComp.item.ParentInventory.Owner);
287 if (recipientClient != null)
288 {
289 GameMain.Server.SendDirectChatMessage(
290 ChatMessage.Create(signal.source?.Name ?? "", chatMsg, ChatMessageType.Radio, null), recipientClient);
291 }
292 }
293#endif
294 chatMsgSent = true;
295 }
296 }
297 }
298 if (chatMsgSent)
299 {
300 chatMsgCooldown = MinChatMessageInterval;
301 IsActive = true;
302 }
303
304 prevSignal = signal.value;
305 }

307 public override void ReceiveSignal(Signal signal, Connection connection)
308 {
309 if (connection == null) { return; }
310
311 switch (connection.Name)
312 {
313 case "signal_in":
314 TransmitSignal(signal, false);
315 break;
316 case "set_channel":
317 if (int.TryParse(signal.value, out int newChannel))
318 {
319 int prevChannel = Channel;
320 Channel = newChannel;
321 if (prevChannel != Channel)
322 {
323#if SERVER
324 item.CreateServerEvent(this);
325#endif
326 }
327 }
328 break;
329 case "set_range":
330 if (float.TryParse(signal.value, NumberStyles.Float, CultureInfo.InvariantCulture, out float newRange))
331 {
332 Range = newRange;
333 }
334 break;
335 }
