67 public override void Update(float deltaTime, Camera cam)
68 {
69 if (signalQueue.Count == 0)
70 {
71 IsActive = false;
72 return;
73 }
74
75 foreach (var val in signalQueue)
76 {
77 val.SendTimer -= 1;
78 }
79 while (signalQueue.Count > 0 && signalQueue.Peek().SendTimer <= 0)
80 {
81 var signalOut = signalQueue.Peek();
82 signalOut.SendDuration -= 1;
83 item.SendSignal(new Signal(signalOut.Signal.value, sender: signalOut.Signal.sender, strength: signalOut.Signal.strength), "signal_out");
84 if (signalOut.SendDuration <= 0)
85 {
86 signalQueue.Dequeue();
87 }
88 else
89 {
90 break;
91 }
92 }
93 }

95 public override void ReceiveSignal(Signal signal, Connection connection)
96 {
97 switch (connection.Name)
98 {
99 case "signal_in":
100 if (signalQueue.Count >= signalQueueSize) { return; }
101 if (ResetWhenSignalReceived) { prevQueuedSignal = null; signalQueue.Clear(); }
102 if (ResetWhenDifferentSignalReceived && signalQueue.Count > 0 && signalQueue.Peek().Signal.value != signal.value)
103 {
104 prevQueuedSignal = null;
105 signalQueue.Clear();
106 }
107
108 if (prevQueuedSignal != null &&
109 prevQueuedSignal.Signal.value == signal.value &&
110 MathUtils.NearlyEqual(prevQueuedSignal.Signal.strength, signal.strength) &&
111 ((prevQueuedSignal.SendTimer + prevQueuedSignal.SendDuration == delayTicks) || (prevQueuedSignal.SendTimer <= 0 && prevQueuedSignal.SendDuration > 0)))
112 {
113 prevQueuedSignal.SendDuration += 1;
114 return;
115 }
116
117 prevQueuedSignal = new DelayedSignal(signal, delayTicks)
118 {
119 SendDuration = 1
120 };
121 signalQueue.Enqueue(prevQueuedSignal);
122 IsActive = true;
123 break;
124 case "set_delay":
125 if (float.TryParse(signal.value, NumberStyles.Any, CultureInfo.InvariantCulture, out float newDelay))
126 {
127 newDelay = MathHelper.Clamp(newDelay, 0, 60);
128 if (signalQueue.Count > 0 && newDelay != Delay)
129 {
130 prevQueuedSignal = null;
131 signalQueue.Clear();
132 }
133 Delay = newDelay;
134 }
135 break;
136 }
137 }
