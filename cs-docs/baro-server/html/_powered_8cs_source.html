294 public static void UpdateGrids(bool useCache = true)
295 {
296 //don't use cache if there are no existing grids
297 if (Grids.Count > 0 && useCache)
298 {
299 //delete all grids that were affected
300 foreach (Connection c in ChangedConnections)
301 {
302 if (c.Grid != null)
303 {
304 Grids.Remove(c.Grid.ID);
305 c.Grid = null;
306 }
307 }
308
309 foreach (Connection c in ChangedConnections)
310 {
311 //Make sure the connection grid hasn't been resolved by another connection update
312 //Ensure the connection has other connections
313 if (c.Grid == null && c.Recipients.Count > 0 && c.Item.Condition > 0.0f)
314 {
315 GridInfo grid = PropagateGrid(c);
316 Grids[grid.ID] = grid;
317 }
318 }
319 }
320 else
321 {
322 //Clear all grid IDs from connections
323 foreach (Powered powered in poweredList)
324 {
325 //Only check devices with connectors
326 if (powered.powerIn != null)
327 {
328 powered.powerIn.Grid = null;
329 }
330 if (powered.powerOut != null)
331 {
332 powered.powerOut.Grid = null;
333 }
334 }
335
336 Grids.Clear();
337
338 foreach (Powered powered in poweredList)
339 {
340 //Probe through all connections that don't have a gridID
341 if (powered.powerIn != null && powered.powerIn.Grid == null && powered.powerIn != powered.powerOut && powered.Item.Condition > 0.0f)
342 {
343 // Only create grids for networks with more than 1 device
344 if (powered.powerIn.Recipients.Count > 0)
345 {
346 GridInfo grid = PropagateGrid(powered.powerIn);
347 Grids[grid.ID] = grid;
348 }
349 }
350
351 if (powered.powerOut != null && powered.powerOut.Grid == null && powered.Item.Condition > 0.0f)
352 {
353 //Only create grids for networks with more than 1 device
354 if (powered.powerOut.Recipients.Count > 0)
355 {
356 GridInfo grid = PropagateGrid(powered.powerOut);
357 Grids[grid.ID] = grid;
358 }
359 }
360 }
361 }
362
363 //Clear changed connections after each update
364 ChangedConnections.Clear();
365 }

367 private static GridInfo PropagateGrid(Connection conn)
368 {
369 //Generate unique Key
370 int id = Rand.Int(int.MaxValue, Rand.RandSync.Unsynced);
371 while (Grids.ContainsKey(id))
372 {
373 id = Rand.Int(int.MaxValue, Rand.RandSync.Unsynced);
374 }
375
376 return PropagateGrid(conn, id);
377 }

379 private static GridInfo PropagateGrid(Connection conn, int gridID)
380 {
381 Stack<Connection> probeStack = new Stack<Connection>();
382
383 GridInfo grid = new GridInfo(gridID);
384
385 probeStack.Push(conn);
386
387 //Non recursive approach to traversing connection tree
388 while (probeStack.Count > 0)
389 {
390 Connection c = probeStack.Pop();
391 c.Grid = grid;
392 grid.AddConnection(c);
393
394 //Add on recipients
395 foreach (Connection otherC in c.Recipients)
396 {
397 //Only add valid connections
398 if (otherC.Grid != grid && (otherC.Grid == null || !Grids.ContainsKey(otherC.Grid.ID)) && ValidPowerConnection(c, otherC))
399 {
400 otherC.Grid = grid; //Assigning ID early prevents unncessary adding to stack
401 probeStack.Push(otherC);
402 }
403 }
404 }
405
406 return grid;
407 }

424 public static void UpdatePower(float deltaTime)
425 {
426 //Don't update the power if the round is ending
427 if (GameMain.GameSession != null && GameMain.GameSession.RoundEnding)
428 {
429 return;
430 }
431
432 //Only update the power at the given update interval
433 /*
434 //Not use currently as update interval of 1/60
435 if (updateTimer > 0.0f)
436 {
437 updateTimer -= deltaTime;
438 return;
439 }
440 updateTimer = UpdateInterval;
441 */
442
443#if CLIENT
444 System.Diagnostics.Stopwatch sw = new System.Diagnostics.Stopwatch();
445 sw.Start();
446#endif
447 //Ensure all grids are updated correctly and have the correct connections
448 UpdateGrids();
449
450#if CLIENT
451 sw.Stop();
452 GameMain.PerformanceCounter.AddElapsedTicks("GridUpdate", sw.ElapsedTicks);
453 sw.Restart();
454#endif
455
456 //Reset all grids
457 foreach (GridInfo grid in Grids.Values)
458 {
459 //Wipe priority groups as connections can change to not be outputting -- Can be improved caching wise --
460 grid.PowerSourceGroups.Clear();
461 grid.Power = 0;
462 grid.Load = 0;
463 }
464
465 //Determine if devices are adding a load or providing power, also resolve solo nodes
466 foreach (Powered powered in poweredList)
467 {
468 //Make voltage decay to ensure the device powers down.
469 //This only effects devices with no power input (whose voltage is set by other means, e.g. status effects from a contained battery)
470 //or devices that have been disconnected from the power grid - other devices use the voltage of the grid instead.
471 powered.Voltage -= deltaTime;
472
473 //Handle the device if it's got a power connection
474 if (powered.powerIn != null && powered.powerOut != powered.powerIn)
475 {
476 //Get the new load for the connection
477 float currLoad = powered.GetCurrentPowerConsumption(powered.powerIn);
478
479 //If its a load update its grid load
480 if (currLoad >= 0)
481 {
482 if (powered.PoweredByTinkering) { currLoad = 0.0f; }
483 powered.CurrPowerConsumption = currLoad;
484 if (powered.powerIn.Grid != null)
485 {
486 powered.powerIn.Grid.Load += currLoad;
487 }
488 }
489 else if (powered.powerIn.Grid != null)
490 {
491 //If connected to a grid add as a source to be processed
492 powered.powerIn.Grid.AddSrc(powered.powerIn);
493 }
494 else
495 {
496 powered.CurrPowerConsumption = -powered.GetConnectionPowerOut(powered.powerIn, 0, powered.MinMaxPowerOut(powered.powerIn, 0), 0);
497 powered.GridResolved(powered.powerIn);
498 }
499 }
500
501 //Handle the device power depending on if its powerout
502 if (powered.powerOut != null)
503 {
504 //Get the connection's load
505 float currLoad = powered.GetCurrentPowerConsumption(powered.powerOut);
506
507 //Update the device's output load to the correct variable
508 if (powered is PowerTransfer pt)
509 {
510 pt.PowerLoad = currLoad;
511 }
512 else if (powered is PowerContainer pc)
513 {
514 // PowerContainer handle its own output value
515 }
516 else
517 {
518 powered.CurrPowerConsumption = currLoad;
519 }
520
521 if (currLoad >= 0)
522 {
523 //Add to the grid load if possible
524 if (powered.powerOut.Grid != null)
525 {
526 powered.powerOut.Grid.Load += currLoad;
527 }
528 }
529 else if (powered.powerOut.Grid != null)
530 {
531 //Add connection as a source to be processed
532 powered.powerOut.Grid.AddSrc(powered.powerOut);
533 }
534 else
535 {
536 //Perform power calculations for the singular connection
537 float loadOut = -powered.GetConnectionPowerOut(powered.powerOut, 0, powered.MinMaxPowerOut(powered.powerOut, 0), 0);
538 if (powered is PowerTransfer pt2)
539 {
540 pt2.PowerLoad = loadOut;
541 }
542 else if (powered is PowerContainer pc)
543 {
544 //PowerContainer handles its own output value
545 }
546 else
547 {
548 powered.CurrPowerConsumption = loadOut;
549 }
550
551 //Indicate grid is resolved as it was the only device
552 powered.GridResolved(powered.powerOut);
553 }
554 }
555 }
556
557 //Iterate through all grids to determine the power on the grid
558 foreach (GridInfo grid in Grids.Values)
559 {
560 //Iterate through the priority src groups lowest first
561 foreach (PowerSourceGroup scrGroup in grid.PowerSourceGroups.Values)
562 {
563 scrGroup.MinMaxPower = PowerRange.Zero;
564
565 //Iterate through all connections in the group to get their minmax power and sum them
566 foreach (Connection c in scrGroup.Connections)
567 {
568 foreach (var device in c.Item.GetComponents<Powered>())
569 {
570 scrGroup.MinMaxPower += device.MinMaxPowerOut(c, grid.Load);
571 }
572 }
573
574 //Iterate through all connections to get their final power out provided the min max information
575 float addedPower = 0;
576 foreach (Connection c in scrGroup.Connections)
577 {
578 foreach (var device in c.Item.GetComponents<Powered>())
579 {
580 addedPower += device.GetConnectionPowerOut(c, grid.Power, scrGroup.MinMaxPower, grid.Load);
581 }
582 }
583
584 //Add the power to the grid
585 grid.Power += addedPower;
586 }
587
588 //Calculate Grid voltage, limit between 0 - 1000
589 float newVoltage = MathHelper.Min(grid.Power / MathHelper.Max(grid.Load, 1E-10f), 1000);
590 if (float.IsNegative(newVoltage))
591 {
592 newVoltage = 0.0f;
593 }
594
595 grid.Voltage = newVoltage;
596
597 //Iterate through all connections on that grid and run their gridResolved function
598 foreach (Connection c in grid.Connections)
599 {
600 foreach (var device in c.Item.GetComponents<Powered>())
601 {
602 device?.GridResolved(c);
603 }
604 }
605 }

607#if CLIENT
608 sw.Stop();
609 GameMain.PerformanceCounter.AddElapsedTicks("PowerUpdate", sw.ElapsedTicks);
610#endif

617 public virtual float GetCurrentPowerConsumption(Connection connection = null)
618 {
619 // If a handheld device there is no consumption
620 if (powerIn == null && powerOut == null)
621 {
622 return 0;
623 }
624
625 // Add extraload for PowerTransfer devices
626 if (this is PowerTransfer pt)
627 {
628 return PowerConsumption + pt.ExtraLoad;
629 }
630 else if (connection != this.powerIn || !IsActive)
631 {
632 //If not the power in connection or is inactive there is no draw
633 return 0;
634 }
635
636 //Otherwise return the max powerconsumption of the device
637 return PowerConsumption;

678 protected float GetAvailableInstantaneousBatteryPower()
679 {
680 if (item.Connections == null || powerIn == null) { return 0.0f; }
681 float availablePower = 0.0f;
682 var recipients = powerIn.Recipients;
683 foreach (Connection recipient in recipients)
684 {
685 if (!recipient.IsPower || !recipient.IsOutput) { continue; }
686 var battery = recipient.Item?.GetComponent<PowerContainer>();
687 if (battery == null || battery.Item.Condition <= 0.0f) { continue; }
688 float maxOutputPerFrame = battery.MaxOutPut / 60.0f;
689 float framesPerMinute = 3600.0f;
690 availablePower += Math.Min(battery.Charge * framesPerMinute, maxOutputPerFrame);
691 }
692 return availablePower;
693 }
