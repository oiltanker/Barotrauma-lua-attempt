539  public static LevelGenerationParams GetRandom(string seed, LevelData.LevelType type, Identifier biome = default)
540  {
541  Rand.SetSyncedSeed(ToolBox.StringToInt(seed));
542 
543  if (!LevelParams.Any())
544  {
545  throw new InvalidOperationException("Level generation presets not found - using default presets");
546  }
547 
548  var matchingLevelParams = LevelParams.Where(lp =>
549  lp.Type == type &&
550  (lp.AnyBiomeAllowed || lp.AllowedBiomeIdentifiers.Any()) &&
551  !lp.AllowedBiomeIdentifiers.Contains("None".ToIdentifier()));
552  matchingLevelParams = biome.IsEmpty
553  ? matchingLevelParams.Where(lp => lp.AnyBiomeAllowed || !lp.AllowedBiomeIdentifiers.All(b => Biome.Prefabs[b].IsEndBiome))
554  : matchingLevelParams.Where(lp => lp.AnyBiomeAllowed || lp.AllowedBiomeIdentifiers.Contains(biome));
555 
556  if (!matchingLevelParams.Any())
557  {
558  DebugConsole.ThrowError($"Suitable level generation presets not found (biome \"{biome.IfEmpty("null".ToIdentifier())}\", type: \"{type}\")");
559  if (!biome.IsEmpty)
560  {
561  //try to find params that at least have a suitable type
562  matchingLevelParams = LevelParams.Where(lp => lp.Type == type);
563  if (!matchingLevelParams.Any())
564  {
565  //still not found, give up and choose some params randomly
566  matchingLevelParams = LevelParams;
567  }
568  }
569  }
570 
571  return matchingLevelParams.GetRandom(Rand.RandSync.ServerAndClient);
572  }

574  public LevelGenerationParams(ContentXElement element, LevelGenerationParametersFile file) : base(file, element.GetAttributeIdentifier("identifier", element.Name.LocalName))

583  allowedBiomeIdentifiers.Remove("any".ToIdentifier());
