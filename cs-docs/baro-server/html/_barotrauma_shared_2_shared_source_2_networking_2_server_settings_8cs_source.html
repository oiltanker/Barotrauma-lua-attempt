117  public bool PropEquals(object a, object b)
118  {
119  switch (typeString)
120  {
121  case "float":
122  if (!(a is float?)) return false;
123  if (!(b is float?)) return false;
124  return MathUtils.NearlyEqual((float)a, (float)b);
125  case "int":
126  if (!(a is int?)) return false;
127  if (!(b is int?)) return false;
128  return (int)a == (int)b;
129  case "bool":
130  if (!(a is bool?)) return false;
131  if (!(b is bool?)) return false;
132  return (bool)a == (bool)b;
133  case "Enum":
134  if (!(a is Enum)) return false;
135  if (!(b is Enum)) return false;
136  return ((Enum)a).Equals((Enum)b);
137  default:
138  if (a == null || b == null)
139  {
140  return (a == null) == (b == null);
141  }
142  return a.ToString().Equals(b.ToString(), StringComparison.OrdinalIgnoreCase);
143  }
144  }

146  public void Read(IReadMessage msg)
147  {
148  int oldPos = msg.BitPosition;
149  UInt32 size = msg.ReadVariableUInt32();
150 
151  float x; float y; float z; float w;
152  byte r; byte g; byte b; byte a;
153  int ix; int iy; int width; int height;
154 
155  switch (typeString)
156  {
157  case "float":
158  if (size != 4) break;
159  property.SetValue(parentObject, msg.ReadSingle());
160  return;
161  case "int":
162  if (size != 4) break;
163  property.SetValue(parentObject, msg.ReadInt32());
164  return;
165  case "vector2":
166  if (size != 8) break;
167  x = msg.ReadSingle();
168  y = msg.ReadSingle();
169  property.SetValue(parentObject, new Vector2(x, y));
170  return;
171  case "vector3":
172  if (size != 12) break;
173  x = msg.ReadSingle();
174  y = msg.ReadSingle();
175  z = msg.ReadSingle();
176  property.SetValue(parentObject, new Vector3(x, y, z));
177  return;
178  case "vector4":
179  if (size != 16) break;
180  x = msg.ReadSingle();
181  y = msg.ReadSingle();
182  z = msg.ReadSingle();
183  w = msg.ReadSingle();
184  property.SetValue(parentObject, new Vector4(x, y, z, w));
185  return;
186  case "color":
187  if (size != 4) break;
188  r = msg.ReadByte();
189  g = msg.ReadByte();
190  b = msg.ReadByte();
191  a = msg.ReadByte();
192  property.SetValue(parentObject, new Color(r, g, b, a));
193  return;
194  case "rectangle":
195  if (size != 16) break;
196  ix = msg.ReadInt32();
197  iy = msg.ReadInt32();
198  width = msg.ReadInt32();
199  height = msg.ReadInt32();
200  property.SetValue(parentObject, new Rectangle(ix, iy, width, height));
201  return;
202  default:
203  msg.BitPosition = oldPos; //reset position to properly read the string
204  string incVal = msg.ReadString();
205  property.TrySetValue(parentObject, incVal);
206  return;
207  }
208 
209  //size didn't match: skip this
210  msg.BitPosition += (int)(8 * size);
211  }

213  public void Write(IWriteMessage msg, object overrideValue = null)
214  {
215  if (overrideValue == null) { overrideValue = Value; }
216  switch (typeString)
217  {
218  case "float":
219  msg.WriteVariableUInt32(4);
220  msg.Write((float)overrideValue);
221  break;
222  case "int":
223  msg.WriteVariableUInt32(4);
224  msg.Write((int)overrideValue);
225  break;
226  case "vector2":
227  msg.WriteVariableUInt32(8);
228  msg.Write(((Vector2)overrideValue).X);
229  msg.Write(((Vector2)overrideValue).Y);
230  break;
231  case "vector3":
232  msg.WriteVariableUInt32(12);
233  msg.Write(((Vector3)overrideValue).X);
234  msg.Write(((Vector3)overrideValue).Y);
235  msg.Write(((Vector3)overrideValue).Z);
236  break;
237  case "vector4":
238  msg.WriteVariableUInt32(16);
239  msg.Write(((Vector4)overrideValue).X);
240  msg.Write(((Vector4)overrideValue).Y);
241  msg.Write(((Vector4)overrideValue).Z);
242  msg.Write(((Vector4)overrideValue).W);
243  break;
244  case "color":
245  msg.WriteVariableUInt32(4);
246  msg.Write(((Color)overrideValue).R);
247  msg.Write(((Color)overrideValue).G);
248  msg.Write(((Color)overrideValue).B);
249  msg.Write(((Color)overrideValue).A);
250  break;
251  case "rectangle":
252  msg.WriteVariableUInt32(16);
253  msg.Write(((Rectangle)overrideValue).X);
254  msg.Write(((Rectangle)overrideValue).Y);
255  msg.Write(((Rectangle)overrideValue).Width);
256  msg.Write(((Rectangle)overrideValue).Height);
257  break;
258  default:
259  string strVal = overrideValue.ToString();
260 
261  msg.Write(strVal);
262  break;
263  }

299  netProperties = new Dictionary<UInt32, NetPropertyData>();
300 
301  using (MD5 md5 = MD5.Create())
302  {
303  var saveProperties = SerializableProperty.GetProperties<Serialize>(this);
304  foreach (var property in saveProperties)
305  {
306  object value = property.GetValue(this);
307  if (value == null) { continue; }
308 
309  string typeName = SerializableProperty.GetSupportedTypeName(value.GetType());
310  if (typeName != null || property.PropertyType.IsEnum)
311  {
312  NetPropertyData netPropertyData = new NetPropertyData(this, property, typeName);
313  UInt32 key = ToolBox.IdentifierToUint32Hash(netPropertyData.Name, md5);
314  if (key == 0) { key++; } //0 is reserved to indicate the end of the netproperties section of a message
315  if (netProperties.ContainsKey(key)){ throw new Exception("Hashing collision in ServerSettings.netProperties: " + netProperties[key] + " has same key as " + property.Name + " (" + key.ToString() + ")"); }
316  netProperties.Add(key, netPropertyData);
317  }
318  }
319 
320  var karmaProperties = SerializableProperty.GetProperties<Serialize>(networkMember.KarmaManager);
321  foreach (var property in karmaProperties)
322  {
323  object value = property.GetValue(networkMember.KarmaManager);
324  if (value == null) { continue; }
325 
326  string typeName = SerializableProperty.GetSupportedTypeName(value.GetType());
327  if (typeName != null || property.PropertyType.IsEnum)
328  {
329  NetPropertyData netPropertyData = new NetPropertyData(networkMember.KarmaManager, property, typeName);
330  UInt32 key = ToolBox.IdentifierToUint32Hash(netPropertyData.Name, md5);
331  if (netProperties.ContainsKey(key)) { throw new Exception("Hashing collision in ServerSettings.netProperties: " + netProperties[key] + " has same key as " + property.Name + " (" + key.ToString() + ")"); }
332  netProperties.Add(key, netPropertyData);
333  }
334  }
335  }

981  public static byte[] SaltPassword(byte[] password, int salt)
982  {
983  byte[] saltedPw = new byte[password.Length*2];
984  for (int i = 0; i < password.Length; i++)
985  {
986  saltedPw[(i * 2)] = password[i];
987  saltedPw[(i * 2) + 1] = (byte)((salt >> (8 * (i % 4))) & 0xff);
988  }
989  saltedPw = Lidgren.Network.NetUtility.ComputeSHAHash(saltedPw);
990  return saltedPw;
