286  public void Update(float deltaTime)
287  {
288  while (impactQueue.Count > 0)
289  {
290  var impact = impactQueue.Dequeue();
291 
292  if (impact.Target.UserData is VoronoiCell cell)
293  {
294  HandleLevelCollision(impact, cell);
295  }
296  else if (impact.Target.Body.UserData is Structure)
297  {
298  HandleLevelCollision(impact);
299  }
300  else if (impact.Target.Body.UserData is Submarine otherSub)
301  {
302  HandleSubCollision(impact, otherSub);
303  }
304  else if (impact.Target.Body.UserData is Limb limb)
305  {
306  HandleLimbCollision(impact, limb);
307  }
308  }
309 
310  //-------------------------
311 
312  if (Body.FarseerBody.BodyType == BodyType.Static) { return; }
313 
314  ClientUpdatePosition(deltaTime);
315  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
316 
317  Vector2 totalForce = CalculateBuoyancy();
318 
319  //-------------------------
320 
321  //if outside left or right edge of the level
322  if (Level.Loaded != null && (Position.X < 0 || Position.X > Level.Loaded.Size.X))
323  {
324  Rectangle worldBorders = Borders;
325  worldBorders.Location += MathUtils.ToPoint(Position);
326 
327  //push the sub back below the upper "barrier" of the level
328  if (worldBorders.Y > Level.Loaded.Size.Y)
329  {
330  Body.LinearVelocity = new Vector2(
331  Body.LinearVelocity.X,
332  Math.Min(Body.LinearVelocity.Y, ConvertUnits.ToSimUnits(Level.Loaded.Size.Y - worldBorders.Y)));
333  }
334  else if (worldBorders.Y - worldBorders.Height < Level.Loaded.BottomPos)
335  {
336  Body.LinearVelocity = new Vector2(
337  Body.LinearVelocity.X,
338  Math.Max(Body.LinearVelocity.Y, ConvertUnits.ToSimUnits(Level.Loaded.BottomPos - (worldBorders.Y - worldBorders.Height))));
339  }
340 
341  //hard limit for how far outside the level the sub can go
342  float maxDist = 200000.0f;
343  //the force of the current starts to increase exponentially after this point
344  float exponentialForceIncreaseDist = 150000.0f;
345  float distance = Position.X < 0 ? Math.Abs(Position.X) : Position.X - Level.Loaded.Size.X;
346  if (distance > 0)
347  {
348  if (distance > maxDist)
349  {
350  if (Position.X < 0)
351  {
352  Body.LinearVelocity = new Vector2(Math.Max(0, Body.LinearVelocity.X), Body.LinearVelocity.Y);
353  }
354  else
355  {
356  Body.LinearVelocity = new Vector2(Math.Min(0, Body.LinearVelocity.X), Body.LinearVelocity.Y);
357  }
358  }
359  if (distance > exponentialForceIncreaseDist)
360  {
361  distance += (float)Math.Pow((distance - exponentialForceIncreaseDist) * 0.01f, 2.0f);
362  }
363  float force = distance * 0.5f;
364  totalForce += (Position.X < 0 ? Vector2.UnitX : -Vector2.UnitX) * force;
365  if (Character.Controlled != null && Character.Controlled.Submarine == submarine)
366  {
367  GameMain.GameScreen.Cam.Shake = Math.Max(GameMain.GameScreen.Cam.Shake, Math.Min(force * 0.0001f, 5.0f));
368  }
369  }
370  }
371 
372  //-------------------------
373 
374  if (Body.LinearVelocity.LengthSquared() > 0.0001f)
375  {
376  //TODO: sync current drag with clients?
377  float attachedMass = 0.0f;
378  JointEdge jointEdge = Body.FarseerBody.JointList;
379  while (jointEdge != null)
380  {
381  Body otherBody = jointEdge.Joint.BodyA == Body.FarseerBody ? jointEdge.Joint.BodyB : jointEdge.Joint.BodyA;
382  Character character = (otherBody.UserData as Limb)?.character;
383  if (character != null) attachedMass += character.Mass;
384 
385  jointEdge = jointEdge.Next;
386  }
387 
388  float horizontalDragCoefficient = MathHelper.Clamp(HorizontalDrag + attachedMass / 5000.0f, 0.0f, MaxDrag);
389  totalForce.X -= Math.Sign(Body.LinearVelocity.X) * Body.LinearVelocity.X * Body.LinearVelocity.X * horizontalDragCoefficient * Body.Mass;
390 
391  float verticalDragCoefficient = MathHelper.Clamp(VerticalDrag + attachedMass / 5000.0f, 0.0f, MaxDrag);
392  totalForce.Y -= Math.Sign(Body.LinearVelocity.Y) * Body.LinearVelocity.Y * Body.LinearVelocity.Y * verticalDragCoefficient * Body.Mass;
393  }
394 
395  ApplyForce(totalForce);
396 
397  UpdateDepthDamage(deltaTime);
398  }

408  {

410  worldBorders.Location += MathUtils.ToPoint(ConvertUnits.ToDisplayUnits(Body.SimPosition));
411 
412  Vector2 translateDir = Vector2.Normalize(subTranslation);
413  if (!MathUtils.IsValid(translateDir)) translateDir = Vector2.UnitY;
414 
415  foreach (Character c in Character.CharacterList)
416  {
417  if (c.AnimController.CurrentHull != null && c.AnimController.CanEnterSubmarine) { continue; }
418 
419  foreach (Limb limb in c.AnimController.Limbs)
420  {
421  if (limb.IsSevered) { continue; }
422  //if the character isn't inside the bounding box, continue
423  if (!Submarine.RectContains(worldBorders, limb.WorldPosition)) { continue; }
424 
425  //cast a line from the position of the character to the same direction as the translation of the sub
426  //and see where it intersects with the bounding box
427  if (!MathUtils.GetLineRectangleIntersection(limb.WorldPosition,
428  limb.WorldPosition + translateDir * 100000.0f, worldBorders, out Vector2 intersection))
429  {
430  //should never happen when casting a line out from inside the bounding box
431  Debug.Assert(false);
432  continue;
433  }
434 
435 
436  //"+ translatedir" in order to move the character slightly away from the wall
437  c.AnimController.SetPosition(ConvertUnits.ToSimUnits(c.WorldPosition + (intersection - limb.WorldPosition)) + translateDir);

439  return;
440  }

445  private Vector2 CalculateBuoyancy()
446  {
447  float waterVolume = 0.0f;
448  float volume = 0.0f;
449  foreach (Hull hull in Hull.HullList)
450  {
451  if (hull.Submarine != submarine) continue;
452 
453  waterVolume += hull.WaterVolume;
454  volume += hull.Volume;
455  }
456 
457  float waterPercentage = volume <= 0.0f ? 0.0f : waterVolume / volume;
458 
459  float buoyancy = NeutralBallastPercentage - waterPercentage;
460 
461  if (buoyancy > 0.0f)
462  buoyancy *= 2.0f;
463  else
464  buoyancy = Math.Max(buoyancy, -0.5f);
465 
466  return new Vector2(0.0f, buoyancy * Body.Mass * 10.0f);
467  }

560  private bool CheckCharacterCollision(Contact contact, Character character)
561  {
562  //characters that can't enter the sub always collide regardless of gaps
563  if (!character.AnimController.CanEnterSubmarine) { return true; }
564  if (character.Submarine != null) { return false; }
565 
566  contact.GetWorldManifold(out Vector2 contactNormal, out FixedArray2<Vector2> points);
567 
568  Vector2 normalizedVel = character.AnimController.Collider.LinearVelocity == Vector2.Zero ?
569  Vector2.Zero : Vector2.Normalize(character.AnimController.Collider.LinearVelocity);
570 
571  //try to find the hull right next to the contact point
572  Vector2 targetPos = ConvertUnits.ToDisplayUnits(points[0] - contactNormal * 0.1f);
573  Hull newHull = Hull.FindHull(targetPos, null);
574  //not found, try searching a bit further
575  if (newHull == null)
576  {
577  targetPos = ConvertUnits.ToDisplayUnits(points[0] - contactNormal);
578  newHull = Hull.FindHull(targetPos, null);
579  }
580  //still not found, try searching in the direction the character is heading to
581  if (newHull == null)
582  {
583  targetPos = ConvertUnits.ToDisplayUnits(points[0] + normalizedVel);
584  newHull = Hull.FindHull(targetPos, null);
585  }
586 
587  var gaps = newHull?.ConnectedGaps ?? Gap.GapList.Where(g => g.Submarine == submarine);
588  Gap adjacentGap = Gap.FindAdjacent(gaps, ConvertUnits.ToDisplayUnits(points[0]), 200.0f);
589  if (adjacentGap == null) { return true; }
590 
591  if (newHull != null)
592  {
593  CoroutineManager.Invoke(() =>
594  character.AnimController.FindHull(newHull.WorldPosition, setSubmarine: true));
595  }
596 
597  return false;
598  }

600  private void HandleLimbCollision(Impact collision, Limb limb)
601  {
602  if (limb?.body?.FarseerBody == null || limb.character == null) { return; }
603 
604  float impactMass = limb.Mass;
605  var enemyAI = limb.character.AIController as EnemyAIController;
606  float attackMultiplier = 1.0f;
607  if (enemyAI?.ActiveAttack != null)
608  {
609  impactMass = Math.Max(Math.Max(limb.Mass, limb.character.AnimController.MainLimb.Mass), limb.character.AnimController.Collider.Mass);
610  attackMultiplier = enemyAI.ActiveAttack.SubmarineImpactMultiplier;
611  }
612 
613  if (impactMass * attackMultiplier > MinImpactLimbMass)
614  {
615  Vector2 normal =
616  Vector2.DistanceSquared(Body.SimPosition, limb.SimPosition) < 0.0001f ?
617  Vector2.UnitY :
618  Vector2.Normalize(Body.SimPosition - limb.SimPosition);
619 
620  float impact = Math.Min(Vector2.Dot(collision.Velocity, -normal), 50.0f) * Math.Min(impactMass / 300.0f, 1);
621  impact *= attackMultiplier;
622 
623  ApplyImpact(impact, normal, collision.ImpactPos, applyDamage: false);

627  }

633  while (contactEdge?.Contact != null)
634  {
635  if (contactEdge.Contact.Enabled &&
636  contactEdge.Contact.IsTouching &&
637  contactEdge.Other?.UserData is VoronoiCell)
638  {
639  levelContacts.Add(contactEdge.Contact);
640  }
641  contactEdge = contactEdge.Next;
642  }

647  //not a very realistic way to handle the collisions (makes it seem as if the characters were made of reinforced concrete),
648  //but more realistic than bouncing and prevents using characters as "bumpers" that prevent all collision damage
649  Vector2 avgContactNormal = Vector2.Zero;
650  foreach (Contact levelContact in levelContacts)
651  {
652  levelContact.GetWorldManifold(out Vector2 contactNormal, out FixedArray2<Vector2> temp);
653 
654  //if the contact normal is pointing from the limb towards the level cell it's touching, flip the normal
655  VoronoiCell cell = levelContact.FixtureB.UserData is VoronoiCell ?
656  ((VoronoiCell)levelContact.FixtureB.UserData) : ((VoronoiCell)levelContact.FixtureA.UserData);
657 
658  var cellDiff = ConvertUnits.ToDisplayUnits(limb.body.SimPosition) - cell.Center;
659  if (Vector2.Dot(contactNormal, cellDiff) < 0)
660  {
661  contactNormal = -contactNormal;
662  }
663 
664  avgContactNormal += contactNormal;
665 
666  //apply impacts at the positions where this sub is touching the limb
667  ApplyImpact((Vector2.Dot(-collision.Velocity, contactNormal) / 2.0f) / levelContacts.Count, contactNormal, collision.ImpactPos, applyDamage: false);
668  }
669  avgContactNormal /= levelContacts.Count;
670 
671  float contactDot = Vector2.Dot(Body.LinearVelocity, -avgContactNormal);
672  if (contactDot > 0.001f)
673  {
674  Vector2 velChange = Vector2.Normalize(Body.LinearVelocity) * contactDot;
675  if (!MathUtils.IsValid(velChange))
676  {
677  GameAnalyticsManager.AddErrorEventOnce(
678  "SubmarineBody.HandleLimbCollision:" + submarine.ID,
679  GameAnalyticsManager.ErrorSeverity.Error,
680  "Invalid velocity change in SubmarineBody.HandleLimbCollision (submarine velocity: " + Body.LinearVelocity
681  + ", avgContactNormal: " + avgContactNormal
682  + ", contactDot: " + contactDot
683  + ", velChange: " + velChange + ")");
684  return;
685  }
686 
687  Body.LinearVelocity -= velChange;
688

690  {
691  float damageAmount = contactDot * Body.Mass / limb.character.Mass;
692  limb.character.LastDamageSource = submarine;
693  limb.character.DamageLimb(ConvertUnits.ToDisplayUnits(collision.ImpactPos), limb,
694  AfflictionPrefab.ImpactDamage.Instantiate(damageAmount).ToEnumerable(), 0.0f, true, 0.0f);
695

697  {

699  {
700  if (limbJoint.IsSevered || (limbJoint.LimbA != limb && limbJoint.LimbB != limb)) continue;
701  limb.character.AnimController.SeverLimbJoint(limbJoint);
702  }
703  }

752  private void HandleSubCollision(Impact impact, Submarine otherSub)
753  {
754  Debug.Assert(otherSub != submarine);
755 
756  Vector2 normal = impact.Normal;
757  if (impact.Target.Body == otherSub.SubBody.Body.FarseerBody)
758  {
759  normal = -normal;
760  }
761 
762  float thisMass = Body.Mass + submarine.DockedTo.Sum(s => s.PhysicsBody.Mass);
763  float otherMass = otherSub.PhysicsBody.Mass + otherSub.DockedTo.Sum(s => s.PhysicsBody.Mass);
764  float massRatio = otherMass / (thisMass + otherMass);
765 
766  float impulse = (Vector2.Dot(impact.Velocity, normal) / 2.0f) * massRatio;
767 
768  //apply impact to this sub (the other sub takes care of this in its own collision callback)
769  ApplyImpact(impulse, normal, impact.ImpactPos);
770  foreach (Submarine dockedSub in submarine.DockedTo)
771  {
772  dockedSub.SubBody.ApplyImpact(impulse, normal, impact.ImpactPos);
773  }
774 
775  //find all contacts between this sub and level walls
776  List<Contact> levelContacts = new List<Contact>();
777  ContactEdge contactEdge = Body?.FarseerBody?.ContactList;
778  while (contactEdge?.Next != null)
779  {
780  if (contactEdge.Contact.Enabled &&
781  contactEdge.Other.UserData is VoronoiCell &&
782  contactEdge.Contact.IsTouching)
783  {
784  levelContacts.Add(contactEdge.Contact);
785  }
786 
787  contactEdge = contactEdge.Next;
788  }
789 
790  if (levelContacts.Count == 0) { return; }
791 
792  //if this sub is in contact with the level, apply artifical impacts
793  //to both subs to prevent the other sub from bouncing on top of this one
794  //and to fake the other sub "crushing" this one against a wall
795  Vector2 avgContactNormal = Vector2.Zero;
796  foreach (Contact levelContact in levelContacts)
797  {
798  levelContact.GetWorldManifold(out Vector2 contactNormal, out FixedArray2<Vector2> temp);
799 
800  //if the contact normal is pointing from the sub towards the level cell we collided with, flip the normal
801  VoronoiCell cell = levelContact.FixtureB.UserData is VoronoiCell ?
802  ((VoronoiCell)levelContact.FixtureB.UserData) : ((VoronoiCell)levelContact.FixtureA.UserData);
803 
804  var cellDiff = ConvertUnits.ToDisplayUnits(Body.SimPosition) - cell.Center;
805  if (Vector2.Dot(contactNormal, cellDiff) < 0)
806  {
807  contactNormal = -contactNormal;
808  }
809 
810  avgContactNormal += contactNormal;
811 
812  //apply impacts at the positions where this sub is touching the level
813  ApplyImpact((Vector2.Dot(impact.Velocity, contactNormal) / 2.0f) * massRatio / levelContacts.Count, contactNormal, impact.ImpactPos);
814  }
815  avgContactNormal /= levelContacts.Count;
816 
817  //apply an impact to the other sub
818  float contactDot = Vector2.Dot(otherSub.PhysicsBody.LinearVelocity, -avgContactNormal);
819  if (contactDot > 0.0f)
820  {
821  if (otherSub.PhysicsBody.LinearVelocity.LengthSquared() > 0.0001f)
822  {
823  otherSub.PhysicsBody.LinearVelocity -= Vector2.Normalize(otherSub.PhysicsBody.LinearVelocity) * contactDot;
824  }
825 
826  impulse = Vector2.Dot(otherSub.Velocity, normal);
827  otherSub.SubBody.ApplyImpact(impulse, normal, impact.ImpactPos);
828  foreach (Submarine dockedSub in otherSub.DockedTo)
829  {
830  dockedSub.SubBody.ApplyImpact(impulse, normal, impact.ImpactPos);
831  }
832  }

835  private void ApplyImpact(float impact, Vector2 direction, Vector2 impactPos, bool applyDamage = true)
836  {
837  if (impact < MinCollisionImpact) { return; }
838 
839  Vector2 impulse = direction * impact * 0.5f;
840  impulse = impulse.ClampLength(MaxCollisionImpact);
841 
842  float impulseMagnitude = impulse.Length();
843 
844  if (!MathUtils.IsValid(impulse))
845  {
846  string errorMsg =
847  "Invalid impulse in SubmarineBody.ApplyImpact: " + impulse +
848  ". Direction: " + direction + ", body position: " + Body.SimPosition + ", impact: " + impact + ".";
849  if (GameMain.NetworkMember != null)
850  {
851  errorMsg += GameMain.NetworkMember.IsClient ? " Playing as a client." : " Hosting a server.";
852  }
853  if (GameSettings.CurrentConfig.VerboseLogging) DebugConsole.ThrowError(errorMsg);
854  GameAnalyticsManager.AddErrorEventOnce(
855  "SubmarineBody.ApplyImpact:InvalidImpulse",
856  GameAnalyticsManager.ErrorSeverity.Error,
857  errorMsg);
858  return;
859  }
860 
861 #if CLIENT
862  if (Character.Controlled != null && Character.Controlled.Submarine == submarine && Character.Controlled.KnockbackCooldownTimer <= 0.0f)
863  {
864  GameMain.GameScreen.Cam.Shake = Math.Max(impact * 10.0f, GameMain.GameScreen.Cam.Shake);
865  if (submarine.Info.Type == SubmarineType.Player && !submarine.DockedTo.Any(s => s.Info.Type != SubmarineType.Player))
866  {
867  float angularVelocity =
868  (impactPos.X - Body.SimPosition.X) / ConvertUnits.ToSimUnits(submarine.Borders.Width / 2) * impulse.Y
869  - (impactPos.Y - Body.SimPosition.Y) / ConvertUnits.ToSimUnits(submarine.Borders.Height / 2) * impulse.X;
870  GameMain.GameScreen.Cam.AngularVelocity = MathHelper.Clamp(angularVelocity * 0.1f, -1.0f, 1.0f);
871  }
872  }
873 #endif
874 
875  foreach (Character c in Character.CharacterList)
876  {
877  if (c.Submarine != submarine) { continue; }
878  if (c.KnockbackCooldownTimer > 0.0f) { continue; }
879 
880  c.KnockbackCooldownTimer = Character.KnockbackCooldown;
881 
882  foreach (Limb limb in c.AnimController.Limbs)
883  {
884  if (limb.IsSevered) { continue; }
885  limb.body.ApplyLinearImpulse(limb.Mass * impulse, 10.0f);
886  }
887 
888  bool holdingOntoSomething = false;
889  if (c.SelectedConstruction != null)
890  {
891  holdingOntoSomething =
892  c.SelectedConstruction.GetComponent<Ladder>() != null ||
893  (c.SelectedConstruction.GetComponent<Controller>()?.LimbPositions.Any() ?? false);
894  }
895 
896  if (!holdingOntoSomething)
897  {
898  c.AnimController.Collider.ApplyLinearImpulse(c.AnimController.Collider.Mass * impulse, 10.0f);
899  //stun for up to 2 second if the impact equal or higher to the maximum impact
900  if (impact >= MaxCollisionImpact)
901  {
902  c.AddDamage(impactPos, AfflictionPrefab.ImpactDamage.Instantiate(3.0f).ToEnumerable(), stun: Math.Min(impulse.Length() * 0.2f, 2.0f), playSound: true);
903  }
904  }
905  }
906 
907  foreach (Item item in Item.ItemList)
908  {
909  if (item.Submarine != submarine || item.CurrentHull == null || item.body == null || !item.body.Enabled) { continue; }
910  if (item.body.Mass > impulseMagnitude) { continue; }
911 
912  item.body.ApplyLinearImpulse(impulse, 10.0f);
913  item.PositionUpdateInterval = 0.0f;
914  }
915 
916  float dmg = applyDamage ? impact * ImpactDamageMultiplier : 0.0f;
917  var damagedStructures = Explosion.RangedStructureDamage(
918  ConvertUnits.ToDisplayUnits(impactPos),
919  impact * 50.0f,
920  dmg, dmg);
921 
922 #if CLIENT
923  PlayDamageSounds(damagedStructures, impactPos, impact, "StructureBlunt");
924 #endif
925  }
