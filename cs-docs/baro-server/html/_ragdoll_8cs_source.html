649  public bool OnLimbCollision(Fixture f1, Fixture f2, Contact contact)
650  {
651  if (f2.Body.UserData is Submarine && character.Submarine == (Submarine)f2.Body.UserData) { return false; }
652  if (f2.UserData is Hull && character.Submarine != null) { return false; }
653 
654  //using the velocity of the limb would make the impact damage more realistic,
655  //but would also make it harder to edit the animations because the forces/torques
656  //would all have to be balanced in a way that prevents the character from doing
657  //impact damage to itself
658  Vector2 velocity = Collider.LinearVelocity;
659  if (character.Submarine == null && f2.Body.UserData is Submarine)
660  {
661  velocity -= ((Submarine)f2.Body.UserData).Velocity;
662  }
663 
664  //always collides with bodies other than structures
665  if (!(f2.Body.UserData is Structure structure))
666  {
667  if (!f2.IsSensor)
668  {
669  lock (impactQueue)
670  {
671  impactQueue.Enqueue(new Impact(f1, f2, contact, velocity));
672  }
673  }
674  return true;
675  }
676 
677  Vector2 colliderBottom = GetColliderBottom();
678  if (structure.IsPlatform)
679  {
680  if (IgnorePlatforms || currentHull == null) { return false; }
681 
682  if (colliderBottom.Y < ConvertUnits.ToSimUnits(structure.Rect.Y - 5)) { return false; }
683  if (f1.Body.Position.Y < ConvertUnits.ToSimUnits(structure.Rect.Y - 5)) { return false; }
684  }
685  else if (structure.StairDirection != Direction.None)
686  {
687  Stairs = null;
688 
689  //don't collider with stairs if
690 
691  //1. bottom of the collider is at the bottom of the stairs and the character isn't trying to move upwards
692  float stairBottomPos = ConvertUnits.ToSimUnits(structure.Rect.Y - structure.Rect.Height + 10);
693  if (colliderBottom.Y < stairBottomPos && targetMovement.Y < 0.5f) { return false; }
694 
695  //2. bottom of the collider is at the top of the stairs and the character isn't trying to move downwards
696  if (targetMovement.Y >= 0.0f && colliderBottom.Y >= ConvertUnits.ToSimUnits(structure.Rect.Y - Submarine.GridSize.Y * 5)) { return false; }
697 
698  //3. collided with the stairs from below
699  if (contact.Manifold.LocalNormal.Y < 0.0f) { return false; }
700 
701  //4. contact points is above the bottom half of the collider
702  contact.GetWorldManifold(out Vector2 normal, out FarseerPhysics.Common.FixedArray2<Vector2> points);
703  if (points[0].Y > Collider.SimPosition.Y) { return false; }
704 
705  //5. in water
706  if (inWater && targetMovement.Y < 0.5f) { return false; }
707 
708  //---------------
709 
710  //set stairs to that of the one dragging us
711  if (character.SelectedBy != null)
712  Stairs = character.SelectedBy.AnimController.Stairs;
713  else
714  Stairs = structure;
715 
716  if (Stairs == null)
717  return false;
718  }
719 
720  lock (impactQueue)
721  {
722  impactQueue.Enqueue(new Impact(f1, f2, contact, velocity));
723  }
724 
725  return true;
