594 public void RemoveLimb(Limb limb)
595 {
596 if (!Limbs.Contains(limb)) return;
597
598 Limb[] newLimbs = new Limb[Limbs.Length - 1];
599
600 int i = 0;
601 foreach (Limb existingLimb in Limbs)
602 {
603 if (existingLimb == limb) continue;
604 newLimbs[i] = existingLimb;
605 i++;
606 }
607
608 limbs = newLimbs;
609 if (limbDictionary.ContainsKey(limb.type))
610 {
611 limbDictionary.Remove(limb.type);
612 // If there is another limb of the same type, replace the limb in the dictionary.
613 if (HasMultipleLimbsOfSameType)
614 {
615 var otherLimb = Limbs.FirstOrDefault(l => l != limb && l.type == limb.type);
616 if (otherLimb != null)
617 {
618 limbDictionary.Add(otherLimb.type, otherLimb);
619 }
620 }
621 }
622
623 // TODO: this could be optimized if needed, but at least we need to remove the limb from the inversedDrawOrder array.
624 SetupDrawOrder();
625
626 //remove all joints that were attached to the removed limb
627 LimbJoint[] attachedJoints = Array.FindAll(LimbJoints, lj => lj.LimbA == limb || lj.LimbB == limb);
628 if (attachedJoints.Length > 0)
629 {
630 LimbJoint[] newJoints = new LimbJoint[LimbJoints.Length - attachedJoints.Length];
631 i = 0;
632 foreach (LimbJoint limbJoint in LimbJoints)
633 {
634 if (attachedJoints.Contains(limbJoint)) continue;
635 newJoints[i] = limbJoint;
636 i++;
637 }
638 LimbJoints = newJoints;
639 }
640
641 SubtractMass(limb);
642 limb.Remove();
643 foreach (LimbJoint limbJoint in attachedJoints)
644 {
645 GameMain.World.Remove(limbJoint.Joint);
646 }

649 public bool OnLimbCollision(Fixture f1, Fixture f2, Contact contact)
650 {
651 if (f2.Body.UserData is Submarine && character.Submarine == (Submarine)f2.Body.UserData) { return false; }
652 if (f2.UserData is Hull && character.Submarine != null) { return false; }
653
654 //using the velocity of the limb would make the impact damage more realistic,
655 //but would also make it harder to edit the animations because the forces/torques
656 //would all have to be balanced in a way that prevents the character from doing
657 //impact damage to itself
658 Vector2 velocity = Collider.LinearVelocity;
659 if (character.Submarine == null && f2.Body.UserData is Submarine)
660 {
661 velocity -= ((Submarine)f2.Body.UserData).Velocity;
662 }
663
664 //always collides with bodies other than structures
665 if (!(f2.Body.UserData is Structure structure))
666 {
667 if (!f2.IsSensor)
668 {
669 lock (impactQueue)
670 {
671 impactQueue.Enqueue(new Impact(f1, f2, contact, velocity));
672 }
673 }
674 return true;
675 }
676
677 Vector2 colliderBottom = GetColliderBottom();
678 if (structure.IsPlatform)
679 {
680 if (IgnorePlatforms || currentHull == null) { return false; }
681
682 if (colliderBottom.Y < ConvertUnits.ToSimUnits(structure.Rect.Y - 5)) { return false; }
683 if (f1.Body.Position.Y < ConvertUnits.ToSimUnits(structure.Rect.Y - 5)) { return false; }
684 }
685 else if (structure.StairDirection != Direction.None)
686 {
687 Stairs = null;
688
689 //don't collider with stairs if
690
691 //1. bottom of the collider is at the bottom of the stairs and the character isn't trying to move upwards
692 float stairBottomPos = ConvertUnits.ToSimUnits(structure.Rect.Y - structure.Rect.Height + 10);
693 if (colliderBottom.Y < stairBottomPos && targetMovement.Y < 0.5f) { return false; }
694
695 //2. bottom of the collider is at the top of the stairs and the character isn't trying to move downwards
696 if (targetMovement.Y >= 0.0f && colliderBottom.Y >= ConvertUnits.ToSimUnits(structure.Rect.Y - Submarine.GridSize.Y * 5)) { return false; }
697
698 //3. collided with the stairs from below
699 if (contact.Manifold.LocalNormal.Y < 0.0f) { return false; }
700
701 //4. contact points is above the bottom half of the collider
702 contact.GetWorldManifold(out Vector2 normal, out FarseerPhysics.Common.FixedArray2<Vector2> points);
703 if (points[0].Y > Collider.SimPosition.Y) { return false; }
704
705 //5. in water
706 if (inWater && targetMovement.Y < 0.5f) { return false; }
707
708 //---------------
709
710 //set stairs to that of the one dragging us
711 if (character.SelectedBy != null)
712 Stairs = character.SelectedBy.AnimController.Stairs;
713 else
714 Stairs = structure;
715
716 if (Stairs == null)
717 return false;
718 }
719
720 lock (impactQueue)
721 {
722 impactQueue.Enqueue(new Impact(f1, f2, contact, velocity));
723 }
724
725 return true;
726 }

728 private void ApplyImpact(Fixture f1, Fixture f2, Vector2 localNormal, Vector2 impactPos, Vector2 velocity)
729 {
730 if (character.DisableImpactDamageTimer > 0.0f) { return; }
731
732 Vector2 normal = localNormal;
733 float impact = Vector2.Dot(velocity, -normal);
734 if (f1.Body == Collider.FarseerBody || !Collider.Enabled)
735 {
736 bool isNotRemote = true;
737 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { isNotRemote = !character.IsRemotelyControlled; }
738
739 if (isNotRemote)
740 {
741 if (impact > ImpactTolerance)
742 {
743 impactPos = ConvertUnits.ToDisplayUnits(impactPos);
744 if (character.Submarine != null) impactPos += character.Submarine.Position;
745
746 float impactDamage = Math.Min((impact - ImpactTolerance) * ImpactDamageMultiplayer, character.MaxVitality * MaxImpactDamage);
747
748 var should = GameMain.LuaCs.Hook.Call<float?>("changeFallDamage", impactDamage, character, impactPos, velocity);
749
750 if (should != null)
751 {
752 impactDamage = should.Value;
753 }
754
755 character.LastDamageSource = null;
756 character.AddDamage(impactPos, AfflictionPrefab.ImpactDamage.Instantiate(impactDamage).ToEnumerable(), 0.0f, true);
757 strongestImpact = Math.Max(strongestImpact, impact - ImpactTolerance);
758 character.ApplyStatusEffects(ActionType.OnImpact, 1.0f);
759 //briefly disable impact damage
760 //otherwise the character will take damage multiple times when for example falling,
761 //because we use the velocity of the collider to determine the impact
762 //(i.e. the character would take damage until the collider hits the floor and stops)
763 character.DisableImpactDamageTimer = 0.25f;
764 }
765 }
766 }
767
768 ImpactProjSpecific(impact, f1.Body);
769 }

947 public void FindHull(Vector2? worldPosition = null, bool setSubmarine = true)
948 {
949 Vector2 findPos = worldPosition == null ? this.WorldPosition : (Vector2)worldPosition;
950 if (!MathUtils.IsValid(findPos))
951 {
952 GameAnalyticsManager.AddErrorEventOnce(
953 "Ragdoll.FindHull:InvalidPosition",
954 GameAnalyticsManager.ErrorSeverity.Error,
955 "Attempted to find a hull at an invalid position (" + findPos + ")\n" + Environment.StackTrace.CleanupStackTrace());
956 return;
957 }
958
959 Hull newHull = Hull.FindHull(findPos, currentHull);
960
961 if (newHull == currentHull) { return; }
962
963 if (!CanEnterSubmarine || (character.AIController != null && !character.AIController.CanEnterSubmarine))
964 {
965 //character is inside the sub even though it shouldn't be able to enter -> teleport it out
966
967 //far from an ideal solution, but monsters getting lodged inside the sub seems to be
968 //pretty rare during normal gameplay (requires abnormally high velocities), so I think
969 //this is preferable to the cost of using continuous collision detection for the character collider
970 if (newHull?.Submarine != null)
971 {
972 Vector2 hullDiff = WorldPosition - newHull.WorldPosition;
973 Vector2 moveDir = hullDiff.LengthSquared() < 0.001f ? Vector2.UnitY : Vector2.Normalize(hullDiff);
974
975 //find a position 32 units away from the hull
976 if (MathUtils.GetLineRectangleIntersection(
977 newHull.WorldPosition,
978 newHull.WorldPosition + moveDir * Math.Max(newHull.Rect.Width, newHull.Rect.Height),
979 new Rectangle(newHull.WorldRect.X - 32, newHull.WorldRect.Y + 32, newHull.WorldRect.Width + 64, newHull.Rect.Height + 64),
980 out Vector2 intersection))
981 {
982 Collider.SetTransform(ConvertUnits.ToSimUnits(intersection), Collider.Rotation);
983 }
984 return;
985 }
986 }
987
988 if (setSubmarine)
989 {
990 //in -> out
991 if (newHull?.Submarine == null && currentHull?.Submarine != null)
992 {
993 //don't teleport out yet if the character is going through a gap
994 if (Gap.FindAdjacent(Gap.GapList.Where(g => g.Submarine == currentHull.Submarine), findPos, 150.0f) != null) { return; }
995 if (Limbs.Any(l => Gap.FindAdjacent(currentHull.ConnectedGaps, l.WorldPosition, ConvertUnits.ToDisplayUnits(l.body.GetSize().Combine())) != null)) { return; }
996 character.MemLocalState?.Clear();
997 Teleport(ConvertUnits.ToSimUnits(currentHull.Submarine.Position), currentHull.Submarine.Velocity, detachProjectiles: false);
998 }
999 //out -> in
1000 else if (currentHull == null && newHull.Submarine != null)
1001 {
1002 character.MemLocalState?.Clear();
1003 Teleport(-ConvertUnits.ToSimUnits(newHull.Submarine.Position), -newHull.Submarine.Velocity, detachProjectiles: false);
1004 }
1005 //from one sub to another
1006 else if (newHull != null && currentHull != null && newHull.Submarine != currentHull.Submarine)
1007 {
1008 character.MemLocalState?.Clear();
1009 Vector2 newSubPos = newHull.Submarine == null ? Vector2.Zero : newHull.Submarine.Position;
1010 Vector2 prevSubPos = currentHull.Submarine == null ? Vector2.Zero : currentHull.Submarine.Position;
1011 Teleport(ConvertUnits.ToSimUnits(prevSubPos - newSubPos), Vector2.Zero, detachProjectiles: false);
1012 }
1013 }
1014
1015 CurrentHull = newHull;
1016 character.Submarine = currentHull?.Submarine;
1017 character.AttachedProjectiles.ForEach(p => p?.Item?.UpdateTransform());
1018 }

1040 Math.Sign(character.Position.Y - (currentHull.Rect.Y - currentHull.Rect.Height / 2)))
