219  public JobPrefab(ContentXElement element, JobsFile file) : base(file, element.GetAttributeIdentifier("identifier", ""))
220  {
221  SerializableProperty.DeserializeProperties(this, element);
222 
223  Name = TextManager.Get("JobName." + Identifier);
224  Description = TextManager.Get("JobDescription." + Identifier);
225  Element = element;
226 
227  var previewItems = new Dictionary<int, List<PreviewItem>>();
228 
229  int variant = 0;
230  foreach (var subElement in element.Elements())
231  {
232  switch (subElement.Name.ToString().ToLowerInvariant())
233  {
234  case "itemset":
235  ItemSets.Add(variant, subElement);
236  previewItems[variant] = new List<PreviewItem>();
237  loadItemIdentifiers(subElement, variant);
238  variant++;
239  break;
240  case "skills":
241  foreach (var skillElement in subElement.Elements())
242  {
243  Skills.Add(new SkillPrefab(skillElement));
244  }
245  break;
246  case "autonomousobjectives":
247  subElement.Elements().ForEach(order => AutonomousObjectives.Add(new AutonomousObjective(order)));
248  break;
249  case "appropriateobjectives":
250  case "appropriateorders":
251  subElement.Elements().ForEach(order => AppropriateOrders.Add(order.GetAttributeIdentifier("identifier", "")));
252  break;
253  case "jobicon":
254  Icon = new Sprite(subElement.FirstElement());
255  break;
256  case "jobiconsmall":
257  IconSmall = new Sprite(subElement.FirstElement());
258  break;
259  }
260  }

262  void loadItemIdentifiers(XElement parentElement, int variant)
263  {
264  foreach (XElement itemElement in parentElement.GetChildElements("Item"))
265  {
266  if (itemElement.Element("name") != null)
267  {
268  DebugConsole.ThrowError("Error in job config \"" + Name + "\" - use identifiers instead of names to configure the items.");
269  continue;
270  }
271 
272  Identifier itemIdentifier = itemElement.GetAttributeIdentifier("identifier", Identifier.Empty);
273  if (itemIdentifier.IsEmpty)
274  {
275  DebugConsole.ThrowError("Error in job config \"" + Name + "\" - item with no identifier.");
276  }
277  else
278  {
279  previewItems[variant].Add(new PreviewItem(itemIdentifier, itemElement.GetAttributeBool("showpreview", true)));
280  }
281  loadItemIdentifiers(itemElement, variant);
282  }
283  }
284 
285  PreviewItems = previewItems.Select(kvp => (kvp.Key, kvp.Value.ToImmutableArray()))
286  .ToImmutableDictionary();
287 
288  Variants = variant;
289 
290  Skills.Sort((x,y) => y.LevelRange.Start.CompareTo(x.LevelRange.Start));
