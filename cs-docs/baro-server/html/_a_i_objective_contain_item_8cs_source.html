107  protected override void Act(float deltaTime)
108  {
109  if (container?.Item == null)
110  {
111  Abandon = true;
112  return;
113  }
114  ItemToContain = item ?? character.Inventory.FindItem(i => CheckItem(i) && i.Container != container.Item, recursive: true);
115  if (ItemToContain != null)
116  {
117  if (!character.CanInteractWith(ItemToContain, checkLinked: false))
118  {
119  Abandon = true;
120  return;
121  }
122  if (character.CanInteractWith(container.Item, checkLinked: false))
123  {
124  if (RemoveExisting || (RemoveExistingWhenNecessary && !container.Inventory.CanBePut(ItemToContain)))
125  {
126  HumanAIController.UnequipContainedItems(container.Item, predicate: RemoveExistingPredicate, unequipMax: RemoveMax);
127  }
128  else if (RemoveEmpty)
129  {
130  HumanAIController.UnequipEmptyItems(container.Item);
131  }
132  Inventory originalInventory = ItemToContain.ParentInventory;
133  var slots = originalInventory?.FindIndices(ItemToContain);
134  if (container.Inventory.TryPutItem(ItemToContain, null))
135  {
136  if (MoveWholeStack && slots != null)
137  {
138  foreach (int slot in slots)
139  {
140  foreach (Item item in originalInventory.GetItemsAt(slot).ToList())
141  {
142  container.Inventory.TryPutItem(item, null);
143  }
144  }
145  }
146  IsCompleted = item != null || CountItems();
147  }
148  else
149  {
150  if (ItemToContain.ParentInventory == character.Inventory && character.IsInFriendlySub)
151  {
152  ItemToContain.Drop(character);
153  }
154  Abandon = true;
155  }
156  }
157  else
158  {
159  TryAddSubObjective(ref goToObjective, () => new AIObjectiveGoTo(container.Item, character, objectiveManager, getDivingGearIfNeeded: AllowToFindDivingGear)
160  {
161  TargetName = container.Item.Name,
162  AbortCondition = obj =>
163  container?.Item == null || container.Item.Removed || !container.Item.HasAccess(character) ||
164  (container.Item.GetRootContainer()?.OwnInventory?.Locked ?? false) ||
165  ItemToContain == null || ItemToContain.Removed ||
166  !ItemToContain.IsOwnedBy(character) || container.Item.GetRootInventoryOwner() is Character c && c != character,
167  SpeakIfFails = !objectiveManager.IsCurrentOrder<AIObjectiveCleanupItems>()
168  },
169  onAbandon: () => Abandon = true,
170  onCompleted: () => RemoveSubObjective(ref goToObjective));
171  }
172  }
173  else
174  {
175  if (character.Submarine == null)
176  {
177  Abandon = true;
178  }
179  else
180  {
181  // No matching items in the inventory, try to get an item
182  TryAddSubObjective(ref getItemObjective, () =>
183  new AIObjectiveGetItem(character, itemIdentifiers, objectiveManager, equip: Equip, checkInventory: checkInventory, spawnItemIfNotFound: spawnItemIfNotFound)
184  {

192  ItemCount = ItemCount,
