175  public override void Update(float deltaTime, Camera cam)
176  {
177  RefreshConnections();
178 
179  if (Timing.TotalTime > extraLoadSetTime + 1.0)
180  {
181  //Decay the extra load to 0 from either positive or negative
182  if (extraLoad > 0)
183  {
184  extraLoad = Math.Max(extraLoad - 1000.0f * deltaTime, 0);
185  }
186  else
187  {
188  extraLoad = Math.Min(extraLoad + 1000.0f * deltaTime, 0);
189  }
190  }
191 
192  if (!CanTransfer) { return; }
193 
194  if (isBroken)
195  {
196  SetAllConnectionsDirty();
197  isBroken = false;
198  }
199 
200  ApplyStatusEffects(ActionType.OnActive, deltaTime, null);
201 
202  float powerReadingOut = 0;
203  float loadReadingOut = ExtraLoad;
204  if (powerLoad < 0)
205  {
206  powerReadingOut = -powerLoad;
207  loadReadingOut = 0;
208  }
209 
210  if (powerOut != null && powerOut.Grid != null)
211  {
212  powerReadingOut = powerOut.Grid.Power;
213  loadReadingOut = powerOut.Grid.Load;
214  }
215 
216  if (prevSentPowerValue != (int)powerReadingOut || powerSignal == null)
217  {
218  prevSentPowerValue = (int)Math.Round(powerReadingOut);
219  powerSignal = prevSentPowerValue.ToString();
220  }
221  if (prevSentLoadValue != (int)loadReadingOut || loadSignal == null)
222  {
223  prevSentLoadValue = (int)Math.Round(loadReadingOut);
224  loadSignal = prevSentLoadValue.ToString();
225  }
226  item.SendSignal(powerSignal, "power_value_out");
227  item.SendSignal(loadSignal, "load_value_out");
228 
229  //if the item can't be fixed, don't allow it to break
230  if (!item.Repairables.Any() || !CanBeOverloaded) { return; }
231 
232  float maxOverVoltage = Math.Max(OverloadVoltage, 1.0f);
233 
234  Overload = Voltage > maxOverVoltage;
235 
236  if (Overload && (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
237  {
238  if (overloadCooldownTimer > 0.0f)
239  {
240  overloadCooldownTimer -= deltaTime;
241  return;
242  }
243 
244  //damage the item if voltage is too high (except if running as a client)
245  float prevCondition = item.Condition;
246  item.Condition -= deltaTime * 10.0f;
247 
248  if (item.Condition <= 0.0f && prevCondition > 0.0f)
249  {
250  overloadCooldownTimer = OverloadCooldown;
251 #if CLIENT
252  SoundPlayer.PlaySound("zap", item.WorldPosition, hullGuess: item.CurrentHull);
253  Vector2 baseVel = Rand.Vector(300.0f);
254  for (int i = 0; i < 10; i++)
255  {
256  var particle = GameMain.ParticleManager.CreateParticle("spark", item.WorldPosition,
257  baseVel + Rand.Vector(100.0f), 0.0f, item.CurrentHull);
258  if (particle != null) particle.Size *= Rand.Range(0.5f, 1.0f);
259  }
260 #endif
261  float currentIntensity = GameMain.GameSession?.EventManager != null ?
262  GameMain.GameSession.EventManager.CurrentIntensity : 0.5f;
263 
264  //higher probability for fires if the current intensity is low
265  if (FireProbability > 0.0f &&
266  Rand.Range(0.0f, 1.0f) < MathHelper.Lerp(FireProbability, FireProbability * 0.1f, currentIntensity))
267  {
268  new FireSource(item.WorldPosition);
269  }
270  }
271  }

284  protected void RefreshConnections()
285  {
286  var connections = item.Connections;
287  foreach (Connection c in connections)
288  {
289  if (!connectionDirty.ContainsKey(c))
290  {
291  connectionDirty[c] = true;
292  }
293  else if (!connectionDirty[c])
294  {
295  continue;
296  }
297 
298  //find all connections that are connected to this one (directly or via another PowerTransfer)
299  HashSet<Connection> tempConnected;
300  if (!connectedRecipients.ContainsKey(c))
301  {
302  tempConnected = new HashSet<Connection>();
303  connectedRecipients.Add(c, tempConnected);
304  }
305  else
306  {
307  tempConnected = connectedRecipients[c];
308  tempConnected.Clear();
309  //mark all previous recipients as dirty
310  foreach (Connection recipient in tempConnected)
311  {
312  var pt = recipient.Item.GetComponent<PowerTransfer>();
313  if (pt != null) { pt.connectionDirty[recipient] = true; }
314  }
315  }
316 
317  tempConnected.Add(c);
318  if (item.Condition > 0.0f)
319  {
320  GetConnected(c, tempConnected);
321  //go through all the PowerTransfers that we're connected to and set their connections to match the ones we just calculated
322  //(no need to go through the recursive GetConnected method again)
323  foreach (Connection recipient in tempConnected)
324  {
325  if (recipient == c) { continue; }
326  var recipientPowerTransfer = recipient.Item.GetComponent<PowerTransfer>();
327  if (recipientPowerTransfer == null) { continue; }
328  if (!recipientPowerTransfer.connectedRecipients.ContainsKey(recipient))
329  {
330  recipientPowerTransfer.connectedRecipients.Add(recipient, new HashSet<Connection>());
331  }
332  else
333  {
334  recipientPowerTransfer.connectedRecipients[recipient].Clear();
335  }
336  foreach (var connection in tempConnected)
337  {
338  recipientPowerTransfer.connectedRecipients[recipient].Add(connection);
339  }
340  recipientPowerTransfer.connectionDirty[recipient] = false;
341  }
342  }
343  connectionDirty[c] = false;
344  }
345  }

348  private void GetConnected(Connection c, HashSet<Connection> connected)
349  {
350  var recipients = c.Recipients;
351 
352  foreach (Connection recipient in recipients)
353  {
354  if (recipient == null || connected.Contains(recipient)) { continue; }
355 
356  Item it = recipient.Item;
357  if (it == null || it.Condition <= 0.0f) { continue; }
358 
359  connected.Add(recipient);
360 
361  var powerTransfer = it.GetComponent<PowerTransfer>();
362  if (powerTransfer != null && powerTransfer.CanTransfer && powerTransfer.IsActive)
363  {
364  GetConnected(recipient, connected);
365  }
366  }

416  public override void ReceiveSignal(Signal signal, Connection connection)
417  {
418  if (item.Condition <= 0.0f || connection.IsPower) { return; }
419  if (!connectedRecipients.ContainsKey(connection)) { return; }
420  if (!signalConnections.Contains(connection)) { return; }
421 
422  foreach (Connection recipient in connectedRecipients[connection])
423  {
424  if (recipient.Item == item || recipient.Item == signal.source) { continue; }
425 
426  signal.source?.LastSentSignalRecipients.Add(recipient);
427 
428  foreach (ItemComponent ic in recipient.Item.Components)
429  {
430  //other junction boxes don't need to receive the signal in the pass-through signal connections
431  //because we relay it straight to the connected items without going through the whole chain of junction boxes
432  if (ic is PowerTransfer && !(ic is RelayComponent)) { continue; }
433  ic.ReceiveSignal(signal, recipient);
434  }
435 
436  if (recipient.Effects != null && signal.value != "0" && !string.IsNullOrEmpty(signal.value))
437  {
438  foreach (StatusEffect effect in recipient.Effects)
439  {
440  recipient.Item.ApplyStatusEffect(effect, ActionType.OnUse, 1.0f);
441  }
