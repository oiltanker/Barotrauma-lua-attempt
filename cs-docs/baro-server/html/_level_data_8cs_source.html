142  Biome = locationConnection.Biome;
143  Type = LevelType.LocationConnection;
144  GenerationParams = LevelGenerationParams.GetRandom(Seed, LevelType.LocationConnection, Biome.Identifier);
145  Difficulty = locationConnection.Difficulty;
146 
147  float sizeFactor = MathUtils.InverseLerp(
148  MapGenerationParams.Instance.SmallLevelConnectionLength,
149  MapGenerationParams.Instance.LargeLevelConnectionLength,
150  locationConnection.Length);
151  int width = (int)MathHelper.Lerp(GenerationParams.MinWidth, GenerationParams.MaxWidth, sizeFactor);
152  Size = new Point(
153  (int)MathUtils.Round(width, Level.GridCellSize),
154  (int)MathUtils.Round(GenerationParams.Height, Level.GridCellSize));
155 
156  var rand = new MTRandom(ToolBox.StringToInt(Seed));
157  InitialDepth = (int)MathHelper.Lerp(GenerationParams.InitialDepthMin, GenerationParams.InitialDepthMax, (float)rand.NextDouble());
158  if (Biome.IsEndBiome)
159  {
160  HasHuntingGrounds = false;
161  HasBeaconStation = false;
162  }
163  else
164  {

166  HasBeaconStation = !HasHuntingGrounds && rand.NextDouble() < locationConnection.Locations.Select(l => l.Type.BeaconStationChance).Max();

190  public static LevelData CreateRandom(string seed = "", float? difficulty = null, LevelGenerationParams generationParams = null, bool requireOutpost = false)
191  {
192  if (string.IsNullOrEmpty(seed))
193  {
194  seed = Rand.Range(0, int.MaxValue, Rand.RandSync.ServerAndClient).ToString();
195  }
196 
197  Rand.SetSyncedSeed(ToolBox.StringToInt(seed));
198 
199  LevelType type = generationParams == null ?
200  (requireOutpost ? LevelType.Outpost : LevelType.LocationConnection) :
201  generationParams.Type;
202 
203  if (generationParams == null) { generationParams = LevelGenerationParams.GetRandom(seed, type); }
204  var biome =
205  Biome.Prefabs.FirstOrDefault(b => generationParams?.AllowedBiomeIdentifiers.Contains(b.Identifier) ?? false) ??
206  Biome.Prefabs.GetRandom(Rand.RandSync.ServerAndClient);
207 
208  var levelData = new LevelData(
209  seed,
210  difficulty ?? Rand.Range(30.0f, 80.0f, Rand.RandSync.ServerAndClient),
211  Rand.Range(0.0f, 1.0f, Rand.RandSync.ServerAndClient),
212  generationParams,
213  biome);
214  if (type == LevelType.LocationConnection)
215  {
216  float beaconRng = Rand.Range(0.0f, 1.0f, Rand.RandSync.ServerAndClient);
217  levelData.HasBeaconStation = beaconRng < 0.5f;
218  levelData.IsBeaconActive = beaconRng > 0.25f;
219  }
220  if (GameMain.GameSession?.GameMode != null)
221  {
222  foreach (Mission mission in GameMain.GameSession.GameMode.Missions)
223  {
224  mission.AdjustLevelData(levelData);
225  }
