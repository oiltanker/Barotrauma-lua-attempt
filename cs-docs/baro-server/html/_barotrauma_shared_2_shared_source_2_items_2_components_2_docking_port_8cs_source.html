369  public int GetDir(DockingPort dockingTarget = null)
370  {
371  int forcedDockingDir = GetForcedDockingDir();
372  if (forcedDockingDir != 0) { return forcedDockingDir; }
373  if (dockingTarget != null)
374  {
375  forcedDockingDir = -dockingTarget.GetForcedDockingDir();
376  if (forcedDockingDir != 0) { return forcedDockingDir; }
377  }
378 
379  if (DockingDir != 0) { return DockingDir; }
380 
381  if (Door != null && Door.LinkedGap.linkedTo.Count > 0)
382  {
383  Hull refHull = null;
384  float largestHullSize = 0.0f;
385  foreach (MapEntity linked in Door.LinkedGap.linkedTo)
386  {
387  if (!(linked is Hull hull)) { continue; }
388  if (hull.Volume > largestHullSize)
389  {
390  refHull = hull;
391  largestHullSize = hull.Volume;
392  }
393  }
394  if (refHull != null)
395  {
396  return IsHorizontal ?
397  Math.Sign(Door.Item.WorldPosition.X - refHull.WorldPosition.X) :
398  Math.Sign(Door.Item.WorldPosition.Y - refHull.WorldPosition.Y);
399  }
400  }
401  if (dockingTarget?.Door?.LinkedGap != null && dockingTarget.Door.LinkedGap.linkedTo.Count > 0)
402  {
403  Hull refHull = null;
404  float largestHullSize = 0.0f;
405  foreach (MapEntity linked in dockingTarget.Door.LinkedGap.linkedTo)
406  {
407  if (!(linked is Hull hull)) { continue; }
408  if (hull.Volume > largestHullSize)
409  {
410  refHull = hull;
411  largestHullSize = hull.Volume;
412  }
413  }
414  if (refHull != null)
415  {
416  return IsHorizontal ?
417  Math.Sign(refHull.WorldPosition.X - dockingTarget.Door.Item.WorldPosition.X) :
418  Math.Sign(refHull.WorldPosition.Y - dockingTarget.Door.Item.WorldPosition.Y);
419  }
420  }
421  if (dockingTarget != null)
422  {
423  int dir = IsHorizontal ?
424  Math.Sign(dockingTarget.item.WorldPosition.X - item.WorldPosition.X) :
425  Math.Sign(dockingTarget.item.WorldPosition.Y - item.WorldPosition.Y);
426  if (dir != 0) { return dir; }
427  }
428  if (item.Submarine != null)
429  {
430  return IsHorizontal ?
431  Math.Sign(item.WorldPosition.X - item.Submarine.WorldPosition.X) :
432  Math.Sign(item.WorldPosition.Y - item.Submarine.WorldPosition.Y);
433  }
434 
435  return 0;
436  }

460  wire.Hidden = true;
461 
462  if (Item.Connections == null) { return; }

465  if (powerConnection == null) { return; }

520  private void CreateHulls()
521  {
522  var hullRects = new Rectangle[] { item.WorldRect, DockingTarget.item.WorldRect };
523  var subs = new Submarine[] { item.Submarine, DockingTarget.item.Submarine };
524 
525  bodies = new Body[4];
526 
527  if (DockingTarget.Door != null)
528  {
529  CreateDoorBody();
530  }
531 
532  if (Door != null)
533  {
534  DockingTarget.CreateDoorBody();
535  }
536 
537  if (IsHorizontal)
538  {
539  if (hullRects[0].Center.X > hullRects[1].Center.X)
540  {
541  hullRects = new Rectangle[] { DockingTarget.item.WorldRect, item.WorldRect };
542  subs = new Submarine[] { DockingTarget.item.Submarine, item.Submarine };
543  }
544 
545  int scaledDockedDistance = (int)(DockedDistance / 2 * item.Scale);
546  hullRects[0] = new Rectangle(hullRects[0].Center.X, hullRects[0].Y, scaledDockedDistance, hullRects[0].Height);
547  hullRects[1] = new Rectangle(hullRects[1].Center.X - scaledDockedDistance, hullRects[1].Y, scaledDockedDistance, hullRects[1].Height);
548 
549  //expand hulls if needed, so there's no empty space between the sub's hulls and docking port hulls
550  int leftSubRightSide = int.MinValue, rightSubLeftSide = int.MaxValue;
551  foreach (Hull hull in Hull.HullList)
552  {
553  for (int i = 0; i < 2; i++)
554  {
555  if (hull.Submarine != subs[i]) { continue; }
556  if (hull.WorldRect.Y - 5 < hullRects[i].Y - hullRects[i].Height) { continue; }
557  if (hull.WorldRect.Y - hull.WorldRect.Height + 5 > hullRects[i].Y) { continue; }
558 
559  if (i == 0) //left hull
560  {
561  if (hull.WorldPosition.X > hullRects[0].Center.X) { continue; }
562  leftSubRightSide = Math.Max(hull.WorldRect.Right, leftSubRightSide);
563  }
564  else //upper hull
565  {
566  if (hull.WorldPosition.X < hullRects[1].Center.X) { continue; }
567  rightSubLeftSide = Math.Min(hull.WorldRect.X, rightSubLeftSide);
568  }
569  }
570  }
571 
572  if (leftSubRightSide == int.MinValue || rightSubLeftSide == int.MaxValue)
573  {
574  DebugConsole.NewMessage("Creating hulls between docking ports failed. Could not find a hull next to the docking port.");
575  return;
576  }
577 
578  //expand left hull to the rightmost hull of the sub at the left side
579  //(unless the difference is more than 100 units - if the distance is very large
580  //there's something wrong with the positioning of the docking ports or submarine hulls)
581  int leftHullDiff = (hullRects[0].X - leftSubRightSide) + 5;
582  if (leftHullDiff > 0)
583  {
584  if (leftHullDiff > 100)
585  {
586  DebugConsole.NewMessage("Creating hulls between docking ports failed. The leftmost docking port seems to be very far from any hulls in the left-side submarine.");
587  return;
588  }
589  else
590  {
591  hullRects[0].X -= leftHullDiff;
592  hullRects[0].Width += leftHullDiff;
593  }
594  }
595 
596  int rightHullDiff = (rightSubLeftSide - hullRects[1].Right) + 5;
597  if (rightHullDiff > 0)
598  {
599  if (rightHullDiff > 100)
600  {
601  DebugConsole.NewMessage("Creating hulls between docking ports failed. The rightmost docking port seems to be very far from any hulls in the right-side submarine.");
602  return;
603  }
604  else
605  {
606  hullRects[1].Width += rightHullDiff;
607  }
608  }
609 
610  int expand = 5;
611  for (int i = 0; i < 2; i++)
612  {
613  hullRects[i].X -= expand;
614  hullRects[i].Width += expand * 2;
615  hullRects[i].Location -= MathUtils.ToPoint(subs[i].WorldPosition - subs[i].HiddenSubPosition);
616  hulls[i] = new Hull(MapEntityPrefab.Find(null, "hull"), hullRects[i], subs[i]);
617  hulls[i].RoomName = IsHorizontal ? "entityname.dockingport" : "entityname.dockinghatch";
618  hulls[i].AddToGrid(subs[i]);
619  hulls[i].FreeID();
620 
621  for (int j = 0; j < 2; j++)
622  {
623  bodies[i + j * 2] = GameMain.World.CreateEdge(
624  ConvertUnits.ToSimUnits(new Vector2(hullRects[i].X, hullRects[i].Y - hullRects[i].Height * j)),
625  ConvertUnits.ToSimUnits(new Vector2(hullRects[i].Right, hullRects[i].Y - hullRects[i].Height * j)));
626  }
627  }
628 
629  if (rightHullDiff <= 100 && hulls[0].Submarine != null)
630  {
631  outsideBlocker = hulls[0].Submarine.PhysicsBody.FarseerBody.CreateRectangle(
632  ConvertUnits.ToSimUnits(hullRects[0].Width + hullRects[1].Width),
633  ConvertUnits.ToSimUnits(hullRects[0].Height),
634  density: 0.0f,
635  offset: ConvertUnits.ToSimUnits(new Vector2(hullRects[0].Right, hullRects[0].Y - hullRects[0].Height / 2) - hulls[0].Submarine.HiddenSubPosition));
636  outsideBlocker.UserData = this;
637  }
638 
639  gap = new Gap(new Rectangle(hullRects[0].Right - 2, hullRects[0].Y, 4, hullRects[0].Height), true, subs[0]);
640  }
641  else
642  {
643  if (hullRects[0].Center.Y > hullRects[1].Center.Y)
644  {
645  hullRects = new Rectangle[] { DockingTarget.item.WorldRect, item.WorldRect };
646  subs = new Submarine[] { DockingTarget.item.Submarine, item.Submarine };
647  }
648 
649  int scaledDockedDistance = (int)(DockedDistance / 2 * item.Scale);
650  hullRects[0] = new Rectangle(hullRects[0].X, hullRects[0].Y - hullRects[0].Height / 2 + scaledDockedDistance, hullRects[0].Width, scaledDockedDistance);
651  hullRects[1] = new Rectangle(hullRects[1].X, hullRects[1].Y - hullRects[1].Height / 2, hullRects[1].Width, scaledDockedDistance);
652 
653  //expand hulls if needed, so there's no empty space between the sub's hulls and docking port hulls
654  int upperSubBottom = int.MaxValue, lowerSubTop = int.MinValue;
655  foreach (Hull hull in Hull.HullList)
656  {
657  for (int i = 0; i < 2; i++)
658  {
659  if (hull.Submarine != subs[i]) { continue; }
660  if (hull.WorldRect.Right - 5 < hullRects[i].X) { continue; }
661  if (hull.WorldRect.X + 5 > hullRects[i].Right) { continue; }
662 
663  if (i == 0) //lower hull
664  {
