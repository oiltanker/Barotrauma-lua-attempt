33 class PermissionPreset
34 {
35 public static List<PermissionPreset> List = new List<PermissionPreset>();
36
37 public readonly LocalizedString Name;
38 public readonly LocalizedString Description;
39 public readonly ClientPermissions Permissions;
40 public readonly HashSet<DebugConsole.Command> PermittedCommands;
41
42 public PermissionPreset(XElement element)
43 {
44 string name = element.GetAttributeString("name", "");
45 Name = TextManager.Get("permissionpresetname." + name).Fallback(name);
46 Description = TextManager.Get("permissionpresetdescription." + name) .Fallback(element.GetAttributeString("description", ""));
47
48 string permissionsStr = element.GetAttributeString("permissions", "");
49 if (!Enum.TryParse(permissionsStr, out Permissions))
50 {
51 DebugConsole.ThrowError("Error in permission preset \"" + Name + "\" - " + permissionsStr + " is not a valid permission!");
52 }
53
54 PermittedCommands = new HashSet<DebugConsole.Command>();
55 if (Permissions.HasFlag(ClientPermissions.ConsoleCommands))
56 {
57 foreach (var subElement in element.Elements())
58 {
59 if (!subElement.Name.ToString().Equals("command", StringComparison.OrdinalIgnoreCase)) { continue; }
60 string commandName = subElement.GetAttributeString("name", "");
61
62 DebugConsole.Command command = DebugConsole.FindCommand(commandName);
63 if (command == null)
64 {
65#if SERVER
66 DebugConsole.ThrowError("Error in permission preset \"" + Name + "\" - " + commandName + "\" is not a valid console command.");
67#endif
68 continue;
69 }
70
71 PermittedCommands.Add(command);
72 }
73 }
74 }
75
76 public static void LoadAll(string file)
77 {
78 if (!File.Exists(file)) { return; }
79
80 XDocument doc = XMLExtensions.TryLoadXml(file);
81 if (doc == null) { return; }
82
83 List.Clear();
84 foreach (XElement element in doc.Root.Elements())
85 {
86 List.Add(new PermissionPreset(element));
87 }
88 }
89
90 public bool MatchesPermissions(ClientPermissions permissions, HashSet<DebugConsole.Command> permittedConsoleCommands)
91 {
92 return permissions == this.Permissions && PermittedCommands.SequenceEqual(permittedConsoleCommands);
93 }
