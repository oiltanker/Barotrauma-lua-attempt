495  private void ReduceMatchingAfflictions(float amount, ActionType? treatmentAction)
496  {
497  if (matchingAfflictions.Count == 0) { return; }
498 
499  float reduceAmount = amount / matchingAfflictions.Count;
500  for (int i = matchingAfflictions.Count - 1; i >= 0; i--)
501  {
502  var matchingAffliction = matchingAfflictions[i];
503 
504  if (matchingAffliction.Strength < reduceAmount)
505  {
506  float surplus = reduceAmount - matchingAffliction.Strength;
507  amount -= matchingAffliction.Strength;
508  matchingAffliction.Strength = 0.0f;
509  matchingAfflictions.RemoveAt(i);
510  if (i == 0) { i = matchingAfflictions.Count; }
511  if (i > 0) { reduceAmount += surplus / i; }
512  SteamAchievementManager.OnAfflictionRemoved(matchingAffliction, Character);
513  }
514  else
515  {
516  matchingAffliction.Strength -= reduceAmount;
517  amount -= reduceAmount;
518  if (treatmentAction != null)
519  {
520  if (treatmentAction.Value == ActionType.OnUse)
521  {
522  matchingAffliction.AppliedAsSuccessfulTreatmentTime = Timing.TotalTime;
523  }
524  else if (treatmentAction.Value == ActionType.OnFailure)
525  {

589  public float GetLimbDamage(Limb limb, string afflictionType = null)
590  {
591  float damageStrength;
592  if (limb.IsSevered)
593  {
594  return 1;
595  }
596  else
597  {
598  // Instead of using the limbhealth count here, I think it's best to define the max vitality per limb roughly with a constant value.
599  // Therefore with e.g. 80 health, the max damage per limb would be 40.
600  // Having at least 40 damage on both legs would cause maximum limping.
601  float max = MaxVitality / 2;
602  if (string.IsNullOrEmpty(afflictionType))
603  {
604  float damage = GetAfflictionStrength("damage", limb, true);
605  float bleeding = GetAfflictionStrength("bleeding", limb, true);
606  float burn = GetAfflictionStrength("burn", limb, true);
607  damageStrength = Math.Min(damage + bleeding + burn, max);
608  }
609  else
610  {
611  damageStrength = Math.Min(GetAfflictionStrength("damage", limb, true), max);
612  }
613  return damageStrength / max;
614  }
