83  public void ServerEventWrite(IWriteMessage msg, Client c, NetEntityEvent.IData extraData = null)
84  {
85  if (!(extraData is IEventData eventData)) { throw new Exception($"Malformed hull event: expected {nameof(Hull)}.{nameof(IEventData)}"); }
86  msg.WriteRangedInteger((int)eventData.EventType, (int)EventType.MinValue, (int)EventType.MaxValue);
87 
88  switch (eventData)
89  {
90  case StatusEventData statusEventData:
91  msg.WriteRangedSingle(MathHelper.Clamp(OxygenPercentage, 0.0f, 100.0f), 0.0f, 100.0f, 8);
92  SharedStatusWrite(msg);
93  break;
94  case BackgroundSectionsEventData backgroundSectionsEventData:
95  SharedBackgroundSectionsWrite(msg, backgroundSectionsEventData);
96  break;
97  case DecalEventData decalEventData:
98  msg.WriteRangedInteger(decals.Count, 0, MaxDecalsPerHull);
99  foreach (Decal decal in decals)
100  {
101  msg.Write(decal.Prefab.UintIdentifier);
102  msg.Write((byte)decal.SpriteIndex);
103  float normalizedXPos = MathHelper.Clamp(MathUtils.InverseLerp(0.0f, rect.Width, decal.CenterPosition.X), 0.0f, 1.0f);
104  float normalizedYPos = MathHelper.Clamp(MathUtils.InverseLerp(-rect.Height, 0.0f, decal.CenterPosition.Y), 0.0f, 1.0f);
105  msg.WriteRangedSingle(normalizedXPos, 0.0f, 1.0f, 8);
106  msg.WriteRangedSingle(normalizedYPos, 0.0f, 1.0f, 8);
107  msg.WriteRangedSingle(decal.Scale, 0f, 2f, 12);
108  }
109  break;
110  case BallastFloraEventData ballastFloraEventData:
111  ballastFloraEventData.Behavior.ServerWrite(msg, ballastFloraEventData.SubEventData);
112  break;
113  default:
114  throw new Exception($"Malformed hull event: did not expect {eventData.GetType().Name}");
115  }

119  public void ServerEventRead(IReadMessage msg, Client c)
120  {
121  EventType eventType = (EventType)msg.ReadRangedInteger((int)EventType.MinValue, (int)EventType.MaxValue);
122  switch (eventType)
123  {
124  case EventType.Status:
125  SharedStatusRead(
126  msg,
127  out float newWaterVolume,
128  out NetworkFireSource[] newFireSources);
129 
130  if (!c.HasPermission(ClientPermissions.ConsoleCommands) ||
131  !c.PermittedConsoleCommands.Any(command => command.names.Contains("fire") || command.names.Contains("editfire")))
132  {
133  return;
134  }
135 
136  WaterVolume = newWaterVolume;
137 
138  for (int i = 0; i < newFireSources.Length; i++)
139  {
140  Vector2 pos = newFireSources[i].Position;
141  float size = newFireSources[i].Size;
142 
143  var newFire = i < FireSources.Count ?
144  FireSources[i] :
145  new FireSource(Submarine == null ? pos : pos + Submarine.Position, null, true);
146  newFire.Position = pos;
147  newFire.Size = new Vector2(size, newFire.Size.Y);
148 
149  //ignore if the fire wasn't added to this room (invalid position)?
150  if (!FireSources.Contains(newFire))
151  {
152  newFire.Remove();
153  continue;
154  }
155  }
156 
157  for (int i = FireSources.Count - 1; i >= newFireSources.Length; i--)
158  {
159  FireSources[i].Remove();
160  if (i < FireSources.Count)
161  {
162  FireSources.RemoveAt(i);
163  }
164  }
165  break;
166  case EventType.BackgroundSections:
167  SharedBackgroundSectionRead(
168  msg,
169  bsnu =>
170  {
171  int i = bsnu.SectionIndex;
172  Color color = bsnu.Color;
173  float colorStrength = bsnu.ColorStrength;
174 
175  #warning TODO: verify the client is close enough to this hull to paint it, that the sprayer is functional and that the color matches
176  if (!(c.Character is { AllowInput: true })) { return; }
177  if (c.Character.HeldItems.All(it => it.GetComponent<Sprayer>() == null)) { return; }
178 
179  BackgroundSections[i].SetColorStrength(colorStrength);
180  BackgroundSections[i].SetColor(color);
181  },
182  out int sectorToUpdate);
183  //add to pending updates to notify other clients as well
184  pendingSectionUpdates.Add(sectorToUpdate);
185  break;

187  byte decalIndex = msg.ReadByte();
188  float decalAlpha = msg.ReadRangedSingle(0.0f, 1.0f, 255);
189  if (decalIndex < 0 || decalIndex >= decals.Count) { return; }
190  if (c.Character != null && c.Character.AllowInput && c.Character.HeldItems.Any(it => it.GetComponent<Sprayer>() != null))
191  {
192  decals[decalIndex].BaseAlpha = decalAlpha;
193  }
194  decalUpdatePending = true;
195  break;
196  default:
197  throw new Exception($"Malformed incoming hull event: {eventType} is not a supported event type");
198  }
