173  foreach (PathNode node in nodes)

176  }
177 
178  // First calculate the temp positions for all nodes.
179  foreach (PathNode node in nodes)
180  {
181  node.TempPosition = node.Position;
182  var wpSub = node.Waypoint.Submarine;
183  if (hostSub != null && wpSub == null)
184  {
185  // inside and targeting outside
186  node.TempPosition -= hostSub.SimPosition;
187  }
188  else if (wpSub != null && hostSub != null && wpSub != hostSub)
189  {
190  // different subs
191  node.TempPosition -= hostSub.SimPosition - wpSub.SimPosition;
192  }
193  else if (hostSub == null && wpSub != null)
194  {
195  // Outside and targeting inside
196  node.TempPosition += wpSub.SimPosition;
197  }
198  }
199 
200  //sort nodes roughly according to distance
201  sortedNodes.Clear();
202  PathNode startNode = null;
203  foreach (PathNode node in nodes)
204  {
205  float xDiff = Math.Abs(start.X - node.TempPosition.X);
206  float yDiff = Math.Abs(start.Y - node.TempPosition.Y);
207  if (InsideSubmarine && !(node.Waypoint.Submarine?.Info?.IsRuin ?? false))
208  {
209  //higher cost for vertical movement when inside the sub
210  if (yDiff > 1.0f && node.Waypoint.Ladders == null && node.Waypoint.Stairs == null)
211  {
212  yDiff += 10.0f;
213  }
214  node.TempDistance = xDiff + yDiff * 10.0f;
215  }
216  else
217  {
218  node.TempDistance = xDiff + yDiff;
219  }
220 
221  //much higher cost to waypoints that are outside
222  if (node.Waypoint.CurrentHull == null && ApplyPenaltyToOutsideNodes) { node.TempDistance *= 10.0f; }
223 
224  //optimization: node extremely far, don't try to use it as a start node
225  if (node.TempDistance > (InsideSubmarine ? 100.0f : 800.0f))
226  {
227  continue;
228  }
229  //optimization: node extremely close (< 1 m). If it's valid, choose it as the start node and skip the more exhaustive search for the closest one
230  if (node.TempDistance < 1.0f)
231  {
232  if (IsValidStartNode(node))
233  {
234  startNode = node;
235  break;
236  }
237  }
238  //prefer nodes that are closer to the end position
239  node.TempDistance += (Math.Abs(end.X - node.TempPosition.X) + Math.Abs(end.Y - node.TempPosition.Y)) / 100.0f;
240 
241  int i = 0;
242  while (i < sortedNodes.Count && sortedNodes[i].TempDistance < node.TempDistance)
243  {
244  i++;
245  }
246  sortedNodes.Insert(i, node);
247  }
248 
249  //find the most suitable start node, starting from the ones that are the closest
250  if (startNode == null)
251  {
252  foreach (PathNode node in sortedNodes)
253  {
254  if (IsValidStartNode(node))
255  {
256  startNode = node;
257  break;

259  }
260  }
261 
262  if (startNode == null)
263  {
264 #if DEBUG

270  //sort nodes again, now based on distance from the end position
271  sortedNodes.Clear();
272  PathNode endNode = null;
273  foreach (PathNode node in nodes)
274  {
275  node.TempDistance = Vector2.DistanceSquared(end, node.TempPosition);
276  if (InsideSubmarine)
277  {
278  if (ApplyPenaltyToOutsideNodes)
279  {
280  //much higher cost to waypoints that are outside
281  if (node.Waypoint.CurrentHull == null) { node.TempDistance *= 10.0f; }
282  }
283  //avoid stopping at a doorway
284  if (node.Waypoint.ConnectedDoor != null) { node.TempDistance *= 10.0f; }
285  //avoid stopping at a ladder
286  if (node.Waypoint.Ladders != null) { node.TempDistance *= 10.0f; }
287  }
288  //optimization: node extremely far (> 100m / 800 m) from the end position, don't try to use it as an end node
289  if (node.TempDistance > (InsideSubmarine ? 100.0f * 100.0f : 800.0f * 800.0f))
290  {
291  continue;
292  }
293  //optimization: node extremely close (< 1 m). If it's valid, choose it as the end node and skip the more exhaustive search for the closest one
294  if (node.TempDistance < 1.0f)
295  {
296  if (IsValidEndNode(node))
297  {
298  endNode = node;
299  break;
300  }
301  }
302  int i = 0;
303  while (i < sortedNodes.Count && sortedNodes[i].TempDistance < node.TempDistance)
304  {
305  i++;
306  }
307  sortedNodes.Insert(i, node);
308  }
309  if (endNode == null)
310  {
311  //find the most suitable end node, starting from the ones closest to the end position
312  foreach (PathNode node in sortedNodes)
313  {
314  if (IsValidEndNode(node))

320  }

379  private SteeringPath FindPath(PathNode start, PathNode end, Func<PathNode, bool> filter = null, string errorMsgStr = "", float minGapSize = 0)
380  {
381  if (start == end)
382  {
383  var path1 = new SteeringPath();
384  path1.AddNode(start.Waypoint);
385  return path1;
386  }
387 
388  foreach (PathNode node in nodes)
389  {
390  node.Parent = null;
391  node.state = 0;
392  node.F = 0.0f;
393  node.G = 0.0f;
394  node.H = 0.0f;
395  }
396 
397  start.state = 1;
398  while (true)
399  {
400  PathNode currNode = null;
401  float dist = float.MaxValue;
402  foreach (PathNode node in nodes)
403  {
404  if (node.state != 1 || node.F > dist) { continue; }
405  if (isCharacter && node.Waypoint.isObstructed) { continue; }
406  if (filter != null && !filter(node)) { continue; }
407  if (node.IsBlocked()) { continue; }
408  if (node.Waypoint.ConnectedGap != null)
409  {
410  if (!CanFitThroughGap(node.Waypoint.ConnectedGap, minGapSize)) { continue; }
411  }
412  dist = node.F;
413  currNode = node;
414  }
415 
416  if (currNode == null || currNode == end) { break; }
417 
418  currNode.state = 2;
419 
420  for (int i = 0; i < currNode.connections.Count; i++)
421  {
422  PathNode nextNode = currNode.connections[i];
423 
424  //a node that hasn't been searched yet
425  if (nextNode.state == 0)
426  {
427  nextNode.H = Vector2.Distance(nextNode.Position, end.Position);
428 
429  float penalty = 0.0f;
430  if (GetNodePenalty != null)
431  {
432  float? nodePenalty = GetNodePenalty(currNode, nextNode);
433  if (nodePenalty == null)
434  {
435  nextNode.state = -1;
436  continue;
437  }
438  penalty = nodePenalty.Value;
439  }
440 
441  nextNode.G = currNode.G + currNode.distances[i] + penalty;
442  nextNode.F = nextNode.G + nextNode.H;
443  nextNode.Parent = currNode;
444  nextNode.state = 1;
445  }
446  //node that has been searched
447  else if (nextNode.state == 1 || nextNode.state == -1)
448  {
449  float tempG = currNode.G + currNode.distances[i];
450 
451  if (GetNodePenalty != null)
452  {
453  float? nodePenalty = GetNodePenalty(currNode, nextNode);
454  if (nodePenalty == null) { continue; }
455  tempG += nodePenalty.Value;
456  }
457 
458  //only use if this new route is better than the
459  //route the node was a part of
460  if (tempG < nextNode.G)
461  {
462  nextNode.G = tempG;
463  nextNode.F = nextNode.G + nextNode.H;
464  nextNode.Parent = currNode;
465  nextNode.state = 1;
466  }
467  }
468  }
469  }
470 
471  if (end.state == 0 || end.Parent == null)
472  {
473 #if DEBUG
474  if (errorMsgStr != null)
475  {
476  DebugConsole.NewMessage("Path not found. " + errorMsgStr, Color.Yellow);
477  }
478 #endif
479  return new SteeringPath(true);
480  }
481 
482  SteeringPath path = new SteeringPath();
483  List<WayPoint> finalPath = new List<WayPoint>();
484 
485  PathNode pathNode = end;
486  while (pathNode != start && pathNode != null)
487  {
488  finalPath.Add(pathNode.Waypoint);
489 
490  //(there was one bug report that seems to have been caused by this loop never terminating:
491  //couldn't reproduce or figure out what caused it, but here's a workaround that prevents the game from crashing in case it happens again)
492 
493  //should be fixed now, was most likely caused by the parent fields of the nodes not being cleared before starting the pathfinding
494  if (finalPath.Count > nodes.Count)
495  {
496 #if DEBUG
497  DebugConsole.ThrowError("Pathfinding error: constructing final path failed");
498 #endif
499  return new SteeringPath(true);
500  }
501 
502  path.Cost += pathNode.F;
503  pathNode = pathNode.Parent;
504  }
505 
506  finalPath.Add(start.Waypoint);
507  for (int i = finalPath.Count - 1; i >= 0; i--)
508  {
509  path.AddNode(finalPath[i]);
510  }
511  System.Diagnostics.Debug.Assert(finalPath.Count == path.Nodes.Count);
512 
513  return path;
