79  public void Update(float deltaTime)
80  {
81  if (ShouldEndRound) { return; }
82 
83 #if DISABLE_MISSIONS
84  return;
85 #endif
86  if (Missions.Any())
87  {
88  bool missionCompleted = false;
89  bool gameShouldEnd = false;
90  CharacterTeamType winningTeam = CharacterTeamType.None;
91  foreach (var mission in Missions)
92  {
93  mission.Value.Update(deltaTime, () =>
94  {
95  switch (mission.Key)
96  {
97  case CharacterTeamType.Team1:
98  winningTeam = (winningTeam == CharacterTeamType.None) ? CharacterTeamType.Team2 : CharacterTeamType.None;
99  break;
100  case CharacterTeamType.Team2:
101  winningTeam = (winningTeam == CharacterTeamType.None) ? CharacterTeamType.Team1 : CharacterTeamType.None;
102  break;
103  default:
104  break;
105  }
106  gameShouldEnd = true;
107  });
108  if (!gameShouldEnd && mission.Value.IsCompleted)
109  {
110  missionCompleted = true;
111  foreach (var traitor in mission.Value.Traitors.Values)
112  {
113  traitor.UpdateCurrentObjective("", mission.Value.Identifier);
114  }
115  }
116  }
117  if (gameShouldEnd)
118  {
119  GameMain.GameSession.WinningTeam = winningTeam;
120  ShouldEndRound = true;
121  return;
122  }
123  if (missionCompleted)
124  {
125  Missions.Clear();
126  startCountdown = MathHelper.Lerp(server.ServerSettings.TraitorsMinRestartDelay, server.ServerSettings.TraitorsMaxRestartDelay, (float)RandomDouble());
127  }
128  }
129  else if (startCountdown > 0.0f && server.GameStarted)
130  {
131  startCountdown -= deltaTime;
132  if (startCountdown <= 0.0f)
133  {
134  int playerCharactersCount = server.ConnectedClients.Sum(client => client.Character != null && !client.Character.IsDead ? 1 : 0);
135  if (playerCharactersCount < server.ServerSettings.TraitorsMinPlayerCount)
136  {
137  startCountdown = MathHelper.Lerp(server.ServerSettings.TraitorsMinRestartDelay, server.ServerSettings.TraitorsMaxRestartDelay, (float)RandomDouble());
138  return;
139  }
140  if (Character.CharacterList.Count(c => !c.IsDead && c.TeamID == CharacterTeamType.Team1 || c.TeamID == CharacterTeamType.Team2) <= 1)
141  {
142  return;
143  }
144  if (GameMain.GameSession.Missions.Any(m => m is CombatMission))
145  {
146  var teamIds = new[] { CharacterTeamType.Team1, CharacterTeamType.Team2 };
147  foreach (var teamId in teamIds)
148  {
149  if (server.ConnectedClients.Count(c => c.Character != null && !c.Character.IsDead && c.TeamID == teamId) < 2)
150  {
151  continue;
152  }
153  var mission = TraitorMissionPrefab.RandomPrefab()?.Instantiate();
154  if (mission != null)
155  {
156  Missions.Add(teamId, mission);
157  }
158  }
159  var canBeStartedCount = Missions.Sum(mission => mission.Value.CanBeStarted(server, this, mission.Key) ? 1 : 0);
160  if (canBeStartedCount >= Missions.Count)
161  {
162  var startSuccessCount = Missions.Sum(mission => mission.Value.Start(server, this, mission.Key) ? 1 : 0);
163  if (startSuccessCount >= Missions.Count)
164  {
165  return;
166  }
167  }
168  }
169  else
170  {
171  var mission = TraitorMissionPrefab.RandomPrefab()?.Instantiate();
172  if (mission != null)
173  {
174  if (mission.CanBeStarted(server, this, CharacterTeamType.None))
175  {
176  if (mission.Start(server, this, CharacterTeamType.None))
177  {
178  Missions.Add(CharacterTeamType.None, mission);
179  return;
180  }
181  }
182  }
183  }
184  Missions.Clear();
185  startCountdown = MathHelper.Lerp(server.ServerSettings.TraitorsMinRestartDelay, server.ServerSettings.TraitorsMaxRestartDelay, (float)RandomDouble());
186  }
187  }
188  }
