386 foreach (List<MapEntity> entityList in entities.Values)
387 {
388 allEntities.AddRange(entityList);
389 }
390

392 {

395 Submarine.RepositionEntities(module.Offset + sub.HiddenSubPosition, entities[module]);

417 }

419 {
420 hull.WaterVolume = hull.Volume * Rand.Range(generationParams.MinWaterPercentage, generationParams.MaxWaterPercentage, Rand.RandSync.ServerAndClient) * 0.01f;
421 }
422 }
423 }
424 }
425 }
426
427 return allEntities;
428 }

434 private static List<Identifier> SelectModules(IEnumerable<SubmarineInfo> modules, OutpostGenerationParams generationParams)
435 {
436 int totalModuleCount = generationParams.TotalModuleCount;
437 var pendingModuleFlags = new List<Identifier>();
438 bool availableModulesFound = true;
439
440 Identifier initialModuleFlag = generationParams.ModuleCounts.FirstOrDefault().Key;
441 pendingModuleFlags.Add(initialModuleFlag);
442 while (pendingModuleFlags.Count < totalModuleCount && availableModulesFound)
443 {
444 availableModulesFound = false;
445 foreach (var moduleFlag in generationParams.ModuleCounts)
446 {
447 if (pendingModuleFlags.Count(m => m == moduleFlag.Key) >= generationParams.GetModuleCount(moduleFlag.Key))
448 {
449 continue;
450 }
451 if (!modules.Any(m => m.OutpostModuleInfo.ModuleFlags.Contains(moduleFlag.Key)))
452 {
453 DebugConsole.ThrowError($"Failed to add a module to the outpost (no modules with the flag \"{moduleFlag.Key}\" found).");
454 continue;
455 }
456 availableModulesFound = true;
457 pendingModuleFlags.Add(moduleFlag.Key);
458 }
459 }
460 using (MD5 md5 = MD5.Create())
461 {
462 pendingModuleFlags.Sort((i1, i2) => (int)ToolBox.StringToUInt32Hash(i1.Value.ToLowerInvariant(), md5) - (int)ToolBox.StringToUInt32Hash(i2.Value.ToLowerInvariant(), md5));
463 }
464 pendingModuleFlags.Shuffle(Rand.RandSync.ServerAndClient);
465 while (pendingModuleFlags.Count < totalModuleCount)
466 {
467 //don't place "none" modules at the end because
468 // a. "filler rooms" at the end of a hallway are pointless
469 // b. placing the unnecessary filler rooms first give more options for the placement of the more important modules
470 pendingModuleFlags.Insert(Rand.Int(pendingModuleFlags.Count - 1, Rand.RandSync.ServerAndClient), "none".ToIdentifier());
471 }
472
473 //make sure the initial module is inserted first
474 pendingModuleFlags.Remove(initialModuleFlag);
475 pendingModuleFlags.Insert(0, initialModuleFlag);
476
477 if (pendingModuleFlags.Count > totalModuleCount)
478 {
479 DebugConsole.ThrowError($"Error during outpost generation. {pendingModuleFlags.Count} modules set to be used the outpost, but total module count is only {totalModuleCount}. Leaving out some of the modules...");
480 int removeCount = pendingModuleFlags.Count - totalModuleCount;
481 for (int i = 0; i < removeCount; i++)
482 {
483 pendingModuleFlags.Remove(pendingModuleFlags.Last());
484 }
485 }
486
487 return pendingModuleFlags;
488 }

498 private static bool AppendToModule(PlacedModule currentModule,
499 List<SubmarineInfo> availableModules,
500 List<Identifier> pendingModuleFlags,
501 List<PlacedModule> selectedModules,
502 LocationType locationType,
503 bool retry = true,
504 bool allowExtendBelowInitialModule = false)
505 {
506 if (pendingModuleFlags.Count == 0) { return true; }
507
508 List<PlacedModule> placedModules = new List<PlacedModule>();
509 for (int i = 0; i < 2; i++)
510 {
511 //try placing a module meant for this location type first, and if that fails, try choosing whatever fits
512 bool allowDifferentLocationType = i > 0;
513 foreach (OutpostModuleInfo.GapPosition gapPosition in GapPositions.Randomize(Rand.RandSync.ServerAndClient))
514 {
515 if (currentModule.UsedGapPositions.HasFlag(gapPosition)) { continue; }
516 if (!allowExtendBelowInitialModule)
517 {
518 //don't continue downwards if it'd extend below the airlock
519 if (gapPosition == OutpostModuleInfo.GapPosition.Bottom && currentModule.Offset.Y <= 1) { continue; }
520 }
521
522 PlacedModule newModule = null;
523 //try appending to the current module if possible
524 if (currentModule.Info.OutpostModuleInfo.GapPositions.HasFlag(gapPosition))
525 {
526 newModule = AppendModule(currentModule, GetOpposingGapPosition(gapPosition), availableModules, pendingModuleFlags, selectedModules, locationType, allowDifferentLocationType);
527 }
528
529 if (newModule != null)
530 {
531 placedModules.Add(newModule);
532 }
533 else
534 {
535 //couldn't append to current module, try one of the other placed modules
536 foreach (PlacedModule otherModule in selectedModules)
537 {
538 if (otherModule == currentModule) { continue; }
539 foreach (OutpostModuleInfo.GapPosition otherGapPosition in
540 GapPositions.Where(g => !otherModule.UsedGapPositions.HasFlag(g) && otherModule.Info.OutpostModuleInfo.GapPositions.HasFlag(g)))
541 {
542 newModule = AppendModule(otherModule, GetOpposingGapPosition(otherGapPosition), availableModules, pendingModuleFlags, selectedModules, locationType, allowDifferentLocationType);
543 if (newModule != null)
544 {
545 placedModules.Add(newModule);
546 break;
547 }
548 }
549 if (newModule != null) { break; }
550 }
551 }
552 if (pendingModuleFlags.Count == 0) { return true; }
553 }
554 }
555
556 //couldn't place a module anywhere, we're probably fucked!
557 if (placedModules.Count == 0 && retry && currentModule.PreviousModule != null && !selectedModules.Any(m => m != currentModule && m.PreviousModule == currentModule))
558 {
559 //try to replace the previously placed module with something else that we can append to
560 for (int i = 0; i < 10; i++)
561 {
562 selectedModules.Remove(currentModule);
563 assertAllPreviousModulesPresent();
564 //readd the module types that the previous module was supposed to fulfill to the pending module types
565 pendingModuleFlags.AddRange(currentModule.FulfilledModuleTypes);
566 if (!availableModules.Contains(currentModule.Info)) { availableModules.Add(currentModule.Info); }
567 //retry
568 currentModule = AppendModule(currentModule.PreviousModule, currentModule.ThisGapPosition, availableModules, pendingModuleFlags, selectedModules, locationType, allowDifferentLocationType: true);
569 assertAllPreviousModulesPresent();
570 if (currentModule == null) { break; }
571 if (AppendToModule(currentModule, availableModules, pendingModuleFlags, selectedModules, locationType, retry: false, allowExtendBelowInitialModule: allowExtendBelowInitialModule))
572 {
573 assertAllPreviousModulesPresent();
574 return true;
575 }
576 }
577 return false;
578 }
579
580 foreach (PlacedModule placedModule in placedModules)
581 {
582 AppendToModule(placedModule, availableModules, pendingModuleFlags, selectedModules, locationType, allowExtendBelowInitialModule: allowExtendBelowInitialModule);
583 }
584 return placedModules.Count > 0;
585
586 void assertAllPreviousModulesPresent()
587 {
588 System.Diagnostics.Debug.Assert(selectedModules.All(m => m.PreviousModule == null || selectedModules.Contains(m.PreviousModule)));
589 }
590 }

600 private static PlacedModule AppendModule(
601 PlacedModule currentModule,
602 OutpostModuleInfo.GapPosition gapPosition,
603 List<SubmarineInfo> availableModules,
604 List<Identifier> pendingModuleFlags,
605 List<PlacedModule> selectedModules,
606 LocationType locationType,
607 bool allowDifferentLocationType)
608 {
609 if (pendingModuleFlags.Count == 0) { return null; }
610
611 Identifier flagToPlace = "none".ToIdentifier();
612 SubmarineInfo nextModule = null;
613 foreach (Identifier moduleFlag in pendingModuleFlags)
614 {
615 flagToPlace = moduleFlag;
616 nextModule = GetRandomModule(currentModule?.Info?.OutpostModuleInfo, availableModules, flagToPlace, gapPosition, locationType, allowDifferentLocationType);
617 if (nextModule != null) { break; }
618 }
619
620 if (nextModule != null)
621 {
622 var newModule = new PlacedModule(nextModule, currentModule, gapPosition)
623 {
624 Offset = currentModule.Offset + GetMoveDir(gapPosition),
625 };
626 foreach (Identifier moduleFlag in nextModule.OutpostModuleInfo.ModuleFlags)
627 {
628 if (!pendingModuleFlags.Contains(moduleFlag)) { continue; }
629 if (moduleFlag != "none" || flagToPlace == "none")
630 {
631 newModule.FulfilledModuleTypes.Add(moduleFlag);
632 pendingModuleFlags.Remove(moduleFlag);
633 }
634 }
635 selectedModules.Add(newModule);
636 if (selectedModules.Count(m => m.Info == nextModule) >= nextModule.OutpostModuleInfo.MaxCount)
637 {
638 availableModules.Remove(nextModule);
639 }
640 return newModule;
641 }
642 return null;
643 }

648 private static bool FindOverlap(IEnumerable<PlacedModule> modules1, IEnumerable<PlacedModule> modules2, out PlacedModule module1, out PlacedModule module2)
649 {
650 module1 = null;
651 module2 = null;
652 foreach (PlacedModule module in modules1)
653 {
654 foreach (PlacedModule otherModule in modules2)
655 {
656 if (module == otherModule) { continue; }
657 if (module.PreviousModule == otherModule && module.PreviousGap.ConnectedDoor == null && module.ThisGap.ConnectedDoor == null) { continue; }
658 if (ModulesOverlap(module, otherModule))
659 {
660 module1 = module;
661 module2 = otherModule;
662 return true;
663 }
664 }
665 }
666 return false;
667 }

672 private static bool ModulesOverlap(PlacedModule module1, PlacedModule module2)
673 {
674 Rectangle bounds1 = module1.Bounds;
675 bounds1.Location += (module1.Offset + module1.MoveOffset).ToPoint();
676 Rectangle bounds2 = module2.Bounds;
677 bounds2.Location += (module2.Offset + module2.MoveOffset).ToPoint();
678
679 //more tolerance on adjacent modules to prevent generating an unnecessary, small hallway between them
680 if (module1.PreviousModule == module2 || module2.PreviousModule == module1)
681 {
682 bounds1.Inflate(-16, -16);
683 bounds2.Inflate(-16, -16);
684 }
685
686 Rectangle hullBounds1 = module1.HullBounds;
687 hullBounds1.Location += (module1.Offset + module1.MoveOffset).ToPoint();
688 Rectangle hullBounds2 = module2.HullBounds;
689 hullBounds2.Location += (module2.Offset + module2.MoveOffset).ToPoint();
690
691 hullBounds1.Inflate(-32, -32);
692 hullBounds2.Inflate(-32, -32);
693
694 return hullBounds1.Intersects(hullBounds2) || hullBounds1.Intersects(bounds2) || hullBounds2.Intersects(bounds1);
695 }

700 private static bool ModuleOverlapsWithModuleConnections(IEnumerable<PlacedModule> modules)
701 {
702 foreach (PlacedModule module in modules)
703 {
704 Rectangle rect = module.Bounds;
705 rect.Location += (module.Offset + module.MoveOffset).ToPoint();
706 rect.Y += module.Bounds.Height;
707
708 foreach (PlacedModule otherModule in modules)
709 {
710 if (otherModule == module || otherModule.PreviousModule == null || otherModule.PreviousModule == module) { continue; }
711
712 //cast at both edges of the gap and see if it overlaps with anything
713 for (int i = -1; i <= 1; i += 2)
714 {
715 Vector2 gapEdgeOffset =
716 otherModule.ThisGap.IsHorizontal ?
717 Vector2.UnitY * otherModule.ThisGap.Rect.Height / 2 * i * 0.9f :
718 Vector2.UnitX * otherModule.ThisGap.Rect.Width / 2 * i * 0.9f;
719
720 Vector2 gapPos1 = otherModule.Offset + otherModule.ThisGap.Position + gapEdgeOffset + otherModule.MoveOffset;
721 Vector2 gapPos2 = otherModule.PreviousModule.Offset + otherModule.PreviousGap.Position + gapEdgeOffset + otherModule.PreviousModule.MoveOffset;
722 if (Submarine.RectContains(rect, gapPos1) || Submarine.RectContains(rect, gapPos2) || MathUtils.GetLineRectangleIntersection(gapPos1, gapPos2, rect, out _))
723 {
724 return true;
725 }
726 }
727 }
728 }
729 return false;
730 }

743 private static bool FindOverlapSolution(IEnumerable<PlacedModule> movableModules, PlacedModule module1, PlacedModule module2, IEnumerable<PlacedModule> allmodules, out Dictionary<PlacedModule, Vector2> solution)
744 {
745 solution = new Dictionary<PlacedModule, Vector2>();
746 foreach (PlacedModule module in movableModules)
747 {
748 solution[module] = Vector2.Zero;
749 }
750
751 Vector2 shortestMove = new Vector2(float.MaxValue, float.MaxValue);
752 bool solutionFound = false;
753 foreach (PlacedModule module in movableModules)
754 {
755 if (module.ThisGap.ConnectedDoor == null && module.PreviousGap.ConnectedDoor == null) { continue; }
756 Vector2 moveDir = GetMoveDir(module.ThisGapPosition);
757 Vector2 moveStep = moveDir * 50.0f;
758 Vector2 currentMove = Vector2.Zero;
759 float maxMoveAmount = 2000.0f;
760
761 List<PlacedModule> subsequentModules2 = new List<PlacedModule>();
762 GetSubsequentModules(module, movableModules, ref subsequentModules2);
763 while (currentMove.LengthSquared() < maxMoveAmount * maxMoveAmount)
764 {
765 currentMove += moveStep;
766 foreach (PlacedModule movedModule in subsequentModules2)
767 {
768 movedModule.MoveOffset = currentMove;
769 }
770 if (!ModulesOverlap(module1, module2) &&
771 !ModuleOverlapsWithModuleConnections(allmodules) &&
772 currentMove.LengthSquared() < shortestMove.LengthSquared())
773 {
774 shortestMove = currentMove;
775 foreach (PlacedModule movedModule in allmodules)
776 {
777 solution[movedModule] = subsequentModules2.Contains(movedModule) ? currentMove : Vector2.Zero;
778 solutionFound = true;
779 }
780 break;
781 }
782 }
783 foreach (PlacedModule movedModule in allmodules)
784 {
785 movedModule.MoveOffset = Vector2.Zero;
786 }
787 }
788
789 return solutionFound;
790 }

1102 gapToRemove.ConnectedDoor?.Item.Remove();
