27 {
28 get { return deattachTimer; }
29 set
30 {
31 //clients don't deattach the item until the server says so (handled in ClientRead)
32 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
33 {
34 return;
35 }
36
37 if (holdable == null) { return; }
38
39 deattachTimer = Math.Max(0.0f, value);
40#if SERVER
41 if (deattachTimer >= DeattachDuration)
42 {
43 if (holdable.Attached) { item.CreateServerEvent(this); }
44 holdable.DeattachFromWall();
45 }
46 else if (Math.Abs(lastSentDeattachTimer - deattachTimer) > 0.1f)
47 {
48 item.CreateServerEvent(this);
49 lastSentDeattachTimer = deattachTimer;
50 }
51#else
52 if (deattachTimer >= DeattachDuration)
53 {
54 if (holdable.Attached)
55 {
56 GameAnalyticsManager.AddDesignEvent("ResourceCollected:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "none") + ":" + item.Prefab.Identifier);

58 }
59 trigger.Enabled = false;
60 }
61#endif

107 public override void OnItemLoaded()
108 {

110 if (holdable == null)
111 {
112 IsActive = false;

116 if (requiredItems.Any())
117 {
118 holdable.PickingTime = float.MaxValue;
119 }
120
121 var body = item.body ?? holdable.Body;
122
123 if (body != null)
124 {
125 trigger = new PhysicsBody(body.width, body.height, body.radius, body.Density)
126 {
127 UserData = item
128 };
129 trigger.FarseerBody.SetIsSensor(true);
130 trigger.FarseerBody.BodyType = BodyType.Static;
131 trigger.FarseerBody.CollisionCategories = Physics.CollisionWall;
132 trigger.FarseerBody.CollidesWith = Physics.CollisionNone;
133 }
