143 public static bool IsValidName(string name, ServerSettings serverSettings)
144 {
145 if (string.IsNullOrWhiteSpace(name)) { return false; }
146
147 char[] disallowedChars = new char[] { ';', ',', '<', '>', '/', '\\', '[', ']', '"', '?' };
148 if (name.Any(c => disallowedChars.Contains(c))) { return false; }
149
150 foreach (char character in name)
151 {
152 if (!serverSettings.AllowedClientNameChars.Any(charRange => (int)character >= charRange.Start && (int)character <= charRange.End)) { return false; }
153 }
154
155 return true;
156 }
