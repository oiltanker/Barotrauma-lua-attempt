89 protected void ReadConnectionInitializationStep(PendingClient pendingClient, IReadMessage inc)
90 {
91 pendingClient.TimeOut = NetworkConnection.TimeoutThreshold;
92
93 ConnectionInitialization initializationStep = (ConnectionInitialization)inc.ReadByte();
94
95 if (pendingClient.InitializationStep != initializationStep) return;
96
97 pendingClient.UpdateTime = Timing.TotalTime + Timing.Step;
98
99 switch (initializationStep)
100 {
101 case ConnectionInitialization.SteamTicketAndVersion:
102 string name = Client.SanitizeName(inc.ReadString());
103 int ownerKey = inc.ReadInt32();
104 UInt64 steamId = inc.ReadUInt64();
105 UInt16 ticketLength = inc.ReadUInt16();
106 byte[] ticketBytes = inc.ReadBytes(ticketLength);
107
108 if (!Client.IsValidName(name, serverSettings))
109 {
110 RemovePendingClient(pendingClient, DisconnectReason.InvalidName, "");
111 return;
112 }
113
114 string version = inc.ReadString();
115 bool isCompatibleVersion = NetworkMember.IsCompatible(version, GameMain.Version.ToString()) ?? false;
116 if (!isCompatibleVersion)
117 {
118 RemovePendingClient(pendingClient, DisconnectReason.InvalidVersion,
119 $"DisconnectMessage.InvalidVersion~[version]={GameMain.Version}~[clientversion]={version}");
120
121 GameServer.Log($"{name} ({steamId}) couldn't join the server (incompatible game version)", ServerLog.MessageType.Error);
122 DebugConsole.NewMessage($"{name} ({steamId}) couldn't join the server (incompatible game version)", Microsoft.Xna.Framework.Color.Red);
123 return;
124 }
125
126 LanguageIdentifier language = inc.ReadIdentifier().ToLanguageIdentifier();
127 pendingClient.Connection.Language = language;
128
129 Client nameTaken = GameMain.Server.ConnectedClients.Find(c => Homoglyphs.Compare(c.Name.ToLower(), name.ToLower()));
130 if (nameTaken != null)
131 {
132 RemovePendingClient(pendingClient, DisconnectReason.NameTaken, "");
133 GameServer.Log($"{name} ({steamId}) couldn't join the server (name too similar to the name of the client \"" + nameTaken.Name + "\").", ServerLog.MessageType.Error);
134 return;
135 }
136
137 if (!pendingClient.AuthSessionStarted)
138 {
139 ProcessAuthTicket(name, ownerKey, steamId, pendingClient, ticketBytes);
140 }
141 break;
142 case ConnectionInitialization.Password:
143 int pwLength = inc.ReadByte();
144 byte[] incPassword = inc.ReadBytes(pwLength);
145 if (pendingClient.PasswordSalt == null)
146 {
147 DebugConsole.ThrowError("Received password message from client without salt");
148 return;
149 }
150 if (serverSettings.IsPasswordCorrect(incPassword, pendingClient.PasswordSalt.Value))
151 {
152 pendingClient.InitializationStep = ConnectionInitialization.ContentPackageOrder;
153 }
154 else
155 {
156 pendingClient.Retries++;
157 if (serverSettings.BanAfterWrongPassword && pendingClient.Retries > serverSettings.MaxPasswordRetriesBeforeBan)
158 {
159 string banMsg = "Failed to enter correct password too many times";
160 BanPendingClient(pendingClient, banMsg, null);
161
162 RemovePendingClient(pendingClient, DisconnectReason.Banned, banMsg);
163 return;
164 }
165 }
166 pendingClient.UpdateTime = Timing.TotalTime;
167 break;
168 case ConnectionInitialization.ContentPackageOrder:
169 pendingClient.InitializationStep = ConnectionInitialization.Success;
170 pendingClient.UpdateTime = Timing.TotalTime;
171 break;
172 }
173 }

177 protected void BanPendingClient(PendingClient pendingClient, string banReason, TimeSpan? duration)
178 {
179 if (pendingClient.Connection is LidgrenConnection l)
180 {
181 serverSettings.BanList.BanPlayer(pendingClient.Name, l.NetConnection.RemoteEndPoint.Address, banReason, duration);
182 }
183 else if (pendingClient.Connection is SteamP2PConnection s)
184 {
185 serverSettings.BanList.BanPlayer(pendingClient.Name, s.SteamID, banReason, duration);
186 serverSettings.BanList.BanPlayer(pendingClient.Name, s.OwnerSteamID, banReason, duration);
187 }
188 }

190 protected bool IsPendingClientBanned(PendingClient pendingClient, out string banReason)
191 {
192 if (pendingClient.Connection is LidgrenConnection l)
193 {
194 return serverSettings.BanList.IsBanned(l.NetConnection.RemoteEndPoint.Address, out banReason);
195 }
196 else if (pendingClient.Connection is SteamP2PConnection s)
197 {
198 return serverSettings.BanList.IsBanned(s.SteamID, out banReason) ||
199 serverSettings.BanList.IsBanned(s.OwnerSteamID, out banReason);
200 }
201 banReason = null;
202 return false;

207 protected void UpdatePendingClient(PendingClient pendingClient)
208 {
209 var skipRemove = false;
210 var result = GameMain.LuaCs.Hook.Call<bool?>("handlePendingClient", pendingClient);
211
212 if (result != null) skipRemove = result.Value;
213
214 if (!skipRemove && connectedClients.Count >= serverSettings.MaxPlayers)
215 {
216 RemovePendingClient(pendingClient, DisconnectReason.ServerFull, "");
217 }
218
219 if (IsPendingClientBanned(pendingClient, out string banReason))
220 {
221 RemovePendingClient(pendingClient, DisconnectReason.Banned, banReason);
222 return;
223 }
224
225 if (pendingClient.InitializationStep == ConnectionInitialization.Success)
226 {
227 NetworkConnection newConnection = pendingClient.Connection;
228 connectedClients.Add(newConnection);
229 pendingClients.Remove(pendingClient);
230
231 CheckOwnership(pendingClient);
232
233 OnInitializationComplete?.Invoke(newConnection);
234 }
235
236 pendingClient.TimeOut -= Timing.Step;
237 if (pendingClient.TimeOut < 0.0)
238 {
239 RemovePendingClient(pendingClient, DisconnectReason.Unknown, Lidgren.Network.NetConnection.NoResponseMessage);
240 }
241
242 if (Timing.TotalTime < pendingClient.UpdateTime) { return; }
243 pendingClient.UpdateTime = Timing.TotalTime + 1.0;
244
245 IWriteMessage outMsg = new WriteOnlyMessage();
246 outMsg.Write((byte)(PacketHeader.IsConnectionInitializationStep |
247 PacketHeader.IsServerMessage));
248 outMsg.Write((byte)pendingClient.InitializationStep);
249 switch (pendingClient.InitializationStep)
250 {
251 case ConnectionInitialization.ContentPackageOrder:
252 outMsg.Write(GameMain.Server.ServerName);
253
254 var mpContentPackages = ContentPackageManager.EnabledPackages.All.Where(cp => cp.HasMultiplayerSyncedContent).ToList();
255 outMsg.WriteVariableUInt32((UInt32)mpContentPackages.Count);
256 for (int i = 0; i < mpContentPackages.Count; i++)
257 {
258 outMsg.Write(mpContentPackages[i].Name);
259 byte[] hashBytes = mpContentPackages[i].Hash.ByteRepresentation;
260 outMsg.WriteVariableUInt32((UInt32)hashBytes.Length);
261 outMsg.Write(hashBytes, 0, hashBytes.Length);
262 outMsg.Write(mpContentPackages[i].SteamWorkshopId);
263 UInt32 installTimeDiffSeconds = (UInt32)((mpContentPackages[i].InstallTime ?? DateTime.UtcNow) - DateTime.UtcNow).TotalSeconds;
264 outMsg.Write(installTimeDiffSeconds);
265 }
266 break;
267 case ConnectionInitialization.Password:
268 outMsg.Write(pendingClient.PasswordSalt == null); outMsg.WritePadBits();
269 if (pendingClient.PasswordSalt == null)
270 {
271 pendingClient.PasswordSalt = Lidgren.Network.CryptoRandom.Instance.Next();
272 outMsg.Write(pendingClient.PasswordSalt.Value);
273 }
274 else
275 {
276 outMsg.Write(pendingClient.Retries);
277 }
278 break;
279 }
280
281 SendMsgInternal(pendingClient.Connection, DeliveryMethod.Reliable, outMsg);
282 }

286 public void RemovePendingClient(PendingClient pendingClient, DisconnectReason reason, string msg)
287 {

291
292 pendingClients.Remove(pendingClient);

294 if (pendingClient.AuthSessionStarted)
295 {
296 Steam.SteamManager.StopAuthSession(pendingClient.SteamID.Value);
297 pendingClient.SteamID = null;
298 pendingClient.OwnerSteamID = null;
299 pendingClient.AuthSessionStarted = false;
300 }
