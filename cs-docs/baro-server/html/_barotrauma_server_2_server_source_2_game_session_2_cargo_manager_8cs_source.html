43 public void SellItems(Identifier storeIdentifier, List<SoldItem> itemsToSell, Client client)
44 {
45 var store = Location.GetStore(storeIdentifier);
46 if (store == null) { return; }
47 bool canAddToRemoveQueue = (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer) && Entity.Spawner != null;
48 IEnumerable<Item> sellableItemsInSub = Enumerable.Empty<Item>();
49 if (canAddToRemoveQueue && itemsToSell.Any(i => i.Origin == SoldItem.SellOrigin.Submarine && i.ID == Entity.NullEntityID && !i.Removed))
50 {
51 sellableItemsInSub = GetSellableItemsFromSub();
52 }
53 var itemsSoldAtStore = SoldItems.GetValueOrDefault(storeIdentifier);
54 // Check all the prices before starting the transaction to make sure the modifiers stay the same for the whole transaction
55 var sellValues = GetSellValuesAtCurrentLocation(storeIdentifier, itemsToSell.Select(i => i.ItemPrefab));
56 foreach (var item in itemsToSell)
57 {
58 int itemValue = sellValues[item.ItemPrefab];
59 // check if the store can afford the item and if the item hasn't been removed already
60 if (store.Balance < itemValue || item.Removed) { continue; }
61 // Server determines the items that are sold from the sub in multiplayer
62 if (item.Origin == SoldItem.SellOrigin.Submarine && item.ID == Entity.NullEntityID && !item.Removed)
63 {
64 var matchingItem = sellableItemsInSub.FirstOrDefault(i => !i.Removed && i.Prefab == item.ItemPrefab &&
65 itemsToSell.None(itemToSell => itemToSell.ItemPrefab == i.Prefab && itemToSell.ID == i.ID));
66 // This is a failsafe for scenarios where a client is trying to sell more items than there's available on the sub
67 if (matchingItem == null) { continue; }
68 item.SetItemId(matchingItem.ID);
69 }
70 if (!item.Removed && canAddToRemoveQueue && Entity.FindEntityByID(item.ID) is Item entity)
71 {
72 item.Removed = true;
73 Entity.Spawner.AddItemToRemoveQueue(entity);
74 }
75 itemsSoldAtStore?.Add(item);
76 store.Balance -= itemValue;
77 campaign.GetWallet(client).Give(itemValue);
78 GameAnalyticsManager.AddMoneyGainedEvent(itemValue, GameAnalyticsManager.MoneySource.Store, item.ItemPrefab.Identifier.Value);
79 }
80 OnSoldItemsChanged?.Invoke();
81 }
