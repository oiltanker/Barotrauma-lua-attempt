72 public VoipQueue(byte id, bool canSend, bool canReceive)
73 {
74 BufferToQueue = new byte[VoipConfig.MAX_COMPRESSED_SIZE];
75 newestBufferInd = BUFFER_COUNT - 1;
76 bufferLengths = new int[BUFFER_COUNT];
77 buffers = new byte[BUFFER_COUNT][];

91 public void EnqueueBuffer(int length)
92 {
93 if (length > byte.MaxValue) { return; }
94
95 newestBufferInd = (newestBufferInd + 1) % BUFFER_COUNT;
96
97 int enqueuedTotalLength = EnqueuedTotalLength;
98
99 bufferLengths[newestBufferInd] = length;
100 BufferToQueue.CopyTo(buffers[newestBufferInd], 0);
101
102 if ((enqueuedTotalLength + length) > 0) { LatestBufferID++; }
103 }

105 public void RetrieveBuffer(int id, out int outSize, out byte[] outBuf)
106 {
107 lock (buffers)
108 {
109 if (id >= LatestBufferID - (BUFFER_COUNT - 1) && id <= LatestBufferID)
110 {
111 int index = newestBufferInd - (LatestBufferID - id);
112 if (index < 0) { index += BUFFER_COUNT; }
113 outSize = bufferLengths[index];
114 outBuf = buffers[index];
115 return;
116 }
117 }
118 outSize = -1;

122 public virtual void Write(IWriteMessage msg)

124 if (!CanSend) { throw new Exception("Called Write on a VoipQueue not set up for sending"); }

126 msg.Write((UInt16)LatestBufferID);

140 public virtual bool Read(IReadMessage msg, bool discardData = false)
141 {
142 if (!CanReceive) { throw new Exception("Called Read on a VoipQueue not set up for receiving"); }
143
144 UInt16 incLatestBufferID = msg.ReadUInt16();
145 if ((firstRead || NetIdUtils.IdMoreRecent(incLatestBufferID, LatestBufferID)) && !discardData)
146 {
147 ForceLocal = msg.ReadBoolean(); msg.ReadPadBits();
148
149 firstRead = false;
150 lock (buffers)
151 {
152 for (int i = 0; i < BUFFER_COUNT; i++)
153 {
154 bufferLengths[i] = msg.ReadByte();
155 buffers[i] = msg.ReadBytes(bufferLengths[i]);
156 }
157 }
158 newestBufferInd = BUFFER_COUNT - 1;
159 LatestBufferID = incLatestBufferID;
160 LastReadTime = DateTime.Now;
161 return true;
162 }
163 else
164 {
165 msg.ReadBoolean(); msg.ReadPadBits();
166 for (int i = 0; i < BUFFER_COUNT; i++)
167 {
168 byte len = msg.ReadByte();
169 msg.BitPosition += len * 8;
170 }
171 return false;
172 }
