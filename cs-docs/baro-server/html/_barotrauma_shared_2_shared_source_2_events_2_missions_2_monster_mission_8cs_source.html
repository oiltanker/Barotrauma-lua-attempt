36 public MonsterMission(MissionPrefab prefab, Location[] locations, Submarine sub)
37 : base(prefab, locations, sub)
38 {
39 Identifier speciesName = prefab.ConfigElement.GetAttributeIdentifier("monsterfile", Identifier.Empty);
40 if (!speciesName.IsEmpty)
41 {
42 var characterPrefab = CharacterPrefab.FindBySpeciesName(speciesName);
43 if (characterPrefab != null)
44 {
45 int monsterCount = Math.Min(prefab.ConfigElement.GetAttributeInt("monstercount", 1), 255);
46 monsterPrefabs.Add((characterPrefab, new Point(monsterCount)));
47 }
48 else
49 {
50 DebugConsole.ThrowError($"Error in monster mission \"{prefab.Identifier}\". Could not find a character prefab with the name \"{speciesName}\".");
51 }
52 }
53
54 maxSonarMarkerDistance = prefab.ConfigElement.GetAttributeFloat("maxsonarmarkerdistance", 10000.0f);
55
56 var spawnPosTypeStr = prefab.ConfigElement.GetAttributeString("spawntype", "");
57 if (string.IsNullOrWhiteSpace(spawnPosTypeStr) ||
58 !Enum.TryParse(spawnPosTypeStr, true, out spawnPosType))
59 {
60 spawnPosType = Level.PositionType.MainPath | Level.PositionType.SidePath;
61 }
62
63 foreach (var monsterElement in prefab.ConfigElement.GetChildElements("monster"))
64 {
65 speciesName = monsterElement.GetAttributeIdentifier("character", Identifier.Empty);
66 int defaultCount = monsterElement.GetAttributeInt("count", -1);
67 if (defaultCount < 0)
68 {
69 defaultCount = monsterElement.GetAttributeInt("amount", 1);
70 }
71 int min = Math.Min(monsterElement.GetAttributeInt("min", defaultCount), 255);
72 int max = Math.Min(Math.Max(min, monsterElement.GetAttributeInt("max", defaultCount)), 255);
73 var characterPrefab = CharacterPrefab.FindBySpeciesName(speciesName);
74 if (characterPrefab != null)
75 {
76 monsterPrefabs.Add((characterPrefab, new Point(min, max)));
77 }
78 else
79 {
80 DebugConsole.ThrowError($"Error in monster mission \"{prefab.Identifier}\". Could not find a character prefab with the name \"{speciesName}\".");
81 }
82 }
83
84 if (monsterPrefabs.Any())
85 {
86 var characterParams = new CharacterParams(monsterPrefabs.First().character.ContentFile as CharacterFile);
87 description = description.Replace("[monster]",
88 TextManager.Get("character." + characterParams.SpeciesTranslationOverride).Fallback(
89 TextManager.Get("character." + characterParams.SpeciesName)));
90 }

131 private void InitializeMonsters(IEnumerable<Character> monsters)
132 {
133 foreach (var monster in monsters)
134 {
135 monster.Enabled = false;
136 if (monster.Params.AI != null && monster.Params.AI.EnforceAggressiveBehaviorForMissions)
137 {
138 foreach (var targetParam in monster.Params.AI.Targets)
139 {
140 switch (targetParam.State)
141 {
142 case AIState.Avoid:
143 case AIState.Escape:
144 case AIState.Flee:
145 case AIState.PassiveAggressive:
146 targetParam.State = AIState.Attack;
147 break;
148 }
149 }
150 }
151 }
152 SwarmBehavior.CreateSwarm(monsters.Cast<AICharacter>());
153 foreach (Character monster in monsters)
154 {
155 tempSonarPositions.Add(monster.WorldPosition + Rand.Vector(maxSonarMarkerDistance));
156 }
157 if (monsters.Count() != tempSonarPositions.Count)
158 {
159 throw new Exception($"monsters.Count != tempSonarPositions.Count ({monsters.Count()} != {tempSonarPositions.Count})");
160 }

163 protected override void UpdateMissionSpecific(float deltaTime)
164 {
165 switch (State)
166 {
167 case 0:
168 //keep sonar markers within maxSonarMarkerDistance from the monster(s)
169 for (int i = 0; i < tempSonarPositions.Count; i++)
170 {
171 if (monsters.Count != tempSonarPositions.Count)
172 {
173 throw new Exception($"monsters.Count != tempSonarPositions.Count ({monsters.Count} != {tempSonarPositions.Count})");
174 }
175
176 if (i < 0 || i >= monsters.Count)
177 {
178 throw new Exception($"Index {i} outside of bounds 0-{monsters.Count} ({tempSonarPositions.Count})");
179 }
180
181 if (monsters[i].Removed || monsters[i].IsDead) { continue; }
182 Vector2 diff = tempSonarPositions[i] - monsters[i].Position;
183
184 float maxDist = maxSonarMarkerDistance;
185 Submarine refSub = Character.Controlled?.Submarine ?? Submarine.MainSub;
186 if (refSub != null)
187 {
188 Vector2 refPos = refSub == null ? Vector2.Zero : refSub.WorldPosition;
189 float subDist = Vector2.Distance(refPos, tempSonarPositions[i]) / maxDist;
190
191 maxDist = Math.Min(subDist * subDist * maxDist, maxDist);
192 maxDist = Math.Min(Vector2.Distance(refPos, monsters[i].Position), maxDist);
193 }
194
195 if (diff.LengthSquared() > maxDist * maxDist)
196 {
197 tempSonarPositions[i] = monsters[i].Position + Vector2.Normalize(diff) * maxDist;
198 }
199 }
200
201 sonarPositions.Clear();
202 for (int i = 0; i < monsters.Count; i++)
203 {
204 if (monsters[i].Removed || monsters[i].IsDead) { continue; }
205 //don't add another label if there's another monster roughly at the same spot
206 if (sonarPositions.All(p => Vector2.DistanceSquared(p, tempSonarPositions[i]) > 1000.0f * 1000.0f))
207 {
208 sonarPositions.Add(tempSonarPositions[i]);
209 }
210 }
211 if (!IsClient && monsters.All(m => IsEliminated(m)))
212 {
213 State = 1;
214 }
215 break;
216 }
217 }

219 public override void End()
220 {
221 tempSonarPositions.Clear();
222 monsters.Clear();
223 if (State < 1) { return; }
224
225 if (Prefab.LocationTypeChangeOnCompleted != null)
226 {
227 ChangeLocationType(Prefab.LocationTypeChangeOnCompleted);

231 if (level?.LevelData != null && Prefab.Tags.Any(t => t.Equals("huntinggrounds", StringComparison.OrdinalIgnoreCase)))
232 {
233 level.LevelData.HasHuntingGrounds = false;
234 }

239 enemy.Removed ||
