141  public Sprite(ContentXElement element, string path = "", string file = "", bool lazyLoad = false)
142  {
143  if (element is null) { return; }
144  this.LazyLoad = lazyLoad;
145  SourceElement = element;
146  if (!ParseTexturePath(path, file)) { return; }
147  Name = SourceElement.GetAttributeString("name", null);
148  Vector4 sourceVector = SourceElement.GetAttributeVector4("sourcerect", Vector4.Zero);
149  var overrideElement = GetLocalizationOverrideElement();
150  if (overrideElement != null && overrideElement.Attribute("sourcerect") != null)
151  {
152  sourceVector = overrideElement.GetAttributeVector4("sourcerect", Vector4.Zero);
153  }
154  if ((overrideElement ?? SourceElement).Attribute("sheetindex") != null)
155  {
156  Point sheetElementSize = (overrideElement ?? SourceElement).GetAttributePoint("sheetelementsize", Point.Zero);
157  Point sheetIndex = (overrideElement ?? SourceElement).GetAttributePoint("sheetindex", Point.Zero);
158  sourceVector = new Vector4(sheetIndex.X * sheetElementSize.X, sheetIndex.Y * sheetElementSize.Y, sheetElementSize.X, sheetElementSize.Y);
159  }
160  Compress = SourceElement.GetAttributeBool("compress", true);
161  bool shouldReturn = false;
162  if (!lazyLoad)
163  {
164  LoadTexture(ref sourceVector, ref shouldReturn);
165  }
166  if (shouldReturn) { return; }
167  sourceRect = new Rectangle((int)sourceVector.X, (int)sourceVector.Y, (int)sourceVector.Z, (int)sourceVector.W);
168  size = SourceElement.GetAttributeVector2("size", Vector2.One);
169  RelativeSize = size;
170  size.X *= sourceRect.Width;
171  size.Y *= sourceRect.Height;
172  RelativeOrigin = SourceElement.GetAttributeVector2("origin", new Vector2(0.5f, 0.5f));
173  Depth = SourceElement.GetAttributeFloat("depth", 0.001f);
174  Identifier = GetIdentifier(SourceElement);
175  AddToList(this);
176  }

257  public void ReloadXML()
258  {
259  if (SourceElement == null) { return; }
260  string path = SourceElement.ParseContentPathFromUri();
261  if (string.IsNullOrWhiteSpace(path))
262  {
263  DebugConsole.NewMessage($"[Sprite] Could not parse the content path from the source element ({SourceElement}) uri: {SourceElement.BaseUri}", Color.Yellow);
264  return;
265  }
266  var doc = XMLExtensions.TryLoadXml(path);
267  if (doc == null) { return; }
268  if (string.IsNullOrWhiteSpace(Name) && string.IsNullOrWhiteSpace(EntityIdentifier.Value)) { return; }
269  var spriteElements = doc.Descendants("sprite").Concat(doc.Descendants("Sprite"));
270  var sourceElements = spriteElements.Where(e => e.GetAttributeString("name", null) == Name);
271  if (sourceElements.None())
272  {
273  // Try parents by first comparing the entity id and then the name, if no match was found.
274  sourceElements = spriteElements.Where(e => e.Parent?.GetAttributeString("identifier", null) == EntityIdentifier);
275  if (sourceElements.None())
276  {
277  sourceElements = spriteElements.Where(e => e.Parent?.GetAttributeString("name", null) == Name);
278  }
279  }
280  if (sourceElements.Multiple())
281  {
282  DebugConsole.NewMessage($"[Sprite] Multiple matching elements found by name ({Name}) or identifier ({EntityIdentifier})!: {SourceElement}", Color.Yellow);
283  }
284  else if (sourceElements.None())
285  {
286  DebugConsole.NewMessage($"[Sprite] Cannot find matching source element by comparing the name attribute ({Name}) or identifier ({EntityIdentifier})! Cannot reload the xml for sprite element \"{SourceElement.ToString()}\"!", Color.Yellow);
287  }
288  else
289  {
290  SourceElement = sourceElements.Single().FromPackage(SourceElement.ContentPackage);
291  }
292  if (SourceElement != null)
293  {
294  sourceRect = SourceElement.GetAttributeRect("sourcerect", Rectangle.Empty);
295  var overrideElement = GetLocalizationOverrideElement();
296  if (overrideElement != null && overrideElement.Attribute("sourcerect") != null)
297  {
298  sourceRect = overrideElement.GetAttributeRect("sourcerect", Rectangle.Empty);
299  }
300  if ((overrideElement ?? SourceElement).Attribute("sheetindex") != null)
301  {
302  Point sheetElementSize = (overrideElement ?? SourceElement).GetAttributePoint("sheetelementsize", Point.Zero);
303  Point sheetIndex = (overrideElement ?? SourceElement).GetAttributePoint("sheetindex", Point.Zero);
304  sourceRect = new Rectangle(sheetIndex.X * sheetElementSize.X, sheetIndex.Y * sheetElementSize.Y, sheetElementSize.X, sheetElementSize.Y);
305  }
306  size = SourceElement.GetAttributeVector2("size", Vector2.One);
307  size.X *= sourceRect.Width;
308  size.Y *= sourceRect.Height;
309  RelativeOrigin = SourceElement.GetAttributeVector2("origin", new Vector2(0.5f, 0.5f));
310  Depth = SourceElement.GetAttributeFloat("depth", 0.001f);
311  Identifier = GetIdentifier(SourceElement);
312  }
313  }

315  public bool ParseTexturePath(string path = "", string file = "")
316  {
317  if (file == "")
318  {
319  file = SourceElement.GetAttributeStringUnrestricted("texture", "");
320  var overrideElement = GetLocalizationOverrideElement();
321  if (overrideElement != null)
322  {
323  string overrideFile = overrideElement.GetAttributeStringUnrestricted("texture", "");
324  if (!string.IsNullOrEmpty(overrideFile)) { file = overrideFile; }
325  }
326  }
327  if (file == "")
328  {
329  DebugConsole.ThrowError("Sprite " + SourceElement + " doesn't have a texture specified!");
330  return false;
331  }
332  if (!string.IsNullOrEmpty(path))
333  {
334  if (!path.EndsWith("/")) path += "/";
335  }
336  FilePath = ContentPath.FromRaw(SourceElement.ContentPackage, (path + file).CleanUpPathCrossPlatform(correctFilenameCase: true));
337  return true;
338  }

340  private XElement GetLocalizationOverrideElement()
341  {

343  {

349  return subElement;

351  }
352  }
353  return null;
