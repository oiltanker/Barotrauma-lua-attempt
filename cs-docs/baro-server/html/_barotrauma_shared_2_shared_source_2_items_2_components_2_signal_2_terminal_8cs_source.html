103 public override void ReceiveSignal(Signal signal, Connection connection)
104 {
105 switch (connection.Name)
106 {
107 case "set_text":
108 case "signal_in":
109 if (signal.value.Length > MaxMessageLength)
110 {
111 signal.value = signal.value.Substring(0, MaxMessageLength);
112 }
113
114 string inputSignal = signal.value.Replace("\\n", "\n");
115 ShowOnDisplay(inputSignal, addToHistory: true, TextColor);
116 break;
117 case "set_text_color":
118 if (signal.value != prevColorSignal)
119 {
120 TextColor = XMLExtensions.ParseColor(signal.value, false);
121 prevColorSignal = signal.value;
122 }
123 break;
124 case "clear_text" when signal.value != "0":
125 messageHistory.Clear();
126#if CLIENT
127 if (historyBox?.Content is { } history)
128 {
129 history.ClearChildren();
130 }
131
132 CreateFillerBlock();
133#endif
134 break;
135 }
136 }
