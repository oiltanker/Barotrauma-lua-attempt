19 protected override bool Filter(PowerContainer battery)
20 {
21 if (battery == null) { return false; }
22 var item = battery.Item;
23 if (item.IgnoreByAI(character)) { return false; }
24 if (!item.IsInteractable(character)) { return false; }
25 if (item.Submarine == null) { return false; }
26 if (item.CurrentHull == null) { return false; }
27 if (item.Submarine.TeamID != character.TeamID) { return false; }
28 if (character.Submarine != null)
29 {
30 if (!character.Submarine.IsConnectedTo(item.Submarine)) { return false; }
31 }
32 if (item.ConditionPercentage <= 0) { return false; }
33 if (item.IsClaimedByBallastFlora) { return false; }
34 if (Character.CharacterList.Any(c => c.CurrentHull == item.CurrentHull && !HumanAIController.IsFriendly(c) && HumanAIController.IsActive(c))) { return false; }
35 if (IsReady(battery)) { return false; }
36 return true;
37 }
38
39 protected override float TargetEvaluation()
40 {
41 if (Targets.None()) { return 0; }
42 if (Option == "charge")
43 {
44 return Targets.Max(t => MathHelper.Lerp(100, 0, Math.Abs(PowerContainer.aiRechargeTargetRatio - t.RechargeRatio)));
45 }
46 else
47 {
48
49 return Targets.Max(t => MathHelper.Lerp(0, 100, t.RechargeRatio));
50 }
51 }

53 protected override IEnumerable<PowerContainer> GetList()
54 {

59 return Array.Empty<PowerContainer>();

62 }
63 return batteryList;

72 }
73 else
74 {
75 return battery.RechargeRatio <= 0;
