160  public override void Update(float deltaTime, Camera cam)
161  {
162  UpdateOnActiveEffects(deltaTime);
163 
164  if (UseTransducers)
165  {
166  foreach (ConnectedTransducer transducer in connectedTransducers)
167  {
168  transducer.DisconnectTimer -= deltaTime;
169  }
170  connectedTransducers.RemoveAll(t => t.DisconnectTimer <= 0.0f);
171  }
172 
173  for (var pingIndex = 0; pingIndex < activePingsCount; ++pingIndex)
174  {
175  activePings[pingIndex].State += deltaTime * PingFrequency;
176  }
177 
178  if (currentMode == Mode.Active)
179  {
180  if ((Voltage >= MinVoltage) &&
181  (!UseTransducers || connectedTransducers.Count > 0))
182  {
183  if (currentPingIndex != -1)
184  {
185  var activePing = activePings[currentPingIndex];
186  if (activePing.State > 1.0f)
187  {
188  if (item.AiTarget != null)
189  {
190  float range = MathUtils.InverseLerp(item.AiTarget.MinSoundRange, item.AiTarget.MaxSoundRange, Range * activePing.State / zoom);
191  item.AiTarget.SoundRange = MathHelper.Lerp(item.AiTarget.MinSoundRange, item.AiTarget.MaxSoundRange, range);
192  item.AiTarget.SectorDegrees = activePing.IsDirectional ? DirectionalPingSector : 360.0f;
193  item.AiTarget.SectorDir = new Vector2(pingDirection.X, -pingDirection.Y);
194  }
195  aiPingCheckPending = true;
196  currentPingIndex = -1;
197  }
198  }
199  if (currentPingIndex == -1 && activePingsCount < activePings.Length)
200  {
201  currentPingIndex = activePingsCount++;
202  if (activePings[currentPingIndex] == null)
203  {
204  activePings[currentPingIndex] = new ActivePing();
205  }
206  activePings[currentPingIndex].IsDirectional = useDirectionalPing;
207  activePings[currentPingIndex].Direction = pingDirection;
208  activePings[currentPingIndex].State = 0.0f;
209  activePings[currentPingIndex].PrevPingRadius = 0.0f;
210  item.Use(deltaTime);
211  }
212  }
213  else
214  {
215  if (item.AiTarget != null)
216  {
217  item.AiTarget.SectorDegrees = 360.0f;
218  }
219  aiPingCheckPending = false;
220  }
221  }
222 
223  for (var pingIndex = 0; pingIndex < activePingsCount;)
224  {
225  if (activePings[pingIndex].State > 1.0f)
226  {
227  var lastIndex = --activePingsCount;
228  var oldActivePing = activePings[pingIndex];
229  activePings[pingIndex] = activePings[lastIndex];
230  activePings[lastIndex] = oldActivePing;
231  if (currentPingIndex == lastIndex)
232  {
233  currentPingIndex = pingIndex;
234  }
235  }
236  else
237  {
238  ++pingIndex;
239  }
240  }
241  }

263  public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
264  {
265  if (currentMode == Mode.Passive || !aiPingCheckPending) { return false; }
266 
267  foreach (List<Character> targetGroup in targetGroups.Values)
268  {
269  targetGroup.Clear();
270  }
271  foreach (Character c in Character.CharacterList)
272  {
273  if (c.IsDead || c.Removed || !c.Enabled) { continue; }
274  if (c.AnimController.CurrentHull != null || c.Params.HideInSonar) { continue; }
275  if (DetectSubmarineWalls && c.AnimController.CurrentHull == null && item.CurrentHull != null) { continue; }
276  if (Vector2.DistanceSquared(c.WorldPosition, item.WorldPosition) > range * range) { continue; }
277 
278  #warning This is not the best key for a dictionary.
279  string directionName = GetDirectionName(c.WorldPosition - item.WorldPosition).Value;
280  if (!targetGroups.ContainsKey(directionName))
281  {
282  targetGroups.Add(directionName, new List<Character>());
283  }
284  targetGroups[directionName].Add(c);
285  }
286 
287  foreach (KeyValuePair<string, List<Character>> targetGroup in targetGroups)
288  {
289  if (!targetGroup.Value.Any()) { continue; }
290  string dialogTag = "DialogSonarTarget";
291  if (targetGroup.Value.Count > 1)

294  }

305  null, 0, $"sonartarget{targetGroup.Value[0].ID}".ToIdentifier(), 60);

376  useDirectionalPing = directionalPing;
377  if (useDirectionalPing)

381  }
