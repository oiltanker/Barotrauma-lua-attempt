138  private void UpdateClient(Client client, float deltaTime)
139  {
140  if (client.Character != null && !client.Character.Removed && !client.Character.IsDead)
141  {
142  if (client.Karma > KarmaDecayThreshold)
143  {
144  client.Karma -= KarmaDecay * deltaTime;
145  }
146  else if (client.Karma < KarmaIncreaseThreshold)
147  {
148  client.Karma += KarmaIncrease * deltaTime;
149  }
150 
151  //increase the strength of the herpes affliction in steps instead of linearly
152  //otherwise clients could determine their exact karma value from the strength
153  float herpesStrength = 0.0f;
154  if (client.Karma < 20)
155  herpesStrength = 100.0f;
156  else if (client.Karma < 30)
157  herpesStrength = 60.0f;
158  else if (client.Karma < 40.0f)
159  herpesStrength = 30.0f;
160 
161  var existingAffliction = client.Character.CharacterHealth.GetAffliction<AfflictionSpaceHerpes>("spaceherpes");
162  if (existingAffliction == null && herpesStrength > 0.0f)
163  {
164  client.Character.CharacterHealth.ApplyAffliction(null, new Affliction(herpesAffliction, herpesStrength));
165  GameServer.Log($"{GameServer.ClientLogName(client)} has contracted space herpes due to low karma.", ServerLog.MessageType.Karma);
166  GameMain.NetworkMember.LastClientListUpdateID++;
167  }
168  else if (existingAffliction != null)
169  {
170  existingAffliction.Strength = herpesStrength;
171  if (herpesStrength <= 0.0f)
172  {
173  client.Character.CharacterHealth.ReduceAfflictionOnAllLimbs("invertcontrols".ToIdentifier(), 100.0f);
174  }
175  }
176 
177  //check if the client has disconnected an excessive number of wires
178  var clientMemory = GetClientMemory(client);
179  if (clientMemory.WireDisconnectTime.Count > (int)AllowedWireDisconnectionsPerMinute)
180  {
181  clientMemory.WireDisconnectTime.RemoveRange(0, clientMemory.WireDisconnectTime.Count - (int)AllowedWireDisconnectionsPerMinute);
182  if (clientMemory.WireDisconnectTime.All(w => Timing.TotalTime - w.Second < 60.0f))
183  {
184  float karmaDecrease = -WireDisconnectionKarmaDecrease;
185  //engineers don't lose as much karma for removing lots of wires
186  if (client.Character.Info?.Job.Prefab.Identifier == "engineer") { karmaDecrease *= 0.5f; }
187  AdjustKarma(client.Character, karmaDecrease, "Disconnected excessive number of wires");
188  }
189  }
190 
191  if (client.Character?.Info?.Job.Prefab.Identifier == "captain" && client.Character.SelectedConstruction != null)
192  {
193  if (client.Character.SelectedConstruction.GetComponent<Steering>() != null)
194  {
195  AdjustKarma(client.Character, SteerSubKarmaIncrease * deltaTime, "Steering the sub");
196  }
197  }
198  }
199 
200  if (client.Karma < KickBanThreshold && client.Connection != GameMain.Server.OwnerConnection)
201  {
202  if (TestMode)
203  {
204  client.Karma = 50.0f;
205  GameMain.Server.SendDirectChatMessage("BANNED! (not really because karma test mode is enabled)", client);
206  }
207  else

209  bannedClients.Add(client);

279  static bool isValid(Item item)
280  {
281  return item.Prefab.Identifier == "idcard" || item.GetComponent<RangedWeapon>() != null || item.GetComponent<MeleeWeapon>() != null;
282  }
283 
284  if (foundItem == null) { return; }
285 
286  bool isIdCard = ((MapEntity)foundItem).Prefab.Identifier == "idcard";
287  bool isWeapon = foundItem.GetComponent<RangedWeapon>() != null || foundItem.GetComponent<MeleeWeapon>() != null;
288 
289  if (isIdCard)
290  {
291  string name = string.Empty;
292 
293  foreach (var tag in foundItem.Tags.Split(','))
294  {
295  string[] split = tag.Split(':');
296  string key = split.Length > 0 ? split[0] : string.Empty;
297  string value = split.Length > 1 ? split[1] : string.Empty;
298  if (key == "name") { name = value; }
299  }
300 
301  // Name tag doesn't belong to anyone in particular or we own the ID card
302  if (name == null || name == yoinkerCharacter.Name) { return; }
303  }
304 
305  if (MathUtils.NearlyEqual(DangerousItemStealKarmaDecrease, 0)) { return; }
306 
307  const float calcUpper = 1, calcLower = -1;
308 
309  float upper = DangerousItemStealKarmaDecrease + 10.0f;
310  float lower = DangerousItemStealKarmaDecrease - 10.0f;
311 
312  if (lower < 0)
313  {
314  upper += Math.Abs(lower);
315  lower = 0;
316  }
317 
318  // If we're stealing from a bot assume the bot has 50 karma
319  var targetKarma = targetClient?.Karma ?? 50;
320 
321  float karmaDifference = Math.Clamp((targetKarma - yoinker.Karma) / 50.0f, calcLower, calcUpper);
322  float karmaDecrease = lower + (karmaDifference - calcLower) * (upper - lower) / (calcUpper - calcLower);
323 
324  JobPrefab clientJob = yoinker.CharacterInfo?.Job?.Prefab;
325 
326  // security officers receive less karma penalty
327  if (clientJob != null && clientJob.Identifier == "securityofficer" && isWeapon)
328  {
329  karmaDecrease *= 0.5f;
330  }
331 
332  AdjustKarma(yoinkerCharacter, -karmaDecrease, "Stolen dangerous item");
333  }

335  public void OnCharacterHealthChanged(Character target, Character attacker, float damage, float stun, IEnumerable<Affliction> appliedAfflictions = null)
336  {
337  if (target == null || attacker == null) { return; }
338  if (target == attacker) { return; }
339 
340  //damaging dead characters doesn't affect karma
341  if (target.IsDead || target.Removed) { return; }
342 
343  bool isEnemy = target.AIController is EnemyAIController || target.TeamID != attacker.TeamID;
344  if (GameMain.Server.TraitorManager?.Traitors != null)
345  {
346  if (GameMain.Server.TraitorManager.Traitors.Any(t => t.Character == target))
347  {
348  //traitors always count as enemies
349  isEnemy = true;
350  }
351  if (GameMain.Server.TraitorManager.Traitors.Any(t =>
352  t.Character == attacker &&
353  t.CurrentObjective != null &&
354  t.CurrentObjective.IsEnemy(target)))
355  {
356  //target counts as an enemy to the traitor
357  isEnemy = true;
358  }
359  }
360 
361  bool targetIsHusk = target.CharacterHealth?.GetAffliction<AfflictionHusk>("huskinfection")?.State == AfflictionHusk.InfectionState.Active;
362  bool attackerIsHusk = attacker.CharacterHealth?.GetAffliction<AfflictionHusk>("huskinfection")?.State == AfflictionHusk.InfectionState.Active;
363  //huskified characters count as enemies to healthy characters and vice versa
364  if (targetIsHusk != attackerIsHusk) { isEnemy = true; }
365 
366  if (appliedAfflictions != null)
367  {
368  foreach (Affliction affliction in appliedAfflictions)
369  {
370  if (MathUtils.NearlyEqual(affliction.Prefab.KarmaChangeOnApplied, 0.0f)) { continue; }
371  damage -= affliction.Prefab.KarmaChangeOnApplied * affliction.Strength;
372  }
373  }
374 
375  Client targetClient = GameMain.Server.ConnectedClients.Find(c => c.Character == target);
376  if (damage > 0 && targetClient != null)
377  {
378  var targetMemory = GetClientMemory(targetClient);
379  targetMemory.LastAttackTime[attacker] = Timing.TotalTime;
380  }
381 
382  Client attackerClient = GameMain.Server.ConnectedClients.Find(c => c.Character == attacker);
383  ClientMemory attackerMemory = GetClientMemory(attackerClient);
384  if (attackerMemory != null)
385  {
386  //if the attacker has been attacked by the target within the last x seconds, ignore the damage
387  //(= no karma penalty from retaliating against someone who attacked you)
388  if (attackerMemory.LastAttackTime.ContainsKey(target) &&
389  attackerMemory.LastAttackTime[target] > Timing.TotalTime - AllowedRetaliationTime)
390  {
391  damage = Math.Min(damage, 0);
392  stun = 0.0f;
393  }
394  }
395 
396  //attacking/healing clowns has a smaller effect on karma
397  if (target.HasEquippedItem("clownmask".ToIdentifier()) &&
398  target.HasEquippedItem("clowncostume".ToIdentifier()))
399  {
400  damage *= 0.5f;
401  stun *= 0.5f;
402  }
403 
404  //smaller karma penalty for attacking someone who's aiming with a weapon
405  if (damage > 0.0f &&
406  target.IsKeyDown(InputType.Aim) &&
407  target.HeldItems.Any(it => it.GetComponent<MeleeWeapon>() != null || it.GetComponent<RangedWeapon>() != null))
408  {
409  damage *= 0.5f;
410  stun *= 0.5f;
411  }
412 
413  //damage scales according to the karma of the target
414  //(= smaller karma penalty from attacking someone who has a low karma)
415  if (damage > 0 && targetClient != null)
416  {
417  damage *= MathUtils.InverseLerp(0.0f, 50.0f, targetClient.Karma);
418  }
419 
420  if (isEnemy)
421  {
422  if (damage > 0)
423  {
424  float karmaIncrease = damage * DamageEnemyKarmaIncrease;
425  if (attacker?.Info?.Job.Prefab.Identifier == "securityofficer") { karmaIncrease *= 2.0f; }
426  AdjustKarma(attacker, karmaIncrease, "Damaged enemy");
427  }
428  }
429  else
430  {
431  if (stun > 0 && attackerMemory != null)
432  {
433  //GameServer.Log(GameServer.CharacterLogName(attacker) + " stunned " + GameServer.CharacterLogName(target) + $" ({stun})", ServerLog.MessageType.Karma);
434  attackerMemory.StunsInPastMinute.Add(new ClientMemory.TimeAmount() { Time = Timing.TotalTime, Amount = stun });
435 
436  if (attackerMemory.StunsInPastMinute.Count > 1)
437  {
438  float avgStunsInflicted = attackerMemory.StunsInPastMinute[0].Amount / (float)(attackerMemory.StunsInPastMinute[1].Time - attackerMemory.StunsInPastMinute[0].Time);
439  for (int i = 1; i < attackerMemory.StunsInPastMinute.Count; i++)
440  {
441  avgStunsInflicted += attackerMemory.StunsInPastMinute[i].Amount / (float)(attackerMemory.StunsInPastMinute[i].Time - attackerMemory.StunsInPastMinute[i - 1].Time);
442  }
443 
444  //GameServer.Log(avgStunsInflicted.ToString(), ServerLog.MessageType.Karma);
445 
446  if (avgStunsInflicted > StunFriendlyKarmaDecreaseThreshold ||
447  attackerMemory.StunKarmaDecreaseMultiplier > 1.0f)
448  {
449  AdjustKarma(attacker, -StunFriendlyKarmaDecrease * attackerMemory.StunKarmaDecreaseMultiplier, "Stunned friendly");
450  attackerMemory.StunKarmaDecreaseMultiplier *= 2.0f;
451  }
452  }
453  }
454 
455  if (damage > 0)
456  {
457  AdjustKarma(attacker, -damage * DamageFriendlyKarmaDecrease, "Damaged friendly");
458  }
459  else
460  {
461  float karmaIncrease = -damage * HealFriendlyKarmaIncrease;
462  if (attacker?.Info?.Job.Prefab.Identifier == "medicaldoctor") { karmaIncrease *= 2.0f; }
463  AdjustKarma(attacker, karmaIncrease, "Healed friendly");
464  }
465  }

484  t.Character == attacker &&

488  //traitor tasked to flood the sub -> damaging structures is ok
489  return;

491  }
492 
493  Client client = GameMain.Server.ConnectedClients.Find(c => c.Character == attacker);
494  if (client != null)
495  {
496  //cap the damage so the karma can't decrease by more than MaxStructureDamageKarmaDecreasePerSecond per second
497  var clientMemory = GetClientMemory(client);
498  if (clientMemory.StructureDamagePerSecond + damageAmount >= MaxStructureDamageKarmaDecreasePerSecond / StructureDamageKarmaDecrease)
499  {

501  if (damageAmount <= 0.0f) { return; }
502  }
503  clientMemory.StructureDamageAccumulator += damageAmount;
504  }
505  AdjustKarma(attacker, -damageAmount * StructureDamageKarmaDecrease, "Damaged structures");
506  }
507  else

509  float karmaIncrease = -damageAmount * StructureRepairKarmaIncrease;
510  //mechanics get twice as much karma for repairing walls

599  private void AdjustKarma(Character target, float amount, string debugKarmaChangeReason = "")
600  {
601  if (target == null) { return; }
602 
603  Client client = GameMain.Server.ConnectedClients.Find(c => c.Character == target);
604  if (client == null) { return; }
605 
606  //all penalties/rewards are halved when wearing a clown costume
607  if (target.HasEquippedItem("clownmask".ToIdentifier()) &&
608  target.HasEquippedItem("clowncostume".ToIdentifier()))
609  {
610  amount *= 0.5f;
611  }
612 
613  client.Karma += amount;
614 
615  if (amount < 0.0f)
616  {
617  float? herpesStrength = client.Character?.CharacterHealth.GetAfflictionStrength("spaceherpes");
