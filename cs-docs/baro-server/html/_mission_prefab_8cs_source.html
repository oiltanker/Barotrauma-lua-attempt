141  public MissionPrefab(ContentXElement element, MissionsFile file) : base(file, element.GetAttributeIdentifier("identifier", ""))
142  {
143  ConfigElement = element;
144 
145  TextIdentifier = element.GetAttributeIdentifier("textidentifier", Identifier);
146 
147  tags = element.GetAttributeStringArray("tags", Array.Empty<string>(), convertToLowerInvariant: true);
148 
149  Name = TextManager.Get($"MissionName.{TextIdentifier}").Fallback(element.GetAttributeString("name", ""));
150  Description = TextManager.Get($"MissionDescription.{TextIdentifier}").Fallback(element.GetAttributeString("description", ""));
151  Reward = element.GetAttributeInt("reward", 1);
152  AllowRetry = element.GetAttributeBool("allowretry", false);
153  IsSideObjective = element.GetAttributeBool("sideobjective", false);
154  RequireWreck = element.GetAttributeBool("requirewreck", false);
155  RequireRuin = element.GetAttributeBool("requireruin", false);
156  Commonness = element.GetAttributeInt("commonness", 1);
157  if (element.GetAttribute("difficulty") != null)
158  {
159  int difficulty = element.GetAttributeInt("difficulty", MinDifficulty);
160  Difficulty = Math.Clamp(difficulty, MinDifficulty, MaxDifficulty);
161  }
162 
163  SuccessMessage = TextManager.Get($"MissionSuccess.{TextIdentifier}").Fallback(element.GetAttributeString("successmessage", "Mission completed successfully"));
164  FailureMessage = TextManager.Get($"MissionFailure.{TextIdentifier}").Fallback(
165  TextManager.Get("missionfailed")).Fallback(
166  GameSettings.CurrentConfig.Language == TextManager.DefaultLanguage ? element.GetAttributeString("failuremessage", "") : "");
167 
168  string sonarLabelTag = element.GetAttributeString("sonarlabel", "");
169 
170  SonarLabel =
171  TextManager.Get($"MissionSonarLabel.{sonarLabelTag}")
172  .Fallback(TextManager.Get(sonarLabelTag))
173  .Fallback(TextManager.Get($"MissionSonarLabel.{TextIdentifier}"))
174  .Fallback(element.GetAttributeString("sonarlabel", ""));
175 
176  SonarIconIdentifier = element.GetAttributeIdentifier("sonaricon", "");
177 
178  MultiplayerOnly = element.GetAttributeBool("multiplayeronly", false);
179  SingleplayerOnly = element.GetAttributeBool("singleplayeronly", false);
180 
181  AchievementIdentifier = element.GetAttributeIdentifier("achievementidentifier", "");
182 
183  UnhideEntitySubCategories = element.GetAttributeStringArray("unhideentitysubcategories", Array.Empty<string>()).ToList();
184 
185  var headers = new List<LocalizedString>();
186  var messages = new List<LocalizedString>();
187  AllowedConnectionTypes = new List<(Identifier from, Identifier to)>();
188 
189  for (int i = 0; i < 100; i++)
190  {
191  LocalizedString header = TextManager.Get($"MissionHeader{i}.{TextIdentifier}");
192  LocalizedString message = TextManager.Get($"MissionMessage{i}.{TextIdentifier}");
193  if (!message.IsNullOrEmpty())
194  {
195  headers.Add(header);
196  messages.Add(message);
197  }
198  }
199 
200  int messageIndex = 0;
201  foreach (var subElement in element.Elements())
202  {
203  switch (subElement.Name.ToString().ToLowerInvariant())
204  {
205  case "message":
206  if (messageIndex > headers.Count - 1)
207  {
208  headers.Add(string.Empty);
209  messages.Add(string.Empty);
210  }
211  headers[messageIndex] = TextManager.Get($"MissionHeader{messageIndex}.{TextIdentifier}").Fallback(subElement.GetAttributeString("header", ""));
212  messages[messageIndex] = TextManager.Get($"MissionMessage{messageIndex}.{TextIdentifier}").Fallback(subElement.GetAttributeString("text", ""));
213  messageIndex++;
214  break;
215  case "locationtype":
216  case "connectiontype":
217  if (subElement.GetAttribute("identifier") != null)
218  {
219  AllowedLocationTypes.Add(subElement.GetAttributeIdentifier("identifier", ""));
220  }
221  else
222  {
223  AllowedConnectionTypes.Add((
224  subElement.GetAttributeIdentifier("from", ""),
225  subElement.GetAttributeIdentifier("to", "")));
226  }
227  break;
228  case "locationtypechange":
229  LocationTypeChangeOnCompleted = new LocationTypeChange(subElement.GetAttributeIdentifier("from", ""), subElement, requireChangeMessages: false, defaultProbability: 1.0f);
230  break;
231  case "reputation":
232  case "reputationreward":
233  Identifier factionIdentifier = subElement.GetAttributeIdentifier("identifier", Identifier.Empty);
234  float amount = subElement.GetAttributeFloat("amount", 0.0f);
235  if (ReputationRewards.ContainsKey(factionIdentifier))
236  {
237  DebugConsole.ThrowError($"Error in mission prefab \"{Identifier}\". Multiple reputation changes defined for the identifier \"{factionIdentifier}\".");
238  continue;
239  }
240  ReputationRewards.Add(factionIdentifier, amount);
241  break;
242  case "metadata":
243  Identifier identifier = subElement.GetAttributeIdentifier("identifier", Identifier.Empty);
244  string stringValue = subElement.GetAttributeString("value", string.Empty);
245  if (!string.IsNullOrWhiteSpace(stringValue) && !identifier.IsEmpty)
246  {
247  object value = SetDataAction.ConvertXMLValue(stringValue);
248  SetDataAction.OperationType operation = SetDataAction.OperationType.Set;
249 
250  string operatingString = subElement.GetAttributeString("operation", string.Empty);
251  if (!string.IsNullOrWhiteSpace(operatingString))
252  {
253  operation = (SetDataAction.OperationType) Enum.Parse(typeof(SetDataAction.OperationType), operatingString);
254  }
255 
256  DataRewards.Add((identifier, value, operation));
257  }
258  break;
259  case "triggerevent":
260  TriggerEvents.Add(new TriggerEvent(subElement));
261  break;
262  }
263  }
264  Headers = headers.ToImmutableArray();
265  Messages = messages.ToImmutableArray();
266 
267  Identifier missionTypeName = element.GetAttributeIdentifier("type", Identifier.Empty);
268  //backwards compatibility
269  if (missionTypeName == "outpostdestroy" || missionTypeName == "outpostrescue")
270  {
271  missionTypeName = "AbandonedOutpost".ToIdentifier();
272  }
273 
274  if (!Enum.TryParse(missionTypeName.Value, true, out Type))
275  {
276  DebugConsole.ThrowError("Error in mission prefab \"" + Name + "\" - \"" + missionTypeName + "\" is not a valid mission type.");
277  return;
278  }
279  if (Type == MissionType.None)
280  {
281  DebugConsole.ThrowError("Error in mission prefab \"" + Name + "\" - mission type cannot be none.");
282  return;
283  }
284 
285  if (CoOpMissionClasses.ContainsKey(Type))
286  {
287  constructor = CoOpMissionClasses[Type].GetConstructor(new[] { typeof(MissionPrefab), typeof(Location[]), typeof(Submarine) });
288  }
289  else if (PvPMissionClasses.ContainsKey(Type))
290  {
291  constructor = PvPMissionClasses[Type].GetConstructor(new[] { typeof(MissionPrefab), typeof(Location[]), typeof(Submarine) });
292  }
293  else
294  {
295  DebugConsole.ThrowError("Error in mission prefab \"" + Name + "\" - unsupported mission type \"" + Type.ToString() + "\"");
296  }
297  if (constructor == null)
298  {
299  DebugConsole.ThrowError($"Failed to find a constructor for the mission type \"{Type}\"!");
300  }
301 
302  InitProjSpecific(element);
303  }

307  public bool IsAllowed(Location from, Location to)
308  {
309  if (from == to)
310  {
311  return
312  AllowedLocationTypes.Any(lt => lt == "any") ||
313  AllowedLocationTypes.Any(lt => lt == from.Type.Identifier);
314  }
315 
316  foreach (var (fromType, toType) in AllowedConnectionTypes)
317  {
318  if (fromType == "any" ||
319  fromType == from.Type.Identifier ||
320  (fromType == "anyoutpost" && from.HasOutpost()))
321  {
322  if (toType == "any" ||
323  toType == to.Type.Identifier ||
324  (toType == "anyoutpost" && to.HasOutpost()))
325  {
326  return true;
327  }
328  }
329  }
330 
331  if (Type == MissionType.Beacon)
332  {
333  var connection = from.Connections.Find(c => c.Locations.Contains(from) && c.Locations.Contains(to));
334  if (connection?.LevelData == null || !connection.LevelData.HasBeaconStation || connection.LevelData.IsBeaconActive) { return false; }
335  }
336  else if (Type == MissionType.ScanAlienRuins || Type == MissionType.ClearAlienRuins)
337  {
338  var connection = from.Connections.Find(c => c.Locations.Contains(from) && c.Locations.Contains(to));
339  if (connection?.LevelData == null || connection.LevelData.GenerationParams.RuinCount < 1) { return false; }
340  }
341 
342  return false;
343  }
