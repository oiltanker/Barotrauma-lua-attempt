182 private GameMode InstantiateGameMode(GameModePreset gameModePreset, string? seed, SubmarineInfo selectedSub, CampaignSettings settings, IEnumerable<MissionPrefab>? missionPrefabs = null, MissionType missionType = MissionType.None)
183 {
184 if (gameModePreset.GameModeType == typeof(CoOpMode) || gameModePreset.GameModeType == typeof(PvPMode))
185 {
186 //don't allow hidden mission types (e.g. GoTo) in single mission modes
187 var missionTypes = (MissionType[])Enum.GetValues(typeof(MissionType));
188 for (int i = 0; i < missionTypes.Length; i++)
189 {
190 if (MissionPrefab.HiddenMissionClasses.Contains(missionTypes[i]))
191 {
192 missionType &= ~missionTypes[i];
193 }
194 }
195 }
196 if (gameModePreset.GameModeType == typeof(CoOpMode))
197 {
198 return missionPrefabs != null ?
199 new CoOpMode(gameModePreset, missionPrefabs) :
200 new CoOpMode(gameModePreset, missionType, seed ?? ToolBox.RandomSeed(8));
201 }
202 else if (gameModePreset.GameModeType == typeof(PvPMode))
203 {
204 return missionPrefabs != null ?
205 new PvPMode(gameModePreset, missionPrefabs) :
206 new PvPMode(gameModePreset, missionType, seed ?? ToolBox.RandomSeed(8));
207 }
208 else if (gameModePreset.GameModeType == typeof(MultiPlayerCampaign))
209 {
210 var campaign = MultiPlayerCampaign.StartNew(seed ?? ToolBox.RandomSeed(8), selectedSub, settings);
211 if (selectedSub != null)
212 {
213 campaign.Bank.Deduct(selectedSub.Price);
214 campaign.Bank.Balance = Math.Max(campaign.Bank.Balance, MultiPlayerCampaign.MinimumInitialMoney);
215 }
216 return campaign;
217 }
218#if CLIENT
219 else if (gameModePreset.GameModeType == typeof(SinglePlayerCampaign))
220 {
221 var campaign = SinglePlayerCampaign.StartNew(seed ?? ToolBox.RandomSeed(8), selectedSub, settings);
222 if (selectedSub != null)
223 {
224 campaign.Bank.TryDeduct(selectedSub.Price);
225 campaign.Bank.Balance = Math.Max(campaign.Bank.Balance, MultiPlayerCampaign.MinimumInitialMoney);
226 }
227 return campaign;
228 }
229 else if (gameModePreset.GameModeType == typeof(TutorialMode))
230 {
231 return new TutorialMode(gameModePreset);
232 }
233 else if (gameModePreset.GameModeType == typeof(TestGameMode))
234 {
235 return new TestGameMode(gameModePreset);
236 }
237#endif
238 else if (gameModePreset.GameModeType == typeof(GameMode))
239 {
240 return new GameMode(gameModePreset);
241 }
242 else
243 {
244 throw new Exception($"Could not find a game mode of the type \"{gameModePreset.GameModeType}\"");
245 }

339 public void StartRound(string levelSeed, float? difficulty = null, LevelGenerationParams? levelGenerationParams = null)
340 {
341 if (GameMode == null) { return; }
342 LevelData? randomLevel = null;
343 foreach (Mission mission in Missions.Union(GameMode.Missions))
344 {
345 MissionPrefab missionPrefab = mission.Prefab;
346 if (missionPrefab != null &&
347 missionPrefab.AllowedLocationTypes.Any() &&
348 !missionPrefab.AllowedConnectionTypes.Any())
349 {
350 LocationType? locationType = LocationType.Prefabs.FirstOrDefault(lt => missionPrefab.AllowedLocationTypes.Any(m => m == lt.Identifier));
351 CreateDummyLocations(locationType);
352 randomLevel = LevelData.CreateRandom(levelSeed, difficulty, levelGenerationParams, requireOutpost: true);
353 break;
354 }
355 }
356 randomLevel ??= LevelData.CreateRandom(levelSeed, difficulty, levelGenerationParams);
357 StartRound(randomLevel);

625 public void PlaceSubAtStart(Level? level)
626 {
627 if (level == null || Submarine == null)
628 {
629 Submarine?.SetPosition(Vector2.Zero);
630 return;
631 }
632
633 var originalSubPos = Submarine.WorldPosition;
634
635 if (level.StartOutpost != null)
636 {
637 //start by placing the sub below the outpost
638 Rectangle outpostBorders = Level.Loaded.StartOutpost.GetDockedBorders();
639 Rectangle subBorders = Submarine.GetDockedBorders();
640
641 Submarine.SetPosition(
642 Level.Loaded.StartOutpost.WorldPosition -
643 new Vector2(0.0f, outpostBorders.Height / 2 + subBorders.Height / 2));
644
645 //find the port that's the nearest to the outpost and dock if one is found
646 float closestDistance = 0.0f;
647 DockingPort? myPort = null, outPostPort = null;
648 foreach (DockingPort port in DockingPort.List)
649 {
650 if (port.IsHorizontal || port.Docked) { continue; }
651 if (port.Item.Submarine == level.StartOutpost)
652 {
653 if (port.DockingTarget == null)
654 {
655 outPostPort = port;
656 }
657 continue;
658 }
659 if (port.Item.Submarine != Submarine) { continue; }
660
661 //the submarine port has to be at the top of the sub
662 if (port.Item.WorldPosition.Y < Submarine.WorldPosition.Y) { continue; }
663
664 float dist = Vector2.DistanceSquared(port.Item.WorldPosition, level.StartOutpost.WorldPosition);
665 if ((myPort == null || dist < closestDistance || port.MainDockingPort) && !(myPort?.MainDockingPort ?? false))
666 {
667 myPort = port;
668 closestDistance = dist;
669 }
670 }
671
672 if (myPort != null && outPostPort != null)
673 {
674 Vector2 portDiff = myPort.Item.WorldPosition - Submarine.WorldPosition;
675 Vector2 spawnPos = (outPostPort.Item.WorldPosition - portDiff) - Vector2.UnitY * outPostPort.DockedDistance;
676
677 bool startDocked = level.Type == LevelData.LevelType.Outpost;
678#if CLIENT
679 startDocked |= GameMode is TutorialMode;
680#endif
681 if (startDocked)
682 {
683 Submarine.SetPosition(spawnPos);
684 myPort.Dock(outPostPort);
685 myPort.Lock(isNetworkMessage: true, applyEffects: false);
686 }
687 else
688 {
689 Submarine.SetPosition(spawnPos - Vector2.UnitY * 100.0f);
690 Submarine.NeutralizeBallast();
691 Submarine.EnableMaintainPosition();
692 }
693 }
694 else
695 {
696 Submarine.NeutralizeBallast();
697 Submarine.EnableMaintainPosition();
698 }
699 }
700 else
701 {
702 Submarine.SetPosition(Submarine.FindSpawnPos(level.StartPosition));
703 Submarine.NeutralizeBallast();
704 Submarine.EnableMaintainPosition();
705 }
706
707 // Make sure that linked subs which are NOT docked to the main sub
708 // (but still close enough to NOT be considered as 'left behind')
709 // are also moved to keep their relative position to the main sub

1026 excessPackages.Add(cp.Name);
1027 }
1028 }

1040 }
