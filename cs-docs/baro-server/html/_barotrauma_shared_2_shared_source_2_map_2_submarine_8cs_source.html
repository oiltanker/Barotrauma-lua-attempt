317 List<Pump> pumps = new List<Pump>();

326 pumps.Add(pump);

328 }
329
330 if (!pumps.Any()) { return; }
331
332 Pump randomPump = pumps.GetRandom(Rand.RandSync.Unsynced);
333 if (randomPump.IsOn && randomPump.HasPower && randomPump.FlowPercentage > 0 && randomPump.Item.Condition > 0.0f)
334 {
335 randomPump.InfectBallast(identifier);

433 public Vector2 FindSpawnPos(Vector2 spawnPos, Point? submarineSize = null, float subDockingPortOffset = 0.0f, int verticalMoveDir = 0)
434 {
435 Rectangle dockedBorders = GetDockedBorders();
436 Vector2 diffFromDockedBorders =
437 new Vector2(dockedBorders.Center.X, dockedBorders.Y - dockedBorders.Height / 2)
438 - new Vector2(Borders.Center.X, Borders.Y - Borders.Height / 2);
439
440 int minWidth = Math.Max(submarineSize.HasValue ? submarineSize.Value.X : dockedBorders.Width, 500);
441 int minHeight = Math.Max(submarineSize.HasValue ? submarineSize.Value.Y : dockedBorders.Height, 1000);
442 //a bit of extra padding to prevent the sub from spawning in a super tight gap between walls
443 int padding = 100;
444 minWidth += padding;
445 minHeight += padding;
446
447 Vector2 limits = GetHorizontalLimits(spawnPos, minWidth, minHeight, 0);
448 if (verticalMoveDir != 0)
449 {
450 verticalMoveDir = Math.Sign(verticalMoveDir);
451 //do a raycast towards the top/bottom of the level depending on direction
452 Vector2 potentialPos = new Vector2(spawnPos.X, verticalMoveDir > 0 ? Level.Loaded.Size.Y : 0);
453 if (PickBody(ConvertUnits.ToSimUnits(spawnPos), ConvertUnits.ToSimUnits(potentialPos), collisionCategory: Physics.CollisionLevel | Physics.CollisionWall) != null)
454 {
455 //if the raycast hit a wall, attempt to place the spawnpos there
456 potentialPos.Y = ConvertUnits.ToDisplayUnits(LastPickedPosition.Y) - 10;
457 }
458
459 //step away from the top/bottom of the level, or from whatever wall the raycast hit,
460 //until we found a spot where there's enough room to place the sub
461 float dist = Math.Abs(potentialPos.Y - spawnPos.Y);
462 for (float d = dist; d > 0; d -= 100.0f)
463 {
464 float y = spawnPos.Y + verticalMoveDir * d;
465 limits = GetHorizontalLimits(new Vector2(spawnPos.X, y), minWidth, minHeight, verticalMoveDir);
466 if (limits.Y - limits.X > minWidth)
467 {
468 spawnPos = new Vector2(spawnPos.X, y - (dockedBorders.Height * 0.5f * verticalMoveDir));
469 break;
470 }
471 }
472 }
473
474 static Vector2 GetHorizontalLimits(Vector2 spawnPos, float minWidth, float minHeight, int verticalMoveDir)
475 {
476 Vector2 refPos = spawnPos - Vector2.UnitY * minHeight * 0.5f * Math.Sign(verticalMoveDir);
477
478 float minX = float.MinValue, maxX = float.MaxValue;
479 foreach (VoronoiCell cell in Level.Loaded.GetAllCells())
480 {
481 foreach (GraphEdge e in cell.Edges)
482 {
483 if ((e.Point1.Y < refPos.Y - minHeight * 0.5f && e.Point2.Y < refPos.Y - minHeight * 0.5f) ||
484 (e.Point1.Y > refPos.Y + minHeight * 0.5f && e.Point2.Y > refPos.Y + minHeight * 0.5f))
485 {
486 continue;
487 }
488
489 if (cell.Site.Coord.X < refPos.X)
490 {
491 minX = Math.Max(minX, Math.Max(e.Point1.X, e.Point2.X));
492 }
493 else
494 {
495 maxX = Math.Min(maxX, Math.Min(e.Point1.X, e.Point2.X));
496 }
497 }
498 }
499
500 foreach (var ruin in Level.Loaded.Ruins)
501 {
502 if (Math.Abs(ruin.Area.Center.Y - refPos.Y) > (minHeight + ruin.Area.Height) * 0.5f) { continue; }
503 if (ruin.Area.Center.X < refPos.X)
504 {
505 minX = Math.Max(minX, ruin.Area.Right + 100.0f);
506 }
507 else
508 {
509 maxX = Math.Min(maxX, ruin.Area.X - 100.0f);
510 }
511 }
512 return new Vector2(Math.Max(minX, spawnPos.X - minWidth), Math.Min(maxX, spawnPos.X + minWidth));
513 }
514
515 if (limits.X < 0.0f && limits.Y > Level.Loaded.Size.X)
516 {
517 //no walls found at either side, just use the initial spawnpos and hope for the best
518 }
519 else if (limits.X < 0)
520 {
521 //no wall found at the left side, spawn to the left from the right-side wall
522 spawnPos.X = limits.Y - minWidth * 0.5f - 100.0f + subDockingPortOffset;
523 }
524 else if (limits.Y > Level.Loaded.Size.X)
525 {
526 //no wall found at right side, spawn to the right from the left-side wall
527 spawnPos.X = limits.X + minWidth * 0.5f + 100.0f + subDockingPortOffset;
528 }
529 else
530 {
531 //walls found at both sides, use their midpoint
532 spawnPos.X = (limits.X + limits.Y) / 2 + subDockingPortOffset;
533 }
534
535 spawnPos.Y = MathHelper.Clamp(spawnPos.Y, dockedBorders.Height / 2 + 10, Level.Loaded.Size.Y - dockedBorders.Height / 2 - padding * 2);
536 return spawnPos - diffFromDockedBorders;
537 }

557 public Rectangle CalculateDimensions(bool onlyHulls = true)
558 {
559 List<MapEntity> entities = onlyHulls ?
560 Hull.HullList.FindAll(h => h.Submarine == this).Cast<MapEntity>().ToList() :
561 MapEntity.mapEntityList.FindAll(me => me.Submarine == this);
562
563 //ignore items whose body is disabled (wires, items inside cabinets)
564 entities.RemoveAll(e =>
565 {
566 if (e is Item item)
567 {
568 if (item.GetComponent<Turret>() != null) { return false; }
569 if (item.body != null && !item.body.Enabled) { return true; }
570 }
571 if (e.HiddenInGame) { return true; }
572 return false;
573 });
574
575 if (entities.Count == 0) { return Rectangle.Empty; }
576
577 float minX = entities[0].Rect.X, minY = entities[0].Rect.Y - entities[0].Rect.Height;
578 float maxX = entities[0].Rect.Right, maxY = entities[0].Rect.Y;
579
580 for (int i = 1; i < entities.Count; i++)
581 {
582 if (entities[i] is Item item)
583 {
584 var turret = item.GetComponent<Turret>();
585 if (turret != null)
586 {
587 minX = Math.Min(minX, entities[i].Rect.X + turret.TransformedBarrelPos.X * 2f);
588 minY = Math.Min(minY, entities[i].Rect.Y - entities[i].Rect.Height - turret.TransformedBarrelPos.Y * 2f);
589 maxX = Math.Max(maxX, entities[i].Rect.Right + turret.TransformedBarrelPos.X * 2f);
590 maxY = Math.Max(maxY, entities[i].Rect.Y - turret.TransformedBarrelPos.Y * 2f);
591 }
592 }
593 minX = Math.Min(minX, entities[i].Rect.X);
594 minY = Math.Min(minY, entities[i].Rect.Y - entities[i].Rect.Height);
595 maxX = Math.Max(maxX, entities[i].Rect.Right);
596 maxY = Math.Max(maxY, entities[i].Rect.Y);
597 }
598
599 return new Rectangle((int)minX, (int)minY, (int)(maxX - minX), (int)(maxY - minY));
600 }

646 public static Body PickBody(Vector2 rayStart, Vector2 rayEnd, IEnumerable<Body> ignoredBodies = null, Category? collisionCategory = null, bool ignoreSensors = true, Predicate<Fixture> customPredicate = null, bool allowInsideFixture = false)
647 {
648 if (Vector2.DistanceSquared(rayStart, rayEnd) < 0.0001f)
649 {
650 return null;
651 }
652
653 float closestFraction = 1.0f;
654 Vector2 closestNormal = Vector2.Zero;
655 Fixture closestFixture = null;
656 Body closestBody = null;
657 if (allowInsideFixture)
658 {
659 var aabb = new FarseerPhysics.Collision.AABB(rayStart - Vector2.One * 0.001f, rayStart + Vector2.One * 0.001f);
660 GameMain.World.QueryAABB((fixture) =>
661 {
662 if (!CheckFixtureCollision(fixture, ignoredBodies, collisionCategory, ignoreSensors, customPredicate)) { return true; }
663
664 fixture.Body.GetTransform(out FarseerPhysics.Common.Transform transform);
665 if (!fixture.Shape.TestPoint(ref transform, ref rayStart)) { return true; }
666
667 closestFraction = 0.0f;
668 closestNormal = Vector2.Normalize(rayEnd - rayStart);
669 closestFixture = fixture;
670 if (fixture.Body != null) { closestBody = fixture.Body; }
671 return false;
672 }, ref aabb);
673 if (closestFraction <= 0.0f)
674 {
675 lastPickedPosition = rayStart;
676 lastPickedFraction = closestFraction;
677 lastPickedFixture = closestFixture;
678 lastPickedNormal = closestNormal;
679 return closestBody;
680 }
681 }
682
683 GameMain.World.RayCast((fixture, point, normal, fraction) =>
684 {
685 if (!CheckFixtureCollision(fixture, ignoredBodies, collisionCategory, ignoreSensors, customPredicate)) { return -1; }
686
687 if (fraction < closestFraction)
688 {
689 closestFraction = fraction;
690 closestNormal = normal;
691 closestFixture = fixture;
692 if (fixture.Body != null) closestBody = fixture.Body;
693 }
694 return fraction;
695 }, rayStart, rayEnd, collisionCategory ?? Category.All);
696
697 lastPickedPosition = rayStart + (rayEnd - rayStart) * closestFraction;
698 lastPickedFraction = closestFraction;
699 lastPickedFixture = closestFixture;
700 lastPickedNormal = closestNormal;
701
702 return closestBody;
703 }

719 public static IEnumerable<Body> PickBodies(Vector2 rayStart, Vector2 rayEnd, IEnumerable<Body> ignoredBodies = null, Category? collisionCategory = null, bool ignoreSensors = true, Predicate<Fixture> customPredicate = null, bool allowInsideFixture = false)
720 {
721 if (Vector2.DistanceSquared(rayStart, rayEnd) < 0.00001f)
722 {
723 rayEnd += Vector2.UnitX * 0.001f;
724 }
725
726 float closestFraction = 1.0f;
727 bodies.Clear();
728 bodyDist.Clear();
729 GameMain.World.RayCast((fixture, point, normal, fraction) =>
730 {
731 if (!CheckFixtureCollision(fixture, ignoredBodies, collisionCategory, ignoreSensors, customPredicate)) { return -1; }
732
733 if (fixture.Body != null)
734 {
735 bodies.Add(fixture.Body);
736 bodyDist[fixture.Body] = fraction;
737 }
738 if (fraction < closestFraction)
739 {
740 lastPickedPosition = rayStart + (rayEnd - rayStart) * fraction;
741 lastPickedFraction = fraction;
742 lastPickedNormal = normal;
743 lastPickedFixture = fixture;
744 }
745 //continue
746 return -1;
747 }, rayStart, rayEnd, collisionCategory ?? Category.All);
748
749 if (allowInsideFixture)
750 {
751 var aabb = new FarseerPhysics.Collision.AABB(rayStart - Vector2.One * 0.001f, rayStart + Vector2.One * 0.001f);
752 GameMain.World.QueryAABB((fixture) =>
753 {
754 if (bodies.Contains(fixture.Body) || fixture.Body == null) { return true; }
755 if (!CheckFixtureCollision(fixture, ignoredBodies, collisionCategory, ignoreSensors, customPredicate)) { return true; }
756
757 fixture.Body.GetTransform(out FarseerPhysics.Common.Transform transform);
758 if (!fixture.Shape.TestPoint(ref transform, ref rayStart)) { return true; }
759
760 closestFraction = 0.0f;
761 lastPickedPosition = rayStart;
762 lastPickedFraction = 0.0f;
763 lastPickedNormal = Vector2.Normalize(rayEnd - rayStart);
764 lastPickedFixture = fixture;
765 bodies.Add(fixture.Body);
766 bodyDist[fixture.Body] = 0.0f;
767 return false;
