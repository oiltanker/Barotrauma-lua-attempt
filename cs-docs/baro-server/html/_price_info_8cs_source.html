69 public static List<PriceInfo> CreatePriceInfos(XElement element, out PriceInfo defaultPrice)
70 {
71 var priceInfos = new List<PriceInfo>();
72 defaultPrice = null;
73 int basePrice = element.GetAttributeInt("baseprice", 0);
74 int minAmount = GetMinAmount(element);
75 int maxAmount = GetMaxAmount(element);
76 int minLevelDifficulty = element.GetAttributeInt("minleveldifficulty", 0);
77 bool canBeSpecial = element.GetAttributeBool("canbespecial", true);
78 float buyingPriceMultiplier = element.GetAttributeFloat("buyingpricemultiplier", 1f);
79 bool displayNonEmpty = element.GetAttributeBool("displaynonempty", false);
80 bool soldByDefault = element.GetAttributeBool("sold", element.GetAttributeBool("soldbydefault", true));
81 foreach (XElement childElement in element.GetChildElements("price"))
82 {
83 float priceMultiplier = childElement.GetAttributeFloat("multiplier", 1.0f);
84 bool sold = childElement.GetAttributeBool("sold", soldByDefault);
85 int storeMinLevelDifficulty = childElement.GetAttributeInt("minleveldifficulty", minLevelDifficulty);
86 float storeBuyingMultiplier = childElement.GetAttributeFloat("buyingpricemultiplier", buyingPriceMultiplier);
87 string backwardsCompatibleIdentifier = childElement.GetAttributeString("locationtype", "");
88 if (!string.IsNullOrEmpty(backwardsCompatibleIdentifier))
89 {
90 backwardsCompatibleIdentifier = $"merchant{backwardsCompatibleIdentifier}";
91 }
92 string storeIdentifier = childElement.GetAttributeString("storeidentifier", backwardsCompatibleIdentifier);
93 // TODO: Add some error messages if we have defined the min or max amount while the item is not sold
94 var priceInfo = new PriceInfo((int)(priceMultiplier * basePrice),
95 sold,
96 sold ? GetMinAmount(childElement, minAmount) : 0,
97 sold ? GetMaxAmount(childElement, maxAmount) : 0,
98 canBeSpecial,
99 storeMinLevelDifficulty,
100 storeBuyingMultiplier,
101 displayNonEmpty,
102 storeIdentifier);
103 priceInfos.Add(priceInfo);
104 }
105 bool soldElsewhere = soldByDefault && element.GetAttributeBool("soldelsewhere", element.GetAttributeBool("soldeverywhere", false));
106 defaultPrice = new PriceInfo(basePrice,
107 soldElsewhere,
108 soldElsewhere ? minAmount : 0,
109 soldElsewhere ? maxAmount : 0,
110 canBeSpecial,
111 minLevelDifficulty,
112 buyingPriceMultiplier,
113 displayNonEmpty);
114 return priceInfos;
115 }
