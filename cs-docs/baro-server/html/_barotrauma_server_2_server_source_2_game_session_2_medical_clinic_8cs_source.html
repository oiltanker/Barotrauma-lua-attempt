128  private RateLimitResult CheckRateLimit(Client client)
129  {
130  if (rateLimits.TryGetValue(client, out RateLimitInfo rateLimitInfo))
131  {
132  if (rateLimitInfo.Expiry < DateTimeOffset.Now)
133  {
134  rateLimitInfo.Expiry = DateTimeOffset.Now.AddSeconds(5);
135  rateLimitInfo.Requests = 1;
