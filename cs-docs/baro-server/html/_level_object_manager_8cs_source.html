58 public SpawnPosition(GraphEdge graphEdge, Vector2 normal, IEnumerable<LevelObjectPrefab.SpawnPosType> spawnPosTypes, Alignment alignment)
59 {
60 GraphEdge = graphEdge;
61 Normal = normal.NearlyEquals(Vector2.Zero) ? Vector2.UnitY : Vector2.Normalize(normal);
62 SpawnPosTypes.AddRange(spawnPosTypes);
63
64 if (spawnPosTypes.Contains(LevelObjectPrefab.SpawnPosType.MainPath) ||
65 spawnPosTypes.Contains(LevelObjectPrefab.SpawnPosType.LevelStart) ||
66 spawnPosTypes.Contains(LevelObjectPrefab.SpawnPosType.LevelEnd))
67 {
68 Length = 1000.0f;
69 Normal = Vector2.Zero;
70 Alignment = Alignment.Any;
71 }
72 else
73 {
74 Alignment = alignment;
75 Length = Vector2.Distance(graphEdge.Point1, graphEdge.Point2);
76 }
77
78 noiseVal =
79 (float)(PerlinNoise.CalculatePerlin(GraphEdge.Point1.X / 10000.0f, GraphEdge.Point1.Y / 10000.0f, 0.5f) +
80 PerlinNoise.CalculatePerlin(GraphEdge.Point1.X / 20000.0f, GraphEdge.Point1.Y / 20000.0f, 0.5f));
81 }
82
83 public float GetSpawnProbability(LevelObjectPrefab prefab)
84 {
85 if (prefab.ClusteringAmount <= 0.0f) { return Length; }
86 float noise = (noiseVal + PerlinNoise.GetPerlin(prefab.ClusteringGroup, prefab.ClusteringGroup * 0.3f)) % 1.0f;
87 return Length * (float)Math.Pow(noise, prefab.ClusteringAmount);
88 }

282 private void PlaceObject(LevelObjectPrefab prefab, SpawnPosition spawnPosition, Level level, Level.Cave parentCave = null)
283 {
284 float rotation = 0.0f;
285 if (prefab.AlignWithSurface && spawnPosition != null && spawnPosition.Normal.LengthSquared() > 0.001f)
286 {
287 rotation = MathUtils.VectorToAngle(new Vector2(spawnPosition.Normal.Y, spawnPosition.Normal.X));
288 }
289 rotation += Rand.Range(prefab.RandomRotationRad.X, prefab.RandomRotationRad.Y, Rand.RandSync.ServerAndClient);
290
291 Vector2 position = Vector2.Zero;
292 Vector2 edgeDir = Vector2.UnitX;
293 if (spawnPosition == null)
294 {
295 position = new Vector2(
296 Rand.Range(0.0f, level.Size.X, Rand.RandSync.ServerAndClient),
297 Rand.Range(0.0f, level.Size.Y, Rand.RandSync.ServerAndClient));
298 }
299 else
300 {
301 edgeDir = (spawnPosition.GraphEdge.Point1 - spawnPosition.GraphEdge.Point2) / spawnPosition.Length;
302 position = spawnPosition.GraphEdge.Point2 + edgeDir * Rand.Range(prefab.MinSurfaceWidth / 2.0f, spawnPosition.Length - prefab.MinSurfaceWidth / 2.0f, Rand.RandSync.ServerAndClient);
303 }
304
305 if (!MathUtils.NearlyEqual(prefab.RandomOffset.X, 0.0f) || !MathUtils.NearlyEqual(prefab.RandomOffset.Y, 0.0f))
306 {
307 Vector2 offsetDir = spawnPosition.Normal.LengthSquared() > 0.001f ? spawnPosition.Normal : Rand.Vector(1.0f, Rand.RandSync.ServerAndClient);
308 position += offsetDir * Rand.Range(prefab.RandomOffset.X, prefab.RandomOffset.Y, Rand.RandSync.ServerAndClient);
309 }
310
311 var newObject = new LevelObject(prefab,
312 new Vector3(position, Rand.Range(prefab.DepthRange.X, prefab.DepthRange.Y, Rand.RandSync.ServerAndClient)), Rand.Range(prefab.MinSize, prefab.MaxSize, Rand.RandSync.ServerAndClient), rotation);
313 AddObject(newObject, level);
314 newObject.ParentCave = parentCave;
315
316 foreach (LevelObjectPrefab.ChildObject child in prefab.ChildObjects)
317 {
318 int childCount = Rand.Range(child.MinCount, child.MaxCount + 1, Rand.RandSync.ServerAndClient);
319 for (int j = 0; j < childCount; j++)
320 {
321 var matchingPrefabs = LevelObjectPrefab.Prefabs.Where(p => child.AllowedNames.Contains(p.Name));
322 int prefabCount = matchingPrefabs.Count();
323 var childPrefab = prefabCount == 0 ? null : matchingPrefabs.ElementAt(Rand.Range(0, prefabCount, Rand.RandSync.ServerAndClient));
324 if (childPrefab == null) { continue; }
325
326 Vector2 childPos = position + edgeDir * Rand.Range(-0.5f, 0.5f, Rand.RandSync.ServerAndClient) * prefab.MinSurfaceWidth;
327
328 var childObject = new LevelObject(childPrefab,
329 new Vector3(childPos, Rand.Range(childPrefab.DepthRange.X, childPrefab.DepthRange.Y, Rand.RandSync.ServerAndClient)),
330 Rand.Range(childPrefab.MinSize, childPrefab.MaxSize, Rand.RandSync.ServerAndClient),
331 rotation + Rand.Range(childPrefab.RandomRotationRad.X, childPrefab.RandomRotationRad.Y, Rand.RandSync.ServerAndClient));
332
333 AddObject(childObject, level);
334 childObject.ParentCave = parentCave;
335 }
336 }

339 private void AddObject(LevelObject newObject, Level level)
340 {
341 if (newObject.Triggers != null)
342 {
343 foreach (LevelTrigger trigger in newObject.Triggers)
344 {
345 trigger.OnTriggered += (levelTrigger, obj) =>
346 {
347 OnObjectTriggered(newObject, levelTrigger, obj);
348 };
349 }
350 }
351
352 var spriteCorners = new List<Vector2>
353 {
354 Vector2.Zero, Vector2.Zero, Vector2.Zero, Vector2.Zero
355 };
356
357 Sprite sprite = newObject.Sprite ?? newObject.Prefab.DeformableSprite?.Sprite;
358
359 //calculate the positions of the corners of the rotated sprite
360 if (sprite != null)
361 {
362 Vector2 halfSize = sprite.size * newObject.Scale / 2;
363 spriteCorners[0] = -halfSize;
364 spriteCorners[1] = new Vector2(-halfSize.X, halfSize.Y);
365 spriteCorners[2] = halfSize;
366 spriteCorners[3] = new Vector2(halfSize.X, -halfSize.Y);
367
368 Vector2 pivotOffset = sprite.Origin * newObject.Scale - halfSize;
369 pivotOffset.X = -pivotOffset.X;
370 pivotOffset = new Vector2(
371 (float)(pivotOffset.X * Math.Cos(-newObject.Rotation) - pivotOffset.Y * Math.Sin(-newObject.Rotation)),
372 (float)(pivotOffset.X * Math.Sin(-newObject.Rotation) + pivotOffset.Y * Math.Cos(-newObject.Rotation)));
373
374 for (int j = 0; j < 4; j++)
375 {
376 spriteCorners[j] = new Vector2(
377 (float)(spriteCorners[j].X * Math.Cos(-newObject.Rotation) - spriteCorners[j].Y * Math.Sin(-newObject.Rotation)),
378 (float)(spriteCorners[j].X * Math.Sin(-newObject.Rotation) + spriteCorners[j].Y * Math.Cos(-newObject.Rotation)));
379
380 spriteCorners[j] += new Vector2(newObject.Position.X, newObject.Position.Y) + pivotOffset;
381 }
382 }
383
384 float minX = spriteCorners.Min(c => c.X) - newObject.Position.Z;
385 float maxX = spriteCorners.Max(c => c.X) + newObject.Position.Z;
386
387 float minY = spriteCorners.Min(c => c.Y) - newObject.Position.Z - level.BottomPos;
388 float maxY = spriteCorners.Max(c => c.Y) + newObject.Position.Z - level.BottomPos;
389
390 if (newObject.Triggers != null)
391 {
392 foreach (LevelTrigger trigger in newObject.Triggers)
393 {
394 if (trigger.PhysicsBody == null) { continue; }
395 for (int i = 0; i < trigger.PhysicsBody.FarseerBody.FixtureList.Count; i++)
396 {
397 trigger.PhysicsBody.FarseerBody.GetTransform(out FarseerPhysics.Common.Transform transform);
398 trigger.PhysicsBody.FarseerBody.FixtureList[i].Shape.ComputeAABB(out FarseerPhysics.Collision.AABB aabb, ref transform, i);
399
400 minX = Math.Min(minX, ConvertUnits.ToDisplayUnits(aabb.LowerBound.X));
401 maxX = Math.Max(maxX, ConvertUnits.ToDisplayUnits(aabb.UpperBound.X));
402 minY = Math.Min(minY, ConvertUnits.ToDisplayUnits(aabb.LowerBound.Y) - level.BottomPos);
403 maxY = Math.Max(maxY, ConvertUnits.ToDisplayUnits(aabb.UpperBound.Y) - level.BottomPos);
404 }
405 }
406 }
407
408#if CLIENT
409 if (newObject.ParticleEmitters != null)
410 {
411 foreach (ParticleEmitter emitter in newObject.ParticleEmitters)
412 {
413 Rectangle particleBounds = emitter.CalculateParticleBounds(new Vector2(newObject.Position.X, newObject.Position.Y));
414 minX = Math.Min(minX, particleBounds.X);
415 maxX = Math.Max(maxX, particleBounds.Right);
416 minY = Math.Min(minY, particleBounds.Y - level.BottomPos);
417 maxY = Math.Max(maxY, particleBounds.Bottom - level.BottomPos);
418 }
419 }
420#endif
421 objects.Add(newObject);
422 if (newObject.NeedsUpdate) { updateableObjects.Add(newObject); }
423 newObject.Position.Z += (minX + minY) % 100.0f * 0.00001f;
424
425 int xStart = (int)Math.Floor(minX / GridSize);
426 int xEnd = (int)Math.Floor(maxX / GridSize);
427 if (xEnd < 0 || xStart >= objectGrid.GetLength(0)) return;
428
429 int yStart = (int)Math.Floor(minY / GridSize);
430 int yEnd = (int)Math.Floor(maxY / GridSize);
431 if (yEnd < 0 || yStart >= objectGrid.GetLength(1)) return;
432
433 xStart = Math.Max(xStart, 0);
434 xEnd = Math.Min(xEnd, objectGrid.GetLength(0) - 1);
435 yStart = Math.Max(yStart, 0);
436 yEnd = Math.Min(yEnd, objectGrid.GetLength(1) - 1);
437
438 for (int x = xStart; x <= xEnd; x++)
439 {
440 for (int y = yStart; y <= yEnd; y++)
441 {
442 if (objectGrid[x, y] == null) objectGrid[x, y] = new List<LevelObject>();
443 objectGrid[x, y].Add(newObject);
444 }
445 }
446 }

460 private readonly static HashSet<LevelObject> objectsInRange = new HashSet<LevelObject>();
461 public IEnumerable<LevelObject> GetAllObjects(Vector2 worldPosition, float radius)
462 {
463 var minIndices = GetGridIndices(worldPosition - Vector2.One * radius);
464 if (minIndices.X >= objectGrid.GetLength(0) || minIndices.Y >= objectGrid.GetLength(1)) { return Enumerable.Empty<LevelObject>(); }
465
466 var maxIndices = GetGridIndices(worldPosition + Vector2.One * radius);
467 if (maxIndices.X < 0 || maxIndices.Y < 0) { return Enumerable.Empty<LevelObject>(); }
468
469 minIndices.X = Math.Max(0, minIndices.X);
470 minIndices.Y = Math.Max(0, minIndices.Y);
471 maxIndices.X = Math.Min(objectGrid.GetLength(0) - 1, maxIndices.X);
472 maxIndices.Y = Math.Min(objectGrid.GetLength(1) - 1, maxIndices.Y);
473
474 objectsInRange.Clear();
475 for (int x = minIndices.X; x <= maxIndices.X; x++)
476 {
477 for (int y = minIndices.Y; y <= maxIndices.Y; y++)
478 {
479 if (objectGrid[x, y] == null) { continue; }
480 foreach (LevelObject obj in objectGrid[x, y])
481 {
482 if (obj.Prefab.HideWhenBroken && obj.Health <= 0.0f) { continue; }
483 objectsInRange.Add(obj);
484 }
485 }
486 }
487
488 return objectsInRange;
489 }

491 private List<SpawnPosition> GetAvailableSpawnPositions(IEnumerable<VoronoiCell> cells, LevelObjectPrefab.SpawnPosType spawnPosType, bool checkFlags = true)
492 {
493 List<LevelObjectPrefab.SpawnPosType> spawnPosTypes = new List<LevelObjectPrefab.SpawnPosType>(4);
494 List<SpawnPosition> availableSpawnPositions = new List<SpawnPosition>();
495 foreach (var cell in cells)
496 {
497 foreach (var edge in cell.Edges)
498 {
499 if (!edge.IsSolid || edge.OutsideLevel) { continue; }
500 if (spawnPosType != LevelObjectPrefab.SpawnPosType.CaveWall && edge.NextToCave) { continue; }
501 Vector2 normal = edge.GetNormal(cell);
502
503 Alignment edgeAlignment = 0;
504 if (normal.Y < -0.5f)
505 edgeAlignment |= Alignment.Bottom;
506 else if (normal.Y > 0.5f)
507 edgeAlignment |= Alignment.Top;
508 else if (normal.X < -0.5f)
509 edgeAlignment |= Alignment.Left;
510 else if(normal.X > 0.5f)
511 edgeAlignment |= Alignment.Right;
512
513 spawnPosTypes.Clear();
514 spawnPosTypes.Add(spawnPosType);
515 if (spawnPosType.HasFlag(LevelObjectPrefab.SpawnPosType.MainPathWall) && edge.NextToMainPath) { spawnPosTypes.Add(LevelObjectPrefab.SpawnPosType.MainPathWall); }
516 if (spawnPosType.HasFlag(LevelObjectPrefab.SpawnPosType.SidePathWall) && edge.NextToSidePath) { spawnPosTypes.Add(LevelObjectPrefab.SpawnPosType.SidePathWall); }
517 if (spawnPosType.HasFlag(LevelObjectPrefab.SpawnPosType.CaveWall) && edge.NextToCave) { spawnPosTypes.Add(LevelObjectPrefab.SpawnPosType.CaveWall); }
518
519 availableSpawnPositions.Add(new SpawnPosition(edge, normal, spawnPosTypes, edgeAlignment));
520 }
521 }
522 return availableSpawnPositions;
523 }

525 public void Update(float deltaTime)
526 {
527 GlobalForceDecreaseTimer += deltaTime;
528 if (GlobalForceDecreaseTimer > 1000000.0f)
529 {
530 GlobalForceDecreaseTimer = 0.0f;
531 }
532
533 foreach (LevelObject obj in updateableObjects)
534 {
535 if (GameMain.NetworkMember is { IsServer: true })
536 {
537 obj.NetworkUpdateTimer -= deltaTime;
538 if (obj.NeedsNetworkSyncing && obj.NetworkUpdateTimer <= 0.0f)
539 {
540 GameMain.NetworkMember.CreateEntityEvent(this, new EventData(obj));
541 obj.NeedsNetworkSyncing = false;
542 obj.NetworkUpdateTimer = NetConfig.LevelObjectUpdateInterval;
543 }
544 }
