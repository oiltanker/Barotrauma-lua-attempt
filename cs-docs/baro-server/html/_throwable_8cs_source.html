28 : base(item, element)

30 //throwForce = ToolBox.GetAttributeFloat(element, "throwforce", 1.0f);

62 public override void Update(float deltaTime, Camera cam)
63 {
64 if (!item.body.Enabled) { return; }
65 if (midAir)
66 {
67 if (item.body.FarseerBody.IsBullet)
68 {
69 if (item.body.LinearVelocity.LengthSquared() < ContinuousCollisionThreshold * ContinuousCollisionThreshold)
70 {
71 item.body.FarseerBody.IsBullet = false;
72 }
73 }
74 if (item.body.LinearVelocity.LengthSquared() < 0.01f)
75 {
76 CurrentThrower = null;
77 if (statusEffectLists?.ContainsKey(ActionType.OnImpact) ?? false)
78 {
79 foreach (var statusEffect in statusEffectLists[ActionType.OnImpact])
80 {
81 statusEffect.SetUser(null);
82 }
83 }
84 if (statusEffectLists?.ContainsKey(ActionType.OnBroken) ?? false)
85 {
86 foreach (var statusEffect in statusEffectLists[ActionType.OnBroken])
87 {
88 statusEffect.SetUser(null);
89 }
90 }
91 item.body.CollidesWith = Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionPlatform;
92 midAir = false;
93 }
94 return;
95 }
96
97 if (picker == null || picker.Removed || !picker.HeldItems.Contains(item))
98 {
99 IsActive = false;
100 return;
101 }
102
103 if (picker.IsKeyDown(InputType.Aim) && picker.IsKeyHit(InputType.Shoot)) { throwing = true; }
104 if (!picker.IsKeyDown(InputType.Aim) && !throwing) { throwPos = 0.0f; }
105 bool aim = picker.IsKeyDown(InputType.Aim) && picker.CanAim;
106
107 if (picker.IsDead || !picker.AllowInput)
108 {
109 throwing = false;
110 aim = false;
111 }
112
113 ApplyStatusEffects(ActionType.OnActive, deltaTime, picker);
114
115 if (item.body.Dir != picker.AnimController.Dir) { item.FlipX(relativeToSub: false); }
116
117 AnimController ac = picker.AnimController;
118
119 item.Submarine = picker.Submarine;
120
121 if (!throwing)
122 {
123 if (aim)
124 {
125 throwPos = MathUtils.WrapAnglePi(System.Math.Min(throwPos + deltaTime * 5.0f, MathHelper.PiOver2));
126 ac.HoldItem(deltaTime, item, handlePos, aimPos, Vector2.Zero, aim: false, throwPos);
127 }
128 else
129 {
130 throwPos = 0;
131 ac.HoldItem(deltaTime, item, handlePos, holdPos, Vector2.Zero, aim: false, holdAngle);
132 }
133 }
134 else
135 {
136 throwPos = MathUtils.WrapAnglePi(throwPos - deltaTime * 15.0f);
137 ac.HoldItem(deltaTime, item, handlePos, aimPos, Vector2.Zero, aim: false, throwPos);
138
139 if (throwPos < 0)
140 {
141 Vector2 throwVector = Vector2.Normalize(picker.CursorWorldPosition - picker.WorldPosition);
142 //throw upwards if cursor is at the position of the character
143 if (!MathUtils.IsValid(throwVector)) { throwVector = Vector2.UnitY; }
144
145#if SERVER
146 GameServer.Log(GameServer.CharacterLogName(picker) + " threw " + item.Name, ServerLog.MessageType.ItemInteraction);
147#endif
148 CurrentThrower = picker;
149 if (statusEffectLists?.ContainsKey(ActionType.OnImpact) ?? false)
150 {
151 foreach (var statusEffect in statusEffectLists[ActionType.OnImpact])
152 {
153 statusEffect.SetUser(CurrentThrower);
154 }
155 }
156 if (statusEffectLists?.ContainsKey(ActionType.OnBroken) ?? false)
157 {
158 foreach (var statusEffect in statusEffectLists[ActionType.OnBroken])
159 {
160 statusEffect.SetUser(CurrentThrower);
161 }
162 }
163
164 item.Drop(CurrentThrower, createNetworkEvent: GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer);
165 item.body.ApplyLinearImpulse(throwVector * ThrowForce * item.body.Mass * 3.0f, maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
166
167 //disable platform collisions until the item comes back to rest again
168 item.body.CollidesWith = Physics.CollisionWall | Physics.CollisionLevel;
169 item.body.FarseerBody.IsBullet = true;
170 midAir = true;
171
172 ac.GetLimb(LimbType.Head)?.body.ApplyLinearImpulse(throwVector * 10.0f, maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
173 ac.GetLimb(LimbType.Torso)?.body.ApplyLinearImpulse(throwVector * 10.0f, maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
174
175 Limb rightHand = ac.GetLimb(LimbType.RightHand);
176 item.body.AngularVelocity = rightHand.body.AngularVelocity;
177 throwPos = 0;
178 throwDone = true;
179 IsActive = true;
180
181 if (GameMain.NetworkMember is { IsServer: true })
182 {
183 GameMain.NetworkMember.CreateEntityEvent(item, new Item.ApplyStatusEffectEventData(ActionType.OnSecondaryUse, this, CurrentThrower));
184 }
185 if (!(GameMain.NetworkMember is { IsClient: true }))
186 {
187 //Stun grenades, flares, etc. all have their throw-related things handled in "onSecondaryUse"
188 ApplyStatusEffects(ActionType.OnSecondaryUse, deltaTime, CurrentThrower, user: CurrentThrower);
189 }
190 throwing = false;
191 }
192 }
193 }
194 }
