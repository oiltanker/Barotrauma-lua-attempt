99 subs = new Submarine[] { Submarine.MainSubs[0], Submarine.MainSubs[1] };

101 subs[0].NeutralizeBallast();
102 subs[0].TeamID = CharacterTeamType.Team1;
103 subs[0].GetConnectedSubs().ForEach(s => s.TeamID = CharacterTeamType.Team1);
104
105 subs[1].NeutralizeBallast();
106 subs[1].TeamID = CharacterTeamType.Team2;
107 subs[1].GetConnectedSubs().ForEach(s => s.TeamID = CharacterTeamType.Team2);
108 subs[1].SetPosition(subs[1].FindSpawnPos(Level.Loaded.EndPosition));
109 subs[1].FlipX();
110#if SERVER
111 crews = new List<Character>[] { new List<Character>(), new List<Character>() };
112#endif
113 }
