304  {

306  {
307  if (outMsg == null) { throw new ArgumentException("OutMsg was null"); }

319  outMsg.WritePadBits();

321  for (int i = 0; i < bannedPlayers.Count; i++)
322  {
323  BannedPlayer bannedPlayer = bannedPlayers[i];
324 
325  outMsg.Write(bannedPlayer.Name);
326  outMsg.Write(bannedPlayer.UniqueIdentifier);
327  outMsg.Write(bannedPlayer.IsRangeBan);
328  outMsg.Write(bannedPlayer.ExpirationTime != null);
329  outMsg.WritePadBits();
330  if (bannedPlayer.ExpirationTime != null)
331  {
332  double hoursFromNow = (bannedPlayer.ExpirationTime.Value - DateTime.Now).TotalHours;
333  outMsg.Write(hoursFromNow);
334  }
335 
336  outMsg.Write(bannedPlayer.Reason ?? "");
337 
338  if (c.Connection == GameMain.Server.OwnerConnection)
339  {
340  outMsg.Write(bannedPlayer.EndPoint);
341  outMsg.Write(bannedPlayer.SteamID);

343  }
344  }

354  public bool ServerAdminRead(IReadMessage incMsg, Client c)
355  {
356  if (!c.HasPermission(ClientPermissions.Ban))
357  {
358  UInt16 removeCount = incMsg.ReadUInt16();
359  incMsg.BitPosition += removeCount * 4 * 8;
360  UInt16 rangeBanCount = incMsg.ReadUInt16();
361  incMsg.BitPosition += rangeBanCount * 4 * 8;
362  return false;
363  }
364  else
365  {
366  UInt16 removeCount = incMsg.ReadUInt16();
367  for (int i = 0; i < removeCount; i++)
368  {
369  UInt16 id = incMsg.ReadUInt16();
370  BannedPlayer bannedPlayer = bannedPlayers.Find(p => p.UniqueIdentifier == id);
371  if (bannedPlayer != null)
372  {
373  GameServer.Log(GameServer.ClientLogName(c) + " unbanned " + bannedPlayer.Name + " (" + bannedPlayer.EndPoint + ")", ServerLog.MessageType.ConsoleUsage);
374  RemoveBan(bannedPlayer);
375  }
376  }
377  Int16 rangeBanCount = incMsg.ReadInt16();
378  for (int i = 0; i < rangeBanCount; i++)
379  {
380  UInt16 id = incMsg.ReadUInt16();
381  BannedPlayer bannedPlayer = bannedPlayers.Find(p => p.UniqueIdentifier == id);
382  if (bannedPlayer != null)
383  {
384  GameServer.Log(GameServer.ClientLogName(c) + " rangebanned " + bannedPlayer.Name + " (" + bannedPlayer.EndPoint + ")", ServerLog.MessageType.ConsoleUsage);
385  RangeBan(bannedPlayer);
386  }
387  }
388 
389  return removeCount > 0 || rangeBanCount > 0;
390  }
