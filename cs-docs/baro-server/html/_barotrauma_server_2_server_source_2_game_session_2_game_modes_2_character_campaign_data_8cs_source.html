55 public CharacterCampaignData(XElement element)
56 {
57 Name = element.GetAttributeString("name", "Unnamed");
58 ClientEndPoint = element.GetAttributeString("endpoint", null) ?? element.GetAttributeString("ip", "");
59 string steamID = element.GetAttributeString("steamid", "");
60 if (!string.IsNullOrEmpty(steamID))
61 {
62 ulong.TryParse(steamID, out ulong parsedID);
63 SteamID = parsedID;
64 }
65
66 foreach (XElement subElement in element.Elements())
67 {
68 switch (subElement.Name.ToString().ToLowerInvariant())
69 {
70 case "character":
71 case "characterinfo":
72 CharacterInfo = new CharacterInfo(subElement);
73 break;
74 case "inventory":
75 itemData = subElement;
76 break;
77 case "health":
78 healthData = subElement;
79 break;
80 case "orders":
81 OrderData = subElement;
82 break;
83 case Wallet.LowerCaseSaveElementName:
84 WalletData = subElement;
85 break;
86 }
87 }
