144  protected virtual Vector2 DoSteeringAvoid(float deltaTime, float lookAheadDistance, float weight, Vector2? heading = null)
145  {
146  if (steering == Vector2.Zero || host.Steering == Vector2.Zero)
147  {
148  return Vector2.Zero;
149  }
150 
151  float maxDistance = lookAheadDistance;
152  if (Timing.TotalTime >= lastRayCastTime + RayCastInterval)
153  {
154  avoidRayCastHit = false;
155  AvoidLookAheadPos = host.SimPosition + Vector2.Normalize(host.Steering) * maxDistance;
156  lastRayCastTime = (float)Timing.TotalTime;
157  Body closestBody = Submarine.CheckVisibility(host.SimPosition, AvoidLookAheadPos);
158  if (closestBody != null)
159  {
160  avoidRayCastHit = true;
161  AvoidRayCastHitPosition = Submarine.LastPickedPosition;
162  AvoidDir = Submarine.LastPickedNormal;
163  //add a bit of randomness
164  AvoidDir = MathUtils.RotatePoint(AvoidDir, Rand.Range(-0.15f, 0.15f));
165  //wait a bit longer for the next raycast
166  lastRayCastTime += RayCastInterval;
167  }
168  }
169 
170  if (AvoidDir.LengthSquared() < 0.0001f) { return Vector2.Zero; }
171 
172  //if raycast hit nothing, lerp avoid dir to zero
173  if (!avoidRayCastHit)
174  {
175  AvoidDir -= Vector2.Normalize(AvoidDir) * deltaTime * 0.5f;
176  }
177 
178  Vector2 diff = AvoidRayCastHitPosition - host.SimPosition;
179  float dist = diff.Length();
180 
181  //> 0 when heading in the same direction as the obstacle, < 0 when away from it
182  float dot = MathHelper.Clamp(Vector2.Dot(diff / dist, host.Steering), 0.0f, 1.0f);
183  if (dot < 0) { return Vector2.Zero; }
184 
185  return AvoidDir * dot * weight * MathHelper.Clamp(1.0f - dist / lookAheadDistance, 0.0f, 1.0f);
186  }
