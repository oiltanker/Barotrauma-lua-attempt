81 public void AddObjective<T>(T objective) where T : AIObjective
82 {
83 var result = GameMain.LuaCs.Hook.Call<bool?>("AI.AddObjective", this, objective);
84
85 if (result != null && result.Value) return;
86
87 if (objective == null)
88 {
89#if DEBUG
90 DebugConsole.ThrowError("Attempted to add a null objective to AIObjectiveManager\n" + Environment.StackTrace.CleanupStackTrace());
91#endif
92 return;
93 }
94 // Can't use the generic type, because it's possible that the user of this method uses the base type AIObjective.
95 // We need to get the highest type.
96 var type = objective.GetType();
97 if (objective.AllowMultipleInstances)
98 {
99 if (Objectives.FirstOrDefault(o => o.GetType() == type) is T existingObjective && existingObjective.IsDuplicate(objective))
100 {
101 Objectives.Remove(existingObjective);
102 }
103 }
104 else
105 {
106 Objectives.RemoveAll(o => o.GetType() == type);
107 }
108 Objectives.Add(objective);
109 }

123 public void CreateAutonomousObjectives()
124 {
125 if (character.IsDead)
126 {
127#if DEBUG
128 DebugConsole.ThrowError("Attempted to create autonomous orders for a dead character");
129#else
130 return;
131#endif
132 }
133 foreach (var delayedObjective in DelayedObjectives)
134 {
135 CoroutineManager.StopCoroutines(delayedObjective.Value);
136 }
137 DelayedObjectives.Clear();
138 Objectives.Clear();
139 FailedAutonomousObjectives = false;
140 AddObjective(new AIObjectiveFindSafety(character, this));
141 AddObjective(new AIObjectiveIdle(character, this));
142 int objectiveCount = Objectives.Count;
143 foreach (var autonomousObjective in character.Info.Job.Prefab.AutonomousObjectives)
144 {
145 var orderPrefab = OrderPrefab.Prefabs[autonomousObjective.Identifier];
146 if (orderPrefab == null) { throw new Exception($"Could not find a matching prefab by the identifier: '{autonomousObjective.Identifier}'"); }
147 Item item = null;
148 if (orderPrefab.MustSetTarget)
149 {
150 item = orderPrefab.GetMatchingItems(character.Submarine, mustBelongToPlayerSub: false, requiredTeam: character.Info.TeamID, interactableFor: character)?.GetRandomUnsynced();
151 }
152 var order = new Order(orderPrefab, autonomousObjective.Option, item ?? character.CurrentHull as Entity, orderPrefab.GetTargetItemComponent(item), orderGiver: character);
153 if (order == null) { continue; }
154 if ((order.IgnoreAtOutpost || autonomousObjective.IgnoreAtOutpost) && Level.IsLoadedOutpost && character.TeamID != CharacterTeamType.FriendlyNPC)
155 {
156 if (Submarine.MainSub != null && Submarine.MainSub.DockedTo.None(s => s.TeamID != CharacterTeamType.FriendlyNPC && s.TeamID != character.TeamID))
157 {
158 continue;
159 }
160 }
161 var objective = CreateObjective(order, autonomousObjective.PriorityModifier);
162 if (objective != null && objective.CanBeCompleted)
163 {
164 AddObjective(objective, delay: Rand.Value() / 2);
165 objectiveCount++;
166 }
167 }
168 _waitTimer = Math.Max(_waitTimer, Rand.Range(0.5f, 1f) * objectiveCount);
169 }

171 public void AddObjective<T>(T objective, float delay, Action callback = null) where T : AIObjective
172 {
173 if (objective == null)
174 {
175#if DEBUG
176 DebugConsole.ThrowError($"{character.Name}: Attempted to add a null objective to AIObjectiveManager\n" + Environment.StackTrace.CleanupStackTrace());
177#endif
178 return;
179 }
180 if (DelayedObjectives.TryGetValue(objective, out CoroutineHandle coroutine))
181 {
182 CoroutineManager.StopCoroutines(coroutine);
183 DelayedObjectives.Remove(objective);
184 }
185 coroutine = CoroutineManager.Invoke(() =>
186 {
187 //round ended before the coroutine finished
188#if CLIENT
189 if (GameMain.GameSession == null || Level.Loaded == null && !(GameMain.GameSession.GameMode is TestGameMode)) { return; }
190#else
191 if (GameMain.GameSession == null || Level.Loaded == null) { return; }
192#endif
193 DelayedObjectives.Remove(objective);
194 AddObjective(objective);
195 callback?.Invoke();
196 }, delay);
197 DelayedObjectives.Add(objective, coroutine);

242 void UpdateOrderObjective(AIObjective orderObjective)
243 {
244 if (orderObjective == null) { return; }
245#if DEBUG
246 // Note: don't automatically remove orders here. Removing orders needs to be done via dismissing.
247 if (!orderObjective.CanBeCompleted)
248 {
249 DebugConsole.NewMessage($"{character.Name}: ORDER {orderObjective.DebugTag}, CANNOT BE COMPLETED.", Color.Red);
250 }
251#endif
252 orderObjective.Update(deltaTime);
253 }
254
255 if (WaitTimer > 0)
256 {
257 WaitTimer -= deltaTime;
258 return;
259 }
260 for (int i = 0; i < Objectives.Count; i++)
261 {
262 var objective = Objectives[i];
263 if (objective.IsCompleted)
264 {
265#if DEBUG
266 DebugConsole.NewMessage($"{character.Name}: Removing objective {objective.DebugTag}, because it is completed.", Color.LightBlue);
267#endif
268 Objectives.Remove(objective);
269 }
270 else if (!objective.CanBeCompleted)
271 {
272#if DEBUG
273 DebugConsole.NewMessage($"{character.Name}: Removing objective {objective.DebugTag}, because it cannot be completed.", Color.Red);
274#endif
275 Objectives.Remove(objective);
276 FailedAutonomousObjectives = true;
277 }
278 else
279 {
280 objective.Update(deltaTime);
281 }
282 }

286 public void SortObjectives()
287 {
288 ForcedOrder?.CalculatePriority();
289 AIObjective orderWithHighestPriority = null;
290 float highestPriority = 0;
291 for (int i = CurrentOrders.Count - 1; i >= 0; i--)
292 {
293 if (CurrentOrders.Count <= i) { break; }
294 var orderObjective = CurrentOrders[i].Objective;
295 if (orderObjective == null) { continue; }
296 orderObjective.CalculatePriority();
297 if (orderWithHighestPriority == null || orderObjective.Priority > highestPriority)
298 {
299 orderWithHighestPriority = orderObjective;
300 highestPriority = orderObjective.Priority;
301 }
302 }
303 CurrentOrder = orderWithHighestPriority;
304 for (int i = Objectives.Count - 1; i >= 0; i--)
305 {
306 Objectives[i].CalculatePriority();
307 }
308 if (Objectives.Any())
309 {
310 Objectives.Sort((x, y) => y.Priority.CompareTo(x.Priority));
311 }
312 GetCurrentObjective()?.SortSubObjectives();
313 }

338 public void SetOrder(Order order, bool speak)
339 {
340 if (character.IsDead)
341 {
342#if DEBUG
343 DebugConsole.ThrowError("Attempted to set an order for a dead character");
344#else
345 return;
346#endif
347 }
348 ClearIgnored();
349
350 if (order == null || order.IsDismissal)
351 {
352 if (order.Option != Identifier.Empty)
353 {
354 if (CurrentOrders.Any(o => o.MatchesDismissedOrder(order.Option)))
355 {
356 var dismissedOrderInfo = CurrentOrders.First(o => o.MatchesDismissedOrder(order.Option));
357 CurrentOrders.Remove(dismissedOrderInfo);
358 }
359 }
360 else
361 {
362 CurrentOrders.Clear();
363 }
364 }
365
366 // Make sure the order priorities reflect those set by the player
367 for (int i = CurrentOrders.Count - 1; i >= 0; i--)
368 {
369 if (CurrentOrders.Count <= i) { break; }
370 var currentOrder = CurrentOrders[i];
371 if (currentOrder.Objective == null || currentOrder.MatchesOrder(order))
372 {
373 CurrentOrders.RemoveAt(i);
374 continue;
375 }
376 var currentOrderInfo = character.GetCurrentOrder(currentOrder);
377 if (currentOrderInfo is Order)
378 {
379 int currentPriority = currentOrderInfo.ManualPriority;
380 if (currentOrder.ManualPriority != currentPriority)
381 {
382 CurrentOrders[i] = currentOrder.WithManualPriority(currentPriority);
383 }
384 }
385 else
386 {
387 CurrentOrders.RemoveAt(i);
388 }
389 }
390
391 var newCurrentObjective = CreateObjective(order);
392 if (newCurrentObjective != null)
393 {
394 newCurrentObjective.Abandoned += () => DismissSelf(order);
395 CurrentOrders.Add(order.WithObjective(newCurrentObjective));
396 }
397 if (!HasOrders())
398 {
399 // Recreate objectives, because some of them may be removed, if impossible to complete (e.g. due to path finding)
400 CreateAutonomousObjectives();
401 }
402 else if (newCurrentObjective != null)
403 {
404 if (speak && character.IsOnPlayerTeam)
405 {
406 LocalizedString msg = newCurrentObjective.IsAllowed ? TextManager.Get("DialogAffirmative") : TextManager.Get("DialogNegative");
407 character.Speak(msg.Value, delay: 1.0f);
408 }
409 }

412 public AIObjective CreateObjective(Order order, float priorityModifier = 1)
413 {
414 if (order == null || order.IsDismissal) { return null; }
415 AIObjective newObjective;
416 switch (order.Identifier.Value.ToLowerInvariant())
417 {
418 case "follow":
419 if (order.OrderGiver == null) { return null; }
420 newObjective = new AIObjectiveGoTo(order.OrderGiver, character, this, repeat: true, priorityModifier: priorityModifier)
421 {
422 CloseEnough = Rand.Range(80f, 100f),
423 CloseEnoughMultiplier = Math.Min(1 + HumanAIController.CountCrew(c => c.ObjectiveManager.HasOrder<AIObjectiveGoTo>(o => o.Target == order.OrderGiver), onlyBots: true) * Rand.Range(0.8f, 1f), 4),
424 ExtraDistanceOutsideSub = 100,
425 ExtraDistanceWhileSwimming = 100,
426 AllowGoingOutside = true,
427 IgnoreIfTargetDead = true,
428 IsFollowOrderObjective = true,
429 Mimic = character.IsOnPlayerTeam,
430 DialogueIdentifier = "dialogcannotreachplace".ToIdentifier()
431 };
432 break;
433 case "wait":
434 newObjective = new AIObjectiveGoTo(order.TargetSpatialEntity ?? character, character, this, repeat: true, priorityModifier: priorityModifier)
435 {
436 AllowGoingOutside = true
437 };
438 break;
439 case "return":
440 newObjective = new AIObjectiveReturn(character, order.OrderGiver, this, priorityModifier: priorityModifier);
441 newObjective.Completed += () => DismissSelf(order);
442 break;
443 case "fixleaks":
444 newObjective = new AIObjectiveFixLeaks(character, this, priorityModifier: priorityModifier, prioritizedHull: order.TargetEntity as Hull);
445 break;
446 case "chargebatteries":
447 newObjective = new AIObjectiveChargeBatteries(character, this, order.Option, priorityModifier);
448 break;
449 case "rescue":
450 newObjective = new AIObjectiveRescueAll(character, this, priorityModifier);
451 break;
452 case "repairsystems":
453 case "repairmechanical":
454 case "repairelectrical":
455 newObjective = new AIObjectiveRepairItems(character, this, priorityModifier: priorityModifier, prioritizedItem: order.TargetEntity as Item)
456 {
457 RelevantSkill = order.AppropriateSkill,
458 };
459 break;
460 case "pumpwater":
461 if (order.TargetItemComponent is Pump targetPump)
462 {
463 if (!order.TargetItemComponent.Item.IsInteractable(character)) { return null; }
464 newObjective = new AIObjectiveOperateItem(targetPump, character, this, order.Option, false, priorityModifier: priorityModifier)
465 {
466 IsLoop = false,
467 Override = order.OrderGiver is { IsCommanding: true }
468 };
469 newObjective.Completed += () => DismissSelf(order);
470 }
471 else
472 {
473 newObjective = new AIObjectivePumpWater(character, this, order.Option, priorityModifier: priorityModifier);
474 }
475 break;
476 case "extinguishfires":
477 newObjective = new AIObjectiveExtinguishFires(character, this, priorityModifier);
478 break;
479 case "fightintruders":
480 newObjective = new AIObjectiveFightIntruders(character, this, priorityModifier);
481 break;
482 case "assaultenemy":
483 newObjective = new AIObjectiveFightIntruders(character, this, priorityModifier)
484 {
485 TargetCharactersInOtherSubs = true
486 };
487 break;
488 case "steer":
489 var steering = (order?.TargetEntity as Item)?.GetComponent<Steering>();
490 if (steering != null) { steering.PosToMaintain = steering.Item.Submarine?.WorldPosition; }
491 if (order.TargetItemComponent == null) { return null; }
492 if (!order.TargetItemComponent.Item.IsInteractable(character)) { return null; }
493 newObjective = new AIObjectiveOperateItem(order.TargetItemComponent, character, this, order.Option,
494 requireEquip: false, useController: order.UseController, controller: order.ConnectedController, priorityModifier: priorityModifier)
495 {
496 IsLoop = true,
497 // Don't override unless it's an order by a player
498 Override = order.OrderGiver != null && order.OrderGiver.IsCommanding
499 };
500 break;
501 case "setchargepct":
502 newObjective = new AIObjectiveOperateItem(order.TargetItemComponent, character, this, order.Option, false, priorityModifier: priorityModifier)
503 {
504 IsLoop = false,
505 Override = !character.IsDismissed,
506 completionCondition = () =>
507 {
508 if (float.TryParse(order.Option.Value, out float pct))
509 {
510 var targetRatio = Math.Clamp(pct, 0f, 1f);
511 var currentRatio = (order.TargetItemComponent as PowerContainer).RechargeRatio;
512 return Math.Abs(targetRatio - currentRatio) < 0.05f;
513 }
514 return true;
515 }
516 };
517 break;
518 case "getitem":
519 newObjective = new AIObjectiveGetItem(character, order.TargetEntity as Item ?? order.TargetItemComponent?.Item, this, false, priorityModifier: priorityModifier)
520 {
521 MustBeSpecificItem = true
522 };
523 break;
524 case "cleanupitems":
525 if (order.TargetEntity is Item targetItem)
526 {
527 if (targetItem.HasTag("allowcleanup") && targetItem.ParentInventory == null && targetItem.OwnInventory != null)
528 {
529 // Target all items inside the container
530 newObjective = new AIObjectiveCleanupItems(character, this, targetItem.OwnInventory.AllItems, priorityModifier);
531 }
532 else
533 {
534 newObjective = new AIObjectiveCleanupItems(character, this, targetItem, priorityModifier);
535 }
536 }
537 else
538 {
539 newObjective = new AIObjectiveCleanupItems(character, this, priorityModifier: priorityModifier);
540 }
541 break;
542 case "escapehandcuffs":
543 newObjective = new AIObjectiveEscapeHandcuffs(character, this, priorityModifier: priorityModifier);
544 break;
545 case "prepareforexpedition":
546 newObjective = new AIObjectivePrepare(character, this, order.GetTargetItems(order.Option), order.RequireItems)
547 {
548 KeepActiveWhenReady = true,
549 CheckInventory = true,
550 Equip = false,
551 FindAllItems = true

553 break;

578 default:
579 if (order.TargetItemComponent == null) { return null; }
580 if (!order.TargetItemComponent.Item.IsInteractable(character)) { return null; }
581 newObjective = new AIObjectiveOperateItem(order.TargetItemComponent, character, this, order.Option,

584 IsLoop = true,

588 if (newObjective.Abandon) { return null; }
589 break;
590 }
591 if (newObjective != null)
592 {
593 newObjective.Identifier = order.Identifier;
594 }
595 newObjective.IgnoreAtOutpost = order.IgnoreAtOutpost;

599 private void DismissSelf(Order order)
600 {
601 var currentOrder = CurrentOrders.FirstOrDefault(oi => oi.MatchesOrder(order.Identifier, order.Option));
602 if (currentOrder == null)
603 {
604#if DEBUG
605 DebugConsole.ThrowError("Tried to self-dismiss an order, but no matching current order was found");
606#endif
607 return;
608 }
609
610 Order dismissOrder = currentOrder.GetDismissal();
611#if CLIENT
612 if (GameMain.GameSession?.CrewManager != null && GameMain.GameSession.CrewManager.IsSinglePlayer)
613 {
614 GameMain.GameSession.CrewManager.SetCharacterOrder(character, dismissOrder);
615 }
616#else
617 GameMain.Server?.SendOrderChatMessage(new OrderChatMessage(dismissOrder, character, character));
618 SetOrder(dismissOrder, speak: false);
619#endif

622 private bool IsAllowedToWait()
623 {
624 if (!character.IsOnPlayerTeam) { return false; }
625 if (HasOrders()) { return false; }
626 if (CurrentObjective is AIObjectiveCombat || CurrentObjective is AIObjectiveFindSafety) { return false; }
627 if (character.AnimController.InWater) { return false; }
628 if (character.IsClimbing) { return false; }

679 public float GetOrderPriority(AIObjective objective)
680 {
681 if (objective == ForcedOrder)
682 {
683 return HighestOrderPriority;
684 }
685 var currentOrder = CurrentOrders.FirstOrDefault(o => o.Objective == objective);
686 if (currentOrder.Objective == null)
687 {
688 return HighestOrderPriority;
689 }
690 else if (currentOrder.ManualPriority > 0)
691 {
692 if (objective.ForceHighestPriority)
693 {
694 return HighestOrderPriority;
695 }
696 if (objective.PrioritizeIfSubObjectivesActive && objective.SubObjectives.Any())
697 {
698 return HighestOrderPriority;
699 }
700 return MathHelper.Lerp(LowestOrderPriority, HighestOrderPriority - 1, MathUtils.InverseLerp(1, CharacterInfo.HighestManualOrderPriority, currentOrder.ManualPriority));
701 }
702#if DEBUG
703 DebugConsole.AddWarning("Error in order priority: shouldn't return 0!");
704#endif
705 return 0;
706 }
