41  protected override bool MatchesConditionSpecific(AbilityObject abilityObject)
42  {
43  if (abilityObject is AbilityAttackData attackData)
44  {
45  if (ignoreNonHarmfulAttacks && attackData.SourceAttack != null)
46  {
47  if (attackData.SourceAttack.Stun <= 0.0f && (attackData.SourceAttack.Afflictions?.All(a => a.Key.Prefab.IsBuff) ?? true))
48  {
49  return false;
50  }
51  }
52 
53  Item item = attackData?.SourceAttack?.SourceItem;
54 
55  if (!string.IsNullOrEmpty(itemIdentifier))
56  {
57  if (item?.Prefab.Identifier != itemIdentifier)
58  {
59  return false;
60  }
61  }
62 
63  if (tags.Any())
64  {
65  if (!tags.All(t => item?.HasTag(t) ?? false))
66  {
67  return false;
68  }
69  }
70 
71  if (weapontype != WeaponType.Any)
72  {
73  foreach (WeaponType wt in WeaponTypeValues)
74  {
75  if (wt == WeaponType.Any || !weapontype.HasFlag(wt)) { continue; }
76  switch (wt)
77  {
78  // it is possible that an item that has both a melee and a projectile component will return true
79  // even when not used as a melee/ranged weapon respectively
80  // attackdata should contain data regarding whether the attack is melee or not
81  case WeaponType.Melee:
82  if (item?.GetComponent<MeleeWeapon>() != null) { return true; }
83  break;
84  case WeaponType.Ranged:
85  if (item?.GetComponent<Projectile>() != null) { return true; }
86  break;
87  case WeaponType.HandheldRanged:
88  {
89  var projectile = item?.GetComponent<Projectile>();
90  if (projectile?.Launcher?.GetComponent<Holdable>() != null) { return true; }
91  }
92  break;
93  case WeaponType.Turret:
94  {
95  var projectile = item?.GetComponent<Projectile>();
96  if (projectile?.Launcher?.GetComponent<Turret>() != null) { return true; }
97  }
98  break;
99  case WeaponType.NoWeapon:
100  if (item == null) { return true; }
101  break;
102  }
103  }
104  return false;
105  }
106 
107  return true;
108  }
109  else
110  {
111  LogAbilityConditionError(abilityObject, typeof(AbilityAttackData));
112  return false;
113  }
114  }
