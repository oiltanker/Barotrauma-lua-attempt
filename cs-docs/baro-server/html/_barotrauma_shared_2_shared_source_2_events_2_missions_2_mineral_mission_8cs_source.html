204  private bool HasBeenCollected(Item item)
205  {
206  if (item == null) { return false; }
207  if (item.Removed) { return false; }
208  var owner = item.GetRootInventoryOwner();
209  if (owner.Submarine != null && owner.Submarine.Info.Type == SubmarineType.Player)
210  {
211  return true;
212  }
213  else if (owner is Character c)
214  {
215  return c.Info != null && GameMain.GameSession.CrewManager.CharacterInfos.Contains(c.Info);
216  }
217  return false;

223  private void CalculateMissionClusterPositions()
224  {
225  missionClusterPositions.Clear();

227  {

231  foreach (var i in kvp.Value.Where(i => i != null && !i.Removed))
232  {
233  pos += i.WorldPosition;
234  itemCount++;
235  }
236  pos /= itemCount;
