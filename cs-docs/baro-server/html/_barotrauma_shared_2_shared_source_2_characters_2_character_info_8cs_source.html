478  private void GetName(Rand.RandSync randSync, out string name)
479  {
480  var nameElement = CharacterConfigElement.GetChildElement("names") ?? CharacterConfigElement.GetChildElement("name");
481  ContentPath namesXmlFile = nameElement?.GetAttributeContentPath("path") ?? ContentPath.Empty;
482  XElement namesXml = null;
483  if (!namesXmlFile.IsNullOrEmpty()) //names.xml is defined
484  {
485  XDocument doc = XMLExtensions.TryLoadXml(namesXmlFile);
486  namesXml = doc.Root;
487  }
488  else //the legacy firstnames.txt/lastnames.txt shit is defined
489  {
490  namesXml = new XElement("names", new XAttribute("format", "[firstname] [lastname]"));
491  var firstNamesPath = ReplaceVars(nameElement.GetAttributeContentPath("firstname")?.Value ?? "");
492  var lastNamesPath = ReplaceVars(nameElement.GetAttributeContentPath("lastname")?.Value ?? "");
493  if (File.Exists(firstNamesPath) && File.Exists(lastNamesPath))
494  {
495  var firstNames = File.ReadAllLines(firstNamesPath);
496  var lastNames = File.ReadAllLines(lastNamesPath);
497  namesXml.Add(firstNames.Select(n => new XElement("firstname", new XAttribute("value", n))));
498  namesXml.Add(lastNames.Select(n => new XElement("lastname", new XAttribute("value", n))));
499  }
500  else //the files don't exist, just fall back to the vanilla names
501  {
502  XDocument doc = XMLExtensions.TryLoadXml("Content/Characters/Human/names.xml");
503  namesXml = doc.Root;
504  }
505  }
506  name = namesXml.GetAttributeString("format", "");
507  Dictionary<Identifier, List<string>> entries = new Dictionary<Identifier, List<string>>();
508  foreach (var subElement in namesXml.Elements())
509  {
510  Identifier elemName = subElement.NameAsIdentifier();
511  if (!entries.ContainsKey(elemName))
512  {
513  entries.Add(elemName, new List<string>());
514  }
515  ImmutableHashSet<Identifier> identifiers = subElement.GetAttributeIdentifierArray("tags", Array.Empty<Identifier>()).ToImmutableHashSet();
516  if (identifiers.IsSubsetOf(Head.Preset.TagSet))
517  {
518  entries[elemName].Add(subElement.GetAttributeString("value", ""));
519  }
520  }
521 
522  foreach (var k in entries.Keys)
523  {
524  name = name.Replace($"[{k}]", entries[k].GetRandom(randSync), StringComparison.OrdinalIgnoreCase);
525  }
526  }

1406  public static List<Order> LoadOrders(XElement ordersElement)
1407  {
1408  var orders = new List<Order>();
1409  if (ordersElement == null) { return orders; }
1410  // If an order is invalid, we discard the order and increase the priority of the following orders so
1411  // 1) the highest priority value will remain equal to CharacterInfo.HighestManualOrderPriority; and
1412  // 2) the order priorities will remain sequential.
1413  int priorityIncrease = 0;
1414  var linkedSubs = GetLinkedSubmarines();
1415  foreach (var orderElement in ordersElement.GetChildElements("order"))
1416  {
1417  Order order = null;
1418  string orderIdentifier = orderElement.GetAttributeString("id", "");
1419  var orderPrefab = OrderPrefab.Prefabs[orderIdentifier];
1420  if (orderPrefab == null)
1421  {
1422  DebugConsole.ThrowError($"Error loading a previously saved order - can't find an order prefab with the identifier \"{orderIdentifier}\"");
1423  priorityIncrease++;
1424  continue;
1425  }
1426  var targetType = (Order.OrderTargetType)orderElement.GetAttributeInt("targettype", 0);
1427  int orderGiverInfoId = orderElement.GetAttributeInt("ordergiverinfoid", -1);
1428  var orderGiver = orderGiverInfoId >= 0 ? Character.CharacterList.FirstOrDefault(c => c.Info?.ID == orderGiverInfoId) : null;
1429  Entity targetEntity = null;
1430  switch (targetType)
1431  {
1432  case Order.OrderTargetType.Entity:
1433  ushort targetId = (ushort)orderElement.GetAttributeUInt("targetid", Entity.NullEntityID);
1434  if (!GetTargetEntity(targetId, out targetEntity)) { continue; }
1435  var targetComponent = orderPrefab.GetTargetItemComponent(targetEntity as Item);
1436  order = new Order(orderPrefab, targetEntity, targetComponent, orderGiver: orderGiver);
1437  break;
1438  case Order.OrderTargetType.Position:
1439  var orderTargetElement = orderElement.GetChildElement("ordertarget");
1440  var position = orderTargetElement.GetAttributeVector2("position", Vector2.Zero);
1441  ushort hullId = (ushort)orderTargetElement.GetAttributeUInt("hullid", 0);
1442  if (!GetTargetEntity(hullId, out targetEntity)) { continue; }
1443  if (!(targetEntity is Hull targetPositionHull))
1444  {
1445  DebugConsole.ThrowError($"Error loading a previously saved order ({orderIdentifier}) - entity with the ID {hullId} is of type {targetEntity?.GetType()} instead of Hull");
1446  priorityIncrease++;
1447  continue;
1448  }
1449  var orderTarget = new OrderTarget(targetPositionHull.WorldPosition + position, targetPositionHull);
1450  order = new Order(orderPrefab, orderTarget, orderGiver: orderGiver);
1451  break;
1452  case Order.OrderTargetType.WallSection:
1453  ushort structureId = (ushort)orderElement.GetAttributeInt("structureid", Entity.NullEntityID);
1454  if (!GetTargetEntity(structureId, out targetEntity)) { continue; }
1455  int wallSectionIndex = orderElement.GetAttributeInt("wallsectionindex", 0);
1456  if (!(targetEntity is Structure targetStructure))
1457  {
1458  DebugConsole.ThrowError($"Error loading a previously saved order ({orderIdentifier}) - entity with the ID {structureId} is of type {targetEntity?.GetType()} instead of Structure");
1459  priorityIncrease++;
1460  continue;
1461  }
1462  order = new Order(orderPrefab, targetStructure, wallSectionIndex, orderGiver: orderGiver);
1463  break;
1464  }
1465  Identifier orderOption = orderElement.GetAttributeIdentifier("option", "");
1466  int manualPriority = orderElement.GetAttributeInt("priority", 0) + priorityIncrease;
1467  var orderInfo = order.WithOption(orderOption).WithManualPriority(manualPriority);
1468  orders.Add(orderInfo);

1470  bool GetTargetEntity(ushort targetId, out Entity targetEntity)
1471  {
1472  targetEntity = null;
1473  if (targetId == Entity.NullEntityID) { return true; }
1474  Submarine parentSub = null;
1475  if (orderElement.GetAttributeBool("onmainsub", false))
1476  {
1477  parentSub = Submarine.MainSub;
1478  }
1479  else
1480  {
1481  int linkedSubIndex = orderElement.GetAttributeInt("linkedsubindex", -1);
1482  if (linkedSubIndex >= 0 && linkedSubIndex < linkedSubs.Count &&
1483  linkedSubs[linkedSubIndex] is LinkedSubmarine linkedSub && linkedSub.LoadSub)
1484  {
1485  parentSub = linkedSub.Sub;
1486  }
1487  }
1488  if (parentSub != null)
1489  {
1490  targetId = GetOffsetId(parentSub, targetId);
1491  targetEntity = Entity.FindEntityByID(targetId);
1492  }
1493  else
1494  {
1495  if (!orderPrefab.CanBeGeneralized)
1496  {
1497  DebugConsole.ThrowError($"Error loading a previously saved order ({orderIdentifier}). Can't find the parent sub of the target entity. The order requires a target so it can't be loaded at all.");
1498  priorityIncrease++;
1499  return false;
1500  }
1501  else
1502  {
1503  DebugConsole.AddWarning($"Trying to load a previously saved order ({orderIdentifier}). Can't find the parent sub of the target entity. The order doesn't require a target so a more generic version of the order will be loaded instead.");
1504  }
1505  }
1506  return true;
1507  }
1508  }
1509  return orders;
