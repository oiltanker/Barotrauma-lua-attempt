70 public override void Update(float deltaTime, Camera cam)
71 {
72 MoveInputQueue();
73
74 if (inputContainer == null || inputContainer.Inventory.IsEmpty())
75 {
76 SetActive(false);
77 return;
78 }
79
80 hasPower = Voltage >= MinVoltage;
81 if (!hasPower) { return; }
82
83 var repairable = item.GetComponent<Repairable>();
84 if (repairable != null)
85 {
86 repairable.LastActiveTime = (float)Timing.TotalTime + 10.0f;
87 }
88
89 ApplyStatusEffects(ActionType.OnActive, deltaTime, null);
90
91 progressTimer += deltaTime * Math.Min(powerConsumption <= 0.0f ? 1 : Voltage, 1.0f);
92
93 float tinkeringStrength = 0f;
94 if (repairable.IsTinkering)
95 {
96 tinkeringStrength = repairable.TinkeringStrength;
97 }
98 // doesn't quite work properly, remaining time changes if tinkering stops
99 float deconstructionSpeedModifier = userDeconstructorSpeedMultiplier * (1f + tinkeringStrength * TinkeringSpeedIncrease);
100
101 if (DeconstructItemsSimultaneously)
102 {
103 float deconstructTime = 0.0f;
104 foreach (Item targetItem in inputContainer.Inventory.AllItems)
105 {
106 deconstructTime += targetItem.Prefab.DeconstructTime / (DeconstructionSpeed * deconstructionSpeedModifier);
107 }
108
109 progressState = Math.Min(progressTimer / deconstructTime, 1.0f);
110 if (progressTimer > deconstructTime)
111 {
112 List<Item> items = inputContainer.Inventory.AllItems.ToList();
113 foreach (Item targetItem in items)
114 {
115 if ((Entity.Spawner?.IsInRemoveQueue(targetItem) ?? false) || !inputContainer.Inventory.AllItems.Contains(targetItem)) { continue; }
116 var validDeconstructItems = targetItem.Prefab.DeconstructItems.Where(it =>
117 (it.RequiredDeconstructor.Length == 0 || it.RequiredDeconstructor.Any(r => item.HasTag(r) || item.Prefab.Identifier == r)) &&
118 (it.RequiredOtherItem.Length == 0 || it.RequiredOtherItem.Any(r => items.Any(it => it != targetItem && (it.HasTag(r) || it.Prefab.Identifier == r))))).ToList();
119
120 ProcessItem(targetItem, items, validDeconstructItems, allowRemove: validDeconstructItems.Any() || !targetItem.Prefab.DeconstructItems.Any());
121 }
122#if SERVER
123 item.CreateServerEvent(this);
124#endif
125 progressTimer = 0.0f;
126 progressState = 0.0f;
127
128 }
129 }
130 else
131 {
132 var targetItem = inputContainer.Inventory.LastOrDefault();
133 if (targetItem == null) { return; }
134
135 var validDeconstructItems = targetItem.Prefab.DeconstructItems.Where(it =>
136 it.RequiredDeconstructor.Length == 0 || it.RequiredDeconstructor.Any(r => item.HasTag(r) || item.Prefab.Identifier == r)).ToList();
137
138 float deconstructTime = validDeconstructItems.Any() ? targetItem.Prefab.DeconstructTime / (DeconstructionSpeed * deconstructionSpeedModifier) : 1.0f;
139
140 progressState = Math.Min(progressTimer / deconstructTime, 1.0f);
141 if (progressTimer > deconstructTime)
142 {
143 ProcessItem(targetItem, inputContainer.Inventory.AllItemsMod, validDeconstructItems, allowRemove: validDeconstructItems.Any() || !targetItem.Prefab.DeconstructItems.Any());
144
145#if SERVER
146 item.CreateServerEvent(this);
147#endif
148 progressTimer = 0.0f;
149 progressState = 0.0f;
150
151 }
152 }

155 private void ProcessItem(Item targetItem, IEnumerable<Item> inputItems, List<DeconstructItem> validDeconstructItems, bool allowRemove = true)
156 {
157 // In multiplayer, the server handles the deconstruction into new items
158 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
159
160 float amountMultiplier = 1f;
161
162 if (user != null && !user.Removed)
163 {
164 var abilityTargetItem = new AbilityDeconstructedItem(targetItem, user);
165 user.CheckTalents(AbilityEffectType.OnItemDeconstructed, abilityTargetItem);
166
167 foreach (Character character in Character.GetFriendlyCrew(user))
168 {
169 character.CheckTalents(AbilityEffectType.OnItemDeconstructedByAlly, abilityTargetItem);
170 }
171
172 var itemCreationMultiplier = new AbilityItemCreationMultiplier(targetItem.Prefab, amountMultiplier);
173 user.CheckTalents(AbilityEffectType.OnItemDeconstructedMaterial, itemCreationMultiplier);
174 amountMultiplier = (int)itemCreationMultiplier.Value;
175 }
176
177 if (targetItem.Prefab.RandomDeconstructionOutput)
178 {
179 int amount = targetItem.Prefab.RandomDeconstructionOutputAmount;
180 List<int> deconstructItemIndexes = new List<int>();
181 for (int i = 0; i < validDeconstructItems.Count; i++)
182 {
183 deconstructItemIndexes.Add(i);
184 }
185 List<float> commonness = validDeconstructItems.Select(i => i.Commonness).ToList();
186 List<DeconstructItem> products = new List<DeconstructItem>();
187
188 for (int i = 0; i < amount; i++)
189 {
190 if (deconstructItemIndexes.Count < 1) { break; }
191 var itemIndex = ToolBox.SelectWeightedRandom(deconstructItemIndexes, commonness, Rand.RandSync.Unsynced);
192 products.Add(validDeconstructItems[itemIndex]);
193 var removeIndex = deconstructItemIndexes.IndexOf(itemIndex);
194 deconstructItemIndexes.RemoveAt(removeIndex);
195 commonness.RemoveAt(removeIndex);
196 }
197
198 foreach (DeconstructItem deconstructProduct in products)
199 {
200 CreateDeconstructProduct(deconstructProduct, inputItems, amountMultiplier);
201 }
202 }
203 else
204 {
205 foreach (DeconstructItem deconstructProduct in validDeconstructItems)
206 {
207 CreateDeconstructProduct(deconstructProduct, inputItems, amountMultiplier);
208 }

211 void CreateDeconstructProduct(DeconstructItem deconstructProduct, IEnumerable<Item> inputItems, float amountMultiplier)
212 {
213 float percentageHealth = targetItem.Condition / targetItem.MaxCondition;
214
215 if (percentageHealth < deconstructProduct.MinCondition || percentageHealth > deconstructProduct.MaxCondition) { return; }
216
217 if (!(MapEntityPrefab.Find(null, deconstructProduct.ItemIdentifier) is ItemPrefab itemPrefab))
218 {
219 DebugConsole.ThrowError("Tried to deconstruct item \"" + targetItem.Name + "\" but couldn't find item prefab \"" + deconstructProduct.ItemIdentifier + "\"!");
220 return;
221 }
222
223 float condition = deconstructProduct.CopyCondition ?
224 percentageHealth * itemPrefab.Health * deconstructProduct.OutConditionMax :
225 itemPrefab.Health * Rand.Range(deconstructProduct.OutConditionMin, deconstructProduct.OutConditionMax);
226
227 if (DeconstructItemsSimultaneously && deconstructProduct.RequiredOtherItem.Length > 0)
228 {
229 foreach (Item otherItem in inputItems)
230 {
231 if (targetItem == otherItem) { continue; }
232 if (deconstructProduct.RequiredOtherItem.Any(r => otherItem.HasTag(r) || r == otherItem.Prefab.Identifier))
233 {
234 user?.CheckTalents(AbilityEffectType.OnGeneticMaterialCombinedOrRefined);
235 foreach (Character character in Character.GetFriendlyCrew(user))
236 {
237 character.CheckTalents(AbilityEffectType.OnCrewGeneticMaterialCombinedOrRefined);
238 }
239
240 var geneticMaterial1 = targetItem.GetComponent<GeneticMaterial>();
241 var geneticMaterial2 = otherItem.GetComponent<GeneticMaterial>();
242 if (geneticMaterial1 != null && geneticMaterial2 != null)
243 {
244 if (geneticMaterial1.Combine(geneticMaterial2, user))
245 {
246 inputContainer.Inventory.RemoveItem(otherItem);
247 OutputContainer.Inventory.RemoveItem(otherItem);
248 Entity.Spawner.AddItemToRemoveQueue(otherItem);
249 }
250 allowRemove = false;
251 return;
252 }
253 inputContainer.Inventory.RemoveItem(otherItem);
254 OutputContainer.Inventory.RemoveItem(otherItem);
255 Entity.Spawner.AddItemToRemoveQueue(otherItem);
256 }
257 }
258 }
259
260 if (user != null && !user.Removed)
261 {
262 // used to spawn items directly into the deconstructor
263 var itemDeconstructedInventory = new AbilityItemDeconstructedInventory(targetItem.Prefab, item);
264 user.CheckTalents(AbilityEffectType.OnItemDeconstructedInventory, itemDeconstructedInventory);
265 }
266
267 int amount = (int)amountMultiplier;
268 for (int i = 0; i < amount; i++)
269 {
270 Entity.Spawner.AddItemToSpawnQueue(itemPrefab, outputContainer.Inventory, condition, onSpawned: (Item spawnedItem) =>
271 {
272 spawnedItem.StolenDuringRound = targetItem.StolenDuringRound;
273 spawnedItem.AllowStealing = targetItem.AllowStealing;
274 for (int i = 0; i < outputContainer.Capacity; i++)
275 {
276 var containedItem = outputContainer.Inventory.GetItemAt(i);
277 bool combined = false;
278 if (containedItem?.OwnInventory != null)
279 {
280 foreach (Item subItem in containedItem.ContainedItems.ToList())
281 {
282 if (subItem.Combine(spawnedItem, null))
283 {
284 combined = true;
285 break;
286 }
287 }
288 }
289 if (!combined)
290 {
291 if (containedItem?.Combine(spawnedItem, null) ?? false)
292 {
293 break;
294 }
295 }
296 }
297 PutItemsToLinkedContainer();
298 });
299 }
300 }
301
302 GameAnalyticsManager.AddDesignEvent("ItemDeconstructed:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "none") + ":" + targetItem.Prefab.Identifier);
303
304 if (targetItem.AllowDeconstruct && allowRemove)
305 {
306 //drop all items that are inside the deconstructed item
307 foreach (ItemContainer ic in targetItem.GetComponents<ItemContainer>())
308 {
309 if (ic?.Inventory == null || ic.RemoveContainedItemsOnDeconstruct) { continue; }
310 foreach (Item outputItem in ic.Inventory.AllItemsMod)
311 {
312 tryPutInOutputSlots(outputItem);
313 }
314 }
315 inputContainer.Inventory.RemoveItem(targetItem);
316 Entity.Spawner.AddItemToRemoveQueue(targetItem);
317 MoveInputQueue();
318 PutItemsToLinkedContainer();
319 }
320 else
321 {
322 if (Entity.Spawner?.IsInRemoveQueue(targetItem) ?? false)
323 {
324 targetItem.Drop(dropper: null);
325 }
326 else
327 {
328 tryPutInOutputSlots(targetItem);
329 }
330 }

332 void tryPutInOutputSlots(Item item)
333 {
334 for (int i = 0; i < outputContainer.Capacity; i++)
335 {
336 var containedItem = outputContainer.Inventory.GetItemAt(i);
337 if (containedItem?.OwnInventory != null && containedItem.GetComponent<GeneticMaterial>() == null && containedItem.OwnInventory.TryPutItem(item, user: null))
338 {
339 return;
340 }
341 }
342 if (!outputContainer.Inventory.TryPutItem(item, user: null))
343 {
344 item.Drop(dropper: null);
345 }

349 private void PutItemsToLinkedContainer()
350 {
351 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
352 if (outputContainer.Inventory.IsEmpty()) { return; }
353
354 foreach (MapEntity linkedTo in item.linkedTo)
355 {
356 if (linkedTo is Item linkedItem)
357 {
358 var fabricator = linkedItem.GetComponent<Fabricator>();
359 if (fabricator != null) { continue; }
360 var itemContainer = linkedItem.GetComponent<ItemContainer>();
361 if (itemContainer == null) { continue; }
362 outputContainer.Inventory.AllItemsMod.ForEach(containedItem => itemContainer.Inventory.TryPutItem(containedItem, user: null, createNetworkEvent: true));
363 }
364 }

370 private void MoveInputQueue()
371 {
372 for (int i = inputContainer.Inventory.Capacity - 2; i >= 0; i--)
373 {
374 while (inputContainer.Inventory.GetItemAt(i) is Item item1 && inputContainer.Inventory.CanBePutInSlot(item1, i + 1))
375 {
376 if (!inputContainer.Inventory.TryPutItem(item1, i + 1, allowSwapping: false, allowCombine: false, user: null, createNetworkEvent: true))
377 {
378 break;
379 }
380 }

384 private IEnumerable<(Item item, DeconstructItem output)> GetAvailableOutputs(bool checkRequiredOtherItems = true)
385 {
386 var items = inputContainer.Inventory.AllItems;
387 foreach (Item inputItem in items)
388 {
389 if (!inputItem.AllowDeconstruct) { continue; }
390 foreach (var deconstructItem in inputItem.Prefab.DeconstructItems)
391 {
392 if (deconstructItem.RequiredDeconstructor.Length > 0)
393 {
394 if (!deconstructItem.RequiredDeconstructor.Any(r => item.HasTag(r) || item.Prefab.Identifier == r)) { continue; }
395 }
396 if (deconstructItem.RequiredOtherItem.Length > 0 && checkRequiredOtherItems)
397 {
398 if (!deconstructItem.RequiredOtherItem.Any(r => items.Any(it => it.HasTag(r) || it.Prefab.Identifier == r))) { continue; }
399 bool validOtherItemFound = false;
400 foreach (Item otherInputItem in items)
401 {
402 if (otherInputItem == inputItem) { continue; }
403 if (!deconstructItem.RequiredOtherItem.Any(r => otherInputItem.HasTag(r) || otherInputItem.Prefab.Identifier == r)) { continue; }
404
405 var geneticMaterial1 = inputItem.GetComponent<GeneticMaterial>();
406 var geneticMaterial2 = otherInputItem.GetComponent<GeneticMaterial>();
407 if (geneticMaterial1 != null && geneticMaterial2 != null)
408 {
409 if (!geneticMaterial1.CanBeCombinedWith(geneticMaterial2)) { continue; }
410 }
411 validOtherItemFound = true;
412 }
413 if (!validOtherItemFound) { continue; }
414 }
415 yield return (inputItem, deconstructItem);
416 }
417 }

421 {
422 PutItemsToLinkedContainer();
423
424 this.user = user;
425
426 if (inputContainer.Inventory.IsEmpty()) { active = false; }
427
428 IsActive = active;
429 //currPowerConsumption = IsActive ? powerConsumption : 0.0f;
430 userDeconstructorSpeedMultiplier = user != null ? 1f + user.GetStatValue(StatTypes.DeconstructorSpeedMultiplier) : 1f;
431
432#if SERVER
433 if (user != null)
434 {
435 GameServer.Log(GameServer.CharacterLogName(user) + (IsActive ? " activated " : " deactivated ") + item.Name, ServerLog.MessageType.ItemInteraction);
436 }
437#endif
438 if (!IsActive)
439 {
440 progressTimer = 0.0f;
441 progressState = 0.0f;
442 }
443
444 inputContainer.Inventory.Locked = IsActive;
