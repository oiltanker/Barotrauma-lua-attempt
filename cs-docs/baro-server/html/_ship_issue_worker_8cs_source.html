96 // if there are other orders of the same type already being attended to, such as fixing leaks
97 // reduce the relative importance of this issue
98 CurrentRedundancy = 1f;
99 foreach (ShipIssueWorker shipIssueWorker in shipCommandManager.ShipIssueWorkers)
