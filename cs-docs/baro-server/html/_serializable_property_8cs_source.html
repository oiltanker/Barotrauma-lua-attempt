202  public bool TrySetValue(object parentObject, string value)
203  {
204  if (value == null) { return false; }
205 
206  if (!supportedTypes.TryGetValue(PropertyType, out string typeName))
207  {
208  if (PropertyType.IsEnum)
209  {
210  object enumVal;
211  try
212  {
213  enumVal = Enum.Parse(PropertyInfo.PropertyType, value, true);
214  }
215  catch (Exception e)
216  {
217  DebugConsole.ThrowError("Failed to set the value of the property \"" + Name + "\" of \"" + parentObject + "\" to " + value + " (not a valid " + PropertyInfo.PropertyType + ")", e);
218  return false;
219  }
220  try
221  {
222  PropertyInfo.SetValue(parentObject, enumVal);
223  }
224  catch (Exception e)
225  {
226  DebugConsole.ThrowError("Failed to set the value of the property \"" + Name + "\" of \"" + parentObject.ToString() + "\" to " + value.ToString(), e);
227  return false;
228  }
229  }
230  else
231  {
232  DebugConsole.ThrowError($"Failed to set the value of the property \"{Name}\" of \"{parentObject}\" to {value} (Type \"{PropertyType.Name}\" not supported)");
233 
234  return false;
235  }
236  }
237 
238  try
239  {
240  switch (typeName)
241  {
242  case "bool":
243  bool boolValue = value == "true" || value == "True";
244  if (TrySetBoolValueWithoutReflection(parentObject, boolValue)) { return true; }
245  PropertyInfo.SetValue(parentObject, boolValue, null);
246  break;
247  case "int":
248  if (int.TryParse(value, out int intVal))
249  {
250  if (TrySetFloatValueWithoutReflection(parentObject, intVal)) { return true; }
251  PropertyInfo.SetValue(parentObject, intVal, null);
252  }
253  else
254  {
255  return false;
256  }
257  break;
258  case "float":
259  if (float.TryParse(value, NumberStyles.Float, CultureInfo.InvariantCulture, out float floatVal))
260  {
261  if (TrySetFloatValueWithoutReflection(parentObject, floatVal)) { return true; }
262  PropertyInfo.SetValue(parentObject, floatVal, null);
263  }
264  else
265  {
266  return false;
267  }
268  break;
269  case "string":
270  PropertyInfo.SetValue(parentObject, value, null);
271  break;
272  case "point":
273  PropertyInfo.SetValue(parentObject, XMLExtensions.ParsePoint(value));
274  break;
275  case "vector2":
276  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseVector2(value));
277  break;
278  case "vector3":
279  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseVector3(value));
280  break;
281  case "vector4":
282  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseVector4(value));
283  break;
284  case "color":
285  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseColor(value));
286  break;
287  case "rectangle":
288  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseRect(value, true));
289  break;
290  case "identifier":

314  public bool TrySetValue(object parentObject, object value)
315  {
316  if (value == null || parentObject == null || PropertyInfo == null) return false;
317 
318  try
319  {
320  if (!supportedTypes.TryGetValue(PropertyType, out string typeName))
321  {
322  if (PropertyType.IsEnum)
323  {
324  object enumVal;
325  try
326  {
327  enumVal = Enum.Parse(PropertyInfo.PropertyType, value.ToString(), true);
328  }
329  catch (Exception e)
330  {
331  DebugConsole.ThrowError(
332  $"Failed to set the value of the property \"{Name}\" of \"{parentObject}\" to {value} (not a valid {PropertyInfo.PropertyType})", e);
333  return false;
334  }
335  PropertyInfo.SetValue(parentObject, enumVal);
336  return true;
337  }
338  else
339  {
340  DebugConsole.ThrowError($"Failed to set the value of the property \"{Name}\" of \"{parentObject}\" to {value} (Type \"{PropertyType.Name}\" not supported)");
341 
342  return false;
343  }
344  }
345 
346  try
347  {
348  if (value.GetType() == typeof(string))
349  {

352  case "string":

354  return true;

366  return true;
367  case "color":
368  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseColor((string)value));
369  return true;
370  case "rectangle":
371  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseRect((string)value, false));
372  return true;
373  case "identifier":
374  PropertyInfo.SetValue(parentObject, new Identifier((string)value));
375  return true;
376  case "localizedstring":
377  PropertyInfo.SetValue(parentObject, new RawLString((string)value));
378  return true;
379  case "stringarray":
380  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseStringArray((string)value));
381  return true;
382  case "identifierarray":
383  PropertyInfo.SetValue(parentObject, XMLExtensions.ParseStringArray((string)value).ToIdentifiers().ToArray());
384  return true;
385  default:
386  DebugConsole.ThrowError("Failed to set the value of the property \"" + Name + "\" of \"" + parentObject.ToString() + "\" to " + value.ToString());
387  DebugConsole.ThrowError("(Cannot convert a string to a " + PropertyType.ToString() + ")");
388  return false;
389  }
390  }
391  else if (PropertyType != value.GetType())
392  {
393  DebugConsole.ThrowError("Failed to set the value of the property \"" + Name + "\" of \"" + parentObject.ToString() + "\" to " + value.ToString());
394  DebugConsole.ThrowError("(Non-matching type, should be " + PropertyType + " instead of " + value.GetType() + ")");
395  return false;
396  }
397 
398  PropertyInfo.SetValue(parentObject, value, null);
399  }
400 
401  catch (Exception e)
402  {
403  DebugConsole.ThrowError("Failed to set the value of the property \"" + Name + "\" of \"" + parentObject.ToString() + "\" to " + value.ToString(), e);
404  return false;
405  }
406 
407  return true;
408  }
409  catch (Exception e)
410  {
411  DebugConsole.ThrowError("Error in SerializableProperty.TrySetValue", e);
412  return false;

873  public static void SerializeProperties(ISerializableEntity obj, XElement element, bool saveIfDefault = false, bool ignoreEditable = false)
874  {
875  var saveProperties = GetProperties<Serialize>(obj);
876  foreach (var property in saveProperties)
877  {
878  object value = property.GetValue(obj);
879  if (value == null) continue;
880 
881  if (!saveIfDefault)
882  {
883  //only save
884  // - if the attribute is saveable and it's different from the default value
885  // - or can be changed in-game or in the editor
886  bool save = false;
887  foreach (var attribute in property.Attributes.OfType<Serialize>())
888  {
889  if ((attribute.IsSaveable == IsPropertySaveable.Yes && !attribute.DefaultValue.Equals(value)) ||
890  (!ignoreEditable && property.Attributes.OfType<Editable>().Any()))
891  {
892  save = true;
893  break;
894  }
895  }
896 
897  if (!save) continue;
898  }
899 
900  string stringValue;
901  if (!supportedTypes.TryGetValue(value.GetType(), out string typeName))
902  {
903  if (property.PropertyType.IsEnum)
904  {
905  stringValue = value.ToString();
906  }
907  else
908  {
909  DebugConsole.ThrowError("Failed to serialize the property \"" + property.Name + "\" of \"" + obj + "\" (type " + property.PropertyType + " not supported)");
910  continue;
911  }
912  }
913  else
914  {
915  switch (typeName)
916  {
917  case "float":
918  //make sure the decimal point isn't converted to a comma or anything else
919  stringValue = ((float)value).ToString("G", CultureInfo.InvariantCulture);
920  break;
921  case "point":
922  stringValue = XMLExtensions.PointToString((Point)value);
923  break;
924  case "vector2":
925  stringValue = XMLExtensions.Vector2ToString((Vector2)value);
926  break;
927  case "vector3":
928  stringValue = XMLExtensions.Vector3ToString((Vector3)value);
929  break;
930  case "vector4":
931  stringValue = XMLExtensions.Vector4ToString((Vector4)value);
932  break;
933  case "color":
934  stringValue = XMLExtensions.ColorToString((Color)value);
935  break;
936  case "rectangle":
937  stringValue = XMLExtensions.RectToString((Rectangle)value);
938  break;
939  case "stringarray":
940  string[] stringArray = (string[])value;
941  stringValue = stringArray != null ? string.Join(';', stringArray) : "";
942  break;
943  case "identifierarray":
944  Identifier[] identifierArray = (Identifier[])value;
945  stringValue = identifierArray != null ? string.Join(';', identifierArray) : "";
946  break;
947  default:
948  stringValue = value.ToString();
949  break;
950  }
951  }
952  element.GetAttribute(property.Name)?.Remove();
953  element.SetAttributeValue(property.Name, stringValue);
954  }

1035  static void FixValue(SerializableProperty property, object parentObject, XAttribute attribute)
1036  {
1037  if (attribute.Value.Length > 0 && attribute.Value[0] == '*')
1038  {
1039  float.TryParse(attribute.Value.Substring(1), NumberStyles.Float, CultureInfo.InvariantCulture, out float multiplier);
1040 
1041  if (property.PropertyType == typeof(int))
1042  {
1043  property.TrySetValue(parentObject, (int)(((int)property.GetValue(parentObject)) * multiplier));
1044  }
1045  else if (property.PropertyType == typeof(float))
1046  {
1047  property.TrySetValue(parentObject, (float)property.GetValue(parentObject) * multiplier);
1048  }
1049  else if (property.PropertyType == typeof(Vector2))
1050  {
1051  property.TrySetValue(parentObject, (Vector2)property.GetValue(parentObject) * multiplier);
1052  }
1053  else if (property.PropertyType == typeof(Point))
1054  {
1055  property.TrySetValue(parentObject, ((Point)property.GetValue(parentObject)).Multiply(multiplier));
1056  }
1057  }
1058  else if (attribute.Value.Length > 0 && attribute.Value[0] == '+')
1059  {
1060  if (property.PropertyType == typeof(int))
1061  {
1062  float.TryParse(attribute.Value.Substring(1), NumberStyles.Float, CultureInfo.InvariantCulture, out float addition);
1063  property.TrySetValue(parentObject, (int)(((int)property.GetValue(parentObject)) + addition));
1064  }
1065  else if (property.PropertyType == typeof(float))
1066  {
1067  float.TryParse(attribute.Value.Substring(1), NumberStyles.Float, CultureInfo.InvariantCulture, out float addition);
1068  property.TrySetValue(parentObject, (float)property.GetValue(parentObject) + addition);
1069  }
1070  else if (property.PropertyType == typeof(Vector2))
1071  {
1072  var addition = XMLExtensions.ParseVector2(attribute.Value.Substring(1));
1073  property.TrySetValue(parentObject, (Vector2)property.GetValue(parentObject) + addition);
1074  }
1075  else if (property.PropertyType == typeof(Point))
1076  {
1077  var addition = XMLExtensions.ParsePoint(attribute.Value.Substring(1));
1078  property.TrySetValue(parentObject, ((Point)property.GetValue(parentObject)) + addition);
1079  }
1080  }
1081  else
1082  {
1083  property.TrySetValue(parentObject, attribute.Value);
1084  }
1085  }
1086 
1087  if (entity is Item item2)
1088  {
1089  var componentElement = subElement.FirstElement();
1090  if (componentElement == null) { continue; }
1091  ItemComponent itemComponent = item2.Components.First(c => c.Name == componentElement.Name.ToString());
1092  if (itemComponent == null) { continue; }
1093  foreach (XAttribute attribute in componentElement.Attributes())
1094  {
1095  var attributeName = attribute.NameAsIdentifier();
1096  if (itemComponent.SerializableProperties.TryGetValue(attributeName, out SerializableProperty property))
1097  {
1098  FixValue(property, itemComponent, attribute);
1099  }
1100  }
1101  foreach (var element in componentElement.Elements())
1102  {
1103  switch (element.Name.ToString().ToLowerInvariant())
1104  {
1105  case "requireditem":
1106  case "requireditems":
1107  itemComponent.requiredItems.Clear();
1108  itemComponent.DisabledRequiredItems.Clear();
1109 
1110  itemComponent.SetRequiredItems(element);
1111  break;
1112  }
1113  }
1114  }
