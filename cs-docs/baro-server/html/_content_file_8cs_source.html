72 public static Result<ContentFile, string> CreateFromXElement(ContentPackage contentPackage, XElement element)
73 {
74 static Result<ContentFile, string> fail(string error, string? stackTrace = null)
75 => Result<ContentFile, string>.Failure(error, stackTrace);
76
77 Identifier elemName = element.NameAsIdentifier();
78 var type = Types.FirstOrDefault(t => t.Names.Contains(elemName));
79 var filePath = element.GetAttributeContentPath("file", contentPackage);
80 if (type is null)
81 {
82 return fail($"Invalid content type \"{elemName}\"");
83 }
84 if (filePath is null)
85 {
86 return fail($"No content path defined for file of type \"{elemName}\"");
87 }
88 try
89 {
90 filePath = type.MutateContentPath(filePath);
91 if (!File.Exists(filePath.FullPath))
92 {
93 return fail($"Failed to load file \"{filePath}\" of type \"{elemName}\": file not found.");
94 }
95 var file = type.CreateInstance(contentPackage, filePath);
96 return file is null
97 ? throw new Exception($"Content type is not implemented correctly")
98 : Result<ContentFile, string>.Success(file);
99 }
100 catch (Exception e)
101 {
102 return fail($"Failed to load file \"{filePath}\" of type \"{elemName}\": {e.Message}", e.StackTrace.CleanupStackTrace());
103 }
