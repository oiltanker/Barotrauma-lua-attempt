39 private RunType GetRunType(ContentPackage cp, string path)
40 {
41 if (!Directory.Exists(path + "CSharp")) return RunType.None;
42
43 var isEnabled = ContentPackageManager.EnabledPackages.All.Contains(cp);
44 if (File.Exists(path + "CSharp/RunConfig.xml"))
45 {
46 var doc = XDocument.Load(File.Open(path + "CSharp/RunConfig.xml", FileMode.Open, FileAccess.Read));
47 var elems = doc.Root.Elements().ToArray();
48 var elem = elems.FirstOrDefault(e => e.Name.LocalName.Equals(LuaCsSetup.IsServer ? "Server" : (LuaCsSetup.IsClient ? "Client" : "None"), StringComparison.OrdinalIgnoreCase));
49
50 if (elem != null && Enum.TryParse(typeof(RunType), elem.Value, out object enumValue) && enumValue is RunType rtValue)
51 {
52 if (rtValue == RunType.Standard && isEnabled) LuaCsSetup.PrintCsMessage($"Standard run C# of {cp.Name}");
53 else if (rtValue == RunType.Forced) LuaCsSetup.PrintCsMessage($"Forced run C# of {cp.Name}");
54 return rtValue;
55 }
56 }
57
58 if (isEnabled)
59 {
60 LuaCsSetup.PrintCsMessage($"Assumed run C# of {cp.Name}");
61 return RunType.Standard;
62 }
63 else return RunType.None;

78 private enum SourceCategory { Shared, Server, Client };
79 private Regex rMaskPathValid = new Regex(@"^(((?!csharp)[^/])+/)+csharp(/(shared|client|server))?(/[^/]+)+\.cs$", RegexOptions.IgnoreCase | RegexOptions.Compiled);
80 private Regex rMaskPathCategory1 = new Regex(@"/(shared|client|server)(/[^/]+)+\.cs$", RegexOptions.IgnoreCase | RegexOptions.Compiled);
81 private Regex rMaskPathCategory2 = new Regex(@"^/(shared|client|server)", RegexOptions.IgnoreCase | RegexOptions.Compiled);
82 private void RunFolder(string folder)
83 {
84 var offendingSources = new List<string>();
85 foreach (var str in DirSearch(folder))
86 {
87 var s = str.Replace("\\", "/");
88
89 if (s.EndsWith(".cs") && LuaCsFile.IsPathAllowedCsException(s))
90 {
91 if (rMaskPathValid.IsMatch(s)) // valid path
92 {
93 var sourceCategory = SourceCategory.Shared;
94 {
95 var match = rMaskPathCategory1.Match(s);
96 if (match.Success) match = rMaskPathCategory2.Match(match.Value);
97 if (match.Success)
98 {
99 if (match.Value.EndsWith("shared", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Shared;
100 else if (match.Value.EndsWith("server", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Server;
101 else if (match.Value.EndsWith("client", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Client;
102 }
103 }
104
105 var belongsInAssembly = false;
106 {
107 if (sourceCategory == SourceCategory.Shared) belongsInAssembly = true;
108 else if (sourceCategory == SourceCategory.Server && LuaCsSetup.IsServer) belongsInAssembly = true;
109 else if (sourceCategory == SourceCategory.Client && LuaCsSetup.IsClient) belongsInAssembly = true;
110 }
111
112 if (belongsInAssembly)
113 {
114 if (sources.ContainsKey(folder)) sources[folder].Add(s);
115 else sources.Add(folder, new List<string> { s });
116 }
117 }
118 else offendingSources.Add(s);
119 }
120 }
121 if (offendingSources.Count > 0)
122 {
123 LuaCsSetup.PrintCsError($"All C# sources must belong to <mod_folder>/CSharp/*\n Offending sources:{offendingSources.Select(s => $"\n {s}").Aggregate((s1, s2) => $"{s1}{s2}")}");
124 }
125 }

127 private IEnumerable<SyntaxTree> ParseSources() {
128 var syntaxTrees = new List<SyntaxTree>();
129
130 if (sources.Count <= 0) throw new Exception("No Cs sources detected");
131 syntaxTrees.Add(AssemblyInfoSyntaxTree(NET_SCRIPT_ASSEMBLY));
132 foreach ((var folder, var src) in sources)
133 {
134 try
135 {
136 foreach (var file in src)
137 {
138 var tree = SyntaxFactory.ParseSyntaxTree(File.ReadAllText(file), ParseOptions, file);
139 var error = CsScriptFilter.FilterSyntaxTree(tree as CSharpSyntaxTree); // Check file content for prohibited stuff
140 if (error != null) throw new Exception(error);
141
142 syntaxTrees.Add(tree);
143 }
144 }
145 catch (Exception ex)
146 {
147 LuaCsSetup.PrintCsError("Error loading '" + folder + "':\n" + ex.Message + "\n" + ex.StackTrace);
148 }
149 }
150
151 return syntaxTrees;
152 }

154 public List<Type> Compile()
155 {
156 IEnumerable<SyntaxTree> syntaxTrees = ParseSources();
157
158 var options = new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary)
159 .WithMetadataImportOptions(MetadataImportOptions.All)
160 .WithOptimizationLevel(OptimizationLevel.Release)
161 .WithAllowUnsafe(false);
162 var compilation = CSharpCompilation.Create(NET_SCRIPT_ASSEMBLY,syntaxTrees, defaultReferences, options);
163
164 using (var mem = new MemoryStream())
165 {
166 var result = compilation.Emit(mem);
167 if (!result.Success)
168 {
169 IEnumerable<Diagnostic> failures = result.Diagnostics.Where(d => d.IsWarningAsError || d.Severity == DiagnosticSeverity.Error);
170
171 string errStr = "NET MODS NOT LOADED | Mod compilation errors:";
172 foreach (Diagnostic diagnostic in failures)
173 errStr += $"\n{diagnostic}";
174 LuaCsSetup.PrintCsError(errStr);
175 }
176 else
177 {
178 mem.Seek(0, SeekOrigin.Begin);
179 var errStr = CsScriptFilter.FilterMetadata(new PEReader(mem).GetMetadataReader());
180 if (errStr == null)
181 {
182 mem.Seek(0, SeekOrigin.Begin);
183 Assembly = LoadFromStream(mem);
184 }
185 else LuaCsSetup.PrintCsError(errStr);
186 }
187 }
188
189 if (Assembly != null)
190 return Assembly.GetTypes().Where(t => t.IsSubclassOf(typeof(ACsMod))).ToList();
191 else
192 throw new Exception("Unable to create net mods assembly.");
193 }
