126 public override void Update(float deltaTime, Camera cam)
127 {
128 this.cam = cam;
129 UserInCorrectPosition = false;
130
131 if (IsToggle)
132 {
133 item.SendSignal(State ? "1" : "0", "signal_out");
134 item.SendSignal(State ? "1" : "0", "trigger_out");
135 }
136
137 if (user == null
138 || user.Removed
139 || user.SelectedConstruction != item
140 || item.ParentInventory != null
141 || !user.CanInteractWith(item)
142 || (UsableIn == UseEnvironment.Water && !user.AnimController.InWater)
143 || (UsableIn == UseEnvironment.Air && user.AnimController.InWater))
144 {
145 if (user != null)
146 {
147 CancelUsing(user);
148 user = null;
149 }
150 if (!IsToggle) { IsActive = false; }
151 return;
152 }
153
154 user.AnimController.Anim = AnimController.Animation.UsingConstruction;
155
156 if (userPos != Vector2.Zero)
157 {
158 Vector2 diff = (item.WorldPosition + userPos) - user.WorldPosition;
159
160 if (user.AnimController.InWater)
161 {
162 if (diff.LengthSquared() > 30.0f * 30.0f)
163 {
164 user.AnimController.TargetMovement = Vector2.Clamp(diff * 0.01f, -Vector2.One, Vector2.One);
165 user.AnimController.TargetDir = diff.X > 0.0f ? Direction.Right : Direction.Left;
166 }
167 else
168 {
169 user.AnimController.TargetMovement = Vector2.Zero;
170 UserInCorrectPosition = true;
171 }
172 }
173 else
174 {
175 diff.Y = 0.0f;
176 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient && user != Character.Controlled)
177 {
178 if (Math.Abs(diff.X) > 20.0f)
179 {
180 //wait for the character to walk to the correct position
181 return;
182 }
183 else if (Math.Abs(diff.X) > 0.1f)
184 {
185 //aim to keep the collider at the correct position once close enough
186 user.AnimController.Collider.LinearVelocity = new Vector2(
187 diff.X * 0.1f,
188 user.AnimController.Collider.LinearVelocity.Y);
189 }
190 }
191 else
192 {
193 if (Math.Abs(diff.X) > 10.0f)
194 {
195 user.AnimController.TargetMovement = Vector2.Normalize(diff);
196 user.AnimController.TargetDir = diff.X > 0.0f ? Direction.Right : Direction.Left;
197 return;
198 }
199 }
200 user.AnimController.TargetMovement = Vector2.Zero;
201 UserInCorrectPosition = true;
202 }
203 }
204
205 ApplyStatusEffects(ActionType.OnActive, deltaTime, user);
206
207 if (limbPositions.Count == 0) { return; }
208
209 user.AnimController.Anim = AnimController.Animation.UsingConstruction;
210
211 user.AnimController.ResetPullJoints();
212
213 if (dir != 0) { user.AnimController.TargetDir = dir; }
214
215 foreach (LimbPos lb in limbPositions)
216 {
217 Limb limb = user.AnimController.GetLimb(lb.LimbType);
218 if (limb == null || !limb.body.Enabled) { continue; }
219
220 if (lb.AllowUsingLimb)
221 {
222 switch (lb.LimbType)
223 {
224 case LimbType.RightHand:
225 case LimbType.RightForearm:
226 case LimbType.RightArm:
227 if (user.Inventory.GetItemInLimbSlot(InvSlotType.RightHand) != null) { continue; }
228 break;
229 case LimbType.LeftHand:
230 case LimbType.LeftForearm:
231 case LimbType.LeftArm:
232 if (user.Inventory.GetItemInLimbSlot(InvSlotType.LeftHand) != null) { continue; }
233 break;
234 }
235 }
236
237 limb.Disabled = true;
238
239 Vector2 worldPosition = new Vector2(item.WorldRect.X, item.WorldRect.Y) + lb.Position * item.Scale;
240 Vector2 diff = worldPosition - limb.WorldPosition;
241
242 limb.PullJointEnabled = true;
243 limb.PullJointWorldAnchorB = limb.SimPosition + ConvertUnits.ToSimUnits(diff);
244 }
245 }

249 public override bool Use(float deltaTime, Character activator = null)
250 {
251 if (activator != user)
252 {
253 return false;
254 }
255
256 if (user == null || user.Removed ||
257 user.SelectedConstruction != item || !user.CanInteractWith(item))
258 {
259 user = null;
260 return false;
261 }
262
263 if (IsToggle && (activator == null || lastUsed < Timing.TotalTime - 0.1))
264 {
265 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
266 {
267 State = !State;
268#if SERVER
269 item.CreateServerEvent(this);
270#endif
271 }
272 }
273 else
274 {
275 item.SendSignal(new Signal("1", sender: user), "trigger_out");
276 }
277
278 lastUsed = Timing.TotalTime;
279
280 ApplyStatusEffects(ActionType.OnUse, 1.0f, activator);
281
282 return true;
283 }

285 public override bool SecondaryUse(float deltaTime, Character character = null)
286 {
287 if (this.user != character)
288 {
289 return false;
290 }
291
292 if (this.user == null || character.Removed ||
293 this.user.SelectedConstruction != item || !character.CanInteractWith(item))
294 {
295 this.user = null;
296 return false;
297 }
298 if (character == null) return false;
299
300 focusTarget = GetFocusTarget();
301 if (focusTarget == null)
302 {
303 Vector2 centerPos = new Vector2(item.WorldRect.Center.X, item.WorldRect.Center.Y);
304
305 Vector2 offset = character.CursorWorldPosition - centerPos;
306 offset.Y = -offset.Y;
307
308 targetRotation = MathUtils.WrapAngleTwoPi(MathUtils.VectorToAngle(offset));
309 return false;
310 }
311
312 character.ViewTarget = focusTarget;
313#if CLIENT
314 if (character == Character.Controlled && cam != null)
315 {
316 Lights.LightManager.ViewTarget = focusTarget;
317 cam.TargetPos = focusTarget.WorldPosition;
318
319 cam.OffsetAmount = MathHelper.Lerp(cam.OffsetAmount, (focusTarget as Item).Prefab.OffsetOnSelected * focusTarget.OffsetOnSelectedMultiplier, deltaTime * 10.0f);
320 HideHUDs(true);
321 }
322#endif
323
324 if (!character.IsRemotePlayer || character.ViewTarget == focusTarget)
325 {
326 Vector2 centerPos = new Vector2(focusTarget.WorldRect.Center.X, focusTarget.WorldRect.Center.Y);
327
328 Turret turret = focusTarget.GetComponent<Turret>();
329 if (turret != null)
330 {
331 centerPos = new Vector2(focusTarget.WorldRect.X + turret.TransformedBarrelPos.X, focusTarget.WorldRect.Y - turret.TransformedBarrelPos.Y);
332 }
333
334 Vector2 offset = character.CursorWorldPosition - centerPos;
335 offset.Y = -offset.Y;
336
337 targetRotation = MathUtils.WrapAngleTwoPi(MathUtils.VectorToAngle(offset));
338 }
339 return true;
340 }

342 public Item GetFocusTarget()
343 {
344 var positionOut = item.Connections?.Find(c => c.Name == "position_out");
345 if (positionOut == null) { return null; }
346
347 item.SendSignal(new Signal(MathHelper.ToDegrees(targetRotation).ToString("G", CultureInfo.InvariantCulture), sender: user), positionOut);
348
349 for (int i = item.LastSentSignalRecipients.Count - 1; i >= 0; i--)
350 {
351 if (item.LastSentSignalRecipients[i].Item.Condition <= 0.0f || item.LastSentSignalRecipients[i].IsPower) { continue; }
352 if (item.LastSentSignalRecipients[i].Item.Prefab.FocusOnSelected)
353 {
354 return item.LastSentSignalRecipients[i].Item;
355 }
356 }
357
358 foreach (var recipientPanel in item.GetConnectedComponentsRecursive<ConnectionPanel>(positionOut, allowTraversingBackwards: false))
359 {
360 if (recipientPanel.Item.Condition <= 0.0f) { continue; }
361 if (recipientPanel.Item.Prefab.FocusOnSelected)
362 {
363 return recipientPanel.Item;
364 }
365 }
366
367 return null;
368 }

395 private void CancelUsing(Character character)
396 {
397 if (character == null || character.Removed) { return; }
398
399 foreach (LimbPos lb in limbPositions)
400 {
401 Limb limb = character.AnimController.GetLimb(lb.LimbType);
402 if (limb == null) { continue; }
403
404 limb.Disabled = false;
405 limb.PullJointEnabled = false;
406 }
407
408 //disable flipping for 0.5 seconds, because flipping the character when it's in a weird pose (e.g. lying in bed) can mess up the ragdoll
409 if (character.AnimController is HumanoidAnimController humanoidAnim)
410 {
411 humanoidAnim.LockFlippingUntil = (float)Timing.TotalTime + 0.5f;
412 }
413
414 if (character.SelectedConstruction == this.item) { character.SelectedConstruction = null; }
415
416 character.AnimController.Anim = AnimController.Animation.None;
417 if (character == Character.Controlled)
418 {
419 HideHUDs(false);
420 }
421#if SERVER
422 item.CreateServerEvent(this);
423#endif

426 public override bool Select(Character activator)
427 {
428 if (activator == null || activator.Removed) { return false; }
429
430 if (UsableIn == UseEnvironment.Water && !activator.AnimController.InWater ||
431 UsableIn == UseEnvironment.Air && activator.AnimController.InWater)
432 {
433 return false;
434 }
435
436 //someone already using the item
437 if (user != null && !user.Removed)
438 {
439 if (user == activator)
440 {
441 IsActive = false;
442 CancelUsing(user);
443 user = null;
444 return false;
445 }
446 }
447 else
448 {
449 user = activator;
450 IsActive = true;
451 }
452#if SERVER
453 item.CreateServerEvent(this);
454#endif
455 item.SendSignal(new Signal("1", sender: user), "signal_out");
456 return true;

459 public override void FlipX(bool relativeToSub)
460 {
461 if (dir != Direction.None)
462 {
463 dir = dir == Direction.Left ? Direction.Right : Direction.Left;
464 }
465
466 userPos.X = -UserPos.X;
467
468 for (int i = 0; i < limbPositions.Count; i++)
469 {
470 float diff = (item.Rect.X + limbPositions[i].Position.X * item.Scale) - item.Rect.Center.X;
471
472 Vector2 flippedPos =
473 new Vector2(
474 (item.Rect.Center.X - diff - item.Rect.X) / item.Scale,
475 limbPositions[i].Position.Y);
476 limbPositions[i] = new LimbPos(limbPositions[i].LimbType, flippedPos, limbPositions[i].AllowUsingLimb);
477 }
478 }

480 public override void FlipY(bool relativeToSub)
481 {
482 userPos.Y = -UserPos.Y;
483
484 for (int i = 0; i < limbPositions.Count; i++)
485 {
486 float diff = (item.Rect.Y + limbPositions[i].Position.Y) - item.Rect.Center.Y;
487
488 Vector2 flippedPos =
489 new Vector2(
490 limbPositions[i].Position.X,
491 item.Rect.Center.Y - diff - item.Rect.Y);
492 limbPositions[i] = new LimbPos(limbPositions[i].LimbType, flippedPos, limbPositions[i].AllowUsingLimb);

533 private void LoadLimbPositions(XElement element)
534 {
535 limbPositions.Clear();
536 foreach (var subElement in element.Elements())
537 {
538 if (subElement.Name != "limbposition") { continue; }
539 string limbStr = subElement.GetAttributeString("limb", "");
540 if (!Enum.TryParse(subElement.GetAttribute("limb").Value, out LimbType limbType))
541 {
542 DebugConsole.ThrowError($"Error in item \"{item.Name}\" - {limbStr} is not a valid limb type.");
543 }
544 else
545 {
546 LimbPos limbPos = new LimbPos(limbType,
547 subElement.GetAttributeVector2("position", Vector2.Zero),
548 subElement.GetAttributeBool("allowusinglimb", false));
549 limbPositions.Add(limbPos);
550 if (!limbPos.AllowUsingLimb)
551 {
552 if (limbType == LimbType.RightHand || limbType == LimbType.RightForearm || limbType == LimbType.RightArm ||
553 limbType == LimbType.LeftHand || limbType == LimbType.LeftForearm || limbType == LimbType.LeftArm)
554 {
555 AllowAiming = false;
556 }
557 }
558 }
559 }
