60 public void AddElapsedTicks(string identifier, long ticks)

62 lock (mutex)

120 public bool Update(double deltaTime)
121 {
122 if (deltaTime == 0.0f) { return false; }
123
124 CurrentFramesPerSecond = 1.0 / deltaTime;
125
126 sampleBuffer.Enqueue(CurrentFramesPerSecond);
127
128 if (sampleBuffer.Count > MaximumSamples)
129 {
130 sampleBuffer.Dequeue();
131 AverageFramesPerSecond = sampleBuffer.Average(i => i);
132 }
133 else
134 {
135 AverageFramesPerSecond = CurrentFramesPerSecond;
136 }
137
138 return true;
