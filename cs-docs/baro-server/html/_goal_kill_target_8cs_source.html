35  private bool DoesDeathMatchCriteria()
36  {
37  if (Targets == null || Targets.Any(t => !t.IsDead)) return false;
38 
39  bool typeMatch = false;
40 
41  for (int i = 0; i < Targets.Count; i++)
42  {
43  // No specified cause of death required or missing cause of death
44  if (requiredCauseOfDeath == CauseOfDeathType.Unknown || Targets[i].CauseOfDeath == null)
45  {
46  typeMatch = true;
47  }
48  else
49  {
50  switch (Targets[i].CauseOfDeath.Type)
51  {
52  // If a cause of death is labeled as unknown, side with the traitor and accept this regardless of the required type
53  case CauseOfDeathType.Unknown:
54  typeMatch = true;
55  break;
56  case CauseOfDeathType.Pressure:
57  case CauseOfDeathType.Suffocation:
58  case CauseOfDeathType.Drowning:
59  typeMatch = requiredCauseOfDeath == Targets[i].CauseOfDeath.Type;
60  break;
61  case CauseOfDeathType.Affliction:
62  typeMatch = Targets[i].CauseOfDeath.Type == requiredCauseOfDeath && Targets[i].CauseOfDeath.Affliction.Identifier == afflictionId;
63  break;
64  case CauseOfDeathType.Disconnected:
65  typeMatch = false;
66  break;
67  }
68  }
69 
70  if (targetHull != null)
71  {
72  if (Targets[i].CurrentHull != null)
73  {
74  if (typeMatch && Targets[i].CurrentHull.RoomName == targetHull || Targets[i].CurrentHull.RoomName.Contains(targetHull))
75  {
76  continue;
77  }
78  else
79  {
80  return false;
81  }
82  }
83  else
84  {
85  // Outside the submarine, not supported for now
86  return false;
87  }
88  }
89  else
90  {
91  if (typeMatch)
92  {
93  continue;
94  }
95  else
96  {
97  return false;
98  }
99  }
100  }
101 
102  return true;
103  }
