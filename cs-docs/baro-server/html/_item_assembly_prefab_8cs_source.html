54  List<ushort> containedItemIDs = new List<ushort>();
55  foreach (XElement entityElement in element.Elements())
56  {
57  var containerElement = entityElement.GetChildElement("itemcontainer");
58  if (containerElement == null) { continue; }
59 
60  string containedString = containerElement.GetAttributeString("contained", "");
61  string[] itemIdStrings = containedString.Split(',');
62  var itemIds = new List<ushort>[itemIdStrings.Length];
63  for (int i = 0; i < itemIdStrings.Length; i++)
64  {
65  itemIds[i] ??= new List<ushort>();
66  foreach (string idStr in itemIdStrings[i].Split(';'))
67  {
68  if (int.TryParse(idStr, out int id))
69  {
70  itemIds[i].Add((ushort)id);
71  containedItemIDs.Add((ushort)id);
72  }
73  }
74  }
75  }

77  int minX = int.MaxValue, minY = int.MaxValue;
78  int maxX = int.MinValue, maxY = int.MinValue;
79  var displayEntities = new List<(Identifier, Rectangle)>();
80  foreach (XElement entityElement in element.Elements())
81  {
82  ushort id = (ushort)entityElement.GetAttributeInt("ID", 0);
83  if (id > 0 && containedItemIDs.Contains(id)) { continue; }
84 
85  Identifier identifier = entityElement.GetAttributeIdentifier("identifier", entityElement.Name.ToString().ToLowerInvariant());
86

88  if (!entityElement.Elements().Any(e => e.Name.LocalName.Equals("wire", StringComparison.OrdinalIgnoreCase)))
89  {
90  if (!entityElement.GetAttributeBool("hideinassemblypreview", false)) { displayEntities.Add((identifier, rect)); }
91  minX = Math.Min(minX, rect.X);
92  minY = Math.Min(minY, rect.Y - rect.Height);

122  public static List<MapEntity> PasteEntities(Vector2 position, Submarine sub, XElement configElement, string filePath = null, bool selectInstance = false)
123  {
124  int idOffset = Entity.FindFreeIdBlock(configElement.Elements().Count());
125  List<MapEntity> entities = MapEntity.LoadAll(sub, configElement, filePath, idOffset);
126  if (entities.Count == 0) { return entities; }
127 
128  Vector2 offset = sub?.HiddenSubPosition ?? Vector2.Zero;
129 
130  foreach (MapEntity me in entities)
131  {
132  me.Move(position);
133  me.Submarine = sub;
134  if (!(me is Item item)) { continue; }
135  Wire wire = item.GetComponent<Wire>();
136  //Vector2 subPosition = Submarine == null ? Vector2.Zero : Submarine.HiddenSubPosition;
137  if (wire != null)
138  {
139  //fix wires that have been erroneously saved at the "hidden position"
140  if (sub != null && Vector2.Distance(me.Position, sub.HiddenSubPosition) > sub.HiddenSubPosition.Length() / 2)
141  {
142  me.Move(position);
143  }
144  wire.MoveNodes(position - offset);
145  }
146  }
147 
148  MapEntity.MapLoaded(entities, true);
149 #if CLIENT
150  if (Screen.Selected == GameMain.SubEditorScreen && selectInstance)
151  {
152  MapEntity.SelectedList.Clear();
153  entities.ForEach(MapEntity.AddSelection);
154  }
155 #endif
156  return entities;
157  }
