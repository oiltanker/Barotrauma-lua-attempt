68 currentTarget = null;
69 if (user == null || (item.Submarine == null && OnlyInOwnSub))
70 {
71 IsActive = false;

73 }
74
75 float closestDist = float.PositiveInfinity;
76 foreach (Item targetItem in Item.ItemList)

79 if (OnlyInOwnSub)
80 {
81 if (targetItem.Submarine != item.Submarine) { continue; }
82 if (targetItem.Submarine.TeamID != user.TeamID) { continue; }
83 }
84 if (!targetItem.HasTag(Target) && ((MapEntity)targetItem).Prefab.Identifier != Target) { continue; }
85
86 float distSqr = Vector2.DistanceSquared(item.WorldPosition, targetItem.WorldPosition);
87 if (distSqr > Range * Range || distSqr > closestDist) { continue; }

89 currentTarget = targetItem;
