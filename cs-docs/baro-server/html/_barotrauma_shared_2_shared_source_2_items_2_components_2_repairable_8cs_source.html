213  public float RepairDegreeOfSuccess(Character character, List<Skill> skills)
214  {
215  if (skills.Count == 0) { return 1.0f; }
216 
217  float skillSum = (from t in skills let characterLevel = character.GetSkillLevel(t.Identifier) select (characterLevel - (t.Level * SkillRequirementMultiplier))).Sum();
218  float average = skillSum / skills.Count;
219 
220  return ((average + 100.0f) / 2.0f) / 100.0f;

361  public override void Update(float deltaTime, Camera cam)
362  {
363  UpdateProjSpecific(deltaTime);
364  IsTinkering = false;
365 
366  if (prevSentConditionValue != (int)item.ConditionPercentage || conditionSignal == null)
367  {
368  prevSentConditionValue = (int)item.ConditionPercentage;
369  conditionSignal = prevSentConditionValue.ToString();
370  }
371 
372  item.SendSignal(conditionSignal, "condition_out");
373 
374  if (CurrentFixer == null)
375  {
376  if (deteriorateAlwaysResetTimer > 0.0f)
377  {
378  deteriorateAlwaysResetTimer -= deltaTime;
379  if (deteriorateAlwaysResetTimer <= 0.0f)
380  {
381  DeteriorateAlways = false;
382 #if SERVER
383  //let the clients know the deterioration delay
384  item.CreateServerEvent(this);
385 #endif
386  }
387  }
388  if (!ShouldDeteriorate()) { return; }
389  if (item.Condition > 0.0f)
390  {
391  if (deteriorationTimer > 0.0f)
392  {
393  if (GameMain.NetworkMember == null || !GameMain.NetworkMember.IsClient)
394  {
395  deteriorationTimer -= deltaTime * GetDeteriorationDelayMultiplier();
396 #if SERVER
397  if (deteriorationTimer <= 0.0f) { item.CreateServerEvent(this); }
398 #endif
399  }
400  return;
401  }
402 
403  if (item.ConditionPercentage > MinDeteriorationCondition)
404  {
405  item.Condition -= DeteriorationSpeed * deltaTime;
406  }
407  }
408  return;
409  }
410 
411  UpdateFixAnimation(CurrentFixer);
412 
413  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
414 
415  if (CurrentFixer != null && (CurrentFixer.SelectedConstruction != item || !CurrentFixer.CanInteractWith(item) || CurrentFixer.IsDead))
416  {
417  StopRepairing(CurrentFixer);
418  return;
419  }
420 
421  if (currentFixerAction == FixActions.Tinker)
422  {
423  tinkeringDuration -= deltaTime;
424  // not great to interject it here, should be less reliant on returning
425 
426  float conditionDecrease = deltaTime * (CurrentFixer.GetStatValue(StatTypes.TinkeringDamage) / item.Prefab.Health) * 100f;
427  item.Condition -= conditionDecrease;
428 
429  if (!CanTinker(CurrentFixer) || tinkeringDuration <= 0f)
430  {
431  StopRepairing(CurrentFixer);
432  }
433  else
434  {
435  IsTinkering = true;
436  }
437  return;
438  }
439 
440  float successFactor = requiredSkills.Count == 0 ? 1.0f : RepairDegreeOfSuccess(CurrentFixer, requiredSkills);
441 
442  //item must have been below the repair threshold for the player to get an achievement or XP for repairing it
443  if (IsBelowRepairThreshold)
444  {
445  wasBroken = true;
446  }
447  if (item.ConditionPercentage > MinSabotageCondition)
448  {
449  wasGoodCondition = true;
450  }
451 
452  float fixDuration = MathHelper.Lerp(FixDurationLowSkill, FixDurationHighSkill, successFactor);
453  fixDuration /= 1 + CurrentFixer.GetStatValue(StatTypes.RepairSpeed) + currentRepairItem?.Prefab.AddedRepairSpeedMultiplier ?? 0f;
454  fixDuration /= 1 + item.GetQualityModifier(Quality.StatType.RepairSpeed);
455 
456  item.MaxRepairConditionMultiplier = GetMaxRepairConditionMultiplier(CurrentFixer);
457 
458  if (currentFixerAction == FixActions.Repair)
459  {
460  if (fixDuration <= 0.0f)
461  {
462  item.Condition = item.MaxCondition;
463  }
464  else
465  {
466  // scale with prefab's health instead of real health to ensure repair speed remains static with upgrades
467  float conditionIncrease = deltaTime / (fixDuration / item.Prefab.Health);
468  item.Condition += conditionIncrease;
469 #if SERVER
470  GameMain.Server.KarmaManager.OnItemRepaired(CurrentFixer, this, conditionIncrease);
471 #endif
472  }
473 
474  if (item.IsFullCondition)
475  {
476  if (wasBroken)
477  {
478  foreach (Skill skill in requiredSkills)
479  {
480  float characterSkillLevel = CurrentFixer.GetSkillLevel(skill.Identifier);
481  CurrentFixer.Info?.IncreaseSkillLevel(skill.Identifier,
482  SkillSettings.Current.SkillIncreasePerRepair / Math.Max(characterSkillLevel, 1.0f));
483  }
484  SteamAchievementManager.OnItemRepaired(item, CurrentFixer);
485  CurrentFixer.CheckTalents(AbilityEffectType.OnRepairComplete);
486  }
487  if (CurrentFixer?.SelectedConstruction == item) { CurrentFixer.SelectedConstruction = null; }
488  deteriorationTimer = Rand.Range(MinDeteriorationDelay, MaxDeteriorationDelay);
489  wasBroken = false;
490  StopRepairing(CurrentFixer);
491 #if SERVER
492  prevLoggedFixer = null;
493  prevLoggedFixAction = FixActions.None;
494 #endif
495  }
496  }
497  else if (currentFixerAction == FixActions.Sabotage)
498  {
499  if (fixDuration <= 0.0f)
500  {
501  item.Condition = item.MaxCondition * (MinSabotageCondition / 100);
502  }
503  else
504  {
505  // scale with prefab's health instead of real health to ensure sabotage speed remains static with (any) upgrades
506  float conditionDecrease = deltaTime / (fixDuration / item.Prefab.Health);
507  item.Condition -= conditionDecrease;
508  }
509 
510  if (item.ConditionPercentage <= MinSabotageCondition)
511  {
512  if (wasGoodCondition)
513  {
514  foreach (Skill skill in requiredSkills)
515  {
516  float characterSkillLevel = CurrentFixer.GetSkillLevel(skill.Identifier);
517  CurrentFixer.Info?.IncreaseSkillLevel(skill.Identifier,
518  SkillSettings.Current.SkillIncreasePerSabotage / Math.Max(characterSkillLevel, 1.0f));
519  }
520 
521  deteriorationTimer = 0.0f;
522  deteriorateAlwaysResetTimer = item.Condition / DeteriorationSpeed;
523  DeteriorateAlways = true;
524  item.Condition = item.MaxCondition * (MinSabotageCondition / 100);
525  wasGoodCondition = false;
526  }
527  StopRepairing(CurrentFixer);
528  }
529  }

532  throw new NotImplementedException(currentFixerAction.ToString());
533  }

585  private bool ShouldDeteriorate()
586  {
587  if (Level.IsLoadedOutpost) { return false; }
588 
589  if (LastActiveTime > Timing.TotalTime) { return true; }
590  foreach (ItemComponent ic in item.Components)
591  {
592  if (ic is Fabricator || ic is Deconstructor)
593  {
594  //fabricators and deconstructors rely on LastActiveTime
595  return false;
596  }
597  else if (ic is PowerTransfer pt)
598  {
599  //power transfer items (junction boxes, relays) don't deteriorate if they're no carrying any power
600  if (pt.Voltage > 0.1f) { return true; }
601  }
602  else if (ic is PowerContainer pc)
603  {
604  //batteries don't deteriorate if they're not charging/discharging
605  if (Math.Abs(pc.CurrPowerConsumption) > 0.1f || Math.Abs(pc.CurrPowerOutput) > 0.1f) { return true; }
606  }
607  else if (ic is Engine engine)
608  {
609  //engines don't deteriorate if they're not running
610  if (Math.Abs(engine.Force) > 1.0f) { return true; }
611  }
612  else if (ic is Pump pump)
613  {
614  //pumps don't deteriorate if they're not running
615  if (Math.Abs(pump.FlowPercentage) > 1.0f && pump.IsActive && pump.HasPower) { return true; }
616  }
617  else if (ic is Reactor reactor)
618  {
619  //reactors don't deteriorate if they're not powered up
620  if (reactor.Temperature > 0.1f) { return true; }
621  }
622  else if (ic is OxygenGenerator oxyGenerator)
623  {
624  //oxygen generators don't deteriorate if they're not running
625  if (oxyGenerator.CurrFlow > 0.1f) { return true; }
626  }
627  else if (ic is Powered powered && !(powered is LightComponent))
628  {
629  if (powered.Voltage >= powered.MinVoltage) { return true; }
630  }
631  }
632 
633  return DeteriorateAlways;
634  }

636  private float GetDeteriorationDelayMultiplier()
637  {
638  foreach (ItemComponent ic in item.Components)
639  {
640  if (ic is Engine engine)
641  {
642  return Math.Abs(engine.Force) / 100.0f;
643  }
644  else if (ic is Pump pump)
645  {
646  return Math.Abs(pump.FlowPercentage) / 100.0f;
647  }
648  else if (ic is Reactor reactor)

651  }
652  }
653  return 1.0f;
