33 (string value, bool loaded) tryLoad(LanguageIdentifier lang)
34 {
35 IReadOnlyList<string> candidates = Array.Empty<string>();
36 int tagIndex = 0;
37
38 if (TextManager.TextPacks.TryGetValue(lang, out var packs))
39 {
40 while (candidates.Count == 0 && tagIndex < tags.Length)
41 {
42 foreach (var pack in packs)
43 {
44 if (pack.Texts.TryGetValue(tags[tagIndex], out var texts))
45 {
46 candidates = candidates.ListConcat(texts);
47 }
48 }
49 tagIndex++;
50 }
51 }
52
53 bool loaded = candidates.Count > 0;
54 return (loaded ? candidates.GetRandomUnsynced() : "", loaded);
55 }
56
57 var (value, loaded) = tryLoad(language);
58 loadedSuccessfully = loaded ? LoadedSuccessfully.Yes : LoadedSuccessfully.No;
59 cachedValue = value;
60 if (!loaded && language != TextManager.DefaultLanguage)
61 {
62 (value, _) = tryLoad(TextManager.DefaultLanguage);
63 cachedValue = value;
64 //Notice how we don't set loadedSuccessfully again here.
