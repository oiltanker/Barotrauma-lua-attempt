24 public static void Shuffle<T>(this IList<T> list, Random rng)
25 {
26 int n = list.Count;
27 while (n > 1)
28 {
29 n--;
30 int k = rng.Next(n + 1);
31 T value = list[k];
32 list[k] = list[n];
33 list[n] = value;
34 }
35 }

99 public static T RandomElementByWeight<T>(this IList<T> source, Func<T, float> weightSelector, Rand.RandSync randSync = Rand.RandSync.Unsynced)
100 {
101 float totalWeight = source.Sum(weightSelector);
102
103 float itemWeightIndex = Rand.Range(0f, 1f, randSync) * totalWeight;
104 float currentWeightIndex = 0;
105
106 for (int i = 0; i < source.Count; i++)
107 {
108 T weightedItem = source[i];
109 float weight = weightSelector(weightedItem);
110 currentWeightIndex += weight;
111
112 if (currentWeightIndex >= itemWeightIndex)
113 {
114 return weightedItem;
115 }
116 }
117
118 return default;
119 }
