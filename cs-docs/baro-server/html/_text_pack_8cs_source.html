55 public TextPack(TextFile file, ContentXElement mainElement, LanguageIdentifier language)
56 {
57 ContentFile = file;
58
59 var languageName = mainElement.GetAttributeIdentifier("language", TextManager.DefaultLanguage.Value);
60 Language = language;
61 TranslatedName = mainElement.GetAttributeString("translatedname", languageName.Value);
62 NoWhitespace = mainElement.GetAttributeBool("nowhitespace", false);
63
64 Dictionary<Identifier, List<string>> texts = new Dictionary<Identifier, List<string>>();
65 foreach (var element in mainElement.Elements())
66 {
67 Identifier elemName = element.NameAsIdentifier();
68 if (!texts.ContainsKey(elemName)) { texts.Add(elemName, new List<string>()); }
69 texts[elemName].Add(element.ElementInnerText()
70 .Replace(@"\n", "\n")
71 .Replace("&amp;", "&")
72 .Replace("&lt;", "<")
73 .Replace("&gt;", ">")
74 .Replace("&quot;", "\"")
75 .Replace("&apos;", "'"));
76 }
77 Texts = texts.Select(kvp => (kvp.Key, kvp.Value.ToImmutableArray())).ToImmutableDictionary();
78 }

150
151 XDocument doc = XMLExtensions.TryLoadXml(ContentFile.Path);
152 if (doc == null) { return; }
153
154 List<(string key, string value)> texts = new List<(string key, string value)>();
155
156 foreach (var element in doc.Root.Elements())
157 {
158 string text = element.ElementInnerText()
159 .Replace("&amp;", "&")
160 .Replace("&lt;", "<")
161 .Replace("&gt;", ">")
162 .Replace("&quot;", "\"")
163 .Replace("&apos;", "'");
164
165 texts.Add((element.Name.ToString(), text));
166 }
167
168 foreach ((string key, string value) in texts)
169 {
170 sb.Append(key); // ID
171 sb.Append('*');
172 sb.Append(value); // Original
173 sb.Append('*');
174 // Translated
175 sb.Append('*');
176 // Comments
177 sb.AppendLine();
178
179 }
