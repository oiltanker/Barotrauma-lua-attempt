38 public static Dictionary<Identifier, string> ParseQueryString(string query)
39 {
40 Dictionary<Identifier, string> collection = new Dictionary<Identifier, string>();
41 var splitGet = query.Split('?');
42 if (splitGet.Length > 1)
43 {
44 var get = splitGet[1];
45 foreach (string kvp in get.Split('&'))
46 {
47 var splitKeyValue = kvp.Split('=');
48 if (splitKeyValue.Length > 1)
49 {
50 collection.Add(splitKeyValue[0].ToIdentifier(), splitKeyValue[1]);
51 }
52 }
53 }
54 return collection;
55 }
56 }
