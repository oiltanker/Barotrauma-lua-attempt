136  {

140  string itemName = itemElement.Attribute("name").Value;

142  itemPrefab = MapEntityPrefab.Find(itemName) as ItemPrefab;
143  if (itemPrefab == null)
144  {
145  DebugConsole.ThrowError("Tried to spawn \"" + Name + "\" with the item \"" + itemName + "\". Matching item prefab not found.");
146  return;
147  }
148  }
149  else
150  {
151  string itemIdentifier = itemElement.GetAttributeString("identifier", "");
152  itemPrefab = MapEntityPrefab.FindByIdentifier(itemIdentifier.ToIdentifier()) as ItemPrefab;
153  if (itemPrefab == null)
154  {
155  DebugConsole.ThrowError("Tried to spawn \"" + Name + "\" with the item \"" + itemIdentifier + "\". Matching item prefab not found.");
156  return;
157  }
158  }
159 
160  Item item = new Item(itemPrefab, character.Position, null);
161 
162 #if SERVER
163  if (GameMain.Server != null && Entity.Spawner != null)
164  {
165  if (GameMain.Server.EntityEventManager.UniqueEvents.Any(ev => ev.Entity == item))
166  {
167  string errorMsg = $"Error while spawning job items. Item {item.Name} created network events before the spawn event had been created.";
168  DebugConsole.ThrowError(errorMsg);
169  GameAnalyticsManager.AddErrorEventOnce("Job.InitializeJobItem:EventsBeforeSpawning", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
170  GameMain.Server.EntityEventManager.UniqueEvents.RemoveAll(ev => ev.Entity == item);
171  GameMain.Server.EntityEventManager.Events.RemoveAll(ev => ev.Entity == item);

173 
174  Entity.Spawner.CreateNetworkEvent(new EntitySpawner.SpawnEntity(item));
175  }
176 #endif
177 
178  if (itemElement.GetAttributeBool("equip", false))
179  {
180  //if the item is both pickable and wearable, try to wear it instead of picking it up
181  List<InvSlotType> allowedSlots =
182  item.GetComponents<Pickable>().Count() > 1 ?
183  new List<InvSlotType>(item.GetComponent<Wearable>()?.AllowedSlots ?? item.GetComponent<Pickable>().AllowedSlots) :
184  new List<InvSlotType>(item.AllowedSlots);

187  }
