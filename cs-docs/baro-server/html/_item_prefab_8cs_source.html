129 public FabricationRecipe(XElement element, Identifier itemPrefab)
130 {
131 TargetItemPrefabIdentifier = itemPrefab;
132 var displayNameIdentifier = element.GetAttributeIdentifier("displayname", "");
133 displayName = new Lazy<LocalizedString>(() => displayNameIdentifier.IsEmpty
134 ? TargetItem.Name
135 : TextManager.GetWithVariable($"DisplayName.{displayNameIdentifier}", "[itemname]", TargetItem.Name));
136
137 SuitableFabricatorIdentifiers = element.GetAttributeIdentifierArray("suitablefabricators", Array.Empty<Identifier>()).ToImmutableArray();
138
139 var requiredSkills = new List<Skill>();
140 RequiredTime = element.GetAttributeFloat("requiredtime", 1.0f);
141 RequiredMoney = element.GetAttributeInt("requiredmoney", 0);
142 OutCondition = element.GetAttributeFloat("outcondition", 1.0f);
143 if (OutCondition > 1.0f)
144 {
145 DebugConsole.AddWarning($"Error in \"{itemPrefab}\"'s fabrication recipe: out condition is above 100% ({OutCondition * 100}).");
146 }
147 var requiredItems = new List<RequiredItem>();
148 RequiresRecipe = element.GetAttributeBool("requiresrecipe", false);
149 Amount = element.GetAttributeInt("amount", 1);
150
151 int limitDefault = element.GetAttributeInt("fabricationlimit", -1);
152 FabricationLimitMin = element.GetAttributeInt(nameof(FabricationLimitMin), limitDefault);
153 FabricationLimitMax = element.GetAttributeInt(nameof(FabricationLimitMax), limitDefault);
154
155 if (element.GetAttribute(nameof(Quality)) != null)
156 {
157 Quality = element.GetAttributeInt(nameof(Quality), 0);
158 }
159
160 foreach (var subElement in element.Elements())
161 {
162 switch (subElement.Name.ToString().ToLowerInvariant())
163 {
164 case "requiredskill":
165 if (subElement.Attribute("name") != null)
166 {
167 DebugConsole.ThrowError("Error in fabricable item " + itemPrefab + "! Use skill identifiers instead of names.");
168 continue;
169 }
170
171 requiredSkills.Add(new Skill(
172 subElement.GetAttributeIdentifier("identifier", ""),
173 subElement.GetAttributeInt("level", 0)));
174 break;
175 case "item":
176 case "requireditem":
177 Identifier requiredItemIdentifier = subElement.GetAttributeIdentifier("identifier", Identifier.Empty);
178 Identifier requiredItemTag = subElement.GetAttributeIdentifier("tag", Identifier.Empty);
179 if (requiredItemIdentifier == Identifier.Empty && requiredItemTag == Identifier.Empty)
180 {
181 DebugConsole.ThrowError("Error in fabricable item " + itemPrefab + "! One of the required items has no identifier or tag.");
182 continue;
183 }
184
185 float minCondition = subElement.GetAttributeFloat("mincondition", 1.0f);
186 float maxCondition = subElement.GetAttributeFloat("maxcondition", 1.0f);
187 //Substract mincondition from required item's condition or delete it regardless?
188 bool useCondition = subElement.GetAttributeBool("usecondition", true);
189 int amount = subElement.GetAttributeInt("count", subElement.GetAttributeInt("amount", 1));
190
191 if (requiredItemIdentifier != Identifier.Empty)
192 {
193 var existing = requiredItems.FindIndex(r =>
194 r is RequiredItemByIdentifier ri &&
195 ri.ItemPrefabIdentifier == requiredItemIdentifier &&
196 MathUtils.NearlyEqual(r.MinCondition, minCondition) &&
197 MathUtils.NearlyEqual(r.MaxCondition, maxCondition));
198 if (existing >= 0)
199 {
200 amount += requiredItems[existing].Amount;
201 requiredItems.RemoveAt(existing);
202 }
203 requiredItems.Add(new RequiredItemByIdentifier(requiredItemIdentifier, amount, minCondition, maxCondition, useCondition));
204 }
205 else
206 {
207 var existing = requiredItems.FindIndex(r =>
208 r is RequiredItemByTag rt &&
209 rt.Tag == requiredItemTag &&
210 MathUtils.NearlyEqual(r.MinCondition, minCondition) &&
211 MathUtils.NearlyEqual(r.MaxCondition, maxCondition));
212 if (existing >= 0)
213 {
214 amount += requiredItems[existing].Amount;
215 requiredItems.RemoveAt(existing);
216 }
217 requiredItems.Add(new RequiredItemByTag(requiredItemTag, amount, minCondition, maxCondition, useCondition));
218 }
219 break;
220 }
221 }
222
223 this.RequiredSkills = requiredSkills.ToImmutableArray();
224 this.RequiredItems = requiredItems.ToImmutableArray();
225
226 RecipeHash = GenerateHash();
227 }
228
229 private uint GenerateHash()
230 {
231 using var md5 = MD5.Create();
232 uint outputId = ToolBox.IdentifierToUint32Hash(TargetItemPrefabIdentifier, md5);
233
234 var requiredItems = string.Join(':', RequiredItems
235 .Select(i => i.UintIdentifier)
236 .Select(i => string.Join(',', i)));
237
238 var requiredSkills = string.Join(':', RequiredSkills.Select(s => $"{s.Identifier}:{s.Level}"));
239
240 uint retVal = ToolBox.StringToUInt32Hash($"{Amount}|{outputId}|{RequiredTime}|{requiredItems}|{requiredSkills}", md5);
241 if (retVal == 0) { retVal = 1; }
242 return retVal;
243 }

643 private void ParseConfigElement(ItemPrefab variantOf)
644 {
645 string categoryStr = ConfigElement.GetAttributeString("category", "Misc");
646 this.category = Enum.TryParse(categoryStr, true, out MapEntityCategory category)
647 ? category
648 : MapEntityCategory.Misc;
649
650 var parentType = ConfigElement.Parent?.GetAttributeIdentifier("itemtype", "");
651
652 //nameidentifier can be used to make multiple items use the same names and descriptions
653 Identifier nameIdentifier = ConfigElement.GetAttributeIdentifier("nameidentifier", "");
654
655 //only used if the item doesn't have a name/description defined in the currently selected language
656 string fallbackNameIdentifier = ConfigElement.GetAttributeString("fallbacknameidentifier", "");
657
658 //works the same as nameIdentifier, but just replaces the description
659 Identifier descriptionIdentifier = ConfigElement.GetAttributeIdentifier("descriptionidentifier", "");
660
661 if (string.IsNullOrEmpty(OriginalName))
662 {
663 name = TextManager.Get(nameIdentifier.IsEmpty
664 ? $"EntityName.{Identifier}"
665 : $"EntityName.{nameIdentifier}",
666 $"EntityName.{fallbackNameIdentifier}");
667 }
668 else if (Category.HasFlag(MapEntityCategory.Legacy))

671 name = TextManager.Get(nameIdentifier.IsEmpty
672 ? $"EntityName.{Identifier}"
673 : $"EntityName.{nameIdentifier}");
674 name = name.Fallback(OriginalName);
675 }
676
677 if (parentType == "wrecked")
678 {
679 name = TextManager.GetWithVariable("wreckeditemformat", "[name]", name);
680 }
681
682 name = GeneticMaterial.TryCreateName(this, ConfigElement);
683
684 this.aliases =
685 (ConfigElement.GetAttributeStringArray("aliases", null, convertToLowerInvariant: true) ??
686 ConfigElement.GetAttributeStringArray("Aliases", Array.Empty<string>(), convertToLowerInvariant: true))
687 .ToImmutableHashSet()

695 var preferredContainers = new List<PreferredContainer>();
696 DeconstructTime = 1.0f;
697
698 if (ConfigElement.GetAttribute("allowasextracargo") != null)

700 AllowAsExtraCargo = ConfigElement.GetAttributeBool("allowasextracargo", false);
701 }
702
703 this.tags = ConfigElement.GetAttributeIdentifierArray("tags", Array.Empty<Identifier>()).ToImmutableHashSet();
