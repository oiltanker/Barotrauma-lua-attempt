151 public void StartServer()
152 {
153 string name = "Server";
154 int port = NetConfig.DefaultPort;
155 int queryPort = NetConfig.DefaultQueryPort;
156 bool publiclyVisible = false;
157 string password = "";
158 bool enableUpnp = false;
159
160 int maxPlayers = 10;
161 int? ownerKey = null;
162 UInt64 steamId = 0;
163 IPAddress listenIp = IPAddress.Any;
164
165 XDocument doc = XMLExtensions.TryLoadXml(ServerSettings.SettingsFile);
166 if (doc?.Root == null)
167 {
168 DebugConsole.ThrowError("File \"" + ServerSettings.SettingsFile + "\" not found. Starting the server with default settings.");
169 }
170 else
171 {
172 name = doc.Root.GetAttributeString("name", "Server");
173 port = doc.Root.GetAttributeInt("port", NetConfig.DefaultPort);
174 queryPort = doc.Root.GetAttributeInt("queryport", NetConfig.DefaultQueryPort);
175 publiclyVisible = doc.Root.GetAttributeBool("public", false);
176 password = doc.Root.GetAttributeString("password", "");
177 enableUpnp = doc.Root.GetAttributeBool("enableupnp", false);
178 maxPlayers = doc.Root.GetAttributeInt("maxplayers", 10);
179 ownerKey = null;
180 }
181
182#if DEBUG
183 foreach (string s in CommandLineArgs)
184 {
185 Console.WriteLine(s);
186 }
187#endif
188
189 for (int i = 0; i < CommandLineArgs.Length; i++)
190 {
191 switch (CommandLineArgs[i].Trim())
192 {
193 case "-name":
194 name = CommandLineArgs[i + 1];
195 i++;
196 break;
197 case "-ip":
198 if (IPAddress.TryParse(CommandLineArgs[i + 1], out IPAddress address))
199 listenIp = address;
200 else
201 DebugConsole.ThrowError($"Invalid Ip Address '{CommandLineArgs[i + 1]}'.");
202 break;
203 case "-port":
204 int.TryParse(CommandLineArgs[i + 1], out port);
205 i++;
206 break;
207 case "-queryport":
208 int.TryParse(CommandLineArgs[i + 1], out queryPort);
209 i++;
210 break;
211 case "-public":
212 bool.TryParse(CommandLineArgs[i + 1], out publiclyVisible);
213 i++;
214 break;
215 case "-password":
216 password = CommandLineArgs[i + 1];
217 i++;
218 break;
219 case "-nopassword":
220 password = "";
221 break;
222 case "-upnp":
223 case "-enableupnp":
224 bool.TryParse(CommandLineArgs[i + 1], out enableUpnp);
225 i++;
226 break;
227 case "-maxplayers":
228 int.TryParse(CommandLineArgs[i + 1], out maxPlayers);
229 i++;
230 break;
231 case "-ownerkey":
232 if (int.TryParse(CommandLineArgs[i + 1], out int key))
233 {
234 ownerKey = key;
235 }
236 i++;
237 break;
238 case "-steamid":
239 UInt64.TryParse(CommandLineArgs[i + 1], out steamId);
240 i++;
241 break;
242 case "-pipes":
243 //handled in TryStartChildServerRelay
244 i += 2;
245 break;
246 }
247 }
248
249 Server = new GameServer(
250 name,
251 listenIp,
252 port,
253 queryPort,
254 publiclyVisible,
255 password,
256 enableUpnp,
257 maxPlayers,
258 ownerKey,
259 steamId);
260
261 for (int i = 0; i < CommandLineArgs.Length; i++)
262 {
263 switch (CommandLineArgs[i].Trim())
264 {
265 case "-playstyle":
266 Enum.TryParse(CommandLineArgs[i + 1], out PlayStyle playStyle);
267 Server.ServerSettings.PlayStyle = playStyle;
268 i++;
269 break;
270 case "-banafterwrongpassword":
271 bool.TryParse(CommandLineArgs[i + 1], out bool banAfterWrongPassword);
272 Server.ServerSettings.BanAfterWrongPassword = banAfterWrongPassword;
273 break;
274 case "-karma":
275 case "-karmaenabled":
276 bool.TryParse(CommandLineArgs[i + 1], out bool karmaEnabled);
277 Server.ServerSettings.KarmaEnabled = karmaEnabled;
278 i++;
279 break;
280 case "-karmapreset":
281 string karmaPresetName = CommandLineArgs[i + 1];
282 Server.ServerSettings.KarmaPreset = karmaPresetName;
283 i++;
284 break;
285 }
286 }
287 }

296 public void Run()
297 {
298 Hyper.ComponentModel.HyperTypeDescriptionProvider.Add(typeof(Character));
299 Hyper.ComponentModel.HyperTypeDescriptionProvider.Add(typeof(Item));
300 Hyper.ComponentModel.HyperTypeDescriptionProvider.Add(typeof(Items.Components.ItemComponent));
301 Hyper.ComponentModel.HyperTypeDescriptionProvider.Add(typeof(Hull));
302
303 Init();
304 StartServer();
305
306 ResetFrameTime();
307
308 double frequency = (double)Stopwatch.Frequency;
309 if (frequency <= 1500)
310 {
311 DebugConsole.NewMessage("WARNING: Stopwatch frequency under 1500 ticks per second. Expect significant syncing accuracy issues.", Color.Yellow);
312 }
313
314 Stopwatch performanceCounterTimer = Stopwatch.StartNew();
315
316 stopwatch = Stopwatch.StartNew();
317 long prevTicks = stopwatch.ElapsedTicks;
318 while (ShouldRun)
319 {
320 long currTicks = stopwatch.ElapsedTicks;
321 double elapsedTime = Math.Max(currTicks - prevTicks, 0) / frequency;
322 Timing.Accumulator += elapsedTime;
323 if (Timing.Accumulator > Timing.AccumulatorMax)
324 {
325 //prevent spiral of death:
326 //if the game's running too slowly then we have no choice but to skip a bunch of steps
327 //otherwise it snowballs and becomes unplayable
328 Timing.Accumulator = Timing.Step;
329 }
330
331 CrossThread.ProcessTasks();
332
333 prevTicks = currTicks;
334 while (Timing.Accumulator >= Timing.Step)
335 {
336 performanceCounterTimer.Start();
337
338 Timing.TotalTime += Timing.Step;
339 DebugConsole.Update();
340 if (GameSession?.GameMode == null || !GameSession.GameMode.Paused)
341 {
342 Screen.Selected?.Update((float)Timing.Step);
343 }
344 Server.Update((float)Timing.Step);
345 if (Server == null) { break; }
346 SteamManager.Update((float)Timing.Step);
347 TaskPool.Update();
348 CoroutineManager.Update((float)Timing.Step, (float)Timing.Step);
349
350 GameMain.LuaCs.Update();
351 GameMain.LuaCs.Hook.Call("think", new object[] { });
352 performanceCounterTimer.Stop();
353 LuaCsTimer.LastUpdateTime = performanceCounterTimer.ElapsedMilliseconds;
354 performanceCounterTimer.Reset();
355
356 Timing.Accumulator -= Timing.Step;
357 updateCount++;
358 }
359
360#if !DEBUG
361 if (Server?.OwnerConnection == null)
362 {
363 DebugConsole.UpdateCommandLine((int)(Timing.Accumulator * 800));
364 }
365 else
366 {
367 DebugConsole.Clear();
368 }
369#else
370 DebugConsole.UpdateCommandLine((int)(Timing.Accumulator * 800));
371#endif
372
373 int frameTime = (int)((stopwatch.ElapsedTicks - prevTicks) / frequency * 1000.0);
374 frameTime = Math.Max(0, frameTime);
375
376 Thread.Sleep(Math.Max(((int)(Timing.Step * 1000.0) - frameTime) / 2, 0));
377
378 if (performanceCounterTimer.ElapsedMilliseconds > 1000)
379 {
380 int updateRate = (int)Math.Round(updateCount / (double)(performanceCounterTimer.ElapsedMilliseconds / 1000.0));
381 prevUpdateRates.Enqueue(updateRate);
382 if (prevUpdateRates.Count >= 10)
383 {
384 int avgUpdateRate = (int)prevUpdateRates.Average();
385 if (avgUpdateRate < Timing.FixedUpdateRate * 0.98 && GameSession != null && Timing.TotalTime > GameSession.RoundStartTime + 1.0)
386 {
387 DebugConsole.AddWarning($"Running slowly ({avgUpdateRate} updates/s)!");
388 if (Server != null)
389 {
390 foreach (Client c in Server.ConnectedClients)
391 {
392 if (c.Connection == Server.OwnerConnection || c.Permissions != ClientPermissions.None)
393 {
394 Server.SendConsoleMessage($"Server running slowly ({avgUpdateRate} updates/s)!", c, Color.Orange);
395 }
396 }
397 }
398 }
399 prevUpdateRates.Clear();
400 }
401 performanceCounterTimer.Restart();
402 updateCount = 0;
403 }
404 }
405 stopwatch.Stop();
406
407 CloseServer();
408
409 SteamManager.ShutDown();
410
411 SaveUtil.CleanUnnecessarySaveFiles();
412
413 if (GameSettings.CurrentConfig.SaveDebugConsoleLogs
414 || GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.SaveLogs(); }
415 if (GameAnalyticsManager.SendUserStatistics) { GameAnalyticsManager.ShutDown(); }
416
417 MainThread = null;
418 }
