147 private bool? CompareFloat(float val1, float val2)
148 {
149 value1 = val1;
150 value2 = val2;
151 switch (Operator)
152 {
153 case PropertyConditional.OperatorType.Equals:
154 return MathUtils.NearlyEqual(val1, val2);
155 case PropertyConditional.OperatorType.GreaterThan:
156 return val1 > val2;
157 case PropertyConditional.OperatorType.GreaterThanEquals:
158 return val1 >= val2;
159 case PropertyConditional.OperatorType.LessThan:
160 return val1 < val2;
161 case PropertyConditional.OperatorType.LessThanEquals:
162 return val1 <= val2;
163 case PropertyConditional.OperatorType.NotEquals:
164 return !MathUtils.NearlyEqual(val1, val2);
165 }

167 return null;

175 private bool? CompareString(string val1, string val2)
176 {
177 value1 = val1;
178 value2 = val2;

183 return equals;
184 case PropertyConditional.OperatorType.NotEquals:
185 return !equals;
186 default:
187 DebugConsole.Log($"Only \"Equals\" and \"Not equals\" operators are allowed for a string (was {Operator} for {val2}).");
