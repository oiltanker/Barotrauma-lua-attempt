346 public override void Update(float deltaTime, Camera cam)
347 {
348 UpdateProjSpecific(deltaTime);
349 toggleCooldownTimer -= deltaTime;
350 damageSoundCooldown -= deltaTime;
351
352 if (isBroken)
353 {
354 lastBrokenTime = Timing.TotalTime;
355 //the door has to be restored to 50% health before collision detection on the body is re-enabled
356 if (item.ConditionPercentage > 50.0f && (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
357 {
358 IsBroken = false;
359 }
360 return;
361 }
362
363 bool isClosing = false;
364 if ((!IsStuck && !IsJammed) || !isOpen)
365 {
366 if (PredictedState == null)
367 {
368 OpenState += deltaTime * (isOpen ? OpeningSpeed : -ClosingSpeed);
369 isClosing = openState > 0.0f && openState < 1.0f && !isOpen;
370 }
371 else
372 {
373 OpenState += deltaTime * ((bool)PredictedState ? OpeningSpeed : -ClosingSpeed);
374 isClosing = openState > 0.0f && openState < 1.0f && !(bool)PredictedState;
375
376 resetPredictionTimer -= deltaTime;
377 if (resetPredictionTimer <= 0.0f)
378 {
379 PredictedState = null;
380 }
381 }
382 LinkedGap.Open = isBroken ? 1.0f : openState;
383 }
384
385 if (isClosing)
386 {
387 if (OpenState < 0.9f) { PushCharactersAway(); }
388 }
389 else
390 {
391 bool wasEnabled = Body.Enabled;
392 Body.Enabled = Impassable || openState < 1.0f;
393 if (wasEnabled && !Body.Enabled && IsHorizontal)
394 {
395 //when opening a hatch, force characters above it to refresh the floor position
396 //(otherwise the character won't fall through the hatch until it moves)
397 foreach (Character c in Character.CharacterList)
398 {
399 if (c.WorldPosition.Y < item.WorldPosition.Y) { continue; }
400 if (c.WorldPosition.X < item.WorldRect.X || c.WorldPosition.X > item.WorldRect.Right) { continue; }
401 c.AnimController?.ForceRefreshFloorY();
402 }
403 }
404 }
405
406 //don't use the predicted state here, because it might set
407 //other items to an incorrect state if the prediction is wrong
408 item.SendSignal(isOpen ? "1" : "0", "state_out");
409 }

533 bool itemPosErrorShown;
534 private readonly HashSet<Character> characterPosErrorShown = new HashSet<Character>();
535 private void PushCharactersAway()
536 {
537 if (!MathUtils.IsValid(item.SimPosition))
538 {
539 if (!itemPosErrorShown)
540 {
541 DebugConsole.ThrowError("Failed to push a character out of a doorway - position of the door is not valid (" + item.SimPosition + ")");
542 GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:DoorPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
543 "Failed to push a character out of a doorway - position of the door is not valid (" + item.SimPosition + ").");
544 itemPosErrorShown = true;
545 }
546 return;
547 }
548
549 Vector2 simPos = ConvertUnits.ToSimUnits(new Vector2(item.Rect.X, item.Rect.Y));
550
551 Vector2 currSize = IsHorizontal ?
552 new Vector2(item.Rect.Width * (1.0f - openState), doorSprite.size.Y * item.Scale) :
553 new Vector2(doorSprite.size.X * item.Scale, item.Rect.Height * (1.0f - openState));
554
555 Vector2 simSize = ConvertUnits.ToSimUnits(currSize);
556
557 foreach (Character c in Character.CharacterList)
558 {
559 if (!c.Enabled) { continue; }
560 if (!MathUtils.IsValid(c.SimPosition))
561 {
562 if (!characterPosErrorShown.Contains(c))
563 {
564 if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError("Failed to push a character out of a doorway - position of the character \"" + c.Name + "\" is not valid (" + c.SimPosition + ")"); }
565 GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:CharacterPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
566 "Failed to push a character out of a doorway - position of the character \"" + c.SpeciesName + "\" is not valid (" + c.SimPosition + ")." +
567 " Removed: " + c.Removed +
568 " Remoteplayer: " + c.IsRemotePlayer);
569 characterPosErrorShown.Add(c);
570 }
571 continue;
572 }
573 int dir = IsHorizontal ? Math.Sign(c.SimPosition.Y - item.SimPosition.Y) : Math.Sign(c.SimPosition.X - item.SimPosition.X);
574
575 foreach (Limb limb in c.AnimController.Limbs)
576 {
577 if (limb.IsSevered) { continue; }
578 if (PushBodyOutOfDoorway(c, limb.body, dir, simPos, simSize) && damageSoundCooldown <= 0.0f)
579 {
580#if CLIENT
581 SoundPlayer.PlayDamageSound("LimbBlunt", 1.0f, limb.body);
582#endif
583 damageSoundCooldown = 0.5f;
584 }
585 }
586 PushBodyOutOfDoorway(c, c.AnimController.Collider, dir, simPos, simSize);
587 }

590 private bool PushBodyOutOfDoorway(Character c, PhysicsBody body, int dir, Vector2 doorRectSimPos, Vector2 doorRectSimSize)
591 {
592 if (!MathUtils.IsValid(body.SimPosition))
593 {
594 DebugConsole.ThrowError("Failed to push a limb out of a doorway - position of the body (character \"" + c.Name + "\") is not valid (" + body.SimPosition + ")");
595 GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:LimbPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
596 "Failed to push a character out of a doorway - position of the character \"" + c.SpeciesName + "\" is not valid (" + body.SimPosition + ")." +
597 " Removed: " + c.Removed +
598 " Remoteplayer: " + c.IsRemotePlayer);
599 return false;
600 }
601
602 float diff;
603 if (IsHorizontal)
604 {
605 if (body.SimPosition.X < doorRectSimPos.X || body.SimPosition.X > doorRectSimPos.X + doorRectSimSize.X) { return false; }
606 diff = body.SimPosition.Y - item.SimPosition.Y;
607 }
608 else
609 {
610 if (body.SimPosition.Y > doorRectSimPos.Y || body.SimPosition.Y < doorRectSimPos.Y - doorRectSimSize.Y) { return false; }
611 diff = body.SimPosition.X - item.SimPosition.X;
612 }
613
614 //if the limb is at a different side of the door than the character (collider),
615 //immediately teleport it to the correct side
616 if (Math.Sign(diff) != dir)
617 {
618 if (IsHorizontal)
619 {
620 body.SetTransform(new Vector2(body.SimPosition.X, item.SimPosition.Y + dir * doorRectSimSize.Y * 2.0f), body.Rotation);
621 }
622 else
623 {
624 body.SetTransform(new Vector2(item.SimPosition.X + dir * doorRectSimSize.X * 1.2f, body.SimPosition.Y), body.Rotation);
625 }
626 }
627
628 //apply an impulse to push the limb further from the door
629 if (IsHorizontal)
630 {
631 if (Math.Abs(body.SimPosition.Y - item.SimPosition.Y) > doorRectSimSize.Y * 0.5f) { return false; }
632 body.ApplyLinearImpulse(new Vector2(isOpen ? 0.0f : 1.0f, dir * 2.0f), maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
633 }
634 else
635 {
636 if (Math.Abs(body.SimPosition.X - item.SimPosition.X) > doorRectSimSize.X * 0.5f) { return false; }
637 body.ApplyLinearImpulse(new Vector2(dir * 2.0f, isOpen ? 0.0f : -1.0f), maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
638 }
639
640 //don't stun if the door was broken a moment ago
641 //otherwise enabling the door's collider and pushing the character away will interrupt repairing
642 if (lastBrokenTime < Timing.TotalTime - 1.0f)
643 {
644 c.SetStun(0.2f);
645 }
646 return true;
647 }

664 public override void ReceiveSignal(Signal signal, Connection connection)
665 {
666 if (IsStuck || IsJammed) { return; }
667
668 bool wasOpen = PredictedState == null ? isOpen : PredictedState.Value;
669
670 if (connection.Name == "toggle")
671 {
672 if (signal.value == "0") { return; }
673 if (toggleCooldownTimer > 0.0f && signal.sender != lastUser) { OnFailedToOpen(); return; }
674 if (IsStuck) { toggleCooldownTimer = 1.0f; OnFailedToOpen(); return; }
675 toggleCooldownTimer = ToggleCoolDown;
676 lastUser = signal.sender;
677 SetState(!wasOpen, false, true, forcedOpen: false);
678 }
679 else if (connection.Name == "set_state")
680 {
681 bool signalOpen = signal.value != "0";
682 if (IsStuck && signalOpen != wasOpen) { toggleCooldownTimer = 1.0f; OnFailedToOpen(); return; }
683 SetState(signalOpen, false, true, forcedOpen: false);
684 }
685
686#if SERVER
687 if (signal.sender != null && wasOpen != isOpen)
688 {
689 GameServer.Log(GameServer.CharacterLogName(signal.sender) + (isOpen ? " opened " : " closed ") + item.Name, ServerLog.MessageType.ItemInteraction);
690 }
691#endif
692 }
