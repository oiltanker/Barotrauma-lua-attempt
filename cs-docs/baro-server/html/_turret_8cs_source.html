395 
396  if (SingleChargedShot && reload > 0f)
397  {
398  // single charged shot guns will decharge after firing
399  // for cosmetic reasons, this is done by lerping in half the reload time
400  currentChargeTime = Math.Max(0f, MaxChargeTime * (reload / reloadTime - 0.5f));
401  }
402  else
403  {
404  float chargeDeltaTime = tryingToCharge ? deltaTime : -deltaTime;
405  if (chargeDeltaTime > 0f && user != null)
406  {
407  chargeDeltaTime *= 1f + user.GetStatValue(StatTypes.TurretChargeSpeed);
408  }
409  currentChargeTime = Math.Clamp(currentChargeTime + chargeDeltaTime, 0f, MaxChargeTime);
410  }
411  tryingToCharge = false;
412 
413  if (currentChargeTime == 0f)
414  {
415  currentChargingState = ChargingState.Inactive;
416  }
417  else if (currentChargeTime < previousChargeTime)
418  {
419  currentChargingState = ChargingState.WindingDown;
420  }
421  else
422  {
423  // if we are charging up or at maxed charge, remain winding up
424  currentChargingState = ChargingState.WindingUp;
425  }
426 
427  UpdateProjSpecific(deltaTime);
428 
429  if (MathUtils.NearlyEqual(minRotation, maxRotation))
430  {
431  UpdateLightComponent();
432  return;
433  }
434 
435  float targetMidDiff = MathHelper.WrapAngle(targetRotation - (minRotation + maxRotation) / 2.0f);
436 
437  float maxDist = (maxRotation - minRotation) / 2.0f;
438 
439  if (Math.Abs(targetMidDiff) > maxDist)
440  {
441  targetRotation = (targetMidDiff < 0.0f) ? minRotation : maxRotation;
442  }
443 
444  float degreeOfSuccess = user == null ? 0.5f : DegreeOfSuccess(user);
445  if (degreeOfSuccess < 0.5f) { degreeOfSuccess *= degreeOfSuccess; } //the ease of aiming drops quickly with insufficient skill levels
446  float springStiffness = MathHelper.Lerp(SpringStiffnessLowSkill, SpringStiffnessHighSkill, degreeOfSuccess);
447  float springDamping = MathHelper.Lerp(SpringDampingLowSkill, SpringDampingHighSkill, degreeOfSuccess);
448  float rotationSpeed = MathHelper.Lerp(RotationSpeedLowSkill, RotationSpeedHighSkill, degreeOfSuccess);
449  if (MaxChargeTime > 0)
450  {
451  rotationSpeed *= MathHelper.Lerp(1f, FiringRotationSpeedModifier, MathUtils.EaseIn(currentChargeTime / MaxChargeTime));
452  }
453 
454  // Do not increase the weapons skill when operating a turret in an outpost level
455  if (user?.Info != null && (GameMain.GameSession?.Campaign == null || !Level.IsLoadedOutpost))
456  {
457  user.Info.IncreaseSkillLevel("weapons".ToIdentifier(),
458  SkillSettings.Current.SkillIncreasePerSecondWhenOperatingTurret * deltaTime / Math.Max(user.GetSkillLevel("weapons"), 1.0f));
459  }
460 
461  float rotMidDiff = MathHelper.WrapAngle(rotation - (minRotation + maxRotation) / 2.0f);
462 
463  float targetRotationDiff = MathHelper.WrapAngle(targetRotation - rotation);
464 
465  if ((maxRotation - minRotation) < MathHelper.TwoPi)
466  {
467  float targetRotationMaxDiff = MathHelper.WrapAngle(targetRotation - maxRotation);
468  float targetRotationMinDiff = MathHelper.WrapAngle(targetRotation - minRotation);
469 
470  if (Math.Abs(targetRotationMaxDiff) < Math.Abs(targetRotationMinDiff) &&
471  rotMidDiff < 0.0f &&
472  targetRotationDiff < 0.0f)
473  {
474  targetRotationDiff += MathHelper.TwoPi;
475  }
476  else if (Math.Abs(targetRotationMaxDiff) > Math.Abs(targetRotationMinDiff) &&
477  rotMidDiff > 0.0f &&
478  targetRotationDiff > 0.0f)
479  {
480  targetRotationDiff -= MathHelper.TwoPi;
481  }
482  }
483 
484  angularVelocity +=
485  (targetRotationDiff * springStiffness - angularVelocity * springDamping) * deltaTime;
486  angularVelocity = MathHelper.Clamp(angularVelocity, -rotationSpeed, rotationSpeed);
487 
488  rotation += angularVelocity * deltaTime;
489 
490  rotMidDiff = MathHelper.WrapAngle(rotation - (minRotation + maxRotation) / 2.0f);
491 
492  if (rotMidDiff < -maxDist)
493  {
494  rotation = minRotation;
495  angularVelocity *= -0.5f;
496  }
497  else if (rotMidDiff > maxDist)
498  {
499  rotation = maxRotation;
500  angularVelocity *= -0.5f;
501  }
502 
503  if (aiTargetingGraceTimer > 0f)
504  {
505  aiTargetingGraceTimer -= deltaTime;
506  }
507  if (aiFindTargetTimer > 0.0f)
508  {
509  aiFindTargetTimer -= deltaTime;
510  }
511 
512  UpdateLightComponent();
513  }

546  private bool TryLaunch(float deltaTime, Character character = null, bool ignorePower = false)
547  {
548  tryingToCharge = true;
549  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return false; }
550 
551  if (currentChargeTime < MaxChargeTime) { return false; }
552 
553  if (reload > 0.0f) { return false; }
554 
555  if (MaxActiveProjectiles >= 0)
556  {
557  activeProjectiles.RemoveAll(it => it.Removed);
558  if (activeProjectiles.Count >= MaxActiveProjectiles)
559  {
560  return false;
561  }
562  }
563 
564  if (!ignorePower)
565  {
566  if (!HasPowerToShoot())
567  {
568 #if CLIENT
569  if (!flashLowPower && character != null && character == Character.Controlled)
570  {
571  flashLowPower = true;
572  SoundPlayer.PlayUISound(GUISoundType.PickItemFail);
573  }
574 #endif
575  return false;
576  }
577  }

579  Projectile launchedProjectile = null;
580  bool loaderBroken = false;
581  float tinkeringStrength = 0f;
582 
583  for (int i = 0; i < ProjectileCount; i++)
584  {
585  var projectiles = GetLoadedProjectiles();
586  if (projectiles.Any())
587  {
588  ItemContainer projectileContainer = projectiles.First().Item.Container?.GetComponent<ItemContainer>();
589  if (projectileContainer != null && projectileContainer.Item != item)
590  {
591  projectileContainer?.Item.Use(deltaTime, null);
592  }
593  }
594  else
595  {
596  for (int j = 0; j < item.linkedTo.Count; j++)
597  {
598  var e = item.linkedTo[(j + currentLoaderIndex) % item.linkedTo.Count];
599  //use linked projectile containers in case they have to react to the turret being launched somehow
600  //(play a sound, spawn more projectiles)
601  if (!(e is Item linkedItem)) { continue; }
602  if (!item.Prefab.IsLinkAllowed(e.Prefab)) { continue; }
603  if (linkedItem.Condition <= 0.0f)
604  {
605  loaderBroken = true;
606  continue;
607  }
608  ItemContainer projectileContainer = linkedItem.GetComponent<ItemContainer>();
609  if (projectileContainer != null)
610  {
611  linkedItem.Use(deltaTime, null);
612  projectiles = GetLoadedProjectiles();
613  if (projectiles.Any()) { break; }
614  }
615  }
616  }
617  if (projectiles.Count == 0 && !LaunchWithoutProjectile)
618  {
619  //coilguns spawns ammo in the ammo boxes with the OnUse statuseffect when the turret is launched,
620  //causing a one frame delay before the gun can be launched (or more in multiplayer where there may be a longer delay)
621  // -> attempt to launch the gun multiple times before showing the "no ammo" flash
622  failedLaunchAttempts++;
623 #if CLIENT
624  if (!flashNoAmmo && !flashLoaderBroken && character != null && character == Character.Controlled && failedLaunchAttempts > 20)
625  {
626  if (loaderBroken)
627  {
628  flashLoaderBroken = true;
629  }
630  else
631  {
632  flashNoAmmo = true;
633  }
634  failedLaunchAttempts = 0;
635  SoundPlayer.PlayUISound(GUISoundType.PickItemFail);
636  }
637 #endif
638  return false;
639  }
640  failedLaunchAttempts = 0;
641 
642  foreach (MapEntity e in item.linkedTo)
643  {
644  if (!(e is Item linkedItem)) { continue; }
645  if (!((MapEntity)item).Prefab.IsLinkAllowed(e.Prefab)) { continue; }
646  if (linkedItem.GetComponent<Repairable>() is Repairable repairable && repairable.IsTinkering && linkedItem.HasTag("turretammosource"))
647  {
648  tinkeringStrength = repairable.TinkeringStrength;
649  }
650  }
651 
652  if (!ignorePower)
653  {
654  List<PowerContainer> batteries = GetConnectedBatteries();
655  float neededPower = GetPowerRequiredToShoot();
656 
657  // tinkering is currently not factored into the common method as it is checked only when shooting
658  // but this is a minor issue that causes mostly cosmetic woes. might still be worth refactoring later
659  neededPower /= 1f + (tinkeringStrength * TinkeringPowerCostReduction);
660 
661  while (neededPower > 0.0001f && batteries.Count > 0)
662  {
663  batteries.RemoveAll(b => b.Charge <= 0.0001f || b.MaxOutPut <= 0.0001f);
664  float takePower = neededPower / batteries.Count;
665  takePower = Math.Min(takePower, batteries.Min(b => Math.Min(b.Charge * 3600.0f, b.MaxOutPut)));
666  foreach (PowerContainer battery in batteries)
667  {
668  neededPower -= takePower;
669  battery.Charge -= takePower / 3600.0f;
670 #if SERVER
671  battery.Item.CreateServerEvent(battery);
672 #endif
673  }
674  }
675  }

677  launchedProjectile = projectiles.FirstOrDefault();
678  Item container = launchedProjectile?.Item.Container;
679  if (container != null)
680  {
681  var repairable = launchedProjectile?.Item.Container.GetComponent<Repairable>();
682  if (repairable != null)
683  {
684  repairable.LastActiveTime = (float)Timing.TotalTime + 1.0f;
685  }
686  }

691  {
692  foreach (Projectile projectile in projectiles)
693  {
694  Launch(projectile.Item, character, tinkeringStrength: tinkeringStrength);
695  }
696  }
697  else
698  {
699  Launch(null, character, tinkeringStrength: tinkeringStrength);
700  }
701  if (item.AiTarget != null)
702  {
703  item.AiTarget.SoundRange = item.AiTarget.MaxSoundRange;
704  // Turrets also have a light component, which handles the sight range.
705  }
706  if (container != null)
707  {
708  ShiftItemsInProjectileContainer(container.GetComponent<ItemContainer>());
709  }
710  if (item.linkedTo.Count > 0)
711  {
712  currentLoaderIndex = (currentLoaderIndex + 1) % item.linkedTo.Count;
713  }
714  }
715  }
716 
717 #if SERVER
718  if (character != null && launchedProjectile != null)
719  {
720  string msg = GameServer.CharacterLogName(character) + " launched " + item.Name + " (projectile: " + launchedProjectile.Item.Name;
721  var containedItems = launchedProjectile.Item.ContainedItems;
