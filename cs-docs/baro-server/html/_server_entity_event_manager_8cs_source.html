114  public void CreateEvent(IServerSerializable entity, NetEntityEvent.IData extraData = null)
115  {
116  if (!ValidateEntity(entity)) { return; }
117 
118  var newEvent = new ServerEntityEvent(entity, (UInt16)(ID + 1));
119  if (extraData != null) newEvent.SetData(extraData);
120 
121  bool inGameClientsPresent = server.ConnectedClients.Count(c => c.InGame) > 0;
122 
123  //remove old events that have been sent to all clients, they are redundant now
124  // keep at least one event in the list (lastSentToAll == e.ID) so we can use it to keep track of the latest ID
125  // and events less than 15 seconds old to give disconnected clients a bit of time to reconnect without getting desynced
126  if (Timing.TotalTime > GameMain.GameSession.RoundStartTime + NetConfig.RoundStartSyncDuration)
127  {
128  events.RemoveAll(e =>
129  (NetIdUtils.IdMoreRecent(lastSentToAll, e.ID) || !inGameClientsPresent) &&
130  e.CreateTime < Timing.TotalTime - NetConfig.EventRemovalTime);
131  }
132 
133  for (int i = events.Count - 1; i >= 0; i--)
134  {
135  //we already have an identical event that's waiting to be sent
136  // -> no need to add a new one
137  if (events[i].IsDuplicate(newEvent) && !events[i].Sent) return;
138  }
139 
140  ID++;
141 
142  events.Add(newEvent);
143 
144  if (!uniqueEvents.Any(e => e.IsDuplicate(newEvent)))
145  {
146  //create a copy of the event and give it a new ID
147  var uniqueEvent = new ServerEntityEvent(entity, (UInt16)(uniqueEvents.Count + 1));
148  uniqueEvent.SetData(extraData);
149 
150  uniqueEvents.Add(uniqueEvent);

154  public void Update(List<Client> clients)
155  {
156  foreach (BufferedEvent bufferedEvent in bufferedEvents)
157  {
158  if (bufferedEvent.Character == null || bufferedEvent.Character.IsDead)
159  {
160  bufferedEvent.IsProcessed = true;
161  continue;
162  }
163 
164  //delay reading the events until the inputs for the corresponding frame have been processed
165 
166  //UNLESS the character is unconscious, in which case we'll read the messages immediately (because further inputs will be ignored)
167  //atm the "give in" command is the only thing unconscious characters can do, other types of events are ignored
168  if (!bufferedEvent.Character.IsIncapacitated &&
169  NetIdUtils.IdMoreRecent(bufferedEvent.CharacterStateID, bufferedEvent.Character.LastProcessedID))
170  {
171  continue;
172  }
173 
174  try
175  {
176  ReadEvent(bufferedEvent.Data, bufferedEvent.TargetEntity, bufferedEvent.Sender);
177  }
178 
179  catch (Exception e)
180  {
181  string entityName = bufferedEvent.TargetEntity == null ? "null" : bufferedEvent.TargetEntity.ToString();
182  if (GameSettings.CurrentConfig.VerboseLogging)
183  {
184  string errorMsg = "Failed to read server event for entity \"" + entityName + "\"!";
185  GameServer.Log(errorMsg + "\n" + e.StackTrace.CleanupStackTrace(), ServerLog.MessageType.Error);
186  DebugConsole.ThrowError(errorMsg, e);
187  }
188  GameAnalyticsManager.AddErrorEventOnce("ServerEntityEventManager.Read:ReadFailed" + entityName,
189  GameAnalyticsManager.ErrorSeverity.Error,
190  "Failed to read server event for entity \"" + entityName + "\"!\n" + e.StackTrace.CleanupStackTrace());
191  }
192 
193  bufferedEvent.IsProcessed = true;
194  }
195 
196  var inGameClients = clients.FindAll(c => c.InGame && !c.NeedsMidRoundSync);
197  if (inGameClients.Count > 0)
198  {
199  lastSentToAnyone = inGameClients[0].LastRecvEntityEventID;
200  lastSentToAll = inGameClients[0].LastRecvEntityEventID;
201 
202  if (server.OwnerConnection != null)
203  {
204  var owner = clients.Find(c => c.Connection == server.OwnerConnection);
205  if (owner != null)
206  {
207  lastSentToAll = owner.LastRecvEntityEventID;
208  }
209  }
210  inGameClients.ForEach(c =>
211  {
212  if (NetIdUtils.IdMoreRecent(lastSentToAll, c.LastRecvEntityEventID)) { lastSentToAll = c.LastRecvEntityEventID; }
213  if (NetIdUtils.IdMoreRecent(c.LastRecvEntityEventID, lastSentToAnyone)) { lastSentToAnyone = c.LastRecvEntityEventID; }
214  });
215  lastSentToAnyoneTime = events.Find(e => e.ID == lastSentToAnyone)?.CreateTime ?? Timing.TotalTime;
216 
217  if (Timing.TotalTime - lastWarningTime > 5.0 &&
218  Timing.TotalTime - lastSentToAnyoneTime > 10.0 &&
219  Timing.TotalTime > GameMain.GameSession.RoundStartTime + NetConfig.RoundStartSyncDuration)
220  {
221  lastWarningTime = Timing.TotalTime;
222  GameServer.Log("WARNING: ServerEntityEventManager is lagging behind! Last sent id: " + lastSentToAnyone.ToString() + ", latest create id: " + ID.ToString(), ServerLog.MessageType.ServerMessage);
223  events.ForEach(e => e.ResetCreateTime());
224  //TODO: reset clients if this happens, maybe do it if a majority are behind rather than all of them?
225  }
226 
227  clients.Where(c => c.NeedsMidRoundSync).ForEach(c => { if (NetIdUtils.IdMoreRecent(lastSentToAll, c.FirstNewEventID)) lastSentToAll = (ushort)(c.FirstNewEventID - 1); });
228 
229  ServerEntityEvent firstEventToResend = events.Find(e => e.ID == (ushort)(lastSentToAll + 1));
230  if (firstEventToResend != null &&
231  Timing.TotalTime > GameMain.GameSession.RoundStartTime + NetConfig.RoundStartSyncDuration &&
232  ((lastSentToAnyoneTime - firstEventToResend.CreateTime) > NetConfig.OldReceivedEventKickTime || (Timing.TotalTime - firstEventToResend.CreateTime) > NetConfig.OldEventKickTime))
233  {
234  // This event is 10 seconds older than the last one we've successfully sent,
235  // kick everyone that hasn't received it yet, this is way too old
236  // UNLESS the event was created when the client was still midround syncing,
237  // in which case we'll wait until the timeout runs out before kicking the client
238  List<Client> toKick = inGameClients.FindAll(c =>
239  NetIdUtils.IdMoreRecent((UInt16)(lastSentToAll + 1), c.LastRecvEntityEventID) &&
240  (firstEventToResend.CreateTime > c.MidRoundSyncTimeOut || lastSentToAnyoneTime > c.MidRoundSyncTimeOut || Timing.TotalTime > c.MidRoundSyncTimeOut + 10.0));
241  toKick.ForEach(c =>
242  {
243  DebugConsole.NewMessage(c.Name + " was kicked because they were expecting a very old network event (" + (c.LastRecvEntityEventID + 1).ToString() + ")", Color.Red);
244  GameServer.Log(GameServer.ClientLogName(c) + " was kicked because they were expecting a very old network event ("
245  + (c.LastRecvEntityEventID + 1).ToString() +
246  " (created " + (Timing.TotalTime - firstEventToResend.CreateTime).ToString("0.##") + " s ago, " +
247  (lastSentToAnyoneTime - firstEventToResend.CreateTime).ToString("0.##") + " s older than last event sent to anyone)" +
248  " Events queued: " + events.Count + ", last sent to all: " + lastSentToAll, ServerLog.MessageType.Error);
249  server.DisconnectClient(c, "", DisconnectReason.ExcessiveDesyncOldEvent + "/ServerMessage.ExcessiveDesyncOldEvent");
250  }
251  );
252  }
253 
254  if (events.Count > 0)
255  {
256  //the client is waiting for an event that we don't have anymore
257  //(the ID they're expecting is smaller than the ID of the first event in our list)
258  List<Client> toKick = inGameClients.FindAll(c => NetIdUtils.IdMoreRecent(events[0].ID, (UInt16)(c.LastRecvEntityEventID + 1)));
259  toKick.ForEach(c =>
260  {
261  DebugConsole.NewMessage(c.Name + " was kicked because they were expecting a removed network event (" + (c.LastRecvEntityEventID + 1).ToString() + ", last available is " + events[0].ID.ToString() + ")", Color.Red);
262  GameServer.Log(GameServer.ClientLogName(c) + " was kicked because they were expecting a removed network event (" + (c.LastRecvEntityEventID + 1).ToString() + ", last available is " + events[0].ID.ToString() + ")", ServerLog.MessageType.Error);
263  server.DisconnectClient(c, "", DisconnectReason.ExcessiveDesyncRemovedEvent + "/ServerMessage.ExcessiveDesyncRemovedEvent");
264  });
265  }

305  public void Write(Client client, IWriteMessage msg, out List<NetEntityEvent> sentEvents)
306  {
307  List<NetEntityEvent> eventsToSync = GetEventsToSync(client);
308 
309  if (eventsToSync.Count == 0)
310  {
311  sentEvents = eventsToSync;
312  return;
313  }
314 
315  //too many events for one packet
316  //(normal right after a round has just started, don't show a warning if it's been less than 10 seconds)
317  if (eventsToSync.Count > 200 && GameMain.GameSession != null && Timing.TotalTime > GameMain.GameSession.RoundStartTime + 10.0)
318  {
319  if (eventsToSync.Count > 200 && !client.NeedsMidRoundSync && Timing.TotalTime > lastEventCountHighWarning + 2.0)
320  {
321  Color color = eventsToSync.Count > 500 ? Color.Red : Color.Orange;
322  if (eventsToSync.Count < 300) { color = Color.Yellow; }
323  string warningMsg = "WARNING: event count very high: " + eventsToSync.Count;
324 
325  var sortedEvents = eventsToSync.GroupBy(e => e.Entity.ToString())
326  .Select(e => new { Value = e.Key, Count = e.Count() })
327  .OrderByDescending(e => e.Count);
328 
329  int count = 1;
330  foreach (var sortedEvent in sortedEvents)
331  {
332  warningMsg += "\n" + count + ". " + (sortedEvent.Value?.ToString() ?? "null") + " x" + sortedEvent.Count;
333  count++;
334  if (count > 3) { break; }
335  }
336  if (GameSettings.CurrentConfig.VerboseLogging)
337  {
338  GameServer.Log(warningMsg, ServerLog.MessageType.Error);
339  }
340  DebugConsole.NewMessage(warningMsg, color);
341  lastEventCountHighWarning = Timing.TotalTime;
342  }
343  }
344 
345  if (client.NeedsMidRoundSync)
346  {
347  msg.Write((byte)ServerNetObject.ENTITY_EVENT_INITIAL);
348  msg.Write(client.UnreceivedEntityEventCount);
349  msg.Write(client.FirstNewEventID);
350 
351  Write(msg, eventsToSync, out sentEvents, client);
352  }
353  else
354  {
355  msg.Write((byte)ServerNetObject.ENTITY_EVENT);
356  Write(msg, eventsToSync, out sentEvents, client);
357  }
358 
359  foreach (NetEntityEvent entityEvent in sentEvents)
360  {
361  (entityEvent as ServerEntityEvent).Sent = true;
362  client.EntityEventLastSent[entityEvent.ID] = Lidgren.Network.NetTime.Now;
363  }

369  private List<NetEntityEvent> GetEventsToSync(Client client)
370  {
371  List<NetEntityEvent> eventsToSync = new List<NetEntityEvent>();
372 
373  var eventList = client.NeedsMidRoundSync ? uniqueEvents : events;
374 
375  if (eventList.Count == 0) { return eventsToSync; }
376 
377  //find the index of the first event the client hasn't received
378  int startIndex = eventList.Count;
379  while (startIndex > 0 &&
380  NetIdUtils.IdMoreRecent(eventList[startIndex - 1].ID, client.LastRecvEntityEventID))
381  {
382  startIndex--;
383  }
384 
385  for (int i = startIndex; i < eventList.Count; i++)
386  {
387  //find the first event that hasn't been sent in roundtriptime or at all
388  client.EntityEventLastSent.TryGetValue(eventList[i].ID, out double lastSent);
389 
390  float avgRoundtripTime = 0.01f; //TODO: reimplement client.Connection.AverageRoundtripTime
391  float minInterval = Math.Max(avgRoundtripTime, (float)server.UpdateInterval.TotalSeconds * 2);
392 
393  if (lastSent > Lidgren.Network.NetTime.Now - Math.Min(minInterval, 0.5f))
394  {
395  continue;
396  }
397 
398  if (client.NeedsMidRoundSync)
399  {
400  if (i <= client.UnreceivedEntityEventCount)
401  {
402  eventsToSync.AddRange(eventList.GetRange(i, client.UnreceivedEntityEventCount - i));
403  }
404  }
405  else
406  {
407  eventsToSync.AddRange(eventList.GetRange(i, eventList.Count - i));
408  }
409 
410  break;
411  }
412 
413  return eventsToSync;
414  }

447  public void Read(IReadMessage msg, Client sender = null)
448  {
449  msg.ReadPadBits();
450  UInt16 firstEventID = msg.ReadUInt16();
451  int eventCount = msg.ReadByte();
452 
453  for (int i = 0; i < eventCount; i++)
454  {
455  UInt16 thisEventID = (UInt16)(firstEventID + (UInt16)i);
456  UInt16 entityID = msg.ReadUInt16();
457 
458  if (entityID == Entity.NullEntityID)
459  {
460  if (thisEventID == (UInt16)(sender.LastSentEntityEventID + 1)) sender.LastSentEntityEventID++;
461  continue;
462  }
463 
464  int msgLength = (int)msg.ReadVariableUInt32();
465 
466  IClientSerializable entity = Entity.FindEntityByID(entityID) as IClientSerializable;
467 
468  //skip the event if we've already received it
469  if (thisEventID != (UInt16)(sender.LastSentEntityEventID + 1))
470  {
471  if (GameSettings.CurrentConfig.VerboseLogging)
472  {
473  DebugConsole.NewMessage("Received msg " + thisEventID + ", expecting " + sender.LastSentEntityEventID, Color.Red);
474  }
475  msg.BitPosition += msgLength * 8;
476  }
477  else if (entity == null)
478  {
479  //entity not found -> consider the event read and skip over it
480  //(can happen, for example, when a client uses a medical item repeatedly
481  //and creates an event for it before receiving the event about it being removed)
482  if (GameSettings.CurrentConfig.VerboseLogging)
483  {
484  DebugConsole.NewMessage(
485  "Received msg " + thisEventID + ", entity " + entityID + " not found",
486  Microsoft.Xna.Framework.Color.Orange);
487  }
488  sender.LastSentEntityEventID++;
489  msg.BitPosition += msgLength * 8;
490  }
491  else
492  {
493  if (GameSettings.CurrentConfig.VerboseLogging)
494  {
495  DebugConsole.NewMessage("Received msg " + thisEventID, Microsoft.Xna.Framework.Color.Green);
496  }
497 
498  UInt16 characterStateID = msg.ReadUInt16();
499 
500  ReadWriteMessage buffer = new ReadWriteMessage();
501  byte[] temp = msg.ReadBytes(msgLength - 2);
502  buffer.Write(temp, 0, msgLength - 2);
503  buffer.BitPosition = 0;
504  BufferEvent(new BufferedEvent(sender, sender.Character, characterStateID, entity, buffer));
505 
506  sender.LastSentEntityEventID++;
507  }
508  }
