174  public void ServerRead(IReadMessage inc, Client sender)
175  {
176  if (GameMain.Server == null || sender == null) { return; }
177 
178  byte voteTypeByte = inc.ReadByte();
179  VoteType voteType = VoteType.Unknown;
180  try
181  {
182  voteType = (VoteType)voteTypeByte;
183  }
184  catch (Exception e)
185  {
186  DebugConsole.ThrowError("Failed to cast vote type \"" + voteTypeByte + "\"", e);
187  return;
188  }
189 
190  switch (voteType)
191  {
192  case VoteType.Sub:
193  int equalityCheckVal = inc.ReadInt32();
194  string hash = equalityCheckVal > 0 ? string.Empty : inc.ReadString();
195  SubmarineInfo sub = equalityCheckVal > 0 ?
196  SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Type == SubmarineType.Player && s.EqualityCheckVal == equalityCheckVal) :
197  SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Type == SubmarineType.Player && s.MD5Hash.StringRepresentation == hash);
198  sender.SetVote(voteType, sub);
199  break;
200  case VoteType.Mode:
201  string modeIdentifier = inc.ReadString();
202  GameModePreset mode = GameModePreset.List.Find(gm => gm.Identifier == modeIdentifier);
203  if (mode == null || !mode.Votable) { break; }
204  sender.SetVote(voteType, mode);
205  break;
206  case VoteType.EndRound:
207  if (!sender.HasSpawned) { return; }
208  sender.SetVote(voteType, inc.ReadBoolean());
209  break;
210  case VoteType.Kick:
211  byte kickedClientID = inc.ReadByte();
212 
213  Client kicked = GameMain.Server.ConnectedClients.Find(c => c.ID == kickedClientID);
214  if (kicked != null && kicked.Connection != GameMain.Server.OwnerConnection && !kicked.HasKickVoteFrom(sender))
215  {
216  kicked.AddKickVote(sender);
217  Client.UpdateKickVotes(GameMain.Server.ConnectedClients);
218  GameMain.Server.SendChatMessage($"ServerMessage.HasVotedToKick~[initiator]={sender.Name}~[target]={kicked.Name}", ChatMessageType.Server, null);
219  }
220 
221  break;
222  case VoteType.StartRound:
223  bool ready = inc.ReadBoolean();
224  if (ready != sender.GetVote<bool>(VoteType.StartRound))
225  {
226  sender.SetVote(VoteType.StartRound, ready);
227  GameServer.Log(GameServer.ClientLogName(sender) + (ready ? " is ready to start the game." : " is not ready to start the game."), ServerLog.MessageType.ServerMessage);
228  }
229  break;
230 
231  case VoteType.PurchaseAndSwitchSub:
232  case VoteType.PurchaseSub:
233  case VoteType.SwitchSub:
234  case VoteType.TransferMoney:
235  bool startVote = inc.ReadBoolean();
236  if (startVote)
237  {
238  if (voteType == VoteType.TransferMoney)
239  {
240  int amount = inc.ReadInt32();
241  int fromClientId = inc.ReadByte();
242  int toClientId = inc.ReadByte();
243  pendingVotes.Enqueue(new TransferVote(sender,
244  GameMain.Server.ConnectedClients.Find(c => c.ID == fromClientId),
245  amount,
246  GameMain.Server.ConnectedClients.Find(c => c.ID == toClientId)));
247  }
248  else
249  {
250  string subName = inc.ReadString();
251  SubmarineInfo subInfo = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Name == subName);
252  if (GameMain.GameSession?.Campaign is MultiPlayerCampaign campaign && (campaign.CanPurchaseSub(subInfo, sender) || GameMain.GameSession.IsSubmarineOwned(subInfo)))
253  {
254  StartSubmarineVote(subInfo, voteType, sender);
255  }
256  }
257  }
258  else
259  {
260  sender.SetVote(voteType, (int)inc.ReadByte());
261  }
262  break;
263  }
264 
265  inc.ReadPadBits();
266 
267  GameMain.Server.UpdateVoteStatus();
268  }

270  public void ServerWrite(IWriteMessage msg)
271  {
272  if (GameMain.Server == null) { return; }
273 
274  msg.Write(GameMain.Server.ServerSettings.AllowSubVoting);
275  if (GameMain.Server.ServerSettings.AllowSubVoting)
276  {
277  IReadOnlyDictionary<SubmarineInfo, int> voteList = GetVoteCounts<SubmarineInfo>(VoteType.Sub, GameMain.Server.ConnectedClients);
278  msg.Write((byte)voteList.Count);
279  foreach (KeyValuePair<SubmarineInfo, int> vote in voteList)
280  {
281  msg.Write((byte)vote.Value);
282  msg.Write(vote.Key.Name);
283  }
284  }
285  msg.Write(GameMain.Server.ServerSettings.AllowModeVoting);
286  if (GameMain.Server.ServerSettings.AllowModeVoting)
287  {
288  IReadOnlyDictionary<GameModePreset, int> voteList = GetVoteCounts<GameModePreset>(VoteType.Mode, GameMain.Server.ConnectedClients);
289  msg.Write((byte)voteList.Count);
290  foreach (KeyValuePair<GameModePreset, int> vote in voteList)
291  {
292  msg.Write((byte)vote.Value);
293  msg.Write(vote.Key.Identifier);
294  }
295  }
296  msg.Write(GameMain.Server.ServerSettings.AllowEndVoting);
297  if (GameMain.Server.ServerSettings.AllowEndVoting)
298  {
299  msg.Write((byte)GameMain.Server.ConnectedClients.Count(c => c.HasSpawned && c.GetVote<bool>(VoteType.EndRound)));
300  msg.Write((byte)GameMain.Server.ConnectedClients.Count(c => c.HasSpawned));
301  }
302 
303  msg.Write(GameMain.Server.ServerSettings.AllowVoteKick);
304 
305  msg.Write((byte)(ActiveVote?.State ?? VoteState.None));
306  if (ActiveVote != null)
307  {
308  msg.Write((byte)ActiveVote.VoteType);
309  if (ActiveVote.State != VoteState.None && ActiveVote.VoteType != VoteType.Unknown)
310  {
311  var yesClients = GameMain.Server.ConnectedClients.FindAll(c => c.InGame && c.GetVote<int>(ActiveVote.VoteType) == 2);
312  msg.Write((byte)yesClients.Count);
313  foreach (Client c in yesClients)
314  {
315  msg.Write(c.ID);
316  }
317 
318  var noClients = GameMain.Server.ConnectedClients.FindAll(c => c.InGame && c.GetVote<int>(ActiveVote.VoteType) == 1);
319  msg.Write((byte)noClients.Count);
320  foreach (Client c in noClients)
321  {
322  msg.Write(c.ID);
323  }
324 
325  msg.Write((byte)GameMain.Server.ConnectedClients.Count(c => c.InGame));
326 
327  switch (ActiveVote.State)
328  {
329  case VoteState.Started:
330  msg.Write(ActiveVote.VoteStarter.ID);
331  msg.Write((byte)GameMain.Server.ServerSettings.VoteTimeout);
332 
333  switch (ActiveVote.VoteType)
334  {
335  case VoteType.PurchaseSub:
336  case VoteType.PurchaseAndSwitchSub:
337  case VoteType.SwitchSub:
338  msg.Write((ActiveVote as SubmarineVote).Sub.Name);
339  break;
340  case VoteType.TransferMoney:
341  var transferVote = (ActiveVote as TransferVote);
342  msg.Write(transferVote.From?.ID ?? 0);
343  msg.Write(transferVote.To?.ID ?? 0);
344  msg.Write(transferVote.TransferAmount);
345  break;
346  }
347 
348  break;
349  case VoteState.Running:
350  // Nothing specific
351  break;
352  case VoteState.Passed:
353  case VoteState.Failed:
354  msg.Write(ActiveVote.State == VoteState.Passed);
355  switch (ActiveVote.VoteType)
356  {
357  case VoteType.PurchaseSub:
358  case VoteType.PurchaseAndSwitchSub:
359  case VoteType.SwitchSub:
360  msg.Write((ActiveVote as SubmarineVote).Sub.Name);
361  msg.Write((short)(ActiveVote as SubmarineVote).DeliveryFee);
362  break;
363  }
364  break;
365  }
366  }
367  }
368 
369  var readyClients = GameMain.Server.ConnectedClients.FindAll(c => c.GetVote<bool>(VoteType.StartRound));
370  msg.Write((byte)readyClients.Count);
371  foreach (Client c in readyClients)
372  {
373  msg.Write(c.ID);
374  }
375 
376  msg.WritePadBits();
377  }
378  }
