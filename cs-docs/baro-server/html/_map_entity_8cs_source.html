257 protected void ParseLinks(XElement element, IdRemap idRemap)
258 {
259 string linkedToString = element.GetAttributeString("linked", "");
260 if (!string.IsNullOrEmpty(linkedToString))
261 {
262 string[] linkedToIds = linkedToString.Split(',');
263 for (int i = 0; i < linkedToIds.Length; i++)
264 {
265 int srcId = int.Parse(linkedToIds[i]);
266 int targetId = idRemap.GetOffsetId(srcId);
267 if (targetId <= 0)
268 {
269 unresolvedLinkedToID ??= new List<ushort>();
270 unresolvedLinkedToID.Add((ushort)srcId);
271 continue;
272 }
273 linkedToID.Add((ushort)targetId);
274 }
275 }
276 }

278 public void ResolveLinks(IdRemap childRemap)
279 {
280 if (unresolvedLinkedToID == null) { return; }
281 for (int i = 0; i < unresolvedLinkedToID.Count; i++)
282 {
283 int srcId = unresolvedLinkedToID[i];
284 int targetId = childRemap.GetOffsetId(srcId);
285 if (targetId > 0)
286 {
287 var otherEntity = FindEntityByID((ushort)targetId) as MapEntity;
288 linkedTo.Add(otherEntity);
289 if (otherEntity.Linkable && otherEntity.linkedTo != null) otherEntity.linkedTo.Add(this);
290 unresolvedLinkedToID.RemoveAt(i);
291 i--;
292 }
293 }

379 public static List<MapEntity> Clone(List<MapEntity> entitiesToClone)
380 {
381 List<MapEntity> clones = new List<MapEntity>();
382 foreach (MapEntity e in entitiesToClone)
383 {
384 Debug.Assert(e != null);
385 try
386 {
387 clones.Add(e.Clone());
388 }
389 catch (Exception ex)
390 {
391 DebugConsole.ThrowError("Cloning entity \"" + e.Name + "\" failed.", ex);
392 GameAnalyticsManager.AddErrorEventOnce(
393 "MapEntity.Clone:" + e.Name,
394 GameAnalyticsManager.ErrorSeverity.Error,
395 "Cloning entity \"" + e.Name + "\" failed (" + ex.Message + ").\n" + ex.StackTrace.CleanupStackTrace());
396 return clones;
397 }
398 Debug.Assert(clones.Last() != null);
399 }
400
401 Debug.Assert(clones.Count == entitiesToClone.Count);
402
403 //clone links between the entities
404 for (int i = 0; i < clones.Count; i++)
405 {
406 if (entitiesToClone[i].linkedTo == null) { continue; }
407 foreach (MapEntity linked in entitiesToClone[i].linkedTo)
408 {
409 if (!entitiesToClone.Contains(linked)) { continue; }
410 clones[i].linkedTo.Add(clones[entitiesToClone.IndexOf(linked)]);
411 }
412 }
413
414 //connect clone wires to the clone items and refresh links between doors and gaps
415 List<Wire> orphanedWires = new List<Wire>();
416 for (int i = 0; i < clones.Count; i++)
417 {
418 if (!(clones[i] is Item cloneItem)) { continue; }
419
420 var door = cloneItem.GetComponent<Door>();
421 door?.RefreshLinkedGap();
422
423 var cloneWire = cloneItem.GetComponent<Wire>();
424 if (cloneWire == null) { continue; }
425
426 var originalWire = ((Item)entitiesToClone[i]).GetComponent<Wire>();
427
428 cloneWire.SetNodes(originalWire.GetNodes());
429
430 for (int n = 0; n < 2; n++)
431 {
432 if (originalWire.Connections[n] == null)
433 {
434 var disconnectedFrom = entitiesToClone.Find(e => e is Item item && (item.GetComponent<ConnectionPanel>()?.DisconnectedWires.Contains(originalWire) ?? false));
435 if (disconnectedFrom == null) { continue; }
436
437 int disconnectedFromIndex = entitiesToClone.IndexOf(disconnectedFrom);
438 var disconnectedFromClone = (clones[disconnectedFromIndex] as Item)?.GetComponent<ConnectionPanel>();
439 if (disconnectedFromClone == null) { continue; }
440
441 disconnectedFromClone.DisconnectedWires.Add(cloneWire);
442 if (cloneWire.Item.body != null) { cloneWire.Item.body.Enabled = false; }
443 cloneWire.IsActive = false;
444 continue;
445 }
446
447 var connectedItem = originalWire.Connections[n].Item;
448 if (connectedItem == null || !entitiesToClone.Contains(connectedItem)) { continue; }
449
450 //index of the item the wire is connected to
451 int itemIndex = entitiesToClone.IndexOf(connectedItem);
452 if (itemIndex < 0)
453 {
454 DebugConsole.ThrowError("Error while cloning wires - item \"" + connectedItem.Name + "\" was not found in entities to clone.");
455 GameAnalyticsManager.AddErrorEventOnce("MapEntity.Clone:ConnectedNotFound" + connectedItem.ID,
456 GameAnalyticsManager.ErrorSeverity.Error,
457 "Error while cloning wires - item \"" + connectedItem.Name + "\" was not found in entities to clone.");
458 continue;
459 }
460
461 //index of the connection in the connectionpanel of the target item
462 int connectionIndex = connectedItem.Connections.IndexOf(originalWire.Connections[n]);
463 if (connectionIndex < 0)
464 {
465 DebugConsole.ThrowError("Error while cloning wires - connection \"" + originalWire.Connections[n].Name + "\" was not found in connected item \"" + connectedItem.Name + "\".");
466 GameAnalyticsManager.AddErrorEventOnce("MapEntity.Clone:ConnectionNotFound" + connectedItem.ID,
467 GameAnalyticsManager.ErrorSeverity.Error,
468 "Error while cloning wires - connection \"" + originalWire.Connections[n].Name + "\" was not found in connected item \"" + connectedItem.Name + "\".");
469 continue;
470 }
471
472 (clones[itemIndex] as Item).Connections[connectionIndex].TryAddLink(cloneWire);
473 cloneWire.Connect((clones[itemIndex] as Item).Connections[connectionIndex], false);
474 }
475
476 if ((cloneWire.Connections[0] == null || cloneWire.Connections[1] == null) && cloneItem.GetComponent<DockingPort>() == null)
477 {
478 if (!clones.Any(c => (c as Item)?.GetComponent<ConnectionPanel>()?.DisconnectedWires.Contains(cloneWire) ?? false))
479 {
480 orphanedWires.Add(cloneWire);
481 }
482 }
483 }
484
485 foreach (var orphanedWire in orphanedWires)
486 {
487 orphanedWire.Item.Remove();
488 clones.Remove(orphanedWire.Item);
489 }
490
491 return clones;
492 }

562 public static void UpdateAll(float deltaTime, Camera cam)
563 {
564 mapEntityUpdateTick++;
565
566 if (mapEntityUpdateTick % MapEntityUpdateInterval == 0)
567 {
568
569 foreach (Hull hull in Hull.HullList)
570 {
571 hull.Update(deltaTime * MapEntityUpdateInterval, cam);
572 }
573#if CLIENT
574 Hull.UpdateCheats(deltaTime * MapEntityUpdateInterval, cam);
575#endif
576
577 foreach (Structure structure in Structure.WallList)
578 {
579 structure.Update(deltaTime * MapEntityUpdateInterval, cam);
580 }
581 }
582
583 //update gaps in random order, because otherwise in rooms with multiple gaps
584 //the water/air will always tend to flow through the first gap in the list,
585 //which may lead to weird behavior like water draining down only through
586 //one gap in a room even if there are several
587 if (mapEntityUpdateTick % GapUpdateInterval == 0)
588 {
589 foreach (Gap gap in Gap.GapList.OrderBy(g => Rand.Int(int.MaxValue)))
590 {
591 gap.Update(deltaTime * GapUpdateInterval, cam);
592 }
593 }
594
595 if (mapEntityUpdateTick % PoweredUpdateInterval == 0)
596 {
597 Powered.UpdatePower(deltaTime * PoweredUpdateInterval);
598 }
599
600 if (mapEntityUpdateTick % MapEntityUpdateInterval == 0)
601 {
602 foreach (Item item in Item.ItemList)
603 {
604 if (GameMain.LuaCs.Game.updatePriorityItems.Contains(item)) continue;
605 item.Update(deltaTime * MapEntityUpdateInterval, cam);
606 }
607 }
608
609 foreach (var item in GameMain.LuaCs.Game.updatePriorityItems)
610 {
611 if (item.Removed) continue;
612
613 item.Update(deltaTime, cam);
614 }
615
616 if (mapEntityUpdateTick % MapEntityUpdateInterval == 0)
617 {
618
619 UpdateAllProjSpecific(deltaTime * MapEntityUpdateInterval);
620
621 Spawner?.Update();
622 }

657 public static List<MapEntity> LoadAll(Submarine submarine, XElement parentElement, string filePath, int idOffset)
658 {
659 IdRemap idRemap = new IdRemap(parentElement, idOffset);
660
661 List<MapEntity> entities = new List<MapEntity>();
662 foreach (var element in parentElement.Elements())
663 {
664 string typeName = element.Name.ToString();
665
666 Type t;
667 try
668 {
669 t = Type.GetType("Barotrauma." + typeName, true, true);
670 if (t == null)
671 {
672 DebugConsole.ThrowError("Error in " + filePath + "! Could not find a entity of the type \"" + typeName + "\".");
673 continue;
674 }
675 }
676 catch (Exception e)
677 {
678 DebugConsole.ThrowError("Error in " + filePath + "! Could not find a entity of the type \"" + typeName + "\".", e);
679 continue;
680 }
681
682 if (t == typeof(Structure))
683 {
684 string name = element.Attribute("name").Value;
685 Identifier identifier = element.GetAttributeIdentifier("identifier", "");
686 StructurePrefab structurePrefab = Structure.FindPrefab(name, identifier);
687 if (structurePrefab == null)
688 {
689 ItemPrefab itemPrefab = ItemPrefab.Find(name, identifier);
690 if (itemPrefab != null)
691 {
692 t = typeof(Item);
693 }
694 }
695 }
696
697 try
698 {
699 MethodInfo loadMethod = t.GetMethod("Load", new[] { typeof(ContentXElement), typeof(Submarine), typeof(IdRemap) });
700 if (loadMethod == null)
701 {
702 DebugConsole.ThrowError("Could not find the method \"Load\" in " + t + ".");
703 }
704 else if (!loadMethod.ReturnType.IsSubclassOf(typeof(MapEntity)))
705 {
706 DebugConsole.ThrowError("Error loading entity of the type \"" + t.ToString() + "\" - load method does not return a valid map entity.");
707 }
708 else
709 {
710 object newEntity = loadMethod.Invoke(t, new object[] { element.FromPackage(null), submarine, idRemap });
711 if (newEntity != null)
712 {
713 entities.Add((MapEntity)newEntity);
714 }
715 }
716 }
717 catch (TargetInvocationException e)
718 {
719 DebugConsole.ThrowError("Error while loading entity of the type " + t + ".", e.InnerException);
720 }
721 catch (Exception e)
722 {
723 DebugConsole.ThrowError("Error while loading entity of the type " + t + ".", e);
724 }
725 }
726 return entities;
727 }

823 private static void GetLinkedEntitiesRecursive<T>(MapEntity mapEntity, HashSet<T> linkedTargets, ref int depth, int? maxDepth = null, Func<T, bool> filter = null)
824 where T : MapEntity
825 {
826 if (depth > maxDepth) { return; }
827 foreach (var linkedEntity in mapEntity.linkedTo)
828 {
829 if (linkedEntity is T linkedTarget)
830 {
831 if (!linkedTargets.Contains(linkedTarget) && (filter == null || filter(linkedTarget)))
832 {
833 linkedTargets.Add(linkedTarget);
834 depth++;
835 GetLinkedEntitiesRecursive(linkedEntity, linkedTargets, ref depth, maxDepth, filter);
836 }
837 }
