9 public static void ServerRead(IReadMessage msg, Client c)
10 {
11 c.KickAFKTimer = 0.0f;
12
13 UInt16 ID = msg.ReadUInt16();
14 ChatMessageType type = (ChatMessageType)msg.ReadByte();
15 string txt;
16
17 Character orderTargetCharacter = null;
18 Entity orderTargetEntity = null;
19 OrderChatMessage orderMsg = null;
20 Order.OrderTargetType orderTargetType = Order.OrderTargetType.Entity;
21 int? wallSectionIndex = null;
22 Order order = null;
23 bool isNewOrder = false;
24 if (type == ChatMessageType.Order)
25 {
26 var orderMessageInfo = OrderChatMessage.ReadOrder(msg);
27 if (orderMessageInfo.OrderIdentifier == Identifier.Empty)
28 {
29 DebugConsole.ThrowError($"Invalid order message from client \"{c.Name}\" - order identifier is empty.");
30 if (NetIdUtils.IdMoreRecent(ID, c.LastSentChatMsgID)) { c.LastSentChatMsgID = ID; }
31 return;
32 }
33 isNewOrder = orderMessageInfo.IsNewOrder;
34 orderTargetCharacter = orderMessageInfo.TargetCharacter;
35 orderTargetEntity = orderMessageInfo.TargetEntity;
36 OrderTarget orderTargetPosition = orderMessageInfo.TargetPosition;
37 orderTargetType = orderMessageInfo.TargetType;
38 wallSectionIndex = orderMessageInfo.WallSectionIndex;
39 var orderPrefab = orderMessageInfo.OrderPrefab ?? OrderPrefab.Prefabs[orderMessageInfo.OrderIdentifier];
40 Identifier orderOption = orderMessageInfo.OrderOption;
41 if (orderOption.IsEmpty)
42 {
43 orderOption = orderMessageInfo.OrderOptionIndex == null || orderMessageInfo.OrderOptionIndex < 0 || orderMessageInfo.OrderOptionIndex >= orderPrefab.Options.Length ?
44 Identifier.Empty : orderPrefab.Options[orderMessageInfo.OrderOptionIndex.Value];
45 }
46 if (orderTargetType == Order.OrderTargetType.Position)
47 {
48 order = new Order(orderPrefab, orderOption, orderTargetPosition, orderGiver: c.Character)
49 .WithManualPriority(orderMessageInfo.Priority);
50 }
51 else if (orderTargetType == Order.OrderTargetType.WallSection)
52 {
53 order = new Order(orderPrefab, orderOption, orderTargetEntity as Structure, wallSectionIndex, orderGiver: c.Character)
54 .WithManualPriority(orderMessageInfo.Priority);
55 }
56 else
57 {
58 order = new Order(orderPrefab, orderOption, orderTargetEntity, orderPrefab.GetTargetItemComponent(orderTargetEntity as Item), orderGiver: c.Character)
59 .WithManualPriority(orderMessageInfo.Priority);
60 }
61 orderMsg = new OrderChatMessage(order, orderTargetCharacter, c.Character);
62 txt = orderMsg.Text;
63 }
64 else
65 {
66 txt = msg.ReadString() ?? "";
67 }
68
69 if (!NetIdUtils.IdMoreRecent(ID, c.LastSentChatMsgID)) { return; }
70
71 c.LastSentChatMsgID = ID;
72
73 if (txt.Length > MaxLength)
74 {
75 txt = txt.Substring(0, MaxLength);
76 }
77
78 c.LastSentChatMessages.Add(txt);
79 if (c.LastSentChatMessages.Count > 10)
80 {
81 c.LastSentChatMessages.RemoveRange(0, c.LastSentChatMessages.Count - 10);
82 }
83
84 float similarity = 0.0f;
85 for (int i = 0; i < c.LastSentChatMessages.Count; i++)
86 {
87 float closeFactor = 1.0f / (c.LastSentChatMessages.Count - i);
88
89 if (string.IsNullOrEmpty(txt))
90 {
91 similarity += closeFactor;
92 }
93 else
94 {
95 int levenshteinDist = ToolBox.LevenshteinDistance(txt, c.LastSentChatMessages[i]);
96 similarity += Math.Max((txt.Length - levenshteinDist) / (float)txt.Length * closeFactor, 0.0f);
97 }
98 }
99 //order/report messages can be sent a little faster than normal messages without triggering the spam filter
100 if (orderMsg != null)
101 {
102 similarity *= 0.25f;
103 }
104
105 bool isOwner = GameMain.Server.OwnerConnection != null && c.Connection == GameMain.Server.OwnerConnection;
106
107 if (similarity + c.ChatSpamSpeed > 5.0f && !isOwner && !GameMain.LuaCs.Game.disableSpamFilter)
108 {
109 GameMain.Server.KarmaManager.OnSpamFilterTriggered(c);
110
111 c.ChatSpamCount++;
112 if (c.ChatSpamCount > 3)
113 {
114 //kick for spamming too much
115 GameMain.Server.KickClient(c, TextManager.Get("SpamFilterKicked").Value);
116 }
117 else
118 {
119 ChatMessage denyMsg = Create("", TextManager.Get("SpamFilterBlocked").Value, ChatMessageType.Server, null);
120 c.ChatSpamTimer = 10.0f;
121 GameMain.Server.SendDirectChatMessage(denyMsg, c);
122 }
123 return;
124 }
125
126 c.ChatSpamSpeed += similarity + 0.5f;
127
128 if (c.ChatSpamTimer > 0.0f && !isOwner && !GameMain.LuaCs.Game.disableSpamFilter)
129 {
130 ChatMessage denyMsg = Create("", TextManager.Get("SpamFilterBlocked").Value, ChatMessageType.Server, null);
131 c.ChatSpamTimer = 10.0f;
132 GameMain.Server.SendDirectChatMessage(denyMsg, c);
133 return;
134 }
135
136 var should = GameMain.LuaCs.Hook.Call<bool?>("chatMessage", txt, c, type);
137
138 if (should != null && should.Value)
139 {
140 return;
141 }
142
143 if (type == ChatMessageType.Order)
144 {
145 if (c.Character == null || c.Character.SpeechImpediment >= 100.0f || c.Character.IsDead) { return; }
146 if (orderMsg.Order.IsReport)
147 {
148 HumanAIController.ReportProblem(orderMsg.Sender, orderMsg.Order);
149 }
150 if (order != null)
151 {
152 if (order.TargetAllCharacters)
153 {
154 if (order.IsIgnoreOrder)
155 {
156 switch (orderTargetType)
157 {
158 case Order.OrderTargetType.Entity:
159 if (!(orderTargetEntity is IIgnorable ignorableEntity)) { break; }
160 ignorableEntity.OrderedToBeIgnored = order.Identifier == "ignorethis";
161 break;
162 case Order.OrderTargetType.Position:
163 throw new NotImplementedException();
164 case Order.OrderTargetType.WallSection:
165 if (!wallSectionIndex.HasValue) { break; }
166 if (!(orderTargetEntity is Structure s)) { break; }
167 if (!(s.GetSection(wallSectionIndex.Value) is IIgnorable ignorableWall)) { break; }
168 ignorableWall.OrderedToBeIgnored = order.Identifier == "ignorethis";
169 break;
170 }
171 }
172 GameMain.GameSession?.CrewManager?.AddOrder(order, order.IsIgnoreOrder ? (float?)null : order.FadeOutTime);
173 }
174 else if (orderTargetCharacter != null)
175 {
176 orderTargetCharacter.SetOrder(order, isNewOrder);
177 }
178 }
179 GameMain.Server.SendOrderChatMessage(orderMsg);
180 }
181 else
182 {
183 GameMain.Server.SendChatMessage(txt, null, c);
184 }

189 public int EstimateLengthBytesServer(Client c)
190 {
191 int length = 1 + //(byte)ServerNetObject.CHAT_MESSAGE
192 2 + //(UInt16)NetStateID
193 1 + //(byte)Type
194 (Text == null ? 0 : Encoding.UTF8.GetBytes(Text).Length) + 2;
195
196 if (SenderClient != null)
197 {
198 length += 8; //SteamID or local ID (ulong)
199 }
200 if (Sender != null && c.InGame)
201 {
202 length += 2; //sender ID (UInt16)
203 }
204 if (SenderName != null)
205 {
206 length += Encoding.UTF8.GetBytes(SenderName).Length + 2;
207 }
208
209 return length;
210 }
