193 public static Mission LoadRandom(Location[] locations, MTRandom rand, bool requireCorrectLocationType, MissionType missionType, bool isSinglePlayer = false)
194 {
195 List<MissionPrefab> allowedMissions = new List<MissionPrefab>();
196 if (missionType == MissionType.None)
197 {
198 return null;
199 }
200 else
201 {
202 allowedMissions.AddRange(MissionPrefab.Prefabs.Where(m => ((int)(missionType & m.Type)) != 0));
203 }
204
205 allowedMissions.RemoveAll(m => isSinglePlayer ? m.MultiplayerOnly : m.SingleplayerOnly);
206 if (requireCorrectLocationType)
207 {
208 allowedMissions.RemoveAll(m => !m.IsAllowed(locations[0], locations[1]));
209 }
210
211 if (allowedMissions.Count == 0)
212 {
213 return null;
214 }
215
216 int probabilitySum = allowedMissions.Sum(m => m.Commonness);
217 int randomNumber = rand.NextInt32() % probabilitySum;
218 foreach (MissionPrefab missionPrefab in allowedMissions)
219 {
220 if (randomNumber <= missionPrefab.Commonness)
221 {
222 return missionPrefab.Instantiate(locations, Submarine.MainSub);
223 }
224 randomNumber -= missionPrefab.Commonness;
225 }
226
227 return null;
228 }

295 {

343 public void GiveReward()
344 {
345 if (!(GameMain.GameSession.GameMode is CampaignMode campaign)) { return; }
346 int reward = GetReward(Submarine.MainSub);
347
348 float baseExperienceGain = reward * 0.09f;
349
350 float difficultyMultiplier = 1 + level.Difficulty / 100f;
351 baseExperienceGain *= difficultyMultiplier;
352
353 IEnumerable<Character> crewCharacters = GameSession.GetSessionCrewCharacters(CharacterType.Both);
354
355 // use multipliers here so that we can easily add them together without introducing multiplicative XP stacking
356 var experienceGainMultiplier = new AbilityExperienceGainMultiplier(1f);
357 crewCharacters.ForEach(c => c.CheckTalents(AbilityEffectType.OnAllyGainMissionExperience, experienceGainMultiplier));
358 crewCharacters.ForEach(c => experienceGainMultiplier.Value += c.GetStatValue(StatTypes.MissionExperienceGainMultiplier));
359
360 int experienceGain = (int)(baseExperienceGain * experienceGainMultiplier.Value);
361#if CLIENT
362 foreach (Character character in crewCharacters)
363 {
364 character.Info?.GiveExperience(experienceGain, isMissionExperience: true);
365 }
366#else
367 foreach (Barotrauma.Networking.Client c in GameMain.Server.ConnectedClients)
368 {
369 //give the experience to the stored characterinfo if the client isn't currently controlling a character
370 (c.Character?.Info ?? c.CharacterInfo)?.GiveExperience(experienceGain, isMissionExperience: true);
371 }
372#endif
373
374 // apply money gains afterwards to prevent them from affecting XP gains
375 var missionMoneyGainMultiplier = new AbilityMissionMoneyGainMultiplier(this, 1f);
376 crewCharacters.ForEach(c => c.CheckTalents(AbilityEffectType.OnGainMissionMoney, missionMoneyGainMultiplier));
377 crewCharacters.ForEach(c => missionMoneyGainMultiplier.Value += c.GetStatValue(StatTypes.MissionMoneyGainMultiplier));
378
379 int totalReward = (int)(reward * missionMoneyGainMultiplier.Value);
380 GameAnalyticsManager.AddMoneyGainedEvent(totalReward, GameAnalyticsManager.MoneySource.MissionReward, Prefab.Identifier.Value);
381
382#if SERVER
383 totalReward = DistributeRewardsToCrew(GameSession.GetSessionCrewCharacters(CharacterType.Player), totalReward);
384#endif
385 if (totalReward > 0)
386 {
387 campaign.Bank.Give(totalReward);
388 }
389
390 foreach (Character character in crewCharacters)
391 {
392 character.Info.MissionsCompletedSinceDeath++;
393 }
394
395 foreach (KeyValuePair<Identifier, float> reputationReward in ReputationRewards)
396 {
397 if (reputationReward.Key == "location")
398 {
399 Locations[0].Reputation.AddReputation(reputationReward.Value);
400 Locations[1].Reputation.AddReputation(reputationReward.Value);
401 }
402 else
403 {
404 Faction faction = campaign.Factions.Find(faction1 => faction1.Prefab.Identifier == reputationReward.Key);
405 if (faction != null) { faction.Reputation.AddReputation(reputationReward.Value); }
406 }
407 }
408
409 if (Prefab.DataRewards != null)
410 {
411 foreach (var (identifier, value, operation) in Prefab.DataRewards)
412 {
413 SetDataAction.PerformOperation(campaign.CampaignMetadata, identifier, value, operation);
414 }
415 }

419 public static int DistributeRewardsToCrew(IEnumerable<Character> crew, int totalReward)
420 {
421 int remainingRewards = totalReward;
422 float sum = GetRewardDistibutionSum(crew);
423 if (MathUtils.NearlyEqual(sum, 0)) { return remainingRewards; }
424 foreach (Character character in crew)
425 {
426 int rewardDistribution = character.Wallet.RewardDistribution;
427 float rewardWeight = sum > 100 ? rewardDistribution / sum : rewardDistribution / 100f;
428 int reward = (int)(totalReward * rewardWeight);
429 reward = Math.Min(remainingRewards, reward);
430 character.Wallet.Give(reward);
431 remainingRewards -= reward;
432 if (remainingRewards <= 0) { break; }
433 }
434
435 return remainingRewards;
436 }
437#endif

442 public static (int Amount, int Percentage, float Sum) GetRewardShare(int rewardDistribution, IEnumerable<Character> crew, Option<int> reward)
443 {
444 float sum = GetRewardDistibutionSum(crew, rewardDistribution);
445 if (MathUtils.NearlyEqual(sum, 0)) { return (0, 0, sum); }
446
447 float rewardWeight = sum > 100 ? rewardDistribution / sum : rewardDistribution / 100f;
448 int rewardPercentage = (int)(rewardWeight * 100);
449
450 return reward switch
451 {
452 Some<int> { Value: var amount } => ((int)(amount * rewardWeight), rewardPercentage, sum),
453 None<int> _ => (0, rewardPercentage, sum),
454 _ => throw new ArgumentOutOfRangeException()
455 };
