540 void UseScooter(Vector2 targetWorldPos)
541 {
542 if (!character.HasEquippedItem("scooter".ToIdentifier())) { return; }
543 SteeringManager.Reset();
544 character.CursorPosition = targetWorldPos;
545 if (character.Submarine != null)
546 {
547 character.CursorPosition -= character.Submarine.Position;
548 }
549 Vector2 diff = character.CursorPosition - character.Position;
550 Vector2 dir = Vector2.Normalize(diff);
551 if (character.CurrentHull == null && IsFollowOrderObjective)
552 {
553 float sqrDist = diff.LengthSquared();
554 if (sqrDist > MathUtils.Pow2(CloseEnough * 1.5f))
555 {
556 SteeringManager.SteeringManual(1.0f, dir);
557 }
558 else
559 {
560 float dot = Vector2.Dot(dir, VectorExtensions.Forward(character.AnimController.Collider.Rotation + MathHelper.PiOver2));
561 bool isFacing = dot > 0.9f;
562 if (!isFacing && sqrDist > MathUtils.Pow2(CloseEnough))
563 {
564 SteeringManager.SteeringManual(1.0f, dir);
565 }
566 }
567 }
568 else
569 {
570 SteeringManager.SteeringManual(1.0f, dir);
571 }
572 character.SetInput(InputType.Aim, false, true);
573 character.SetInput(InputType.Shoot, false, true);
574 }

617 private void SeekGaps(float maxDistance)
618 {
619 Gap selectedGap = null;
620 float selectedDistance = -1;
621 Vector2 toTargetNormalized = Vector2.Normalize(Target.WorldPosition - character.WorldPosition);
622 foreach (Gap gap in Gap.GapList)
623 {
624 if (gap.Open < 1) { continue; }
625 if (gap.Submarine == null) { continue; }
626 if (!IsFollowOrderObjective)
627 {
628 if (gap.FlowTargetHull == null) { continue; }
629 if (gap.Submarine != Target.Submarine) { continue; }
630 }
631 Vector2 toGap = gap.WorldPosition - character.WorldPosition;
632 if (Vector2.Dot(Vector2.Normalize(toGap), toTargetNormalized) < 0) { continue; }
633 float squaredDistance = toGap.LengthSquared();
634 if (squaredDistance > maxDistance * maxDistance) { continue; }
635 if (selectedGap == null || squaredDistance < selectedDistance)
636 {
637 selectedGap = gap;
638 selectedDistance = squaredDistance;
639 }
640 }
641 TargetGap = selectedGap;
642 }

644 public bool IsCloseEnough
645 {
646 get
647 {
648 if (character.IsClimbing)
649 {
650 if (SteeringManager == PathSteering && PathSteering.CurrentPath != null && !PathSteering.CurrentPath.Finished && PathSteering.IsCurrentNodeLadder)
651 {
652 if (Target.WorldPosition.Y > character.WorldPosition.Y)
653 {
654 // The target is still above us
655 return false;
656 }
657 if (!character.AnimController.IsAboveFloor)
658 {
659 // Going through a hatch
660 return false;
661 }
662 }
663 }
664 if (!AlwaysUseEuclideanDistance && !character.AnimController.InWater)
665 {
666 float yDist = Math.Abs(Target.WorldPosition.Y - character.WorldPosition.Y);
667 if (yDist > CloseEnough) { return false; }
668 float xDist = Math.Abs(Target.WorldPosition.X - character.WorldPosition.X);
669 return xDist <= CloseEnough;
670 }
671 Vector2 sourcePos = UseDistanceRelativeToAimSourcePos ? character.AnimController.AimSourceWorldPos : character.WorldPosition;
672 return Vector2.DistanceSquared(Target.WorldPosition, sourcePos) < CloseEnough * CloseEnough;
673 }
