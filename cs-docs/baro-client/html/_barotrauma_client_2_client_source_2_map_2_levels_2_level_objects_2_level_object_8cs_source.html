89 partial void InitProjSpecific()
90 {
91 Sprite?.EnsureLazyLoaded();
92 Prefab.DeformableSprite?.EnsureLazyLoaded();
93
94 CurrentSwingAmount = Prefab.SwingAmountRad;
95 CurrentScaleOscillation = Prefab.ScaleOscillation;
96
97 SwingTimer = Rand.Range(0.0f, MathHelper.TwoPi);
98 ScaleOscillateTimer = Rand.Range(0.0f, MathHelper.TwoPi);
99
100 if (Prefab.ParticleEmitterPrefabs != null)
101 {
102 ParticleEmitters = new ParticleEmitter[Prefab.ParticleEmitterPrefabs.Count];
103 ParticleEmitterTriggers = new LevelTrigger[Prefab.ParticleEmitterPrefabs.Count];
104 for (int i = 0; i < Prefab.ParticleEmitterPrefabs.Count; i++)
105 {
106 ParticleEmitters[i] = new ParticleEmitter(Prefab.ParticleEmitterPrefabs[i]);
107 ParticleEmitterTriggers[i] = Prefab.ParticleEmitterTriggerIndex[i] > -1 ?
108 Triggers[Prefab.ParticleEmitterTriggerIndex[i]] : null;
109 }
110 }
111
112 if (Prefab.LightSourceParams != null && Prefab.LightSourceParams.Count > 0)
113 {
114 LightSources = new LightSource[Prefab.LightSourceParams.Count];
115 LightSourceTriggers = new LevelTrigger[Prefab.LightSourceParams.Count];
116 for (int i = 0; i < Prefab.LightSourceParams.Count; i++)
117 {
118 LightSources[i] = new LightSource(Prefab.LightSourceParams[i])
119 {
120 Position = new Vector2(Position.X, Position.Y),
121 IsBackground = true
122 };
123 LightSourceTriggers[i] = Prefab.LightSourceTriggerIndex[i] > -1 ?
124 Triggers[Prefab.LightSourceTriggerIndex[i]] : null;
125 }
126 }
127
128 Sounds = new RoundSound[Prefab.Sounds.Count];
129 SoundChannels = new SoundChannel[Prefab.Sounds.Count];
130 SoundTriggers = new LevelTrigger[Prefab.Sounds.Count];
131 for (int i = 0; i < Prefab.Sounds.Count; i++)
132 {
133 Sounds[i] = RoundSound.Load(Prefab.Sounds[i].SoundElement, false);
134 SoundTriggers[i] = Prefab.Sounds[i].TriggerIndex > -1 ? Triggers[Prefab.Sounds[i].TriggerIndex] : null;
135 }
136
137 int j = 0;
138 foreach (var subElement in Prefab.Config.Elements())
139 {
140 if (!subElement.Name.ToString().Equals("deformablesprite", StringComparison.OrdinalIgnoreCase)) { continue; }
141 foreach (XElement animationElement in subElement.Elements())
142 {
143 var newDeformation = SpriteDeformation.Load(animationElement, Prefab.Name);
144 if (newDeformation != null)
145 {
146 newDeformation.Params = Prefab.SpriteDeformations[j].Params;
147 spriteDeformations.Add(newDeformation);
148 j++;
149 }
150 }
151 }
152
153 VisibleOnSonar = Prefab.SonarDisruption > 0.0f || Prefab.OverrideProperties.Any(p => p != null && p.SonarDisruption > 0.0f) ||
154 (Triggers != null && Triggers.Any(t => !MathUtils.NearlyEqual(t.Force, Vector2.Zero) && t.ForceMode != LevelTrigger.TriggerForceMode.LimitVelocity || !t.InfectIdentifier.IsEmpty));
155 if (VisibleOnSonar && Triggers.Any())
156 {
157 SonarRadius = Triggers.Select(t => t.ColliderRadius * 1.5f).Max();
158 }
159 }

161 public void Update(float deltaTime)
162 {
163 CurrentRotation = Rotation;
164 if (ActivePrefab.SwingFrequency > 0.0f)
165 {
166 SwingTimer += deltaTime * ActivePrefab.SwingFrequency;
167 SwingTimer = SwingTimer % MathHelper.TwoPi;
168 //lerp the swing amount to the correct value to prevent it from abruptly changing to a different value
169 //when a trigger changes the swing amoung
170 CurrentSwingAmount = MathHelper.Lerp(CurrentSwingAmount, ActivePrefab.SwingAmountRad, deltaTime * 10.0f);
171
172 if (ActivePrefab.SwingAmountRad > 0.0f)
173 {
174 CurrentRotation += (float)Math.Sin(SwingTimer) * CurrentSwingAmount;
175 }
176 }
177
178 CurrentScale = Vector2.One * Scale;
179 if (ActivePrefab.ScaleOscillationFrequency > 0.0f)
180 {
181 ScaleOscillateTimer += deltaTime * ActivePrefab.ScaleOscillationFrequency;
182 ScaleOscillateTimer = ScaleOscillateTimer % MathHelper.TwoPi;
183 CurrentScaleOscillation = Vector2.Lerp(CurrentScaleOscillation, ActivePrefab.ScaleOscillation, deltaTime * 10.0f);
184
185 float sin = (float)Math.Sin(ScaleOscillateTimer);
186 CurrentScale *= new Vector2(
187 1.0f + sin * CurrentScaleOscillation.X,
188 1.0f + sin * CurrentScaleOscillation.Y);
189 }
190
191 if (LightSources != null)
192 {
193 for (int i = 0; i < LightSources.Length; i++)
194 {
195 if (LightSourceTriggers[i] != null) LightSources[i].Enabled = LightSourceTriggers[i].IsTriggered;
196 LightSources[i].Rotation = -CurrentRotation;
197 LightSources[i].SpriteScale = CurrentScale;
198 }
199 }
200
201 if (spriteDeformations.Count > 0)
202 {
203 UpdateDeformations(deltaTime);
204 }
205
206 if (ParticleEmitters != null)
207 {
208 for (int i = 0; i < ParticleEmitters.Length; i++)
209 {
210 if (ParticleEmitterTriggers[i] != null && !ParticleEmitterTriggers[i].IsTriggered) { continue; }
211 Vector2 emitterPos = LocalToWorld(Prefab.EmitterPositions[i]);
212 ParticleEmitters[i].Emit(deltaTime, emitterPos, hullGuess: null,
213 angle: ParticleEmitters[i].Prefab.Properties.CopyEntityAngle ? -CurrentRotation + MathHelper.Pi : 0.0f);
214 }
215 }
216
217 for (int i = 0; i < Sounds.Length; i++)
218 {
219 if (Sounds[i] == null) { continue; }
220 if (SoundTriggers[i] == null || SoundTriggers[i].IsTriggered)
221 {
222 RoundSound roundSound = Sounds[i];
223 Vector2 soundPos = LocalToWorld(new Vector2(Prefab.Sounds[i].Position.X, Prefab.Sounds[i].Position.Y));
224 if (Vector2.DistanceSquared(new Vector2(GameMain.SoundManager.ListenerPosition.X, GameMain.SoundManager.ListenerPosition.Y), soundPos) <
225 roundSound.Range * roundSound.Range)
226 {
227 if (SoundChannels[i] == null || !SoundChannels[i].IsPlaying)
228 {
229 SoundChannels[i] = roundSound.Sound.Play(roundSound.Volume, roundSound.Range, soundPos);
230 }
231 SoundChannels[i].Position = new Vector3(soundPos.X, soundPos.Y, 0.0f);
232 }
233 }
234 else if (SoundChannels[i] != null && SoundChannels[i].IsPlaying)
235 {
236 SoundChannels[i].FadeOutAndDispose();
237 SoundChannels[i] = null;
238 }
239 }
240 }

242 private void UpdateDeformations(float deltaTime)
243 {
244 if (ActivePrefab.DeformableSprite == null) { return; }
245 foreach (SpriteDeformation deformation in spriteDeformations)
246 {
247 if (deformation is PositionalDeformation positionalDeformation)
248 {
249 UpdatePositionalDeformation(positionalDeformation, deltaTime);
250 }
251 deformation.Update(deltaTime);
252 }
253 CurrentSpriteDeformation = SpriteDeformation.GetDeformation(spriteDeformations, ActivePrefab.DeformableSprite.Size);
254 if (LightSources != null)
255 {
256 foreach (LightSource lightSource in LightSources)
257 {
258 if (lightSource?.DeformableLightSprite != null)
259 {
260 lightSource.DeformableLightSprite.Deform(CurrentSpriteDeformation);
261 }
262 }
263 }
264 }

266 private void UpdatePositionalDeformation(PositionalDeformation positionalDeformation, float deltaTime)
267 {
268 if (Triggers == null) { return; }
269 Matrix matrix = ActivePrefab.DeformableSprite.GetTransform(
270 Position,
271 ActivePrefab.DeformableSprite.Origin,
272 CurrentRotation,
273 Vector2.One * Scale);
274
275 Matrix rotationMatrix = Matrix.CreateRotationZ(CurrentRotation);
276
277 foreach (LevelTrigger trigger in Triggers)
278 {
279 foreach (Entity triggerer in trigger.Triggerers)
280 {
281 Vector2 moveAmount = triggerer.WorldPosition - trigger.TriggererPosition[triggerer];
282
283 moveAmount = Vector2.Transform(moveAmount, rotationMatrix);
284 moveAmount /= ActivePrefab.DeformableSprite.Size * Scale;
285 moveAmount.Y = -moveAmount.Y;
286
287 positionalDeformation.Deform(trigger.WorldPosition, moveAmount, deltaTime, Matrix.Invert(matrix) *
288 Matrix.CreateScale(1.0f / ActivePrefab.DeformableSprite.Size.X, 1.0f / ActivePrefab.DeformableSprite.Size.Y, 1));
289 }
290 }

293 public void ClientRead(IReadMessage msg)
294 {
295 if (Triggers == null) { return; }
296
297 if (Prefab.TakeLevelWallDamage)
298 {
299 float newHealth = msg.ReadRangedSingle(0.0f, Prefab.Health, 8);
300 AddDamage(Health - newHealth, 1.0f, null, isNetworkEvent: true);
301 }
302 for (int i = 0; i < Triggers.Count; i++)
303 {
304 if (!Triggers[i].UseNetworkSyncing) { continue; }
305 Triggers[i].ClientRead(msg);
306 }
