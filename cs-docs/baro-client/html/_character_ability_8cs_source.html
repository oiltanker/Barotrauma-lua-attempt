96 public static CharacterAbility Load(ContentXElement abilityElement, CharacterAbilityGroup characterAbilityGroup, bool errorMessages = true)
97 {
98 Type abilityType;
99 string type = abilityElement.Name.ToString().ToLowerInvariant();
100 try
101 {
102 abilityType = Type.GetType("Barotrauma.Abilities." + type + "", false, true);
103 if (abilityType == null)
104 {
105 if (errorMessages) DebugConsole.ThrowError("Could not find the CharacterAbility \"" + type + "\" (" + characterAbilityGroup.CharacterTalent.DebugIdentifier + ")");
106 return null;
107 }
108 }
109 catch (Exception e)
110 {
111 if (errorMessages) DebugConsole.ThrowError("Could not find the CharacterAbility \"" + type + "\" (" + characterAbilityGroup.CharacterTalent.DebugIdentifier + ")", e);
112 return null;
113 }
114
115 object[] args = { characterAbilityGroup, abilityElement };
116 CharacterAbility characterAbility;
117
118 try
119 {
120 characterAbility = (CharacterAbility)Activator.CreateInstance(abilityType, args);
121 }
122 catch (TargetInvocationException e)
123 {
124 DebugConsole.ThrowError("Error while creating an instance of a CharacterAbility of the type " + abilityType + ".", e.InnerException);
125 return null;
126 }
127
128 return characterAbility;
129 }
