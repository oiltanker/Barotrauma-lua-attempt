213  public ItemContainer(Item item, ContentXElement element)
214  : base(item, element)
215  {
216  int totalCapacity = capacity;
217 
218  foreach (var subElement in element.Elements())
219  {
220  switch (subElement.Name.ToString().ToLowerInvariant())
221  {
222  case "containable":
223  RelatedItem containable = RelatedItem.Load(subElement, returnEmpty: false, parentDebugName: item.Name);
224  if (containable == null)
225  {
226  DebugConsole.ThrowError("Error in item config \"" + item.ConfigFilePath + "\" - containable with no identifiers.");
227  continue;
228  }
229  ContainableItems ??= new List<RelatedItem>();
230  ContainableItems.Add(containable);
231  break;
232  case "subcontainer":
233  totalCapacity += subElement.GetAttributeInt("capacity", 1);
234  break;
235  }
236  }
237  Inventory = new ItemInventory(item, this, totalCapacity, SlotsPerRow);
238  slotRestrictions = new SlotRestrictions[totalCapacity];
239  for (int i = 0; i < capacity; i++)
240  {
241  slotRestrictions[i] = new SlotRestrictions(maxStackSize, ContainableItems);
242  }
243 
244  int subContainerIndex = capacity;
245  foreach (var subElement in element.Elements())
246  {
247  if (subElement.Name.ToString().ToLowerInvariant() != "subcontainer") { continue; }
248 
249  int subCapacity = subElement.GetAttributeInt("capacity", 1);
250  int subMaxStackSize = subElement.GetAttributeInt("maxstacksize", maxStackSize);
251 
252  List<RelatedItem> subContainableItems = null;
253  foreach (var subSubElement in subElement.Elements())
254  {
255  if (subSubElement.Name.ToString().ToLowerInvariant() != "containable") { continue; }
256 
257  RelatedItem containable = RelatedItem.Load(subSubElement, returnEmpty: false, parentDebugName: item.Name);
258  if (containable == null)
259  {
260  DebugConsole.ThrowError("Error in item config \"" + item.ConfigFilePath + "\" - containable with no identifiers.");
261  continue;
262  }
263  subContainableItems ??= new List<RelatedItem>();
264  subContainableItems.Add(containable);
265  }
266 
267  for (int i = subContainerIndex; i < subContainerIndex + subCapacity; i++)
268  {
269  slotRestrictions[i] = new SlotRestrictions(subMaxStackSize, subContainableItems);
270  }
271  subContainerIndex += subCapacity;
272  }
273  capacity = totalCapacity;
274  InitProjSpecific(element);
275  }

368  public override void Update(float deltaTime, Camera cam)
369  {
370  if (!string.IsNullOrEmpty(SpawnWithId) && !alwaysContainedItemsSpawned)
371  {
372  SpawnAlwaysContainedItems();
373  }
374 
375  if (item.ParentInventory is CharacterInventory ownerInventory)
376  {
377  item.SetContainedItemPositions();
378 
379  if (AutoInject)
380  {
381  if (ownerInventory?.Owner is Character ownerCharacter &&
382  ownerCharacter.HealthPercentage / 100f <= AutoInjectThreshold &&
383  ownerCharacter.HasEquippedItem(item))
384  {
385  foreach (Item item in Inventory.AllItemsMod)
386  {
387  item.ApplyStatusEffects(ActionType.OnUse, 1.0f, ownerCharacter);
388  item.GetComponent<GeneticMaterial>()?.Equip(ownerCharacter);
389  }
390  }
391  }
392 
393  }
394  else if (item.body != null &&
395  item.body.Enabled &&
396  item.body.FarseerBody.Awake)
397  {
398  item.SetContainedItemPositions();
399  }
400  else if (activeContainedItems.Count == 0)
401  {
402  IsActive = false;
403  return;
404  }
405 
406  foreach (var activeContainedItem in activeContainedItems)
407  {
408  Item contained = activeContainedItem.Item;
409 
410  if (activeContainedItem.ExcludeBroken && contained.Condition <= 0.0f) { continue; }
411  StatusEffect effect = activeContainedItem.StatusEffect;
412 
413  if (effect.HasTargetType(StatusEffect.TargetType.This))
414  effect.Apply(ActionType.OnContaining, deltaTime, item, item.AllPropertyObjects);
415  if (effect.HasTargetType(StatusEffect.TargetType.Contained))
416  effect.Apply(ActionType.OnContaining, deltaTime, item, contained.AllPropertyObjects);
417  if (effect.HasTargetType(StatusEffect.TargetType.NearbyItems) ||
418  effect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
419  {
420  targets.Clear();
421  targets.AddRange(effect.GetNearbyTargets(item.WorldPosition, targets));
422  effect.Apply(ActionType.OnActive, deltaTime, item, targets);
423  }
424  }
425  }

540  public void SetContainedItemPositions()
541  {
542  Vector2 transformedItemPos = ItemPos * item.Scale;
543  Vector2 transformedItemInterval = ItemInterval * item.Scale;
544  Vector2 transformedItemIntervalHorizontal = new Vector2(transformedItemInterval.X, 0.0f);
545  Vector2 transformedItemIntervalVertical = new Vector2(0.0f, transformedItemInterval.Y);
546 
547  if (ItemPos == Vector2.Zero && ItemInterval == Vector2.Zero)
548  {
549  transformedItemPos = item.Position;
550  }
551  else
552  {
553  if (item.body == null)
554  {
555  if (item.FlippedX)
556  {
557  transformedItemPos.X = -transformedItemPos.X;
558  transformedItemPos.X += item.Rect.Width;
559  transformedItemInterval.X = -transformedItemInterval.X;
560  transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
561  }
562  if (item.FlippedY)
563  {
564  transformedItemPos.Y = -transformedItemPos.Y;
565  transformedItemPos.Y -= item.Rect.Height;
566  transformedItemInterval.Y = -transformedItemInterval.Y;
567  transformedItemIntervalVertical.Y = -transformedItemIntervalVertical.Y;
568  }
569  transformedItemPos += new Vector2(item.Rect.X, item.Rect.Y);
570  if (Math.Abs(item.Rotation) > 0.01f)
571  {
572  Matrix transform = Matrix.CreateRotationZ(MathHelper.ToRadians(-item.Rotation));
573  transformedItemPos = Vector2.Transform(transformedItemPos - item.Position, transform) + item.Position;
574  transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
575  transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
576  transformedItemIntervalVertical = Vector2.Transform(transformedItemIntervalVertical, transform);
577  }
578  }
579  else
580  {
581  Matrix transform = Matrix.CreateRotationZ(item.body.Rotation);
582  if (item.body.Dir == -1.0f)
583  {
584  transformedItemPos.X = -transformedItemPos.X;
585  transformedItemInterval.X = -transformedItemInterval.X;
586  transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
587  }
588  transformedItemPos = Vector2.Transform(transformedItemPos, transform);
589  transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
590  transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
591  transformedItemPos += item.Position;
592  }
593  }
594 
595  float currentRotation = itemRotation;
596  if (item.body != null)
597  {
598  currentRotation *= item.body.Dir;
599  currentRotation += item.body.Rotation;
600  }
601  else
602  {
603  currentRotation += MathHelper.ToRadians(-item.Rotation);
604  }
605 
606  int i = 0;
607  Vector2 currentItemPos = transformedItemPos;
608  foreach (Item contained in Inventory.AllItems)
609  {
610  if (contained.body != null)
611  {
612  try
613  {
614  Vector2 simPos = ConvertUnits.ToSimUnits(currentItemPos);
615  contained.body.FarseerBody.SetTransformIgnoreContacts(ref simPos, currentRotation);
616  contained.body.SetPrevTransform(contained.body.SimPosition, contained.body.Rotation);
617  contained.body.UpdateDrawPosition();
618  }
619  catch (Exception e)
620  {
621  DebugConsole.Log("SetTransformIgnoreContacts threw an exception in SetContainedItemPositions (" + e.Message + ")\n" + e.StackTrace.CleanupStackTrace());
622  GameAnalyticsManager.AddErrorEventOnce("ItemContainer.SetContainedItemPositions.InvalidPosition:" + contained.Name,
623  GameAnalyticsManager.ErrorSeverity.Error,
624  "SetTransformIgnoreContacts threw an exception in SetContainedItemPositions (" + e.Message + ")\n" + e.StackTrace.CleanupStackTrace());
625  }
626  contained.body.Submarine = item.Submarine;
627  }
628 
629  contained.Rect =
630  new Rectangle(
631  (int)(currentItemPos.X - contained.Rect.Width / 2.0f),
632  (int)(currentItemPos.Y + contained.Rect.Height / 2.0f),
633  contained.Rect.Width, contained.Rect.Height);
634 
635  contained.Submarine = item.Submarine;
636  contained.CurrentHull = item.CurrentHull;
637  contained.SetContainedItemPositions();
638 
639  i++;
640  if (Math.Abs(ItemInterval.X) > 0.001f && Math.Abs(ItemInterval.Y) > 0.001f)
641  {
642  //interval set on both axes -> use a grid layout
643  currentItemPos += transformedItemIntervalHorizontal;
644  if (i % ItemsPerRow == 0)
645  {
646  currentItemPos = transformedItemPos;
647  currentItemPos += transformedItemIntervalVertical * (i / ItemsPerRow);
648  }
649  }
650  else
651  {
652  currentItemPos += transformedItemInterval;
653  }
654  }
655  }
