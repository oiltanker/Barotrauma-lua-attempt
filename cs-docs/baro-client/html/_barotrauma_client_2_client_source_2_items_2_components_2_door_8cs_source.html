238  partial void SetState(bool open, bool isNetworkMessage, bool sendNetworkMessage, bool forcedOpen)
239  {
240  if ((IsStuck && !isNetworkMessage) ||
241  (PredictedState == null && isOpen == open) ||
242  (PredictedState != null && isOpen == PredictedState.Value && isOpen == open))
243  {
244  return;
245  }
246 
247  if (GameMain.Client != null && !isNetworkMessage)
248  {
249  bool stateChanged = open != PredictedState;
250 
251  //clients can "predict" that the door opens/closes when a signal is received
252  //the prediction will be reset after 1 second, setting the door to a state
253  //sent by the server, or reverting it back to its old state if no msg from server was received
254  PredictedState = open;
255  resetPredictionTimer = CorrectionDelay;
256  if (stateChanged) PlaySound(forcedOpen ? ActionType.OnPicked : ActionType.OnUse);
257  }
258  else
259  {
260  isOpen = open;
261  if (!isNetworkMessage || open != PredictedState)
262  {
263  StopPicking(null);
264  ActionType actionType = ActionType.OnUse;
265  if (forcedOpen)
266  {
267  actionType = ActionType.OnPicked;
268  }
269  else
270  {
271  if (open && HasSoundsOfType[(int)ActionType.OnOpen])
272  {
273  actionType = ActionType.OnOpen;
274  }
275  else if (!open && HasSoundsOfType[(int)ActionType.OnClose])
276  {
277  actionType = ActionType.OnClose;
278  }
279  }
280  PlaySound(actionType);
281  if (isOpen) { stuck = MathHelper.Clamp(stuck - StuckReductionOnOpen, 0.0f, 100.0f); }
282  }
