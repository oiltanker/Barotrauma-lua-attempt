35 private Vector2[] GetConvexHullCorners(Rectangle rect)
36 {
37 Point shadowSize = rect.Size.Multiply(ShadowScale);
38 Vector2 center = new Vector2(rect.Center.X, rect.Y - rect.Height / 2);
39
40 Vector2[] corners = new Vector2[4];
41 corners[0] = center + new Vector2(-shadowSize.X, -shadowSize.Y) / 2;
42 corners[1] = center + new Vector2(-shadowSize.X, shadowSize.Y) / 2;
43 corners[2] = center + new Vector2(shadowSize.X, shadowSize.Y) / 2;
44 corners[3] = center + new Vector2(shadowSize.X, -shadowSize.Y) / 2;
45
46 return corners;
47 }

49 private void UpdateConvexHulls()
50 {
51 if (item.Removed) { return; }
52
53 doorRect = new Rectangle(
54 item.Rect.Center.X - (int)(doorSprite.size.X / 2 * item.Scale),
55 item.Rect.Y - item.Rect.Height / 2 + (int)(doorSprite.size.Y / 2.0f * item.Scale),
56 (int)(doorSprite.size.X * item.Scale),
57 (int)(doorSprite.size.Y * item.Scale));
58
59 Rectangle rect = doorRect;
60 if (IsHorizontal)
61 {
62 rect.Width = (int)(rect.Width * (1.0f - openState));
63 }
64 else
65 {
66 rect.Height = (int)(rect.Height * (1.0f - openState));
67 }
68
69 if (Window.Height > 0 && Window.Width > 0)
70 {
71 if (IsHorizontal)
72 {
73 rect.Width = (int)(Window.X * item.Scale);
74 rect.X -= (int)(doorRect.Width * openState);
75 rect.Width = Math.Max(rect.Width - (doorRect.X - rect.X), 0);
76 rect.X = Math.Max(doorRect.X, rect.X);
77 if (convexHull2 != null)
78 {
79 Rectangle rect2 = doorRect;
80 rect2.X += (int)(Window.Right * item.Scale);
81 rect2.X -= (int)(doorRect.Width * openState);
82 rect2.X = Math.Max(doorRect.X, rect2.X);
83 rect2.Width = doorRect.Right - (int)(doorRect.Width * openState) - rect2.X;
84 if (rect2.Width == 0)
85 {
86 convexHull2.Enabled = false;
87 }
88 else
89 {
90 convexHull2.Enabled = true;
91 convexHull2.SetVertices(GetConvexHullCorners(rect2));
92 }
93 }
94 }
95 else
96 {
97 rect.Height = -(int)(Window.Y * item.Scale);
98 rect.Y += (int)(doorRect.Height * openState);
99 rect.Height = Math.Max(rect.Height - (rect.Y - doorRect.Y), 0);
100 rect.Y = Math.Min(doorRect.Y, rect.Y);
101 if (convexHull2 != null)
102 {
103 Rectangle rect2 = doorRect;
104 rect2.Y += (int)(Window.Y * item.Scale - Window.Height * item.Scale);
105 rect2.Y += (int)(doorRect.Height * openState);
106 rect2.Y = Math.Min(doorRect.Y, rect2.Y);
107 rect2.Height = rect2.Y - (doorRect.Y - (int)(doorRect.Height * (1.0f - openState)));
108 if (rect2.Height == 0)
109 {
110 convexHull2.Enabled = false;
111 }
112 else
113 {
114 convexHull2.Enabled = true;
115 convexHull2.SetVertices(GetConvexHullCorners(rect2));
116 }
117 }
118 }
119 }
120
121 if (convexHull == null) { return; }
122
123 if (rect.Height == 0 || rect.Width == 0)
124 {
125 convexHull.Enabled = false;
126 }
127 else
128 {
129 convexHull.Enabled = true;
130 convexHull.SetVertices(GetConvexHullCorners(rect));
131 }

238 partial void SetState(bool open, bool isNetworkMessage, bool sendNetworkMessage, bool forcedOpen)
239 {
240 if ((IsStuck && !isNetworkMessage) ||
241 (PredictedState == null && isOpen == open) ||
242 (PredictedState != null && isOpen == PredictedState.Value && isOpen == open))
243 {
244 return;
245 }
246
247 if (GameMain.Client != null && !isNetworkMessage)
248 {
249 bool stateChanged = open != PredictedState;
250
251 //clients can "predict" that the door opens/closes when a signal is received
252 //the prediction will be reset after 1 second, setting the door to a state
253 //sent by the server, or reverting it back to its old state if no msg from server was received
254 PredictedState = open;
255 resetPredictionTimer = CorrectionDelay;
256 if (stateChanged) PlaySound(forcedOpen ? ActionType.OnPicked : ActionType.OnUse);
257 }
258 else
259 {
260 isOpen = open;
261 if (!isNetworkMessage || open != PredictedState)
262 {
263 StopPicking(null);
264 ActionType actionType = ActionType.OnUse;
265 if (forcedOpen)
266 {
267 actionType = ActionType.OnPicked;
268 }
269 else
270 {
271 if (open && HasSoundsOfType[(int)ActionType.OnOpen])
272 {
273 actionType = ActionType.OnOpen;
274 }
275 else if (!open && HasSoundsOfType[(int)ActionType.OnClose])
276 {
277 actionType = ActionType.OnClose;
278 }
279 }
280 PlaySound(actionType);
281 if (isOpen) { stuck = MathHelper.Clamp(stuck - StuckReductionOnOpen, 0.0f, 100.0f); }
282 }
283 }

286 public override void ClientEventRead(IReadMessage msg, float sendingTime)
287 {
288 base.ClientEventRead(msg, sendingTime);
289
290 bool open = msg.ReadBoolean();
291 bool broken = msg.ReadBoolean();
292 bool forcedOpen = msg.ReadBoolean();
293 bool isStuck = msg.ReadBoolean();
294 bool isJammed = msg.ReadBoolean();
295 SetState(open, isNetworkMessage: true, sendNetworkMessage: false, forcedOpen: forcedOpen);
296 stuck = msg.ReadRangedSingle(0.0f, 100.0f, 8);
297 UInt16 lastUserID = msg.ReadUInt16();
298 Character user = lastUserID == 0 ? null : Entity.FindEntityByID(lastUserID) as Character;
299 if (user != lastUser)
300 {
301 lastUser = user;
302 toggleCooldownTimer = ToggleCoolDown;
303 }
304 this.isStuck = isStuck;
305 this.isJammed = isJammed;
306 if (isStuck) { OpenState = 0.0f; }
307 IsBroken = broken;
308 PredictedState = null;
309 }
