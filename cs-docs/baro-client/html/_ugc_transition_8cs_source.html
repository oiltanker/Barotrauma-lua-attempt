251  private static async Task<(OldSubs Subs, OldItemAssemblies ItemAssemblies, OldMods Mods)> DetermineItemsToTransition()
252  {
253  string[] subs = Array.Empty<string>();
254  string[] itemAssemblies = Array.Empty<string>();
255  List<(string Dir, string Name, Steamworks.Ugc.Item? Item, DateTime InstallTime)> mods
256  = new List<(string Dir, string Name, Steamworks.Ugc.Item? Item, DateTime InstallTime)>();
257  if (FolderShouldBeTransitioned(oldModsPath))
258  {
259  string[] getFiles(string path, string pattern)
260  => Directory.Exists(path)
261  ? Directory.GetFiles(path, pattern, SearchOption.TopDirectoryOnly)
262  : Array.Empty<string>();
263 
264  subs = getFiles(oldSubsPath, "*.sub");
265  itemAssemblies = getFiles(oldItemAssembliesPath, "*.xml");
266 
267  string[] allOldMods = Directory.GetDirectories(oldModsPath, "*", SearchOption.TopDirectoryOnly);
268 
269  var publishedItems = await SteamManager.Workshop.GetPublishedItems();
270  foreach (var modDir in allOldMods)
271  {
272  var fileList = XMLExtensions.TryLoadXml(Path.Combine(modDir, ContentPackage.FileListFileName), out _);
273  if (fileList?.Root is null) { continue; }
274 
275  var oldId = fileList.Root.GetAttributeUInt64("steamworkshopid", 0);
276  var updateTime = File.GetLastWriteTime(modDir).ToUniversalTime();
277  var oldName = fileList.Root.GetAttributeString("name", "");
278 
279  var item = oldId != 0 ? publishedItems.FirstOrNull(it => it.Id == oldId) : null;
280  if (oldId == 0 || item.HasValue)
281  {
282  mods.Add((modDir, oldName, item, updateTime));
283  }
284  }
285  }
286 
287  while (!(Screen.Selected is MainMenuScreen)) { await Task.Delay(50); }
288 
289  return (new OldSubs(subs), new OldItemAssemblies(itemAssemblies), new OldMods(mods));
290  }

309  private static async Task<string?> TransferMod(string path, GUITickBox tickbox)
310  {
311  if (!tickbox.Selected) { return null; }
312  string dirName = Path.GetFileNameWithoutExtension(path);
313  string destPath = Path.Combine(ContentPackage.LocalModsDir, dirName);
314 
315  //find unique path to save in
316  for (int i = 0;;i++)
317  {
318  if (!Directory.Exists(destPath)) { break; }
319  destPath = Path.Combine(ContentPackage.LocalModsDir, $"{dirName}.{i}");
320  }
321 
322  bool isSub = path.StartsWith(oldSubsPath, StringComparison.OrdinalIgnoreCase);
323  bool isItemAssembly = path.StartsWith(oldItemAssembliesPath, StringComparison.OrdinalIgnoreCase);
324  if (isSub || isItemAssembly)
325  {
326  //copying a sub or item assembly: manually create filelist.xml
327  ModProject modProject = new ModProject
328  {
329  Name = dirName,
330  ModVersion = ContentPackage.DefaultModVersion
331  };
332 
333  Type fileType;
334  if (isSub)
335  {
336  fileType = typeof(SubmarineFile);
337  XDocument? doc = SubmarineInfo.OpenFile(path, out _);
338  if (doc?.Root != null)
339  {
340  SubmarineType subType = doc.Root.GetAttributeEnum("type", SubmarineType.Player);
341  fileType = SubEditorScreen.DetermineSubFileType(subType);
342  }
343  }
344  else
345  {
346  fileType = typeof(ItemAssemblyFile);
347  }
348 
349  modProject.AddFile(ModProject.File.FromPath(
350  Path.Combine(ContentPath.ModDirStr, $"{dirName}.{(isSub ? "sub" : "xml")}"),
351  fileType));
352 
353  Directory.CreateDirectory(destPath);
354  File.Copy(path, Path.Combine(destPath, $"{dirName}.{(isSub ? "sub" : "xml")}"));
355  modProject.Save(Path.Combine(destPath, ContentPackage.FileListFileName));
356 
357  return destPath.CleanUpPathCrossPlatform(correctFilenameCase: false);
358  }
359  else
360  {
361  //copying a mod: we have a neat method for that!
362  await SteamManager.Workshop.CopyDirectory(path, Path.GetFileName(path), path, destPath);
363 
364  return null;
365  }
366  }
