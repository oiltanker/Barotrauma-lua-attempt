111  break;

114  break;
115  case "sloticon":
116  int index = subElement.GetAttributeInt("slotindex", -1);

257  public void DrawContainedItems(SpriteBatch spriteBatch, float itemDepth)
258  {
259  Vector2 transformedItemPos = ItemPos * item.Scale;
260  Vector2 transformedItemInterval = ItemInterval * item.Scale;
261  Vector2 transformedItemIntervalHorizontal = new Vector2(transformedItemInterval.X, 0.0f);
262  Vector2 transformedItemIntervalVertical = new Vector2(0.0f, transformedItemInterval.Y);
263 
264  if (item.body == null)
265  {
266  if (item.FlippedX)
267  {
268  transformedItemPos.X = -transformedItemPos.X;
269  transformedItemPos.X += item.Rect.Width;
270  transformedItemInterval.X = -transformedItemInterval.X;
271  transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
272  }
273  if (item.FlippedY)
274  {
275  transformedItemPos.Y = -transformedItemPos.Y;
276  transformedItemPos.Y -= item.Rect.Height;
277  transformedItemInterval.Y = -transformedItemInterval.Y;
278  transformedItemIntervalVertical.Y = -transformedItemIntervalVertical.Y;
279  }
280  transformedItemPos += new Vector2(item.Rect.X, item.Rect.Y);
281  if (item.Submarine != null) { transformedItemPos += item.Submarine.DrawPosition; }
282 
283  if (Math.Abs(item.Rotation) > 0.01f)
284  {
285  Matrix transform = Matrix.CreateRotationZ(MathHelper.ToRadians(-item.Rotation));
286  transformedItemPos = Vector2.Transform(transformedItemPos - item.DrawPosition, transform) + item.DrawPosition;
287  transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
288  transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
289  transformedItemIntervalVertical = Vector2.Transform(transformedItemIntervalVertical, transform);
290  }
291  }
292  else
293  {
294  Matrix transform = Matrix.CreateRotationZ(item.body.DrawRotation);
295  if (item.body.Dir == -1.0f)
296  {
297  transformedItemPos.X = -transformedItemPos.X;
298  transformedItemInterval.X = -transformedItemInterval.X;
299  transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
300  }
301 
302  transformedItemPos = Vector2.Transform(transformedItemPos, transform);
303  transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
304  transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
305  transformedItemPos += item.body.DrawPosition;
306  }
307 
308  Vector2 currentItemPos = transformedItemPos;
309 
310  SpriteEffects spriteEffects = SpriteEffects.None;
311  if ((item.body != null && item.body.Dir == -1) || item.FlippedX)
312  {
313  spriteEffects |= MathUtils.NearlyEqual(ItemRotation % 180, 90.0f) ? SpriteEffects.FlipVertically : SpriteEffects.FlipHorizontally;
314  }
315  if (item.FlippedY)
316  {
317  spriteEffects |= MathUtils.NearlyEqual(ItemRotation % 180, 90.0f) ? SpriteEffects.FlipHorizontally : SpriteEffects.FlipVertically;
318  }
319 
320  bool isWiringMode = SubEditorScreen.TransparentWiringMode && SubEditorScreen.IsWiringMode();
321 
322  int i = 0;
323  foreach (Item containedItem in Inventory.AllItems)
324  {
325  if (AutoInteractWithContained)
326  {
327  containedItem.IsHighlighted = item.IsHighlighted;
328  item.IsHighlighted = false;
329  }
330 
331  Vector2 origin = containedItem.Sprite.Origin;
332  if (item.FlippedX) { origin.X = containedItem.Sprite.SourceRect.Width - origin.X; }
333  if (item.FlippedY) { origin.Y = containedItem.Sprite.SourceRect.Height - origin.Y; }
334 
335  float containedSpriteDepth = ContainedSpriteDepth < 0.0f ? containedItem.Sprite.Depth : ContainedSpriteDepth;
336  if (i < containedSpriteDepths.Length)
337  {
338  containedSpriteDepth = containedSpriteDepths[i];
339  }
340  containedSpriteDepth = itemDepth + (containedSpriteDepth - (item.Sprite?.Depth ?? item.SpriteDepth)) / 10000.0f;
341 
342  containedItem.Sprite.Draw(
343  spriteBatch,
344  new Vector2(currentItemPos.X, -currentItemPos.Y),
345  isWiringMode ? containedItem.GetSpriteColor() * 0.15f : containedItem.GetSpriteColor(),
346  origin,
347  -(containedItem.body == null ? 0.0f : containedItem.body.DrawRotation ),
348  containedItem.Scale,
349  spriteEffects,
350  depth: containedSpriteDepth);
351 
352  foreach (ItemContainer ic in containedItem.GetComponents<ItemContainer>())
353  {
354  if (ic.hideItems) continue;
355  ic.DrawContainedItems(spriteBatch, containedSpriteDepth);
356  }
357 
358  i++;
359  if (Math.Abs(ItemInterval.X) > 0.001f && Math.Abs(ItemInterval.Y) > 0.001f)
360  {
361  //interval set on both axes -> use a grid layout
362  currentItemPos += transformedItemIntervalHorizontal;
363  if (i % ItemsPerRow == 0)
364  {
365  currentItemPos = transformedItemPos;
366  currentItemPos += transformedItemIntervalVertical * (i / ItemsPerRow);
367  }
368  }
369  else
370  {
371  currentItemPos += transformedItemInterval;
372  }
373  }
374  }
