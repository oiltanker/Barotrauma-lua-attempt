93 partial void InitProjSpecific(ContentXElement element)
94 {
95 slotIcons = new Sprite[capacity];
96 foreach (var subElement in element.Elements())
97 {
98 switch (subElement.Name.ToString().ToLowerInvariant())
99 {
100 case "topsprite":
101 inventoryTopSprite = new Sprite(subElement);
102 break;
103 case "backsprite":
104 inventoryBackSprite = new Sprite(subElement);
105 break;
106 case "bottomsprite":
107 inventoryBottomSprite = new Sprite(subElement);
108 break;
109 case "containedstateindicator":
110 ContainedStateIndicator = new Sprite(subElement);
111 break;
112 case "containedstateindicatorempty":
113 ContainedStateIndicatorEmpty = new Sprite(subElement);
114 break;
115 case "sloticon":
116 int index = subElement.GetAttributeInt("slotindex", -1);
117 Sprite icon = new Sprite(subElement);
118 for (int i = 0; i < capacity; i++)
119 {
120 if (i == index || index == -1)
121 {
122 slotIcons[i] = icon;
123 }
124 }
125 break;
126 }
127 }
128
129 if (string.IsNullOrEmpty(ContainedStateIndicatorStyle))
130 {
131 //if neither a style or a custom sprite is defined, use default style
132 if (ContainedStateIndicator == null)
133 {
134 IndicatorStyle = GUIStyle.GetComponentStyle("ContainedStateIndicator.Default");
135 }
136 }
137 else
138 {
139 IndicatorStyle = GUIStyle.GetComponentStyle("ContainedStateIndicator." + ContainedStateIndicatorStyle);
140 if (ContainedStateIndicator != null || ContainedStateIndicatorEmpty != null)
141 {
142 DebugConsole.AddWarning($"Item \"{item.Name}\" defines both a contained state indicator style and a custom indicator sprite. Will use the custom sprite...");
143 }
144 }
145 if (GuiFrame == null)
146 {
147 //if a GUIFrame is not defined in the xml,
148 //we create a full-screen frame and let the inventory position itself on it
149 GuiFrame = new GUIFrame(new RectTransform(Vector2.One, GUI.Canvas), style: null)
150 {
151 CanBeFocused = false
152 };
153 guiCustomComponent = new GUICustomComponent(new RectTransform(Vector2.One, GuiFrame.RectTransform),
154 onDraw: (SpriteBatch spriteBatch, GUICustomComponent component) => { Inventory.Draw(spriteBatch); },
155 onUpdate: null)
156 {
157 CanBeFocused = false
158 };
159 GuiFrame.RectTransform.ParentChanged += OnGUIParentChanged;
160 }
161 else
162 {
163 //if a GUIFrame has been defined, draw the inventory inside it
164 CreateGUI();
165 }
166
167 containedSpriteDepths = element.GetAttributeFloatArray("containedspritedepths", Array.Empty<float>());
168 }

170 protected override void CreateGUI()
171 {
172 var content = new GUIFrame(new RectTransform(GuiFrame.Rect.Size - GUIStyle.ItemFrameMargin, GuiFrame.RectTransform, Anchor.Center) { AbsoluteOffset = GUIStyle.ItemFrameOffset },
173 style: null)
174 {
175 CanBeFocused = false
176 };
177
178 LocalizedString labelText = GetUILabel();
179 GUITextBlock label = null;
180 if (!labelText.IsNullOrEmpty())
181 {
182 label = new GUITextBlock(new RectTransform(new Vector2(1.0f, 0.0f), content.RectTransform, Anchor.TopCenter),
183 labelText, font: GUIStyle.SubHeadingFont, textAlignment: Alignment.Center, wrap: true);
184 }
185
186 float minInventoryAreaSize = 0.5f;
187 guiCustomComponent = new GUICustomComponent(
188 new RectTransform(new Vector2(1.0f, label == null ? 1.0f : Math.Max(1.0f - label.RectTransform.RelativeSize.Y, minInventoryAreaSize)), content.RectTransform, Anchor.BottomCenter),
189 onDraw: (SpriteBatch spriteBatch, GUICustomComponent component) => { Inventory.Draw(spriteBatch); },
190 onUpdate: null)
191 {
192 CanBeFocused = false
193 };
194
195 // Expand the frame vertically if it's too small to fit the text
196 if (label != null && label.RectTransform.RelativeSize.Y > 0.5f)
197 {
198 int newHeight = (int)(GuiFrame.Rect.Height + (2 * (label.RectTransform.RelativeSize.Y - 0.5f) * content.Rect.Height));
199 if (newHeight > GuiFrame.RectTransform.MaxSize.Y)
200 {
201 Point newMaxSize = GuiFrame.RectTransform.MaxSize;
202 newMaxSize.Y = newHeight;
203 GuiFrame.RectTransform.MaxSize = newMaxSize;
204 }
205 GuiFrame.RectTransform.Resize(new Point(GuiFrame.Rect.Width, newHeight));
206 content.RectTransform.Resize(GuiFrame.Rect.Size - GUIStyle.ItemFrameMargin);
207 label.CalculateHeightFromText();
208 guiCustomComponent.RectTransform.Resize(new Vector2(1.0f, Math.Max(1.0f - label.RectTransform.RelativeSize.Y, minInventoryAreaSize)));
209 }
210
211 Inventory.RectTransform = guiCustomComponent.RectTransform;
212 }

233 public bool KeepOpenWhenEquippedBy(Character character)
234 {
235 if (!character.CanAccessInventory(Inventory) ||
236 !KeepOpenWhenEquipped ||
237 !character.HasEquippedItem(Item))
238 {
239 return false;
240 }
241
242 //if holding 2 different "always open" items in different hands, don't force them to stay open
243 if (character.HeldItems.Count() > 1 && character.HeldItems.All(it => it.GetComponent<ItemContainer>()?.KeepOpenWhenEquipped ?? false))
244 {
245 return false;
246 }
247
248 return true;

257 public void DrawContainedItems(SpriteBatch spriteBatch, float itemDepth)
258 {
259 Vector2 transformedItemPos = ItemPos * item.Scale;
260 Vector2 transformedItemInterval = ItemInterval * item.Scale;
261 Vector2 transformedItemIntervalHorizontal = new Vector2(transformedItemInterval.X, 0.0f);
262 Vector2 transformedItemIntervalVertical = new Vector2(0.0f, transformedItemInterval.Y);
263
264 if (item.body == null)
265 {
266 if (item.FlippedX)
267 {
268 transformedItemPos.X = -transformedItemPos.X;
269 transformedItemPos.X += item.Rect.Width;
270 transformedItemInterval.X = -transformedItemInterval.X;
271 transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
272 }
273 if (item.FlippedY)
274 {
275 transformedItemPos.Y = -transformedItemPos.Y;
276 transformedItemPos.Y -= item.Rect.Height;
277 transformedItemInterval.Y = -transformedItemInterval.Y;
278 transformedItemIntervalVertical.Y = -transformedItemIntervalVertical.Y;
279 }
280 transformedItemPos += new Vector2(item.Rect.X, item.Rect.Y);
281 if (item.Submarine != null) { transformedItemPos += item.Submarine.DrawPosition; }
282
283 if (Math.Abs(item.Rotation) > 0.01f)
284 {
285 Matrix transform = Matrix.CreateRotationZ(MathHelper.ToRadians(-item.Rotation));
286 transformedItemPos = Vector2.Transform(transformedItemPos - item.DrawPosition, transform) + item.DrawPosition;
287 transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
288 transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
289 transformedItemIntervalVertical = Vector2.Transform(transformedItemIntervalVertical, transform);
290 }
291 }
292 else
293 {
294 Matrix transform = Matrix.CreateRotationZ(item.body.DrawRotation);
295 if (item.body.Dir == -1.0f)
296 {
297 transformedItemPos.X = -transformedItemPos.X;
298 transformedItemInterval.X = -transformedItemInterval.X;
299 transformedItemIntervalHorizontal.X = -transformedItemIntervalHorizontal.X;
300 }
301
302 transformedItemPos = Vector2.Transform(transformedItemPos, transform);
303 transformedItemInterval = Vector2.Transform(transformedItemInterval, transform);
304 transformedItemIntervalHorizontal = Vector2.Transform(transformedItemIntervalHorizontal, transform);
305 transformedItemPos += item.body.DrawPosition;
306 }
307
308 Vector2 currentItemPos = transformedItemPos;
309
310 SpriteEffects spriteEffects = SpriteEffects.None;
311 if ((item.body != null && item.body.Dir == -1) || item.FlippedX)
312 {
313 spriteEffects |= MathUtils.NearlyEqual(ItemRotation % 180, 90.0f) ? SpriteEffects.FlipVertically : SpriteEffects.FlipHorizontally;
314 }
315 if (item.FlippedY)
316 {
317 spriteEffects |= MathUtils.NearlyEqual(ItemRotation % 180, 90.0f) ? SpriteEffects.FlipHorizontally : SpriteEffects.FlipVertically;
318 }
319
320 bool isWiringMode = SubEditorScreen.TransparentWiringMode && SubEditorScreen.IsWiringMode();
321
322 int i = 0;
323 foreach (Item containedItem in Inventory.AllItems)
324 {
325 if (AutoInteractWithContained)
326 {
327 containedItem.IsHighlighted = item.IsHighlighted;
328 item.IsHighlighted = false;
329 }
330
331 Vector2 origin = containedItem.Sprite.Origin;
332 if (item.FlippedX) { origin.X = containedItem.Sprite.SourceRect.Width - origin.X; }
333 if (item.FlippedY) { origin.Y = containedItem.Sprite.SourceRect.Height - origin.Y; }
334
335 float containedSpriteDepth = ContainedSpriteDepth < 0.0f ? containedItem.Sprite.Depth : ContainedSpriteDepth;
336 if (i < containedSpriteDepths.Length)
337 {
338 containedSpriteDepth = containedSpriteDepths[i];
339 }
340 containedSpriteDepth = itemDepth + (containedSpriteDepth - (item.Sprite?.Depth ?? item.SpriteDepth)) / 10000.0f;
341
342 containedItem.Sprite.Draw(
343 spriteBatch,
344 new Vector2(currentItemPos.X, -currentItemPos.Y),
345 isWiringMode ? containedItem.GetSpriteColor() * 0.15f : containedItem.GetSpriteColor(),
346 origin,
347 -(containedItem.body == null ? 0.0f : containedItem.body.DrawRotation ),
348 containedItem.Scale,
349 spriteEffects,
350 depth: containedSpriteDepth);
351
352 foreach (ItemContainer ic in containedItem.GetComponents<ItemContainer>())
353 {
354 if (ic.hideItems) continue;
355 ic.DrawContainedItems(spriteBatch, containedSpriteDepth);
356 }
357
358 i++;
359 if (Math.Abs(ItemInterval.X) > 0.001f && Math.Abs(ItemInterval.Y) > 0.001f)
360 {
361 //interval set on both axes -> use a grid layout
362 currentItemPos += transformedItemIntervalHorizontal;
363 if (i % ItemsPerRow == 0)
364 {
365 currentItemPos = transformedItemPos;
366 currentItemPos += transformedItemIntervalVertical * (i / ItemsPerRow);
367 }
368 }
369 else
370 {
371 currentItemPos += transformedItemInterval;
372 }
373 }
374 }

376 public override void UpdateHUD(Character character, float deltaTime, Camera cam)
377 {

387 if (!guiCustomComponent.Visible) { return; }
388
389 Inventory.Update(deltaTime, cam);
