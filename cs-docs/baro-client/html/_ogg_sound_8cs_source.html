81 int bufferSize = (int)reader.TotalSamples * reader.Channels;

86 int readSamples = reader.ReadSamples(floatBuffer, 0, bufferSize);
87
88 playbackAmplitude = new List<float>();
89 for (int i = 0; i < bufferSize; i += reader.Channels * AMPLITUDE_SAMPLE_COUNT)
90 {
91 float maxAmplitude = 0.0f;
92 for (int j = i; j < i + reader.Channels * AMPLITUDE_SAMPLE_COUNT; j++)
93 {
94 if (j >= bufferSize) { break; }
95 maxAmplitude = Math.Max(maxAmplitude, Math.Abs(floatBuffer[j]));
96 }
97 playbackAmplitude.Add(maxAmplitude);
98 }
