77 private void OnAuthChange(Steamworks.SteamId steamID, Steamworks.SteamId ownerID, Steamworks.AuthResponse status)
78 {
79 RemotePeer remotePeer = remotePeers.Find(p => p.SteamID == steamID);
80 DebugConsole.Log(steamID + " validation: " + status + ", " + (remotePeer != null));
81
82 if (remotePeer == null) { return; }
83
84 if (remotePeer.Authenticated)
85 {
86 if (status != Steamworks.AuthResponse.OK)
87 {
88 DisconnectPeer(remotePeer, DisconnectReason.SteamAuthenticationFailed.ToString() + "/ Steam authentication status changed: " + status.ToString());
89 }
90 return;
91 }
92
93 if (status == Steamworks.AuthResponse.OK)
94 {
95 remotePeer.OwnerSteamID = ownerID;
96 remotePeer.Authenticated = true;
97 remotePeer.Authenticating = false;
98 foreach (var msg in remotePeer.UnauthedMessages)
99 {
100 //rewrite the owner id before
101 //forwarding the messages to
102 //the server, since it's only
103 //known now
104 int prevBitPosition = msg.Message.BitPosition;
105 msg.Message.BitPosition = sizeof(ulong) * 8;
106 msg.Message.Write(ownerID);
107 msg.Message.BitPosition = prevBitPosition;
108 byte[] msgToSend = (byte[])msg.Message.Buffer.Clone();
109 Array.Resize(ref msgToSend, msg.Message.LengthBytes);
110 ChildServerRelay.Write(msgToSend);
111 }
112 remotePeer.UnauthedMessages.Clear();
113 }
114 else
115 {
116 DisconnectPeer(remotePeer, DisconnectReason.SteamAuthenticationFailed.ToString() + "/ Steam authentication failed: " + status.ToString());
117 return;
118 }
119 }

133 private void OnP2PData(ulong steamId, byte[] data, int dataLength, int channel)
134 {
135 if (!isActive) { return; }
136
137 RemotePeer remotePeer = remotePeers.Find(p => p.SteamID == steamId);
138 if (remotePeer == null || remotePeer.DisconnectTime != null)
139 {
140 return;
141 }
142
143 IWriteMessage outMsg = new WriteOnlyMessage();
144 outMsg.Write(steamId);
145 outMsg.Write(remotePeer.OwnerSteamID);
146 outMsg.Write(data, 1, dataLength - 1);
147
148 DeliveryMethod deliveryMethod = (DeliveryMethod)data[0];
149
150 PacketHeader packetHeader = (PacketHeader)data[1];
151
152 if (!remotePeer.Authenticated & !remotePeer.Authenticating && packetHeader.IsConnectionInitializationStep())
153 {
154 remotePeer.DisconnectTime = null;
155
156 IReadMessage authMsg = new ReadOnlyMessage(data, packetHeader.IsCompressed(), 2, dataLength - 2, null);
157 ConnectionInitialization initializationStep = (ConnectionInitialization)authMsg.ReadByte();
158 if (initializationStep == ConnectionInitialization.SteamTicketAndVersion)
159 {
160 remotePeer.Authenticating = true;
161
162 authMsg.ReadString(); //skip name
163 authMsg.ReadInt32(); //skip owner key
164 authMsg.ReadUInt64(); //skip steamid
165 UInt16 ticketLength = authMsg.ReadUInt16();
166 byte[] ticket = authMsg.ReadBytes(ticketLength);
167
168 Steamworks.BeginAuthResult authSessionStartState = Steam.SteamManager.StartAuthSession(ticket, steamId);
169 if (authSessionStartState != Steamworks.BeginAuthResult.OK)
170 {
171 DisconnectPeer(remotePeer, DisconnectReason.SteamAuthenticationFailed.ToString() + "/ Steam auth session failed to start: " + authSessionStartState.ToString());
172 return;
173 }
174 }
175 }
176
177 if (remotePeer.Authenticating)
178 {
179 remotePeer.UnauthedMessages.Add(new RemotePeer.UnauthedMessage() { DeliveryMethod = deliveryMethod, Message = outMsg });
180 }
181 else
182 {
183 byte[] msgToSend = (byte[])outMsg.Buffer.Clone();
184 Array.Resize(ref msgToSend, outMsg.LengthBytes);
185 ChildServerRelay.Write(msgToSend);
186 }
187 }

189 public override void Update(float deltaTime)
190 {
191 if (!isActive) { return; }
192
193 if (ChildServerRelay.HasShutDown || (ChildServerRelay.Process?.HasExited ?? true))
194 {
195 Close();
196 var msgBox = new GUIMessageBox(TextManager.Get("ConnectionLost"), ChildServerRelay.CrashMessage);
197 msgBox.Buttons[0].OnClicked += (btn, obj) => { GameMain.MainMenuScreen.Select(); return false; };
198 return;
199 }
200
201 for (int i = remotePeers.Count - 1; i >= 0; i--)
202 {
203 if (remotePeers[i].DisconnectTime != null && remotePeers[i].DisconnectTime < Timing.TotalTime)
204 {
205 ClosePeerSession(remotePeers[i]);
206 }
207 }
208
209 for (int i = 0; i < 100; i++)
210 {
211 if (!Steamworks.SteamNetworking.IsP2PPacketAvailable()) { break; }
212 var packet = Steamworks.SteamNetworking.ReadP2PPacket();
213 if (packet.HasValue)
214 {
215 OnP2PData(packet?.SteamId ?? 0, packet?.Data, packet?.Data.Length ?? 0, 0);
216 receivedBytes += packet?.Data.Length ?? 0;
217 }
218 }
219
220 GameMain.Client?.NetStats?.AddValue(NetStats.NetStatType.ReceivedBytes, receivedBytes);
221 GameMain.Client?.NetStats?.AddValue(NetStats.NetStatType.SentBytes, sentBytes);
222
223 while (ChildServerRelay.Read(out byte[] incBuf))
224 {
225 ChildServerRelay.DisposeLocalHandles();
226 IReadMessage inc = new ReadOnlyMessage(incBuf, false, 0, incBuf.Length, ServerConnection);
227 HandleDataMessage(inc);
228 }
229 }

231 private void HandleDataMessage(IReadMessage inc)
232 {
233 if (!isActive) { return; }
234
235 UInt64 recipientSteamId = inc.ReadUInt64();
236 DeliveryMethod deliveryMethod = (DeliveryMethod)inc.ReadByte();
237
238 int p2pDataStart = inc.BytePosition;
239
240 PacketHeader packetHeader = (PacketHeader)inc.ReadByte();
241
242 if (recipientSteamId != selfSteamID)
243 {
244 if (!packetHeader.IsServerMessage())
245 {
246 DebugConsole.ThrowError("Received non-server message meant for remote peer");
247 return;
248 }
249
250 RemotePeer peer = remotePeers.Find(p => p.SteamID == recipientSteamId);
251
252 if (peer == null) { return; }
253
254 if (packetHeader.IsDisconnectMessage())
255 {
256 DisconnectPeer(peer, inc.ReadString());
257 return;
258 }
259
260 Steamworks.P2PSend sendType;
261 switch (deliveryMethod)
262 {
263 case DeliveryMethod.Reliable:
264 case DeliveryMethod.ReliableOrdered:
265 //the documentation seems to suggest that the
266 //Reliable send type enforces packet order
267 sendType = Steamworks.P2PSend.Reliable;
268 break;
269 default:
270 sendType = Steamworks.P2PSend.Unreliable;
271 break;
272 }
273
274 byte[] p2pData;
275
276 if (packetHeader.IsConnectionInitializationStep())
277 {
278 p2pData = new byte[inc.LengthBytes - p2pDataStart + 8];
279 p2pData[0] = inc.Buffer[p2pDataStart];
280 Lidgren.Network.NetBitWriter.WriteUInt64(SteamManager.CurrentLobbyID, 8 * 8, p2pData, 1 * 8);
281 Array.Copy(inc.Buffer, p2pDataStart+1, p2pData, 1 + 8, inc.LengthBytes - p2pDataStart - 1);
282 }
283 else
284 {
285 p2pData = new byte[inc.LengthBytes - p2pDataStart];
286 Array.Copy(inc.Buffer, p2pDataStart, p2pData, 0, p2pData.Length);
287
288 if (!packetHeader.IsHeartbeatMessage() && !packetHeader.IsDisconnectMessage())
289 {
290 UInt16 length = Lidgren.Network.NetBitWriter.ReadUInt16(p2pData, 16, 8);
291 if (length > p2pData.Length - 2)
292 {
293 string errorMsg = $"Length written in message to send to client is larger than buffer size ({length} > {p2pData.Length - 2})";
294 DebugConsole.ThrowError(errorMsg);
295 GameAnalyticsManager.AddErrorEventOnce(
296 "SteamP2POwnerPeerLengthValidationFail",
297 GameAnalyticsManager.ErrorSeverity.Error,
298 errorMsg);
299 }
300 }
301 }
302
303 if (p2pData.Length + 4 >= MsgConstants.MTU)
304 {
305 DebugConsole.Log("WARNING: message length comes close to exceeding MTU, forcing reliable send (" + p2pData.Length.ToString() + " bytes)");
306 sendType = Steamworks.P2PSend.Reliable;
307 }
308
309 bool successSend = Steamworks.SteamNetworking.SendP2PPacket(recipientSteamId, p2pData, p2pData.Length, 0, sendType);
310 sentBytes += p2pData.Length;
311
312 if (!successSend)
313 {
314 if (sendType != Steamworks.P2PSend.Reliable)
315 {
316 DebugConsole.Log("WARNING: message couldn't be sent unreliably, forcing reliable send (" + p2pData.Length.ToString() + " bytes)");
317 sendType = Steamworks.P2PSend.Reliable;
318 successSend = Steamworks.SteamNetworking.SendP2PPacket(recipientSteamId, p2pData, p2pData.Length, 0, sendType);
319 sentBytes += p2pData.Length;
320 }
321 if (!successSend)
322 {
323 DebugConsole.AddWarning("Failed to send message to remote peer! (" + p2pData.Length.ToString() + " bytes)");
324 }
325 }
326 }
327 else
328 {
329 if (packetHeader.IsDisconnectMessage())
330 {
331 DebugConsole.ThrowError("Received disconnect message from owned server");
332 return;
333 }
334 if (!packetHeader.IsServerMessage())
335 {
336 DebugConsole.ThrowError("Received non-server message from owned server");
337 return;
338 }
339 if (packetHeader.IsHeartbeatMessage())
340 {
341 return; //no timeout since we're using pipes, ignore this message
342 }
343 if (packetHeader.IsConnectionInitializationStep())
344 {
345 IWriteMessage outMsg = new WriteOnlyMessage();
346 outMsg.Write(selfSteamID);
347 outMsg.Write(selfSteamID);
348 outMsg.Write((byte)(PacketHeader.IsConnectionInitializationStep));
349 outMsg.Write(Name);
350
351 byte[] msgToSend = (byte[])outMsg.Buffer.Clone();
352 Array.Resize(ref msgToSend, outMsg.LengthBytes);
353 ChildServerRelay.Write(msgToSend);
354 return;
355 }
356 else
357 {
358 if (initializationStep != ConnectionInitialization.Success)
359 {
360 OnInitializationComplete?.Invoke();
361 initializationStep = ConnectionInitialization.Success;
362 }
363 UInt16 length = inc.ReadUInt16();
364 IReadMessage msg = new ReadOnlyMessage(inc.Buffer, packetHeader.IsCompressed(), inc.BytePosition, length, ServerConnection);
365 OnMessageReceived?.Invoke(msg);
366
367 return;
368 }
369 }

372 private void DisconnectPeer(RemotePeer peer, string msg)
373 {
374 if (!string.IsNullOrWhiteSpace(msg))
375 {
376 if (peer.DisconnectTime == null)
377 {
378 peer.DisconnectTime = Timing.TotalTime + 1.0;
379 }
380
381 IWriteMessage outMsg = new WriteOnlyMessage();
382 outMsg.Write((byte)(PacketHeader.IsServerMessage | PacketHeader.IsDisconnectMessage));
383 outMsg.Write(msg);
384
385 Steamworks.SteamNetworking.SendP2PPacket(peer.SteamID, outMsg.Buffer, outMsg.LengthBytes, 0, Steamworks.P2PSend.Reliable);
386 sentBytes += outMsg.LengthBytes;
387 }
388 else
389 {
390 ClosePeerSession(peer);
391 }
392 }

405 public override void Close(string msg = null, bool disableReconnect = false)
406 {
407 if (!isActive) { return; }
408
409 isActive = false;
410
411 for (int i = remotePeers.Count - 1; i >= 0; i--)
412 {
413 DisconnectPeer(remotePeers[i], msg ?? DisconnectReason.ServerShutdown.ToString());
414 }
415
416 Thread.Sleep(100);
417
418 for (int i = remotePeers.Count - 1; i >= 0; i--)
419 {
420 ClosePeerSession(remotePeers[i]);
421 }
422
423 ChildServerRelay.ClosePipes();
424
425 OnDisconnect?.Invoke(disableReconnect);
426
427 SteamManager.LeaveLobby();
428 Steamworks.SteamNetworking.ResetActions();
429 Steamworks.SteamUser.OnValidateAuthTicketResponse -= OnAuthChange;
430 }

432 public override void Send(IWriteMessage msg, DeliveryMethod deliveryMethod, bool compressPastThreshold = true)
433 {
434 if (!isActive) { return; }
435
436 IWriteMessage msgToSend = new WriteOnlyMessage();
437 byte[] msgData = new byte[msg.LengthBytes];
438 msg.PrepareForSending(ref msgData, compressPastThreshold, out bool isCompressed, out int length);
439 msgToSend.Write(selfSteamID);
440 msgToSend.Write(selfSteamID);
441 msgToSend.Write((byte)(isCompressed ? PacketHeader.IsCompressed : PacketHeader.None));
442 msgToSend.Write((UInt16)length);
443 msgToSend.Write(msgData, 0, length);
444
445 byte[] bufToSend = (byte[])msgToSend.Buffer.Clone();
446 Array.Resize(ref bufToSend, msgToSend.LengthBytes);
447 ChildServerRelay.Write(bufToSend);
448 }
