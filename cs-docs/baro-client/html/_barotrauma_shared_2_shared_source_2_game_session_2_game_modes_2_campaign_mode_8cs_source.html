480  public TransitionType GetAvailableTransition(out LevelData nextLevel, out Submarine leavingSub)
481  {
482  if (Level.Loaded == null || Submarine.MainSub == null)
483  {
484  nextLevel = null;
485  leavingSub = null;
486  return TransitionType.None;
487  }
488 
489  leavingSub = GetLeavingSub();
490  if (leavingSub == null)
491  {
492  nextLevel = null;
493  return TransitionType.None;
494  }
495 
496  //currently travelling from location to another
497  if (Level.Loaded.Type == LevelData.LevelType.LocationConnection)
498  {
499  if (leavingSub.AtEndExit)
500  {
501  if (Map.EndLocation != null &&
502  map.SelectedLocation == Map.EndLocation &&
503  Map.EndLocation.Connections.Any(c => c.LevelData == Level.Loaded.LevelData))
504  {
505  nextLevel = map.StartLocation.LevelData;
506  return TransitionType.End;
507  }
508  if (Level.Loaded.EndLocation != null && Level.Loaded.EndLocation.Type.HasOutpost && Level.Loaded.EndOutpost != null)
509  {
510  nextLevel = Level.Loaded.EndLocation.LevelData;
511  return TransitionType.ProgressToNextLocation;
512  }
513  else if (map.SelectedConnection != null)
514  {
515  nextLevel = Level.Loaded.LevelData != map.SelectedConnection?.LevelData || (map.SelectedConnection.Locations[0] == Level.Loaded.EndLocation == Level.Loaded.Mirrored) ?
516  map.SelectedConnection.LevelData : null;
517  return TransitionType.ProgressToNextEmptyLocation;
518  }
519  else
520  {
521  nextLevel = null;
522  return TransitionType.ProgressToNextEmptyLocation;
523  }
524  }
525  else if (leavingSub.AtStartExit)
526  {
527  if (map.CurrentLocation.Type.HasOutpost && Level.Loaded.StartOutpost != null)
528  {
529  nextLevel = map.CurrentLocation.LevelData;
530  return TransitionType.ReturnToPreviousLocation;
531  }
532  else if (map.SelectedLocation != null && map.SelectedLocation != map.CurrentLocation && !map.CurrentLocation.Type.HasOutpost &&
533  map.SelectedConnection != null && Level.Loaded.LevelData != map.SelectedConnection.LevelData)
534  {
535  nextLevel = map.SelectedConnection.LevelData;
536  return TransitionType.LeaveLocation;
537  }
538  else
539  {
540  nextLevel = map.SelectedConnection?.LevelData;
541  return TransitionType.ReturnToPreviousEmptyLocation;
542  }
543  }
544  else
545  {
546  nextLevel = null;
547  return TransitionType.None;
548  }
549  }
550  else if (Level.Loaded.Type == LevelData.LevelType.Outpost)
551  {
552  nextLevel = map.SelectedLocation == null ? null : map.SelectedConnection?.LevelData;
553  return nextLevel == null ? TransitionType.None : TransitionType.LeaveLocation;
554  }
555  else
556  {
557  throw new NotImplementedException();
558  }

564  private Submarine GetLeavingSub()
565  {
566  if (Level.IsLoadedOutpost)
567  {
568  return Submarine.MainSub;
569  }
570  //in single player, only the sub the controlled character is inside can transition between levels
571  //in multiplayer, if there's subs at both ends of the level, only the one with more players inside can transition
572  //TODO: ignore players who don't have the permission to trigger a transition between levels?
573  var leavingPlayers = Character.CharacterList.Where(c => !c.IsDead && (c == Character.Controlled || c.IsRemotePlayer));
574 
575  //allow leaving if inside an outpost, and the submarine is either docked to it or close enough
576  Submarine leavingSubAtStart = GetLeavingSubAtStart(leavingPlayers);
577  Submarine leavingSubAtEnd = GetLeavingSubAtEnd(leavingPlayers);
578 
579  int playersInSubAtStart = leavingSubAtStart == null || !leavingSubAtStart.AtStartExit ? 0 :
580  leavingPlayers.Count(c => c.Submarine == leavingSubAtStart || leavingSubAtStart.DockedTo.Contains(c.Submarine) || (Level.Loaded.StartOutpost != null && c.Submarine == Level.Loaded.StartOutpost));
581  int playersInSubAtEnd = leavingSubAtEnd == null || !leavingSubAtEnd.AtEndExit ? 0 :
582  leavingPlayers.Count(c => c.Submarine == leavingSubAtEnd || leavingSubAtEnd.DockedTo.Contains(c.Submarine) || (Level.Loaded.EndOutpost != null && c.Submarine == Level.Loaded.EndOutpost));
583 
584  if (playersInSubAtStart == 0 && playersInSubAtEnd == 0)
585  {
586  return null;
587  }
588 
589  return playersInSubAtStart > playersInSubAtEnd ? leavingSubAtStart : leavingSubAtEnd;

591  static Submarine GetLeavingSubAtStart(IEnumerable<Character> leavingPlayers)
592  {
593  if (Level.Loaded.StartOutpost == null)
594  {
595  Submarine closestSub = Submarine.FindClosest(Level.Loaded.StartExitPosition, ignoreOutposts: true, ignoreRespawnShuttle: true, teamType: leavingPlayers.FirstOrDefault()?.TeamID);
596  return closestSub.DockedTo.Contains(Submarine.MainSub) ? Submarine.MainSub : closestSub;
597  }
598  else
599  {
600  //if there's a sub docked to the outpost, we can leave the level
601  if (Level.Loaded.StartOutpost.DockedTo.Any())
602  {
603  var dockedSub = Level.Loaded.StartOutpost.DockedTo.FirstOrDefault();
604  if (dockedSub == GameMain.NetworkMember?.RespawnManager?.RespawnShuttle || dockedSub.TeamID != leavingPlayers.FirstOrDefault()?.TeamID) { return null; }
605  return dockedSub.DockedTo.Contains(Submarine.MainSub) ? Submarine.MainSub : dockedSub;
606  }
607 
608  //nothing docked, check if there's a sub close enough to the outpost and someone inside the outpost
609  if (Level.Loaded.Type == LevelData.LevelType.LocationConnection && !leavingPlayers.Any(s => s.Submarine == Level.Loaded.StartOutpost)) { return null; }
610  Submarine closestSub = Submarine.FindClosest(Level.Loaded.StartOutpost.WorldPosition, ignoreOutposts: true, ignoreRespawnShuttle: true, teamType: leavingPlayers.FirstOrDefault()?.TeamID);
611  if (closestSub == null || !closestSub.AtStartExit) { return null; }
612  return closestSub.DockedTo.Contains(Submarine.MainSub) ? Submarine.MainSub : closestSub;
613  }

616  static Submarine GetLeavingSubAtEnd(IEnumerable<Character> leavingPlayers)
617  {
618  //no "end" in outpost levels
619  if (Level.Loaded.Type == LevelData.LevelType.Outpost) { return null; }
620 
621  if (Level.Loaded.EndOutpost == null)
622  {
623  Submarine closestSub = Submarine.FindClosest(Level.Loaded.EndExitPosition, ignoreOutposts: true, ignoreRespawnShuttle: true, teamType: leavingPlayers.FirstOrDefault()?.TeamID);
624  if (closestSub == null) { return null; }
625  return closestSub.DockedTo.Contains(Submarine.MainSub) ? Submarine.MainSub : closestSub;
626  }
627  else
628  {
629  //if there's a sub docked to the outpost, we can leave the level
630  if (Level.Loaded.EndOutpost.DockedTo.Any())
631  {
632  var dockedSub = Level.Loaded.EndOutpost.DockedTo.FirstOrDefault();
633  if (dockedSub == GameMain.NetworkMember?.RespawnManager?.RespawnShuttle || dockedSub.TeamID != leavingPlayers.FirstOrDefault()?.TeamID) { return null; }
634  return dockedSub.DockedTo.Contains(Submarine.MainSub) ? Submarine.MainSub : dockedSub;
635  }
636
