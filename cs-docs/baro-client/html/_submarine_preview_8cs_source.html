199 private async Task GeneratePreviewMeshesInternal()
200 {
201 await Task.Yield();
202 spriteRecorder.Begin(SpriteSortMode.BackToFront);
203
204 HashSet<int> toIgnore = new HashSet<int>();
205
206 foreach (var subElement in submarineInfo.SubmarineElement.Elements())
207 {
208 switch (subElement.Name.LocalName.ToLowerInvariant())
209 {
210 case "item":
211 foreach (var component in subElement.Elements())
212 {
213 switch (component.Name.LocalName.ToLowerInvariant())
214 {
215 case "itemcontainer":
216 ExtractItemContainerIds(component, toIgnore);
217 break;
218 case "connectionpanel":
219 ExtractConnectionPanelLinks(component, toIgnore);
220 break;
221 }
222 }
223 break;
224 }
225 if (isDisposed) { return; }
226 await Task.Yield();
227 }
228
229 foreach (var subElement in submarineInfo.SubmarineElement.Elements())
230 {
231 if (subElement.GetAttributeBool("hiddeningame", false)) { continue; }
232 switch (subElement.Name.LocalName.ToLowerInvariant())
233 {
234 case "item":
235 if (!toIgnore.Contains(subElement.GetAttributeInt("ID", 0)))
236 {
237 BakeMapEntity(subElement);
238 }
239 break;
240 case "structure":
241 BakeMapEntity(subElement);
242 break;
243 case "hull":
244 Identifier identifier = subElement.GetAttributeIdentifier("roomname", "");
245 if (!identifier.IsEmpty)
246 {
247 if (!hullCollections.TryGetValue(identifier, out HullCollection hullCollection))
248 {
249 hullCollection = new HullCollection(identifier);
250 hullCollections.Add(identifier, hullCollection);
251 }
252 hullCollection.AddRect(subElement);
253 }
254 break;
255 }
256 if (isDisposed) { return; }
257 await Task.Yield();
258 }
259 spriteRecorder.End();
260
261 camera.Position = (spriteRecorder.Min + spriteRecorder.Max) * 0.5f;
262 float scaledSpan = (spriteRecorder.Max - spriteRecorder.Min).X / camera.Resolution.X;
263 camera.Zoom = 0.8f / scaledSpan;
264 camera.StopMovement();
265 }

267 private void ExtractItemContainerIds(XElement component, HashSet<int> ids)
268 {
269 string containedString = component.GetAttributeString("contained", "");
270 string[] itemIdStrings = containedString.Split(',');
271 for (int i = 0; i < itemIdStrings.Length; i++)
272 {
273 foreach (string idStr in itemIdStrings[i].Split(';'))
274 {
275 if (!int.TryParse(idStr, NumberStyles.Any, CultureInfo.InvariantCulture, out int id)) { continue; }
276 if (id != 0 && !ids.Contains(id)) { ids.Add(id); }
277 }

295 private void BakeMapEntity(XElement element)
296 {
297 string identifier = element.GetAttributeString("identifier", "");
298 if (string.IsNullOrEmpty(identifier)) { return; }
299 Rectangle rect = element.GetAttributeRect("rect", Rectangle.Empty);
300 if (rect.Equals(Rectangle.Empty)) { return; }
301
302 float depth = element.GetAttributeFloat("spritedepth", 1f);
303 bool flippedX = element.GetAttributeBool("flippedx", false);
304 bool flippedY = element.GetAttributeBool("flippedy", false);
305
306 float scale = element.GetAttributeFloat("scale", 1f);
307 Color color = element.GetAttributeColor("spritecolor", Color.White);
308
309 float rotation = element.GetAttributeFloat("rotation", 0f);
310
311 MapEntityPrefab prefab = MapEntityPrefab.List.FirstOrDefault(p => p.Identifier == identifier);
312 if (prefab == null) { return; }
313
314 var texture = prefab.Sprite.Texture;
315 var srcRect = prefab.Sprite.SourceRect;
316
317 SpriteEffects spriteEffects = SpriteEffects.None;
318 if (flippedX && ((prefab as ItemPrefab)?.CanSpriteFlipX ?? true))
319 {
320 spriteEffects |= SpriteEffects.FlipHorizontally;
321 }
322 if (flippedY && ((prefab as ItemPrefab)?.CanSpriteFlipY ?? true))
323 {
324 spriteEffects |= SpriteEffects.FlipVertically;
325 }
326
327 var prevEffects = prefab.Sprite.effects;
328 prefab.Sprite.effects ^= spriteEffects;
329
330 bool overrideSprite = false;
331 ItemPrefab itemPrefab = prefab as ItemPrefab;
332 StructurePrefab structurePrefab = prefab as StructurePrefab;
333 if (itemPrefab != null)
334 {
335 BakeItemComponents(itemPrefab, rect, color, scale, rotation, depth, out overrideSprite);
336 }
337
338 if (!overrideSprite)
339 {
340 if (structurePrefab != null)
341 {
342 ParseUpgrades(structurePrefab.ConfigElement, ref scale);
343
344 if (!prefab.ResizeVertical)
345 {
346 rect.Height = (int)(rect.Height * scale / prefab.Scale);
347 }
348 if (!prefab.ResizeHorizontal)
349 {
350 rect.Width = (int)(rect.Width * scale / prefab.Scale);
351 }
352 var textureScale = element.GetAttributeVector2("texturescale", Vector2.One);
353
354 Vector2 backGroundOffset = Vector2.Zero;
355
356 Vector2 textureOffset = element.GetAttributeVector2("textureoffset", Vector2.Zero);
357 if (flippedX) { textureOffset.X = -textureOffset.X; }
358 if (flippedY) { textureOffset.Y = -textureOffset.Y; }
359
360 backGroundOffset = new Vector2(
361 MathUtils.PositiveModulo((int)-textureOffset.X, prefab.Sprite.SourceRect.Width),
362 MathUtils.PositiveModulo((int)-textureOffset.Y, prefab.Sprite.SourceRect.Height));
363
364 prefab.Sprite.DrawTiled(
365 spriteRecorder,
366 rect.Location.ToVector2() * new Vector2(1f, -1f),
367 rect.Size.ToVector2(),
368 color: color,
369 startOffset: backGroundOffset,
370 textureScale: textureScale * scale,
371 depth: depth);
372 }
373 else if (itemPrefab != null)
374 {
375 bool usePrefabValues = element.GetAttributeBool("isoverride", false) != itemPrefab.IsOverride;
376 if (usePrefabValues)
377 {
378 scale = itemPrefab.ConfigElement.GetAttributeFloat(scale, "scale", "Scale");
379 }
380
381 ParseUpgrades(itemPrefab.ConfigElement, ref scale);
382
383 if (prefab.ResizeVertical || prefab.ResizeHorizontal)
384 {
385 if (!prefab.ResizeHorizontal)
386 {
387 rect.Width = (int)(prefab.Sprite.size.X * scale);
388 }
389 if (!prefab.ResizeVertical)
390 {
391 rect.Height = (int)(prefab.Sprite.size.Y * scale);
392 }
393
394 var spritePos = rect.Center.ToVector2();
395 //spritePos.Y = rect.Height - spritePos.Y;
396
397 prefab.Sprite.DrawTiled(
398 spriteRecorder,
399 rect.Location.ToVector2() * new Vector2(1f, -1f),
400 rect.Size.ToVector2(),
401 color: color,
402 textureScale: Vector2.One * scale,
403 depth: depth);
404
405 foreach (var decorativeSprite in itemPrefab.DecorativeSprites)
406 {
407 float offsetState = 0f;
408 Vector2 offset = decorativeSprite.GetOffset(ref offsetState, Vector2.Zero) * scale;
409 if (flippedX && itemPrefab.CanSpriteFlipX) { offset.X = -offset.X; }
410 if (flippedY && itemPrefab.CanSpriteFlipY) { offset.Y = -offset.Y; }
411 decorativeSprite.Sprite.DrawTiled(spriteRecorder,
412 new Vector2(spritePos.X + offset.X - rect.Width / 2, -(spritePos.Y + offset.Y + rect.Height / 2)),
413 rect.Size.ToVector2(), color: color,
414 textureScale: Vector2.One * scale,
415 depth: Math.Min(depth + (decorativeSprite.Sprite.Depth - prefab.Sprite.Depth), 0.999f));
416 }
417 }
418 else
419 {
420 rect.Width = (int)(rect.Width * scale / prefab.Scale);
421 rect.Height = (int)(rect.Height * scale / prefab.Scale);
422
423 var spritePos = rect.Center.ToVector2();
424 spritePos.Y -= rect.Height;
425 //spritePos.Y = rect.Height - spritePos.Y;
426
427 prefab.Sprite.Draw(
428 spriteRecorder,
429 spritePos * new Vector2(1f, -1f),
430 color,
431 prefab.Sprite.Origin,
432 rotation,
433 scale,
434 prefab.Sprite.effects, depth);
435
436 foreach (var decorativeSprite in itemPrefab.DecorativeSprites)
437 {
438 float rotationState = 0f; float offsetState = 0f;
439 float rot = decorativeSprite.GetRotation(ref rotationState, 0f);
440 Vector2 offset = decorativeSprite.GetOffset(ref offsetState, Vector2.Zero) * scale;
441 if (flippedX && itemPrefab.CanSpriteFlipX) { offset.X = -offset.X; }
442 if (flippedY && itemPrefab.CanSpriteFlipY) { offset.Y = -offset.Y; }
443 decorativeSprite.Sprite.Draw(spriteRecorder, new Vector2(spritePos.X + offset.X, -(spritePos.Y + offset.Y)), color,
444 MathHelper.ToRadians(rotation) + rot, decorativeSprite.GetScale(0f) * scale, prefab.Sprite.effects,
445 depth: Math.Min(depth + (decorativeSprite.Sprite.Depth - prefab.Sprite.Depth), 0.999f));
446 }
447 }
448 }
449 }
450
451 prefab.Sprite.effects = prevEffects;

454 private void BakeItemComponents(
455 ItemPrefab prefab,
456 Rectangle rect, Color color,
457 float scale, float rotation, float depth,
458 out bool overrideSprite)
459 {
460 overrideSprite = false;
461
462 foreach (var subElement in prefab.ConfigElement.Elements())
463 {
464 switch (subElement.Name.LocalName.ToLowerInvariant())
465 {
466 case "turret":
467 Sprite barrelSprite = null;
468 Sprite railSprite = null;
469 foreach (var turretSubElem in subElement.Elements())
470 {
471 switch (turretSubElem.Name.ToString().ToLowerInvariant())
472 {
473 case "barrelsprite":
474 barrelSprite = new Sprite(turretSubElem);
475 break;
476 case "railsprite":
477 railSprite = new Sprite(turretSubElem);
478 break;
479 }
480 }
481
482 Vector2 barrelPos = subElement.GetAttributeVector2("barrelpos", Vector2.Zero);
483 Vector2 relativeBarrelPos = barrelPos * prefab.Scale - new Vector2(rect.Width / 2, rect.Height / 2);
484 var transformedBarrelPos = MathUtils.RotatePoint(
485 relativeBarrelPos,
486 MathHelper.ToRadians(rotation));
487
488 float relativeScale = scale / prefab.Scale;
489 Vector2 drawPos = new Vector2(rect.X + rect.Width * relativeScale / 2 + transformedBarrelPos.X * relativeScale, rect.Y - rect.Height * relativeScale / 2 - transformedBarrelPos.Y * relativeScale);
490 drawPos.Y = -drawPos.Y;
491
492 railSprite?.Draw(spriteRecorder,
493 drawPos,
494 color,
495 rotation + MathHelper.PiOver2, scale,
496 SpriteEffects.None, depth + (railSprite.Depth - prefab.Sprite.Depth));
497
498 barrelSprite?.Draw(spriteRecorder,
499 drawPos,
500 color,
501 rotation + MathHelper.PiOver2, scale,
502 SpriteEffects.None, depth + (barrelSprite.Depth - prefab.Sprite.Depth));
503
504 break;
505 case "door":
506 doors.Add(new Door(rect));
507
508 var doorSpriteElem = subElement.Elements().FirstOrDefault(e => e.Name.LocalName.Equals("sprite", StringComparison.OrdinalIgnoreCase));
509 if (doorSpriteElem != null)
510 {
511 string texturePath = doorSpriteElem.GetAttributeString("texture", "");
512 Vector2 pos = rect.Location.ToVector2() * new Vector2(1f, -1f);
513 if (subElement.GetAttributeBool("horizontal", false))
514 {
515 pos.Y += (float)rect.Height * 0.5f;
516 }
517 else
518 {
519 pos.X += (float)rect.Width * 0.5f;
520 }
521 Sprite doorSprite = new Sprite(doorSpriteElem, texturePath.Contains("/") ? "" : Path.GetDirectoryName(prefab.FilePath));
522 spriteRecorder.Draw(doorSprite.Texture, pos,
523 new Rectangle((int)doorSprite.SourceRect.X,
524 (int)doorSprite.SourceRect.Y,
525 (int)doorSprite.size.X, (int)doorSprite.size.Y),
526 color, 0.0f, doorSprite.Origin, new Vector2(scale), SpriteEffects.None, doorSprite.Depth);
527 }
528 break;
529 case "ladder":
530 var backgroundSprElem = subElement.Elements().FirstOrDefault(e => e.Name.LocalName.Equals("backgroundsprite", StringComparison.OrdinalIgnoreCase));
531 if (backgroundSprElem != null)
532 {
533 Sprite backgroundSprite = new Sprite(backgroundSprElem);
534 backgroundSprite.DrawTiled(spriteRecorder,
535 new Vector2(rect.Left, -rect.Top) - backgroundSprite.Origin * scale,
536 new Vector2(backgroundSprite.size.X * scale, rect.Height), color: color,
537 textureScale: Vector2.One * scale,
538 depth: depth + 0.1f);
539 }
540 break;
541 }
542 }

545 public void ParseUpgrades(XElement prefabConfigElement, ref float scale)
546 {
547 foreach (var upgrade in prefabConfigElement.Elements("Upgrade"))
548 {
549 var upgradeVersion = new Version(upgrade.GetAttributeString("gameversion", "0.0.0.0"));
550 if (upgradeVersion >= submarineInfo.GameVersion)
551 {
552 string scaleModifier = upgrade.GetAttributeString("scale", "*1");
553
554 if (scaleModifier.StartsWith("*"))
555 {
556 if (float.TryParse(scaleModifier.Substring(1), NumberStyles.Any, CultureInfo.InvariantCulture, out float parsedScale))
557 {
558 scale *= parsedScale;
559 }
560 }
561 else
562 {
563 if (float.TryParse(scaleModifier, NumberStyles.Any, CultureInfo.InvariantCulture, out float parsedScale))
564 {
565 scale = parsedScale;
566 }
567 }
568 }
569 }
