161  public virtual bool SteerThroughGap(Structure wall, WallSection section, Vector2 targetWorldPos, float deltaTime)
162  {
163  if (wall == null) { return false; }
164  if (section == null) { return false; }
165  Gap gap = section.gap;
166  if (gap == null) { return false; }
167  float maxDistance = Math.Min(wall.Rect.Width, wall.Rect.Height);
168  if (Vector2.DistanceSquared(Character.WorldPosition, targetWorldPos) > maxDistance * maxDistance) { return false; }
169  Hull targetHull = gap.FlowTargetHull;
170  if (targetHull == null) { return false; }
171  if (wall.IsHorizontal)
172  {
173  targetWorldPos.Y = targetHull.WorldRect.Y - targetHull.Rect.Height / 2;
174  }
175  else
176  {
177  targetWorldPos.X = targetHull.WorldRect.Center.X;
178  }
179  return SteerThroughGap(gap, targetWorldPos, deltaTime, maxDistance: -1);
180  }

242  public bool TakeItem(Item item, CharacterInventory targetInventory, bool equip, bool wear = false, bool dropOtherIfCannotMove = true, bool allowSwapping = false, bool storeUnequipped = false)
243  {
244  var pickable = item.GetComponent<Pickable>();
245  if (pickable == null) { return false; }
246  if (wear)
247  {
248  var wearable = item.GetComponent<Wearable>();
249  if (wearable != null)
250  {
251  pickable = wearable;
252  }
253  }
254  else
255  {
256  var holdable = item.GetComponent<Holdable>();
257  if (holdable != null)
258  {
259  pickable = holdable;
260  }
261  }
262  if (item.ParentInventory is ItemInventory itemInventory)
263  {
264  if (!itemInventory.Container.HasRequiredItems(Character, addMessage: false)) { return false; }
265  }
266  if (equip)
267  {
268  int targetSlot = -1;
269  //check if all the slots required by the item are free
270  foreach (InvSlotType slots in pickable.AllowedSlots)
271  {
272  if (slots.HasFlag(InvSlotType.Any)) { continue; }
273  for (int i = 0; i < targetInventory.Capacity; i++)
274  {
275  if (targetInventory is CharacterInventory characterInventory)
276  {
277  //slot not needed by the item, continue
278  if (!slots.HasFlag(characterInventory.SlotTypes[i])) { continue; }
279  }
280  targetSlot = i;
281  //slot free, continue
282  var otherItem = targetInventory.GetItemAt(i);
283  if (otherItem == null) { continue; }
284  //try to move the existing item to LimbSlot.Any and continue if successful
285  if (otherItem.AllowedSlots.Contains(InvSlotType.Any) && targetInventory.TryPutItem(otherItem, Character, CharacterInventory.anySlot))
286  {
287  if (storeUnequipped && targetInventory.Owner == Character)
288  {
289  unequippedItems.Add(otherItem);
290  }
291  continue;
292  }
293  if (dropOtherIfCannotMove)
294  {
295  //if everything else fails, simply drop the existing item
296  otherItem.Drop(Character);
297  }
298  }
299  }
300  return targetInventory.TryPutItem(item, targetSlot, allowSwapping, allowCombine: false, Character);
301  }
302  else
303  {
304  return targetInventory.TryPutItem(item, Character, CharacterInventory.anySlot);
305  }
306  }

314  public static void UnequipContainedItems(Character character, Item parentItem, Func<Item, bool> predicate, bool avoidDroppingInSea = true, int? unequipMax = null)
315  {
316  var inventory = parentItem.OwnInventory;
317  if (inventory == null) { return; }
318  int removed = 0;
319  if (predicate == null || inventory.AllItems.Any(predicate))
320  {
321  foreach (Item containedItem in inventory.AllItemsMod)
322  {
323  if (containedItem == null) { continue; }
324  if (predicate == null || predicate(containedItem))
325  {
326  if (avoidDroppingInSea && !character.IsInFriendlySub)
327  {
328  // If we are not inside a friendly sub (= same team), try to put the item in the inventory instead dropping it.
329  if (character.Inventory.TryPutItem(containedItem, character, CharacterInventory.anySlot))
330  {
331  if (unequipMax.HasValue && ++removed >= unequipMax) { return; }
332  continue;
333  }
334  }
335  containedItem.Drop(character);
336  if (unequipMax.HasValue && ++removed >= unequipMax) { return; }
337  }
338  }
339  }

359  private readonly float escapeTargetSeekInterval = 2;
360  private float escapeTimer;
361  protected bool allGapsSearched;
362  protected readonly HashSet<Gap> unreachableGaps = new HashSet<Gap>();
363  protected bool UpdateEscape(float deltaTime, bool canAttackDoors)
364  {
365  IndoorsSteeringManager pathSteering = SteeringManager as IndoorsSteeringManager;
366  if (allGapsSearched)
367  {
368  escapeTimer -= deltaTime;
369  if (escapeTimer <= 0)
370  {
371  allGapsSearched = false;
372  }
373  }
374  if (Character.CurrentHull != null && pathSteering != null)
375  {
376  // Seek exit if inside
377  if (!allGapsSearched)
378  {
379  float closestDistance = 0;
380  foreach (Gap gap in Gap.GapList)
381  {
382  if (gap == null || gap.Removed) { continue; }
383  if (EscapeTarget == gap) { continue; }
384  if (unreachableGaps.Contains(gap)) { continue; }
385  if (gap.Submarine != Character.Submarine) { continue; }
386  if (gap.IsRoomToRoom) { continue; }
387  float multiplier = 1;
388  var door = gap.ConnectedDoor;
389  if (door != null)
390  {
391  if (!door.CanBeTraversed)
392  {
393  if (!door.HasAccess(Character))
394  {
395  if (!canAttackDoors) { continue; }
396  // Treat doors that don't have access to like they were farther, because it will take time to break them.
397  multiplier = 5;
398  }
399  }
400  }
401  else
402  {
403  if (gap.Open < 1) { continue; }
404  if (gap.Size < minGapSize) { continue; }
405  }
406  if (gap.FlowTargetHull == Character.CurrentHull)
407  {
408  // If the gap is in the same room, it's close enough.
409  EscapeTarget = gap;
410  break;
411  }
412  float distance = Vector2.DistanceSquared(Character.WorldPosition, gap.WorldPosition) * multiplier;
413  if (EscapeTarget == null || distance < closestDistance)
414  {
415  EscapeTarget = gap;
416  closestDistance = distance;
417  }
418  }
419  allGapsSearched = true;
420  escapeTimer = escapeTargetSeekInterval;
421  }
422  else if (EscapeTarget != null && EscapeTarget.FlowTargetHull != Character.CurrentHull)
423  {
424  if (IsCurrentPathUnreachable)
425  {
426  unreachableGaps.Add(EscapeTarget);
427  EscapeTarget = null;
428  allGapsSearched = false;
429  }
430  }
431  }
432  if (EscapeTarget != null)
433  {
434  var door = EscapeTarget.ConnectedDoor;
435  bool isClosedDoor = door != null && !door.IsOpen;
436  Vector2 diff = EscapeTarget.WorldPosition - Character.WorldPosition;
437  float sqrDist = diff.LengthSquared();
438  bool isClose = sqrDist < MathUtils.Pow2(100);
439  if (Character.CurrentHull == null || isClose && !isClosedDoor || pathSteering == null || IsCurrentPathNullOrUnreachable || IsCurrentPathFinished)
440  {
441  // Very close to the target, outside, or at the end of the path -> try to steer through the gap
442  SteeringManager.Reset();
443  pathSteering?.ResetPath();
444  Vector2 dir = Vector2.Normalize(diff);
445  if (Character.CurrentHull == null || isClose)
446  {
447  // Outside -> steer away from the target
448  if (EscapeTarget.FlowTargetHull != null)
449  {
450  SteeringManager.SteeringManual(deltaTime, Vector2.Normalize(EscapeTarget.WorldPosition - EscapeTarget.FlowTargetHull.WorldPosition));
451  }
452  else
453  {
454  SteeringManager.SteeringManual(deltaTime, -dir);
455  }
456  }
457  else
458  {
459  // Still inside -> steer towards the target
460  SteeringManager.SteeringManual(deltaTime, dir);
461  }
462  return sqrDist < MathUtils.Pow2(250);
463  }
464  else if (pathSteering != null)
465  {
466  pathSteering.SteeringSeek(EscapeTarget.SimPosition, weight: 1, minGapSize);
467  }
468  else
469  {
470  SteeringManager.SteeringSeek(EscapeTarget.SimPosition, 10);
471  }
472  }
473  else
474  {
475  // Can't find the target
476  EscapeTarget = null;
477  allGapsSearched = false;
478  unreachableGaps.Clear();
479  }
480  return false;
481  }
