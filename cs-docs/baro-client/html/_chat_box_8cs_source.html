274 public bool TypingChatMessage(GUITextBox textBox, string text)
275 {
276 string command = ChatMessage.GetChatMessageCommand(text, out _);
277 if (IsSinglePlayer)
278 {
279 //radio is the only allowed special message type in single player
280 if (command != "r" && command != "radio")
281 {
282 command = "";
283 }
284 }
285
286 Color textColor;
287 switch (command)
288 {
289 case "r":
290 case "radio":
291 textColor = ChatMessage.MessageColor[(int)ChatMessageType.Radio];
292 break;
293 case "d":
294 case "dead":
295 textColor = ChatMessage.MessageColor[(int)ChatMessageType.Dead];
296 break;
297 default:
298 if (Character.Controlled != null && (Character.Controlled.IsDead || Character.Controlled.SpeechImpediment >= 100.0f))
299 {
300 textColor = ChatMessage.MessageColor[(int)ChatMessageType.Dead];
301 }
302 else if (command != "") //PMing
303 {
304 textColor = ChatMessage.MessageColor[(int)ChatMessageType.Private];
305 }
306 else
307 {
308 textColor = ChatMessage.MessageColor[(int)ChatMessageType.Default];
309 }
310 break;
311 }
312 textBox.TextColor = textBox.TextBlock.SelectedTextColor = textColor;
313
314 return true;
315 }

501 private IEnumerable<CoroutineStatus> UpdateMessageAnimation(GUIComponent message, float animDuration)
502 {
503 float timer = 0.0f;
504 while (timer < animDuration)
505 {
506 timer += CoroutineManager.DeltaTime;
507 float wavePhase = timer / animDuration * MathHelper.TwoPi;
508 message.RectTransform.ScreenSpaceOffset =
509 new Point((int)(Math.Sin(wavePhase) * (1.0f - timer / animDuration) * 50.0f), 0);
510 yield return CoroutineStatus.Running;
511 }
512 message.RectTransform.ScreenSpaceOffset = Point.Zero;
513 yield return CoroutineStatus.Success;
514 }

592 openState += deltaTime * 5.0f;
593 //delete all popup messages when the chatbox is open

609

611
612 if (popupMsg == popupMessages.First())
613 {
614 popupMsg.UserData = msgTimer + deltaTime * Math.Max(popupMessages.Count / 2, 1);
615 if (msgTimer > PopupMessageDuration)
616 {
617 //move the message out of the screen and delete it
618 popupMsg.RectTransform.ScreenSpaceOffset =
619 new Point((int)MathHelper.SmoothStep(popupMessageOffset, 10, (msgTimer - PopupMessageDuration) * 5.0f), targetYOffset);
620 if (msgTimer > PopupMessageDuration + 0.2f)
621 {
622 popupMsg.Parent?.RemoveChild(popupMsg);
623 }
624 }
625 }
626 if (msgTimer < PopupMessageDuration)
627 {
628 if (popupMsg != popupMessages.First()) { popupMsg.UserData = Math.Min(msgTimer + deltaTime, 1.0f); }
629 //move the message on the screen
630 popupMsg.RectTransform.ScreenSpaceOffset = new Point(
631 (int)MathHelper.SmoothStep(0, popupMessageOffset, msgTimer * 5.0f), targetYOffset);
632 }
633 yOffset += popupMsg.Rect.Height + GUI.IntScale(10);
634 }
635
636 popupMessages.RemoveAll(p => p.Parent == null);
637
638 //make the first popup message visible
639 /*var popupMsg = popupMessages.Count > 0 ? popupMessages.Peek() : null;
640 if (popupMsg != null)
641 {
642 popupMsg.Visible = true;
643 //popup messages appear and disappear faster when there's more pending messages
644 popupMessageTimer += deltaTime * popupMessages.Count * popupMessages.Count;
645 if (popupMessageTimer > PopupMessageDuration)
646 {
647 //move the message out of the screen and delete it
648 popupMsg.RectTransform.ScreenSpaceOffset =
649 new Point((int)MathHelper.SmoothStep(popupMessageOffset, 10, (popupMessageTimer - PopupMessageDuration) * 5.0f), 0);
