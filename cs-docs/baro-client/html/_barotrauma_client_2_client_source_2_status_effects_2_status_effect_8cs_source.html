27 partial void InitProjSpecific(ContentXElement element, string parentDebugName)
28 {
29 particleEmitters = new List<ParticleEmitter>();
30
31 foreach (var subElement in element.Elements())
32 {
33 switch (subElement.Name.ToString().ToLowerInvariant())
34 {
35 case "particleemitter":
36 particleEmitters.Add(new ParticleEmitter(subElement));
37 break;
38 case "sound":
39 var sound = RoundSound.Load(subElement);
40 if (sound?.Sound != null)
41 {
42 loopSound = subElement.GetAttributeBool("loop", false);
43 if (subElement.GetAttribute("selectionmode") != null)
44 {
45 if (Enum.TryParse(subElement.GetAttributeString("selectionmode", "Random"), out SoundSelectionMode selectionMode))
46 {
47 soundSelectionMode = selectionMode;
48 }
49 }
50 sounds.Add(sound);
51 }
52 break;
53 }
54 }

57 partial void ApplyProjSpecific(float deltaTime, Entity entity, IReadOnlyList<ISerializableEntity> targets, Hull hull, Vector2 worldPosition, bool playSound)
58 {
59 if (playSound)
60 {
61 PlaySound(entity, hull, worldPosition);
62 }
63
64 foreach (ParticleEmitter emitter in particleEmitters)
65 {
66 float angle = 0.0f;
67 float particleRotation = 0.0f;
68 bool mirrorAngle = false;
69 if (emitter.Prefab.Properties.CopyEntityAngle)
70 {
71 Limb targetLimb = null;
72 if (entity is Item item && item.body != null)
73 {
74 angle = item.body.Rotation + ((item.body.Dir > 0.0f) ? 0.0f : MathHelper.Pi);
75 particleRotation = -item.body.Rotation;
76 if (item.body.Dir < 0.0f)
77 {
78 particleRotation += MathHelper.Pi;
79 mirrorAngle = true;
80 }
81 }
82 else if (entity is Character c && !c.Removed && targetLimbs?.FirstOrDefault(l => l != LimbType.None) is LimbType l)
83 {
84 targetLimb = c.AnimController.GetLimb(l);
85 }
86 else
87 {
88 for (int i = 0; i < targets.Count; i++)
89 {
90 if (targets[i] is Limb limb)
91 {
92 targetLimb = limb;
93 break;
94 }
95 }
96 }
97 if (targetLimb != null && !targetLimb.Removed)
98 {
99 angle = targetLimb.body.Rotation + ((targetLimb.body.Dir > 0.0f) ? 0.0f : MathHelper.Pi);
100 particleRotation = -targetLimb.body.Rotation;
101 if (targetLimb.body.Dir < 0.0f)
102 {
103 particleRotation += MathHelper.Pi;
104 mirrorAngle = true;
105 }
106 }
107 }
108
109 emitter.Emit(deltaTime, worldPosition, hull, angle: angle, particleRotation: particleRotation, mirrorAngle: mirrorAngle);
110 }

115 private void PlaySound(Entity entity, Hull hull, Vector2 worldPosition)
116 {
117 if (sounds.Count == 0) return;
118
119 if (soundChannel == null || !soundChannel.IsPlaying)
120 {
121 if (soundSelectionMode == SoundSelectionMode.All)
122 {
123 foreach (RoundSound sound in sounds)
124 {
125 if (sound?.Sound == null)
126 {
127 string errorMsg = $"Error in StatusEffect.ApplyProjSpecific1 (sound \"{sound?.Filename ?? "unknown"}\" was null)\n" + Environment.StackTrace.CleanupStackTrace();
128 GameAnalyticsManager.AddErrorEventOnce("StatusEffect.ApplyProjSpecific:SoundNull1" + Environment.StackTrace.CleanupStackTrace(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
129 return;
130 }
131 soundChannel = SoundPlayer.PlaySound(sound.Sound, worldPosition, sound.Volume, sound.Range, hullGuess: hull, ignoreMuffling: sound.IgnoreMuffling);
132 ignoreMuffling = sound.IgnoreMuffling;
133 if (soundChannel != null) { soundChannel.Looping = loopSound; }
134 }
135 }
136 else
137 {
138 int selectedSoundIndex;
139 if (soundSelectionMode == SoundSelectionMode.ItemSpecific && entity is Item item)
140 {
141 selectedSoundIndex = item.ID % sounds.Count;
142 }
143 else if (soundSelectionMode == SoundSelectionMode.CharacterSpecific && entity is Character user)
144 {
145 selectedSoundIndex = user.ID % sounds.Count;
146 }
147 else
148 {
149 selectedSoundIndex = Rand.Int(sounds.Count);
150 }
151 var selectedSound = sounds[selectedSoundIndex];
152 if (selectedSound?.Sound == null)
153 {
154 string errorMsg = $"Error in StatusEffect.ApplyProjSpecific2 (sound \"{selectedSound?.Filename ?? "unknown"}\" was null)\n" + Environment.StackTrace.CleanupStackTrace();
155 GameAnalyticsManager.AddErrorEventOnce("StatusEffect.ApplyProjSpecific:SoundNull2" + Environment.StackTrace.CleanupStackTrace(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
156 return;
157 }
158 soundChannel = SoundPlayer.PlaySound(selectedSound.Sound, worldPosition, selectedSound.Volume, selectedSound.Range, hullGuess: hull, ignoreMuffling: selectedSound.IgnoreMuffling);
159 ignoreMuffling = selectedSound.IgnoreMuffling;
160 if (soundChannel != null) { soundChannel.Looping = loopSound; }
161 }
162 }
163 else
164 {
165 soundChannel.Position = new Vector3(worldPosition, 0.0f);
166 }
167
168 if (soundChannel != null && soundChannel.Looping)
169 {
170 ActiveLoopingSounds.Add(this);
171 soundEmitter = entity;
172 loopStartTime = Timing.TotalTime;
173 }
174 }

176 static partial void UpdateAllProjSpecific(float deltaTime)
177 {
178 bool doMuffleCheck = Timing.TotalTime > LastMuffleCheckTime + 0.2;
179 if (doMuffleCheck) { LastMuffleCheckTime = Timing.TotalTime; }
180 foreach (StatusEffect statusEffect in ActiveLoopingSounds)
181 {
182 if (statusEffect.soundChannel == null) { continue; }
183
184 //stop looping sounds if the statuseffect hasn't been applied in 0.1
185 //= keeping the sound looping requires continuously applying the statuseffect
186 if (Timing.TotalTime > statusEffect.loopStartTime + 0.1 && !DurationList.Any(e => e.Parent == statusEffect))
187 {
188 statusEffect.soundChannel.FadeOutAndDispose();
189 statusEffect.soundChannel = null;
190 }
191 else
192 {
193 statusEffect.soundChannel.Position = new Vector3(statusEffect.soundEmitter.WorldPosition, 0.0f);
194 if (doMuffleCheck && !statusEffect.ignoreMuffling)
195 {
196 statusEffect.soundChannel.Muffled = SoundPlayer.ShouldMuffleSound(
197 Character.Controlled, statusEffect.soundEmitter.WorldPosition, statusEffect.soundChannel.Far, Character.Controlled?.CurrentHull);
198 }
199 }
200 }
201 ActiveLoopingSounds.RemoveWhere(s => s.soundChannel == null);
202 }
