336  private Identifier GetTargetingTag(AITarget aiTarget)
337  {
338  if (aiTarget?.Entity == null) { return Identifier.Empty; }
339  string targetingTag = string.Empty;
340  if (aiTarget.Entity is Character targetCharacter)
341  {
342  if (targetCharacter.IsDead)
343  {
344  targetingTag = "dead";
345  }
346  else if (AIParams.TryGetTarget(targetCharacter.CharacterHealth.GetActiveAfflictionTags(), out CharacterParams.TargetParams tp) && tp.Threshold >= Character.GetDamageDoneByAttacker(targetCharacter))
347  {
348  targetingTag = tp.Tag;
349  }
350  else if (PetBehavior != null && aiTarget.Entity == PetBehavior.Owner)
351  {
352  targetingTag = "owner";
353  }
354  else if (AIParams.TryGetTarget(targetCharacter, out CharacterParams.TargetParams tP))
355  {
356  targetingTag = tP.Tag;
357  }
358  else if (targetCharacter.AIController is EnemyAIController enemy)
359  {
360  if (targetCharacter.IsHusk && AIParams.HasTag("husk"))
361  {
362  targetingTag = "husk";
363  }
364  else if (!Character.IsFriendly(targetCharacter))
365  {
366  if (enemy.CombatStrength > CombatStrength)
367  {
368  targetingTag = "stronger";
369  }
370  else if (enemy.CombatStrength < CombatStrength)
371  {
372  targetingTag = "weaker";
373  }
374  else
375  {
376  targetingTag = "equal";
377  }
378  }
379  }
380  }
381  else if (aiTarget.Entity is Item targetItem)
382  {
383  foreach (var prio in AIParams.Targets)
384  {
385  if (targetItem.HasTag(prio.Tag))
386  {
387  targetingTag = prio.Tag;
388  break;
389  }
390  }
391  if (targetingTag.IsNullOrEmpty())
392  {
393  if (targetItem.GetComponent<Sonar>() != null)
394  {
395  targetingTag = "sonar";
396  }
397  if (targetItem.GetComponent<Door>() != null)
398  {
399  targetingTag = "door";
400  }
401  }
402  }
403  else if (aiTarget.Entity is Structure)
404  {
405  targetingTag = "wall";
406  }
407  else if (aiTarget.Entity is Hull)
408  {
409  targetingTag = "room";
410  }

435  public override void Update(float deltaTime)
436  {
437  if (DisableEnemyAI) { return; }
438  base.Update(deltaTime);
439  UpdateTriggers(deltaTime);
440  Character.ClearInputs();
441 
442  bool ignorePlatforms = Character.AnimController.TargetMovement.Y < -0.5f && (-Character.AnimController.TargetMovement.Y > Math.Abs(Character.AnimController.TargetMovement.X));
443  if (steeringManager == insideSteering)
444  {
445  var currPath = PathSteering.CurrentPath;
446  if (currPath != null && currPath.CurrentNode != null)
447  {
448  if (currPath.CurrentNode.SimPosition.Y < Character.AnimController.GetColliderBottom().Y)
449  {
450  // Don't allow to jump from too high.
451  float allowedJumpHeight = Character.AnimController.ImpactTolerance / 2;
452  float height = Math.Abs(currPath.CurrentNode.SimPosition.Y - Character.SimPosition.Y);
453  ignorePlatforms = height < allowedJumpHeight;
454  }
455  }
456  if (Character.IsClimbing && PathSteering.IsNextLadderSameAsCurrent)
457  {
458  Character.AnimController.TargetMovement = new Vector2(0.0f, Math.Sign(Character.AnimController.TargetMovement.Y));
459  }
460  }
461  Character.AnimController.IgnorePlatforms = ignorePlatforms;
462 
463  if (Math.Abs(Character.AnimController.movement.X) > 0.1f && !Character.AnimController.InWater &&
464  (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer || Character.Controlled == Character))
465  {
466  if (SelectedAiTarget?.Entity != null || EscapeTarget != null)
467  {
468  Entity t = SelectedAiTarget?.Entity ?? EscapeTarget;
469  float referencePos = Vector2.DistanceSquared(Character.WorldPosition, t.WorldPosition) > 100 * 100 && HasValidPath(requireNonDirty: true) ? PathSteering.CurrentPath.CurrentNode.WorldPosition.X : t.WorldPosition.X;
470  Character.AnimController.TargetDir = Character.WorldPosition.X < referencePos ? Direction.Right : Direction.Left;
471  }
472  else
473  {
474  Character.AnimController.TargetDir = Character.AnimController.movement.X > 0.0f ? Direction.Right : Direction.Left;
475  }
476  }
477  if (isStateChanged)
478  {
479  if (State == AIState.Idle || State == AIState.Patrol)
480  {
481  stateResetTimer -= deltaTime;
482  if (stateResetTimer <= 0)
483  {
484  ResetOriginalState();
485  }
486  }
487  }
488  if (targetIgnoreTimer > 0)
489  {
490  targetIgnoreTimer -= deltaTime;
491  }
492  else
493  {
494  ignoredTargets.Clear();
495  targetIgnoreTimer = targetIgnoreTime;
496  }
497  avoidTimer -= deltaTime;
498  if (avoidTimer < 0)
499  {
500  avoidTimer = 0;
501  }
502  UpdateCurrentMemoryLocation();
503  if (updateMemoriesTimer > 0)
504  {
505  updateMemoriesTimer -= deltaTime;
506  }
507  else
508  {
509  FadeMemories(updateMemoriesInverval);
510  updateMemoriesTimer = updateMemoriesInverval;
511  }
512  if (Math.Max(Character.HealthPercentage, 0) < FleeHealthThreshold && SelectedAiTarget != null &&
513  SelectedAiTarget.Entity is Character target && (target.IsHuman && CanPerceive(SelectedAiTarget) || IsBeingChasedBy(target)))
514  {
515  // Keep fleeing if being chased
516  State = AIState.Flee;
517  wallTarget = null;
518  }
519  else
520  {
521  if (updateTargetsTimer > 0)
522  {
523  updateTargetsTimer -= deltaTime;
524  }
525  else if (avoidTimer <= 0 || activeTriggers.Any() && returnTimer <= 0)
526  {
527  UpdateTargets(out CharacterParams.TargetParams targetingParams);
528  updateTargetsTimer = updateTargetsInterval * Rand.Range(0.75f, 1.25f);
529  if (SelectedAiTarget == null)
530  {
531  State = AIState.Idle;
532  }
533  else if (targetingParams != null)
534  {
535  selectedTargetingParams = targetingParams;
536  State = targetingParams.State;
537  }
538  if ((LatchOntoAI == null || !LatchOntoAI.IsAttached || wallTarget != null) &&
539  (State == AIState.Attack || State == AIState.Aggressive || State == AIState.PassiveAggressive))
540  {
541  UpdateWallTarget(requiredHoleCount);
542  }
543  }
544  }
545 
546  if (AIParams.CanOpenDoors)
547  {
548  bool IsCloseEnoughToTargetSub(float threshold) => SelectedAiTarget?.Entity?.Submarine is Submarine sub && sub != null && Vector2.DistanceSquared(Character.WorldPosition, sub.WorldPosition) < MathUtils.Pow(Math.Max(sub.Borders.Size.X, sub.Borders.Size.Y) / 2 + threshold, 2);
549 
550  if (Character.Submarine != null || HasValidPath() && IsCloseEnoughToTargetSub(maxSteeringBuffer) || IsCloseEnoughToTargetSub(steeringBuffer))
551  {
552  if (steeringManager != insideSteering)
553  {
554  insideSteering.Reset();
555  }
556  steeringManager = insideSteering;
557  steeringBuffer += steeringBufferIncreaseSpeed * deltaTime;
558  }
559  else
560  {
561  if (steeringManager != outsideSteering)
562  {
563  outsideSteering.Reset();
564  }
565  steeringManager = outsideSteering;
566  steeringBuffer = minSteeringBuffer;
567  }
568  steeringBuffer = Math.Clamp(steeringBuffer, minSteeringBuffer, maxSteeringBuffer);
569  }
570  else
571  {
572  if (Character.Submarine != null && Character.Params.UsePathFinding)
573  {
574  if (steeringManager != insideSteering)
575  {
576  insideSteering.Reset();
577  }
578  steeringManager = insideSteering;
579  }
580  else
581  {
582  if (steeringManager != outsideSteering)
583  {
584  outsideSteering.Reset();
585  }
586  steeringManager = outsideSteering;
587  }
588  }
589 
590  bool useSteeringLengthAsMovementSpeed = State == AIState.Idle && Character.AnimController.InWater;
591  bool run = false;
592  switch (State)
593  {
594  case AIState.Freeze:
595  SteeringManager.Reset();
596  break;
597  case AIState.Idle:
598  UpdateIdle(deltaTime);
599  break;
600  case AIState.Patrol:
601  UpdatePatrol(deltaTime);
602  break;
603  case AIState.Attack:
604  run = !IsCoolDownRunning || AttackLimb != null && AttackLimb.attack.FullSpeedAfterAttack;
605  UpdateAttack(deltaTime);
606  break;
607  case AIState.Eat:
608  UpdateEating(deltaTime);
609  break;
610  case AIState.Escape:
611  case AIState.Flee:
612  run = true;
613  Escape(deltaTime);
614  break;
615  case AIState.Avoid:
616  case AIState.PassiveAggressive:
617  case AIState.Aggressive:
618  if (SelectedAiTarget == null || SelectedAiTarget.Entity == null || SelectedAiTarget.Entity.Removed)
619  {
620  State = AIState.Idle;
621  return;
622  }
623  float squaredDistance = Vector2.DistanceSquared(WorldPosition, SelectedAiTarget.WorldPosition);
624  var attackLimb = AttackLimb ?? GetAttackLimb(SelectedAiTarget.WorldPosition);
625  if (attackLimb != null && squaredDistance <= Math.Pow(attackLimb.attack.Range, 2))
626  {
627  run = true;
628  if (State == AIState.Avoid)
629  {
630  Escape(deltaTime);
631  }
632  else
633  {
634  UpdateAttack(deltaTime);
635  }
636  }
637  else
638  {
639  bool isBeingChased = IsBeingChased;
640  float reactDistance = !isBeingChased && selectedTargetingParams != null && selectedTargetingParams.ReactDistance > 0 ? selectedTargetingParams.ReactDistance : GetPerceivingRange(SelectedAiTarget);
641  if (squaredDistance <= Math.Pow(reactDistance, 2))
642  {
643  float halfReactDistance = reactDistance / 2;
644  float attackDistance = selectedTargetingParams != null && selectedTargetingParams.AttackDistance > 0 ? selectedTargetingParams.AttackDistance : halfReactDistance;
645  if (State == AIState.Aggressive || State == AIState.PassiveAggressive && squaredDistance < Math.Pow(attackDistance, 2))
646  {
647  run = true;
648  UpdateAttack(deltaTime);
649  }
650  else
651  {
652  run = isBeingChased || squaredDistance < Math.Pow(halfReactDistance, 2);
653  State = AIState.Escape;
654  avoidTimer = AIParams.AvoidTime * 0.5f * Rand.Range(0.75f, 1.25f);
655  }
656  }
657  else
658  {
659  UpdateIdle(deltaTime);
660  }
661  }
662  break;
663  case AIState.Protect:
664  case AIState.Follow:
665  case AIState.FleeTo:
666  if (SelectedAiTarget == null || SelectedAiTarget.Entity == null || SelectedAiTarget.Entity.Removed)
667  {
668  State = AIState.Idle;
669  return;
670  }
671  if (State == AIState.Protect)
672  {
673  if (SelectedAiTarget.Entity is Character targetCharacter)
674  {
675  bool IsValid(Character.Attacker a)
676  {
677  Character c = a.Character;
678  if (c.IsDead || c.Removed) { return false; }
679  if (!Character.IsFriendly(c)) { return true; }
680  if (!c.IsPlayer) { return false; }
681  // Only apply the threshold to players
682  return a.Damage >= selectedTargetingParams.Threshold;
683  }
684  Character attacker = targetCharacter.LastAttackers.LastOrDefault(IsValid)?.Character;
685  //if the attacker has the same targeting tag as the character we're protecting, we can't change the TargetState
686  //otherwise e.g. a pet that's set to follow humans would start attacking all humans (and other pets, since they're considered part of the same group) when a hostile human attacks it
687  //TODO: a way for pets to differentiate hostile and friendly humans?
688  if (attacker?.AiTarget != null && targetCharacter.SpeciesName != GetTargetingTag(attacker.AiTarget))
689  {
690  // Attack the character that attacked the target we are protecting
691  ChangeTargetState(attacker, AIState.Attack, selectedTargetingParams.Priority * 2);
692  SelectTarget(attacker.AiTarget);
693  State = AIState.Attack;
694  UpdateWallTarget(requiredHoleCount);
695  return;
