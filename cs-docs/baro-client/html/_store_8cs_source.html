139 private bool HasPermissionToUseTab(StoreTab tab)
140 {
141 return tab switch
142 {
143 StoreTab.Buy => true,
144 StoreTab.Sell => campaignUI.Campaign.AllowedToManageCampaign(Networking.ClientPermissions.SellInventoryItems),
145 StoreTab.SellSub => campaignUI.Campaign.AllowedToManageCampaign(Networking.ClientPermissions.SellSubItems),
146 _ => false,
147 };
148 }
149
150 private void UpdatePermissions()
151 {
152 HasBuyPermissions = HasPermissionToUseTab(StoreTab.Buy);
153 HasSellInventoryPermissions = HasPermissionToUseTab(StoreTab.Sell);
154 HasSellSubPermissions = HasPermissionToUseTab(StoreTab.SellSub);
155 }
156
157 private bool HasTabPermissions(StoreTab tab)
158 {
159 return tab switch
160 {
161 StoreTab.Buy => HasBuyPermissions,
162 StoreTab.Sell => HasSellInventoryPermissions,
163 StoreTab.SellSub => HasSellSubPermissions,
164 _ => false
165 };
166 }
167
168 private bool HasActiveTabPermissions()
169 {
170 return HasTabPermissions(activeTab);
171 }
172
173 private bool HavePermissionsChanged(StoreTab tab)
174 {
175 bool hadTabPermissions = tab switch
176 {
177 StoreTab.Buy => hadBuyPermissions,
178 StoreTab.Sell => hadSellInventoryPermissions,
179 StoreTab.SellSub => hadSellSubPermissions,
180 _ => false
181 };
182 return hadTabPermissions != HasTabPermissions(tab);
183 }

210 public void SelectStore(Identifier identifier)
211 {
212 if (CurrentLocation?.Stores != null)
213 {
214 if (!identifier.IsEmpty && CurrentLocation.GetStore(identifier) is { } store)
215 {
216 ActiveStore = store;
217 if (storeNameBlock != null)
218 {
219 var storeName = TextManager.Get($"storename.{store.Identifier}");
220 if (storeName.IsNullOrEmpty())
221 {
222 storeName = TextManager.Get("store");
223 }
224 storeNameBlock.SetRichText(storeName);
225 }
226 }
227 else
228 {
229 ActiveStore = null;
230 string errorId, msg;
231 if (identifier.IsEmpty)
232 {
233 errorId = "Store.SelectStore:IdentifierEmpty";
234 msg = $"Error selecting store at {CurrentLocation}: identifier is empty.";
235 }
236 else
237 {
238 errorId = "Store.SelectStore:StoreDoesntExist";
239 msg = $"Error selecting store with identifier \"{identifier}\" at {CurrentLocation}: store with the identifier doesn't exist at the location.";
240 }
241 DebugConsole.ShowError(msg);
242 GameAnalyticsManager.AddErrorEventOnce(errorId, GameAnalyticsManager.ErrorSeverity.Error, msg);
243 }
244 }
245 else
246 {
247 ActiveStore = null;
248 string errorId = "", msg = "";
249 if (campaignUI.Campaign.Map == null)
250 {
251 errorId = "Store.SelectStore:MapNull";
252 msg = $"Error selecting store with identifier \"{identifier}\": Map is null.";
253 }
254 else if (CurrentLocation == null)
255 {
256 errorId = "Store.SelectStore:CurrentLocationNull";
257 msg = $"Error selecting store with identifier \"{identifier}\": CurrentLocation is null.";
258 }
259 else if (CurrentLocation.Stores == null)
260 {
261 errorId = "Store.SelectStore:StoresNull";
262 msg = $"Error selecting store with identifier \"{identifier}\": CurrentLocation.Stores is null.";
263 }
264 if (!msg.IsNullOrEmpty())
265 {
266 DebugConsole.ShowError(msg);
267 GameAnalyticsManager.AddErrorEventOnce(errorId, GameAnalyticsManager.ErrorSeverity.Error, msg);
268 }
269 }
270 RefreshItemsToSell();
271 Refresh();
