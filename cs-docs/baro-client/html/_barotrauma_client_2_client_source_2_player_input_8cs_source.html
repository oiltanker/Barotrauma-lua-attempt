544 public static void Update(double deltaTime)
545 {
546 timeSinceClick += deltaTime;
547
548 if (!GameMain.WindowActive)
549 {
550 wasWindowActive = false;
551 return;
552 }
553
554 //window was not active during the previous frame -> ignore inputs from this frame
555 if (!wasWindowActive)
556 {
557 wasWindowActive = true;
558 allowInput = false;
559 }
560 else
561 {
562 allowInput = true;
563 }
564
565 GameMain.LuaCs.Hook.Call("keyUpdate", deltaTime);
566
567 oldMouseState = mouseState;
568 mouseState = latestMouseState;
569 UpdateVariable();
570
571 oldKeyboardState = keyboardState;
572 keyboardState = Keyboard.GetState();
573
574 MouseSpeedPerSecond = MouseSpeed / (float)deltaTime;
575
576 // Split into two to not accept drag & drop releasing as part of a double-click
577 doubleClicked = false;
578 if (PrimaryMouseButtonClicked())
579 {
580 float dist = (mouseState.Position - lastClickPosition).ToVector2().Length();
581
582 if (timeSinceClick < DoubleClickDelay && dist < MaxDoubleClickDistance)
583 {
584 doubleClicked = true;
585 timeSinceClick = DoubleClickDelay;
586 }
587 else if (timeSinceClick < DoubleClickDelay)
588 {
589 lastClickPosition = mouseState.Position;
590 }
591 if (!doubleClicked && dist < MaxDoubleClickDistance)
592 {
593 timeSinceClick = 0.0;
594 }
595 }
596
597 if (PrimaryMouseButtonDown())
598 {
599 lastClickPosition = mouseState.Position;
600 }
601 }
