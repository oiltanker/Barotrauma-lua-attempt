31 class ContainedItemSprite
32 {
33 public enum DecorativeSpriteBehaviorType
34 {
35 None, HideWhenVisible, HideWhenNotVisible
36 }
37
38 public readonly Sprite Sprite;
39 public readonly bool UseWhenAttached;
40 public readonly DecorativeSpriteBehaviorType DecorativeSpriteBehavior;
41 public readonly ImmutableHashSet<Identifier> AllowedContainerIdentifiers;
42 public readonly ImmutableHashSet<Identifier> AllowedContainerTags;
43
44 public ContainedItemSprite(ContentXElement element, string path = "", bool lazyLoad = false)
45 {
46 Sprite = new Sprite(element, path, lazyLoad: lazyLoad);
47 UseWhenAttached = element.GetAttributeBool("usewhenattached", false);
48 Enum.TryParse(element.GetAttributeString("decorativespritebehavior", "None"), ignoreCase: true, out DecorativeSpriteBehavior);
49 AllowedContainerIdentifiers = element.GetAttributeIdentifierArray("allowedcontaineridentifiers", Array.Empty<Identifier>()).ToImmutableHashSet();
50 AllowedContainerTags = element.GetAttributeIdentifierArray("allowedcontainertags", Array.Empty<Identifier>()).ToImmutableHashSet();
51 }
52
53 public bool MatchesContainer(Item container)
54 {
55 if (container == null) { return false; }
56 return AllowedContainerIdentifiers.Contains(container.Prefab.Identifier) ||
57 AllowedContainerTags.Any(t => container.Prefab.Tags.Contains(t));
58 }

89 private void ParseSubElementsClient(ContentXElement element, ItemPrefab variantOf)
90 {
91 UpgradePreviewSprite = null;
92 UpgradePreviewScale = 1f;
93 InventoryIcon = null;
94 MinimapIcon = null;
95 InfectedSprite = null;
96 DamagedInfectedSprite = null;
97
98 var upgradeOverrideSprites = new Dictionary<Identifier, List<DecorativeSprite>>();
99 var brokenSprites = new List<BrokenItemSprite>();
100 var decorativeSprites = new List<DecorativeSprite>();
101 var containedSprites = new List<ContainedItemSprite>();
102 var decorativeSpriteGroups = new Dictionary<int, List<DecorativeSprite>>();
103
104 foreach (var subElement in element.Elements())
105 {
106 switch (subElement.Name.LocalName.ToLowerInvariant())
107 {
108 case "upgradeoverride":
109 {
110
111 var sprites = new List<DecorativeSprite>();
112 foreach (var decorSprite in subElement.Elements())
113 {
114 if (decorSprite.NameAsIdentifier() == "decorativesprite")
115 {
116 sprites.Add(new DecorativeSprite(decorSprite));
117 }
118 }
119 upgradeOverrideSprites.Add(subElement.GetAttributeIdentifier("identifier", Identifier.Empty), sprites);
120 break;
121 }
122 case "upgradepreviewsprite":
123 {
124 string iconFolder = GetTexturePath(subElement, variantOf);
125 UpgradePreviewSprite = new Sprite(subElement, iconFolder, lazyLoad: true);
126 UpgradePreviewScale = subElement.GetAttributeFloat("scale", 1.0f);
127 }
128 break;
129 case "inventoryicon":
130 {
131 string iconFolder = GetTexturePath(subElement, variantOf);
132 InventoryIcon = new Sprite(subElement, iconFolder, lazyLoad: true);
133 }
134 break;
135 case "minimapicon":
136 {
137 string iconFolder = GetTexturePath(subElement, variantOf);
138 MinimapIcon = new Sprite(subElement, iconFolder, lazyLoad: true);
139 }
140 break;
141 case "infectedsprite":
142 {
143 string iconFolder = GetTexturePath(subElement, variantOf);
144
145 InfectedSprite = new Sprite(subElement, iconFolder, lazyLoad: true);
146 }
147 break;
148 case "damagedinfectedsprite":
149 {
150 string iconFolder = GetTexturePath(subElement, variantOf);
151
152 DamagedInfectedSprite = new Sprite(subElement, iconFolder, lazyLoad: true);
153 }
154 break;
155 case "brokensprite":
156 string brokenSpriteFolder = GetTexturePath(subElement, variantOf);
157
158 var brokenSprite = new BrokenItemSprite(
159 new Sprite(subElement, brokenSpriteFolder, lazyLoad: true),
160 subElement.GetAttributeFloat("maxcondition", 0.0f),
161 subElement.GetAttributeBool("fadein", false),
162 subElement.GetAttributePoint("offset", Point.Zero));
163
164 int spriteIndex = 0;
165 for (int i = 0; i < brokenSprites.Count && brokenSprites[i].MaxConditionPercentage < brokenSprite.MaxConditionPercentage; i++)
166 {
167 spriteIndex = i;
168 }
169 brokenSprites.Insert(spriteIndex, brokenSprite);
170 break;
171 case "decorativesprite":
172 string decorativeSpriteFolder = GetTexturePath(subElement, variantOf);
173
174 int groupID = 0;
175 DecorativeSprite decorativeSprite = null;
176 if (subElement.GetAttribute("texture") == null)
177 {
178 groupID = subElement.GetAttributeInt("randomgroupid", 0);
179 }
180 else
181 {
182 decorativeSprite = new DecorativeSprite(subElement, decorativeSpriteFolder, lazyLoad: true);
183 decorativeSprites.Add(decorativeSprite);
184 groupID = decorativeSprite.RandomGroupID;
185 }
186 if (!decorativeSpriteGroups.ContainsKey(groupID))
187 {
188 decorativeSpriteGroups.Add(groupID, new List<DecorativeSprite>());
189 }
190 decorativeSpriteGroups[groupID].Add(decorativeSprite);
191
192 break;
193 case "containedsprite":
194 string containedSpriteFolder = GetTexturePath(subElement, variantOf);
195 var containedSprite = new ContainedItemSprite(subElement, containedSpriteFolder, lazyLoad: true);
196 if (containedSprite.Sprite != null)
197 {
198 containedSprites.Add(containedSprite);
199 }
200 break;
201 }
202 }
203
204 UpgradeOverrideSprites = upgradeOverrideSprites.Select(kvp => (kvp.Key, kvp.Value.ToImmutableArray())).ToImmutableDictionary();
205 BrokenSprites = brokenSprites.ToImmutableArray();
206 DecorativeSprites = decorativeSprites.ToImmutableArray();
207 ContainedSprites = containedSprites.ToImmutableArray();
208 DecorativeSpriteGroups = decorativeSpriteGroups.Select(kvp => (kvp.Key, kvp.Value.ToImmutableArray())).ToImmutableDictionary();
209 }

211 public override void UpdatePlacing(Camera cam)
212 {
213 Vector2 position = Submarine.MouseToWorldGrid(cam, Submarine.MainSub);
214
215 if (PlayerInput.SecondaryMouseButtonClicked())
216 {
217 Selected = null;
218 return;
219 }
220
221 var potentialContainer = MapEntity.GetPotentialContainer(position);
222
223 if (!ResizeHorizontal && !ResizeVertical)
224 {
225 if (PlayerInput.PrimaryMouseButtonClicked())
226 {
227 var item = new Item(new Rectangle((int)position.X, (int)position.Y, (int)(Sprite.size.X * Scale), (int)(Sprite.size.Y * Scale)), this, Submarine.MainSub)
228 {
229 Submarine = Submarine.MainSub
230 };
231 item.SetTransform(ConvertUnits.ToSimUnits(Submarine.MainSub == null ? item.Position : item.Position - Submarine.MainSub.Position), 0.0f);
232 item.FindHull();
233 item.Submarine = Submarine.MainSub;
234
235 if (PlayerInput.IsShiftDown())
236 {
237 if (potentialContainer?.OwnInventory?.TryPutItem(item, Character.Controlled) ?? false)
238 {
239 SoundPlayer.PlayUISound(GUISoundType.PickItem);
240 }
241 }
242
243 SubEditorScreen.StoreCommand(new AddOrDeleteCommand(new List<MapEntity> {item}, false));
244
245 placePosition = Vector2.Zero;
246 return;
247 }
248 }
249 else
250 {
251 Vector2 placeSize = Size * Scale;
252
253 if (placePosition == Vector2.Zero)
254 {
255 if (PlayerInput.PrimaryMouseButtonHeld()) placePosition = position;
256 }
257 else
258 {
259 if (ResizeHorizontal)
260 placeSize.X = Math.Max(position.X - placePosition.X, Size.X);
261 if (ResizeVertical)
262 placeSize.Y = Math.Max(placePosition.Y - position.Y, Size.Y);
263
264 if (PlayerInput.PrimaryMouseButtonReleased())
265 {
266 var item = new Item(new Rectangle((int)placePosition.X, (int)placePosition.Y, (int)placeSize.X, (int)placeSize.Y), this, Submarine.MainSub);
267 placePosition = Vector2.Zero;
268
269 item.Submarine = Submarine.MainSub;
270 item.SetTransform(ConvertUnits.ToSimUnits(Submarine.MainSub == null ? item.Position : item.Position - Submarine.MainSub.Position), 0.0f);
271 item.FindHull();
272
273 //selected = null;
274 return;
275 }
276
277 position = placePosition;
278 }
279 }
280
281 if (potentialContainer != null)
282 {
283 potentialContainer.IsHighlighted = true;
284 }
285
286
287 //if (PlayerInput.GetMouseState.RightButton == ButtonState.Pressed) selected = null;
288
289 }

291 public override void DrawPlacing(SpriteBatch spriteBatch, Camera cam)
292 {
293 Vector2 position = Submarine.MouseToWorldGrid(cam, Submarine.MainSub);
294
295 if (PlayerInput.SecondaryMouseButtonClicked())
296 {
297 Selected = null;
298 return;
299 }
300
301 if (!ResizeHorizontal && !ResizeVertical)
302 {
303 Sprite.Draw(spriteBatch, new Vector2(position.X, -position.Y) + Sprite.size / 2.0f * Scale, SpriteColor, scale: Scale);
304 }
305 else
306 {
307 Vector2 placeSize = Size * Scale;
308 if (placePosition != Vector2.Zero)
309 {
310 if (ResizeHorizontal) { placeSize.X = Math.Max(position.X - placePosition.X, placeSize.X); }
311 if (ResizeVertical) { placeSize.Y = Math.Max(placePosition.Y - position.Y, placeSize.Y); }
312 position = placePosition;
313 }
314 Sprite?.DrawTiled(spriteBatch, new Vector2(position.X, -position.Y), placeSize, color: SpriteColor);
315 }
316 }
