251  velocity.X += velocityChange.X * deltaTime;
252  velocity.Y += velocityChange.Y * deltaTime;
253  if (prefab.Drag > 0.0f)

255  ApplyDrag(prefab.Drag, deltaTime);

271  }
272  }

332  private UpdateResult CollisionUpdate()
333  {
334  if (currentHull == null)
335  {
336  Hull collidedHull = Hull.FindHull(position);
337  if (collidedHull != null)
338  {
339  if (prefab.DeleteOnCollision) { return UpdateResult.Delete; }
340  OnWallCollisionOutside(collidedHull);
341  }
342  }
343  else
344  {
345  Rectangle hullRect = currentHull.WorldRect;
346  Vector2 collisionNormal = Vector2.Zero;
347  if (velocity.Y < 0.0f && position.Y - prefab.CollisionRadius * size.Y < hullRect.Y - hullRect.Height)
348  {
349  collisionNormal = new Vector2(0.0f, 1.0f);
350  }
351  else if (velocity.Y > 0.0f && position.Y + prefab.CollisionRadius * size.Y > hullRect.Y)
352  {
353  collisionNormal = new Vector2(0.0f, -1.0f);
354  }
355 
356  if (collisionNormal != Vector2.Zero)
357  {
358  bool gapFound = false;
359  foreach (Gap gap in hullGaps)
360  {
361  if (gap.Open <= 0.9f || gap.IsHorizontal) { continue; }
362 
363  if (gap.WorldRect.X > position.X || gap.WorldRect.Right < position.X) { continue; }
364  float hullCenterY = currentHull.WorldRect.Y - currentHull.WorldRect.Height / 2;
365  int gapDir = Math.Sign(gap.WorldRect.Y - hullCenterY);
366  if (Math.Sign(velocity.Y) != gapDir || Math.Sign(position.Y - hullCenterY) != gapDir) { continue; }
367 
368  gapFound = true;
369  break;
370  }
371 
372  if (prefab.DeleteOnCollision && !gapFound)
373  {
374  OnCollision?.Invoke(position, currentHull);
375  return UpdateResult.Delete;
376  }
377  handleCollision(gapFound, collisionNormal);
378  }
379 
380  collisionNormal = Vector2.Zero;
381  if (velocity.X < 0.0f && position.X - prefab.CollisionRadius * size.X < hullRect.X)
382  {
383  collisionNormal = new Vector2(1.0f, 0.0f);
384  }
385  else if (velocity.X > 0.0f && position.X + prefab.CollisionRadius * size.X > hullRect.Right)
386  {
387  collisionNormal = new Vector2(-1.0f, 0.0f);
388  }
389 
390  if (collisionNormal != Vector2.Zero)
391  {
392  bool gapFound = false;
393  foreach (Gap gap in hullGaps)
394  {
395  if (gap.Open <= 0.9f || !gap.IsHorizontal) { continue; }
396 
397  if (gap.WorldRect.Y < position.Y || gap.WorldRect.Y - gap.WorldRect.Height > position.Y) { continue; }
398  int gapDir = Math.Sign(gap.WorldRect.Center.X - currentHull.WorldRect.Center.X);
399  if (Math.Sign(velocity.X) != gapDir || Math.Sign(position.X - currentHull.WorldRect.Center.X) != gapDir) { continue; }
400 
401  gapFound = true;
402  break;
403  }
404  if (prefab.DeleteOnCollision && !gapFound)
405  {
406  OnCollision?.Invoke(position, currentHull);
407  return UpdateResult.Delete;
408  }
409  handleCollision(gapFound, collisionNormal);
410  }

434  private void ApplyDrag(float dragCoefficient, float deltaTime)
435  {
436  Vector2 relativeVel = velocity;
437  if (currentHull?.Submarine != null)
438  {
439  relativeVel = velocity - ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);
440  }
441 
442  float speed = relativeVel.Length();
443 
444  relativeVel /= speed;
445 
446  float drag = speed * speed * dragCoefficient * 0.01f * deltaTime;
447  if (drag > speed)
448  {
449  relativeVel = Vector2.Zero;
450  }
451  else
452  {
453  speed -= drag;
454  relativeVel *= speed;
455  }
456 
457  velocity = relativeVel;
458  if (currentHull?.Submarine != null)
459  {
460  velocity += ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);
461  }
462  }

465  private void OnWallCollisionInside(Hull prevHull, Vector2 collisionNormal)
466  {
467  if (prevHull == null) { return; }
468 
469  Rectangle prevHullRect = prevHull.WorldRect;
470 
471  Vector2 subVel = prevHull?.Submarine != null ? ConvertUnits.ToDisplayUnits(prevHull.Submarine.Velocity) : Vector2.Zero;
472  velocity -= subVel;
473 
474  if (Math.Abs(collisionNormal.X) > Math.Abs(collisionNormal.Y))
475  {
476  if (collisionNormal.X > 0.0f)
477  {
478  position.X = Math.Max(position.X, prevHullRect.X + prefab.CollisionRadius * size.X);
479  }
480  else
481  {
482  position.X = Math.Min(position.X, prevHullRect.Right - prefab.CollisionRadius * size.X);
483  }
484  velocity.X = Math.Sign(collisionNormal.X) * Math.Abs(velocity.X) * prefab.Restitution;
485  velocity.Y *= (1.0f - prefab.Friction);
486  }
487  else
488  {
489  if (collisionNormal.Y > 0.0f)
490  {
491  position.Y = Math.Max(position.Y, prevHullRect.Y - prevHullRect.Height + prefab.CollisionRadius * size.Y);
492  }
493  else
494  {
495  position.Y = Math.Min(position.Y, prevHullRect.Y - prefab.CollisionRadius * size.Y);
496 
497  }
498  velocity.X *= (1.0f - prefab.Friction);
499  velocity.Y = Math.Sign(collisionNormal.Y) * Math.Abs(velocity.Y) * prefab.Restitution;
500  }
501 
502  OnCollision?.Invoke(position, currentHull);
503 
504  velocity += subVel;
505  }

508  private void OnWallCollisionOutside(Hull collisionHull)
509  {
510  Rectangle hullRect = collisionHull.WorldRect;
511 
512  Vector2 center = new Vector2(hullRect.X + hullRect.Width / 2, hullRect.Y - hullRect.Height / 2);
513 
514  if (position.Y < center.Y)
515  {
516  position.Y = hullRect.Y - hullRect.Height - prefab.CollisionRadius;
517  velocity.X *= (1.0f - prefab.Friction);
518  velocity.Y = -velocity.Y * prefab.Restitution;
519  }
520  else if (position.Y > center.Y)
521  {
522  position.Y = hullRect.Y + prefab.CollisionRadius;
523  velocity.X *= (1.0f - prefab.Friction);
524  velocity.Y = -velocity.Y * prefab.Restitution;
525  }
526 
527  if (position.X < center.X)
528  {
529  position.X = hullRect.X - prefab.CollisionRadius;
530  velocity.X = -velocity.X * prefab.Restitution;
531  velocity.Y *= (1.0f - prefab.Friction);
532  }
533  else if (position.X > center.X)
534  {
535  position.X = hullRect.X + hullRect.Width + prefab.CollisionRadius;
536  velocity.X = -velocity.X * prefab.Restitution;
537  velocity.Y *= (1.0f - prefab.Friction);
538  }
539

541
