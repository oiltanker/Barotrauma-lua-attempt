117 public Node? AddNodeAndInheritors(Identifier id)
118 {
119 if (!prefabCollection.TryGet(id, out T? prefab)) { return null; }
120
121 if (!IdToNode.TryGetValue(id, out var node))
122 {
123 node = new Node(id);
124 RootNodes.Add(node);
125 IdToNode.Add(id, node);
126 }
127 else
128 {
129 //if the node already exists, it already contains
130 //all inheritors so let's just return this immediately
131 return node;
132 }
133
134 prefabCollection
135 .Cast<IImplementsVariants<T>>()
136 .Where(p => p.VariantOf == id)
137 .Cast<T>()
138 .ForEach(p =>
139 {
140 var inheritorNode = AddNodeAndInheritors(p.Identifier);
141 if (inheritorNode is null) { return; }
142 RootNodes.Remove(inheritorNode);
143 inheritorNode.Parent = node;
144 node.Inheritors.Add(inheritorNode);
145 });
146
147 return node;
148 }

150 private void FindCycles(in Node node, HashSet<Node> uncheckedNodes)
151 {
152 HashSet<Node> checkedNodes = new HashSet<Node>();
153 List<Node> hierarchyPositions = new List<Node>();
154 Node? currNode = node;
155 do
156 {
157 if (!uncheckedNodes.Contains(currNode)) { break; }
158 if (checkedNodes.Contains(currNode))
159 {
160 int index = hierarchyPositions.IndexOf(currNode);
161 throw new Exception("Inheritance cycle detected: "
162 +string.Join(", ", hierarchyPositions.Skip(index).Select(n => n.Identifier)));
163 }
164 checkedNodes.Add(currNode);
165 hierarchyPositions.Add(currNode);
166 currNode = currNode.Parent;
167 } while (currNode != null);
168 uncheckedNodes.RemoveWhere(i => checkedNodes.Contains(i));
169 }

171 public void AddNodesAndInheritors(IEnumerable<Identifier> ids)
172 => ids.ForEach(id => AddNodeAndInheritors(id));

174 public void InvokeCallbacks()
175 {
176 HashSet<Node> uncheckedNodes = IdToNode.Values.ToHashSet();
177 IdToNode.Values.ForEach(v => FindCycles(v, uncheckedNodes));
178 void invokeCallbacksForNode(Node node)
179 {
180 if (!prefabCollection.TryGet(node.Identifier, out var p) ||
181 !(p is IImplementsVariants<T> prefab)) { return; }
182 if (!prefab.VariantOf.IsEmpty && prefabCollection.TryGet(prefab.VariantOf, out T? parent)) { prefab.InheritFrom(parent!); }
183 node.Inheritors.ForEach(invokeCallbacksForNode);
184 }
185 RootNodes.ForEach(invokeCallbacksForNode);
186 }

327 public void Add(T prefab, bool isOverride)
328 {
329 Prefab.DisallowCallFromConstructor();
330 if (prefab.Identifier.IsEmpty)
331 {
332 throw new ArgumentException($"Prefab has no identifier!");
333 }
334
335 bool selectorExists = prefabs.TryGetValue(prefab.Identifier, out PrefabSelector<T>? selector);
336
337 //Add to list
338 selector ??= new PrefabSelector<T>();
339
340 if (prefab is PrefabWithUintIdentifier prefabWithUintIdentifier)
341 {
342 if (!selector.IsEmpty)
343 {
344 prefabWithUintIdentifier.UintIdentifier = (selector.ActivePrefab as PrefabWithUintIdentifier)!.UintIdentifier;
345 }
346 else
347 {
348 using (MD5 md5 = MD5.Create())
349 {
350 prefabWithUintIdentifier.UintIdentifier = ToolBox.IdentifierToUint32Hash(prefab.Identifier, md5);
351
352 //it's theoretically possible for two different values to generate the same hash, but the probability is astronomically small
353 T? findCollision()
354 => Find(p =>
355 p.Identifier != prefab.Identifier
356 && p is PrefabWithUintIdentifier otherPrefab
357 && otherPrefab.UintIdentifier == prefabWithUintIdentifier.UintIdentifier);
358 for (T? collision = findCollision(); collision != null; collision = findCollision())
359 {
360 DebugConsole.ThrowError($"Hashing collision when generating uint identifiers for {typeof(T).Name}: {prefab.Identifier} has the same UintIdentifier as {collision.Identifier} ({prefabWithUintIdentifier.UintIdentifier})");
361 prefabWithUintIdentifier.UintIdentifier++;
362 }
363 }
364 }
365 }
366 selector.Add(prefab, isOverride);
367
368 if (!selectorExists)
369 {
370 if (!prefabs.TryAdd(prefab.Identifier, selector)) { throw new Exception($"Failed to add selector for \"{prefab.Identifier}\""); }
371 }
372 OnAdd?.Invoke(prefab, isOverride);
373 HandleInheritance(prefab.Identifier);
374 }
