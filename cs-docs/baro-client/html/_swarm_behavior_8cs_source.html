67  public void UpdateSteering(float deltaTime)
68  {
69  if (!IsActive) { return; }
70  if (!IsEnoughMembers) { return; }
71  //calculate the "center of mass" of the swarm and the distance to the closest character in the swarm
72  float closestDistSqr = float.MaxValue;
73  Vector2 center = Vector2.Zero;
74  AICharacter closest = null;
75  foreach (AICharacter member in Members)
76  {
77  center += member.SimPosition;
78  if (member == ai.Character) { continue; }
79  float distSqr = Vector2.DistanceSquared(member.SimPosition, ai.Character.SimPosition);
80  if (distSqr < closestDistSqr)
81  {
82  closestDistSqr = distSqr;
83  closest = member;
84  }
85  }
86  center /= Members.Count;
87 
88  if (closest == null) { return; }
89 
90  //steer away from the closest if too close
91  float closestDist = (float)Math.Sqrt(closestDistSqr);
92  if (closestDist < minDistFromClosest)
93  {
94  Vector2 diff = closest.SimPosition - ai.SimPosition;
95  if (diff.LengthSquared() < 0.0001f)
96  {
97  diff = Vector2.UnitX;
98  }
99  ai.SteeringManager.SteeringManual(deltaTime, -diff);
100  }
101  //steer closer to the center of mass if too far
102  else if (Vector2.DistanceSquared(center, ai.SimPosition) > maxDistFromCenter * maxDistFromCenter)
103  {
104  float distFromCenter = Vector2.Distance(center, ai.SimPosition);
105  ai.SteeringManager.SteeringSeek(center, (distFromCenter - maxDistFromCenter) / 10.0f);
106  }
107 
108  //keep the characters moving in roughly the same direction
109  if (cohesion > 0.0f)
110  {
111  Vector2 avgVel = Vector2.Zero;
112  foreach (AICharacter member in Members)
113  {
114  avgVel += member.AnimController.TargetMovement;
115  }
116  avgVel /= Members.Count;
117  ai.SteeringManager.SteeringManual(deltaTime, avgVel * cohesion);
118  }
119  }
120  }
