160  public static void Apply(GUIComponent targetComponent, Identifier styleName, GUIComponent parent = null)
161  {
162  GUIComponentStyle componentStyle = null;
163  if (parent != null)
164  {
165  GUIComponentStyle parentStyle = parent.Style;
166 
167  if (parentStyle == null)
168  {
169  Identifier parentStyleName = parent.GetType().Name.ToIdentifier();
170 
171  if (!ComponentStyles.ContainsKey(parentStyleName))
172  {
173  DebugConsole.ThrowError($"Couldn't find a GUI style \"{parentStyleName}\"");
174  return;
175  }
176  parentStyle = ComponentStyles[parentStyleName];
177  }
178  Identifier childStyleName = styleName.IsEmpty ? targetComponent.GetType().Name.ToIdentifier() : styleName;
179  parentStyle.ChildStyles.TryGetValue(childStyleName, out componentStyle);
180  }
181  else
182  {
183  Identifier styleIdentifier = styleName.ToIdentifier();
184  if (styleIdentifier == Identifier.Empty)
185  {
186  styleIdentifier = targetComponent.GetType().Name.ToIdentifier();
187  }
188  if (!ComponentStyles.ContainsKey(styleIdentifier))
189  {
190  DebugConsole.ThrowError($"Couldn't find a GUI style \"{styleIdentifier}\"");
191  return;
192  }
193  componentStyle = ComponentStyles[styleIdentifier];
194  }
195 
196  targetComponent.ApplyStyle(componentStyle);
197  }
