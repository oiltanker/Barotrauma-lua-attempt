57  public CustomInterfaceElement(Item item, ContentXElement element, CustomInterface parent)
58  {
59  Label = element.GetAttributeString("text", "");
60  ConnectionName = element.GetAttributeString("connection", "");
61  PropertyName = element.GetAttributeIdentifier("propertyname", "");
62  TargetOnlyParentProperty = element.GetAttributeBool("targetonlyparentproperty", false);
63  NumberInputMin = element.GetAttributeInt("min", DefaultNumberInputMin);
64  NumberInputMax = element.GetAttributeInt("max", DefaultNumberInputMax);
65  MaxTextLength = element.GetAttributeInt("maxtextlength", int.MaxValue);
66  HasPropertyName = !PropertyName.IsEmpty;
67  IsIntegerInput = HasPropertyName && element.Name.ToString().ToLowerInvariant() == "integerinput";
68 
69  if (element.GetAttribute("signal") is XAttribute attribute)
70  {
71  Signal = attribute.Value;
72  ShouldSetProperty = HasPropertyName;
73  }
74  else if (HasPropertyName && parent != null)
75  {
76  if (TargetOnlyParentProperty)
77  {
78  if (parent.SerializableProperties.ContainsKey(PropertyName))
79  {
80  Signal = parent.SerializableProperties[PropertyName].GetValue(parent) as string;
81  }
82  }
83  else
84  {
85  foreach (ISerializableEntity e in parent.item.AllPropertyObjects)
86  {
87  if (!e.SerializableProperties.ContainsKey(PropertyName)) { continue; }
88  Signal = e.SerializableProperties[PropertyName].GetValue(e) as string;
89  break;
90  }
91  }
92  }
93  else
94  {
95  Signal = "1";
96  }
97 
98  foreach (var subElement in element.Elements())
99  {
100  if (subElement.Name.ToString().Equals("statuseffect", System.StringComparison.OrdinalIgnoreCase))
101  {
102  StatusEffects.Add(StatusEffect.Load(subElement, parentDebugName: "custom interface element (label " + Label + ")"));
103  }
104  }
105  }

146  public CustomInterface(Item item, ContentXElement element)
147  : base(item, element)
148  {
149  foreach (var subElement in element.Elements())
150  {
151  switch (subElement.Name.ToString().ToLowerInvariant())
152  {
153  case "button":
154  case "textbox":
155  case "integerinput":
156  var button = new CustomInterfaceElement(item, subElement, this)
157  {
158  ContinuousSignal = false
159  };
160  if (string.IsNullOrEmpty(button.Label))
161  {
162  button.Label = "Signal out " + customInterfaceElementList.Count(e => !e.ContinuousSignal);
163  }
164  customInterfaceElementList.Add(button);
165  break;
166  case "tickbox":
167  var tickBox = new CustomInterfaceElement(item, subElement, this)
168  {
169  ContinuousSignal = true
170  };
171  if (string.IsNullOrEmpty(tickBox.Label))
172  {
173  tickBox.Label = "Signal out " + customInterfaceElementList.Count(e => e.ContinuousSignal);
174  }
175  customInterfaceElementList.Add(tickBox);
176  break;
177  }

203  private void UpdateSignals(string[] newSignals)
204  {
205  signals = new string[customInterfaceElementList.Count];
206  for (int i = 0; i < customInterfaceElementList.Count; i++)
207  {
208  var element = customInterfaceElementList[i];
209  if (i < newSignals.Length)
210  {
211  var newSignal = newSignals[i];
212  signals[i] = newSignal;
213  element.ShouldSetProperty = element.Signal != newSignal;
214  element.Signal = newSignal;
215  }
216  else
217  {
218  signals[i] = element.Signal;
219  }
220 
221  if (element.HasPropertyName && element.ShouldSetProperty)
222  {
223  if (element.TargetOnlyParentProperty)
224  {
225  if (SerializableProperties.ContainsKey(element.PropertyName))
226  {
227  SerializableProperties[element.PropertyName].TrySetValue(this, element.Signal);
228  }
229  }
230  else
231  {
232  foreach (var po in item.AllPropertyObjects)
233  {
234  if (!po.SerializableProperties.ContainsKey(element.PropertyName)) { continue; }
235  po.SerializableProperties[element.PropertyName].TrySetValue(po, element.Signal);
236  }
237  }
238  customInterfaceElementList[i].ShouldSetProperty = false;
239  }
240  }
241  UpdateSignalsProjSpecific();
242  }
