176  public override void Update(float deltaTime, Camera cam)
177  {
178  string signalOut = MotionDetected ? Output : FalseOutput;
179 
180  if (!string.IsNullOrEmpty(signalOut)) { item.SendSignal(new Signal(signalOut, 1), "state_out"); }
181 
182  updateTimer -= deltaTime;
183  if (updateTimer > 0.0f) { return; }
184 
185  MotionDetected = false;
186  updateTimer = UpdateInterval;
187 
188  if (item.body != null && item.body.Enabled && DetectOwnMotion)
189  {
190  if (Math.Abs(item.body.LinearVelocity.X) > MinimumVelocity || Math.Abs(item.body.LinearVelocity.Y) > MinimumVelocity)
191  {
192  MotionDetected = true;
193  return;
194  }
195  }
196 
197  Vector2 detectPos = item.WorldPosition + TransformedDetectOffset;
198  Rectangle detectRect = new Rectangle((int)(detectPos.X - rangeX), (int)(detectPos.Y - rangeY), (int)(rangeX * 2), (int)(rangeY * 2));
199  float broadRangeX = Math.Max(rangeX * 2, 500);
200  float broadRangeY = Math.Max(rangeY * 2, 500);
201 
202  if (item.CurrentHull == null && item.Submarine != null &&
203  (Target == TargetType.Wall || Target == TargetType.Any))
204  {
205  if (Level.Loaded != null && (Math.Abs(item.Submarine.Velocity.X) > MinimumVelocity || Math.Abs(item.Submarine.Velocity.Y) > MinimumVelocity))
206  {
207  var cells = Level.Loaded.GetCells(item.WorldPosition, 1);
208  foreach (var cell in cells)
209  {
210  if (cell.IsPointInside(item.WorldPosition))
211  {
212  MotionDetected = true;
213  return;
214  }
215  foreach (var edge in cell.Edges)
216  {
217  Vector2 e1 = edge.Point1 + cell.Translation;
218  Vector2 e2 = edge.Point2 + cell.Translation;
219  if (MathUtils.LinesIntersect(e1, e2, new Vector2(detectRect.X, detectRect.Y), new Vector2(detectRect.Right, detectRect.Y)) ||
220  MathUtils.LinesIntersect(e1, e2, new Vector2(detectRect.X, detectRect.Bottom), new Vector2(detectRect.Right, detectRect.Bottom)) ||
221  MathUtils.LinesIntersect(e1, e2, new Vector2(detectRect.X, detectRect.Y), new Vector2(detectRect.X, detectRect.Bottom)) ||
222  MathUtils.LinesIntersect(e1, e2, new Vector2(detectRect.Right, detectRect.Y), new Vector2(detectRect.Right, detectRect.Bottom)))
223  {
224  MotionDetected = true;
225  return;
226  }
227  }
228  }
229  }
230  foreach (Submarine sub in Submarine.Loaded)
231  {
232  if (sub == item.Submarine) { continue; }
233 
234  Vector2 relativeVelocity = item.Submarine.Velocity - sub.Velocity;
235  if (Math.Abs(relativeVelocity.X) < MinimumVelocity && Math.Abs(relativeVelocity.Y) < MinimumVelocity) { continue; }
236 
237  Rectangle worldBorders = new Rectangle(
238  sub.Borders.X + (int)sub.WorldPosition.X,
239  sub.Borders.Y + (int)sub.WorldPosition.Y - sub.Borders.Height,
240  sub.Borders.Width,
241  sub.Borders.Height);
242 
243  if (worldBorders.Intersects(detectRect))
244  {
245  MotionDetected = true;
246  return;
247  }
248  }
249  }
250 
251  if (Target != TargetType.Wall)
252  {
253  foreach (Character c in Character.CharacterList)
254  {
255  if (IgnoreDead && c.IsDead) { continue; }
256 
257  //ignore characters that have spawned a second or less ago
258  //makes it possible to detect when a spawned character moves without triggering the detector immediately as the ragdoll spawns and drops to the ground
259  if (c.SpawnTime > Timing.TotalTime - 1.0) { continue; }
260 
261  switch (Target)
262  {
263  case TargetType.Human:
264  if (!c.IsHuman) { continue; }
265  break;
266  case TargetType.Monster:
267  if (c.IsHuman || c.IsPet) { continue; }
268  break;
269  }
270 
271  //do a rough check based on the position of the character's collider first
272  //before the more accurate limb-based check
273  if (Math.Abs(c.WorldPosition.X - detectPos.X) > broadRangeX || Math.Abs(c.WorldPosition.Y - detectPos.Y) > broadRangeY)
274  {
275  continue;
276  }
277 
278  foreach (Limb limb in c.AnimController.Limbs)
279  {
280  if (limb.IsSevered) { continue; }
281  if (limb.LinearVelocity.LengthSquared() < MinimumVelocity * MinimumVelocity) { continue; }
282  if (MathUtils.CircleIntersectsRectangle(limb.WorldPosition, ConvertUnits.ToDisplayUnits(limb.body.GetMaxExtent()), detectRect))
283  {
284  MotionDetected = true;
285  return;
286  }
287  }
288  }
289  }
290  }
