172 public override bool Select(Character character)
173 {
174 // TODO, This works fine as of now but if GUI.PreventElementOverlap ever gets fixed this block of code may become obsolete or detrimental.
175 // Only do this if there's only one linked component. If you link more containers then may
176 // GUI.PreventElementOverlap have mercy on your HUD layout
177 if (GuiFrame != null && item.linkedTo.Count(entity => entity is Item { DisplaySideBySideWhenLinked: true }) == 1)
178 {
179 foreach (MapEntity linkedTo in item.linkedTo)
180 {
181 if (!(linkedTo is Item { DisplaySideBySideWhenLinked: true } linkedItem)) { continue; }
182 if (!linkedItem.Components.Any()) { continue; }
183
184 var itemContainer = linkedItem.GetComponent<ItemContainer>();
185 if (itemContainer?.GuiFrame == null || itemContainer.AllowUIOverlap) { continue; }
186
187 // how much spacing do we want between the components
188 var padding = (int) (8 * GUI.Scale);
189 // Move the linked container to the right and move the deconstructor to the left
190 itemContainer.GuiFrame.RectTransform.AbsoluteOffset = new Point(GuiFrame.Rect.Width / -2 - padding, 0);
191 GuiFrame.RectTransform.AbsoluteOffset = new Point(itemContainer.GuiFrame.Rect.Width / 2 + padding, 0);
192 }
193 }
194 return base.Select(character);
195 }
