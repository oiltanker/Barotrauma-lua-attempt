301

340  for (int i = Connections.Count - 1; i >= 0; i--)

367  for (int i = Locations.Count - 1; i >= 0; i--)
368  {
369  for (int j = Locations.Count - 1; j > i; j--)

447  connectionsBetweenZones[zone1].Add(connection);

468  }

584  private void CreateEndLocation()
585  {
586  float zoneWidth = Width / generationParams.DifficultyZones;
587  Vector2 endPos = new Vector2(Width - zoneWidth / 2, Height / 2);
588  float closestDist = float.MaxValue;
589  EndLocation = Locations.First();
590  foreach (Location location in Locations)
591  {
592  float dist = Vector2.DistanceSquared(endPos, location.MapPosition);
593  if (location.Biome.IsEndBiome && dist < closestDist)
594  {
595  EndLocation = location;
596  closestDist = dist;
597  }
598  }
599 
600  Location previousToEndLocation = null;
601  foreach (Location location in Locations)
602  {
603  if (!location.Biome.IsEndBiome && (previousToEndLocation == null || location.MapPosition.X > previousToEndLocation.MapPosition.X))
604  {
605  previousToEndLocation = location;
606  }
607  }
608 
609  if (EndLocation == null || previousToEndLocation == null) { return; }
610 
611  //remove all locations from the end biome except the end location
612  for (int i = Locations.Count - 1; i >= 0; i--)
613  {
614  if (Locations[i].Biome.IsEndBiome && Locations[i] != EndLocation)
615  {
616  for (int j = Locations[i].Connections.Count - 1; j >= 0; j--)
617  {
618  if (j >= Locations[i].Connections.Count) { continue; }
619  var connection = Locations[i].Connections[j];
620  var otherLocation = connection.OtherLocation(Locations[i]);
621  Locations[i].Connections.RemoveAt(j);
622  otherLocation?.Connections.Remove(connection);
623  Connections.Remove(connection);
624  }
625  Locations.RemoveAt(i);
626  }
627  }
628 
629  //removed all connections from the second-to-last location, need to reconnect it
630  if (!previousToEndLocation.Connections.Any())
631  {
632  Location connectTo = Locations.First();
633  foreach (Location location in Locations)
634  {
635  if (!location.Biome.IsEndBiome && location != previousToEndLocation && location.MapPosition.X > connectTo.MapPosition.X)
636  {
637  connectTo = location;
638  }
639  }
640  var newConnection = new LocationConnection(previousToEndLocation, connectTo)
641  {
642  Biome = EndLocation.Biome,
643  Difficulty = 100.0f
644  };
645  Connections.Add(newConnection);
646  previousToEndLocation.Connections.Add(newConnection);
647  connectTo.Connections.Add(newConnection);
648  }

891  private bool ProgressLocationTypeChanges(Location location)
892  {
893  location.TimeSinceLastTypeChange++;
894  location.LocationTypeChangeCooldown--;
895 
896  if (location.PendingLocationTypeChange != null)
897  {
898  if (location.PendingLocationTypeChange.Value.typeChange.DetermineProbability(location) <= 0.0f)
899  {
900  //remove pending type change if it's no longer allowed
901  location.PendingLocationTypeChange = null;
902  }
903  else
904  {
905  location.PendingLocationTypeChange =
906  (location.PendingLocationTypeChange.Value.typeChange,
907  location.PendingLocationTypeChange.Value.delay - 1,
908  location.PendingLocationTypeChange.Value.parentMission);
909  if (location.PendingLocationTypeChange.Value.delay <= 0)
910  {
911  return ChangeLocationType(location, location.PendingLocationTypeChange.Value.typeChange);
912  }
913  }
914  }
915 
916  //find which types of locations this one can change to
917  Dictionary<LocationTypeChange, float> allowedTypeChanges = new Dictionary<LocationTypeChange, float>();
918  foreach (LocationTypeChange typeChange in location.Type.CanChangeTo)
919  {
920  float probability = typeChange.DetermineProbability(location);
921  if (probability <= 0.0f) { continue; }
922  allowedTypeChanges.Add(typeChange, probability);
923  }
924 
925  //select a random type change
926  if (Rand.Range(0.0f, 1.0f) < allowedTypeChanges.Sum(change => change.Value))
927  {

929  ToolBox.SelectWeightedRandom(
930  allowedTypeChanges.Keys.ToList(),
931  allowedTypeChanges.Values.ToList(),
932  Rand.RandSync.Unsynced);
933  if (selectedTypeChange != null)
934  {
935  if (selectedTypeChange.RequiredDurationRange.X > 0)
936  {
937  location.PendingLocationTypeChange =
938  (selectedTypeChange,
939  Rand.Range(selectedTypeChange.RequiredDurationRange.X, selectedTypeChange.RequiredDurationRange.Y),
940  null);
941  }
942  else
943  {
944  return ChangeLocationType(location, selectedTypeChange);
945  }
946  return false;
947  }
948  }
949 
950  foreach (LocationTypeChange typeChange in location.Type.CanChangeTo)
951  {
952  foreach (var requirement in typeChange.Requirements)
953  {
954  if (requirement.AnyWithinDistance(location, requirement.RequiredProximityForProbabilityIncrease))
955  {

957  location.ProximityTimer[requirement] += 1;
