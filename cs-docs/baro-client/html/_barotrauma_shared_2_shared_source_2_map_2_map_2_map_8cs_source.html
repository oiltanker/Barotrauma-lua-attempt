251 private void Generate()
252 {
253 Connections.Clear();
254 Locations.Clear();
255
256 List<Vector2> voronoiSites = new List<Vector2>();
257 for (float x = 10.0f; x < Width - 10.0f; x += generationParams.VoronoiSiteInterval.X)
258 {
259 for (float y = 10.0f; y < Height - 10.0f; y += generationParams.VoronoiSiteInterval.Y)
260 {
261 voronoiSites.Add(new Vector2(
262 x + generationParams.VoronoiSiteVariance.X * Rand.Range(-0.5f, 0.5f, Rand.RandSync.ServerAndClient),
263 y + generationParams.VoronoiSiteVariance.Y * Rand.Range(-0.5f, 0.5f, Rand.RandSync.ServerAndClient)));
264 }
265 }
266
267 Voronoi voronoi = new Voronoi(0.5f);
268 List<GraphEdge> edges = voronoi.MakeVoronoiGraph(voronoiSites, Width, Height);
269 float zoneWidth = Width / generationParams.DifficultyZones;
270
271 Vector2 margin = new Vector2(
272 Math.Min(10, Width * 0.1f),
273 Math.Min(10, Height * 0.2f));
274
275 float startX = margin.X, endX = Width - margin.X;
276 float startY = margin.Y, endY = Height - margin.Y;
277
278 if (!edges.Any())
279 {
280 throw new Exception($"Generating a campaign map failed (no edges in the voronoi graph). Width: {Width}, height: {Height}, margin: {margin}");
281 }
282
283 voronoiSites.Clear();
284 Dictionary<int, List<Location>> locationsPerZone = new Dictionary<int, List<Location>>();
285 foreach (GraphEdge edge in edges)
286 {
287 if (edge.Point1 == edge.Point2) { continue; }
288
289 if (edge.Point1.X < margin.X || edge.Point1.X > Width - margin.X || edge.Point1.Y < startY || edge.Point1.Y > endY)
290 {
291 continue;
292 }
293 if (edge.Point2.X < margin.X || edge.Point2.X > Width - margin.X || edge.Point2.Y < startY || edge.Point2.Y > endY)
294 {
295 continue;
296 }
297
298 Location[] newLocations = new Location[2];
299 newLocations[0] = Locations.Find(l => l.MapPosition == edge.Point1 || l.MapPosition == edge.Point2);
300 newLocations[1] = Locations.Find(l => l != newLocations[0] && (l.MapPosition == edge.Point1 || l.MapPosition == edge.Point2));
301
302 for (int i = 0; i < 2; i++)
303 {
304 if (newLocations[i] != null) { continue; }
305
306 Vector2[] points = new Vector2[] { edge.Point1, edge.Point2 };
307
308 int positionIndex = Rand.Int(1, Rand.RandSync.ServerAndClient);
309
310 Vector2 position = points[positionIndex];
311 if (newLocations[1 - i] != null && newLocations[1 - i].MapPosition == position) { position = points[1 - positionIndex]; }
312 int zone = GetZoneIndex(position.X);
313 if (!locationsPerZone.ContainsKey(zone))
314 {
315 locationsPerZone[zone] = new List<Location>();
316 }
317
318 LocationType forceLocationType = null;
319 foreach (LocationType locationType in LocationType.Prefabs.OrderBy(lt => lt.Identifier))
320 {
321 if (locationType.MinCountPerZone.TryGetValue(zone, out int minCount) && locationsPerZone[zone].Count(l => l.Type == locationType) < minCount)
322 {
323 forceLocationType = locationType;
324 break;
325 }
326 }
327
328 newLocations[i] = Location.CreateRandom(position, zone, Rand.GetRNG(Rand.RandSync.ServerAndClient),
329 requireOutpost: false, forceLocationType: forceLocationType, existingLocations: Locations);
330 locationsPerZone[zone].Add(newLocations[i]);
331 Locations.Add(newLocations[i]);
332 }
333
334 var newConnection = new LocationConnection(newLocations[0], newLocations[1]);
335 Connections.Add(newConnection);
336 }
337
338 //remove connections that are too short
339 float minConnectionDistanceSqr = generationParams.MinConnectionDistance * generationParams.MinConnectionDistance;
340 for (int i = Connections.Count - 1; i >= 0; i--)
341 {
342 LocationConnection connection = Connections[i];
343
344 if (Vector2.DistanceSquared(connection.Locations[0].MapPosition, connection.Locations[1].MapPosition) > minConnectionDistanceSqr)
345 {
346 continue;
347 }
348
349 //locations.Remove(connection.Locations[0]);
350 Connections.Remove(connection);
351
352 foreach (LocationConnection connection2 in Connections)
353 {
354 if (connection2.Locations[0] == connection.Locations[0]) { connection2.Locations[0] = connection.Locations[1]; }
355 if (connection2.Locations[1] == connection.Locations[0]) { connection2.Locations[1] = connection.Locations[1]; }
356 }
357 }
358
359 foreach (LocationConnection connection in Connections)
360 {
361 connection.Locations[0].Connections.Add(connection);
362 connection.Locations[1].Connections.Add(connection);
363 }
364
365 //remove locations that are too close to each other
366 float minLocationDistanceSqr = generationParams.MinLocationDistance * generationParams.MinLocationDistance;
367 for (int i = Locations.Count - 1; i >= 0; i--)
368 {
369 for (int j = Locations.Count - 1; j > i; j--)
370 {
371 float dist = Vector2.DistanceSquared(Locations[i].MapPosition, Locations[j].MapPosition);
372 if (dist > minLocationDistanceSqr)
373 {
374 continue;
375 }
376 //move connections from Locations[j] to Locations[i]
377 foreach (LocationConnection connection in Locations[j].Connections)
378 {
379 if (connection.Locations[0] == Locations[j])
380 {
381 connection.Locations[0] = Locations[i];
382 }
383 else
384 {
385 connection.Locations[1] = Locations[i];
386 }
387
388 if (connection.Locations[0] != connection.Locations[1])
389 {
390 Locations[i].Connections.Add(connection);
391 }
392 else
393 {
394 Connections.Remove(connection);
395 }
396 }
397 Locations[i].Connections.RemoveAll(c => c.OtherLocation(Locations[i]) == Locations[j]);
398 Locations.RemoveAt(j);
399 }
400 }
401
402 for (int i = Connections.Count - 1; i >= 0; i--)
403 {
404 i = Math.Min(i, Connections.Count - 1);
405 LocationConnection connection = Connections[i];
406 for (int n = Math.Min(i - 1, Connections.Count - 1); n >= 0; n--)
407 {
408 if (connection.Locations.Contains(Connections[n].Locations[0])
409 && connection.Locations.Contains(Connections[n].Locations[1]))
410 {
411 Connections.RemoveAt(n);
412 }
413 }
414 }
415
416 List<LocationConnection>[] connectionsBetweenZones = new List<LocationConnection>[generationParams.DifficultyZones];
417 for (int i = 0; i < generationParams.DifficultyZones; i++)
418 {
419 connectionsBetweenZones[i] = new List<LocationConnection>();
420 }
421 var shuffledConnections = Connections.ToList();
422 shuffledConnections.Shuffle(Rand.RandSync.ServerAndClient);
423 foreach (var connection in shuffledConnections)
424 {
425 int zone1 = GetZoneIndex(connection.Locations[0].MapPosition.X);
426 int zone2 = GetZoneIndex(connection.Locations[1].MapPosition.X);
427 if (zone1 == zone2) { continue; }
428 if (zone1 > zone2)
429 {
430 int temp = zone2;
431 zone2 = zone1;
432 zone1 = temp;
433 }
434
435 if (generationParams.GateCount[zone1] == 0) { continue; }
436
437 if (!connectionsBetweenZones[zone1].Any())
438 {
439 connectionsBetweenZones[zone1].Add(connection);
440 }
441 else if (generationParams.GateCount[zone1] == 1)
442 {
443 //if there's only one connection, place it at the center of the map
444 if (Math.Abs(connection.CenterPos.Y - Height / 2) < Math.Abs(connectionsBetweenZones[zone1].First().CenterPos.Y - Height / 2))
445 {
446 connectionsBetweenZones[zone1].Clear();
447 connectionsBetweenZones[zone1].Add(connection);
448 }
449 }
450 else if (connectionsBetweenZones[zone1].Count() < generationParams.GateCount[zone1])
451 {
452 connectionsBetweenZones[zone1].Add(connection);
453 }
454 }
455
456 for (int i = Connections.Count - 1; i >= 0; i--)
457 {
458 int zone1 = GetZoneIndex(Connections[i].Locations[0].MapPosition.X);
459 int zone2 = GetZoneIndex(Connections[i].Locations[1].MapPosition.X);
460 if (zone1 == zone2) { continue; }
461 if (zone1 == generationParams.DifficultyZones || zone2 == generationParams.DifficultyZones) { continue; }
462
463 if (generationParams.GateCount[Math.Min(zone1, zone2)] == 0) { continue; }
464
465 if (!connectionsBetweenZones[Math.Min(zone1, zone2)].Contains(Connections[i]))
466 {
467 Connections.RemoveAt(i);
468 }
469 else
470 {
471 var leftMostLocation =
472 Connections[i].Locations[0].MapPosition.X < Connections[i].Locations[1].MapPosition.X ?
473 Connections[i].Locations[0] :
474 Connections[i].Locations[1];
475 if (!leftMostLocation.Type.HasOutpost || leftMostLocation.Type.Identifier == "abandoned")
476 {
477 leftMostLocation.ChangeType(LocationType.Prefabs.OrderBy(lt => lt.Identifier).First(lt => lt.HasOutpost && lt.Identifier != "abandoned"));
478 }
479 leftMostLocation.IsGateBetweenBiomes = true;
480 Connections[i].Locked = true;
481 }
482 }
483
484 foreach (Location location in Locations)
485 {
486 for (int i = location.Connections.Count - 1; i >= 0; i--)
487 {
488 if (!Connections.Contains(location.Connections[i]))
489 {
490 location.Connections.RemoveAt(i);
491 }
492 }
493 }
494
495 //remove orphans
496 Locations.RemoveAll(l => !Connections.Any(c => c.Locations.Contains(l)));
497
498 foreach (LocationConnection connection in Connections)
499 {
500 //float difficulty = GetLevelDifficulty(connection.CenterPos.X / Width);
501 //connection.Difficulty = MathHelper.Clamp(difficulty + Rand.Range(-10.0f, 0.0f, Rand.RandSync.ServerAndClient), 1.2f, 100.0f);
502 float difficulty = connection.CenterPos.X / Width * 100;
503 float random = difficulty > 10 ? 5 : 0;
504 connection.Difficulty = MathHelper.Clamp(difficulty + Rand.Range(-random, random, Rand.RandSync.ServerAndClient), 1.0f, 100.0f);
505 }
506
507 AssignBiomes();
508 CreateEndLocation();
509
510 foreach (Location location in Locations)
511 {
512 location.LevelData = new LevelData(location)
513 {
514 Difficulty = MathHelper.Clamp(location.MapPosition.X / Width * 100, 0.0f, 100.0f)
515 //Difficulty = MathHelper.Clamp(GetLevelDifficulty(location.MapPosition.X / Width), 0.0f, 100.0f)
516 };
517 location.UnlockInitialMissions();
518 }
519 foreach (LocationConnection connection in Connections)
520 {
521 connection.LevelData = new LevelData(connection);
522 }

524 float GetLevelDifficulty(float areaDifficulty)
525 {
526 const float CurveModifier = 1.5f;
527 const float DifficultyMultiplier = 1.14f;
528 const float BaseDifficulty = -3f;
529 return (float)(1 - Math.Pow(1 - areaDifficulty, CurveModifier)) * DifficultyMultiplier * 100f + BaseDifficulty;
530 }

554 private void AssignBiomes()
555 {
556 var biomes = Biome.Prefabs;
557 float zoneWidth = Width / generationParams.DifficultyZones;
558
559 List<Biome> allowedBiomes = new List<Biome>(10);
560 for (int i = 0; i < generationParams.DifficultyZones; i++)
561 {
562 allowedBiomes.Clear();
563 allowedBiomes.AddRange(biomes.Where(b => b.AllowedZones.Contains(generationParams.DifficultyZones - i)));
564 float zoneX = zoneWidth * (generationParams.DifficultyZones - i);
565
566 foreach (Location location in Locations)
567 {
568 if (location.MapPosition.X < zoneX)
569 {
570 location.Biome = allowedBiomes[Rand.Range(0, allowedBiomes.Count, Rand.RandSync.ServerAndClient)];
571 }
572 }
573 }
574 foreach (LocationConnection connection in Connections)
575 {
576 if (connection.Biome != null) { continue; }
577 connection.Biome = connection.Locations[0].MapPosition.X > connection.Locations[1].MapPosition.X ? connection.Locations[0].Biome : connection.Locations[1].Biome;
578 }
579
580 System.Diagnostics.Debug.Assert(Locations.All(l => l.Biome != null));
581 System.Diagnostics.Debug.Assert(Connections.All(c => c.Biome != null));
582 }

584 private void CreateEndLocation()
585 {
586 float zoneWidth = Width / generationParams.DifficultyZones;
587 Vector2 endPos = new Vector2(Width - zoneWidth / 2, Height / 2);
588 float closestDist = float.MaxValue;
589 EndLocation = Locations.First();
590 foreach (Location location in Locations)
591 {
592 float dist = Vector2.DistanceSquared(endPos, location.MapPosition);
593 if (location.Biome.IsEndBiome && dist < closestDist)
594 {
595 EndLocation = location;
596 closestDist = dist;
597 }
598 }
599
600 Location previousToEndLocation = null;
601 foreach (Location location in Locations)
602 {
603 if (!location.Biome.IsEndBiome && (previousToEndLocation == null || location.MapPosition.X > previousToEndLocation.MapPosition.X))
604 {
605 previousToEndLocation = location;
606 }
607 }
608
609 if (EndLocation == null || previousToEndLocation == null) { return; }
610
611 //remove all locations from the end biome except the end location
612 for (int i = Locations.Count - 1; i >= 0; i--)
613 {
614 if (Locations[i].Biome.IsEndBiome && Locations[i] != EndLocation)
615 {
616 for (int j = Locations[i].Connections.Count - 1; j >= 0; j--)
617 {
618 if (j >= Locations[i].Connections.Count) { continue; }
619 var connection = Locations[i].Connections[j];
620 var otherLocation = connection.OtherLocation(Locations[i]);
621 Locations[i].Connections.RemoveAt(j);
622 otherLocation?.Connections.Remove(connection);
623 Connections.Remove(connection);
624 }
625 Locations.RemoveAt(i);
626 }
627 }
628
629 //removed all connections from the second-to-last location, need to reconnect it
630 if (!previousToEndLocation.Connections.Any())
631 {
632 Location connectTo = Locations.First();
633 foreach (Location location in Locations)
634 {
635 if (!location.Biome.IsEndBiome && location != previousToEndLocation && location.MapPosition.X > connectTo.MapPosition.X)
636 {
637 connectTo = location;
638 }
639 }
640 var newConnection = new LocationConnection(previousToEndLocation, connectTo)
641 {
642 Biome = EndLocation.Biome,
643 Difficulty = 100.0f
644 };
645 Connections.Add(newConnection);
646 previousToEndLocation.Connections.Add(newConnection);
647 connectTo.Connections.Add(newConnection);
648 }
649
650 var endConnection = new LocationConnection(previousToEndLocation, EndLocation)
651 {
652 Biome = EndLocation.Biome,
653 Difficulty = 100.0f
654 };
655 Connections.Add(endConnection);
656 previousToEndLocation.Connections.Add(endConnection);
657 EndLocation.Connections.Add(endConnection);
658 }

660 private void ExpandBiomes(List<LocationConnection> seeds)
661 {
662 List<LocationConnection> nextSeeds = new List<LocationConnection>();
663 foreach (LocationConnection connection in seeds)
664 {
665 foreach (Location location in connection.Locations)
666 {
667 foreach (LocationConnection otherConnection in location.Connections)
668 {
669 if (otherConnection == connection) continue;
670 if (otherConnection.Biome != null) continue; //already assigned
671
672 otherConnection.Biome = connection.Biome;
673 nextSeeds.Add(otherConnection);
674 }
675 }
676 }
677
678 if (nextSeeds.Count > 0)
679 {
680 ExpandBiomes(nextSeeds);
681 }

687 public void MoveToNextLocation()
688 {
689 if (SelectedConnection == null)
690 {
691 DebugConsole.ThrowError("Could not move to the next location (no connection selected).\n"+Environment.StackTrace.CleanupStackTrace());
692 return;
693 }
694 if (SelectedLocation == null)
695 {
696 DebugConsole.ThrowError("Could not move to the next location (no location selected).\n" + Environment.StackTrace.CleanupStackTrace());
697 return;
698 }
699
700 Location prevLocation = CurrentLocation;
701 SelectedConnection.Passed = true;
702
703 CurrentLocation = SelectedLocation;
704 CurrentLocation.Discover();
705 SelectedLocation = null;
706
707 CurrentLocation.CreateStores();
708 OnLocationChanged?.Invoke(prevLocation, CurrentLocation);
709
710 if (GameMain.GameSession is { Campaign: { CampaignMetadata: { } metadata } })

891 private bool ProgressLocationTypeChanges(Location location)
892 {
893 location.TimeSinceLastTypeChange++;
894 location.LocationTypeChangeCooldown--;
895
896 if (location.PendingLocationTypeChange != null)
897 {
898 if (location.PendingLocationTypeChange.Value.typeChange.DetermineProbability(location) <= 0.0f)
899 {
900 //remove pending type change if it's no longer allowed
901 location.PendingLocationTypeChange = null;
902 }
903 else
904 {
905 location.PendingLocationTypeChange =
906 (location.PendingLocationTypeChange.Value.typeChange,
907 location.PendingLocationTypeChange.Value.delay - 1,
908 location.PendingLocationTypeChange.Value.parentMission);
909 if (location.PendingLocationTypeChange.Value.delay <= 0)
910 {
911 return ChangeLocationType(location, location.PendingLocationTypeChange.Value.typeChange);
912 }
913 }
914 }
915
916 //find which types of locations this one can change to
917 Dictionary<LocationTypeChange, float> allowedTypeChanges = new Dictionary<LocationTypeChange, float>();
918 foreach (LocationTypeChange typeChange in location.Type.CanChangeTo)
919 {
920 float probability = typeChange.DetermineProbability(location);
921 if (probability <= 0.0f) { continue; }
922 allowedTypeChanges.Add(typeChange, probability);
923 }
924
925 //select a random type change
926 if (Rand.Range(0.0f, 1.0f) < allowedTypeChanges.Sum(change => change.Value))
927 {
928 var selectedTypeChange =
929 ToolBox.SelectWeightedRandom(
930 allowedTypeChanges.Keys.ToList(),
931 allowedTypeChanges.Values.ToList(),
932 Rand.RandSync.Unsynced);
933 if (selectedTypeChange != null)
934 {
935 if (selectedTypeChange.RequiredDurationRange.X > 0)
936 {
937 location.PendingLocationTypeChange =
938 (selectedTypeChange,
939 Rand.Range(selectedTypeChange.RequiredDurationRange.X, selectedTypeChange.RequiredDurationRange.Y),
940 null);
941 }
942 else
943 {
944 return ChangeLocationType(location, selectedTypeChange);
945 }
946 return false;
947 }
948 }
949
950 foreach (LocationTypeChange typeChange in location.Type.CanChangeTo)
951 {
952 foreach (var requirement in typeChange.Requirements)
953 {
954 if (requirement.AnyWithinDistance(location, requirement.RequiredProximityForProbabilityIncrease))
955 {
956 if (!location.ProximityTimer.ContainsKey(requirement)) { location.ProximityTimer[requirement] = 0; }
957 location.ProximityTimer[requirement] += 1;
958 }
959 else
960 {
961 location.ProximityTimer.Remove(requirement);
962 }
963 }
964 }
965
966 return false;
967 }

969 public int DistanceToClosestLocationWithOutpost(Location startingLocation, out Location endingLocation)
970 {
971 if (startingLocation.Type.HasOutpost)
972 {
973 endingLocation = startingLocation;
974 return 0;
975 }
976
977 int iterations = 0;
978 int distance = 0;
979 endingLocation = null;
980
981 List<Location> testedLocations = new List<Location>();
982 List<Location> locationsToTest = new List<Location> { startingLocation };
983
984 while (endingLocation == null && iterations < 100)
985 {
986 List<Location> nextTestingBatch = new List<Location>();
987 for (int i = 0; i < locationsToTest.Count; i++)
988 {
989 Location testLocation = locationsToTest[i];
990 for (int j = 0; j < testLocation.Connections.Count; j++)
991 {
992 Location potentialOutpost = testLocation.Connections[j].OtherLocation(testLocation);
993 if (potentialOutpost.Type.HasOutpost)
994 {
995 distance = iterations + 1;
996 endingLocation = potentialOutpost;
997 }
998 else if (!testedLocations.Contains(potentialOutpost))
999 {
1000 nextTestingBatch.Add(potentialOutpost);
1001 }
1002 }
1003
1004 testedLocations.Add(testLocation);
1005 }
1006
1007 locationsToTest = nextTestingBatch;
1008 iterations++;
1009 }
1010
1011 return distance;
1012 }

1062 public void LoadState(XElement element, bool showNotifications)
1063 {
1064 ClearAnimQueue();
1065 SetLocation(element.GetAttributeInt("currentlocation", 0));
1066
1067 if (!Version.TryParse(element.GetAttributeString("version", ""), out _))
1068 {
1069 DebugConsole.ThrowError("Incompatible map save file, loading the game failed.");
1070 return;
1071 }
1072
1073 foreach (var subElement in element.Elements())
1074 {
1075 switch (subElement.Name.ToString().ToLowerInvariant())
1076 {
1077 case "location":
1078 Location location = Locations[subElement.GetAttributeInt("i", 0)];
1079 location.ProximityTimer.Clear();
1080 for (int i = 0; i < location.Type.CanChangeTo.Count; i++)
1081 {
1082 for (int j = 0; j < location.Type.CanChangeTo[i].Requirements.Count; j++)
1083 {
1084 location.ProximityTimer.Add(location.Type.CanChangeTo[i].Requirements[j], subElement.GetAttributeInt("changetimer" + i + "-" + j, 0));
1085 }
1086 }
1087 location.LoadLocationTypeChange(subElement);
1088 if (subElement.GetAttributeBool("discovered", false))
1089 {
1090 location.Discover(checkTalents: false);
1091 }
1092 if (location.Discovered)
1093 {
1094#if CLIENT
1095 RemoveFogOfWar(location);
1096#endif
1097 if (furthestDiscoveredLocation == null || location.MapPosition.X > furthestDiscoveredLocation.MapPosition.X)
1098 {
1099 furthestDiscoveredLocation = location;
1100 }
1101 }
1102
1103 Identifier locationType = subElement.GetAttributeIdentifier("type", Identifier.Empty);
1104 string prevLocationName = location.Name;
1105 LocationType prevLocationType = location.Type;
1106 LocationType newLocationType = LocationType.Prefabs.Find(lt => lt.Identifier == locationType) ?? LocationType.Prefabs.First();
1107 location.ChangeType(newLocationType);
1108 if (showNotifications && prevLocationType != location.Type)
1109 {
1110 var change = prevLocationType.CanChangeTo.Find(c => c.ChangeToType == location.Type.Identifier);
1111 if (change != null)
1112 {
1113 ChangeLocationTypeProjSpecific(location, prevLocationName, change);
1114 location.TimeSinceLastTypeChange = 0;
1115 }
1116 }
1117
1118 location.LoadStores(subElement);
1119 location.LoadMissions(subElement);
1120
1121 break;
1122 case "connection":
1123 int connectionIndex = subElement.GetAttributeInt("i", 0);
1124 Connections[connectionIndex].Passed = subElement.GetAttributeBool("passed", false);
1125 Connections[connectionIndex].Locked = subElement.GetAttributeBool("locked", false);
1126 break;
1127 case "radiation":
1128 Radiation = new Radiation(this, generationParams.RadiationParams, subElement);
1129 break;
1130 }
1131 }
1132
1133 foreach (Location location in Locations)
1134 {
1135 location?.InstantiateLoadedMissions(this);
1136 }
1137
1138 int currentLocationConnection = element.GetAttributeInt("currentlocationconnection", -1);
1139 if (currentLocationConnection >= 0)
1140 {
1141 Connections[currentLocationConnection].Locked = false;
1142 SelectLocation(Connections[currentLocationConnection].OtherLocation(CurrentLocation));
1143 }
1144 else
1145 {
1146 //this should not be possible, you can't enter non-outpost locations (= natural formations)
1147 if (CurrentLocation != null && !CurrentLocation.Type.HasOutpost && SelectedConnection == null)
1148 {
1149 DebugConsole.AddWarning($"Error while loading campaign map state. Submarine in a location with no outpost ({CurrentLocation.Name}). Loading the first adjacent connection...");
1150 SelectLocation(CurrentLocation.Connections[0].OtherLocation(CurrentLocation));
1151 }
1152 }
