405  public override float GetConnectionPowerOut(Connection conn, float power, PowerRange minMaxPower, float load)
406  {
407  //Load must be calculated at this stage instead of at gridResolved to remove influence of lower priority devices
408  float loadLeft = MathHelper.Max(load - power,0);
409  float expectedPower = MathHelper.Clamp(loadLeft, minMaxPower.Min, minMaxPower.Max);
410 
411  //Delta ratio of Min and Max power output capability of the grid
412  float ratio = MathHelper.Max((loadLeft - minMaxPower.Min) / (minMaxPower.Max - minMaxPower.Min), 0);
413  if (float.IsInfinity(ratio))
414  {
415  ratio = 0;
416  }
417 
418  float output = MathHelper.Clamp(ratio * (maxUpdatePowerOut - minUpdatePowerOut) + minUpdatePowerOut, minUpdatePowerOut, maxUpdatePowerOut);
419  float newLoad = loadLeft;
420 
421  //Adjust behaviour for multi reactor setup
422  if (MaxPowerOutput != minMaxPower.ReactorMaxOutput)
423  {
424  float idealLoad = MaxPowerOutput / minMaxPower.ReactorMaxOutput * loadLeft;
425  float loadAdjust = MathHelper.Clamp((ratio - 0.5f) * 25 + idealLoad - (turbineOutput / 100 * MaxPowerOutput), -MaxPowerOutput / 100, MaxPowerOutput / 100);
426  newLoad = MathHelper.Clamp(loadLeft - (expectedPower - output) + loadAdjust, 0, loadLeft);
427  }
428 
429  if (float.IsNegative(newLoad))
430  {
431  newLoad = 0.0f;
432  }
433 
434  Load = newLoad;
435  currPowerConsumption = -output;
436  return output;
437  }

449  private bool NeedMoreFuel(float minimumOutputRatio, float minCondition = 0)
450  {
451  float remainingFuel = item.ContainedItems.Sum(i => i.Condition);
452  if (remainingFuel <= minCondition && Load > 0.0f)
453  {
454  return true;
455  }
456 
457  //fission rate is clamped to the amount of available fuel
458  float maxFissionRate = Math.Min(prevAvailableFuel, 100.0f);
459  if (maxFissionRate >= 100.0f) { return false; }
460 
461  float maxTurbineOutput = 100.0f;
462 
463  //calculate the maximum output if the fission rate is cranked as high as it goes and turbine output is at max
464  float theoreticalMaxHeat = GetGeneratedHeat(fissionRate: maxFissionRate);
465  float temperatureFactor = Math.Min(theoreticalMaxHeat / 50.0f, 1.0f);
466  float theoreticalMaxOutput = Math.Min(maxTurbineOutput / 100.0f, temperatureFactor) * MaxPowerOutput;
467 
468  //maximum output not enough, we need more fuel
469  return theoreticalMaxOutput < Load * minimumOutputRatio;
470  }

525  public void UpdateAutoTemp(float speed, float deltaTime)
526  {
527  float desiredTurbineOutput = (optimalTurbineOutput.X + optimalTurbineOutput.Y) / 2.0f;
528  TargetTurbineOutput += MathHelper.Clamp(desiredTurbineOutput - TargetTurbineOutput, -speed, speed) * deltaTime;
529  TargetTurbineOutput = MathHelper.Clamp(TargetTurbineOutput, 0.0f, 100.0f);
530 
531  float desiredFissionRate = (optimalFissionRate.X + optimalFissionRate.Y) / 2.0f;
532  TargetFissionRate += MathHelper.Clamp(desiredFissionRate - TargetFissionRate, -speed, speed) * deltaTime;
533 
534  if (temperature > (optimalTemperature.X + optimalTemperature.Y) / 2.0f)
535  {
536  TargetFissionRate = Math.Min(TargetFissionRate - speed * 2 * deltaTime, allowedFissionRate.Y);
537  }
538  else if (-currPowerConsumption < Load)
539  {
540  TargetFissionRate = Math.Min(TargetFissionRate + speed * 2 * deltaTime, 100.0f);
541  }
542  TargetFissionRate = MathHelper.Clamp(TargetFissionRate, 0.0f, 100.0f);
543 
544  //don't push the target too far from the current fission rate
545  //otherwise we may "overshoot", cranking the target fission rate all the way up because it takes a while
546  //for the actual fission rate and temperature to follow
547  TargetFissionRate = MathHelper.Clamp(TargetFissionRate, FissionRate - 5, FissionRate + 5);

625  public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
626  {
627  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return false; }
628  character.AIController.SteeringManager.Reset();
629  bool shutDown = objective.Option == "shutdown";
630 
631  IsActive = true;
632 
633  if (!shutDown)
634  {
635  float degreeOfSuccess = Math.Min(DegreeOfSuccess(character), 1.0f);
636  float refuelLimit = 0.3f;
637  //characters with insufficient skill levels don't refuel the reactor
638  if (degreeOfSuccess > refuelLimit)
639  {
640  if (aiUpdateTimer > 0.0f)
641  {
642  aiUpdateTimer -= deltaTime;
643  return false;
644  }
645  aiUpdateTimer = AIUpdateInterval;
646  // load more fuel if the current maximum output is only 50% of the current load
647  // or if the fuel rod is (almost) deplenished
648  float minCondition = fuelConsumptionRate * MathUtils.Pow2((degreeOfSuccess - refuelLimit) * 2);
649  if (NeedMoreFuel(minimumOutputRatio: 0.5f, minCondition: minCondition))
650  {
651  bool outOfFuel = false;
652  var container = item.GetComponent<ItemContainer>();
653  if (objective.SubObjectives.None())
654  {
655  var containObjective = AIContainItems<Reactor>(container, character, objective, itemCount: 1, equip: true, removeEmpty: true, spawnItemIfNotFound: !character.IsOnPlayerTeam, dropItemOnDeselected: true);
