272  public static async Task<List<Steamworks.Data.Lobby>> LobbyQueryRequest()
273  {
274  List<Steamworks.Data.Lobby> allLobbies = new List<Steamworks.Data.Lobby>();
275  Steamworks.Data.LobbyQuery lobbyQuery = Steamworks.SteamMatchmaking.CreateLobbyQuery()
276  .FilterDistanceWorldwide()
277  .WithMaxResults(50);
278  //steamworks seems to unable to retrieve more than 50
279  //lobbies per request; to work around this, we'll make
280  //up to 10 requests, asking to ignore all previous results
281  //in each subsequent request
282  for (int i = 0; i < 10; i++)
283  {
284  Steamworks.Data.Lobby[] lobbies = await lobbyQuery.RequestAsync();
285  if (lobbies == null) { break; }
286  foreach (var l in lobbies)
287  {
288  lobbyQuery = lobbyQuery
289  .WithoutKeyValue("lobbyowner", l.GetData("lobbyowner"));
290  }
291  allLobbies.AddRange(lobbies);
292  }
293 
294  //make sure all returned lobbies are distinct, don't want any duplicates here
295  return allLobbies.Select(l => l.Id).Distinct().Select(i => allLobbies.Find(l => l.Id == i)).ToList();
296  }
