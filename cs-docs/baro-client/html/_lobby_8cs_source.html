154 public static bool GetServers(Action<ServerInfo> addToServerList, Action serverQueryFinished)
155 {
156 if (!IsInitialized) { return false; }
157
158 int doneTasks = 0;
159 void taskDone()
160 {
161 doneTasks++;
162 if (doneTasks >= 2)
163 {
164 serverQueryFinished?.Invoke();
165 serverQueryFinished = null;
166 }
167 }
168
169
170 Steamworks.Dispatch.OnDebugCallback = (callbackType, contents, isServer) =>
171 {
172 DebugConsole.NewMessage($"{callbackType}: " + contents, Color.Yellow);
173 };
174
175 TaskPool.Add("LobbyQueryRequest", LobbyQueryRequest(),
176 (t) =>
177 {
178 Steamworks.Dispatch.OnDebugCallback = null;
179 if (t.Status == TaskStatus.Faulted)
180 {
181 TaskPool.PrintTaskExceptions(t, "Failed to retrieve SteamP2P lobbies");
182 taskDone();
183 return;
184 }
185 var lobbies = ((Task<List<Steamworks.Data.Lobby>>)t).Result;
186 if (lobbies != null)
187 {
188 foreach (var lobby in lobbies)
189 {
190 if (string.IsNullOrEmpty(lobby.GetData("name"))) { continue; }
191
192 ServerInfo serverInfo = new ServerInfo();
193 serverInfo.ServerName = lobby.GetData("name");
194 serverInfo.OwnerID = SteamIDStringToUInt64(lobby.GetData("lobbyowner"));
195 serverInfo.LobbyID = lobby.Id;
196 bool.TryParse(lobby.GetData("haspassword"), out serverInfo.HasPassword);
197 serverInfo.PlayerCount = int.TryParse(lobby.GetData("playercount"), out int playerCount) ? playerCount : 0;
198 serverInfo.MaxPlayers = int.TryParse(lobby.GetData("maxplayernum"), out int maxPlayers) ? maxPlayers : 1;
199 serverInfo.RespondedToSteamQuery = true;
200
201 AssignLobbyDataToServerInfo(lobby, serverInfo);
202
203 addToServerList(serverInfo);
204 }
205 }
206 taskDone();
207 });

210 void onServer(Steamworks.Data.ServerInfo info, bool responsive)
211 {
212 if (string.IsNullOrEmpty(info.Name)) { return; }
213
214 ServerInfo serverInfo = new ServerInfo
215 {
216 ServerName = info.Name,
217 HasPassword = info.Passworded,
218 IP = info.Address.ToString(),
219 Port = info.ConnectionPort.ToString(),
220 PlayerCount = info.Players,
221 MaxPlayers = info.MaxPlayers,
222 RespondedToSteamQuery = responsive
223 };
224
225 if (responsive)
226 {
227 TaskPool.Add($"QueryServerRules (GetServers, {info.Name}, {info.Address})", info.QueryRulesAsync(),
228 (t) =>
229 {
230 if (t.Status == TaskStatus.Faulted)
231 {
232 TaskPool.PrintTaskExceptions(t, "Failed to retrieve rules for " + info.Name);
233 return;
234 }
235
236 var rules = ((Task<Dictionary<string, string>>)t).Result;
237 AssignServerRulesToServerInfo(rules, serverInfo);
238
239 CrossThread.RequestExecutionOnMainThread(() =>
240 {
241 addToServerList(serverInfo);
242 });
243 });
244 }
245 else
246 {
247 CrossThread.RequestExecutionOnMainThread(() =>
248 {
249 addToServerList(serverInfo);
250 });
251 }
252
253 }
254 serverQuery.OnResponsiveServer += (info) => onServer(info, true);
255 serverQuery.OnUnresponsiveServer += (info) => onServer(info, false);
256
257 TaskPool.Add("RunServerQuery", serverQuery.RunQueryAsync(),
258 (t) =>
259 {
260 serverQuery.Dispose();
261 taskDone();
262 if (t.Status == TaskStatus.Faulted)
263 {
264 TaskPool.PrintTaskExceptions(t, "Failed to retrieve servers");
265 return;
266 }
267 });
268
269 return true;
270 }

272 public static async Task<List<Steamworks.Data.Lobby>> LobbyQueryRequest()
273 {
274 List<Steamworks.Data.Lobby> allLobbies = new List<Steamworks.Data.Lobby>();
275 Steamworks.Data.LobbyQuery lobbyQuery = Steamworks.SteamMatchmaking.CreateLobbyQuery()
276 .FilterDistanceWorldwide()
277 .WithMaxResults(50);
278 //steamworks seems to unable to retrieve more than 50
279 //lobbies per request; to work around this, we'll make
280 //up to 10 requests, asking to ignore all previous results
281 //in each subsequent request
282 for (int i = 0; i < 10; i++)
283 {
284 Steamworks.Data.Lobby[] lobbies = await lobbyQuery.RequestAsync();
285 if (lobbies == null) { break; }
286 foreach (var l in lobbies)
287 {
288 lobbyQuery = lobbyQuery
289 .WithoutKeyValue("lobbyowner", l.GetData("lobbyowner"));
290 }
291 allLobbies.AddRange(lobbies);
292 }
293
294 //make sure all returned lobbies are distinct, don't want any duplicates here
295 return allLobbies.Select(l => l.Id).Distinct().Select(i => allLobbies.Find(l => l.Id == i)).ToList();
296 }

298 public static void AssignLobbyDataToServerInfo(Steamworks.Data.Lobby lobby, ServerInfo serverInfo)
299 {
300 serverInfo.OwnerVerified = true;
301
302 serverInfo.ServerMessage = lobby.GetData("message");
303 serverInfo.GameVersion = lobby.GetData("version");

305 serverInfo.ContentPackageNames.AddRange(lobby.GetData("contentpackage").Split(','));
306 serverInfo.ContentPackageHashes.AddRange(lobby.GetData("contentpackagehash").Split(','));
307
308 string workshopIdData = lobby.GetData("contentpackageid");
309 if (!string.IsNullOrEmpty(workshopIdData))
310 {
311 serverInfo.ContentPackageWorkshopIds.AddRange(ParseWorkshopIds(workshopIdData));
312 }
313 else
314 {
315 string[] workshopUrls = lobby.GetData("contentpackageurl").Split(',');
316 serverInfo.ContentPackageWorkshopIds.AddRange(WorkshopUrlsToIds(workshopUrls));
317 }
318
319 serverInfo.UsingWhiteList = getLobbyBool("usingwhitelist");
320 if (Enum.TryParse(lobby.GetData("modeselectionmode"), out SelectionMode selectionMode)) { serverInfo.ModeSelectionMode = selectionMode; }
321 if (Enum.TryParse(lobby.GetData("subselectionmode"), out selectionMode)) { serverInfo.SubSelectionMode = selectionMode; }
322
323 serverInfo.AllowSpectating = getLobbyBool("allowspectating");
324 serverInfo.AllowRespawn = getLobbyBool("allowrespawn");
325 serverInfo.VoipEnabled = getLobbyBool("voicechatenabled");
326 serverInfo.KarmaEnabled = getLobbyBool("karmaenabled");
327 serverInfo.FriendlyFireEnabled = getLobbyBool("friendlyfireenabled");
328 if (Enum.TryParse(lobby.GetData("traitors"), out YesNoMaybe traitorsEnabled)) { serverInfo.TraitorsEnabled = traitorsEnabled; }
329
330 serverInfo.GameStarted = lobby.GetData("gamestarted") == "True";
331 serverInfo.GameMode = (lobby.GetData("gamemode") ?? "").ToIdentifier();
332 if (Enum.TryParse(lobby.GetData("playstyle"), out PlayStyle playStyle)) serverInfo.PlayStyle = playStyle;
333
334 if (serverInfo.ContentPackageNames.Count != serverInfo.ContentPackageHashes.Count ||
335 serverInfo.ContentPackageHashes.Count != serverInfo.ContentPackageWorkshopIds.Count)

339 serverInfo.ContentPackageHashes.Clear();
340 serverInfo.ContentPackageWorkshopIds.Clear();
341 }
342
343 string pingLocation = lobby.GetData("pinglocation");
344 if (!string.IsNullOrEmpty(pingLocation))

347 }
