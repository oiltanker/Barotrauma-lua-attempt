89 public static CoroutineHandle StartCoroutine(IEnumerable<CoroutineStatus> func, string name = "", bool useSeparateThread = false)
90 {
91 var handle = new CoroutineHandle(func.GetEnumerator(), name);
92 lock (Coroutines)
93 {
94 Coroutines.Add(handle);
95 }
96
97 handle.Thread = null;
98 if (useSeparateThread)
99 {
100 handle.Thread = new Thread(() => { ExecuteCoroutineThread(handle); })
101 {
102 Name = "Coroutine Thread (" + handle.Name + ")",
103 IsBackground = true
104 };
105 handle.Thread.Start();
106 }
107
108 return handle;
109 }

111 public static CoroutineHandle Invoke(Action action, float delay = 0f)

116 private static IEnumerable<CoroutineStatus> DoInvokeAfter(Action action, float delay)
117 {
118 if (action == null)
119 {
120 yield return CoroutineStatus.Failure;
121 }
122
123 if (delay > 0.0f)
124 {
125 yield return new WaitForSeconds(delay);

168 private static void HandleCoroutineStopping(Func<CoroutineHandle, bool> filter)
169 {
170 foreach (CoroutineHandle coroutine in Coroutines)
171 {
172 if (filter(coroutine))
173 {
174 coroutine.AbortRequested = true;
175 if (coroutine.Thread != null)
176 {
177 bool joined = false;
178 while (!joined)
179 {
180 CrossThread.ProcessTasks();
181 joined = coroutine.Thread.Join(TimeSpan.FromMilliseconds(500));
182 }
183 }
184 }
185 }

188 private static bool PerformCoroutineStep(CoroutineHandle handle)
189 {
190 var current = handle.Coroutine.Current;
191 if (current != null)
192 {
193 if (current.EndsCoroutine(handle) || handle.AbortRequested) { return true; }
194 if (!current.CheckFinished(UnscaledDeltaTime)) { return false; }
195 }
196 if (!handle.Coroutine.MoveNext()) { return true; }
197 return false;
198 }
199
200 public static void ExecuteCoroutineThread(CoroutineHandle handle)
201 {
202 try
203 {
204 while (!handle.AbortRequested)
205 {
206 if (PerformCoroutineStep(handle)) { return; }
207 Thread.Sleep((int)(UnscaledDeltaTime * 1000));
208 }
209 }
210 catch (ThreadAbortException)
211 {
212 //not an error, don't worry about it
213 }
214 catch (Exception e)
215 {
216 handle.Exception = e;
217 DebugConsole.ThrowError("Coroutine \"" + handle.Name + "\" has thrown an exception", e);
218 }
219 }
220
221 private static bool IsDone(CoroutineHandle handle)
222 {
223#if !DEBUG
224 try
225 {
226#endif
227 if (handle.Thread == null)
228 {
229 return PerformCoroutineStep(handle);
230 }
231 else
232 {
233 if (handle.Thread.ThreadState.HasFlag(ThreadState.Stopped))
234 {
235 if (handle.Exception!=null || handle.Coroutine.Current == CoroutineStatus.Failure)
236 {
237 DebugConsole.ThrowError("Coroutine \"" + handle.Name + "\" has failed");
238 }
239 return true;
240 }
241 return false;
242 }
243#if !DEBUG
244 }
245 catch (Exception e)
246 {
247#if CLIENT && WINDOWS
248 if (e is SharpDX.SharpDXException) { throw; }
249#endif
250 DebugConsole.ThrowError("Coroutine " + handle.Name + " threw an exception: " + e.Message + "\n" + e.StackTrace.CleanupStackTrace());
251 handle.Exception = e;
252 return true;
253 }
254#endif
255 }
256 // Updating just means stepping through all the coroutines
257 public static void Update(float unscaledDeltaTime, float deltaTime)
258 {
259 UnscaledDeltaTime = unscaledDeltaTime;
260 DeltaTime = deltaTime;
261
262 List<CoroutineHandle> coroutineList;
263 lock (Coroutines)
264 {
265 coroutineList = Coroutines.ToList();
266 }
267
268 foreach (var coroutine in coroutineList)
269 {
270 if (IsDone(coroutine))
271 {
272 lock (Coroutines)
273 {
274 Coroutines.Remove(coroutine);
275 }
276 }
277 }
278 }
