113 public override void Update(float deltaTime, Camera cam)
114 {
115 ReloadTimer -= deltaTime;
116
117 if (ReloadTimer < 0.0f)
118 {
119 ReloadTimer = 0.0f;
120 if (MaxChargeTime <= 0f)
121 {
122 IsActive = false;
123 return;
124 }
125 }
126
127 float previousChargeTime = currentChargeTime;
128
129 float chargeDeltaTime = tryingToCharge && ReloadTimer <= 0f ? deltaTime : -deltaTime;
130 currentChargeTime = Math.Clamp(currentChargeTime + chargeDeltaTime, 0f, MaxChargeTime);
131
132 tryingToCharge = false;
133
134 if (currentChargeTime == 0f)
135 {
136 currentChargingState = ChargingState.Inactive;
137 }
138 else if (currentChargeTime < previousChargeTime)
139 {
140 currentChargingState = ChargingState.WindingDown;
141 }
142 else
143 {
144 // if we are charging up or at maxed charge, remain winding up
145 currentChargingState = ChargingState.WindingUp;
146 }
147
148 UpdateProjSpecific(deltaTime);
149 }

153 private float GetSpread(Character user)
154 {
155 float degreeOfFailure = MathHelper.Clamp(1.0f - DegreeOfSuccess(user), 0.0f, 1.0f);
156 degreeOfFailure *= degreeOfFailure;
157 float spread = MathHelper.Lerp(Spread, UnskilledSpread, degreeOfFailure) / (1f + user.GetStatValue(StatTypes.RangedSpreadReduction));
158 return MathHelper.ToRadians(spread);
159 }

161 private readonly List<Body> ignoredBodies = new List<Body>();
162 public override bool Use(float deltaTime, Character character = null)
163 {
164 tryingToCharge = true;
165 if (character == null || character.Removed) { return false; }
166 if ((item.RequireAimToUse && !character.IsKeyDown(InputType.Aim)) || ReloadTimer > 0.0f) { return false; }
167 if (currentChargeTime < MaxChargeTime) { return false; }
168
169 IsActive = true;
170 ReloadTimer = reload / (1 + character?.GetStatValue(StatTypes.RangedAttackSpeed) ?? 0f);
171 currentChargeTime = 0f;
172
173 if (character != null)
174 {
175 var abilityRangedWeapon = new AbilityRangedWeapon(item);
176 character.CheckTalents(AbilityEffectType.OnUseRangedWeapon, abilityRangedWeapon);
177 }
178
179 if (item.AiTarget != null)
180 {
181 item.AiTarget.SoundRange = item.AiTarget.MaxSoundRange;
182 item.AiTarget.SightRange = item.AiTarget.MaxSightRange;
183 }
184
185 ignoredBodies.Clear();
186 foreach (Limb l in character.AnimController.Limbs)
187 {
188 if (l.IsSevered) { continue; }
189 ignoredBodies.Add(l.body.FarseerBody);
190 }
191
192 foreach (Item heldItem in character.HeldItems)
193 {
194 var holdable = heldItem.GetComponent<Holdable>();
195 if (holdable?.Pusher != null)
196 {
197 ignoredBodies.Add(holdable.Pusher.FarseerBody);
198 }
199 }
200
201 float degreeOfFailure = 1.0f - DegreeOfSuccess(character);
202 degreeOfFailure *= degreeOfFailure;
203 if (degreeOfFailure > Rand.Range(0.0f, 1.0f))
204 {
205 ApplyStatusEffects(ActionType.OnFailure, 1.0f, character);
206 }
207
208 for (int i = 0; i < ProjectileCount; i++)
209 {
210 Projectile projectile = FindProjectile(triggerOnUseOnContainers: true);
211 if (projectile != null)
212 {
213 Vector2 barrelPos = TransformedBarrelPos + item.body.SimPosition;
214 float rotation = (Item.body.Dir == 1.0f) ? Item.body.Rotation : Item.body.Rotation - MathHelper.Pi;
215 float spread = GetSpread(character) * Rand.Range(-0.5f, 0.5f);
216 var lastProjectile = LastProjectile;
217 if (lastProjectile != projectile)
218 {
219 lastProjectile?.Item.GetComponent<Rope>()?.Snap();
220 }
221 float damageMultiplier = 1f + item.GetQualityModifier(Quality.StatType.FirepowerMultiplier);
222 projectile.Launcher = item;
223 projectile.Shoot(character, character.AnimController.AimSourceSimPos, barrelPos, rotation + spread, ignoredBodies: ignoredBodies.ToList(), createNetworkEvent: false, damageMultiplier);
224 projectile.Item.GetComponent<Rope>()?.Attach(Item, projectile.Item);
225 if (i == 0)
226 {
227 Item.body.ApplyLinearImpulse(new Vector2((float)Math.Cos(projectile.Item.body.Rotation), (float)Math.Sin(projectile.Item.body.Rotation)) * Item.body.Mass * -50.0f, maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
228 }
229 projectile.Item.body.ApplyTorque(projectile.Item.body.Mass * degreeOfFailure * Rand.Range(-10.0f, 10.0f));
230 Item.RemoveContained(projectile.Item);
231 }
232 LastProjectile = projectile;
233 }
234
235 LaunchProjSpecific();
236
237 return true;
238 }

245 public Projectile FindProjectile(bool triggerOnUseOnContainers = false)
246 {
247 var containedItems = item.OwnInventory?.AllItemsMod;
248 if (containedItems == null) { return null; }
249
250 foreach (Item item in containedItems)
251 {
252 if (item == null) { continue; }
253 Projectile projectile = item.GetComponent<Projectile>();
254 if (projectile != null) { return projectile; }
255 }
256
257 //projectile not found, see if one of the contained items contains projectiles
258 foreach (Item it in containedItems)
259 {
260 if (it == null) { continue; }
261 var containedSubItems = it.OwnInventory?.AllItemsMod;
262 if (containedSubItems == null) { continue; }
263 foreach (Item subItem in containedSubItems)
264 {
265 if (subItem == null) { continue; }
266 Projectile projectile = subItem.GetComponent<Projectile>();
267 //apply OnUse statuseffects to the container in case it has to react to it somehow
268 //(play a sound, spawn more projectiles, reduce condition...)
269 if (triggerOnUseOnContainers && subItem.Condition > 0.0f)
270 {
271 subItem.GetComponent<ItemContainer>()?.Item.ApplyStatusEffects(ActionType.OnUse, 1.0f);
272 }
273 if (projectile != null) { return projectile; }
274 }
275 }
276
277 return null;
