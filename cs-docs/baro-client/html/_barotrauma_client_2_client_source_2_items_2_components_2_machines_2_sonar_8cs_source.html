351 private Vector2 GetTransducerPos()
352 {
353 if (!UseTransducers || connectedTransducers.Count == 0)
354 {
355 //use the position of the sub if the item is static (no body) and inside a sub
356 return item.Submarine != null && item.body == null ? item.Submarine.WorldPosition : item.WorldPosition;
357 }
358
359 Vector2 transducerPosSum = Vector2.Zero;
360 foreach (ConnectedTransducer transducer in connectedTransducers)
361 {
362 if (transducer.Transducer.Item.Submarine != null && !CenterOnTransducers)
363 {
364 return transducer.Transducer.Item.Submarine.WorldPosition;
365 }
366 transducerPosSum += transducer.Transducer.Item.WorldPosition;
367 }
368 return transducerPosSum / connectedTransducers.Count;
369 }

428 public override void UpdateHUD(Character character, float deltaTime, Camera cam)
429 {
430 showDirectionalIndicatorTimer -= deltaTime;
431 if (GameMain.Client != null)
432 {
433 if (unsentChanges)
434 {
435 if (networkUpdateTimer <= 0.0f)
436 {
437 item.CreateClientEvent(this);
438 correctionTimer = CorrectionDelay;
439 networkUpdateTimer = 0.1f;
440 unsentChanges = false;
441 }
442 }
443 networkUpdateTimer -= deltaTime;
444 }
445
446 connectedSubUpdateTimer -= deltaTime;
447 if (connectedSubUpdateTimer <= 0.0f)
448 {
449 connectedSubs.Clear();
450 if (UseTransducers)
451 {
452 foreach (var transducer in connectedTransducers)
453 {
454 if (transducer.Transducer.Item.Submarine == null) { continue; }
455 if (connectedSubs.Contains(transducer.Transducer.Item.Submarine)) { continue; }
456 connectedSubs = transducer.Transducer.Item.Submarine?.GetConnectedSubs();
457 }
458 }
459 else if (item.Submarine != null)
460 {
461 connectedSubs = item.Submarine?.GetConnectedSubs();
462 }
463 connectedSubUpdateTimer = ConnectedSubUpdateInterval;
464 }
465
466 Steering steering = item.GetComponent<Steering>();
467 if (sonarView.Rect.Contains(PlayerInput.MousePosition) &&
468 (GUI.MouseOn == null || GUI.MouseOn == sonarView || sonarView.IsParentOf(GUI.MouseOn) || GUI.MouseOn == steering?.GuiFrame || (steering?.GuiFrame?.IsParentOf(GUI.MouseOn) ?? false)))
469 {
470 float scrollSpeed = PlayerInput.ScrollWheelSpeed / 1000.0f;
471 if (Math.Abs(scrollSpeed) > 0.0001f)
472 {
473 zoomSlider.BarScroll += PlayerInput.ScrollWheelSpeed / 1000.0f;
474 zoomSlider.OnMoved(zoomSlider, zoomSlider.BarScroll);
475 }
476 }
477
478 Vector2 transducerCenter = GetTransducerPos();
479
480 if (steering != null && steering.DockingModeEnabled && steering.ActiveDockingSource != null)
481 {
482 Vector2 worldFocusPos = (steering.ActiveDockingSource.Item.WorldPosition + steering.DockingTarget.Item.WorldPosition) / 2.0f;
483 DisplayOffset = Vector2.Lerp(DisplayOffset, worldFocusPos - transducerCenter, 0.1f);
484 }
485 else
486 {
487 DisplayOffset = Vector2.Lerp(DisplayOffset, Vector2.Zero, 0.1f);
488 }
489 transducerCenter += DisplayOffset;
490
491 float distort = MathHelper.Clamp(1.0f - item.Condition / item.MaxCondition, 0.0f, 1.0f);
492 for (int i = sonarBlips.Count - 1; i >= 0; i--)
493 {
494 sonarBlips[i].FadeTimer -= deltaTime * MathHelper.Lerp(0.5f, 2.0f, distort);
495 sonarBlips[i].Position += sonarBlips[i].Velocity * deltaTime;
496
497 if (sonarBlips[i].FadeTimer <= 0.0f) { sonarBlips.RemoveAt(i); }
498 }

500 //sonar view can only get focus when the cursor is inside the circle
501 sonarView.CanBeFocused =
502 Vector2.DistanceSquared(sonarView.Rect.Center.ToVector2(), PlayerInput.MousePosition) <
503 (sonarView.Rect.Width / 2 * sonarView.Rect.Width / 2);
504
505 if (AllowUsingMineralScanner && Level.Loaded != null && !Level.Loaded.Generating)
506 {
507 if (MineralClusters == null)
508 {
509 MineralClusters = new List<(Vector2, List<Item>)>();
510 Level.Loaded.PathPoints.ForEach(p => p.ClusterLocations.ForEach(c => AddIfValid(c)));
511 Level.Loaded.AbyssResources.ForEach(c => AddIfValid(c));
512
513 void AddIfValid(Level.ClusterLocation c)
514 {
515 if (c.Resources == null) { return; }
516 if (c.Resources.None(i => i != null && !i.Removed && i.Tags.Contains("ore"))) { return; }
517 var pos = Vector2.Zero;
518 foreach (var r in c.Resources)
519 {
520 pos += r.WorldPosition;
521 }
522 pos /= c.Resources.Count;
523 MineralClusters.Add((center: pos, resources: c.Resources));
524 }
525 }
526 else
527 {
528 MineralClusters.RemoveAll(c => c.resources == null || c.resources.None() || c.resources.All(i => i == null || i.Removed));
529 }
530 }
531
532 if (UseTransducers && connectedTransducers.Count == 0)
533 {
534 return;
535 }
536
537 if (Level.Loaded != null)
538 {
539 nearbyObjectUpdateTimer -= deltaTime;
540 if (nearbyObjectUpdateTimer <= 0.0f)
541 {
542 nearbyObjects.Clear();
543 foreach (var nearbyObject in Level.Loaded.LevelObjectManager.GetAllObjects(transducerCenter, range * zoom))
544 {
545 if (!nearbyObject.VisibleOnSonar) { continue; }
546 float objectRange = range + nearbyObject.SonarRadius;
547 if (Vector2.DistanceSquared(transducerCenter, nearbyObject.WorldPosition) < objectRange * objectRange)
548 {
549 nearbyObjects.Add(nearbyObject);
550 }
551 }
552 nearbyObjectUpdateTimer = NearbyObjectUpdateInterval;
553 }
554
555 List<LevelTrigger> ballastFloraSpores = new List<LevelTrigger>();
556 Dictionary<LevelTrigger, Vector2> levelTriggerFlows = new Dictionary<LevelTrigger, Vector2>();
557 for (var pingIndex = 0; pingIndex < activePingsCount; ++pingIndex)
558 {
559 var activePing = activePings[pingIndex];
560 float pingRange = range * activePing.State / zoom;
561 foreach (LevelObject levelObject in nearbyObjects)
562 {
563 if (levelObject.Triggers == null) { continue; }
564 //gather all nearby triggers that are causing the water to flow into the dictionary
565 foreach (LevelTrigger trigger in levelObject.Triggers)
566 {
567 Vector2 flow = trigger.GetWaterFlowVelocity();
568 //ignore ones that are barely doing anything (flow^2 <= 1)
569 if (flow.LengthSquared() >= 1.0f && !levelTriggerFlows.ContainsKey(trigger))
570 {
571 levelTriggerFlows.Add(trigger, flow);
572 }
573 if (!trigger.InfectIdentifier.IsEmpty &&
574 Vector2.DistanceSquared(transducerCenter, trigger.WorldPosition) < pingRange / 2 * pingRange / 2)
575 {
576 ballastFloraSpores.Add(trigger);
577 }
578 }
579 }
580 }
581
582 foreach (KeyValuePair<LevelTrigger, Vector2> triggerFlow in levelTriggerFlows)
583 {
584 LevelTrigger trigger = triggerFlow.Key;
585 Vector2 flow = triggerFlow.Value;
586
587 float flowMagnitude = flow.Length();
588 if (Rand.Range(0.0f, 1.0f) < flowMagnitude / 1000.0f)
589 {
590 float edgeDist = Rand.Range(0.0f, 1.0f);
591 Vector2 blipPos = trigger.WorldPosition + Rand.Vector(trigger.ColliderRadius * edgeDist);
592 Vector2 blipVel = flow;
593
594 //go through other triggers in range and add the flows of the ones that the blip is inside
595 foreach (KeyValuePair<LevelTrigger, Vector2> triggerFlow2 in levelTriggerFlows)
596 {
597 LevelTrigger trigger2 = triggerFlow2.Key;
598 if (trigger2 != trigger && Vector2.DistanceSquared(blipPos, trigger2.WorldPosition) < trigger2.ColliderRadius * trigger2.ColliderRadius)
599 {
600 Vector2 trigger2flow = triggerFlow2.Value;
601 if (trigger2.ForceFalloff) trigger2flow *= 1.0f - Vector2.Distance(blipPos, trigger2.WorldPosition) / trigger2.ColliderRadius;
602 blipVel += trigger2flow;
603 }
604 }
605 var flowBlip = new SonarBlip(blipPos, Rand.Range(0.5f, 1.0f), 1.0f)
606 {
607 Velocity = blipVel * Rand.Range(1.0f, 5.0f),
608 Size = new Vector2(MathHelper.Lerp(0.4f, 5f, flowMagnitude / 500.0f), 0.2f),
609 Rotation = (float)Math.Atan2(-blipVel.Y, blipVel.X)
610 };
611 sonarBlips.Add(flowBlip);
612 }
613 }
614
615 foreach (LevelTrigger spore in ballastFloraSpores)
616 {
617 Vector2 blipPos = spore.WorldPosition + Rand.Vector(spore.ColliderRadius * Rand.Range(0.0f, 1.0f));
618 SonarBlip sporeBlip = new SonarBlip(blipPos, Rand.Range(0.1f, 0.5f), 0.5f)
619 {
620 Rotation = Rand.Range(-MathHelper.TwoPi, MathHelper.TwoPi),
621 BlipType = BlipType.Default,
622 Velocity = Rand.Vector(100f, Rand.RandSync.Unsynced)
623 };
624
625 sonarBlips.Add(sporeBlip);
626 }
627
628 float outsideLevelFlow = 0.0f;
629 if (transducerCenter.X < 0.0f)
630 {
631 outsideLevelFlow = Math.Abs(transducerCenter.X * 0.001f);
632 }
633 else if (transducerCenter.X > Level.Loaded.Size.X)
634 {
635 outsideLevelFlow = -(transducerCenter.X - Level.Loaded.Size.X) * 0.001f;
636 }
637
638 if (Rand.Range(0.0f, 100.0f) < Math.Abs(outsideLevelFlow))

1458 private void CreateBlipsForLine(Vector2 point1, Vector2 point2, Vector2 pingSource, Vector2 transducerPos, float pingRadius, float prevPingRadius,
1459 float lineStep, float zStep, float range, float pingStrength, bool passive, BlipType blipType = BlipType.Default)
1460 {
1461 lineStep /= zoom;
1462 zStep /= zoom;
1463 range *= displayScale;
1464 float length = (point1 - point2).Length();
1465 Vector2 lineDir = (point2 - point1) / length;
1466 for (float x = 0; x < length; x += lineStep * Rand.Range(0.8f, 1.2f))
1467 {
1468 if (Rand.Int(sonarBlips.Count) > 500) { continue; }
1469
1470 Vector2 point = point1 + lineDir * x;
1471
1472 //ignore if outside the display
1473 Vector2 transducerDiff = point - transducerPos;
1474 Vector2 transducerDisplayDiff = transducerDiff * displayScale;
1475 if (transducerDisplayDiff.LengthSquared() > DisplayRadius * DisplayRadius) { continue; }
1476
1477 //ignore if the point is not within the ping
1478 Vector2 pointDiff = point - pingSource;
1479 Vector2 displayPointDiff = pointDiff * displayScale;
1480 float displayPointDistSqr = displayPointDiff.LengthSquared();
1481 if (displayPointDistSqr < prevPingRadius * prevPingRadius || displayPointDistSqr > pingRadius * pingRadius) { continue; }
1482
1483 //ignore if direction is disrupted
1484 float transducerDist = transducerDiff.Length();
1485 Vector2 pingDirection = transducerDiff / transducerDist;
1486 bool disrupted = false;
1487 foreach ((Vector2 disruptPos, float disruptStrength) in disruptedDirections)
1488 {
1489 float dot = Vector2.Dot(pingDirection, disruptPos);
1490 if (dot > 1.0f - disruptStrength)
1491 {
1492 disrupted = true;
1493 break;
1494 }
1495 }
1496 if (disrupted) { continue; }
1497
1498 float displayPointDist = (float)Math.Sqrt(displayPointDistSqr);
1499 float alpha = pingStrength * Rand.Range(1.5f, 2.0f);
1500 for (float z = 0; z < DisplayRadius - transducerDist * displayScale; z += zStep)
1501 {
1502 Vector2 pos = point + Rand.Vector(150.0f / zoom) + pingDirection * z / displayScale;
1503 float fadeTimer = alpha * (1.0f - displayPointDist / range);
1504
1505 int minDist = (int)(200 / zoom);
1506 sonarBlips.RemoveAll(b => b.FadeTimer < fadeTimer && Math.Abs(pos.X - b.Position.X) < minDist && Math.Abs(pos.Y - b.Position.Y) < minDist);
1507
1508 var blip = new SonarBlip(pos, fadeTimer, 1.0f + ((displayPointDist + z) / DisplayRadius), blipType);
1509 if (!passive && !CheckBlipVisibility(blip, transducerPos)) { continue; }
1510
1511 sonarBlips.Add(blip);
1512 zStep += 0.5f / zoom;
1513
1514 if (z == 0)
1515 {
1516 alpha = Math.Min(alpha - 0.5f, 1.5f);
1517 }
1518 else
1519 {
1520 alpha -= 0.1f;
1521 }
1522
1523 if (alpha < 0) { break; }
1524 }
1525 }
