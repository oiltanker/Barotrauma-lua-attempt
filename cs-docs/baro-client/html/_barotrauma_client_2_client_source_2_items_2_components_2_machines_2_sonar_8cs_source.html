513  void AddIfValid(Level.ClusterLocation c)
514  {
515  if (c.Resources == null) { return; }
516  if (c.Resources.None(i => i != null && !i.Removed && i.Tags.Contains("ore"))) { return; }
517  var pos = Vector2.Zero;
518  foreach (var r in c.Resources)
519  {
520  pos += r.WorldPosition;
521  }
522  pos /= c.Resources.Count;
523  MineralClusters.Add((center: pos, resources: c.Resources));
524  }
525  }
526  else
527  {
528  MineralClusters.RemoveAll(c => c.resources == null || c.resources.None() || c.resources.All(i => i == null || i.Removed));
529  }
530  }
531 
532  if (UseTransducers && connectedTransducers.Count == 0)
533  {
534  return;
535  }
536 
537  if (Level.Loaded != null)
538  {
539  nearbyObjectUpdateTimer -= deltaTime;
540  if (nearbyObjectUpdateTimer <= 0.0f)
541  {
542  nearbyObjects.Clear();
543  foreach (var nearbyObject in Level.Loaded.LevelObjectManager.GetAllObjects(transducerCenter, range * zoom))
544  {
545  if (!nearbyObject.VisibleOnSonar) { continue; }
546  float objectRange = range + nearbyObject.SonarRadius;
547  if (Vector2.DistanceSquared(transducerCenter, nearbyObject.WorldPosition) < objectRange * objectRange)
548  {
549  nearbyObjects.Add(nearbyObject);
550  }
551  }
552  nearbyObjectUpdateTimer = NearbyObjectUpdateInterval;
553  }
554 
555  List<LevelTrigger> ballastFloraSpores = new List<LevelTrigger>();
556  Dictionary<LevelTrigger, Vector2> levelTriggerFlows = new Dictionary<LevelTrigger, Vector2>();
557  for (var pingIndex = 0; pingIndex < activePingsCount; ++pingIndex)
558  {
559  var activePing = activePings[pingIndex];
560  float pingRange = range * activePing.State / zoom;
561  foreach (LevelObject levelObject in nearbyObjects)
562  {
563  if (levelObject.Triggers == null) { continue; }
564  //gather all nearby triggers that are causing the water to flow into the dictionary
565  foreach (LevelTrigger trigger in levelObject.Triggers)
566  {
567  Vector2 flow = trigger.GetWaterFlowVelocity();
568  //ignore ones that are barely doing anything (flow^2 <= 1)
569  if (flow.LengthSquared() >= 1.0f && !levelTriggerFlows.ContainsKey(trigger))
570  {
571  levelTriggerFlows.Add(trigger, flow);
572  }
573  if (!trigger.InfectIdentifier.IsEmpty &&
574  Vector2.DistanceSquared(transducerCenter, trigger.WorldPosition) < pingRange / 2 * pingRange / 2)
575  {
576  ballastFloraSpores.Add(trigger);
577  }
578  }
579  }
580  }
581 
582  foreach (KeyValuePair<LevelTrigger, Vector2> triggerFlow in levelTriggerFlows)
583  {
584  LevelTrigger trigger = triggerFlow.Key;
585  Vector2 flow = triggerFlow.Value;
586 
587  float flowMagnitude = flow.Length();
588  if (Rand.Range(0.0f, 1.0f) < flowMagnitude / 1000.0f)
589  {
590  float edgeDist = Rand.Range(0.0f, 1.0f);
591  Vector2 blipPos = trigger.WorldPosition + Rand.Vector(trigger.ColliderRadius * edgeDist);
592  Vector2 blipVel = flow;
593 
594  //go through other triggers in range and add the flows of the ones that the blip is inside
595  foreach (KeyValuePair<LevelTrigger, Vector2> triggerFlow2 in levelTriggerFlows)
596  {
597  LevelTrigger trigger2 = triggerFlow2.Key;
598  if (trigger2 != trigger && Vector2.DistanceSquared(blipPos, trigger2.WorldPosition) < trigger2.ColliderRadius * trigger2.ColliderRadius)
599  {
600  Vector2 trigger2flow = triggerFlow2.Value;
601  if (trigger2.ForceFalloff) trigger2flow *= 1.0f - Vector2.Distance(blipPos, trigger2.WorldPosition) / trigger2.ColliderRadius;
602  blipVel += trigger2flow;
