275 public void RefreshSubmarineDisplay(bool updateSubs)
276 {
277 if (!initialized)
278 {
279 Initialize();
280 }
281 if (GameMain.GraphicsWidth != createdForResolution.X || GameMain.GraphicsHeight != createdForResolution.Y)
282 {
283 CreateGUI();
284 }
285 else
286 {
287 playerBalanceElement = CampaignUI.UpdateBalanceElement(playerBalanceElement);
288 }
289 if (updateSubs)
290 {
291 UpdateSubmarines();
292 }
293
294 if (pageIndicators != null)
295 {
296 for (int i = 0; i < pageIndicators.Length; i++)
297 {
298 pageIndicators[i].Color = i == currentPage - 1 ? Color.White : Color.Gray;
299 }
300 }
301
302 int submarineIndex = (currentPage - 1) * submarinesPerPage;
303
304 for (int i = 0; i < submarineDisplays.Length; i++)
305 {
306 SubmarineInfo subToDisplay = GetSubToDisplay(submarineIndex);
307 if (subToDisplay == null)
308 {
309 submarineDisplays[i].submarineImage.Sprite = null;
310 submarineDisplays[i].submarineName.Text = string.Empty;
311 submarineDisplays[i].submarineFee.Text = string.Empty;
312 submarineDisplays[i].submarineClass.Text = string.Empty;
313 submarineDisplays[i].selectSubmarineButton.Enabled = false;
314 submarineDisplays[i].selectSubmarineButton.OnClicked = null;
315 submarineDisplays[i].displayedSubmarine = null;
316 submarineDisplays[i].middleTextBlock.AutoDraw = false;
317 submarineDisplays[i].previewButton.Visible = false;
318 }
319 else
320 {
321 submarineDisplays[i].displayedSubmarine = subToDisplay;
322 Sprite previewImage = GetPreviewImage(subToDisplay);
323
324 if (previewImage != null)
325 {
326 submarineDisplays[i].submarineImage.Sprite = previewImage;
327 submarineDisplays[i].middleTextBlock.AutoDraw = false;
328 }
329 else
330 {
331 submarineDisplays[i].submarineImage.Sprite = null;
332 submarineDisplays[i].middleTextBlock.Text = missingPreviewText;
333 submarineDisplays[i].middleTextBlock.AutoDraw = true;
334 }
335
336 submarineDisplays[i].selectSubmarineButton.Enabled = true;
337
338 int index = i;
339 submarineDisplays[i].selectSubmarineButton.OnClicked = (button, userData) =>
340 {
341 SelectSubmarine(subToDisplay, submarineDisplays[index].background.Rect);
342 return true;
343 };
344
345 submarineDisplays[i].submarineName.Text = subToDisplay.DisplayName;
346 submarineDisplays[i].submarineClass.Text = TextManager.GetWithVariable("submarineclass.classsuffixformat", "[type]", TextManager.Get($"submarineclass.{subToDisplay.SubmarineClass}"));
347
348 if (!GameMain.GameSession.IsSubmarineOwned(subToDisplay))
349 {
350 LocalizedString amountString = TextManager.FormatCurrency(subToDisplay.Price);
351 submarineDisplays[i].submarineFee.Text = TextManager.GetWithVariable("price", "[amount]", amountString);
352 }
353 else
354 {
355 if (subToDisplay.Name != CurrentOrPendingSubmarine().Name)
356 {
357 if (deliveryFee > 0)
358 {
359 LocalizedString amountString = TextManager.FormatCurrency(deliveryFee);
360 submarineDisplays[i].submarineFee.Text = TextManager.GetWithVariable("deliveryfee", "[amount]", amountString);
361 }
362 else
363 {
364 submarineDisplays[i].submarineFee.Text = string.Empty;
365 }
366 }
367 else
368 {
369 submarineDisplays[i].submarineFee.Text = currentSubText;
370 }
371 }
372
373 if (transferService && subToDisplay.Name == CurrentOrPendingSubmarine().Name && updateSubs)
374 {
375 if (selectedSubmarine == null)
376 {
377 CoroutineManager.StartCoroutine(SelectOwnSubmarineWithDelay(subToDisplay, submarineDisplays[i]));
378 }
379 else
380 {
381 SelectSubmarine(subToDisplay, submarineDisplays[i].background.Rect);
382 }
383 }
384 else if (!transferService && selectedSubmarine == null || !transferService && GameMain.GameSession.IsSubmarineOwned(selectedSubmarine) || subToDisplay == selectedSubmarine)
385 {
386 SelectSubmarine(subToDisplay, submarineDisplays[i].background.Rect);
387 }
388
389 submarineDisplays[i].previewButton.Visible = true;
390 submarineDisplays[i].previewButton.OnClicked = (btn, obj) =>
391 {
392 SubmarinePreview.Create(subToDisplay);
393 return false;
394 };
395 }
396
397 submarineIndex++;
398 }
399
400 if (subsToShow.Count == 0)
401 {
402 SelectSubmarine(null, Rectangle.Empty);
403 }
404 }

406 private void UpdateSubmarines()
407 {
408 subsToShow.Clear();
409 if (transferService)
410 {
411 subsToShow.AddRange(GameMain.GameSession.OwnedSubmarines);
412 subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
413 string currentSubName = CurrentOrPendingSubmarine().Name;
414 int currentIndex = subsToShow.FindIndex(s => s.Name == currentSubName);
415 if (currentIndex != -1)
416 {
417 currentPage = (int)Math.Ceiling((currentIndex + 1) / (float)submarinesPerPage);
418 }
419 }
420 else
421 {
422 subsToShow.AddRange((GameMain.Client is null ? SubmarineInfo.SavedSubmarines : MultiPlayerCampaign.GetCampaignSubs())
423 .Where(s => s.IsCampaignCompatible && !GameMain.GameSession.OwnedSubmarines.Any(os => os.Name == s.Name)));
424 subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
425 }
426
427 if (transferService) SetConfirmButtonState(selectedSubmarine != null && selectedSubmarine.Name != CurrentOrPendingSubmarine().Name);
428
429 subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
430 pageCount = Math.Max(1, (int)Math.Ceiling(subsToShow.Count / (float)submarinesPerPage));
431 UpdatePaging();
432 ContentRefreshRequired = false;
433 }

441 private Sprite GetPreviewImage(SubmarineInfo info)
442 {
443 Sprite preview = info.PreviewImage;
444
445 if (preview == null)
446 {
447 SubmarineInfo potentialMatch = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.EqualityCheckVal == info.EqualityCheckVal);
448
449 preview = potentialMatch?.PreviewImage;
450
451 // Try name comparison as a backup
452 if (preview == null)
453 {
454 potentialMatch = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Name == info.Name);
455 preview = potentialMatch?.PreviewImage;
456 }
457 }
458
459 return preview;
460 }

469 // Selection based on key input
470 private void SelectSubmarine(int index, int direction)
471 {
472 SubmarineInfo nextSub = GetSubToDisplay(index + direction);
473 if (nextSub == null) return;
474
475 for (int i = 0; i < submarineDisplays.Length; i++)
476 {
477 if (submarineDisplays[i].displayedSubmarine == nextSub)
478 {
479 SelectSubmarine(nextSub, submarineDisplays[i].background.Rect);
480 return;
481 }
482 }
483
484 ChangePage(direction);
485
486 for (int i = 0; i < submarineDisplays.Length; i++)
487 {
488 if (submarineDisplays[i].displayedSubmarine == nextSub)
489 {
490 SelectSubmarine(nextSub, submarineDisplays[i].background.Rect);
491 return;
492 }
493 }
