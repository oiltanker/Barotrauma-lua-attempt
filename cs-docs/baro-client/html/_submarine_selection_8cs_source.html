406  private void UpdateSubmarines()
407  {
408  subsToShow.Clear();
409  if (transferService)
410  {
411  subsToShow.AddRange(GameMain.GameSession.OwnedSubmarines);
412  subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
413  string currentSubName = CurrentOrPendingSubmarine().Name;
414  int currentIndex = subsToShow.FindIndex(s => s.Name == currentSubName);
415  if (currentIndex != -1)
416  {
417  currentPage = (int)Math.Ceiling((currentIndex + 1) / (float)submarinesPerPage);
418  }
419  }
420  else
421  {
422  subsToShow.AddRange((GameMain.Client is null ? SubmarineInfo.SavedSubmarines : MultiPlayerCampaign.GetCampaignSubs())
423  .Where(s => s.IsCampaignCompatible && !GameMain.GameSession.OwnedSubmarines.Any(os => os.Name == s.Name)));
424  subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
425  }
426 
427  if (transferService) SetConfirmButtonState(selectedSubmarine != null && selectedSubmarine.Name != CurrentOrPendingSubmarine().Name);
428 
429  subsToShow.Sort((x, y) => x.SubmarineClass.CompareTo(y.SubmarineClass));
430  pageCount = Math.Max(1, (int)Math.Ceiling(subsToShow.Count / (float)submarinesPerPage));
431  UpdatePaging();
432  ContentRefreshRequired = false;
433  }

441  private Sprite GetPreviewImage(SubmarineInfo info)
442  {
443  Sprite preview = info.PreviewImage;
444 
445  if (preview == null)
446  {
447  SubmarineInfo potentialMatch = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.EqualityCheckVal == info.EqualityCheckVal);
448 
449  preview = potentialMatch?.PreviewImage;
450 
451  // Try name comparison as a backup
452  if (preview == null)
453  {
454  potentialMatch = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Name == info.Name);
455  preview = potentialMatch?.PreviewImage;
456  }
457  }
458 
459  return preview;
