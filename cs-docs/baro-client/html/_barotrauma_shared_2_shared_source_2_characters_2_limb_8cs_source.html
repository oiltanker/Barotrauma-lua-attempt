712  private readonly List<DamageModifier> appliedDamageModifiers = new List<DamageModifier>();
713  private readonly List<DamageModifier> tempModifiers = new List<DamageModifier>();
714  private readonly List<Affliction> afflictionsCopy = new List<Affliction>();
715  public AttackResult AddDamage(Vector2 simPosition, IEnumerable<Affliction> afflictions, bool playSound, float damageMultiplier = 1, float penetration = 0f, Character attacker = null)
716  {
717  appliedDamageModifiers.Clear();
718  afflictionsCopy.Clear();
719  foreach (var affliction in afflictions)
720  {
721  tempModifiers.Clear();
722  var newAffliction = affliction;
723  float random = Rand.Value(Rand.RandSync.Unsynced);
724  if (random > affliction.Probability) { continue; }
725  bool applyAffliction = true;
726  foreach (DamageModifier damageModifier in DamageModifiers)
727  {
728  if (!damageModifier.MatchesAffliction(affliction)) { continue; }
729  if (random > affliction.Probability * damageModifier.ProbabilityMultiplier)
730  {
731  applyAffliction = false;
732  continue;
733  }
734  if (SectorHit(damageModifier.ArmorSectorInRadians, simPosition))
735  {
736  tempModifiers.Add(damageModifier);
737  }
738  }
