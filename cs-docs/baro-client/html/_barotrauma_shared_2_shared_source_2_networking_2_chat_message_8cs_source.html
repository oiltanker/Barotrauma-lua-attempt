160  public static float GetGarbleAmount(Entity listener, Entity sender, float range, float obstructionmult = 2.0f)
161  {
162  if (listener == null || sender == null)
163  {
164  return 0.0f;
165  }
166  if (listener.WorldPosition == sender.WorldPosition) { return 0.0f; }
167 
168  float dist = Vector2.Distance(listener.WorldPosition, sender.WorldPosition);
169  if (dist > range) { return 1.0f; }
170 
171  Hull listenerHull = listener == null ? null : Hull.FindHull(listener.WorldPosition);
172  Hull sourceHull = sender == null ? null : Hull.FindHull(sender.WorldPosition);
173  if (sourceHull != listenerHull)
174  {
175  if ((sourceHull == null || !sourceHull.GetConnectedHulls(includingThis: false, searchDepth: 2, ignoreClosedGaps: true).Contains(listenerHull)) &&
176  Submarine.CheckVisibility(listener.SimPosition, sender.SimPosition) != null)
177  {
178  dist = (dist + 100f) * obstructionmult;
179  }
180  }
181  if (dist > range) { return 1.0f; }
182 
183  return dist / range;
184  }

198  public static string ApplyDistanceEffect(string text, float garbleAmount)
199  {
200  if (garbleAmount < 0.3f) return text;
201  if (garbleAmount >= 1.0f) return "";
202 
203  int startIndex = Math.Max(text.IndexOf(':') + 1, 1);
204 
205  StringBuilder sb = new StringBuilder(text.Length);
206  for (int i = 0; i < text.Length; i++)
207  {
208  sb.Append((i > startIndex && Rand.Range(0.0f, 1.0f) < garbleAmount) ? '-' : text[i]);
209  }
210 
211  return sb.ToString();
212  }

214  public static string ApplyDistanceEffect(string message, ChatMessageType type, Character sender, Character receiver)
215  {
216  if (sender == null) { return ""; }
217  float range = SpeakRange;
218  if (type == ChatMessageType.Default && sender.SpeechImpediment > 0)
219  {
220  range *= 1.0f - sender.SpeechImpediment / 100.0f;
221  }
222  string spokenMsg = ApplyDistanceEffect(receiver, sender, message, range, 3.0f);
223 
224  switch (type)
225  {
226  case ChatMessageType.Default:
227  if (receiver != null && !receiver.IsDead)
228  {
229  return spokenMsg;
230  }
231  break;
232  case ChatMessageType.Radio:
233  case ChatMessageType.Order:
234  if (receiver?.Inventory != null && !receiver.IsDead)
235  {
236  foreach (Item receiverItem in receiver.Inventory.AllItems.Where(i => i.GetComponent<WifiComponent>()?.LinkToChat ?? false))
237  {
238  if (sender.Inventory == null || !receiver.HasEquippedItem(receiverItem)) { continue; }
239 
240  foreach (Item senderItem in sender.Inventory.AllItems.Where(i => i.GetComponent<WifiComponent>()?.LinkToChat ?? false))
241  {
242  if (!sender.HasEquippedItem(senderItem)) { continue; }
243 
244  var receiverRadio = receiverItem.GetComponent<WifiComponent>();
245  var senderRadio = senderItem.GetComponent<WifiComponent>();
246  if (!receiverRadio.CanReceive(senderRadio)) { continue; }
247 
248  string msg = ApplyDistanceEffect(receiverItem, senderItem, message, senderRadio.Range);
249  if (sender.SpeechImpediment > 0.0f)
250  {
251  //speech impediment doesn't reduce the range when using a radio, but adds extra garbling
252  msg = ApplyDistanceEffect(msg, sender.SpeechImpediment / 100.0f);
253  }
254  return msg;
255  }
256  }
257  return spokenMsg;
258  }
259  break;
260  }
261 
262  return message;
263  }

288  if (radio == null || wifiComponent.Range > radio.Range)
