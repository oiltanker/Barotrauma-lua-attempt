96 public override void Update(float deltaTime, Camera cam)
97 {
98 base.Update(deltaTime, cam);
99
100 Entity refEntity = equipper;
101 if (isEquippable)
102 {
103 if (equipper == null || equipper.Removed)
104 {
105 IsActive = false;
106 return;
107 }
108 }
109 else
110 {
111 refEntity = item;
112 }
113
114 thermalEffectState += deltaTime;
115 thermalEffectState %= 10000.0f;
116
117 if (updateTimer > 0.0f)
118 {
119 updateTimer -= deltaTime;
120 return;
121 }
122
123 visibleCharacters.Clear();
124 foreach (Character c in Character.CharacterList)
125 {
126 if (c == equipper || !c.Enabled || c.Removed) { continue; }
127 if (!ShowDeadCharacters && c.IsDead) { continue; }
128
129 float dist = Vector2.DistanceSquared(refEntity.WorldPosition, c.WorldPosition);
130 if (dist < Range * Range)
131 {
132 Vector2 diff = c.WorldPosition - refEntity.WorldPosition;
133 if (Submarine.CheckVisibility(refEntity.SimPosition, refEntity.SimPosition + ConvertUnits.ToSimUnits(diff)) == null)
134 {
135 visibleCharacters.Add(c);
136 }
137 }
138 }
139
140 updateTimer = UpdateInterval;
141 }

156 public override void DrawHUD(SpriteBatch spriteBatch, Character character)
157 {
158 if (character == null) { return; }
159
160 if (OverlayColor.A > 0)
161 {
162 GUIStyle.UIGlow.Draw(spriteBatch, new Rectangle(0, 0, GameMain.GraphicsWidth, GameMain.GraphicsHeight),
163 OverlayColor);
164 }
165
166 if (ShowTexts)
167 {
168 Character closestCharacter = null;
169 float closestDist = float.PositiveInfinity;
170 foreach (Character c in visibleCharacters)
171 {
172 if (c == character || !c.Enabled || c.Removed) { continue; }
173
174 float dist = Vector2.DistanceSquared(GameMain.GameScreen.Cam.ScreenToWorld(PlayerInput.MousePosition), c.WorldPosition);
175 if (dist < closestDist)
176 {
177 closestCharacter = c;
178 closestDist = dist;
179 }
180 }
181
182 if (closestCharacter != null)
183 {
184 float dist = Vector2.Distance(GameMain.GameScreen.Cam.ScreenToWorld(PlayerInput.MousePosition), closestCharacter.WorldPosition);
185 DrawCharacterInfo(spriteBatch, closestCharacter, 1.0f - MathHelper.Max((dist - (Range - FadeOutRange)) / FadeOutRange, 0.0f));
186 }
187 }
188
189 if (ThermalGoggles)
190 {
191 spriteBatch.End();
192 GameMain.LightManager.SolidColorEffect.Parameters["color"].SetValue(Color.Red.ToVector4() * (0.3f + MathF.Sin(thermalEffectState) * 0.05f));
193 GameMain.LightManager.SolidColorEffect.CurrentTechnique = GameMain.LightManager.SolidColorEffect.Techniques["SolidColorBlur"];
194 GameMain.LightManager.SolidColorEffect.Parameters["blurDistance"].SetValue(0.01f + MathF.Sin(thermalEffectState) * 0.005f);
195 GameMain.LightManager.SolidColorEffect.CurrentTechnique.Passes[0].Apply();
196 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: Screen.Selected.Cam.Transform, effect: GameMain.LightManager.SolidColorEffect);
197
198 Entity refEntity = equipper;
199 if (!isEquippable || refEntity == null)
200 {
201 refEntity = item;
202 }
203
204 foreach (Character c in Character.CharacterList)
205 {
206 if (c == character || !c.Enabled || c.Removed || c.Params.HideInThermalGoggles) { continue; }
207 if (!ShowDeadCharacters && c.IsDead) { continue; }
208
209 float dist = Vector2.DistanceSquared(refEntity.WorldPosition, c.WorldPosition);
210 if (dist > Range * Range) { continue; }
211
212 Sprite pingCircle = GUIStyle.UIThermalGlow.Value.Sprite;
213 foreach (Limb limb in c.AnimController.Limbs)
214 {
215 if (limb.Mass < 1.0f) { continue; }
216 float noise1 = PerlinNoise.GetPerlin((thermalEffectState + limb.Params.ID + c.ID) * 0.01f, (thermalEffectState + limb.Params.ID + c.ID) * 0.02f);
217 float noise2 = PerlinNoise.GetPerlin((thermalEffectState + limb.Params.ID + c.ID) * 0.01f, (thermalEffectState + limb.Params.ID + c.ID) * 0.008f);
218 Vector2 spriteScale = ConvertUnits.ToDisplayUnits(limb.body.GetSize()) / pingCircle.size * (noise1 * 0.5f + 2f);
219 Vector2 drawPos = new Vector2(limb.body.DrawPosition.X + (noise1 - 0.5f) * 100, -limb.body.DrawPosition.Y + (noise2 - 0.5f) * 100);
220 pingCircle.Draw(spriteBatch, drawPos, 0.0f, scale: Math.Max(spriteScale.X, spriteScale.Y));
221 }
222 }
223
224 spriteBatch.End();
225 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied);
226 }

229 private void DrawCharacterInfo(SpriteBatch spriteBatch, Character target, float alpha = 1.0f)
230 {
231 Vector2 hudPos = GameMain.GameScreen.Cam.WorldToScreen(target.DrawPosition);
232 hudPos += Vector2.UnitX * 50.0f;
233
234 List<LocalizedString> texts = new List<LocalizedString>();
235 List<Color> textColors = new List<Color>();
236 texts.Add(target.Info == null ? target.DisplayName : target.Info.DisplayName);
237 Color nameColor = GUIStyle.TextColorNormal;
238 if (Character.Controlled != null && target.TeamID != Character.Controlled.TeamID)
239 {
240 nameColor = target.TeamID == CharacterTeamType.FriendlyNPC ? Color.SkyBlue : GUIStyle.Red;
241 }
242 textColors.Add(nameColor);
243
244 if (target.IsDead)
245 {
246 texts.Add(TextManager.Get("Deceased"));
247 textColors.Add(GUIStyle.Red);
248 if (target.CauseOfDeath != null)
249 {
250 texts.Add(
251 target.CauseOfDeath.Affliction?.CauseOfDeathDescription ??
252 TextManager.AddPunctuation(':', TextManager.Get("CauseOfDeath"), TextManager.Get("CauseOfDeath." + target.CauseOfDeath.Type.ToString())));
253 textColors.Add(GUIStyle.Red);
254 }
255 }
256 else
257 {
258 if (!target.CustomInteractHUDText.IsNullOrEmpty() && target.AllowCustomInteract)
259 {
260 texts.Add(target.CustomInteractHUDText);
261 textColors.Add(GUIStyle.Green);
262 }
263 if (!target.IsIncapacitated && target.IsPet)
264 {
265 texts.Add(CharacterHUD.GetCachedHudText("PlayHint", InputType.Use));
266 textColors.Add(GUIStyle.Green);
267 }
268 if (target.CharacterHealth.UseHealthWindow && !target.DisableHealthWindow && equipper?.FocusedCharacter == target && equipper.CanInteractWith(target, 160f, false))
269 {
270 texts.Add(CharacterHUD.GetCachedHudText("HealHint", InputType.Health));
271 textColors.Add(GUIStyle.Green);
272 }
273 if (target.CanBeDragged)
274 {
275 texts.Add(CharacterHUD.GetCachedHudText("GrabHint", InputType.Grab));
276 textColors.Add(GUIStyle.Green);
277 }
278
279 if (target.IsUnconscious)
280 {
281 texts.Add(TextManager.Get("Unconscious"));
282 textColors.Add(GUIStyle.Orange);
283 }
284 if (target.Stun > 0.01f)
285 {
286 texts.Add(TextManager.Get("Stunned"));
287 textColors.Add(GUIStyle.Orange);
288 }
289
290 int oxygenTextIndex = MathHelper.Clamp((int)Math.Floor((1.0f - (target.Oxygen / 100.0f)) * OxygenTexts.Length), 0, OxygenTexts.Length - 1);
291 texts.Add(OxygenTexts[oxygenTextIndex]);
292 textColors.Add(Color.Lerp(GUIStyle.Red, GUIStyle.Green, target.Oxygen / 100.0f));
293
294 if (target.Bleeding > 0.0f)
295 {
296 int bleedingTextIndex = MathHelper.Clamp((int)Math.Floor(target.Bleeding / 100.0f) * BleedingTexts.Length, 0, BleedingTexts.Length - 1);
297 texts.Add(BleedingTexts[bleedingTextIndex]);
298 textColors.Add(Color.Lerp(GUIStyle.Orange, GUIStyle.Red, target.Bleeding / 100.0f));
299 }
300
301 var allAfflictions = target.CharacterHealth.GetAllAfflictions();
302 Dictionary<AfflictionPrefab, float> combinedAfflictionStrengths = new Dictionary<AfflictionPrefab, float>();
303 foreach (Affliction affliction in allAfflictions)
304 {
305 if (affliction.Strength < affliction.Prefab.ShowInHealthScannerThreshold || affliction.Strength <= 0.0f) { continue; }
306 if (combinedAfflictionStrengths.ContainsKey(affliction.Prefab))
307 {
308 combinedAfflictionStrengths[affliction.Prefab] += affliction.Strength;
309 }
310 else
311 {
312 combinedAfflictionStrengths[affliction.Prefab] = affliction.Strength;
313 }
314 }
315
316 foreach (AfflictionPrefab affliction in combinedAfflictionStrengths.Keys)
317 {
318 texts.Add(TextManager.AddPunctuation(':', affliction.Name, Math.Max((int)combinedAfflictionStrengths[affliction], 1).ToString() + " %"));
319 textColors.Add(Color.Lerp(GUIStyle.Orange, GUIStyle.Red, combinedAfflictionStrengths[affliction] / affliction.MaxStrength));
320 }
321 }
322
323 GUI.DrawString(spriteBatch, hudPos, texts[0].Value, textColors[0] * alpha, Color.Black * 0.7f * alpha, 2, GUIStyle.SubHeadingFont, ForceUpperCase.No);
324 hudPos.X += 5.0f;
325 hudPos.Y += 24.0f * GameSettings.CurrentConfig.Graphics.TextScale;
326
327 hudPos.X = (int)hudPos.X;
328 hudPos.Y = (int)hudPos.Y;
329
330 for (int i = 1; i < texts.Count; i++)
331 {
332 GUI.DrawString(spriteBatch, hudPos, texts[i], textColors[i] * alpha, Color.Black * 0.7f * alpha, 2, GUIStyle.SmallFont);
333 hudPos.Y += (int)(18.0f * GameSettings.CurrentConfig.Graphics.TextScale);
334 }
335 }
