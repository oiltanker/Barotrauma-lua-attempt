134  private static void UpdateWaterAmbience(float ambienceVolume, float deltaTime)
135  {
136  if (GameMain.SoundManager.Disabled || GameMain.GameScreen?.Cam == null) { return; }
137 
138  //how fast the sub is moving, scaled to 0.0 -> 1.0
139  float movementSoundVolume = 0.0f;
140 
141  float insideSubFactor = 0.0f;
142  foreach (Submarine sub in Submarine.Loaded)
143  {
144  if (sub == null || sub.Removed) { continue; }
145  float movementFactor = (sub.Velocity == Vector2.Zero) ? 0.0f : sub.Velocity.Length() / 10.0f;
146  movementFactor = MathHelper.Clamp(movementFactor, 0.0f, 1.0f);
147 
148  if (Character.Controlled == null || Character.Controlled.Submarine != sub)
149  {
150  float dist = Vector2.Distance(GameMain.GameScreen.Cam.WorldViewCenter, sub.WorldPosition);
151  movementFactor /= Math.Max(dist / 1000.0f, 1.0f);
152  insideSubFactor = Math.Max(1.0f / Math.Max(dist / 1000.0f, 1.0f), insideSubFactor);
153  }
154  else
155  {
156  insideSubFactor = 1.0f;
157  }
158 
159  movementSoundVolume = Math.Max(movementSoundVolume, movementFactor);
160  if (!MathUtils.IsValid(movementSoundVolume))
161  {
162  string errorMsg = "Failed to update water ambience volume - submarine's movement value invalid (" + movementSoundVolume + ", sub velocity: " + sub.Velocity + ")";
163  DebugConsole.Log(errorMsg);
164  GameAnalyticsManager.AddErrorEventOnce("SoundPlayer.UpdateWaterAmbience:InvalidVolume", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
165  movementSoundVolume = 0.0f;
166  }
167  if (!MathUtils.IsValid(insideSubFactor))
168  {
169  string errorMsg = "Failed to update water ambience volume - inside sub value invalid (" + insideSubFactor + ")";
170  DebugConsole.Log(errorMsg);
171  GameAnalyticsManager.AddErrorEventOnce("SoundPlayer.UpdateWaterAmbience:InvalidVolume", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
172  insideSubFactor = 0.0f;
173  }

202  private static void UpdateWaterFlowSounds(float deltaTime)
203  {
204  if (FlowSounds.Count == 0) { return; }
205 
206  for (int i = 0; i < targetFlowLeft.Length; i++)
207  {
208  targetFlowLeft[i] = 0.0f;
209  targetFlowRight[i] = 0.0f;
210  }
211 
212  Vector2 listenerPos = new Vector2(GameMain.SoundManager.ListenerPosition.X, GameMain.SoundManager.ListenerPosition.Y);
213  foreach (Gap gap in Gap.GapList)
214  {
215  Vector2 diff = gap.WorldPosition - listenerPos;
216  if (Math.Abs(diff.X) < FlowSoundRange && Math.Abs(diff.Y) < FlowSoundRange)
217  {
218  if (gap.Open < 0.01f || gap.LerpedFlowForce.LengthSquared() < 100.0f) { continue; }
219  float gapFlow = Math.Abs(gap.LerpedFlowForce.X) + Math.Abs(gap.LerpedFlowForce.Y) * 2.5f;
220  if (!gap.IsRoomToRoom) { gapFlow *= 2.0f; }
221  if (gapFlow < 10.0f) { continue; }
222 
223  if (gap.linkedTo.Count == 2 && gap.linkedTo[0] is Hull hull1 && gap.linkedTo[1] is Hull hull2)
224  {
225  //no flow sounds between linked hulls (= rooms consisting of multiple hulls)
226  if (hull1.linkedTo.Contains(hull2)) { continue; }
227  if (hull1.linkedTo.Any(h => h.linkedTo.Contains(hull1) && h.linkedTo.Contains(hull2))) { continue; }
228  if (hull2.linkedTo.Any(h => h.linkedTo.Contains(hull1) && h.linkedTo.Contains(hull2))) { continue; }
229  }
230 
231  int flowSoundIndex = (int)Math.Floor(MathHelper.Clamp(gapFlow / MaxFlowStrength, 0, FlowSounds.Count));
232  flowSoundIndex = Math.Min(flowSoundIndex, FlowSounds.Count - 1);
233 
234  float dist = diff.Length();
235  float distFallOff = dist / FlowSoundRange;
236  if (distFallOff >= 0.99f) continue;
237 
238  //flow at the left side
239  if (diff.X < 0)
240  {
241  targetFlowLeft[flowSoundIndex] += 1.0f - distFallOff;
242  }
243  else
244  {
245  targetFlowRight[flowSoundIndex] += 1.0f - distFallOff;
246  }
247  }
248  }
249 
250  if (Character.Controlled?.CharacterHealth?.GetAffliction("psychosis") is AfflictionPsychosis psychosis)
251  {
252  if (psychosis.CurrentFloodType == AfflictionPsychosis.FloodType.Minor)
253  {
254  targetFlowLeft[0] = Math.Max(targetFlowLeft[0], 1.0f);
255  targetFlowRight[0] = Math.Max(targetFlowRight[0], 1.0f);
256  }
257  else if (psychosis.CurrentFloodType == AfflictionPsychosis.FloodType.Major)
258  {
259  targetFlowLeft[FlowSounds.Count - 1] = Math.Max(targetFlowLeft[FlowSounds.Count - 1], 1.0f);
260  targetFlowRight[FlowSounds.Count - 1] = Math.Max(targetFlowRight[FlowSounds.Count - 1], 1.0f);
261  }
262  }
263 
264  for (int i = 0; i < FlowSounds.Count; i++)
265  {
266  flowVolumeLeft[i] = (targetFlowLeft[i] < flowVolumeLeft[i]) ?
267  Math.Max(targetFlowLeft[i], flowVolumeLeft[i] - deltaTime) :
268  Math.Min(targetFlowLeft[i], flowVolumeLeft[i] + deltaTime * 10.0f);
269  flowVolumeRight[i] = (targetFlowRight[i] < flowVolumeRight[i]) ?
270  Math.Max(targetFlowRight[i], flowVolumeRight[i] - deltaTime) :
271  Math.Min(targetFlowRight[i], flowVolumeRight[i] + deltaTime * 10.0f);
272 
273  if (flowVolumeLeft[i] < 0.05f && flowVolumeRight[i] < 0.05f)
274  {
275  if (flowSoundChannels[i] != null)
276  {
277  flowSoundChannels[i].Dispose();
278  flowSoundChannels[i] = null;
279  }
280  }
281  else
282  {
283  if (FlowSounds[i] == null) { continue; }
284  Vector2 soundPos = new Vector2(GameMain.SoundManager.ListenerPosition.X + (flowVolumeRight[i] - flowVolumeLeft[i]) * 100, GameMain.SoundManager.ListenerPosition.Y);
285  if (flowSoundChannels[i] == null || !flowSoundChannels[i].IsPlaying)
286  {
287  flowSoundChannels[i] = FlowSounds[i].Sound.Play(1.0f, FlowSoundRange, soundPos);
288  flowSoundChannels[i].Looping = true;
289  }
290  flowSoundChannels[i].Gain = Math.Min(Math.Max(flowVolumeRight[i], flowVolumeLeft[i]), 1.0f);
291  flowSoundChannels[i].Position = new Vector3(soundPos, 0.0f);
292  }
293  }

341  void AddFireVolume(FireSource fs)
342  {
343  Vector2 diff = fs.WorldPosition + fs.Size / 2 - listenerPos;
344  if (Math.Abs(diff.X) < FireSoundRange && Math.Abs(diff.Y) < FireSoundRange)
345  {
346  Vector2 diffLeft = (fs.WorldPosition + new Vector2(fs.Size.X, fs.Size.Y / 2)) - listenerPos;
347  if (Math.Abs(diff.X) < fs.Size.X / 2.0f) { diffLeft.X = 0.0f; }
348  if (diffLeft.X <= 0)
349  {
350  float distFallOffLeft = diffLeft.Length() / FireSoundRange;
351  if (distFallOffLeft < 0.99f)
352  {
353  fireVolumeLeft[0] += (1.0f - distFallOffLeft);
354  if (fs.Size.X > FireSoundLargeLimit)
355  {
356  fireVolumeLeft[2] += (1.0f - distFallOffLeft) * ((fs.Size.X - FireSoundLargeLimit) / FireSoundLargeLimit);
357  }
358  else if (fs.Size.X > FireSoundMediumLimit)
359  {
360  fireVolumeLeft[1] += (1.0f - distFallOffLeft) * ((fs.Size.X - FireSoundMediumLimit) / FireSoundMediumLimit);
361  }
362  }
363  }
364 
365  Vector2 diffRight = (fs.WorldPosition + new Vector2(0.0f, fs.Size.Y / 2)) - listenerPos;
366  if (Math.Abs(diff.X) < fs.Size.X / 2.0f) { diffRight.X = 0.0f; }
367  if (diffRight.X >= 0)
368  {
369  float distFallOffRight = diffRight.Length() / FireSoundRange;
370  if (distFallOffRight < 0.99f)
371  {
372  fireVolumeRight[0] += 1.0f - distFallOffRight;
373  if (fs.Size.X > FireSoundLargeLimit)
374  {
375  fireVolumeRight[2] += (1.0f - distFallOffRight) * ((fs.Size.X - FireSoundLargeLimit) / FireSoundLargeLimit);
376  }
377  else if (fs.Size.X > FireSoundMediumLimit)
378  {
379  fireVolumeRight[1] += (1.0f - distFallOffRight) * ((fs.Size.X - FireSoundMediumLimit) / FireSoundMediumLimit);
380  }
381  }
382  }
383  }
384  }
