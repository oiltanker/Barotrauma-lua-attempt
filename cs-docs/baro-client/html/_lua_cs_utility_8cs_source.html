441  private static object ParseValue(Type[] types, XElement elem)
442  {
443  var type = GetValueType(elem);
444 
445  if (elem.IsEmpty) return null;
446  if (type == ValueType.Enum)
447  {
448  var tType = GetTypeAttr(types, elem);
449  if (tType == null || !tType.IsSubclassOf(typeof(Enum))) return null;
450  if (Enum.TryParse(tType, elem.Value, out object result)) return result;
451  else return null;
452  }
453  if (type == ValueType.Collection)
454  {
455  var tType = GetTypeAttr(types, elem);
456  var tInt = tType.GetInterfaces().FirstOrDefault(i => i.IsGenericType && i.GetGenericTypeDefinition() == typeof(IEnumerable<>));
457  var gArg = tInt.GetGenericArguments()[0];
458  if (tType == null || !tType.GetInterfaces().Any(i => i.IsGenericType && i.GetGenericTypeDefinition() == typeof(IEnumerable<>))) return null;
459 
460  object result = null;
461 
462  if (result == null) {
463  var ctor = tType.GetConstructors(BindingFlags.Public | BindingFlags.Instance).FirstOrDefault(c =>
464  {
465  var param = c.GetParameters();
466  return param.Count() == 1 && param.Any(p => p.ParameterType.IsGenericType && p.ParameterType.GetGenericTypeDefinition() == typeof(IEnumerable<>));
467  });
468  if (ctor != null)
469  {
470  var elements = elem.Elements().Select(x => ParseValue(types, x));
471  var castElems = typeof(Enumerable).GetMethod("Cast").MakeGenericMethod(gArg).Invoke(elements, new object[] { elements });
472  result = ctor.Invoke(new object[] { castElems });
473  }
474  }
475 
476  if (result == null)
477  {
478  var ctor = tType.GetConstructors(BindingFlags.Public | BindingFlags.Instance).FirstOrDefault(c => c.GetParameters().Count() == 0);
479  var addMethod = tType.GetMethods(BindingFlags.Instance | BindingFlags.Public).FirstOrDefault(m =>
480  {
481  if (m.Name != "Add") return false;
482  var param = m.GetParameters();
483  return param.Count() == 1 && param[0].ParameterType == gArg;
484  });
485  if (ctor != null && addMethod != null)
486  {
487  var elements = elem.Elements().Select(x => ParseValue(types, x));
488  result = ctor.Invoke(null);
489  foreach (var el in elements) addMethod.Invoke(result, new object[] { el });
490  }
491  }
492 
493  if (result == null)
494  {
495  var ctor = tType.GetConstructors(BindingFlags.Public | BindingFlags.Instance).FirstOrDefault();
496  var setMethod = tType.GetMethods(BindingFlags.Instance | BindingFlags.Public).FirstOrDefault(m =>
497  {
498  if (m.Name != "Set") return false;
499  var param = m.GetParameters();
500  return param.Count() == 2 && param[0].ParameterType == typeof(int) && param[1].ParameterType == gArg;
501  });
502  if (ctor != null || setMethod != null)
503  {
504  var elements = elem.Elements().Select(x => ParseValue(types, x));
505  result = ctor.Invoke(new object[] { elements.Count() });
506  int i = 0;
507  foreach (var el in elements)
508  {
509  setMethod.Invoke(result, new object[] { i, el });
510  i++;
511  }
512  }
513  }
514 
515  return result;
516  }
517  else if (type == ValueType.Text) return elem.Value;
518  else if (type == ValueType.Integer)
519  {
520  int.TryParse(elem.Value, out var num);
521  return num;
522  }
523  else if (type == ValueType.Decimal)
524  {
525  float.TryParse(elem.Value, out var num);
526  return num;
527  }
528  else if (type == ValueType.Boolean)
529  {
530  bool.TryParse(elem.Value, out var boolean);
531  return boolean;
532  }
533  else if (type == ValueType.Object)
534  {
535  var tType = GetTypeAttr(types, elem);
536  if (tType == null) return null;
537 
538  IEnumerable<FieldInfo> fields = tType.GetFields(BindingFlags.Instance | BindingFlags.Public)
539  .Concat(tType.GetFields(BindingFlags.Instance | BindingFlags.NonPublic));
540  IEnumerable<PropertyInfo> properties = tType.GetProperties(BindingFlags.Instance | BindingFlags.Public).Where(p => p.GetSetMethod() != null)
541  .Concat(tType.GetProperties(BindingFlags.Instance | BindingFlags.NonPublic).Where(p => p.GetSetMethod() != null));
542 
543  object result = null;
544  var ctor = tType.GetConstructors(BindingFlags.Public | BindingFlags.Instance).FirstOrDefault(c => c.GetParameters().Count() == 0);
545  if (ctor == null)
546  {
547  if (!tType.IsValueType) return null;
548  result = Activator.CreateInstance(tType);
549  }
550  else result = ctor.Invoke(null);
551 
552  foreach(var el in elem.Elements())
553  {
554  var value = ParseValue(types, el);
555 
556  var field = fields.FirstOrDefault(f => f.Name == el.Name.LocalName);
557  if (field != null) field.SetValue(result, value);
558  var property = properties.FirstOrDefault(p => p.Name == el.Name.LocalName);
559  if (property != null) property.SetValue(result, value);
560  }
561  return result;
562  }
563  else return elem.Value;
564

573  private static XElement ParseObject(List<Type> types, string name, object value)
574  {
575  XElement result = new XElement(name);
576 
577  if (value != null)
578  {
579  var tType = value.GetType();
580 
581  if (tType.IsEnum)
582  {
583  result.SetAttributeValue("Value", ValueType.Enum);
584  AddTypeAttr(types, tType, result);
585 
586  result.Value = Enum.GetName(tType, value) ?? "";
587  }
588  else if (value is string str)
589  {
590  result.SetAttributeValue("Value", ValueType.Text);
591  result.Value = str;
592  }
593  else if (value is int integer)
594  {
595  result.SetAttributeValue("Value", ValueType.Integer);
596  result.Value = integer.ToString();
597  }
598  else if (value is float || value is double)
599  {
600  result.SetAttributeValue("Value", ValueType.Decimal);
601  result.Value = value.ToString();
602  }
603  else if (value is bool boolean)
604  {
605  result.SetAttributeValue("Value", ValueType.Boolean);
606  result.Value = boolean.ToString();
607  }
608  else if (tType.GetInterfaces().Any(i => i.IsGenericType && i.GetGenericTypeDefinition() == typeof(IEnumerable<>)))
609  {
610  result.SetAttributeValue("Value", ValueType.Collection);
611  AddTypeAttr(types, tType, result);
612 
613  var enumerator = (IEnumerator)tType.GetMethod("GetEnumerator").Invoke(value, null);
614  while (enumerator.MoveNext())
615  {
616  var elVal = ParseObject(types, "Item", enumerator.Current);
617  result.Add(elVal);
618  }
619  }
620  else if (tType.IsClass || tType.IsValueType)
621  {
622  result.SetAttributeValue("Value", ValueType.Object);
623  AddTypeAttr(types, tType, result);
624 
625  IEnumerable<FieldInfo> fields = tType.GetFields(BindingFlags.Instance | BindingFlags.Public)
626  .Concat(tType.GetFields(BindingFlags.Instance | BindingFlags.NonPublic));
627  IEnumerable<PropertyInfo> properties = tType.GetProperties(BindingFlags.Instance | BindingFlags.Public).Where(p => p.GetSetMethod() != null)
628  .Concat(tType.GetProperties(BindingFlags.Instance | BindingFlags.NonPublic).Where(p => p.GetSetMethod() != null));
629 
630  foreach(var field in fields) result.Add(ParseObject(types, field.Name, field.GetValue(value)));
631  foreach (var property in properties) result.Add(ParseObject(types, property.Name, property.GetValue(value)));
632  }
633  else
634  {
635  result.SetAttributeValue("Value", ValueType.None);
636  result.Value = value.ToString();
637  }
638  }
639 
640  return result;
641  }
