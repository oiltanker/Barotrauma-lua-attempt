234  get

236  MissionType retVal = MissionType.None;
237  int index = 0;
238  foreach (MissionType type in Enum.GetValues(typeof(MissionType)))
239  {
240  if (type == MissionType.None || type == MissionType.All) { continue; }
241 
242  if (missionTypeTickBoxes[index].Selected)
243  {
244  retVal = (MissionType)((int)retVal | (int)type);
245  }
246 
247  index++;
248  }
249 
250  return retVal;
251  }
252  set
253  {
254  int index = 0;
255  foreach (MissionType type in Enum.GetValues(typeof(MissionType)))
256  {
257  if (type == MissionType.None || type == MissionType.All) { continue; }
258  missionTypeTickBoxes[index].Selected = ((int)type & (int)value) != 0;
259  index++;
260  }
261  }
