43 int srcRectLeft = srcRect.Left;
44 int srcRectRight = srcRect.Right;
45 int srcRectTop = srcRect.Top;
46 int srcRectBottom = srcRect.Bottom;
47 if (effects.HasFlag(SpriteEffects.FlipHorizontally))
48 {
49 var temp = srcRectRight;

60 rotation = MathHelper.ToRadians(rotation);

62 float cos = (float)Math.Cos(rotation);
63
64 var size = srcRect.Size.ToVector2() * scale;
65
66 Vector2 wAdd = new Vector2(size.X * cos, size.X * sin);
67 Vector2 hAdd = new Vector2(-size.Y * sin, size.Y * cos);
68 pos.X -= origin.X * scale.X * cos - origin.Y * scale.Y * sin;
69 pos.Y -= origin.Y * scale.Y * cos + origin.X * scale.X * sin;

78 VertexBR.Color = color;

81 VertexTR.Position = new Vector3(pos.X + wAdd.X, pos.Y + wAdd.Y, 0f);

185 public void End()
186 {
187 if (isDisposed) { return; }
188 //sort commands according to the sorting
189 //mode given in the last Begin call
190 switch (currentSortMode)
191 {
192 case SpriteSortMode.FrontToBack:
193 commandList.Sort((c1, c2) =>
194 {
195 return c1.Depth < c2.Depth ? -1
196 : c1.Depth > c2.Depth ? 1
197 : c1.Index < c2.Index ? 1
198 : c1.Index > c2.Index ? -1
199 : 0;
200 });
201 break;
202 case SpriteSortMode.BackToFront:
203 commandList.Sort((c1, c2) =>
204 {
205 return c1.Depth < c2.Depth ? 1
206 : c1.Depth > c2.Depth ? -1
207 : c1.Index < c2.Index ? 1
208 : c1.Index > c2.Index ? -1
209 : 0;
210 });
211 break;
212 }
213
214 //try to place commands of the same texture
215 //contiguously for optimal buffer generation
216 //while maintaining the same visual result
217 for (int i = 1; i < commandList.Count; i++)
218 {
219 if (commandList[i].Texture != commandList[i - 1].Texture)
220 {
221 for (int j = i - 1; j >= 0; j--)
222 {
223 if (commandList[j].Texture == commandList[i].Texture)
224 {
225 //no commands between i and j overlap with
226 //i, therefore we can safely sift i down to
227 //make a contiguous block
228 commandList.SiftElement(i, j + 1);
229 break;
230 }
231 else if (commandList[j].Overlaps(commandList[i]))
232 {
233 //an overlapping command was found, therefore
234 //attempting to sift this one down would change
235 //the visual result
236 break;
237 }
238 }
239 }
240 }
241
242 if (isDisposed) { return; }
243 //each contiguous block of commands of the same texture
244 //requires a vertex buffer to be rendered
245 CrossThread.RequestExecutionOnMainThread(() =>
246 {
247 if (isDisposed) { return; }
248 if (commandList.Count == 0) { return; }
249 int startIndex = 0;
250 for (int i = 1; i < commandList.Count; i++)
251 {
252 if (commandList[i].Texture != commandList[startIndex].Texture)
253 {
254 maxSpriteCount = Math.Max(maxSpriteCount, i - startIndex);
255 recordedBuffers.Add(new RecordedBuffer(commandList, startIndex, i - startIndex));
256 startIndex = i;
257 }
258 }
259 recordedBuffers.Add(new RecordedBuffer(commandList, startIndex, commandList.Count - startIndex));
260 maxSpriteCount = Math.Max(maxSpriteCount, commandList.Count - startIndex);
261 });
262
263 commandList.Clear();
264
265 ReadyToRender = true;
266 }

268 public void Render(Camera cam)
269 {
270 if (!ReadyToRender) { return; }
271 var gfxDevice = GameMain.Instance.GraphicsDevice;
272
273 BasicEffect ??= new BasicEffect(gfxDevice);
274 BasicEffect.Projection = Matrix.CreateOrthographicOffCenter(new Rectangle(0, 0, cam.Resolution.X, cam.Resolution.Y), -1f, 1f);
275 BasicEffect.View = cam.Transform;
276 BasicEffect.World = Matrix.Identity;
277 BasicEffect.TextureEnabled = true;
278 BasicEffect.VertexColorEnabled = true;
279 BasicEffect.Alpha = 1f;
280
281 int requiredIndexCount = maxSpriteCount * 6;
282 if (requiredIndexCount > 0 && (indexBuffer == null || indexBuffer.IndexCount < requiredIndexCount))
283 {
284 indexBuffer?.Dispose();
285 indexBuffer = new IndexBuffer(gfxDevice, IndexElementSize.SixteenBits, requiredIndexCount * 2, BufferUsage.WriteOnly);
286 ushort[] indices = new ushort[requiredIndexCount * 2];
287 for (int i = 0; i < indices.Length; i += 6)
288 {
289 indices[i + 0] = (ushort)((i / 6) * 4 + 1);
290 indices[i + 1] = (ushort)((i / 6) * 4 + 0);
291 indices[i + 2] = (ushort)((i / 6) * 4 + 2);
292 indices[i + 3] = (ushort)((i / 6) * 4 + 1);
293 indices[i + 4] = (ushort)((i / 6) * 4 + 2);
294 indices[i + 5] = (ushort)((i / 6) * 4 + 3);
295 }
296 indexBuffer.SetData(indices);
297 }
298
299 gfxDevice.Indices = indexBuffer;
300 for (int i = 0; i < recordedBuffers.Count; i++)
301 {
302 gfxDevice.SetVertexBuffer(recordedBuffers[i].VertexBuffer);
303 BasicEffect.Texture = recordedBuffers[i].Texture;
304 BasicEffect.CurrentTechnique.Passes[0].Apply();
305 gfxDevice.DrawIndexedPrimitives(PrimitiveType.TriangleList, 0, 0, recordedBuffers[i].PolyCount);
306 }
307 }
