209  private void CalculateLegLengths()
210  {
211  //calculate upper and lower leg length (atm this assumes that both legs are the same size)
212  LimbType upperLegType = LimbType.RightThigh;
213  LimbType lowerLegType = LimbType.RightLeg;
214  LimbType footType = LimbType.RightFoot;
215 
216  var waistJoint = GetJointBetweenLimbs(LimbType.Waist, upperLegType) ?? GetJointBetweenLimbs(LimbType.Torso, upperLegType);
217  Vector2 localAnchorWaist = Vector2.Zero;
218  Vector2 localAnchorKnee = Vector2.Zero;
219  if (waistJoint != null)
220  {
221  localAnchorWaist = waistJoint.LimbA.type == upperLegType ? waistJoint.LocalAnchorA : waistJoint.LocalAnchorB;
222  }
223  LimbJoint kneeJoint = GetJointBetweenLimbs(upperLegType, lowerLegType);
224  if (kneeJoint != null)
225  {
226  localAnchorKnee = kneeJoint.LimbA.type == upperLegType ? kneeJoint.LocalAnchorA : kneeJoint.LocalAnchorB;
227  }
228  upperLegLength = Vector2.Distance(localAnchorWaist, localAnchorKnee);
229 
230  LimbJoint ankleJoint = GetJointBetweenLimbs(lowerLegType, footType);
231  if (ankleJoint == null || kneeJoint == null) { return; }
232  lowerLegLength = Vector2.Distance(
233  kneeJoint.LimbA.type == lowerLegType ? kneeJoint.LocalAnchorA : kneeJoint.LocalAnchorB,
234  ankleJoint.LimbA.type == lowerLegType ? ankleJoint.LocalAnchorA : ankleJoint.LocalAnchorB);
235  lowerLegLength += Vector2.Distance(
236  ankleJoint.LimbA.type == footType ? ankleJoint.LocalAnchorA : ankleJoint.LocalAnchorB,
237  GetLimb(footType).PullJointLocalAnchorA);
