165  private static Identifier DetermineIdentifier(EventSet parent, XElement element, RandomEventsFile file)
166  {
167  Identifier retVal = element.GetAttributeIdentifier("identifier", Identifier.Empty);
168 
169  if (retVal.IsEmpty)
170  {
171  if (parent is null)
172  {
173  if (file.ContentPackage is CorePackage)
174  {
175  throw new Exception($"Error in {file.Path}: All root EventSets in a core package must have identifiers");
176  }
177  else
178  {
179  DebugConsole.AddWarning($"{file.Path}: All root EventSets should have an identifier");
180  }
181  }
182 
183  XElement currElement = element;
184  string siblingIndices = "";
185  while (currElement.Parent != null)
186  {
187  int siblingIndex = currElement.ElementsBeforeSelf().Count();
188  siblingIndices = $"-{siblingIndex}{siblingIndices}";
189  if (parent != null) { break; }
190  currElement = currElement.Parent;
191  }
192 
193  retVal =
194  ((parent != null
195  ? parent.Identifier.Value
196  : $"{file.ContentPackage.Name}-{file.Path}")
197  + siblingIndices)
198  .ToIdentifier();
199  }
200  return retVal;
201  }

203  public EventSet(ContentXElement element, RandomEventsFile file, EventSet parentSet = null)
204  : base(file, DetermineIdentifier(parentSet, element, file))
205  {
206  var eventPrefabs = new List<SubEventPrefab>();
207  var childSets = new List<EventSet>();
208  var overrideCommonness = new Dictionary<Identifier, float>();
209 
210  BiomeIdentifier = element.GetAttributeIdentifier("biome", Barotrauma.Identifier.Empty);
211  MinLevelDifficulty = element.GetAttributeFloat("minleveldifficulty", 0);
212  MaxLevelDifficulty = Math.Max(element.GetAttributeFloat("maxleveldifficulty", 100), MinLevelDifficulty);
213 
214  Additive = element.GetAttributeBool("additive", false);
215 
216  string levelTypeStr = element.GetAttributeString("leveltype", parentSet?.LevelType.ToString() ?? "LocationConnection");
217  if (!Enum.TryParse(levelTypeStr, true, out LevelType))
218  {
219  DebugConsole.ThrowError($"Error in event set \"{Identifier}\". \"{levelTypeStr}\" is not a valid level type.");
220  }
221 
222  Identifier[] locationTypeStr = element.GetAttributeIdentifierArray("locationtype", null);
223  if (locationTypeStr != null)
224  {
225  LocationTypeIdentifiers = locationTypeStr.ToImmutableArray();
226  //if (LocationType.List.Any()) { CheckLocationTypeErrors(); } //TODO: perform validation elsewhere
227  }
228 
229  MinIntensity = element.GetAttributeFloat("minintensity", 0.0f);
230  MaxIntensity = Math.Max(element.GetAttributeFloat("maxintensity", 100.0f), MinIntensity);
231 
232  ChooseRandom = element.GetAttributeBool("chooserandom", false);
233  EventCount = element.GetAttributeInt("eventcount", 1);
234  MinDistanceTraveled = element.GetAttributeFloat("mindistancetraveled", 0.0f);
235  MinMissionTime = element.GetAttributeFloat("minmissiontime", 0.0f);
236 
237  AllowAtStart = element.GetAttributeBool("allowatstart", false);
238  PerRuin = element.GetAttributeBool("perruin", false);
239  PerCave = element.GetAttributeBool("percave", false);
240  PerWreck = element.GetAttributeBool("perwreck", false);
241  DisableInHuntingGrounds = element.GetAttributeBool("disableinhuntinggrounds", false);
242  IgnoreCoolDown = element.GetAttributeBool("ignorecooldown", parentSet?.IgnoreCoolDown ?? (PerRuin || PerCave || PerWreck));
243  DelayWhenCrewAway = element.GetAttributeBool("delaywhencrewaway", !PerRuin && !PerCave && !PerWreck);
244  OncePerOutpost = element.GetAttributeBool("onceperoutpost", false);
245  TriggerEventCooldown = element.GetAttributeBool("triggereventcooldown", true);
246  IsCampaignSet = element.GetAttributeBool("campaign", LevelType == LevelData.LevelType.Outpost || (parentSet?.IsCampaignSet ?? false));
247 
248  DefaultCommonness = 1.0f;
249  foreach (var subElement in element.Elements())
250  {
251  switch (subElement.Name.ToString().ToLowerInvariant())
252  {
253  case "commonness":
254  DefaultCommonness = subElement.GetAttributeFloat("commonness", 0.0f);
255  foreach (XElement overrideElement in subElement.Elements())
256  {
257  if (overrideElement.NameAsIdentifier() == "override")
258  {
259  Identifier levelType = overrideElement.GetAttributeIdentifier("leveltype", "");
260  if (!overrideCommonness.ContainsKey(levelType))
261  {
262  overrideCommonness.Add(levelType, overrideElement.GetAttributeFloat("commonness", 0.0f));
263  }
264  }
265  }
266  break;
267  case "eventset":
268  childSets.Add(new EventSet(subElement, file, this));
269  break;
270  default:
271  //an element with just an identifier = reference to an event prefab
272  if (!subElement.HasElements && subElement.Attributes().First().Name.ToString().Equals("identifier", StringComparison.OrdinalIgnoreCase))
273  {
274  Identifier[] identifiers = subElement.GetAttributeIdentifierArray("identifier", Array.Empty<Identifier>());
275  float commonness = subElement.GetAttributeFloat("commonness", -1f);
276  float probability = subElement.GetAttributeFloat("probability", -1f);
277  eventPrefabs.Add(new SubEventPrefab(
278  identifiers,
279  commonness >= 0f ? commonness : (float?)null,
280  probability >= 0f ? probability : (float?)null));
281  }
282  else
283  {
284  var prefab = new EventPrefab(subElement, file, $"{Identifier}-{subElement.ElementsBeforeSelf().Count()}".ToIdentifier());
285  eventPrefabs.Add(new SubEventPrefab(prefab, prefab.Commonness, prefab.Probability));
286  }
287  break;
288  }
289  }
290 
291  EventPrefabs = eventPrefabs.ToImmutableArray();
292  ChildSets = childSets.ToImmutableArray();
293  OverrideCommonness = overrideCommonness.ToImmutableDictionary();
294  }

333  static void CheckEventSet(EventDebugStats stats, EventSet thisSet, Func<MonsterEvent, bool> filter = null)
334  {
335  if (thisSet.ChooseRandom)
336  {
337  var unusedEvents = thisSet.EventPrefabs.ToList();
338  if (unusedEvents.Any())
339  {
340  for (int i = 0; i < thisSet.EventCount; i++)
341  {
342  var eventPrefab = ToolBox.SelectWeightedRandom(unusedEvents, unusedEvents.Select(e => e.Commonness).ToList(), Rand.RandSync.Unsynced);
343  if (eventPrefab.EventPrefabs.Any(p => p != null))
344  {
345  AddEvents(stats, eventPrefab.EventPrefabs, filter);
346  unusedEvents.Remove(eventPrefab);
347  }
348  }
349  }
350  List<float> values = thisSet.ChildSets
351  .SelectMany(s => s.DefaultCommonness.ToEnumerable().Concat(s.OverrideCommonness.Values))
352  .ToList();
353  EventSet childSet = ToolBox.SelectWeightedRandom(thisSet.ChildSets, values, Rand.RandSync.Unsynced);
354  if (childSet != null)
355  {
356  CheckEventSet(stats, childSet, filter);
357  }
358  }
359  else
360  {
361  foreach (var eventPrefab in thisSet.EventPrefabs)
362  {
363  AddEvents(stats, eventPrefab.EventPrefabs, filter);
364  }
365  foreach (var childSet in thisSet.ChildSets)
366  {
367  CheckEventSet(stats, childSet, filter);
368  }
369  }

373  => eventPrefabs.ForEach(p => AddEvent(stats, p, filter));

375  static void AddEvent(EventDebugStats stats, EventPrefab eventPrefab, Func<MonsterEvent, bool> filter = null)
376  {
377  if (eventPrefab.EventType == typeof(MonsterEvent) && eventPrefab.TryCreateInstance(out MonsterEvent monsterEvent))
378  {
379  if (filter != null && !filter(monsterEvent)) { return; }
380  float spawnProbability = monsterEvent.Prefab.Probability;
381  if (Rand.Value() > spawnProbability) { return; }
382  int count = Rand.Range(monsterEvent.MinAmount, monsterEvent.MaxAmount + 1);
383  if (count <= 0) { return; }
384  Identifier character = monsterEvent.SpeciesName;
385  if (stats.MonsterCounts.TryGetValue(character, out int currentCount))
386  {
387  if (currentCount >= monsterEvent.MaxAmountPerLevel) { return; }
388  }
389  else
390  {
391  stats.MonsterCounts[character] = 0;
392  }
393  stats.MonsterCounts[character] += count;
394 
395  var aiElement = CharacterPrefab.FindBySpeciesName(character)?.ConfigElement?.GetChildElement("ai");
396  if (aiElement != null)
397  {
398  stats.MonsterStrength += aiElement.GetAttributeFloat("combatstrength", 0) * count;
399  }
400  }
401  }

403  static void LogEventStats(List<EventDebugStats> stats, List<string> debugLines, bool fullLog)
404  {
405  if (stats.Count == 0 || stats.All(s => s.MonsterCounts.Values.Sum() == 0))
406  {
407  debugLines.Add(" No monster spawns");
408  debugLines.Add($" ");
409  }
410  else
411  {
412  var allMonsters = new Dictionary<Identifier, int>();
413  foreach (var stat in stats)
414  {
415  foreach (var monster in stat.MonsterCounts)
416  {
417  if (!allMonsters.TryAdd(monster.Key, monster.Value))
418  {
419  allMonsters[monster.Key] += monster.Value;
420  }
421  }
422  }
423  allMonsters = allMonsters.OrderBy(m => m.Key).ToDictionary(m => m.Key, m => m.Value);
424  stats.Sort((s1, s2) => s1.MonsterCounts.Values.Sum().CompareTo(s2.MonsterCounts.Values.Sum()));
425  debugLines.Add($" Average monster count: {StringFormatter.FormatZeroDecimal((float)stats.Average(s => s.MonsterCounts.Values.Sum()))} (Min: {stats.First().MonsterCounts.Values.Sum()}, Max: {stats.Last().MonsterCounts.Values.Sum()})");
426  debugLines.Add($" {LogMonsterCounts(allMonsters, divider: stats.Count)}");
427  if (fullLog)
428  {
429  debugLines.Add($" All samples:");
430  stats.ForEach(s => debugLines.Add($" {LogMonsterCounts(s.MonsterCounts)}"));
431  }
432  stats.Sort((s1, s2) => s1.MonsterStrength.CompareTo(s2.MonsterStrength));
433  debugLines.Add($" Average monster strength: {StringFormatter.FormatZeroDecimal(stats.Average(s => s.MonsterStrength))} (Min: {StringFormatter.FormatZeroDecimal(stats.First().MonsterStrength)}, Max: {StringFormatter.FormatZeroDecimal(stats.Last().MonsterStrength)})");
434  debugLines.Add($" ");
435  }
436  }
