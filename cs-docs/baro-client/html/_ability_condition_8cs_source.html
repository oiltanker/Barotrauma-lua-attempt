70 private bool IsViableTarget(TargetType targetType, Character targetCharacter)
71 {
72 switch (targetType)
73 {
74 case TargetType.Enemy:
75 return !HumanAIController.IsFriendly(character, targetCharacter);
76 case TargetType.Ally:
77 return HumanAIController.IsFriendly(character, targetCharacter);
78 case TargetType.NotSelf:
79 return targetCharacter != character;
80 case TargetType.Alive:
81 return !targetCharacter.IsDead;
82 case TargetType.Monster:
83 return !targetCharacter.IsHuman;
84 case TargetType.InFriendlySubmarine:
85 return targetCharacter.Submarine != null && targetCharacter.Submarine.TeamID == character.TeamID;
86 default:
