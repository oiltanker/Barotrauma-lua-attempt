169 protected BiQuad(int sampleRate, double frequency, double q)
170 {
171 if (sampleRate <= 0)
172 throw new ArgumentOutOfRangeException("sampleRate");
173 if (frequency <= 0)
174 throw new ArgumentOutOfRangeException("frequency");

188 public float Process(float input)
189 {
190 double o = input * A0 + Z1;
191 Z1 = input * A1 + Z2 - B1 * o;
192 Z2 = input * A2 - B2 * o;
193 return (float)o;
194 }
195
201 public void Process(float[] input)
202 {
203 for (int i = 0; i < input.Length; i++)
204 {
205 input[i] = Process(input[i]);
206 }

218 public class LowpassFilter : BiQuad
219 {
225 public LowpassFilter(int sampleRate, double frequency)
226 : base(sampleRate, frequency)
227 {
228 }
229
233 protected override void CalculateBiQuadCoefficients()
234 {
235 double k = Math.Tan(Math.PI * Frequency / SampleRate);
236 var norm = 1 / (1 + k / Q + k * k);
237 A0 = k * k * norm;
238 A1 = 2 * A0;
239 A2 = A0;
240 B1 = 2 * (k * k - 1) * norm;
241 B2 = (1 - k / Q + k * k) * norm;
242 }
243 }

338 public class LowShelfFilter : BiQuad
339 {
346 public LowShelfFilter(int sampleRate, double frequency, double gainDB)
347 : base(sampleRate, frequency)
348 {
349 GainDB = gainDB;
350 }
351
355 protected override void CalculateBiQuadCoefficients()
356 {
357 const double sqrt2 = 1.4142135623730951;
358 double k = Math.Tan(Math.PI * Frequency / SampleRate);
359 double v = Math.Pow(10, Math.Abs(GainDB) / 20.0);
360 double norm;
361 if (GainDB >= 0)
362 { // boost
363 norm = 1 / (1 + sqrt2 * k + k * k);
364 A0 = (1 + Math.Sqrt(2 * v) * k + v * k * k) * norm;
365 A1 = 2 * (v * k * k - 1) * norm;
366 A2 = (1 - Math.Sqrt(2 * v) * k + v * k * k) * norm;
367 B1 = 2 * (k * k - 1) * norm;
368 B2 = (1 - sqrt2 * k + k * k) * norm;
369 }
370 else
371 { // cut
372 norm = 1 / (1 + Math.Sqrt(2 * v) * k + v * k * k);
373 A0 = (1 + sqrt2 * k + k * k) * norm;
374 A1 = 2 * (k * k - 1) * norm;
375 A2 = (1 - sqrt2 * k + k * k) * norm;
376 B1 = 2 * (v * k * k - 1) * norm;
377 B2 = (1 - Math.Sqrt(2 * v) * k + v * k * k) * norm;
378 }
379 }

455 public PeakFilter(int sampleRate, double frequency, double bandWidth, double peakGainDB)
456 : base(sampleRate, frequency, bandWidth)
457 {
458 GainDB = peakGainDB;
459 }
460
464 protected override void CalculateBiQuadCoefficients()
465 {
466 double norm;
467 double v = Math.Pow(10, Math.Abs(GainDB) / 20.0);
468 double k = Math.Tan(Math.PI * Frequency / SampleRate);
469 double q = Q;
470
471 if (GainDB >= 0) //boost
472 {
473 norm = 1 / (1 + 1 / q * k + k * k);
474 A0 = (1 + v / q * k + k * k) * norm;
475 A1 = 2 * (k * k - 1) * norm;
476 A2 = (1 - v / q * k + k * k) * norm;
477 B1 = A1;
478 B2 = (1 - 1 / q * k + k * k) * norm;
479 }
480 else //cut
481 {
482 norm = 1 / (1 + v / q * k + k * k);
483 A0 = (1 + 1 / q * k + k * k) * norm;
484 A1 = 2 * (k * k - 1) * norm;
485 A2 = (1 - 1 / q * k + k * k) * norm;
486 B1 = A1;
487 B2 = (1 - v / q * k + k * k) * norm;
488 }
