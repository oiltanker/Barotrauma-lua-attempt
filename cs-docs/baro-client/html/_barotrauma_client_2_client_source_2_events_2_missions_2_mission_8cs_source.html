64 float normalizedValue = MathUtils.InverseLerp(-100.0f, 100.0f, reputationReward.Value);

70 reputationRewardTexts.Add(rewardText.Value);
71 }
72 return RichString.Rich(TextManager.AddPunctuation(':', TextManager.Get("reputation"), LocalizedString.Join(", ", reputationRewardTexts)));
