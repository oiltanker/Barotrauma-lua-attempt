83 for (int i = 0; i < timeSinceReceived.Length; i++)
84 {
85 if (timeSinceReceived[i] > timeFrame) { state = false; }
86 timeSinceReceived[i] += deltaTime;

91 {
92 //deactivate the component if state is false and there's no false output (will be woken up by non-zero signals in ReceiveSignal)
93 if (!state) { IsActive = false; }

95 }

107 signalSender[0] = signal.sender;
108 IsActive = true;

110 case "signal_in2":
111 if (signal.value == "0") { return; }
112 timeSinceReceived[1] = 0.0f;
113 signalSender[1] = signal.sender;
114 IsActive = true;
115 break;
116 case "set_output":
