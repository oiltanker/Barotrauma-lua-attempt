92 public override bool Use(float deltaTime, Character character = null)
93 {
94 if (character == null || reloadTimer > 0.0f) { return false; }
95#if CLIENT
96 if (!Item.RequireAimToUse && character.IsPlayer && (GUI.MouseOn != null || character.Inventory.visualSlots.Any(s => s.MouseOn()) || Inventory.DraggingItems.Any())) { return false; }
97#endif
98 if (Item.RequireAimToUse && !character.IsKeyDown(InputType.Aim) || hitting) { return false; }
99
100 //don't allow hitting if the character is already hitting with another weapon
101 foreach (Item heldItem in character.HeldItems)
102 {
103 var otherWeapon = heldItem.GetComponent<MeleeWeapon>();
104 if (otherWeapon == null) { continue; }
105 if (otherWeapon.hitting) { return false; }
106 }
107
108 SetUser(character);
109
110 if (Item.RequireAimToUse && hitPos < MathHelper.PiOver4) { return false; }
111
112 ActivateNearbySleepingCharacters();
113 reloadTimer = reload;
114 reloadTimer /= (1f + character.GetStatValue(StatTypes.MeleeAttackSpeed));
115 reloadTimer /= (1f + item.GetQualityModifier(Quality.StatType.StrikingSpeedMultiplier));
116
117 item.body.FarseerBody.CollisionCategories = Physics.CollisionProjectile;
118 item.body.FarseerBody.CollidesWith = Physics.CollisionCharacter | Physics.CollisionWall | Physics.CollisionItemBlocking;
119 item.body.FarseerBody.OnCollision += OnCollision;
120 item.body.FarseerBody.IsBullet = true;
121 item.body.PhysEnabled = true;
122
123 if (Swing && !character.AnimController.InWater)
124 {
125 foreach (Limb l in character.AnimController.Limbs)
126 {
127 if (l.IsSevered) { continue; }
128 Vector2 force = new Vector2(character.AnimController.Dir * SwingForce.X, SwingForce.Y) * l.Mass;
129 switch (l.type)
130 {
131 case LimbType.Torso:
132 force *= 2;
133 break;
134 case LimbType.Legs:
135 case LimbType.LeftFoot:
136 case LimbType.LeftThigh:
137 case LimbType.LeftLeg:
138 case LimbType.RightFoot:
139 case LimbType.RightThigh:
140 case LimbType.RightLeg:
141 force = Vector2.Zero;
142 break;
143 }
144 l.body.ApplyLinearImpulse(force);
145 }
146 }
147
148 hitting = true;
149 hitTargets.Clear();
150
151 IsActive = true;
152
153 if (item.AiTarget != null)
154 {
155 item.AiTarget.SoundRange = item.AiTarget.MaxSoundRange;
156 item.AiTarget.SightRange = item.AiTarget.MaxSightRange;
157 }
158 return false;
159 }

178 public override void Update(float deltaTime, Camera cam)
179 {
180 if (!item.body.Enabled)
181 {
182 impactQueue.Clear();
183 return;
184 }
185 if (picker == null && !picker.HeldItems.Contains(item))
186 {
187 impactQueue.Clear();
188 IsActive = false;
189 }
190 while (impactQueue.Count > 0)
191 {
192 var impact = impactQueue.Dequeue();
193 HandleImpact(impact.Body);
194 }
195 //in case handling the impact does something to the picker
196 if (picker == null) { return; }
197 reloadTimer -= deltaTime;
198 if (reloadTimer < 0)
199 {
200 reloadTimer = 0;
201 }
202 if (!picker.IsKeyDown(InputType.Aim) && !hitting)
203 {
204 hitPos = 0.0f;
205 }
206 ApplyStatusEffects(ActionType.OnActive, deltaTime, picker);
207 if (item.body.Dir != picker.AnimController.Dir)
208 {
209 item.FlipX(relativeToSub: false);
210 }
211 AnimController ac = picker.AnimController;
212 if (!hitting)
213 {
214 bool aim = item.RequireAimToUse && picker.AllowInput && picker.IsKeyDown(InputType.Aim) && reloadTimer <= 0 && picker.CanAim;
215 if (aim)
216 {
217 hitPos = MathUtils.WrapAnglePi(Math.Min(hitPos + deltaTime * 3f, MathHelper.PiOver4));
218 ac.HoldItem(deltaTime, item, handlePos, aimPos, Vector2.Zero, aim: false, hitPos, holdAngle + hitPos, aimMelee: true);
219 }
220 else
221 {
222 hitPos = 0;
223 ac.HoldItem(deltaTime, item, handlePos, holdPos, Vector2.Zero, aim: false, holdAngle);
224 }
225 }
226 else
227 {
228 // TODO: We might want to make this configurable
229 hitPos -= deltaTime * 15f;
230 if (Swing)
231 {
232 ac.HoldItem(deltaTime, item, handlePos, SwingPos, Vector2.Zero, aim: false, hitPos, holdAngle);
233 }
234 else
235 {
236 ac.HoldItem(deltaTime, item, handlePos, holdPos, Vector2.Zero, aim: false, holdAngle);
237 }
238 if (hitPos < -MathHelper.Pi)
239 {
240 RestoreCollision();
241 hitting = false;
242 hitTargets.Clear();
243 hitPos = 0;
244 }
245 }
246 }

251 private void ActivateNearbySleepingCharacters()
252 {
253 foreach (Character c in Character.CharacterList)
254 {
255 if (!c.Enabled || !c.AnimController.BodyInRest) { continue; }
256 //do a broad check first
257 if (Math.Abs(c.WorldPosition.X - item.WorldPosition.X) > 1000.0f) { continue; }
258 if (Math.Abs(c.WorldPosition.Y - item.WorldPosition.Y) > 1000.0f) { continue; }
259
260 foreach (Limb limb in c.AnimController.Limbs)
261 {
262 float hitRange = 2.0f;
263 if (Vector2.DistanceSquared(limb.SimPosition, item.SimPosition) < hitRange * hitRange)
264 {
265 c.AnimController.BodyInRest = false;
266 break;
267 }
268 }
269 }

291 private bool OnCollision(Fixture f1, Fixture f2, Contact contact)
292 {
293 if (User == null || User.Removed)
294 {
295 impactQueue.Enqueue(f2);
296 return true;
297 }
298
299 contact.GetWorldManifold(out Vector2 normal, out var points);
300
301 //ignore collision if there's a wall between the user and the contact point to prevent hitting through walls
302 if (Submarine.PickBody(User.AnimController.AimSourceSimPos,
303 points[0],
304 collisionCategory: Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionItemBlocking,
305 allowInsideFixture: true,
306 customPredicate: (Fixture fixture) => { return fixture.CollidesWith.HasFlag(Physics.CollisionItem) && fixture.Body != f2.Body; }) != null)
307 {
308 return false;
309 }
310
311 if (f2.Body.UserData is Limb targetLimb)
312 {
313 if (targetLimb.IsSevered || targetLimb.character == null || targetLimb.character == User) { return false; }
314 var targetCharacter = targetLimb.character;
315 if (targetCharacter == picker) { return false; }
316 if (AllowHitMultiple)
317 {
318 if (hitTargets.Contains(targetCharacter)) { return false; }
319 }
320 else
321 {
322 if (hitTargets.Any(t => t is Character)) { return false; }
323 }
324 hitTargets.Add(targetCharacter);
325 }
326 else if (f2.Body.UserData is Character targetCharacter)
327 {
328 if (targetCharacter == picker || targetCharacter == User) { return false; }
329 targetLimb = targetCharacter.AnimController.GetLimb(LimbType.Torso); //Otherwise armor can be bypassed in strange ways
330 if (AllowHitMultiple)
331 {
332 if (hitTargets.Contains(targetCharacter)) { return false; }
333 }
334 else
335 {
336 if (hitTargets.Any(t => t is Character)) { return false; }
337 }
338 hitTargets.Add(targetCharacter);
339 }
340 else if (f2.Body.UserData is Structure targetStructure)
341 {
342 if (AllowHitMultiple)
343 {
344 if (hitTargets.Contains(targetStructure)) { return true; }
345 }
346 else
347 {
348 if (hitTargets.Any(t => t is Structure)) { return true; }
349 }
350 hitTargets.Add(targetStructure);
351 }
352 else if (f2.Body.UserData is Item targetItem)
353 {
354 if (AllowHitMultiple)
355 {
356 if (hitTargets.Contains(targetItem)) { return true; }
357 }
358 else
359 {
360 if (hitTargets.Any(t => t is Item)) { return true; }
361 }
362 hitTargets.Add(targetItem);
363 }
364 else if (f2.Body.UserData is Holdable holdable && holdable.CanPush)
365 {
366 hitTargets.Add(holdable.Item);
367 }
368 else
369 {
370 return false;
371 }
372
373 impactQueue.Enqueue(f2);
374
375 return true;
376 }

378 private void HandleImpact(Body target)
379 {
380 if (User == null || User.Removed || target == null)
381 {
382 RestoreCollision();
383 hitting = false;
384 User = null;
385 return;
386 }
387
388 Limb targetLimb = target.UserData as Limb;
389 Character targetCharacter = targetLimb?.character ?? target.UserData as Character;
390 GameMain.LuaCs.Hook.Call("meleeWeapon.handleImpact", this, target);
391 if (Attack != null)
392 {
393 Attack.SetUser(User);
394 Attack.DamageMultiplier = 1 + User.GetStatValue(StatTypes.MeleeAttackMultiplier);
395 Attack.DamageMultiplier *= 1.0f + item.GetQualityModifier(Quality.StatType.StrikingPowerMultiplier);
396
397 if (targetLimb != null)
398 {
399 if (targetLimb.character.Removed) { return; }
400 targetLimb.character.LastDamageSource = item;
401 Attack.DoDamageToLimb(User, targetLimb, item.WorldPosition, 1.0f);
402 }
403 else if (targetCharacter != null)
404 {
405 if (targetCharacter.Removed) { return; }
406 targetCharacter.LastDamageSource = item;
407 Attack.DoDamage(User, targetCharacter, item.WorldPosition, 1.0f);
408 }
409 else if (target.UserData is Structure targetStructure)
410 {
411 if (targetStructure.Removed) { return; }
412 Attack.DoDamage(User, targetStructure, item.WorldPosition, 1.0f);
413 }
414 else if (target.UserData is Item targetItem && targetItem.Prefab.DamagedByMeleeWeapons && targetItem.Condition > 0)
415 {
416 if (targetItem.Removed) { return; }
417 Attack.DoDamage(User, targetItem, item.WorldPosition, 1.0f);
418 }
419 else if (target.UserData is Holdable holdable && holdable.CanPush)
420 {
421 if (holdable.Item.Removed) { return; }
422 Attack.DoDamage(User, holdable.Item, item.WorldPosition, 1.0f);
423 RestoreCollision();
424 hitting = false;
425 User = null;
426 }
427 else
428 {
429 return;
430 }
431 }
432
433 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
434
435 bool success = Rand.Range(0.0f, 0.5f) < DegreeOfSuccess(User);
436
437#if SERVER
438 if (GameMain.Server != null && targetCharacter != null) //TODO: Log structure hits
439 {
440 GameMain.Server.CreateEntityEvent(item, new Item.ApplyStatusEffectEventData(
441 success ? ActionType.OnUse : ActionType.OnFailure,
442 targetItemComponent: null,
443 targetCharacter, targetLimb));
444
445 string logStr = picker?.LogName + " used " + item.Name;
446 if (item.ContainedItems != null && item.ContainedItems.Any())
447 {
448 logStr += " (" + string.Join(", ", item.ContainedItems.Select(i => i?.Name)) + ")";
449 }
450 logStr += " on " + targetCharacter.LogName + ".";
451 Networking.GameServer.Log(logStr, Networking.ServerLog.MessageType.Attack);
452 }
453#endif
454
455 if (targetCharacter != null) //TODO: Allow OnUse to happen on structures too maybe??
456 {
457 ApplyStatusEffects(success ? ActionType.OnUse : ActionType.OnFailure, 1.0f, targetCharacter, targetLimb, user: User);
458 }
459
460 if (DeleteOnUse)
461 {
462 Entity.Spawner.AddItemToRemoveQueue(item);
463 }
464 }
465 }
