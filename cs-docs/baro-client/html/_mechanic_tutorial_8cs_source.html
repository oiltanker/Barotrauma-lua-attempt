301 public override IEnumerable<CoroutineStatus> UpdateState()
302 {
303 while (GameMain.Instance.LoadingScreenOpen) yield return null;
304
305 // Room 1
306 SoundPlayer.PlayDamageSound("StructureBlunt", 10, Character.Controlled.WorldPosition);
307 while (shakeTimer > 0.0f) // Wake up, shake
308 {
309 shakeTimer -= 0.1f;
310 GameMain.GameScreen.Cam.Shake = shakeAmount;
311 yield return new WaitForSeconds(0.1f, false);
312 }
313 yield return new WaitForSeconds(2.5f, false);
314
315 mechanic_fabricator.RemoveFabricationRecipes(allowedIdentifiers:
316 new[] { "extinguisher", "wrench", "weldingtool", "weldingfuel", "divingmask", "railgunshell", "nuclearshell", "uex", "harpoongun" }.ToIdentifiers());
317 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Mechanic.Radio.WakeUp"), ChatMessageType.Radio, null);
318
319 yield return new WaitForSeconds(2.5f, false);
320 TriggerTutorialSegment(0, GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Up), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Left), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Down), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Right), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Select), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Select)); // Open door objective
321 yield return new WaitForSeconds(0.0f, false);
322 SetDoorAccess(mechanic_firstDoor, mechanic_firstDoorLight, true);
323 SetHighlight(mechanic_firstDoor.Item, true);
324 do { yield return null; } while (!mechanic_firstDoor.IsOpen);
325 SetHighlight(mechanic_firstDoor.Item, false);
326 yield return new WaitForSeconds(1.5f, false);
327 RemoveCompletedObjective(0);
328
329 // Room 2
330 yield return new WaitForSeconds(0.0f, false);
331 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Mechanic.Radio.Equipment"), ChatMessageType.Radio, null);
332 do { yield return null; } while (!mechanic_equipmentObjectiveSensor.MotionDetected);
333 TriggerTutorialSegment(1, GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Select), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Deselect), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.ToggleInventory)); // Equipment & inventory objective
334 SetHighlight(mechanic_equipmentCabinet.Item, true);
335 bool firstSlotRemoved = false;
336 bool secondSlotRemoved = false;
337 bool thirdSlotRemoved = false;
338 do
339 {
340 if (IsSelectedItem(mechanic_equipmentCabinet.Item))
341 {
342 if (!firstSlotRemoved)
343 {
344 HighlightInventorySlot(mechanic_equipmentCabinet.Inventory, 0, highlightColor, .5f, .5f, 0f);
345 if (mechanic_equipmentCabinet.Inventory.GetItemAt(0) == null) { firstSlotRemoved = true; }
346 }
347
348 if (!secondSlotRemoved)
349 {
350 HighlightInventorySlot(mechanic_equipmentCabinet.Inventory, 1, highlightColor, .5f, .5f, 0f);
351 if (mechanic_equipmentCabinet.Inventory.GetItemAt(1) == null) { secondSlotRemoved = true; }
352 }
353
354 if (!thirdSlotRemoved)
355 {
356 HighlightInventorySlot(mechanic_equipmentCabinet.Inventory, 2, highlightColor, .5f, .5f, 0f);
357 if (mechanic_equipmentCabinet.Inventory.GetItemAt(2) == null) { thirdSlotRemoved = true; }
358 }
359
360 for (int i = 0; i < mechanic.Inventory.Capacity; i++)
361 {
362 if (mechanic.Inventory.GetItemAt(i) == null) { HighlightInventorySlot(mechanic.Inventory, i, highlightColor, .5f, .5f, 0f); }
363 }
364 }
365
366 yield return null;
367 } while (mechanic.Inventory.FindItemByIdentifier("divingmask".ToIdentifier()) == null || mechanic.Inventory.FindItemByIdentifier("weldingtool".ToIdentifier()) == null || mechanic.Inventory.FindItemByIdentifier("wrench".ToIdentifier()) == null); // Wait until looted
368 SetHighlight(mechanic_equipmentCabinet.Item, false);
369 yield return new WaitForSeconds(1.5f, false);
370 RemoveCompletedObjective(1);
371 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Mechanic.Radio.Breach"), ChatMessageType.Radio, null);
372
373 // Room 3
374 do { yield return null; } while (!mechanic_weldingObjectiveSensor.MotionDetected);
375 TriggerTutorialSegment(2, GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Aim), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Shoot), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.ToggleInventory)); // Welding objective
376 do
377 {
378 if (!mechanic.HasEquippedItem("divingmask".ToIdentifier()))
379 {
380 HighlightInventorySlot(mechanic.Inventory, "divingmask".ToIdentifier(), highlightColor, .5f, .5f, 0f);
381 }
382
383 if (!mechanic.HasEquippedItem("weldingtool".ToIdentifier()))
384 {
385 HighlightInventorySlot(mechanic.Inventory, "weldingtool".ToIdentifier(), highlightColor, .5f, .5f, 0f);
386 }
387 yield return null;
388 } while (!mechanic.HasEquippedItem("divingmask".ToIdentifier()) || !mechanic.HasEquippedItem("weldingtool".ToIdentifier())); // Wait until equipped
389 SetDoorAccess(mechanic_secondDoor, mechanic_secondDoorLight, true);
390 mechanic.AddActiveObjectiveEntity(mechanic_brokenWall_1, mechanic_weldIcon, mechanic_repairIconColor);
391 do { yield return null; } while (WallHasDamagedSections(mechanic_brokenWall_1)); // Highlight until repaired
392 mechanic.RemoveActiveObjectiveEntity(mechanic_brokenWall_1);
393 RemoveCompletedObjective(2);
394 yield return new WaitForSeconds(1f, false);
395 TriggerTutorialSegment(3, GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Select)); // Pump objective
396 SetHighlight(mechanic_workingPump.Item, true);
397 do
398 {
399 yield return null;
400 if (IsSelectedItem(mechanic_workingPump.Item))
401 {
402 if (mechanic_workingPump.PowerButton.FlashTimer <= 0)
403 {
404 mechanic_workingPump.PowerButton.Flash(uiHighlightColor, 1.5f, true);
405 }
406 }
407 } while (mechanic_workingPump.FlowPercentage >= 0 || !mechanic_workingPump.IsActive); // Highlight until draining
408 SetHighlight(mechanic_workingPump.Item, false);
409 do { yield return null; } while (mechanic_brokenhull_1.WaterPercentage > waterVolumeBeforeOpening); // Unlock door once drained
410 RemoveCompletedObjective(3);
411 SetDoorAccess(mechanic_thirdDoor, mechanic_thirdDoorLight, true);
412 //TriggerTutorialSegment(11, GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Select], GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Up], GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Down], GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Select]); // Ladder objective
413 //do { yield return null; } while (!mechanic_ladderSensor.MotionDetected);
414 //RemoveCompletedObjective(segments[11]);
415 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Mechanic.Radio.News"), ChatMessageType.Radio, null);
416 yield return new WaitForSeconds(1f, false);
417 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Mechanic.Radio.Fire"), ChatMessageType.Radio, null);
418
419 // Room 4
420 do { yield return null; } while (!mechanic_thirdDoor.IsOpen);
421 yield return new WaitForSeconds(1f, false);
422 mechanic_fire = new DummyFireSource(new Vector2(20f, 2f), Item.ItemList.Find(i => i.HasTag("mechanic_fire")).WorldPosition);
423 //do { yield return null; } while (!mechanic_craftingObjectiveSensor.MotionDetected);
424 TriggerTutorialSegment(4); // Deconstruct
425
426 SetHighlight(mechanic_craftingCabinet.Item, true);
427
428 bool gotOxygenTank = false;
429 bool gotSodium = false;
430 do
431 {
432 if (mechanic.SelectedConstruction == mechanic_craftingCabinet.Item)
433 {
434 for (int i = 0; i < mechanic.Inventory.Capacity; i++)
435 {
436 if (mechanic.Inventory.GetItemAt(i) == null) { HighlightInventorySlot(mechanic.Inventory, i, highlightColor, .5f, .5f, 0f); }
437 }
438
439 if (mechanic.Inventory.FindItemByIdentifier("oxygentank".ToIdentifier()) == null && mechanic.Inventory.FindItemByIdentifier("aluminium".ToIdentifier()) == null)
440 {
441 for (int i = 0; i < mechanic_craftingCabinet.Capacity; i++)
442 {
443 Item item = mechanic_craftingCabinet.Inventory.GetItemAt(i);
444 if (item != null && item.Prefab.Identifier == "oxygentank")
445 {
446 HighlightInventorySlot(mechanic_craftingCabinet.Inventory, i, highlightColor, .5f, .5f, 0f);
447 }
448 }
449 }
450
451 if (mechanic.Inventory.FindItemByIdentifier("sodium".ToIdentifier()) == null)
452 {
453 for (int i = 0; i < mechanic_craftingCabinet.Inventory.Capacity; i++)
454 {
455 Item item = mechanic_craftingCabinet.Inventory.GetItemAt(i);
456 if (item != null && item.Prefab.Identifier == "sodium")
457 {
458 HighlightInventorySlot(mechanic_craftingCabinet.Inventory, i, highlightColor, .5f, .5f, 0f);
459 }
460 }
461 }
462 }
463
464 if (!gotOxygenTank && (mechanic.Inventory.FindItemByIdentifier("oxygentank".ToIdentifier()) != null ||
465 mechanic_deconstructor.InputContainer.Inventory.FindItemByIdentifier("oxygentank".ToIdentifier()) != null))
466 {
467 gotOxygenTank = true;
468 }
469 if (!gotSodium && mechanic.Inventory.FindItemByIdentifier("sodium".ToIdentifier()) != null)
470 {
471 gotSodium = true;
472 }
473 yield return null;
