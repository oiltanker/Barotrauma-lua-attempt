533 private void CreateLevelObjectEditor(LevelObjectPrefab levelObjectPrefab)
534 {
535 editorContainer.ClearChildren();
536
537 var editor = new SerializableEntityEditor(editorContainer.Content.RectTransform, levelObjectPrefab, false, true, elementHeight: 20, titleFont: GUIStyle.LargeFont);
538
539 if (selectedParams != null)
540 {
541 List<Identifier> availableIdentifiers = new List<Identifier>();
542 if (selectedParams != null) { availableIdentifiers.Add(selectedParams.Identifier); }
543 foreach (var caveParam in CaveGenerationParams.CaveParams)
544 {
545 if (selectedParams != null && caveParam.GetCommonness(selectedParams, abyss: false) <= 0.0f) { continue; }
546 availableIdentifiers.Add(caveParam.Identifier);
547 }
548 availableIdentifiers.Reverse();
549
550 foreach (Identifier paramsId in availableIdentifiers)
551 {
552 var commonnessContainer = new GUILayoutGroup(new RectTransform(new Point(editor.Rect.Width, 70)) { IsFixedSize = true },
553 isHorizontal: false, childAnchor: Anchor.TopCenter)
554 {
555 AbsoluteSpacing = 5,
556 Stretch = true
557 };
558 new GUITextBlock(new RectTransform(new Vector2(1.0f, 0.4f), commonnessContainer.RectTransform),
559 TextManager.GetWithVariable("leveleditor.levelobjcommonness", "[leveltype]", paramsId.Value), textAlignment: Alignment.Center);
560 new GUINumberInput(new RectTransform(new Vector2(0.5f, 0.4f), commonnessContainer.RectTransform), GUINumberInput.NumberType.Float)
561 {
562 MinValueFloat = 0,
563 MaxValueFloat = 100,
564 FloatValue = selectedParams.Identifier == paramsId ? levelObjectPrefab.GetCommonness(selectedParams) : levelObjectPrefab.GetCommonness(CaveGenerationParams.CaveParams.Find(p => p.Identifier == paramsId)),
565 OnValueChanged = (numberInput) =>
566 {
567 levelObjectPrefab.OverrideCommonness[paramsId] = numberInput.FloatValue;
568 }
569 };
570 new GUIFrame(new RectTransform(new Vector2(1.0f, 0.2f), commonnessContainer.RectTransform), style: null);
571 editor.AddCustomContent(commonnessContainer, 1);
572 }
573 }
574
575 Sprite sprite = levelObjectPrefab.Sprites.FirstOrDefault() ?? levelObjectPrefab.DeformableSprite?.Sprite;
576 if (sprite != null)
577 {
578 editor.AddCustomContent(new GUIButton(new RectTransform(new Point(editor.Rect.Width / 2, (int)(25 * GUI.Scale))) { IsFixedSize = true },
579 TextManager.Get("leveleditor.editsprite"))
580 {
581 OnClicked = (btn, userdata) =>
582 {
583 editingSprite = sprite;
584 GameMain.SpriteEditorScreen.SelectSprite(editingSprite);
585 return true;
586 }
587 }, 1);
588 }
589
590 if (levelObjectPrefab.DeformableSprite != null)
591 {
592 var deformEditor = levelObjectPrefab.DeformableSprite.CreateEditor(editor, levelObjectPrefab.SpriteDeformations, levelObjectPrefab.Name);
593 deformEditor.GetChild<GUIDropDown>().OnSelected += (selected, userdata) =>
