186 private void AddLine(LogMessage line)
187 {
188 float prevSize = listBox.BarSize;
189
190 GUIFrame textContainer = null;
191
192 Anchor anchor = Anchor.TopLeft;
193 Pivot pivot = Pivot.TopLeft;
194 RichString richString = line.Text as RichString;
195 if (richString != null && richString.RichTextData.HasValue)
196 {
197 foreach (var data in richString.RichTextData.Value)
198 {
199 if (!UInt64.TryParse(data.Metadata, out ulong id)) { return; }
200 Client client = GameMain.Client.ConnectedClients.Find(c => c.SteamID == id)
201 ?? GameMain.Client.ConnectedClients.Find(c => c.ID == id)
202 ?? GameMain.Client.PreviouslyConnectedClients.FirstOrDefault(c => c.SteamID == id)
203 ?? GameMain.Client.PreviouslyConnectedClients.FirstOrDefault(c => c.ID == id);
204 if (client != null && client.Karma < 40.0f)
205 {
206 textContainer = new GUIFrame(new RectTransform(new Vector2(1.0f, 0.0f), listBox.Content.RectTransform),
207 style: null, color: new Color(0xff111155))
208 {
209 CanBeFocused = false
210 };
211 anchor = Anchor.CenterLeft;
212 pivot = Pivot.CenterLeft;
213 break;
214 }
215 }
216 }
217
218 var textBlock = new GUITextBlock(new RectTransform(new Vector2(1.0f, 0.0f), (textContainer ?? listBox.Content).RectTransform, anchor, pivot),
219 line.Text, wrap: true, font: GUIStyle.SmallFont)
220 {
221 TextColor = messageColor[line.Type],
222 Visible = !msgTypeHidden[(int)line.Type],
223 CanBeFocused = false,
224 UserData = line
225 };
226
227 if (textContainer != null)
228 {
229 textContainer.RectTransform.NonScaledSize = new Point(textContainer.RectTransform.NonScaledSize.X, textBlock.RectTransform.NonScaledSize.Y + 5);
230 textBlock.SetTextPos();
231 textBlock.RectTransform.Resize(textContainer.RectTransform.NonScaledSize);
232 }
233
234 if (reverseOrder)
235 {
236 textBlock.RectTransform.SetAsFirstChild();
237 }
238
239 if (richString != null && richString.RichTextData.HasValue)
240 {
241 foreach (var data in richString.RichTextData.Value)
242 {
243 textBlock.ClickableAreas.Add(new GUITextBlock.ClickableArea()

245 Data = data,

249 }

252 if ((prevSize == 1.0f && listBox.BarScroll == 0.0f) || (prevSize < 1.0f && listBox.BarScroll == 1.0f)) listBox.BarScroll = 1.0f;

255 private bool FilterMessages()
256 {
257 string filter = msgFilter == null ? "" : msgFilter.ToLower();
258
259 foreach (GUIComponent child in listBox.Content.Children)
260 {
261 var textBlock = child as GUITextBlock;
262 if (textBlock == null) continue;
263
264 child.Visible = true;
265
266 if (msgTypeHidden[(int)((LogMessage)child.UserData).Type])
267 {
268 child.Visible = false;
269 continue;
270 }
271
272 textBlock.Visible = string.IsNullOrEmpty(filter) || textBlock.Text.ToLower().Contains(filter);
273 }
274 listBox.UpdateScrollBarSize();
275 listBox.BarScroll = 0.0f;
276
277 return true;
278 }
