99  {
100 #if CLIENT
101  SoundPlayer.PlaySound("icebreak", WorldPosition);
102 #endif
103  //generate initial triangles (one triangle from each edge to the center of the cell)
104  List<List<Vector2>> triangles = new List<List<Vector2>>();
105  foreach (var cell in Cells)
106  {
107  foreach (GraphEdge edge in cell.Edges)
108  {
109  List<Vector2> triangleVerts = new List<Vector2>
110  {
111  edge.Point1 + cell.Translation,
112  edge.Point2 + cell.Translation,
113  cell.Center
114  };
115  triangles.Add(triangleVerts);
116  }
117  }
118 
119  //split triangles that have edges more than 1000 units long
120  Pair<int, int> longestEdge = new Pair<int, int>(-1, -1);
121  float longestEdgeLength = 0.0f;
122  do
123  {
124  longestEdge.First = -1;
125  longestEdge.Second = -1;
126  longestEdgeLength = 0.0f;
127  for (int i = 0; i < triangles.Count; i++)
128  {
129  for (int edge = 0; edge < 3; edge++)
130  {
131  float edgeLength = Vector2.Distance(triangles[i][edge], triangles[i][(edge + 1) % 3]);
132  if (edgeLength > longestEdgeLength)
133  {
134  longestEdge.First = i;
135  longestEdge.Second = edge;
136  longestEdgeLength = edgeLength;
137  }
138  }
139  }
140  if (longestEdgeLength < 1000.0f)
141  {
142  break;
143  }
144  Vector2 p0 = triangles[longestEdge.First][longestEdge.Second];
145  Vector2 p1 = triangles[longestEdge.First][(longestEdge.Second + 1) % 3];
146  Vector2 p2 = triangles[longestEdge.First][(longestEdge.Second + 2) % 3];
147  triangles[longestEdge.First] = new List<Vector2> { p0, (p0 + p1) / 2, p2 };
148  triangles.Add(new List<Vector2> { (p0 + p1) / 2, p1, p2 });
149 
150 
151  } while (triangles.Count < 32);
152 
153  //generate fragments
154  foreach (var triangle in triangles)
155  {
156  Vector2 triangleCenter = (triangle[0] + triangle[1]+ triangle[2]) / 3;
157  triangle[0] -= triangleCenter;

159  triangle[2] -= triangleCenter;

161
