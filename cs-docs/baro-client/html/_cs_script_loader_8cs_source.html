48 private enum SourceCategory { Shared, Server, Client };
49 private Regex rMaskPathValid = new Regex(@"^(((?!csharp)[^/])+/)+csharp(/(shared|client|server))?(/[^/]+)+\.cs$", RegexOptions.IgnoreCase | RegexOptions.Compiled);
50 private Regex rMaskPathCategory1 = new Regex(@"/(shared|client|server)(/[^/]+)+\.cs$", RegexOptions.IgnoreCase | RegexOptions.Compiled);
51 private Regex rMaskPathCategory2 = new Regex(@"^/(shared|client|server)", RegexOptions.IgnoreCase | RegexOptions.Compiled);
52 private void RunFolder(string folder)
53 {
54 var offendingSources = new List<string>();
55 foreach (var str in DirSearch(folder))
56 {
57 var s = str.Replace("\\", "/");
58
59 if (s.EndsWith(".cs") && LuaCsFile.IsPathAllowedCsException(s))
60 {
61 if (rMaskPathValid.IsMatch(s)) // valid path
62 {
63 var sourceCategory = SourceCategory.Shared;
64 {
65 var match = rMaskPathCategory1.Match(s);
66 if (match.Success) match = rMaskPathCategory2.Match(match.Value);
67 if (match.Success)
68 {
69 if (match.Value.EndsWith("shared", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Shared;
70 else if (match.Value.EndsWith("server", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Server;
71 else if (match.Value.EndsWith("client", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Client;
72 }
73 }
74
75 var belongsInAssembly = false;
76 {
77 if (sourceCategory == SourceCategory.Shared) belongsInAssembly = true;
78 else if (sourceCategory == SourceCategory.Server && LuaCsSetup.IsServer) belongsInAssembly = true;
79 else if (sourceCategory == SourceCategory.Client && LuaCsSetup.IsClient) belongsInAssembly = true;
80 }
81
82 if (belongsInAssembly)
83 {
84 if (sources.ContainsKey(folder)) sources[folder].Add(s);
85 else sources.Add(folder, new List<string> { s });
86 }
87 }
88 else offendingSources.Add(s);
89 }
90 }
91 if (offendingSources.Count > 0)
92 {
93 LuaCsSetup.PrintCsError($"All C# sources must belong to <mod_folder>/CSharp/*\n Offending sources:{offendingSources.Select(s => $"\n {s}").Aggregate((s1, s2) => $"{s1}{s2}")}");
94 }
95 }

97 private IEnumerable<SyntaxTree> ParseSources() {
98 var syntaxTrees = new List<SyntaxTree>();
99
100 if (sources.Count <= 0) throw new Exception("No Cs sources detected");
101 syntaxTrees.Add(AssemblyInfoSyntaxTree(NET_SCRIPT_ASSEMBLY));
102 foreach ((var folder, var src) in sources)
103 {
104 try
105 {
106 foreach (var file in src)
107 {
108 var tree = SyntaxFactory.ParseSyntaxTree(File.ReadAllText(file), ParseOptions, file);
109 var error = CsScriptFilter.FilterSyntaxTree(tree as CSharpSyntaxTree); // Check file content for prohibited stuff
110 if (error != null) throw new Exception(error);
111
112 syntaxTrees.Add(tree);
113 }
114 }
115 catch (Exception ex)
116 {
117 LuaCsSetup.PrintCsError("Error loading '" + folder + "':\n" + ex.Message + "\n" + ex.StackTrace);
118 }
119 }
120
121 return syntaxTrees;
122 }

124 public List<Type> Compile()
125 {
126 IEnumerable<SyntaxTree> syntaxTrees = ParseSources();
127
128 var options = new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary)
129 .WithMetadataImportOptions(MetadataImportOptions.All)
130 .WithOptimizationLevel(OptimizationLevel.Release)
131 .WithAllowUnsafe(false);
132 var compilation = CSharpCompilation.Create(NET_SCRIPT_ASSEMBLY,syntaxTrees, defaultReferences, options);
133
134 using (var mem = new MemoryStream())
135 {
136 var result = compilation.Emit(mem);
137 if (!result.Success)
138 {
139 IEnumerable<Diagnostic> failures = result.Diagnostics.Where(d => d.IsWarningAsError || d.Severity == DiagnosticSeverity.Error);
140
141 string errStr = "NET MODS NOT LOADED | Mod compilation errors:";
142 foreach (Diagnostic diagnostic in failures)
143 errStr += $"\n{diagnostic}";
144 LuaCsSetup.PrintCsError(errStr);
145 }
146 else
147 {
148 mem.Seek(0, SeekOrigin.Begin);
149 var errStr = CsScriptFilter.FilterMetadata(new PEReader(mem).GetMetadataReader());
150 if (errStr == null)
151 {
152 mem.Seek(0, SeekOrigin.Begin);
153 Assembly = LoadFromStream(mem);
154 }
155 else LuaCsSetup.PrintCsError(errStr);
156 }
157 }
158
159 if (Assembly != null)
160 return Assembly.GetTypes().Where(t => t.IsSubclassOf(typeof(ACsMod))).ToList();
161 else
162 throw new Exception("Unable to create net mods assembly.");
163 }

165 private static string[] DirSearch(string sDir)
166 {
167 List<string> files = new List<string>();
168
169 try
170 {
171 foreach (string f in Directory.GetFiles(sDir))
172 {
173 files.Add(f);
174 }
175
176 foreach (string d in Directory.GetDirectories(sDir))
177 {
178 files.AddRange(DirSearch(d));
179 }
180 }
181 catch (System.Exception excpt)
182 {
183 Console.WriteLine(excpt.Message);
184 }
185
186 return files.ToArray();
