97  private void RunFolder(string folder)
98  {
99  var offendingSources = new List<string>();
100  foreach (var str in DirSearch(folder))
101  {
102  var s = str.Replace("\\", "/");
103 
104  if (s.EndsWith(".cs") && LuaCsFile.IsPathAllowedCsException(s))
105  {
106  if (rMaskPathValid.IsMatch(s)) // valid path
107  {
108  var sourceCategory = SourceCategory.Shared;
109  {
110  var match = rMaskPathCategory1.Match(s);
111  if (match.Success) match = rMaskPathCategory2.Match(match.Value);
112  if (match.Success)
113  {
114  if (match.Value.EndsWith("shared", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Shared;
115  else if (match.Value.EndsWith("server", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Server;
116  else if (match.Value.EndsWith("client", StringComparison.OrdinalIgnoreCase)) sourceCategory = SourceCategory.Client;
117  }
118  }
119 
120  var belongsInAssembly = false;
121  {
122  if (sourceCategory == SourceCategory.Shared) belongsInAssembly = true;
123  else if (sourceCategory == SourceCategory.Server && LuaCsSetup.IsServer) belongsInAssembly = true;
124  else if (sourceCategory == SourceCategory.Client && LuaCsSetup.IsClient) belongsInAssembly = true;
125  }
126 
127  if (belongsInAssembly)
128  {
129  if (sources.ContainsKey(folder)) sources[folder].Add(s);
130  else sources.Add(folder, new List<string> { s });
131  }
132  }
133  else offendingSources.Add(s);
134  }
135  }
136  if (offendingSources.Count > 0)
137  {
138  LuaCsSetup.PrintCsError($"All C# sources must belong to <mod_folder>/CSharp/*\n Offending sources:{offendingSources.Select(s => $"\n {s}").Aggregate((s1, s2) => $"{s1}{s2}")}");
139  }

144 
145  if (sources.Count <= 0) throw new Exception("No Cs sources detected");
146  syntaxTrees.Add(AssemblyInfoSyntaxTree(NET_SCRIPT_ASSEMBLY));
147  foreach ((var folder, var src) in sources)

149  try

154  var error = CsScriptFilter.FilterSyntaxTree(tree as CSharpSyntaxTree); // Check file content for prohibited stuff

157  syntaxTrees.Add(tree);
158  }

160  catch (Exception ex)

169  public List<Type> Compile()
170  {
171  IEnumerable<SyntaxTree> syntaxTrees = ParseSources();
172 
173  var options = new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary)
174  .WithMetadataImportOptions(MetadataImportOptions.All)
175  .WithOptimizationLevel(OptimizationLevel.Release)
176  .WithAllowUnsafe(false);
177  var compilation = CSharpCompilation.Create(NET_SCRIPT_ASSEMBLY,syntaxTrees, defaultReferences, options);
178 
179  using (var mem = new MemoryStream())
180  {
181  var result = compilation.Emit(mem);
182  if (!result.Success)
183  {
184  IEnumerable<Diagnostic> failures = result.Diagnostics.Where(d => d.IsWarningAsError || d.Severity == DiagnosticSeverity.Error);
185 
186  string errStr = "NET MODS NOT LOADED | Mod compilation errors:";
187  foreach (Diagnostic diagnostic in failures)
188  errStr += $"\n{diagnostic}";
189  LuaCsSetup.PrintCsError(errStr);
190  }
191  else
192  {
193  mem.Seek(0, SeekOrigin.Begin);
194  var errStr = CsScriptFilter.FilterMetadata(new PEReader(mem).GetMetadataReader());
195  if (errStr == null)
196  {
197  mem.Seek(0, SeekOrigin.Begin);
198  Assembly = LoadFromStream(mem);
199  }
200  else LuaCsSetup.PrintCsError(errStr);
201  }
202  }
203 
204  if (Assembly != null)
205  return Assembly.GetTypes().Where(t => t.IsSubclassOf(typeof(ACsMod))).ToList();
206  else
207  throw new Exception("Unable to create net mods assembly.");
208  }
