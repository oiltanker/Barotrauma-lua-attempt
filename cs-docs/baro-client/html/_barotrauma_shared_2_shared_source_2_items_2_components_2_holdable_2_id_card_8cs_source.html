72 public void Initialize(WayPoint spawnPoint, Character character)
73 {
74 item.AddTag("name:" + character.Name);
75
76 CharacterInfo info = character.Info;
77 if (info == null) { return; }
78
79 if (spawnPoint != null)
80 {
81 foreach (string s in spawnPoint.IdCardTags)
82 {
83 item.AddTag(s);
84 }
85 if (!string.IsNullOrWhiteSpace(spawnPoint.IdCardDesc))
86 {
87 item.Description = Description = spawnPoint.IdCardDesc;
88 }
89 }
90
91 TeamID = info.TeamID;
92
93 var head = info.Head;
94 if (head == null) { return; }
95
96 OwnerName = info.Name;
97 OwnerJobId = info.Job?.Prefab.Identifier ?? Identifier.Empty;
98 item.AddTag($"jobid:{OwnerJobId}");
99 OwnerTagSet = info.Head.Preset.TagSet;
100 OwnerHairIndex = head.HairIndex;
101 OwnerBeardIndex = head.BeardIndex;
102 OwnerMoustacheIndex = head.MoustacheIndex;
103 OwnerFaceAttachmentIndex = head.FaceAttachmentIndex;
104 OwnerHairColor = head.HairColor;
105 OwnerFacialHairColor = head.FacialHairColor;
106 OwnerSkinColor = head.SkinColor;
107 OwnerSheetIndex = head.SheetIndex;
108 }
