99  public void ClientWrite(IWriteMessage msg, VoteType voteType, object data)
100  {
101  msg.Write((byte)voteType);
102 
103  switch (voteType)
104  {
105  case VoteType.Sub:
106  if (!(data is SubmarineInfo sub)) { return; }
107  msg.Write(sub.EqualityCheckVal);
108  if (sub.EqualityCheckVal == 0)
109  {
110  //sub doesn't exist client-side, use hash to let the server know which one we voted for
111  msg.Write(sub.MD5Hash.StringRepresentation);
112  }
113  break;
114  case VoteType.Mode:
115  if (!(data is GameModePreset gameMode)) { return; }
116  msg.Write(gameMode.Identifier);
117  break;
118  case VoteType.EndRound:
119  if (!(data is bool)) { return; }
120  msg.Write((bool)data);
121  break;
122  case VoteType.Kick:
123  if (!(data is Client votedClient)) { return; }
124 
125  msg.Write(votedClient.ID);
126  break;
127  case VoteType.StartRound:
128  if (!(data is bool)) { return; }
129  msg.Write((bool)data);
130  break;
131  case VoteType.PurchaseAndSwitchSub:
132  case VoteType.PurchaseSub:
133  case VoteType.SwitchSub:
134  if (data is SubmarineInfo voteSub)
135  {
136  //initiate sub vote
137  msg.Write(true);
138  msg.Write(voteSub.Name);
139  }
140  else
141  {
142  if (!(data is int)) { return; }
143  msg.Write(false);
144  msg.Write((int)data);
145  }
146  break;
147  case VoteType.TransferMoney:
148  if (!(data is int)) { return; }
149  msg.Write(false); //not initiating a vote
150  msg.Write((int)data);
151  break;
152  }
153 
154  msg.WritePadBits();
155  }
