108  case "signal_in":
109  if (signal.value.Length > MaxMessageLength)
110  {

112  }
113 
114  string inputSignal = signal.value.Replace("\\n", "\n");
115  ShowOnDisplay(inputSignal, addToHistory: true, TextColor);
116  break;
117  case "set_text_color":

119  {
120  TextColor = XMLExtensions.ParseColor(signal.value, false);

123  break;

125  messageHistory.Clear();
