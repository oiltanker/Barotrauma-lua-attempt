124  private static List<string> ConvertInfoTextToXML(string[] csvContent, string language)
125  {
126  List<string> xmlContent = new List<string>
127  {
128  xmlHeader
129  };
130 
131  string translatedName = GetTranslatedName(language);
132  bool nowhitespace = TextManager.IsCJK(translatedName);
133 
134  xmlContent.Add($"<infotexts language=\"{language}\" nowhitespace=\"{nowhitespace.ToString().ToLower()}\" translatedname=\"{translatedName}\">");
135 
136  for (int i = 1; i < csvContent.Length; i++) // Start at one to ignore header
137  {
138  csvContent[i] = csvContent[i].Trim(separator);
139 
140  if (csvContent[i].Length == 0)
141  {
142  xmlContent.Add(string.Empty);
143  }
144  else
145  {
146  string[] split = csvContent[i].Split(separator, 3);
147 
148  if (split.Length >= 2) // Localization data
149  {
150  if (split.Length > 2 && !split[0].All(char.IsLower)) // Invalid header in line with localization data
151  {
152  split[0] = split[1];
153  split[1] = split[2];
154  split[2] = string.Empty;
155  }
156  split[1] = split[1].Replace(" & ", " &amp; ");
157  xmlContent.Add($"<{split[0]}>{split[1]}</{split[0]}>");
158  }
159  else if (split[0].Contains(".") && !split[0].Any(char.IsUpper)) // An empty field
160  {
161  xmlContent.Add($"<{split[0]}></{split[0]}>");
162  }
163  else // A header
164  {
165  xmlContent.Add($"<!-- {split[0]} -->");
166  }
167  }
168  }
169 
170  xmlContent.Add(string.Empty);
171  xmlContent.Add("</infotexts>");
172 
173  return xmlContent;
174  }

187  private static List<string> ConvertConversationsToXML(string[] csvContent, string language)
188  {
189  List<string> xmlContent = new List<string>();

191

193  bool nowhitespace = TextManager.IsCJK(translatedName);
194 
195  xmlContent.Add($"<Conversations identifier=\"vanillaconversations\" Language=\"{language}\" nowhitespace=\"{nowhitespace}\">");
196  xmlContent.Add(string.Empty);
197  xmlContent.Add("<!-- Personality traits -->");
198

201  {
202  if (csvContent[i].StartsWith("Personality"))

204  traitStart = i + 1;

206  }
207  }
208 
209  int conversationStart = -1;
210  for (int i = 0; i < csvContent.Length; i++)
211  {
212  if (csvContent[i].StartsWith("Generic"))
213  {
214  conversationStart = i;
215  break;
216  }
217  }
218 
219  if (traitStart == -1)
220  {
221  DebugConsole.ThrowError("Invalid formatting of NPCConversations, no traits found!");
222  return null;
223  }
224 
225  //DebugConsole.NewMessage("Count: " + NPCPersonalityTrait.List.Count);
226  var traits = NPCPersonalityTrait.GetAll(language.ToLanguageIdentifier()).ToArray();
227  for (int i = 0; i < traits.Length; i++) // Traits
228  {
229  //string[] split = SplitCSV(csvContent[traitStart + i].Trim(separator));
230  string[] split = csvContent[traitStart + i].Split(separator);
231  xmlContent.Add(
232  $"<PersonalityTrait " +
233  $"{GetVariable("name", split[1])}" +
234  $"{GetVariable("alloweddialogtags", string.Join(",", traits[i].AllowedDialogTags))}" +
235  $"{GetVariable("commonness", traits[i].Commonness.ToString(CultureInfo.InvariantCulture))}/>");
236  }
237 
238  xmlContent.Add(string.Empty);
239 
240  for (int i = conversationStart; i < csvContent.Length; i++) // Conversations
241  {
242  string[] split = csvContent[i].Split(separator);
243 
244  int emptyFields = 0;
245 
246  for (int j = 0; j < split.Length; j++)
247  {
248  if (split[j] == string.Empty) emptyFields++;
249  }
250 
251  if (emptyFields == split.Length) // Empty line with only commas, indicates the end of the previous conversation
252  {
253  HandleClosingElements(xmlContent, 0);
254  xmlContent.Add(string.Empty);
255  continue;
256  }
257  else if (emptyFields == split.Length - 1 && split[0] != string.Empty) // A header
258  {
259  xmlContent.Add($"<!-- {split[0]} -->");
260  continue;
261  }
262 
263  string speaker = split[1];
264  int depthIndex = int.Parse(split[2]);
265  // 3 = original line
266  string line = split[3].Replace("\"", "");

268  string allowedJobs = split[5].Replace("\"", "");
269  string speakerTags = split[6].Replace("\"", "");
270  string minIntensity = split[7].Replace("\"", "").Replace(",", ".");
271  string maxIntensity = split[8].Replace("\"", "").Replace(",", ".");
272 
273  string element =
274  $"{GetIndenting(depthIndex)}" +
275  $"<Conversation line=\"{line}\" " +
276  $"{GetVariable("speaker", speaker)}" +
277  $"{GetVariable("flags", flags)}" +
278  $"{GetVariable("allowedjobs", allowedJobs)}" +
279  $"{GetVariable("speakertags", speakerTags)}" +
280  $"{GetVariable("minintensity", minIntensity)}" +
281  $"{GetVariable("maxintensity", maxIntensity)}";
282 
283  bool nextIsSubConvo = false;
284  int nextDepth = 999;
285 
286  if (i < csvContent.Length - 1) // Not at the end of file
287  {
288  string[] nextConversationElement = csvContent[i + 1].Split(separator);
289 
290  if (nextConversationElement[1] != string.Empty)
291  {
292  nextDepth = int.Parse(nextConversationElement[2]);
293  nextIsSubConvo = nextDepth > depthIndex;
294  }
295 
296  if (!nextIsSubConvo)
297  {
298  xmlContent.Add(element.TrimEnd() + "/>");
299  if (nextDepth < depthIndex)
300  {
301  HandleClosingElements(xmlContent, nextDepth);
302  }
303  }
304  else
305  {
306  xmlContent.Add(element.TrimEnd() + ">");
307  conversationClosingIndent.Add(depthIndex);
308  }
309  }
310  else
311  {
312  xmlContent.Add(element.TrimEnd() + "/>");
313  }
314  }
315 
316  xmlContent.Add(string.Empty);
317  xmlContent.Add("</Conversations>");
318 
319  return xmlContent;
320  }

322  private static void HandleClosingElements(List<string> xmlContent, int targetDepth)
323  {
324  if (conversationClosingIndent.Count == 0) return;
325 
326  for (int k = conversationClosingIndent.Count - 1; k >= 0; k--)
327  {
328  int currentIndent = conversationClosingIndent[k];
329  if (currentIndent < targetDepth) break;
330  xmlContent.Add($"{GetIndenting(currentIndent)}</Conversation>");
331  conversationClosingIndent.RemoveAt(k);
332  }

335  private static string[] SplitCSV(string input) // Splits the .csv with regex, leaving commas inside quotation marks intact
336  {

350  return list.ToArray();
