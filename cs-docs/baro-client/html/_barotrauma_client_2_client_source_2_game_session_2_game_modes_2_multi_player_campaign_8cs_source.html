588 public static void ClientRead(IReadMessage msg)
589 {
590 bool isFirstRound = msg.ReadBoolean();
591 byte campaignID = msg.ReadByte();
592 UInt16 updateID = msg.ReadUInt16();
593 UInt16 saveID = msg.ReadUInt16();
594 string mapSeed = msg.ReadString();
595 UInt16 currentLocIndex = msg.ReadUInt16();
596 UInt16 selectedLocIndex = msg.ReadUInt16();
597
598 byte selectedMissionCount = msg.ReadByte();
599 List<int> selectedMissionIndices = new List<int>();
600 for (int i = 0; i < selectedMissionCount; i++)
601 {
602 selectedMissionIndices.Add(msg.ReadByte());
603 }
604
605 ushort ownedSubCount = msg.ReadUInt16();
606 List<ushort> ownedSubIndices = new List<ushort>();
607 for (int i = 0; i < ownedSubCount; i++)
608 {
609 ownedSubIndices.Add(msg.ReadUInt16());
610 }
611
612 bool allowDebugTeleport = msg.ReadBoolean();
613 float? reputation = null;
614 if (msg.ReadBoolean()) { reputation = msg.ReadSingle(); }
615
616 Dictionary<Identifier, float> factionReps = new Dictionary<Identifier, float>();
617 byte factionsCount = msg.ReadByte();
618 for (int i = 0; i < factionsCount; i++)
619 {
620 factionReps.Add(msg.ReadIdentifier(), msg.ReadSingle());
621 }
622
623 bool forceMapUI = msg.ReadBoolean();
624
625 bool purchasedHullRepairs = msg.ReadBoolean();
626 bool purchasedItemRepairs = msg.ReadBoolean();
627 bool purchasedLostShuttles = msg.ReadBoolean();
628
629 byte missionCount = msg.ReadByte();
630 var availableMissions = new List<(Identifier Identifier, byte ConnectionIndex)>();
631 for (int i = 0; i < missionCount; i++)
632 {
633 Identifier missionIdentifier = msg.ReadIdentifier();
634 byte connectionIndex = msg.ReadByte();
635 availableMissions.Add((missionIdentifier, connectionIndex));
636 }
637
638 var storeBalances = new Dictionary<Identifier, UInt16>();
639 if (msg.ReadBoolean())
640 {
641 byte storeCount = msg.ReadByte();
642 for (int i = 0; i < storeCount; i++)
643 {
644 Identifier identifier = msg.ReadIdentifier();
645 UInt16 storeBalance = msg.ReadUInt16();
646 storeBalances.Add(identifier, storeBalance);
647 }
648 }
649
650 var buyCrateItems = ReadPurchasedItems(msg, sender: null);
651 var subSellCrateItems = ReadPurchasedItems(msg, sender: null);
652 var purchasedItems = ReadPurchasedItems(msg, sender: null);
653 var soldItems = ReadSoldItems(msg);
654
655 ushort pendingUpgradeCount = msg.ReadUInt16();
656 List<PurchasedUpgrade> pendingUpgrades = new List<PurchasedUpgrade>();
657 for (int i = 0; i < pendingUpgradeCount; i++)
658 {
659 Identifier upgradeIdentifier = msg.ReadIdentifier();
660 UpgradePrefab prefab = UpgradePrefab.Find(upgradeIdentifier);
661 Identifier categoryIdentifier = msg.ReadIdentifier();
662 UpgradeCategory category = UpgradeCategory.Find(categoryIdentifier);
663 int upgradeLevel = msg.ReadByte();
664 if (prefab == null || category == null) { continue; }
665 pendingUpgrades.Add(new PurchasedUpgrade(prefab, category, upgradeLevel));
666 }
667
668 ushort purchasedItemSwapCount = msg.ReadUInt16();
669 List<PurchasedItemSwap> purchasedItemSwaps = new List<PurchasedItemSwap>();
670 for (int i = 0; i < purchasedItemSwapCount; i++)
671 {
672 UInt16 itemToRemoveID = msg.ReadUInt16();
673 Identifier itemToInstallIdentifier = msg.ReadIdentifier();
674 ItemPrefab itemToInstall = itemToInstallIdentifier.IsEmpty ? null : ItemPrefab.Find(string.Empty, itemToInstallIdentifier);
675 if (!(Entity.FindEntityByID(itemToRemoveID) is Item itemToRemove)) { continue; }
676 purchasedItemSwaps.Add(new PurchasedItemSwap(itemToRemove, itemToInstall));
677 }
678
679 bool hasCharacterData = msg.ReadBoolean();
680 CharacterInfo myCharacterInfo = null;
681 if (hasCharacterData)
682 {
683 myCharacterInfo = CharacterInfo.ClientRead(CharacterPrefab.HumanSpeciesName, msg);
684 }
685
686 if (!(GameMain.GameSession?.GameMode is MultiPlayerCampaign campaign) || campaignID != campaign.CampaignID)
687 {
688 string savePath = SaveUtil.CreateSavePath(SaveUtil.SaveType.Multiplayer);
689
690 GameMain.GameSession = new GameSession(null, savePath, GameModePreset.MultiPlayerCampaign, CampaignSettings.Unsure, mapSeed);
691 campaign = (MultiPlayerCampaign)GameMain.GameSession.GameMode;
692 campaign.CampaignID = campaignID;
693 GameMain.NetLobbyScreen.ToggleCampaignMode(true);
694 }
695
696 //server has a newer save file
697 if (NetIdUtils.IdMoreRecent(saveID, campaign.PendingSaveID))
698 {
699 campaign.PendingSaveID = saveID;
700 }
701
702 if (NetIdUtils.IdMoreRecent(updateID, campaign.lastUpdateID))
703 {
704 campaign.SuppressStateSending = true;
705 campaign.IsFirstRound = isFirstRound;
706
707 //we need to have the latest save file to display location/mission/store
708 if (campaign.LastSaveID == saveID)
709 {
710 campaign.ForceMapUI = forceMapUI;
711
712 UpgradeStore.WaitForServerUpdate = false;
713
714 campaign.Map.SetLocation(currentLocIndex == UInt16.MaxValue ? -1 : currentLocIndex);
715 campaign.Map.SelectLocation(selectedLocIndex == UInt16.MaxValue ? -1 : selectedLocIndex);
716 campaign.Map.SelectMission(selectedMissionIndices);
717
718 GameMain.GameSession.OwnedSubmarines.Clear();
719 foreach (int ownedSubIndex in ownedSubIndices)
