377 public static RectTransform Load(XElement element, RectTransform parent, Anchor defaultAnchor = Anchor.TopLeft)
378 {
379 Enum.TryParse(element.GetAttributeString("anchor", defaultAnchor.ToString()), out Anchor anchor);
380 Enum.TryParse(element.GetAttributeString("pivot", anchor.ToString()), out Pivot pivot);
381
382 Point? minSize = null, maxSize = null;
383 ScaleBasis scaleBasis = ScaleBasis.Normal;
384 if (element.Attribute("minsize") != null)
385 {
386 minSize = element.GetAttributePoint("minsize", Point.Zero);
387 }
388 if (element.Attribute("maxsize") != null)
389 {
390 maxSize = element.GetAttributePoint("maxsize", new Point(1000, 1000));
391 }
392 string sb = element.GetAttributeString("scalebasis", null);
393 if (sb != null)
394 {
395 Enum.TryParse(sb, ignoreCase: true, out scaleBasis);
396 }
397 RectTransform rectTransform;
398 if (element.Attribute("absolutesize") != null)
399 {
400 rectTransform = new RectTransform(element.GetAttributePoint("absolutesize", new Point(1000, 1000)), parent, anchor, pivot, scaleBasis)
401 {
402 minSize = minSize,
403 maxSize = maxSize
404 };
405 }
406 else
407 {
408 rectTransform = new RectTransform(element.GetAttributeVector2("relativesize", Vector2.One), parent, anchor, pivot, minSize, maxSize, scaleBasis);
409 }
410 rectTransform.RelativeOffset = element.GetAttributeVector2("relativeoffset", Vector2.Zero);
411 rectTransform.AbsoluteOffset = element.GetAttributePoint("absoluteoffset", Point.Zero);
412 return rectTransform;
413 }

425 protected void RecalculateScale()
426 {
427 var scale = LocalScale * globalScale;
428 var parents = GetParents();
429 Scale = parents.Any() ? parents.Select(rt => rt.LocalScale).Aggregate((parent, child) => parent * child) * scale : scale;

453 protected void RecalculateAbsoluteSize()
454 {
455 Point size = NonScaledParentUIRect.Size;
456 switch (ScaleBasis)
457 {
458 case ScaleBasis.BothWidth:
459 size.Y = size.X;
460 break;
461 case ScaleBasis.BothHeight:
462 size.X = size.Y;
463 break;
464 case ScaleBasis.Smallest:
465 if (size.X < size.Y)
466 {
467 size.Y = size.X;
468 }
469 else
470 {
471 size.X = size.Y;
472 }
473 break;
474 case ScaleBasis.Largest:
475 if (size.X > size.Y)
476 {
477 size.Y = size.X;
478 }
479 else
480 {
481 size.X = size.Y;
482 }
483 break;
484 }
485 size = size.Multiply(RelativeSize);
486 nonScaledSize = size.Clamp(MinSize, MaxSize);
487 recalculateRect = true;
488 SizeChanged?.Invoke();
489 }

516 private bool RemoveFromHierarchy(bool displayErrors = true)
517 {
518 if (Parent == null)
519 {
520 if (displayErrors)
521 {
522 DebugConsole.ThrowError("Parent null" + Environment.StackTrace.CleanupStackTrace());
523 }
524 return false;
525 }
526 if (!Parent.children.Contains(this))
527 {
528 if (displayErrors)
529 {
530 DebugConsole.ThrowError("The children of the parent does not contain this child. This should not be possible! " + Environment.StackTrace.CleanupStackTrace());
531 }
532 return false;
533 }
534 if (!Parent.children.Remove(this))
535 {
536 if (displayErrors)
537 {
538 DebugConsole.ThrowError("Unable to remove the child from the parent. " + Environment.StackTrace.CleanupStackTrace());
539 }
540 return false;
541 }
542 return true;

743 private IEnumerable<CoroutineStatus> DoMoveAnimation(Point targetPos, float duration)
744 {
745 Vector2 startPos = AbsoluteOffset.ToVector2();
746 float t = 0.0f;
747 while (t < duration && duration > 0.0f)
748 {
749 t += CoroutineManager.DeltaTime;
750 AbsoluteOffset = Vector2.SmoothStep(startPos, targetPos.ToVector2(), t / duration).ToPoint();
751 yield return CoroutineStatus.Running;
752 }
753 AbsoluteOffset = targetPos;
754 animTargetPos = null;
755 yield return CoroutineStatus.Success;

785 public static Point ConvertOffsetRelativeToAnchor(Point offset, Anchor anchor)
786 {
787 switch (anchor)
788 {
789 case Anchor.BottomRight:
790 return offset.Inverse();
791 case Anchor.BottomLeft:
792 case Anchor.BottomCenter:
793 return new Point(offset.X, -offset.Y);
794 case Anchor.TopRight:
795 case Anchor.CenterRight:
796 return new Point(-offset.X, offset.Y);
797 default:
798 return offset;
799 }

802 public static Point CalculatePivotOffset(Pivot pivot, Point size)
803 {
804 int width = size.X;
805 int height = size.Y;
806 switch (pivot)
807 {
808 case Pivot.TopLeft:
809 return Point.Zero;
810 case Pivot.TopCenter:
811 return new Point(-width / 2, 0);
812 case Pivot.TopRight:
813 return new Point(-width, 0);
814 case Pivot.CenterLeft:
815 return new Point(0, -height / 2);
816 case Pivot.Center:
817 return size.Divide(2).Inverse();
818 case Pivot.CenterRight:
819 return new Point(-width, -height / 2);
820 case Pivot.BottomLeft:
821 return new Point(0, -height);
822 case Pivot.BottomCenter:
823 return new Point(-width / 2, -height);
824 case Pivot.BottomRight:
825 return new Point(-width, -height);
826 default:
827 throw new NotImplementedException(pivot.ToString());
828 }
829 }

831 public static Point CalculateAnchorPoint(Anchor anchor, Rectangle parent)
832 {
833 switch (anchor)
834 {
835 case Anchor.TopLeft:
836 return parent.Location;
837 case Anchor.TopCenter:
838 return new Point(parent.Center.X, parent.Top);
839 case Anchor.TopRight:
840 return new Point(parent.Right, parent.Top);
841 case Anchor.CenterLeft:
842 return new Point(parent.Left, parent.Center.Y);
843 case Anchor.Center:
844 return parent.Center;
845 case Anchor.CenterRight:
846 return new Point(parent.Right, parent.Center.Y);
847 case Anchor.BottomLeft:
848 return new Point(parent.Left, parent.Bottom);
849 case Anchor.BottomCenter:
850 return new Point(parent.Center.X, parent.Bottom);
851 case Anchor.BottomRight:
852 return new Point(parent.Right, parent.Bottom);
853 default:
854 throw new NotImplementedException(anchor.ToString());
855 }
