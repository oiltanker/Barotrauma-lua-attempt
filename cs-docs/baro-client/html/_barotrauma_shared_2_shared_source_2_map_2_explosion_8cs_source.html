121 public void Explode(Vector2 worldPosition, Entity damageSource, Character attacker = null)
122 {
123 Hull hull = Hull.FindHull(worldPosition);
124 ExplodeProjSpecific(worldPosition, hull);
125
126 if (hull != null && !string.IsNullOrWhiteSpace(decal) && decalSize > 0.0f)
127 {
128 hull.AddDecal(decal, worldPosition, decalSize, isNetworkEvent: false);
129 }
130
131 float displayRange = Attack.Range;
132 if (damageSource is Item sourceItem)
133 {
134 displayRange *= 1.0f + sourceItem.GetQualityModifier(Quality.StatType.ExplosionRadius);
135 Attack.DamageMultiplier *= 1.0f + sourceItem.GetQualityModifier(Quality.StatType.ExplosionDamage);
136 Attack.SourceItem ??= sourceItem;
137 }
138
139 Vector2 cameraPos = GameMain.GameScreen.Cam.Position;
140 float cameraDist = Vector2.Distance(cameraPos, worldPosition) / 2.0f;
141 GameMain.GameScreen.Cam.Shake = cameraShake * Math.Max((cameraShakeRange - cameraDist) / cameraShakeRange, 0.0f);
142#if CLIENT
143 if (screenColor != Color.Transparent)
144 {
145 Color flashColor = Color.Lerp(Color.Transparent, screenColor, Math.Max((screenColorRange - cameraDist) / screenColorRange, 0.0f));
146 Screen.Selected.ColorFade(flashColor, Color.Transparent, screenColorDuration);
147 }
148#endif
149
150 if (displayRange < 0.1f) { return; }
151
152 if (!MathUtils.NearlyEqual(Attack.GetStructureDamage(1.0f), 0.0f) || !MathUtils.NearlyEqual(Attack.GetLevelWallDamage(1.0f), 0.0f))
153 {
154 RangedStructureDamage(worldPosition, displayRange, Attack.GetStructureDamage(1.0f), Attack.GetLevelWallDamage(1.0f), attacker, IgnoredSubmarines, Attack.EmitStructureDamageParticles);
155 }
156
157 if (BallastFloraDamage > 0.0f)
158 {
159 RangedBallastFloraDamage(worldPosition, displayRange, BallastFloraDamage, attacker);
160 }
161
162 if (EmpStrength > 0.0f)
163 {
164 float displayRangeSqr = displayRange * displayRange;
165 foreach (Item item in Item.ItemList)
166 {
167 float distSqr = Vector2.DistanceSquared(item.WorldPosition, worldPosition);
168 if (distSqr > displayRangeSqr) continue;
169
170 float distFactor = 1.0f - (float)Math.Sqrt(distSqr) / displayRange;
171
172 //damage repairable power-consuming items
173 var powered = item.GetComponent<Powered>();
174 if (powered == null || !powered.VulnerableToEMP) continue;
175 if (item.Repairables.Any())
176 {
177 item.Condition -= item.MaxCondition * EmpStrength * distFactor;
178 }
179
180 //discharge batteries
181 var powerContainer = item.GetComponent<PowerContainer>();
182 if (powerContainer != null)
183 {
184 powerContainer.Charge -= powerContainer.Capacity * EmpStrength * distFactor;
185 }
186 }
187 }
188
189 if (itemRepairStrength > 0.0f)
190 {
191 float displayRangeSqr = displayRange * displayRange;
192 foreach (Item item in Item.ItemList)
193 {
194 float distSqr = Vector2.DistanceSquared(item.WorldPosition, worldPosition);
195 if (distSqr > displayRangeSqr) continue;
196
197 float distFactor = 1.0f - (float)Math.Sqrt(distSqr) / displayRange;
198 //repair repairable items
199 if (item.Repairables.Any())
200 {
201 item.Condition += itemRepairStrength * distFactor;
202 }
203 }
204 }
205
206 if (MathUtils.NearlyEqual(force, 0.0f) && MathUtils.NearlyEqual(Attack.Stun, 0.0f) && MathUtils.NearlyEqual(Attack.GetTotalDamage(false), 0.0f) && !abilityExplosion)
207 {
208 return;
209 }
210
211 DamageCharacters(worldPosition, Attack, force, damageSource, attacker, applyToSelf);
212
213 if (GameMain.NetworkMember == null || !GameMain.NetworkMember.IsClient)
214 {
215 foreach (Item item in Item.ItemList)
216 {
217 if (item.Condition <= 0.0f) { continue; }
218 float dist = Vector2.Distance(item.WorldPosition, worldPosition);
219 float itemRadius = item.body == null ? 0.0f : item.body.GetMaxExtent();
220 dist = Math.Max(0.0f, dist - ConvertUnits.ToDisplayUnits(itemRadius));
221 if (dist > displayRange) { continue; }
222
223 if (dist < displayRange * 0.5f && applyFireEffects && !item.FireProof && ignoreFireEffectsForTags.None(t => item.HasTag(t)))
224 {
225 //don't apply OnFire effects if the item is inside a fireproof container
226 //(or if it's inside a container that's inside a fireproof container, etc)
227 Item container = item.Container;
228 bool fireProof = false;
229 while (container != null)
230 {
231 if (container.FireProof)
232 {
233 fireProof = true;
234 break;
235 }
236 container = container.Container;
237 }
238 if (!fireProof)
239 {
240 item.ApplyStatusEffects(ActionType.OnFire, 1.0f);
241 if (item.Condition <= 0.0f && GameMain.NetworkMember is { IsServer: true })
242 {
243 GameMain.NetworkMember.CreateEntityEvent(item, new Item.ApplyStatusEffectEventData(ActionType.OnFire));
244 }
245 }
246 }
247
248 if (item.Prefab.DamagedByExplosions && !item.Indestructible)
249 {
250 float distFactor = 1.0f - dist / displayRange;
251 float damageAmount = Attack.GetItemDamage(1.0f) * item.Prefab.ExplosionDamageMultiplier;
252
253 Vector2 explosionPos = worldPosition;
254 if (item.Submarine != null) { explosionPos -= item.Submarine.Position; }
255
256 damageAmount *= GetObstacleDamageMultiplier(ConvertUnits.ToSimUnits(explosionPos), worldPosition, item.SimPosition);
257 item.Condition -= damageAmount * distFactor;
258 }
259 }
260 }

265 private void DamageCharacters(Vector2 worldPosition, Attack attack, float force, Entity damageSource, Character attacker, bool applyToSelf)
266 {
267 if (attack.Range <= 0.0f) { return; }
268
269 //long range for the broad distance check, because large characters may still be in range even if their collider isn't
270 float broadRange = Math.Max(attack.Range * 10.0f, 10000.0f);
271
272 foreach (Character c in Character.CharacterList)
273 {
274 if (!c.Enabled ||
275 Math.Abs(c.WorldPosition.X - worldPosition.X) > broadRange ||
276 Math.Abs(c.WorldPosition.Y - worldPosition.Y) > broadRange)
277 {
278 continue;
279 }
280 if (c == attacker && !applyToSelf) { continue; }
281
282 if (onlyInside && c.Submarine == null) { continue; }
283 else if (onlyOutside && c.Submarine != null) { continue; }
284
285 Vector2 explosionPos = worldPosition;
286 if (c.Submarine != null) { explosionPos -= c.Submarine.Position; }
287
288 Hull hull = Hull.FindHull(explosionPos, null, false);
289 bool underWater = hull == null || explosionPos.Y < hull.Surface;
290
291 explosionPos = ConvertUnits.ToSimUnits(explosionPos);
292
293 Dictionary<Limb, float> distFactors = new Dictionary<Limb, float>();
294 Dictionary<Limb, float> damages = new Dictionary<Limb, float>();
295 List<Affliction> modifiedAfflictions = new List<Affliction>();
296 foreach (Limb limb in c.AnimController.Limbs)
297 {
298 if (limb.IsSevered || limb.IgnoreCollisions || !limb.body.Enabled) { continue; }
299
300 float dist = Vector2.Distance(limb.WorldPosition, worldPosition);
301
302 //calculate distance from the "outer surface" of the physics body
303 //doesn't take the rotation of the limb into account, but should be accurate enough for this purpose
304 float limbRadius = limb.body.GetMaxExtent();
305 dist = Math.Max(0.0f, dist - ConvertUnits.ToDisplayUnits(limbRadius));
306
307 if (dist > attack.Range) { continue; }
308
309 float distFactor = 1.0f - dist / attack.Range;
310
311 //solid obstacles between the explosion and the limb reduce the effect of the explosion
312 if (!ignoreCover)
313 {
314 distFactor *= GetObstacleDamageMultiplier(explosionPos, worldPosition, limb.SimPosition);
315 }
316 distFactors.Add(limb, distFactor);
317
318 modifiedAfflictions.Clear();
319 foreach (Affliction affliction in attack.Afflictions.Keys)
320 {
321 //previously the damage would be divided by the number of limbs (the intention was to prevent characters with more limbs taking more damage from explosions)
322 //that didn't work well on large characters like molochs and endworms: the explosions tend to only damage one or two of their limbs, and since the characters
323 //have lots of limbs, they tended to only take a fraction of the damage they should
324
325 //now we just divide by 10, which keeps the damage to normal-sized characters roughly the same as before and fixes the large characters
326 modifiedAfflictions.Add(affliction.CreateMultiplied(distFactor / 10));
327 }
328 c.LastDamageSource = damageSource;
329 if (attacker == null)
330 {
331 if (damageSource is Item item)
332 {
333 attacker = item.GetComponent<Projectile>()?.User;
334 if (attacker == null)
335 {
336 attacker = item.GetComponent<MeleeWeapon>()?.User;
337 }
338 }
339 }
340
341 AbilityAttackData attackData = new AbilityAttackData(Attack, c, attacker);
342 if (attackData.Afflictions != null)
343 {
344 modifiedAfflictions.AddRange(attackData.Afflictions);
345 }
346
347 //use a position slightly from the limb's position towards the explosion
348 //ensures that the attack hits the correct limb and that the direction of the hit can be determined correctly in the AddDamage methods
349 Vector2 dir = worldPosition - limb.WorldPosition;
350 Vector2 hitPos = limb.WorldPosition + (dir.LengthSquared() <= 0.001f ? Rand.Vector(1.0f) : Vector2.Normalize(dir)) * 0.01f;
351 AttackResult attackResult = c.AddDamage(hitPos, modifiedAfflictions, attack.Stun * distFactor, false, attacker: attacker, damageMultiplier: attack.DamageMultiplier * attackData.DamageMultiplier);
352 damages.Add(limb, attackResult.Damage);
353
354 if (attack.StatusEffects != null && attack.StatusEffects.Any())
355 {
356 attack.SetUser(attacker);
357 var statusEffectTargets = new List<ISerializableEntity>() { c, limb };
358 foreach (StatusEffect statusEffect in attack.StatusEffects)
359 {
360 statusEffect.Apply(ActionType.OnUse, 1.0f, damageSource, statusEffectTargets);
361 statusEffect.Apply(ActionType.Always, 1.0f, damageSource, statusEffectTargets);
362 statusEffect.Apply(underWater ? ActionType.InWater : ActionType.NotInWater, 1.0f, damageSource, statusEffectTargets);
363 }
364 }
365
366 if (limb.WorldPosition != worldPosition && !MathUtils.NearlyEqual(force, 0.0f))
367 {
368 Vector2 limbDiff = Vector2.Normalize(limb.WorldPosition - worldPosition);
369 if (!MathUtils.IsValid(limbDiff)) { limbDiff = Rand.Vector(1.0f); }
370 Vector2 impulse = limbDiff * distFactor * force;
371 Vector2 impulsePoint = limb.SimPosition - limbDiff * limbRadius;
372 limb.body.ApplyLinearImpulse(impulse, impulsePoint, maxVelocity: NetConfig.MaxPhysicsBodyVelocity * 0.2f);
373 }
374 }
375
376 if (c == Character.Controlled && !c.IsDead && playTinnitus)
377 {
378 Limb head = c.AnimController.GetLimb(LimbType.Head);
379 if (head != null && damages.TryGetValue(head, out float headDamage) && headDamage > 0.0f && distFactors.TryGetValue(head, out float headFactor))
380 {
381 PlayTinnitusProjSpecific(headFactor);
382 }
383 }
384
385 //sever joints
386 if (attack.SeverLimbsProbability > 0.0f)
387 {
388 foreach (Limb limb in c.AnimController.Limbs)
389 {
390 if (limb.character.Removed || limb.Removed) { continue; }
391 if (limb.IsSevered) { continue; }
392 if (!c.IsDead && !limb.CanBeSeveredAlive) { continue; }
393 if (distFactors.TryGetValue(limb, out float distFactor))
394 {
395 if (damages.TryGetValue(limb, out float damage))
396 {
397 c.TrySeverLimbJoints(limb, attack.SeverLimbsProbability * distFactor, damage, allowBeheading: true, attacker: attacker);
398 }
399 }
400 }
401 }
402 }

405 private static readonly List<Structure> damagedStructureList = new List<Structure>();
406 private static readonly Dictionary<Structure, float> damagedStructures = new Dictionary<Structure, float>();
410 public static Dictionary<Structure, float> RangedStructureDamage(Vector2 worldPosition, float worldRange, float damage, float levelWallDamage, Character attacker = null, IEnumerable<Submarine> ignoredSubmarines = null, bool emitWallDamageParticles = true)
411 {
412 float dist = 600.0f;
413 damagedStructureList.Clear();
414 foreach (MapEntity entity in MapEntity.mapEntityList)
415 {
416 if (!(entity is Structure structure)) { continue; }
417 if (ignoredSubmarines != null && entity.Submarine != null && ignoredSubmarines.Contains(entity.Submarine)) { continue; }
418
419 if (structure.HasBody &&
420 !structure.IsPlatform &&
421 Vector2.Distance(structure.WorldPosition, worldPosition) < dist * 3.0f)
422 {
423 damagedStructureList.Add(structure);
424 }
425 }
426
427 damagedStructures.Clear();
428 foreach (Structure structure in damagedStructureList)
429 {
430 for (int i = 0; i < structure.SectionCount; i++)
431 {
432 float distFactor = 1.0f - (Vector2.Distance(structure.SectionPosition(i, true), worldPosition) / worldRange);
433 if (distFactor <= 0.0f) { continue; }
434
435 structure.AddDamage(i, damage * distFactor, attacker, emitParticles: emitWallDamageParticles);
436
437 if (damagedStructures.ContainsKey(structure))
438 {
439 damagedStructures[structure] += damage * distFactor;
440 }
441 else
442 {
443 damagedStructures.Add(structure, damage * distFactor);
444 }
445 }
446 }
447
448 if (Level.Loaded != null && !MathUtils.NearlyEqual(levelWallDamage, 0.0f))
449 {
450 if (Level.Loaded?.LevelObjectManager != null)
451 {
452 foreach (var levelObject in Level.Loaded.LevelObjectManager.GetAllObjects(worldPosition, worldRange))
453 {
454 if (levelObject.Prefab.TakeLevelWallDamage)
455 {
456 float distFactor = 1.0f - (Vector2.Distance(levelObject.WorldPosition, worldPosition) / worldRange);
457 if (distFactor <= 0.0f) { continue; }
458 levelObject.AddDamage(levelWallDamage * distFactor, 1.0f, null);
459 }
460 }
461 }
462
463 for (int i = Level.Loaded.ExtraWalls.Count - 1; i >= 0; i--)
464 {
465 if (!(Level.Loaded.ExtraWalls[i] is DestructibleLevelWall destructibleWall)) { continue; }
466 foreach (var cell in destructibleWall.Cells)
467 {
468 if (cell.IsPointInside(worldPosition))
469 {
470 destructibleWall.AddDamage(levelWallDamage, worldPosition);
471 continue;
472 }
473 foreach (var edge in cell.Edges)
474 {
475 if (MathUtils.LineSegmentToPointDistanceSquared((edge.Point1 + cell.Translation).ToPoint(), (edge.Point2 + cell.Translation).ToPoint(), worldPosition.ToPoint()) < worldRange * worldRange)
476 {
477 destructibleWall.AddDamage(levelWallDamage, worldPosition);
478 break;
479 }
480 }
481 }
482 }
483 }
484
485 return damagedStructures;
486 }

488 public void RangedBallastFloraDamage(Vector2 worldPosition, float worldRange, float damage, Character attacker = null)
489 {
490 List<BallastFloraBehavior> ballastFlorae = new List<BallastFloraBehavior>();
491
492 foreach (Hull hull in Hull.HullList)
493 {
494 if (hull.BallastFlora != null) { ballastFlorae.Add(hull.BallastFlora); }
495 }
496
497 foreach (BallastFloraBehavior ballastFlora in ballastFlorae)
498 {
499 float resistanceMuliplier = ballastFlora.HasBrokenThrough ? 1f : 1f - ballastFlora.ExplosionResistance;
500 ballastFlora.Branches.ForEachMod(branch =>
501 {
502 Vector2 branchWorldPos = ballastFlora.GetWorldPosition() + branch.Position;
503 float branchDist = Vector2.Distance(branchWorldPos, worldPosition);
504 if (branchDist < worldRange)
505 {
506 float distFactor = 1.0f - (branchDist / worldRange);
507 if (distFactor <= 0.0f) { return; }
508
509 Vector2 explosionPos = worldPosition;
510 Vector2 branchPos = branchWorldPos;
511 if (ballastFlora.Parent?.Submarine != null)
512 {
513 explosionPos -= ballastFlora.Parent.Submarine.Position;
514 branchPos -= ballastFlora.Parent.Submarine.Position;
515 }
516 distFactor *= GetObstacleDamageMultiplier(ConvertUnits.ToSimUnits(explosionPos), worldPosition, ConvertUnits.ToSimUnits(branchPos));
517 ballastFlora.DamageBranch(branch, damage * distFactor * resistanceMuliplier, BallastFloraBehavior.AttackType.Explosives, attacker);
518 }
519 });
520 }

523 private static float GetObstacleDamageMultiplier(Vector2 explosionSimPos, Vector2 explosionWorldPos, Vector2 targetSimPos)
524 {
525 float damageMultiplier = 1.0f;
526 var obstacles = Submarine.PickBodies(targetSimPos, explosionSimPos, collisionCategory: Physics.CollisionItem | Physics.CollisionItemBlocking | Physics.CollisionWall);
527 foreach (var body in obstacles)
528 {
529 if (body.UserData is Item item)
530 {
531 var door = item.GetComponent<Door>();
532 if (door != null && !door.IsBroken) { damageMultiplier *= 0.01f; }
533 }
534 else if (body.UserData is Structure structure)
535 {
536 int sectionIndex = structure.FindSectionIndex(explosionWorldPos, world: true, clamp: true);
537 if (structure.SectionBodyDisabled(sectionIndex))
538 {
539 continue;
540 }
541 else if (structure.SectionIsLeaking(sectionIndex))
542 {
543 damageMultiplier *= 0.1f;
544 }
545 else
546 {
547 damageMultiplier *= 0.01f;
548 }
549 }
550 else
551 {
552 damageMultiplier *= 0.1f;
553 }
554 }
555 return damageMultiplier;
556 }
