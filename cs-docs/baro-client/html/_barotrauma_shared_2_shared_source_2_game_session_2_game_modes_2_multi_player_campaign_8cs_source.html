101 private void Load(XElement element)
102 {
103 PurchasedLostShuttles = element.GetAttributeBool("purchasedlostshuttles", false);
104 PurchasedHullRepairs = element.GetAttributeBool("purchasedhullrepairs", false);
105 PurchasedItemRepairs = element.GetAttributeBool("purchaseditemrepairs", false);
106 CheatsEnabled = element.GetAttributeBool("cheatsenabled", false);
107 if (CheatsEnabled)
108 {
109 DebugConsole.CheatsEnabled = true;
110#if USE_STEAM
111 if (!SteamAchievementManager.CheatsEnabled)
112 {
113 SteamAchievementManager.CheatsEnabled = true;
114#if CLIENT
115 new GUIMessageBox("Cheats enabled", "Cheat commands have been enabled on the server. You will not receive Steam Achievements until you restart the game.");
116#else
117 DebugConsole.NewMessage("Cheat commands have been enabled.", Color.Red);
118#endif
119 }
120#endif
121 }
122
123 foreach (var subElement in element.Elements())
124 {
125 switch (subElement.Name.ToString().ToLowerInvariant())
126 {
127 case "campaignsettings":
128 Settings = new CampaignSettings(subElement);
129#if CLIENT
130 GameMain.NetworkMember.ServerSettings.MaxMissionCount = Settings.MaxMissionCount;
131 GameMain.NetworkMember.ServerSettings.RadiationEnabled = Settings.RadiationEnabled;
132#endif
133 break;
134 case "map":
135 if (map == null)
136 {
137 //map not created yet, loading this campaign for the first time
138 map = Map.Load(this, subElement, Settings);
139 }
140 else
141 {
142 //map already created, update it
143 //if we're not downloading the initial save file (LastSaveID > 0),
144 //show notifications about location type changes
145 map.LoadState(subElement, LastSaveID > 0);
146 }
147 break;
148 case "metadata":
149 CampaignMetadata = new CampaignMetadata(this, subElement);
150 break;
151 case "upgrademanager":
152 case "pendingupgrades":
153 UpgradeManager = new UpgradeManager(this, subElement, isSingleplayer: false);
154 break;
155 case "bots" when GameMain.NetworkMember != null && GameMain.NetworkMember.IsServer:
156 CrewManager.HasBots = subElement.GetAttributeBool("hasbots", false);
157 CrewManager.AddCharacterElements(subElement);
158 CrewManager.ActiveOrdersElement = subElement.GetChildElement("activeorders");
159 break;
160 case "cargo":
161 CargoManager?.LoadPurchasedItems(subElement);
162 break;
163 case "pets":
164 petsElement = subElement;
165 break;
166 case "stats":
167 LoadStats(subElement);
168 break;
169 case Wallet.LowerCaseSaveElementName:
170 Bank = new Wallet(Option<Character>.None(), subElement);
171 break;
172#if SERVER
173 case "savedexperiencepoints":
174 foreach (XElement savedExp in subElement.Elements())
175 {
176 savedExperiencePoints.Add(new SavedExperiencePoints(savedExp));
177 }
178 break;
179#endif
180 }
181 }
182
183 int oldMoney = element.GetAttributeInt("money", 0);
184 if (oldMoney > 0)
185 {
186 Bank = new Wallet(Option<Character>.None())
187 {
188 Balance = oldMoney
189 };
190 }
191
192 CampaignMetadata ??= new CampaignMetadata(this);
193 UpgradeManager ??= new UpgradeManager(this);
194
195 InitCampaignData();
196#if SERVER
197 characterData.Clear();
198 string characterDataPath = GetCharacterDataSavePath();
199 if (!File.Exists(characterDataPath))
200 {
201 DebugConsole.ThrowError($"Failed to load the character data for the campaign. Could not find the file \"{characterDataPath}\".");
202 }
203 else
204 {
205 var characterDataDoc = XMLExtensions.TryLoadXml(characterDataPath);
206 if (characterDataDoc?.Root == null) { return; }
207 foreach (var subElement in characterDataDoc.Root.Elements())
208 {
209 characterData.Add(new CharacterCampaignData(subElement));
210 }
211 }
212#endif
213 }

215 public static List<SubmarineInfo> GetCampaignSubs()
216 {
217 bool isSubmarineVisible(SubmarineInfo s)
218 => !GameMain.NetworkMember.ServerSettings.HiddenSubs.Any(h
219 => s.Name.Equals(h, StringComparison.OrdinalIgnoreCase));
220
221 List<SubmarineInfo> availableSubs =
222 SubmarineInfo.SavedSubmarines
223 .Where(s =>
224 s.IsCampaignCompatible
225 && isSubmarineVisible(s))
226 .ToList();
227
228 if (!availableSubs.Any())
229 {
230 //None of the available subs were marked as campaign-compatible, just include all visible subs
231 availableSubs.AddRange(
232 SubmarineInfo.SavedSubmarines
233 .Where(isSubmarineVisible));
234 }
235
236 if (!availableSubs.Any())
237 {
238 //No subs are visible at all! Just make the selected one available
239 availableSubs.Add(GameMain.NetLobbyScreen.SelectedSub);
240 }
241
242 return availableSubs;
243 }

260 private static Dictionary<Identifier, List<PurchasedItem>> ReadPurchasedItems(IReadMessage msg, Client sender)
261 {
262 var items = new Dictionary<Identifier, List<PurchasedItem>>();
263 byte storeCount = msg.ReadByte();
264 for (int i = 0; i < storeCount; i++)
265 {
266 Identifier storeId = msg.ReadIdentifier();
267 items.Add(storeId, new List<PurchasedItem>());
268 UInt16 itemCount = msg.ReadUInt16();
269 for (int j = 0; j < itemCount; j++)
270 {
271 Identifier itemId = msg.ReadIdentifier();
272 int quantity = msg.ReadRangedInteger(0, CargoManager.MaxQuantity);
273 items[storeId].Add(new PurchasedItem(itemId, quantity, sender));
274 }
275 }
276 return items;
277 }

297 private static Dictionary<Identifier, List<SoldItem>> ReadSoldItems(IReadMessage msg)
298 {
299 var soldItems = new Dictionary<Identifier, List<SoldItem>>();
300 byte storeCount = msg.ReadByte();
301 for (int i = 0; i < storeCount; i++)
302 {
303 Identifier storeId = msg.ReadIdentifier();
304 soldItems.Add(storeId, new List<SoldItem>());
305 UInt16 itemCount = msg.ReadUInt16();
306 for (int j = 0; j < itemCount; j++)
307 {
308 Identifier prefabId = msg.ReadIdentifier();
309 UInt16 itemId = msg.ReadUInt16();
310 bool removed = msg.ReadBoolean();
311 byte sellerId = msg.ReadByte();
312 byte origin = msg.ReadByte();
313 soldItems[storeId].Add(new SoldItem(ItemPrefab.Prefabs[prefabId], itemId, removed, sellerId, (SoldItem.SellOrigin)origin));
314 }
315 }
316 return soldItems;
317 }
