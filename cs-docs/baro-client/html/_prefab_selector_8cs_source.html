72 #region Underlying implementations of the public methods, done separately to avoid nested locking
73 private T? basePrefabInternal;
74 private readonly List<T> overrides = new List<T>();
75
76 private T? activePrefabInternal => overrides.Any() ? overrides.First() : basePrefabInternal;
77
78 private void AddInternal(T prefab, bool isOverride)
79 {
80 if (isOverride)
81 {
82 if (overrides.Contains(prefab)) { throw new InvalidOperationException($"Duplicate prefab in PrefabSelector ({typeof(T)}, {prefab.Identifier}, {prefab.ContentFile.ContentPackage.Name})"); }
83 overrides.Add(prefab);
84 }
85 else
86 {
87 if (BasePrefab != null)
88 {
89 string prefabName
90 = prefab is MapEntityPrefab mapEntityPrefab
91 ? $"\"{mapEntityPrefab.OriginalName}\", \"{prefab.Identifier}\""
92 : $"\"{prefab.Identifier}\"";
93 throw new InvalidOperationException(
94 $"Failed to add the prefab {prefabName} ({prefab.GetType()}) from \"{prefab.ContentPackage?.Name ?? "[NULL]"}\" ({prefab.ContentPackage?.Dir ?? ""}): "
95 + $"a prefab with the same identifier from \"{ActivePrefab!.ContentPackage?.Name ?? "[NULL]"}\" ({ActivePrefab!.ContentPackage?.Dir ?? ""}) already exists; try overriding");
96 }
97 basePrefabInternal = prefab;
98 }
99 SortInternal();
100 }
101
102 private void RemoveIfContainsInternal(T prefab)
103 {
104 if (!ContainsInternal(prefab)) { return; }
105 RemoveInternal(prefab);
106 }
107
108 private void RemoveInternal(T prefab)
109 {
110 if (basePrefabInternal == prefab) { basePrefabInternal = null; }
111 else if (overrides.Contains(prefab)) { overrides.Remove(prefab); }
112 else { throw new InvalidOperationException($"Can't remove prefab from PrefabSelector ({typeof(T)}, {prefab.Identifier}, {prefab.ContentFile.ContentPackage.Name})"); }
113 prefab.Dispose();
114 SortInternal();
115 }
116
117 private void RemoveByFileInternal(ContentFile file, Action<T>? callback)
118 {
119 for (int i = overrides.Count-1; i >= 0; i--)
120 {
121 var prefab = overrides[i];
122 if (prefab.ContentFile == file)
123 {
124 RemoveInternal(prefab);
125 callback?.Invoke(prefab);
126 }
127 }
128
129 if (basePrefabInternal is { ContentFile: var baseFile } p && baseFile == file)
130 {
131 RemoveInternal(basePrefabInternal);
132 callback?.Invoke(p);
133 }
134 }
135
136 private void SortInternal()
137 {
138 overrides.Sort((p1, p2) => (p1.ContentPackage?.Index ?? int.MaxValue) - (p2.ContentPackage?.Index ?? int.MaxValue));
139 }
140
141 private bool isEmptyInternal => basePrefabInternal is null && !overrides.Any();
142
143 private bool ContainsInternal(T prefab) => basePrefabInternal == prefab || overrides.Contains(prefab);
144
145 private int IndexOfInternal(T prefab) => basePrefabInternal == prefab
146 ? overrides.Count
147 : overrides.IndexOf(prefab);
148
149 private bool IsOverrideInternal(T prefab) => IndexOfInternal(prefab) > 0;
150 #endregion

152 public IEnumerator<T> GetEnumerator()
153 {
154 T? basePrefab;

156 lock (overrides)
157 {
158 basePrefab = basePrefabInternal;

160 }
161 if (basePrefab != null) { yield return basePrefab; }
162 foreach (T prefab in overrideClone)
163 {
164 yield return prefab;
165 }
