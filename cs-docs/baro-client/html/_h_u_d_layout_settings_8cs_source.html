110 Padding = (int)(11 * GUI.Scale);

180 public static bool CloseHUD(Rectangle rect)
181 {
182 // Always close when hitting escape
183 if (PlayerInput.KeyHit(Microsoft.Xna.Framework.Input.Keys.Escape)) { return true; }
184
185 //don't close when the cursor is on a UI element
186 if (GUI.MouseOn != null) { return false; }
187
188 //don't close when hovering over an inventory element
189 if (Inventory.IsMouseOnInventory) { return false; }
190
191 bool input = PlayerInput.PrimaryMouseButtonDown() || PlayerInput.SecondaryMouseButtonClicked();
192 return input && !rect.Contains(PlayerInput.MousePosition);
193 }
