214 public void UpdateSounds()
215 {
216 if (!isActive || item.Condition <= 0.0f)
217 {
218 StopSounds(ActionType.OnActive);
219 }
220
221 if (loopingSound != null && loopingSoundChannel != null && loopingSoundChannel.IsPlaying)
222 {
223 if (Timing.TotalTime > lastMuffleCheckTime + 0.2f)
224 {
225 shouldMuffleLooping = SoundPlayer.ShouldMuffleSound(Character.Controlled, item.WorldPosition, loopingSound.Range, Character.Controlled?.CurrentHull);
226 lastMuffleCheckTime = (float)Timing.TotalTime;
227 }
228 loopingSoundChannel.Muffled = shouldMuffleLooping;
229 float targetGain = GetSoundVolume(loopingSound);
230 float gainDiff = targetGain - loopingSoundChannel.Gain;
231 loopingSoundChannel.Gain += Math.Abs(gainDiff) < 0.1f ? gainDiff : Math.Sign(gainDiff) * 0.1f;
232 loopingSoundChannel.Position = new Vector3(item.WorldPosition, 0.0f);
233 }
234 for (int i = 0; i < playingOneshotSoundChannels.Count; i++)
235 {
236 if (!playingOneshotSoundChannels[i].IsPlaying)
237 {
238 playingOneshotSoundChannels[i].Dispose();
239 playingOneshotSoundChannels[i] = null;
240 }
241 }
242 playingOneshotSoundChannels.RemoveAll(ch => ch == null);
243 foreach (SoundChannel channel in playingOneshotSoundChannels)
244 {
245 channel.Position = new Vector3(item.WorldPosition, 0.0f);
246 }
247 }

249 public void PlaySound(ActionType type, Character user = null)
250 {
251 if (!hasSoundsOfType[(int)type]) { return; }
252 if (GameMain.Client?.MidRoundSyncing ?? false) { return; }
253
254 if (loopingSound != null)
255 {
256 if (Vector3.DistanceSquared(GameMain.SoundManager.ListenerPosition, new Vector3(item.WorldPosition, 0.0f)) > loopingSound.Range * loopingSound.Range ||
257 (GetSoundVolume(loopingSound)) <= 0.0001f)
258 {
259 if (loopingSoundChannel != null)
260 {
261 loopingSoundChannel.FadeOutAndDispose();
262 loopingSoundChannel = null;
263 loopingSound = null;
264 }
265 return;
266 }
267
268 if (loopingSoundChannel != null && loopingSoundChannel.Sound != loopingSound.RoundSound.Sound)
269 {
270 loopingSoundChannel.FadeOutAndDispose();
271 loopingSoundChannel = null;
272 loopingSound = null;
273 }
274
275 if (loopingSoundChannel == null || !loopingSoundChannel.IsPlaying)
276 {
277 loopingSoundChannel = loopingSound.RoundSound.Sound.Play(
278 new Vector3(item.WorldPosition, 0.0f),
279 0.01f,
280 loopingSound.RoundSound.GetRandomFrequencyMultiplier(),
281 SoundPlayer.ShouldMuffleSound(Character.Controlled, item.WorldPosition, loopingSound.Range, Character.Controlled?.CurrentHull));
282 loopingSoundChannel.Looping = true;
283 //TODO: tweak
284 loopingSoundChannel.Near = loopingSound.Range * 0.4f;
285 loopingSoundChannel.Far = loopingSound.Range;
286 }
287
288 // Looping sound with manual selection mode should be changed if value of ManuallySelectedSound has changed
289 // Otherwise the sound won't change until the sound condition (such as being active) is disabled and re-enabled
290 if (loopingSoundChannel != null && loopingSoundChannel.IsPlaying && soundSelectionModes[type] == SoundSelectionMode.Manual)
291 {
292 var playingIndex = sounds[type].IndexOf(loopingSound);
293 var shouldBePlayingIndex = Math.Clamp(ManuallySelectedSound, 0, sounds[type].Count);
294 if (playingIndex != shouldBePlayingIndex)
295 {
296 loopingSoundChannel.FadeOutAndDispose();
297 loopingSoundChannel = null;
298 loopingSound = null;
299 }
300 }
301
302 return;
303 }
304
305 var matchingSounds = sounds[type];
306 if (loopingSoundChannel == null || !loopingSoundChannel.IsPlaying)
307 {
308 SoundSelectionMode soundSelectionMode = soundSelectionModes[type];
309 int index;
310 if (soundSelectionMode == SoundSelectionMode.CharacterSpecific && user != null)
311 {
312 index = user.ID % matchingSounds.Count;
313 }
314 else if (soundSelectionMode == SoundSelectionMode.ItemSpecific)
315 {
316 index = item.ID % matchingSounds.Count;
317 }
318 else if (soundSelectionMode == SoundSelectionMode.All)
319 {
320 foreach (ItemSound sound in matchingSounds)
321 {
322 PlaySound(sound, item.WorldPosition);
323 }
324 return;
325 }
326 else if (soundSelectionMode == SoundSelectionMode.Manual)
327 {
328 index = Math.Clamp(ManuallySelectedSound, 0, matchingSounds.Count);
329 }
330 else
331 {
332 index = Rand.Int(matchingSounds.Count);
333 }
334
335 PlaySound(matchingSounds[index], item.WorldPosition);
336 }

338 private void PlaySound(ItemSound itemSound, Vector2 position)
339 {
340 if (Vector2.DistanceSquared(new Vector2(GameMain.SoundManager.ListenerPosition.X, GameMain.SoundManager.ListenerPosition.Y), position) > itemSound.Range * itemSound.Range)
341 {
342 return;
343 }
344
345 if (itemSound.OnlyPlayInSameSub && item.Submarine != null && Character.Controlled != null)
346 {
347 if (Character.Controlled.Submarine == null || !Character.Controlled.Submarine.IsEntityFoundOnThisSub(item, includingConnectedSubs: true)) { return; }
348 }
349
350 if (itemSound.Loop)
351 {
352 if (loopingSoundChannel != null && loopingSoundChannel.Sound != itemSound.RoundSound.Sound)
353 {
354 loopingSoundChannel.FadeOutAndDispose(); loopingSoundChannel = null;
355 }
356 if (loopingSoundChannel == null || !loopingSoundChannel.IsPlaying)
357 {
358 float volume = GetSoundVolume(itemSound);
359 if (volume <= 0.0001f) { return; }
360 loopingSound = itemSound;
361 loopingSoundChannel = loopingSound.RoundSound.Sound.Play(
362 new Vector3(position.X, position.Y, 0.0f),
363 0.01f,
364 muffle: SoundPlayer.ShouldMuffleSound(Character.Controlled, position, loopingSound.Range, Character.Controlled?.CurrentHull));
365 loopingSoundChannel.Looping = true;
366 //TODO: tweak
367 loopingSoundChannel.Near = loopingSound.Range * 0.4f;
368 loopingSoundChannel.Far = loopingSound.Range;
369 }
370 }
371 else
372 {
373 float volume = GetSoundVolume(itemSound);
374 if (volume <= 0.0001f) { return; }
375 var channel = SoundPlayer.PlaySound(itemSound.RoundSound.Sound, position, volume, itemSound.Range, itemSound.RoundSound.GetRandomFrequencyMultiplier(), item.CurrentHull, ignoreMuffling: itemSound.RoundSound.IgnoreMuffling);
376 if (channel != null) { playingOneshotSoundChannels.Add(channel); }
377 }

394 private float GetSoundVolume(ItemSound sound)
395 {
396 if (sound == null) { return 0.0f; }
397 if (sound.VolumeProperty == "") { return sound.VolumeMultiplier; }
398
399 if (SerializableProperties.TryGetValue(sound.VolumeProperty, out SerializableProperty property))
400 {
401 float newVolume;
402 try
403 {
404 newVolume = property.GetFloatValue(this);
405 }
406 catch
407 {
408 return 0.0f;
409 }
410 newVolume *= sound.VolumeMultiplier;
411
412 if (!MathUtils.IsValid(newVolume))
413 {
414 DebugConsole.Log("Invalid sound volume (item " + item.Name + ", " + GetType().ToString() + "): " + newVolume);
415 GameAnalyticsManager.AddErrorEventOnce(
416 "ItemComponent.PlaySound:" + item.Name + GetType().ToString(),
417 GameAnalyticsManager.ErrorSeverity.Error,
418 "Invalid sound volume (item " + item.Name + ", " + GetType().ToString() + "): " + newVolume);
419 return 0.0f;
420 }
421
422 return MathHelper.Clamp(newVolume, 0.0f, 1.0f);
423 }
424
425 return 0.0f;
426 }

468 private bool LoadElemProjSpecific(ContentXElement subElement)
469 {
470 switch (subElement.Name.ToString().ToLowerInvariant())
471 {
472 case "guiframe":
473 if (subElement.GetAttribute("rect") != null)
474 {
475 DebugConsole.ThrowError($"Error in item config \"{item.ConfigFilePath}\" - GUIFrame defined as rect, use RectTransform instead.");
476 break;
477 }
478 GuiFrameSource = subElement;
479 ReloadGuiFrame();
480 break;
481 case "alternativelayout":
482 AlternativeLayout = GUILayoutSettings.Load(subElement);
483 break;
484 case "itemsound":
485 case "sound":
486 //TODO: this validation stuff should probably go somewhere else
487 string filePath = subElement.GetAttributeStringUnrestricted("file", "");
488
489 if (filePath.IsNullOrEmpty()) { filePath = subElement.GetAttributeStringUnrestricted("sound", ""); }
490
491 if (filePath.IsNullOrEmpty())
492 {
493 DebugConsole.ThrowError(
494 $"Error when instantiating item \"{item.Name}\" - sound with no file path set");
495 break;
496 }
497
498 ActionType type;
499 try
500 {
501 type = (ActionType)Enum.Parse(typeof(ActionType), subElement.GetAttributeString("type", ""), true);
502 }
503 catch (Exception e)
504 {
505 DebugConsole.ThrowError("Invalid sound type in " + subElement + "!", e);
506 break;
507 }
508
509 RoundSound sound = RoundSound.Load(subElement);
510 if (sound == null) { break; }
511 ItemSound itemSound = new ItemSound(sound, type,
512 subElement.GetAttributeBool("loop", false),
513 subElement.GetAttributeBool("onlyinsamesub", false))
514 {
515 VolumeProperty = subElement.GetAttributeIdentifier("volumeproperty", "")
516 };
517
518 if (soundSelectionModes == null) soundSelectionModes = new Dictionary<ActionType, SoundSelectionMode>();
519 if (!soundSelectionModes.ContainsKey(type) || soundSelectionModes[type] == SoundSelectionMode.Random)
520 {
521 SoundSelectionMode selectionMode = SoundSelectionMode.Random;
522 Enum.TryParse(subElement.GetAttributeString("selectionmode", "Random"), out selectionMode);
523 soundSelectionModes[type] = selectionMode;
524 }
525
526 List<ItemSound> soundList = null;
527 if (!sounds.TryGetValue(itemSound.Type, out soundList))
528 {
529 soundList = new List<ItemSound>();
530 sounds.Add(itemSound.Type, soundList);
531 hasSoundsOfType[(int)itemSound.Type] = true;
532 }
533
534 soundList.Add(itemSound);
535 break;
536 default:
537 return false; //unknown element
538 }
539 return true; //element processed
540 }

579 protected void StartDelayedCorrection(IReadMessage buffer, float sendingTime, bool waitForMidRoundSync = false)

586 private IEnumerable<CoroutineStatus> DoDelayedCorrection(IReadMessage buffer, float sendingTime, bool waitForMidRoundSync)
587 {
588 while (GameMain.Client != null &&
589 (correctionTimer > 0.0f || (waitForMidRoundSync && GameMain.Client.MidRoundSyncing)))
590 {
591 correctionTimer -= CoroutineManager.DeltaTime;
592 yield return CoroutineStatus.Running;
593 }
594
595 if (item.Removed || GameMain.Client == null)
596 {
597 yield return CoroutineStatus.Success;
598 }
599
600 ((IServerSerializable)this).ClientEventRead(buffer, sendingTime);
601
602 correctionTimer = 0.0f;
603 delayedCorrectionCoroutine = null;
604
605 yield return CoroutineStatus.Success;
606 }
