58 private void DetermineCargo()
59 {
60 if (this.sub == null || itemConfig == null)
61 {
62 calculatedReward = Prefab.Reward;
63 return;
64 }
65
66 itemsToSpawn.Clear();
67 List<(ItemContainer container, int freeSlots)> containers = sub.GetCargoContainers();
68 containers.Sort((c1, c2) => { return c2.container.Capacity.CompareTo(c1.container.Capacity); });
69
70 previouslySelectedMissions.Clear();
71 if (GameMain.GameSession?.StartLocation?.SelectedMissions != null)
72 {
73 bool isPriorMission = true;
74 foreach (Mission mission in GameMain.GameSession.StartLocation.SelectedMissions)
75 {
76 if (!(mission is CargoMission otherMission)) { continue; }
77 if (mission == this) { isPriorMission = false; }
78 previouslySelectedMissions.Add(otherMission);
79 if (!isPriorMission) { continue; }
80 foreach (var (element, container) in otherMission.itemsToSpawn)
81 {
82 for (int i = 0; i < containers.Count; i++)
83 {
84 if (containers[i].container == container)
85 {
86 containers[i] = (containers[i].container, containers[i].freeSlots - 1);
87 break;
88 }
89 }
90 }
91 }
92 }
93
94 maxItemCount = 0;
95 foreach (var subElement in itemConfig.Elements())
96 {
97 int maxCount = subElement.GetAttributeInt("maxcount", 10);
98 maxItemCount += maxCount;
99 }
100
101 for (int i = 0; i < containers.Count; i++)
102 {
103 foreach (var subElement in itemConfig.Elements())
104 {
105 int maxCount = subElement.GetAttributeInt("maxcount", 10);
106 if (itemsToSpawn.Count(it => it.element == subElement) >= maxCount) { continue; }
107 ItemPrefab itemPrefab = FindItemPrefab(subElement);
108 while (containers[i].freeSlots > 0 && containers[i].container.Inventory.CanBePut(itemPrefab))
109 {
110 containers[i] = (containers[i].container, containers[i].freeSlots - 1);
111 itemsToSpawn.Add((subElement, containers[i].container));
112 if (itemsToSpawn.Count(it => it.element == subElement) >= maxCount) { break; }
113 }
114 }
115 }
116
117 if (!itemsToSpawn.Any())
118 {
119 itemsToSpawn.Add((itemConfig.Elements().First(), null));
120 }
121
122 calculatedReward = 0;
123 foreach (var (element, container) in itemsToSpawn)
124 {
125 int price = element.GetAttributeInt("reward", Prefab.Reward / itemsToSpawn.Count);
126 if (rewardPerCrate.HasValue)
127 {
128 if (price != rewardPerCrate.Value) { rewardPerCrate = -1; }
129 }
130 else
131 {
132 rewardPerCrate = price;
133 }
134 calculatedReward += price;
135 }
136 if (rewardPerCrate.HasValue && rewardPerCrate < 0) { rewardPerCrate = null; }
137
138 string rewardText = $"‖color:gui.orange‖{string.Format(System.Globalization.CultureInfo.InvariantCulture, "{0:N0}", GetReward(sub))}‖end‖";
139 if (descriptionWithoutReward != null) { description = descriptionWithoutReward.Replace("[reward]", rewardText); }
140 }

142 public override int GetReward(Submarine sub)
143 {
144 // If we are not at the location of the mission, skip the calculation of the reward
145 if (GameMain.GameSession?.StartLocation != Locations[0])
146 {
147 return calculatedReward;
148 }
149
150 bool missionsChanged = false;
151 if (GameMain.GameSession?.StartLocation?.SelectedMissions != null)
152 {
153 List<Mission> currentMissions = GameMain.GameSession.StartLocation.SelectedMissions.Where(m => m is CargoMission).ToList();
154 if (currentMissions.Count != previouslySelectedMissions.Count)
155 {
156 missionsChanged = true;
157 }
158 else
159 {
160 for (int i = 0; i < previouslySelectedMissions.Count; i++)
161 {
162 if (previouslySelectedMissions[i] != currentMissions[i])
163 {
164 missionsChanged = true;
165 break;
166 }
167 }
168 }
169 }
170
171 if (sub != this.sub || missionsChanged)
172 {
173 this.sub = sub;
174 DetermineCargo();
175 }
176 return calculatedReward;
177 }

179 private void InitItems()
180 {
181 this.sub = Submarine.MainSub;
182 DetermineCargo();
183
184 items.Clear();
185 parentInventoryIDs.Clear();
186 parentItemContainerIndices.Clear();
187 inventorySlotIndices.Clear();
188
189 if (itemConfig == null)
190 {
191 DebugConsole.ThrowError("Failed to initialize items for cargo mission (itemConfig == null)");
192 return;
193 }
194
195 foreach (var (element, container) in itemsToSpawn)
196 {
197 LoadItemAsChild(element, container?.Item);
198 }
199
200 if (requiredDeliveryAmount <= 0.0f) { requiredDeliveryAmount = 1.0f; }
201 }

203 private void LoadItemAsChild(ContentXElement element, Item parent)
204 {
205 ItemPrefab itemPrefab = FindItemPrefab(element);
206
207 Vector2? position = GetCargoSpawnPosition(itemPrefab, out Submarine cargoRoomSub);
208 if (!position.HasValue) { return; }
209
210 var item = new Item(itemPrefab, position.Value, cargoRoomSub)
211 {
212 SpawnedInCurrentOutpost = true,
213 AllowStealing = false
214 };
215 item.FindHull();
216 items.Add(item);
217
218 if (parent != null && parent.GetComponent<ItemContainer>() != null)
219 {
220 parentInventoryIDs.Add(item, parent.ID);
221 parentItemContainerIndices.Add(item, (byte)parent.GetComponentIndex(parent.GetComponent<ItemContainer>()));
222 parent.Combine(item, user: null);
223 inventorySlotIndices.Add(item, item.ParentInventory?.FindIndex(item) ?? -1);
224 }
225
226 foreach (var subElement in element.Elements())
227 {
228 int amount = subElement.GetAttributeInt("amount", 1);
229 for (int i = 0; i < amount; i++)
230 {
231 LoadItemAsChild(subElement, item);
232 }
233 }

247 public override void End()
248 {
249 if (Submarine.MainSub != null && Submarine.MainSub.AtEndExit)
250 {
251 int deliveredItemCount = items.Count(it => IsItemDelivered(it));
252 if (deliveredItemCount / (float)items.Count >= requiredDeliveryAmount)
253 {
254 GiveReward();
255 completed = true;
256 if (Prefab.LocationTypeChangeOnCompleted != null)
257 {
258 ChangeLocationType(Prefab.LocationTypeChangeOnCompleted);
259 }
260 }
261 }
262
263 foreach (Item item in items)
264 {
265 if (!item.Removed) { item.Remove(); }
266 }
267 items.Clear();
268 failed = !completed;
269 }

271 private bool IsItemDelivered(Item item)
272 {
273 if (item.Removed || item.Condition <= 0.0f || Submarine.MainSub == null) { return false; }
274 var submarine = item.Submarine ?? item.GetRootContainer()?.Submarine;
275 return submarine == Submarine.MainSub || Submarine.MainSub.GetConnectedSubs().Contains(submarine);
276 }
