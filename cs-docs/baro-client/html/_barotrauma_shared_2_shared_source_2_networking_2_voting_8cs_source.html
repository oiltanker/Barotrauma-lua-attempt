32 public T HighestVoted<T>(VoteType voteType, List<Client> voters)
33 {
34 if (voteType == VoteType.Sub && !GameMain.NetworkMember.ServerSettings.AllowSubVoting) { return default; }
35 if (voteType == VoteType.Mode && !GameMain.NetworkMember.ServerSettings.AllowModeVoting) { return default; }
36
37 IReadOnlyDictionary<T, int> voteList = GetVoteCounts<T>(voteType, voters);
38
39 T selected = default;
40 int highestVotes = 0;
41 foreach (KeyValuePair<T, int> votable in voteList)
42 {
43 if (voteType == VoteType.Sub
44 && votable.Key is SubmarineInfo subInfo
45 && GameMain.NetworkMember.ServerSettings.HiddenSubs.Contains(subInfo.Name))
46 {
47 //This sub is hidden so it can't be voted for, skip
48 continue;
49 }
50 if (selected == null || votable.Value > highestVotes)
51 {
52 highestVotes = votable.Value;
53 selected = votable.Key;
54 }
55 }
56
57 return selected;
