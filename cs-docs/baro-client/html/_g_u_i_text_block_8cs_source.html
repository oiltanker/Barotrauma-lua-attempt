467  public ImmutableArray<Vector2> GetAllCaretPositions()
468  {
469  if (cachedCaretPositions.Any())
470  {
471  return cachedCaretPositions;
472  }
473  string textDrawn = Censor ? CensoredText : Text.SanitizedValue;
474  float w = Wrap
475  ? (Rect.Width - Padding.X - Padding.Z) / TextScale
476  : float.PositiveInfinity;
477  string wrapped = Font.WrapText(textDrawn, w, out Vector2[] positions);
478  int textWidth = (int)Font.MeasureString(wrapped).X;
479  int alignmentXDiff
480  = textAlignment.HasFlag(Alignment.Right) ? textWidth
481  : textAlignment.HasFlag(Alignment.Center) ? textWidth / 2
482  : 0;
483  cachedCaretPositions = positions
484  .Select(p => p - new Vector2(alignmentXDiff, 0))
485  .Select(p => p * TextScale + TextPos - Origin * TextScale)
486  .ToImmutableArray();
487  return cachedCaretPositions;

495  public int GetCaretIndexFromLocalPos(in Vector2 pos)
496  {
497  var positions = GetAllCaretPositions();
498  if (positions.Length == 0) { return 0; }
499 
500  float closestXDist = float.PositiveInfinity;
501  float closestYDist = float.PositiveInfinity;
502  int closestIndex = -1;
503  for (int i = 0; i < positions.Length; i++)
504  {
505  float xDist = Math.Abs(pos.X - positions[i].X);
506  float yDist = Math.Abs(pos.Y - (positions[i].Y + Font.LineHeight * 0.5f));
507  if (yDist < closestYDist || (MathUtils.NearlyEqual(yDist, closestYDist) && xDist < closestXDist))
508  {
509  closestIndex = i;
510  closestXDist = xDist;
511  closestYDist = yDist;
512  }
513  }
514 
515  return closestIndex >= 0 ? closestIndex : Text.Length;
