318 public LevelObjectPrefab(ContentXElement element, LevelObjectPrefabsFile file, Identifier identifierOverride = default) : base(file, ParseIdentifier(identifierOverride, element))
319 {
320 ChildObjects = new List<ChildObject>();
321 LevelTriggerElements = new List<ContentXElement>();
322 OverrideProperties = new List<LevelObjectPrefab>();
323 OverrideCommonness = new Dictionary<Identifier, float>();
324
325 SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
326 if (element != null)
327 {
328 Config = element;
329
330 LoadElements(file, element, -1);
331 InitProjSpecific(element);
332 }
333
334 //use the maximum width of the sprite as the minimum surface width if no value is given
335 if (element != null && !element.Attributes("minsurfacewidth").Any())
336 {
337 if (Sprites.Any()) MinSurfaceWidth = Sprites[0].size.X * MaxSize;
338 if (DeformableSprite != null) MinSurfaceWidth = Math.Max(MinSurfaceWidth, DeformableSprite.Size.X * MaxSize);
339 }

342 public static Identifier ParseIdentifier(Identifier identifierOverride, XElement element)
343 {
344 if (!identifierOverride.IsEmpty) { return identifierOverride; }
345 Identifier identifier = element.GetAttributeIdentifier("identifier", "");
346 if (identifier.IsEmpty)
347 {
348#if DEBUG
349 DebugConsole.ThrowError($"Level object prefab \"{element.Name}\" has no identifier! Using the name as the identifier instead...");
350#else
351 DebugConsole.AddWarning($"Level object prefab \"{element.Name}\" has no identifier! Using the name as the identifier instead...");
352#endif
353 identifier = element.NameAsIdentifier();
354 }
355 return identifier;
356 }

358 private void LoadElements(LevelObjectPrefabsFile file, ContentXElement element, int parentTriggerIndex)
359 {
360 int propertyOverrideCount = 0;
361 foreach (var subElement in element.Elements())
362 {
363 switch (subElement.Name.ToString().ToLowerInvariant())
364 {
365 case "sprite":
366 var newSprite = new Sprite(subElement, lazyLoad: true);
367 Sprites.Add(newSprite);
368 var spriteSpecificPhysicsBodyElement =
369 subElement.GetChildElement("PhysicsBody") ?? subElement.GetChildElement("Body") ??
370 subElement.GetChildElement("physicsbody") ?? subElement.GetChildElement("body");
371 if (spriteSpecificPhysicsBodyElement != null)
372 {
373 SpriteSpecificPhysicsBodyElements.Add(newSprite, spriteSpecificPhysicsBodyElement);
374 }
375 break;
376 case "deformablesprite":
377 DeformableSprite = new DeformableSprite(subElement, lazyLoad: true);
378 break;
379 case "overridecommonness":
380 Identifier levelType = subElement.GetAttributeIdentifier("leveltype", Identifier.Empty);
381 if (!OverrideCommonness.ContainsKey(levelType))
382 {
383 OverrideCommonness.Add(levelType, subElement.GetAttributeFloat("commonness", 1.0f));
384 }
385 break;
386 case "leveltrigger":
387 case "trigger":
388 OverrideProperties.Add(null);
389 LevelTriggerElements.Add(subElement);
390 LoadElements(file, subElement, LevelTriggerElements.Count - 1);
391 break;
392 case "childobject":
393 ChildObjects.Add(new ChildObject(subElement));
394 break;
395 case "overrideproperties":
396 var propertyOverride = new LevelObjectPrefab(subElement, file, identifierOverride: $"{Identifier}-{propertyOverrideCount}".ToIdentifier());
397 OverrideProperties[OverrideProperties.Count - 1] = propertyOverride;
398 if (!propertyOverride.Sprites.Any() && propertyOverride.DeformableSprite == null)
399 {
400 propertyOverride.Sprites = Sprites;
401 propertyOverride.DeformableSprite = DeformableSprite;
402 }
403 propertyOverrideCount++;
404 break;
405 case "body":
406 case "physicsbody":
407 PhysicsBodyElement = subElement;
408 PhysicsBodyTriggerIndex = parentTriggerIndex;
409 break;
410 }
411 }
412 }

418 {
419 if (generationParams != null &&
420 generationParams.Identifier != Identifier.Empty &&
421 OverrideCommonness.TryGetValue(generationParams.Identifier, out float commonness))
422 {
423 return commonness;
424 }
425 return requireCaveSpecificOverride ? 0.0f : Commonness;

428 public float GetCommonness(LevelGenerationParams generationParams)
429 {
430 if (generationParams != null &&
431 generationParams.Identifier != Identifier.Empty &&
432 (OverrideCommonness.TryGetValue(generationParams.Identifier, out float commonness) ||
433 (!generationParams.OldIdentifier.IsEmpty && OverrideCommonness.TryGetValue(generationParams.OldIdentifier, out commonness))))
434 {
435 return commonness;
436 }
437 return Commonness;
