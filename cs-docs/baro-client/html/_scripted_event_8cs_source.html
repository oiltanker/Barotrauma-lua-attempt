151  public override void Update(float deltaTime)
152  {
153  int botCount = 0;
154  int playerCount = 0;
155  foreach (Character c in Character.CharacterList)
156  {
157  if (c.IsPlayer)
158  {
159  playerCount++;
160  }
161  else if (c.IsBot)
162  {
163  botCount++;
164  }
165  }
166  if (Entity.EntityCount != prevEntityCount || botCount != prevBotCount || playerCount != prevPlayerCount)
167  {
168  cachedTargets.Clear();
169  prevEntityCount = Entity.EntityCount;
170  prevBotCount = botCount;
171  prevPlayerCount = playerCount;
172  }
173 
174  if (!Actions.Any())
175  {
176  Finished();
177  return;
178  }
179 
180  var currentAction = Actions[CurrentActionIndex];
181  if (!currentAction.CanBeFinished())
182  {
183  Finished();
184  return;
185  }
186 
187  string goTo = null;
188  if (currentAction.IsFinished(ref goTo))
189  {
190  if (string.IsNullOrEmpty(goTo))
191  {
192  CurrentActionIndex++;
193  }
194  else
195  {
196  CurrentActionIndex = -1;
197  Actions.ForEach(a => a.Reset());
198  for (int i = 0; i < Actions.Count; i++)
199  {
200  if (Actions[i].SetGoToTarget(goTo))
201  {
202  CurrentActionIndex = i;
203  break;
204  }
205  }
206  }
207 
208  if (CurrentActionIndex >= Actions.Count || CurrentActionIndex < 0)
209  {
210  Finished();
211  }
212  }
213  else
214  {
215  currentAction.Update(deltaTime);
216  }
217  }

219  public override bool LevelMeetsRequirements()
220  {
221  if (requiredDestinationTypes == null) { return true; }
222  var currLocation = GameMain.GameSession?.Campaign?.Map.CurrentLocation;
223  if (currLocation?.Connections == null) { return true; }
224  foreach (LocationConnection c in currLocation.Connections)
225  {
226  if (RequireBeaconStation && !c.LevelData.HasBeaconStation) { continue; }
227  if (requiredDestinationTypes.Any(t => c.OtherLocation(currLocation).Type.Identifier == t))
228  {
229  return true;
230  }
231  }
232  return false;
233  }
