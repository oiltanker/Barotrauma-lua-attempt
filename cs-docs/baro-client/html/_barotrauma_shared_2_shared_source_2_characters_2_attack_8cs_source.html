359  public Attack(ContentXElement element, string parentDebugName)
360  {
361  Deserialize(element, parentDebugName);
362 
363  if (element.GetAttribute("damage") != null ||
364  element.GetAttribute("bluntdamage") != null ||
365  element.GetAttribute("burndamage") != null ||
366  element.GetAttribute("bleedingdamage") != null)
367  {
368  DebugConsole.ThrowError("Error in Attack (" + parentDebugName + ") - Define damage as afflictions instead of using the damage attribute (e.g. <Affliction identifier=\"internaldamage\" strength=\"10\" />).");
369  }
370 
371  //if level wall damage is not defined, default to the structure damage
372  if (element.GetAttribute("LevelWallDamage") == null &&
373  element.GetAttribute("levelwalldamage") == null)
374  {
375  LevelWallDamage = StructureDamage;
376  }
377 
378  InitProjSpecific(element);
379 
380  foreach (var subElement in element.Elements())
381  {
382  switch (subElement.Name.ToString().ToLowerInvariant())
383  {
384  case "statuseffect":
385  statusEffects.Add(StatusEffect.Load(subElement, parentDebugName));
386  break;
387  case "affliction":
388  AfflictionPrefab afflictionPrefab;
389  if (subElement.GetAttribute("name") != null)
390  {
391  DebugConsole.ThrowError("Error in Attack (" + parentDebugName + ") - define afflictions using identifiers instead of names.");
392  string afflictionName = subElement.GetAttributeString("name", "").ToLowerInvariant();
393  afflictionPrefab = AfflictionPrefab.List.FirstOrDefault(ap => ap.Name.Equals(afflictionName, System.StringComparison.OrdinalIgnoreCase));
394  if (afflictionPrefab == null)
395  {
396  DebugConsole.ThrowError("Error in Attack (" + parentDebugName + ") - Affliction prefab \"" + afflictionName + "\" not found.");
397  continue;
398  }
399  }
400  else
401  {
402  string afflictionIdentifier = subElement.GetAttributeString("identifier", "").ToLowerInvariant();
403  afflictionPrefab = AfflictionPrefab.Prefabs[afflictionIdentifier];
404  if (afflictionPrefab == null)
405  {
406  DebugConsole.ThrowError("Error in Attack (" + parentDebugName + ") - Affliction prefab \"" + afflictionIdentifier + "\" not found.");
407  continue;
408  }
409  }
410  break;
411  case "conditional":
412  foreach (XAttribute attribute in subElement.Attributes())
413  {
414  if (PropertyConditional.IsValid(attribute))
415  {
416  Conditionals.Add(new PropertyConditional(attribute));
417  }
418  }
419  break;
420  }
421  }
422  }

425  public void ReloadAfflictions(XElement element, string parentDebugName)
426  {
427  Afflictions.Clear();
428  foreach (var subElement in element.GetChildElements("affliction"))
429  {
430  AfflictionPrefab afflictionPrefab;
431  Affliction affliction;
432  Identifier afflictionIdentifier = subElement.GetAttributeIdentifier("identifier", "");
433  if (!AfflictionPrefab.Prefabs.ContainsKey(afflictionIdentifier))
434  {
435  DebugConsole.ThrowError($"Error in an Attack defined in \"{parentDebugName}\" - could not find an affliction with the identifier \"{afflictionIdentifier}\".");
436  continue;
437  }
438  afflictionPrefab = AfflictionPrefab.Prefabs[afflictionIdentifier];
439  affliction = afflictionPrefab.Instantiate(0.0f);
440  affliction.Deserialize(subElement);
441  //backwards compatibility
442  if (subElement.Attribute("amount") != null && subElement.Attribute("strength") == null)
443  {
444  affliction.Strength = subElement.GetAttributeFloat("amount", 0.0f);
445  }
446  // add the affliction anyway, so that it can be shown in the editor.
447  Afflictions.Add(affliction, subElement);
448  }

469  public AttackResult DoDamage(Character attacker, IDamageable target, Vector2 worldPosition, float deltaTime, bool playSound = true, PhysicsBody sourceBody = null, Limb sourceLimb = null)
470  {
471  Character targetCharacter = target as Character;
472  if (OnlyHumans)
473  {
474  if (targetCharacter != null && !targetCharacter.IsHuman)
475  {
476  return new AttackResult();
477  }
478  }
479 
480  SetUser(attacker);
481 
482  DamageParticles(deltaTime, worldPosition);
483 
484  var attackResult = target?.AddDamage(attacker, worldPosition, this, deltaTime, playSound) ?? new AttackResult();
485  var effectType = attackResult.Damage > 0.0f ? ActionType.OnUse : ActionType.OnFailure;
486  if (targetCharacter != null && targetCharacter.IsDead)
487  {
488  effectType = ActionType.OnEating;
489  }
490 
491  foreach (StatusEffect effect in statusEffects)
492  {
493  effect.sourceBody = sourceBody;
494  if (effect.HasTargetType(StatusEffect.TargetType.This))
495  {
496  // TODO: do we want to apply the effect at the world position or the entity positions in each cases? -> go through also other cases where status effects are applied
497  effect.Apply(effectType, deltaTime, attacker, sourceLimb ?? attacker as ISerializableEntity, worldPosition);
498  }
499  if (effect.HasTargetType(StatusEffect.TargetType.Parent))
500  {
501  effect.Apply(effectType, deltaTime, attacker, attacker);
502  }
503  if (targetCharacter != null)
504  {
505  if (effect.HasTargetType(StatusEffect.TargetType.Character))
506  {
507  effect.Apply(effectType, deltaTime, targetCharacter, targetCharacter);
508  }
509  if (effect.HasTargetType(StatusEffect.TargetType.Limb))
510  {
511  effect.Apply(effectType, deltaTime, targetCharacter, attackResult.HitLimb);
512  }
513  if (effect.HasTargetType(StatusEffect.TargetType.AllLimbs))
514  {
515  effect.Apply(effectType, deltaTime, targetCharacter, targetCharacter.AnimController.Limbs.Cast<ISerializableEntity>().ToList());
516  }
517  }
518  if (target is Entity targetEntity)
519  {
520  if (effect.HasTargetType(StatusEffect.TargetType.NearbyItems) ||
521  effect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
522  {
523  targets.Clear();
524  targets.AddRange(effect.GetNearbyTargets(worldPosition, targets));
525  effect.Apply(effectType, deltaTime, targetEntity, targets);
526  }
527  if (effect.HasTargetType(StatusEffect.TargetType.UseTarget))
528  {
529  effect.Apply(effectType, deltaTime, targetEntity, attacker, worldPosition);
530  }
531  }
532  }
533 
534  return attackResult;
535  }

538  public AttackResult DoDamageToLimb(Character attacker, Limb targetLimb, Vector2 worldPosition, float deltaTime, bool playSound = true, PhysicsBody sourceBody = null, Limb sourceLimb = null)
539  {
540  if (targetLimb == null)
541  {
542  return new AttackResult();
543  }
544 
545  if (OnlyHumans)
546  {
547  if (targetLimb.character != null && !targetLimb.character.IsHuman)
548  {
549  return new AttackResult();
550  }
551  }
552 
553  SetUser(attacker);
554 
555  DamageParticles(deltaTime, worldPosition);
556 
557  var attackResult = targetLimb.character.ApplyAttack(attacker, worldPosition, this, deltaTime, playSound, targetLimb, penetration: Penetration);
558  var effectType = attackResult.Damage > 0.0f ? ActionType.OnUse : ActionType.OnFailure;
559 
560  foreach (StatusEffect effect in statusEffects)
561  {
562  effect.sourceBody = sourceBody;
563  if (effect.HasTargetType(StatusEffect.TargetType.This))
564  {
565  effect.Apply(effectType, deltaTime, attacker, sourceLimb ?? attacker as ISerializableEntity);
566  }
567  if (effect.HasTargetType(StatusEffect.TargetType.Parent))
568  {
569  effect.Apply(effectType, deltaTime, attacker, attacker);
570  }
571  if (effect.HasTargetType(StatusEffect.TargetType.Character))
572  {
573  effect.Apply(effectType, deltaTime, targetLimb.character, targetLimb.character);
574  }
575  if (effect.HasTargetType(StatusEffect.TargetType.Limb))
576  {
577  effect.Apply(effectType, deltaTime, targetLimb.character, targetLimb);
578  }
579  if (effect.HasTargetType(StatusEffect.TargetType.AllLimbs))
580  {
581  effect.Apply(effectType, deltaTime, targetLimb.character, targetLimb.character.AnimController.Limbs.Cast<ISerializableEntity>().ToList());
582  }
583  if (effect.HasTargetType(StatusEffect.TargetType.NearbyItems) ||
584  effect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
585  {
586  targets.Clear();
587  targets.AddRange(effect.GetNearbyTargets(worldPosition, targets));
588  effect.Apply(effectType, deltaTime, targetLimb.character, targets);
589  }
590  if (effect.HasTargetType(StatusEffect.TargetType.UseTarget))
591  {
592  effect.Apply(effectType, deltaTime, targetLimb.character, attacker, worldPosition);
593  }
594  }

596  return attackResult;
