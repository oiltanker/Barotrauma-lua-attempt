505  public Item FindItemAtPosition(Vector2 simPosition, float aimAssistModifier = 0.0f, Item[] ignoredItems = null)
506  {
507  if (Submarine != null)
508  {
509  simPosition += Submarine.SimPosition;
510  }
511 
512  debugInteractablesInRange.Clear();
513  debugInteractablesAtCursor.Clear();
514  debugInteractablesNearCursor.Clear();
515 
516  bool draggingItemToWorld = CharacterInventory.DraggingItemToWorld;
517 
518  //reduce the amount of aim assist if an item has been selected
519  //= can't switch selection to another item without deselecting the current one first UNLESS the cursor is directly on the item
520  //otherwise it would be too easy to accidentally switch the selected item when rewiring items
521  float aimAssistAmount = SelectedConstruction == null ? 100.0f * aimAssistModifier : 1.0f;
522 
523  Vector2 displayPosition = ConvertUnits.ToDisplayUnits(simPosition);
524 
525  //use the list of visible entities if it exists
526  var entityList = Submarine.VisibleEntities ?? Item.ItemList;
527 
528  Item closestItem = null;
529  float closestItemDistance = Math.Max(aimAssistAmount, 2.0f);
530  foreach (MapEntity entity in entityList)
531  {
532  if (!(entity is Item item))
533  {
534  continue;
535  }
536  if (item.body != null && !item.body.Enabled) { continue; }
537  if (item.ParentInventory != null) { continue; }
538  if (ignoredItems != null && ignoredItems.Contains(item)) { continue; }
539  if (item.Prefab.RequireCampaignInteract && item.CampaignInteractionType == CampaignMode.InteractionType.None) { continue; }
540  if (Screen.Selected is SubEditorScreen editor && editor.WiringMode && item.GetComponent<ConnectionPanel>() == null) { continue; }
541 
542  if (draggingItemToWorld)
543  {
544  if (item.OwnInventory == null ||
545  !item.OwnInventory.Container.AllowDragAndDrop ||
546  !item.OwnInventory.CanBePut(CharacterInventory.DraggingItems.First()) ||
547  !CanAccessInventory(item.OwnInventory))
548  {
549  continue;
550  }
551  }
552 
553  float distanceToItem = float.PositiveInfinity;
554  if (item.IsInsideTrigger(displayPosition, out Rectangle transformedTrigger))
555  {
556  debugInteractablesAtCursor.Add(item);
557  //distance is between 0-1 when the cursor is directly on the item
558  distanceToItem =
559  Math.Abs(transformedTrigger.Center.X - displayPosition.X) / transformedTrigger.Width +
560  Math.Abs((transformedTrigger.Y - transformedTrigger.Height / 2.0f) - displayPosition.Y) / transformedTrigger.Height;
561  //modify the distance based on the size of the trigger (preferring smaller items)
562  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (transformedTrigger.Width + transformedTrigger.Height) / 250.0f);
563  }
564  else if (!item.Prefab.RequireCursorInsideTrigger)
565  {
566  Rectangle itemDisplayRect = new Rectangle(item.InteractionRect.X, item.InteractionRect.Y - item.InteractionRect.Height, item.InteractionRect.Width, item.InteractionRect.Height);
567 
568  if (itemDisplayRect.Contains(displayPosition))
569  {
570  debugInteractablesAtCursor.Add(item);
571  //distance is between 0-1 when the cursor is directly on the item
572  distanceToItem =
573  Math.Abs(itemDisplayRect.Center.X - displayPosition.X) / itemDisplayRect.Width +
574  Math.Abs(itemDisplayRect.Center.Y - displayPosition.Y) / itemDisplayRect.Height;
575  //modify the distance based on the size of the item (preferring smaller ones)
576  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (itemDisplayRect.Width + itemDisplayRect.Height) / 250.0f);
577  }
578  else
579  {
580  if (closestItemDistance < 2.0f) { continue; }
581  //get the point on the itemDisplayRect which is closest to the cursor
582  Vector2 rectIntersectionPoint = new Vector2(
583  MathHelper.Clamp(displayPosition.X, itemDisplayRect.X, itemDisplayRect.Right),
584  MathHelper.Clamp(displayPosition.Y, itemDisplayRect.Y, itemDisplayRect.Bottom));
585  distanceToItem = 2.0f + Vector2.Distance(rectIntersectionPoint, displayPosition);
586  }
587  }
588 
589  if (distanceToItem > closestItemDistance) { continue; }
590  if (!CanInteractWith(item)) { continue; }
591 
592  debugInteractablesNearCursor.Add((item, 1.0f - distanceToItem / (100.0f * aimAssistModifier)));
593  closestItem = item;
594  closestItemDistance = distanceToItem;
595  }
596 
597  return closestItem;
598  }
