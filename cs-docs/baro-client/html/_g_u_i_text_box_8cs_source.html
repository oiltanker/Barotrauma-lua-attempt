321 private bool SetText(string text, bool store = true)
322 {
323 if (textFilterFunction != null)
324 {
325 text = textFilterFunction(text);
326 }
327 if (Text == text) { return false; }
328 textBlock.Text = text;
329 ClearSelection();
330 if (Text == null) textBlock.Text = "";
331 if (Text != "" && !Wrap)
332 {
333 if (maxTextLength != null)
334 {
335 if (textBlock.Text.Length > maxTextLength)
336 {
337 textBlock.Text = Text.Substring(0, (int)maxTextLength);
338 }
339 }
340 else
341 {
342 while (ClampText && textBlock.Text.Length > 0 && Font.MeasureString(textBlock.Text).X * TextBlock.TextScale > (int)(textBlock.Rect.Width - textBlock.Padding.X - textBlock.Padding.Z))
343 {
344 textBlock.Text = Text.Substring(0, textBlock.Text.Length - 1);
345 }
346 }
347 }
348 if (store)
349 {
350 memento.Store(Text);
351 }
352 return true;
353 }

395 protected override void Update(float deltaTime)
396 {
397 if (!Visible) return;
398
399 if (flashTimer > 0.0f) flashTimer -= deltaTime;
400 if (!Enabled) { return; }
401
402 if (skipUpdate)
403 {
404 skipUpdate = false;
405 return;
406 }
407
408 if (MouseRect.Contains(PlayerInput.MousePosition) && (GUI.MouseOn == null || (!(GUI.MouseOn is GUIButton) && GUI.IsMouseOn(this))))
409 {
410 State = ComponentState.Hover;
411 if (PlayerInput.PrimaryMouseButtonDown())
412 {
413 mouseHeldInside = true;
414 Select();
415 }
416 else
417 {
418 isSelecting = PlayerInput.PrimaryMouseButtonHeld();
419 }
420 if (PlayerInput.DoubleClicked())
421 {
422 SelectAll();
423 }
424 if (isSelecting)
425 {
426 if (!MathUtils.NearlyEqual(PlayerInput.MouseSpeed.X, 0))
427 {
428 CaretIndex = textBlock.GetCaretIndexFromScreenPos(PlayerInput.MousePosition);
429 CalculateCaretPos();
430 CalculateSelection();
431 }
432 }
433 }
434 else
435 {
436 if ((PlayerInput.LeftButtonClicked() || PlayerInput.RightButtonClicked()) && selected)
437 {
438 if (!mouseHeldInside) { Deselect(); }
439 mouseHeldInside = false;
440 }
441 isSelecting = false;
442 State = ComponentState.None;
443 }
444 if (!isSelecting)
445 {
446 isSelecting = PlayerInput.IsShiftDown();
447 }
448
449 if (mouseHeldInside && !PlayerInput.PrimaryMouseButtonHeld())
450 {
451 mouseHeldInside = false;
452 }
453
454 if (CaretEnabled)
455 {
456 HandleCaretBoundsOverflow();
457 caretTimer += deltaTime;
458 caretVisible = ((caretTimer * 1000.0f) % 1000) < 500;
459 if (caretVisible && caretPosDirty)
460 {
461 CalculateCaretPos();
462 }
463 }
464
465 if (GUI.KeyboardDispatcher.Subscriber == this)
466 {
467 State = ComponentState.Selected;
468 Character.DisableControls = true;
469 if (OnEnterPressed != null && PlayerInput.KeyHit(Keys.Enter))
470 {
471 OnEnterPressed(this, Text);
472 }
473 }
474 else if (Selected)
475 {
476 Deselect();
477 }
478
479 textBlock.State = State;
480 }

482 private void HandleCaretBoundsOverflow()
483 {
484 if (textBlock.OverflowClipActive)
485 {
486 CalculateCaretPos();
487 float left = textBlock.Rect.X + textBlock.Padding.X;
488 if (CaretScreenPos.X < left)
489 {
490 float diff = left - CaretScreenPos.X;
491 textBlock.TextPos = new Vector2(textBlock.TextPos.X + diff, textBlock.TextPos.Y);
492 CalculateCaretPos();
493 }
494
495 float right = textBlock.Rect.Right - textBlock.Padding.Z;
496 if (CaretScreenPos.X > right)
497 {
498 float diff = CaretScreenPos.X - right;
499 textBlock.TextPos = new Vector2(textBlock.TextPos.X - diff, textBlock.TextPos.Y);
500 CalculateCaretPos();
501 }
502 }

523 private void DrawSelectionRect(SpriteBatch spriteBatch)
524 {
525 var characterPositions = textBlock.GetAllCaretPositions();
526 (int startIndex, int endIndex) = IsLeftToRight
527 ? (selectionStartIndex, selectionEndIndex)
528 : (selectionEndIndex, selectionStartIndex);
529 endIndex--;
530
531 void drawRect(Vector2 topLeft, Vector2 bottomRight)
532 {
533 int minWidth = GUI.IntScale(5);
534 if (bottomRight.X - topLeft.X < minWidth) { bottomRight.X = topLeft.X + minWidth; }
535 GUI.DrawRectangle(spriteBatch,
536 Rect.Location.ToVector2() + topLeft,
537 bottomRight - topLeft,
538 SelectionColor, isFilled: true);
539 }
540
541 Vector2 topLeft = characterPositions[startIndex];
542 for (int i = startIndex+1; i <= endIndex; i++)
543 {
544 Vector2 currPos = characterPositions[i];
545 if (!MathUtils.NearlyEqual(topLeft.Y, currPos.Y))
546 {
547 Vector2 bottomRight = characterPositions[i - 1];
548 bottomRight += Font.MeasureChar(Text[i - 1]) * TextBlock.TextScale;
549 drawRect(topLeft, bottomRight);
550 topLeft = currPos;
551 }
552 }
553 Vector2 finalBottomRight = characterPositions[endIndex];
554 if (Text.Length > endIndex)
555 {
556 finalBottomRight += Font.MeasureChar(Text[endIndex]) * TextBlock.TextScale;
557 }
558 drawRect(topLeft, finalBottomRight);
559 }

604 public void ReceiveCommandInput(char command)
605 {
606 if (Text == null) Text = "";
607
608 // Prevent alt gr from triggering any of these as that combination is often needed for special characters
609 if (PlayerInput.IsAltDown()) return;
610
611 switch (command)
612 {
613 case '\b' when !Readonly: //backspace
614 {
615 using var _ = new TextPosPreservation(this);
616 if (PlayerInput.KeyDown(Keys.LeftControl) || PlayerInput.KeyDown(Keys.RightControl))
617 {
618 SetText(string.Empty, false);
619 CaretIndex = Text.Length;
620 }
621 else if (selectedCharacters > 0)
622 {
623 RemoveSelectedText();
624 }
625 else if (Text.Length > 0 && CaretIndex > 0)
626 {
627 CaretIndex--;
628 SetText(Text.Remove(CaretIndex, 1));
629 CalculateCaretPos();
630 ClearSelection();
631 }
632 OnTextChanged?.Invoke(this, Text);
633 break;
634 }
635 case (char)0x3: // ctrl-c
636 CopySelectedText();
637 break;
638 case (char)0x16 when !Readonly: // ctrl-v
639 string text = GetCopiedText();
640 RemoveSelectedText();
641 if (SetText(Text.Insert(CaretIndex, text)))
642 {
643 CaretIndex = Math.Min(Text.Length, CaretIndex + text.Length);
644 OnTextChanged?.Invoke(this, Text);
645 }
646 break;
647 case (char)0x18: // ctrl-x
648 CopySelectedText();
649 if (!Readonly)
650 {
651 RemoveSelectedText();
652 }
653 break;
654 case (char)0x1: // ctrl-a
655 if (PlayerInput.IsCtrlDown())
656 {
657 SelectAll();
658 }
659 break;
660 case (char)0x1A when !Readonly && !SubEditorScreen.IsSubEditor(): // ctrl-z
661 text = memento.Undo();
662 if (text != Text)
663 {
664 ClearSelection();
665 SetText(text, false);
666 CaretIndex = Text.Length;
667 OnTextChanged?.Invoke(this, Text);
668 }
669 break;
670 case (char)0x12 when !Readonly && !SubEditorScreen.IsSubEditor(): // ctrl-r
671 text = memento.Redo();
672 if (text != Text)
673 {
674 ClearSelection();
675 SetText(text, false);
676 CaretIndex = Text.Length;
677 OnTextChanged?.Invoke(this, Text);
678 }
679 break;
680 }
681 }

683 public void ReceiveSpecialInput(Keys key)
684 {
685 switch (key)
686 {
687 case Keys.Left:
688 if (isSelecting)
689 {
690 InitSelectionStart();
691 }
692 CaretIndex = Math.Max(CaretIndex - 1, 0);
693 caretTimer = 0;
694 HandleSelection();
695 break;
696 case Keys.Right:
697 if (isSelecting)
698 {
699 InitSelectionStart();
700 }
701 CaretIndex = Math.Min(CaretIndex + 1, Text.Length);
702 caretTimer = 0;
703 HandleSelection();
704 break;
705 case Keys.Up:
706 if (isSelecting)
707 {
708 InitSelectionStart();
709 }
710 float lineHeight = Font.LineHeight * TextBlock.TextScale;
711 int newIndex = textBlock.GetCaretIndexFromLocalPos(new Vector2(caretPos.X, caretPos.Y - lineHeight * 0.5f));
712 textBlock.Font.WrapText(
713 textBlock.Text.SanitizedValue,
714 GetWrapWidth(),
715 newIndex,
716 out Vector2 requestedCharPos);
717 requestedCharPos *= TextBlock.TextScale;
718 if (MathUtils.NearlyEqual(requestedCharPos.Y, caretPos.Y)) { newIndex = 0; }
719 CaretIndex = newIndex;
720 caretTimer = 0;
721 HandleSelection();
722 break;
723 case Keys.Down:
724 if (isSelecting)
725 {
726 InitSelectionStart();
727 }
728 lineHeight = Font.LineHeight * TextBlock.TextScale;
729 newIndex = textBlock.GetCaretIndexFromLocalPos(new Vector2(caretPos.X, caretPos.Y + lineHeight * 1.5f));
730 textBlock.Font.WrapText(
731 textBlock.Text.SanitizedValue,
732 GetWrapWidth(),
733 newIndex,
734 out Vector2 requestedCharPos2);
735 requestedCharPos2 *= TextBlock.TextScale;
736 if (MathUtils.NearlyEqual(requestedCharPos2.Y, caretPos.Y)) { newIndex = Text.Length; }
737 CaretIndex = newIndex;
738 caretTimer = 0;
739 HandleSelection();
740 break;
741 case Keys.Delete when !Readonly:
742 if (selectedCharacters > 0)
743 {
744 RemoveSelectedText();
745 }
746 else if (Text.Length > 0 && CaretIndex < Text.Length)
747 {
748 SetText(Text.Remove(CaretIndex, 1));
749 OnTextChanged?.Invoke(this, Text);
750 caretPosDirty = true;
751 }
752 break;
753 case Keys.Tab:
754 // Select the next text box.
755 var editor = RectTransform.GetParents().Select(p => p.GUIComponent as SerializableEntityEditor).FirstOrDefault(e => e != null);
756 if (editor == null) { break; }
757 var allTextBoxes = GetAndSortTextBoxes(editor).ToList();
758 if (allTextBoxes.Any())
759 {
760 int currentIndex = allTextBoxes.IndexOf(this);
761 int nextIndex = Math.Min(allTextBoxes.Count - 1, currentIndex + 1);
762 var next = allTextBoxes[nextIndex];
763 if (next != this)
764 {
765 next.Select();
766 next.Flash(Color.White * 0.5f, 0.5f);
767 }
768 else
769 {
770 // Select the first text box in the next editor that has text boxes.
771 var listBox = RectTransform.GetParents().Select(p => p.GUIComponent as GUIListBox).FirstOrDefault(lb => lb != null);
772 if (listBox == null) { break; }
773 // TODO: The get's out of focus if the selection is out of view.
774 // Not sure how's that possible, but it seems to work when the auto scroll is disabled and you handle the scrolling manually.
775 listBox.SelectNext();
776 while (SelectNextTextBox(listBox) == null)
777 {
778 var previous = listBox.SelectedComponent;
779 listBox.SelectNext();
780 if (listBox.SelectedComponent == previous) { break; }
781 }
782 }
783 }
784 IEnumerable<GUITextBox> GetAndSortTextBoxes(GUIComponent parent) => parent.GetAllChildren<GUITextBox>().OrderBy(t => t.Rect.Y).ThenBy(t => t.Rect.X);
785 GUITextBox SelectNextTextBox(GUIListBox listBox)
786 {
787 var textBoxes = GetAndSortTextBoxes(listBox.SelectedComponent);
788 if (textBoxes.Any())
789 {
790 var next = textBoxes.First();
791 next.Select();
792 next.Flash(Color.White * 0.5f, 0.5f);
793 return next;
794 }
795 return null;
796 }
797 break;
798 }
799 if (caretPosDirty) { CalculateCaretPos(); }

845 private void RemoveSelectedText()
846 {
847 if (selectedText.Length == 0) { return; }
848
849 int targetCaretIndex = Math.Max(0, Math.Min(selectionEndIndex, Math.Min(selectionStartIndex, Text.Length - 1)));
