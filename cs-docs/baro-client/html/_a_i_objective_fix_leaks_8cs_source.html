22  public static float GetLeakSeverity(Gap leak)
23  {
24  if (leak == null) { return 0; }
25  float sizeFactor = MathHelper.Lerp(1, 10, MathUtils.InverseLerp(0, 200, leak.Size));
26  float severity = sizeFactor * leak.Open;
27  if (!leak.IsRoomToRoom)
28  {
29  severity *= 10;
30  // If there is a leak in the outer walls, the severity cannot be lower than 10, no matter how small the leak
31  return MathHelper.Clamp(severity, 10, 100);
32  }
33  else
34  {
35  return MathHelper.Min(severity, 100);
36  }
37  }

39  protected override float TargetEvaluation()
40  {
41  int totalLeaks = Targets.Count();
42  if (totalLeaks == 0) { return 0; }
43  int otherFixers = HumanAIController.CountCrew(c => c != HumanAIController && c.ObjectiveManager.IsCurrentObjective<AIObjectiveFixLeaks>() && !c.Character.IsIncapacitated && c.Character.Submarine == character.Submarine, onlyBots: true);
44  bool anyFixers = otherFixers > 0;
45  if (objectiveManager.IsOrder(this))
46  {
47  float ratio = anyFixers ? totalLeaks / (float)otherFixers : 1;
48  return Targets.Sum(t => GetLeakSeverity(t)) * ratio;
49  }
50  else
51  {
52  int secondaryLeaks = Targets.Count(l => l.IsRoomToRoom);
53  int leaks = totalLeaks - secondaryLeaks;
54  float ratio = leaks == 0 ? 1 : anyFixers ? leaks / (float)otherFixers : 1;
55  if (anyFixers && (ratio <= 1 || otherFixers > 5 || otherFixers / (float)HumanAIController.CountCrew(onlyBots: true) > 0.75f))
56  {
57  // Enough fixers
58  return 0;
59  }
60  return Targets.Sum(t => GetLeakSeverity(t)) * ratio;
61  }
62  }

71  public static bool IsValidTarget(Gap gap, Character character)
72  {
73  if (gap == null) { return false; }
74  // Don't fix a leak on a wall section set to be ignored
75  if (gap.ConnectedWall != null)
76  {
77  if (gap.ConnectedWall.Sections.Any(s => s.gap == gap && s.IgnoreByAI(character))) { return false; }
78  if (gap.ConnectedWall.MaxHealth <= 0.0f) { return false; }
79  }
80  if (gap.ConnectedWall == null || gap.ConnectedDoor != null || gap.Open <= 0 || gap.linkedTo.All(l => l == null)) { return false; }
81  if (gap.Submarine == null || character.Submarine == null) { return false; }
82  // Don't allow going into another sub, unless it's connected and of the same team and type.
83  if (!character.Submarine.IsEntityFoundOnThisSub(gap, includingConnectedSubs: true)) { return false; }
84  return true;
