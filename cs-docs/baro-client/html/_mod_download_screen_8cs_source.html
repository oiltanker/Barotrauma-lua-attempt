265 public override void Update(double deltaTime)
266 {
267 base.Update(deltaTime);
268 if (GameMain.Client is null) { return; }
269 if (!confirmDownload) { return; }
270 if (currentDownload is null)
271 {
272 if (pendingDownloads.TryDequeue(out currentDownload))
273 {
274 GameMain.Client.RequestFile(FileTransferType.Mod, currentDownload.Name, currentDownload.Hash.StringRepresentation);
275 }
276 else
277 {
278 var serverPackages = GameMain.Client.ClientPeer.ServerContentPackages;
279 CorePackage corePackage
280 = downloadedPackages.FirstOrDefault(p => p is CorePackage) as CorePackage
281 ?? serverPackages.FirstOrDefault(p => p.CorePackage != null)
282 ?.CorePackage
283 ?? throw new Exception($"Failed to find core package to enable");
284 RegularPackage[] regularPackages
285 = serverPackages.Where(p => p.CorePackage is null)
286 .Select(p =>
287 p.RegularPackage
288 ?? downloadedPackages.FirstOrDefault(d => d is RegularPackage && d.Hash.Equals(p.Hash))
289 ?? throw new Exception($"Could not find regular package \"{p.Name}\""))
290 .Cast<RegularPackage>()
291 .ToArray();
292 foreach (var regularPackage in regularPackages)
293 {
294 DebugConsole.NewMessage($"Enabling \"{regularPackage.Name}\" ({regularPackage.Dir})", Color.Lime);
295 }
296
297 ContentPackageManager.EnabledPackages.BackUp();
298 ContentPackageManager.EnabledPackages.SetCore(corePackage);
299 ContentPackageManager.EnabledPackages.SetRegular(regularPackages);
300
301 GameMain.NetLobbyScreen.Select();
302 GameMain.LuaCs.Initialize();
303 }
304 }

307 public void CurrentDownloadFinished(FileReceiver.FileTransferIn transfer)
308 {
309 if (currentDownload is null) { throw new Exception("Current download is null"); }
310
311 string path = transfer.FilePath;
312 if (!path.EndsWith(ModReceiver.Extension, StringComparison.OrdinalIgnoreCase))
313 {
314 return;
315 }
316 string dir = path.RemoveFromEnd(ModReceiver.Extension, StringComparison.OrdinalIgnoreCase);
317
318 SaveUtil.DecompressToDirectory(path, dir, file => { });
319 ContentPackage newPackage
320 = ContentPackage.TryLoad($"{dir}/{ContentPackage.FileListFileName}")
321 ?? throw new Exception($"Failed to load downloaded mod \"{currentDownload.Name}\"");
322 if (!currentDownload.Hash.Equals(newPackage.Hash))
323 {
324 throw new Exception($"Hash mismatch for downloaded mod \"{currentDownload.Name}\" (expected {currentDownload.Hash}, got {newPackage.Hash})");
325 }
326 downloadedPackages.Add(newPackage);
327
328 currentDownload = null;
329
330 }
