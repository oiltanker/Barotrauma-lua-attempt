121  if (!itemsToDestroy.Any())
122  {
123  DebugConsole.ThrowError($"Error in mission \"{Prefab.Identifier}\". Could not find an item with the tag \"{itemTag}\".");
124  }
125  else
126  {
127  items.AddRange(itemsToDestroy);

129  }
130 
131  if (itemConfig != null && !IsClient)
132  {
133  foreach (XElement element in itemConfig.Elements())
134  {
135  string itemIdentifier = element.GetAttributeString("identifier", "");
136  if (!(MapEntityPrefab.Find(null, itemIdentifier) is ItemPrefab itemPrefab))
137  {
138  DebugConsole.ThrowError("Couldn't spawn item for outpost destroy mission: item prefab \"" + itemIdentifier + "\" not found");
139  continue;
140  }
141 
142  Identifier[] moduleFlags = element.GetAttributeIdentifierArray("moduleflags", null);
143  Identifier[] spawnPointTags = element.GetAttributeIdentifierArray("spawnpointtags", null);
144  ISpatialEntity spawnPoint = SpawnAction.GetSpawnPos(
145  SpawnAction.SpawnLocationType.Outpost, SpawnType.Human | SpawnType.Enemy,
146  moduleFlags, spawnPointTags, element.GetAttributeBool("asfaraspossible", false));
147  if (spawnPoint == null)

156  wp.CurrentHull.WorldRect.Y - wp.CurrentHull.Rect.Height + 16.0f);
157  }
158  var item = new Item(itemPrefab, spawnPos, null);

281  protected override void UpdateMissionSpecific(float deltaTime)
282  {
283  if (State != HostagesKilledState)
284  {
285  if (requireRescue.Any(r => r.Removed || r.IsDead))
286  {
287  State = HostagesKilledState;
288  return;
289  }
290  }
291  else
292  {
293  endTimer += deltaTime;
294  if (endTimer > EndDelay)
295  {
296 #if SERVER
297  if (!(GameMain.GameSession.GameMode is CampaignMode) && GameMain.Server != null)
298  {
299  GameMain.Server.EndGame();
300  }
301 #endif
302  }
303  }
304 
305  switch (state)
306  {
307  case 0:
308 
309  if (items.All(it => it.Removed || it.Condition <= 0.0f) &&
310  requireKill.All(c => c.Removed || c.IsDead || (c.LockHands && c.Submarine == Submarine.MainSub)) &&
311  requireRescue.All(c => c.Submarine?.Info.Type == SubmarineType.Player))
312  {
313  State = 1;
314  }
315  break;
316 #if SERVER
317  case 1:
318  if (!(GameMain.GameSession.GameMode is CampaignMode) && GameMain.Server != null)
319  {
320  if (!Submarine.MainSub.AtStartExit || (wasDocked && !Submarine.MainSub.DockedTo.Contains(Level.Loaded.StartOutpost)))
321  {
322  GameMain.Server.EndGame();
323  State = 2;
324  }
325  }
326  break;
327 #endif
328  }
