308  public void Shoot(Character user, Vector2 weaponPos, Vector2 spawnPos, float rotation, List<Body> ignoredBodies, bool createNetworkEvent, float damageMultiplier = 1f)
309  {
310  //add the limbs of the shooter to the list of bodies to be ignored
311  //so that the player can't shoot himself
312  IgnoredBodies = ignoredBodies;
313  Vector2 projectilePos = weaponPos;
314  //make sure there's no obstacles between the base of the weapon (or the shoulder of the character) and the end of the barrel
315  if (Submarine.PickBody(weaponPos, spawnPos, IgnoredBodies, Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionItemBlocking,
316  customPredicate: (Fixture f) => { return !IgnoredBodies.Contains(f.Body); }) == null)
317  {
318  //no obstacles -> we can spawn the projectile at the barrel
319  projectilePos = spawnPos;
320  }
321  else if ((weaponPos - spawnPos).LengthSquared() > 0.0001f)
322  {
323  //spawn the projectile body.GetMaxExtent() away from the position where the raycast hit the obstacle
324  Vector2 newPos = weaponPos - Vector2.Normalize(spawnPos - projectilePos) * Math.Max(Item.body.GetMaxExtent(), 0.1f);
325  if (MathUtils.IsValid(newPos))
326  {
327  projectilePos = newPos;
328  }
329  }
330  Launch(user, projectilePos, rotation, damageMultiplier);
331  if (createNetworkEvent && !Item.Removed && GameMain.NetworkMember != null && GameMain.NetworkMember.IsServer)
332  {
333 #if SERVER
334  launchRot = rotation;
335  Item.CreateServerEvent(this, new EventData(launch: true));
336 #endif
337  }

340  public bool Use(Character character = null)
341  {
342  if (character != null && !characterUsable) { return false; }
343 
344  for (int i = 0; i < HitScanCount; i++)
345  {
346  float launchAngle;
347 
348  if (StaticSpread)
349  {
350  float staticSpread = Spread / (HitScanCount - 1);
351  // because the position of the item changes as hitscan are fired, we will set an
352  // initial offset on the first hitscan and then increase the item's angle by a set amount as hitscan are fired
353  float offset = i == 0 ? -staticSpread * (HitScanCount -1) : 0f;
354  launchAngle = item.body.Rotation + MathHelper.ToRadians(staticSpread + offset);
355  }
356  else
357  {
358  launchAngle = item.body.Rotation + MathHelper.ToRadians(Spread * Rand.Range(-0.5f, 0.5f));
359  }
360 
361  Vector2 launchDir = new Vector2((float)Math.Cos(launchAngle), (float)Math.Sin(launchAngle));
362  if (Hitscan)
363  {
364  Vector2 prevSimpos = item.SimPosition;
365  item.body.SetTransformIgnoreContacts(item.body.SimPosition, launchAngle);
366  DoHitscan(launchDir);
367  if (i < HitScanCount - 1)
368  {
369  item.SetTransform(prevSimpos, item.body.Rotation);
370  }
371  }
372  else
373  {
374  item.body.SetTransform(item.body.SimPosition, launchAngle);
375  float modifiedLaunchImpulse = LaunchImpulse * (1 + Rand.Range(-ImpulseSpread, ImpulseSpread));
376  DoLaunch(launchDir * modifiedLaunchImpulse * item.body.Mass);
377  System.Diagnostics.Debug.WriteLine("launch: " + modifiedLaunchImpulse + " - " + item.body.LinearVelocity);
378  }
379  }
380  User = character;
381  return true;
382  }

418  private void DoHitscan(Vector2 dir)
419  {
420  float rotation = item.body.Rotation;
421  Vector2 simPositon = item.SimPosition;
422  Vector2 rayStartWorld = item.WorldPosition;
423  item.Drop(null);
424 
425  item.body.Enabled = true;
426  //set the velocity of the body because the OnProjectileCollision method
427  //uses it to determine the direction from which the projectile hit
428  item.body.LinearVelocity = dir;
429  IsActive = true;
430 
431  Vector2 rayStart = simPositon;
432  Vector2 rayEnd = rayStart + dir * 500.0f;
433 
434  float worldDist = 1000.0f;
435 #if CLIENT
436  worldDist = Screen.Selected?.Cam?.WorldView.Width ?? GameMain.GraphicsWidth;
437 #endif
438  Vector2 rayEndWorld = rayStartWorld + dir * worldDist;
439 
440  List<HitscanResult> hits = new List<HitscanResult>();
441 
442  hits.AddRange(DoRayCast(rayStart, rayEnd, submarine: item.Submarine));
443 
444  if (item.Submarine != null)
445  {
446  //shooting indoors, do a hitscan outside as well
447  hits.AddRange(DoRayCast(rayStart + item.Submarine.SimPosition, rayEnd + item.Submarine.SimPosition, submarine: null));
448  //also in the coordinate space of docked subs
449  foreach (Submarine dockedSub in item.Submarine.DockedTo)
450  {
451  if (dockedSub == item.Submarine) { continue; }
452  hits.AddRange(DoRayCast(rayStart + item.Submarine.SimPosition - dockedSub.SimPosition, rayEnd + item.Submarine.SimPosition - dockedSub.SimPosition, dockedSub));
453  }
454  }
455  else
456  {
457  //shooting outdoors, see if we can hit anything inside a sub
458  foreach (Submarine submarine in Submarine.Loaded)
459  {
460  var inSubHits = DoRayCast(rayStart - submarine.SimPosition, rayEnd - submarine.SimPosition, submarine);
461  //transform back to world coordinates
462  for (int i = 0; i < inSubHits.Count; i++)
463  {
464  inSubHits[i] = new HitscanResult(
465  inSubHits[i].Fixture,
466  inSubHits[i].Point + submarine.SimPosition,
467  inSubHits[i].Normal,
468  inSubHits[i].Fraction);
469  }
470 
471  hits.AddRange(inSubHits);
472  }
473  }
474 
475  int hitCount = 0;
476  Vector2 lastHitPos = item.WorldPosition;
477  hits = hits.OrderBy(h => h.Fraction).ToList();
478  for (int i = 0; i < hits.Count; i++)
479  {
480  var h = hits[i];
481  item.SetTransform(h.Point, rotation);
482  if (HandleProjectileCollision(h.Fixture, h.Normal, Vector2.Zero))
483  {
484  hitCount++;
485  if (hitCount >= MaxTargetsToHit || i == hits.Count - 1)
486  {
487  LaunchProjSpecific(rayStartWorld, item.WorldPosition);
488  break;
489  }
490  }
491  }
492  //the raycast didn't hit anything (or didn't hit enough targets to stop the projectile) -> the projectile flew somewhere outside the level and is permanently lost
493  if (hitCount < MaxTargetsToHit)
494  {
495  item.body.SetTransformIgnoreContacts(item.body.SimPosition, rotation);
496  LaunchProjSpecific(rayStartWorld, rayEndWorld);
497  if (Entity.Spawner == null)
498  {
499  item.Remove();
500  }
501  else
502  {
503  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
504  {
505  //clients aren't allowed to remove items by themselves, so lets hide the projectile until the server tells us to remove it
506  item.HiddenInGame = Hitscan;
507  }
508  else
509  {
510  Entity.Spawner.AddItemToRemoveQueue(item);
511  }
512  }
513  }
514  }

516  private List<HitscanResult> DoRayCast(Vector2 rayStart, Vector2 rayEnd, Submarine submarine)
517  {
518  List<HitscanResult> hits = new List<HitscanResult>();
519 
520  Vector2 dir = rayEnd - rayStart;
521  dir = dir.LengthSquared() < 0.00001f ? Vector2.UnitY : Vector2.Normalize(dir);
522 
523  //do an AABB query first to see if the start of the ray is inside a fixture
524  var aabb = new FarseerPhysics.Collision.AABB(rayStart - Vector2.One * 0.001f, rayStart + Vector2.One * 0.001f);
525  GameMain.World.QueryAABB((fixture) =>
526  {
527  if (fixture?.Body.UserData is LevelObject levelObj)
528  {
529  if (!levelObj.Prefab.TakeLevelWallDamage) { return true; }
530  }
531  else if (fixture?.Body == null || fixture.IsSensor)
532  {
533  //ignore sensors and items
534  return true;
535  }
536  if (fixture.Body.UserData is VineTile) { return true; }
537  if (fixture.Body.UserData is Item item && (item.GetComponent<Door>() == null && !item.Prefab.DamagedByProjectiles || item.Condition <= 0)) { return true; }
538  if (fixture.Body.UserData as string == "ruinroom" || fixture.Body.UserData is Hull || fixture.UserData is Hull) { return true; }
539 
540  //if doing the raycast in a submarine's coordinate space, ignore anything that's not in that sub
541  if (submarine != null)
542  {
543  if (fixture.Body.UserData is VoronoiCell) { return true; }
544  if (fixture.Body.UserData is Entity entity && entity.Submarine != submarine) { return true; }
545  }
546 
547  //ignore everything else than characters, sub walls and level walls
548  if (!fixture.CollisionCategories.HasFlag(Physics.CollisionCharacter) &&
549  !fixture.CollisionCategories.HasFlag(Physics.CollisionWall) &&
550  !fixture.CollisionCategories.HasFlag(Physics.CollisionLevel)) { return true; }
551 
552  if (fixture.Body.UserData is VoronoiCell && (this.item.Submarine != null || submarine != null)) { return true; }
553 
554  fixture.Body.GetTransform(out FarseerPhysics.Common.Transform transform);
555  if (!fixture.Shape.TestPoint(ref transform, ref rayStart)) { return true; }
556 
557  hits.Add(new HitscanResult(fixture, rayStart, -dir, 0.0f));
558  return true;
559  }, ref aabb);
560 
561  GameMain.World.RayCast((fixture, point, normal, fraction) =>
562  {
563  //ignore sensors and items
564  if (fixture?.Body.UserData is LevelObject levelObj)
565  {
566  if (!levelObj.Prefab.TakeLevelWallDamage) { return -1; }
567  }
568  else if (fixture?.Body == null || fixture.IsSensor)
569  {
570  //ignore sensors and items
571  return -1;
572  }
573  if (fixture.Body.UserData is VineTile) { return -1; }
574 
575  if (fixture.Body.UserData is Item item && (item.GetComponent<Door>() == null && !item.Prefab.DamagedByProjectiles || item.Condition <= 0)) { return -1; }
576  if (fixture.Body.UserData as string == "ruinroom" || fixture.Body?.UserData is Hull || fixture.UserData is Hull) { return -1; }
577  if (!(fixture.Body.UserData is Holdable holdable && holdable.CanPush))
578  {
579  //ignore everything else than characters, sub walls and level walls
580  if (!fixture.CollisionCategories.HasFlag(Physics.CollisionCharacter) &&
581  !fixture.CollisionCategories.HasFlag(Physics.CollisionWall) &&
582  !fixture.CollisionCategories.HasFlag(Physics.CollisionLevel)) { return -1; }
583  }
584 
585  //if doing the raycast in a submarine's coordinate space, ignore anything that's not in that sub
586  if (submarine != null)
587  {
588  if (fixture.Body.UserData is VoronoiCell) { return -1; }
589  if (fixture.Body.UserData is Entity entity && entity.Submarine != submarine) { return -1; }
590  if (fixture.Body.UserData is Limb limb && limb.character?.Submarine != submarine) { return -1; }
591  }
592 
593  //ignore level cells if the item and the point of impact are inside a sub
594  if (fixture.Body.UserData is VoronoiCell)
595  {
596  if (Hull.FindHull(ConvertUnits.ToDisplayUnits(point), this.item.CurrentHull) != null && this.item.Submarine != null)
597  {
598  return -1;
599  }
600  }
601 
602  if (hits.Count > 50)
603  {
604  float furthestHit = 0.0f;
605  int furthestHitIndex = -1;
606  for (int i = 0; i < hits.Count; i++)
607  {
608  if (hits[i].Fraction > furthestHit)
609  {
610  furthestHitIndex = i;
611  furthestHit = hits[i].Fraction;
612  }
613  }
614  if (furthestHitIndex > -1)
615  {
616  hits.RemoveAt(furthestHitIndex);
617  }
618  }
619 
620  hits.Add(new HitscanResult(fixture, point, normal, fraction));
621 
622  return 1;
623  }, rayStart, rayEnd, Physics.CollisionCharacter | Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionItemBlocking);
624 
625  return hits;
626  }

638  public override void Update(float deltaTime, Camera cam)
639  {
640  if (DeactivationTime > 0)
641  {
642  deactivationTimer -= deltaTime;
643  if (deactivationTimer < 0)
644  {
645  DisableProjectileCollisions();
646  }
647  }
648  while (impactQueue.Count > 0)
649  {
650  var impact = impactQueue.Dequeue();
651  HandleProjectileCollision(impact.Fixture, impact.Normal, impact.LinearVelocity);
652  }
653 
654  if (!removePending)
655  {
656  Entity useTarget = lastTarget?.Body.UserData is Limb limb ? limb.character : lastTarget?.Body.UserData as Entity;
657  ApplyStatusEffects(ActionType.OnActive, deltaTime, useTarget: useTarget, user: _user);
658  }
659 
660  if (item.body != null && item.body.FarseerBody.IsBullet)
661  {
662  if (item.body.LinearVelocity.LengthSquared() < ContinuousCollisionThreshold * ContinuousCollisionThreshold)
663  {
664  item.body.FarseerBody.IsBullet = false;
665  }
666  }
667  //projectiles with a stickjoint don't become inactive until the stickjoint is detached
668  if (stickJoint == null && !item.body.FarseerBody.IsBullet)
669  {
670  IsActive = false;
671  if (DeactivationTime > 0 && deactivationTimer > 0)
672  {
673  DisableProjectileCollisions();
674  }
675  }
676 
677  if (stickJoint == null) { return; }
678 
679  if (StickDuration > 0 && stickTimer > 0)
680  {
681  stickTimer -= deltaTime;
682  return;
683  }
684 
685  float absoluteMaxTranslation = 100;
686  // Update the item's transform to make sure it's inside the same sub as the target (or outside)
687  if (StickTarget?.UserData is Limb target && target.Submarine != item.Submarine || stickJoint is PrismaticJoint prismaticJoint && Math.Abs(prismaticJoint.JointTranslation) > absoluteMaxTranslation)
688  {
689  item.UpdateTransform();
690  }
691 
692  if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
693  {
694  if (StickTargetRemoved() || stickJoint is PrismaticJoint pJoint && Math.Abs(pJoint.JointTranslation) > maxJointTranslationInSimUnits)
695  {
