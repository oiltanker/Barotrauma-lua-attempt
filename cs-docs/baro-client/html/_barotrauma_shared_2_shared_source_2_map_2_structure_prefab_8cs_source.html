144

150  $"EntityName.{fallbackNameIdentifier}");
151 
152  if (parentType == "wrecked")
153  {
154  Name = TextManager.GetWithVariable("wreckeditemformat", "[name]", Name);
155  }
156  if (!string.IsNullOrEmpty(OriginalName))
157  {
158  Name = Name.Fallback(OriginalName);
159  }
160

162  string joinedTags = element.GetAttributeString("tags", "");
163  if (string.IsNullOrEmpty(joinedTags)) joinedTags = element.GetAttributeString("Tags", "");
164  foreach (string tag in joinedTags.Split(','))
165  {
166  tags.Add(tag.Trim().ToIdentifier());
167  }
168 
169  if (element.GetAttribute("ishorizontal") != null)
170  {
171  IsHorizontal = element.GetAttributeBool("ishorizontal", false);
172  }
173 
174 #if CLIENT
175  var decorativeSprites = new List<DecorativeSprite>();
176  var decorativeSpriteGroups = new Dictionary<int, List<DecorativeSprite>>();
177 #endif
178  foreach (var subElement in element.Elements())
179  {
180  switch (subElement.Name.ToString())
181  {
182  case "sprite":
183  Sprite = new Sprite(subElement, lazyLoad: true);
184  if (subElement.GetAttribute("sourcerect") == null &&
185  subElement.GetAttribute("sheetindex") == null)
186  {
187  DebugConsole.ThrowError("Warning - sprite sourcerect not configured for structure \"" + Name + "\"!");
188  }

192  if (subElement.GetAttributeBool("flipvertical", false))
193  Sprite.effects = SpriteEffects.FlipVertically;
194 #endif
195  CanSpriteFlipX = subElement.GetAttributeBool("canflipx", true);
196  CanSpriteFlipY = subElement.GetAttributeBool("canflipy", true);
197 
198  if (subElement.GetAttribute("name") == null && !Name.IsNullOrWhiteSpace())
199  {
200  Sprite.Name = Name.Value;
201  }
202  Sprite.EntityIdentifier = Identifier;
203  break;
204  case "backgroundsprite":
205  BackgroundSprite = new Sprite(subElement, lazyLoad: true);
206  if (subElement.GetAttribute("sourcerect") == null && Sprite != null)
207  {
208  BackgroundSprite.SourceRect = Sprite.SourceRect;
209  BackgroundSprite.size = Sprite.size;
210  BackgroundSprite.size.X *= Sprite.SourceRect.Width;
211  BackgroundSprite.size.Y *= Sprite.SourceRect.Height;
212  BackgroundSprite.RelativeOrigin = subElement.GetAttributeVector2("origin", new Vector2(0.5f, 0.5f));
213  }
214 #if CLIENT
215  if (subElement.GetAttributeBool("fliphorizontal", false)) { BackgroundSprite.effects = SpriteEffects.FlipHorizontally; }
216  if (subElement.GetAttributeBool("flipvertical", false)) { BackgroundSprite.effects = SpriteEffects.FlipVertically; }
217  BackgroundSpriteColor = subElement.GetAttributeColor("color", Color.White);
218 #endif
219  break;
220  case "decorativesprite":
221 #if CLIENT
222  string decorativeSpriteFolder = "";
223  if (subElement.DoesAttributeReferenceFileNameAlone("texture"))
224  {
225  decorativeSpriteFolder = Path.GetDirectoryName(file.Path);
226  }
227 
228  int groupID = 0;
229  DecorativeSprite decorativeSprite = null;
230  if (subElement.GetAttribute("texture") == null)
231  {
232  groupID = subElement.GetAttributeInt("randomgroupid", 0);
233  }
234  else
235  {
236  decorativeSprite = new DecorativeSprite(subElement, decorativeSpriteFolder, lazyLoad: true);
237  decorativeSprites.Add(decorativeSprite);
238  groupID = decorativeSprite.RandomGroupID;
239  }
240  if (!decorativeSpriteGroups.ContainsKey(groupID))
241  {
242  decorativeSpriteGroups.Add(groupID, new List<DecorativeSprite>());
243  }
244  decorativeSpriteGroups[groupID].Add(decorativeSprite);
245 #endif
246  break;
247  }
248  }
249 #if CLIENT
250  DecorativeSprites = decorativeSprites.ToImmutableArray();
251  DecorativeSpriteGroups = decorativeSpriteGroups.Select(kvp => (kvp.Key, kvp.Value.ToImmutableArray())).ToImmutableDictionary();
252 #endif

255  if (!Enum.TryParse(categoryStr, true, out MapEntityCategory category))
256  {
257  category = MapEntityCategory.Structure;
258  }
259  Category = category;
260 
261  Aliases =
262  (element.GetAttributeStringArray("aliases", null, convertToLowerInvariant: true) ??
263  element.GetAttributeStringArray("Aliases", Array.Empty<string>(), convertToLowerInvariant: true)).ToImmutableHashSet();
264

266  Aliases.Add(nonTranslatedName.ToLowerInvariant());

269  if (Body)

272  }

275  {
