102 public override void Draw(double deltaTime, GraphicsDevice graphics, SpriteBatch spriteBatch)
103 {
104 cam.UpdateTransform(true);
105 Submarine.CullEntities(cam);
106
107 foreach (Character c in Character.CharacterList)
108 {
109 c.AnimController.Limbs.ForEach(l => l.body.UpdateDrawPosition());
110 bool wasVisible = c.IsVisible;
111 c.DoVisibilityCheck(cam);
112 if (c.IsVisible != wasVisible)
113 {
114 c.AnimController.Limbs.ForEach(l =>
115 {
116 if (l.LightSource != null)
117 {
118 l.LightSource.Enabled = c.IsVisible;
119 }
120 });
121 }
122 }
123
124 Stopwatch sw = new Stopwatch();
125 sw.Start();
126
127 DrawMap(graphics, spriteBatch, deltaTime);
128
129 sw.Stop();
130 GameMain.PerformanceCounter.AddElapsedTicks("DrawMap", sw.ElapsedTicks);
131 sw.Restart();
132
133 spriteBatch.Begin(SpriteSortMode.Deferred, null, GUI.SamplerState, null, GameMain.ScissorTestEnable);
134
135 if (Character.Controlled != null && cam != null) { Character.Controlled.DrawHUD(spriteBatch, cam); }
136
137 if (GameMain.GameSession != null) { GameMain.GameSession.Draw(spriteBatch); }
138
139 if (Character.Controlled == null && !GUI.DisableHUD)
140 {
141 for (int i = 0; i < Submarine.MainSubs.Length; i++)
142 {
143 if (Submarine.MainSubs[i] == null) continue;
144 if (Level.Loaded != null && Submarine.MainSubs[i].WorldPosition.Y < Level.MaxEntityDepth) { continue; }
145
146 Vector2 position = Submarine.MainSubs[i].SubBody != null ? Submarine.MainSubs[i].WorldPosition : Submarine.MainSubs[i].HiddenSubPosition;
147
148 Color indicatorColor = i == 0 ? Color.LightBlue * 0.5f : GUIStyle.Red * 0.5f;
149 GUI.DrawIndicator(
150 spriteBatch, position, cam,
151 Math.Max(Submarine.MainSub.Borders.Width, Submarine.MainSub.Borders.Height),
152 GUIStyle.SubmarineLocationIcon.Value.Sprite, indicatorColor);
153 }
154 }
155
156 if (!GUI.DisableHUD)
157 {
158 foreach (Character c in Character.CharacterList)
159 {
160 c.DrawGUIMessages(spriteBatch, cam);
161 }
162 }
163
164 GUI.Draw(cam, spriteBatch);
165
166 spriteBatch.End();
167
168 sw.Stop();
169 GameMain.PerformanceCounter.AddElapsedTicks("DrawHUD", sw.ElapsedTicks);
170 sw.Restart();
171 }

173 public void DrawMap(GraphicsDevice graphics, SpriteBatch spriteBatch, double deltaTime)
174 {
175 foreach (Submarine sub in Submarine.Loaded)
176 {
177 sub.UpdateTransform();
178 }
179
180 GameMain.ParticleManager.UpdateTransforms();
181
182 GameMain.LightManager.ObstructVision =
183 Character.Controlled != null &&
184 Character.Controlled.ObstructVision &&
185 (Character.Controlled.ViewTarget == Character.Controlled || Character.Controlled.ViewTarget == null);
186
187 GameMain.LightManager.UpdateObstructVision(graphics, spriteBatch, cam, Character.Controlled?.CursorWorldPosition ?? Vector2.Zero);

189 //------------------------------------------------------------------------
190 graphics.SetRenderTarget(renderTarget);
191 graphics.Clear(Color.Transparent);
192 //Draw background structures and wall background sprites
193 //(= the background texture that's revealed when a wall is destroyed) into the background render target
194 //These will be visible through the LOS effect.
195 spriteBatch.Begin(SpriteSortMode.BackToFront, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
196 Submarine.DrawBack(spriteBatch, false, e => e is Structure s && (e.SpriteDepth >= 0.9f || s.Prefab.BackgroundSprite != null));
197 Submarine.DrawPaintedColors(spriteBatch, false);
198 spriteBatch.End();
199
200 graphics.SetRenderTarget(null);
201 GameMain.LightManager.RenderLightMap(graphics, spriteBatch, cam, renderTarget);
202
203 //------------------------------------------------------------------------
204 graphics.SetRenderTarget(renderTargetBackground);
205 if (Level.Loaded == null)
206 {
207 graphics.Clear(new Color(11, 18, 26, 255));
208 }
209 else
210 {
211 //graphics.Clear(new Color(255, 255, 255, 255));
212 Level.Loaded.DrawBack(graphics, spriteBatch, cam);

214
215 //draw alpha blended particles that are in water and behind subs
216#if LINUX || OSX
217 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
218#else
219 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
220#endif
221 GameMain.ParticleManager.Draw(spriteBatch, true, false, Particles.ParticleBlendState.AlphaBlend);
222 spriteBatch.End();
223
224 //draw additive particles that are in water and behind subs
225 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, null, DepthStencilState.None, null, null, cam.Transform);
226 GameMain.ParticleManager.Draw(spriteBatch, true, false, Particles.ParticleBlendState.Additive);
227 spriteBatch.End();
228 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.None);
229 spriteBatch.Draw(renderTarget, new Rectangle(0, 0, GameMain.GraphicsWidth, GameMain.GraphicsHeight), Color.White);
230 spriteBatch.End();
231
232 //----------------------------------------------------------------------------
233
234 //Start drawing to the normal render target (stuff that can't be seen through the LOS effect)
235 graphics.SetRenderTarget(renderTarget);
236
237 graphics.BlendState = BlendState.NonPremultiplied;
238 graphics.SamplerStates[0] = SamplerState.LinearWrap;
239 Quad.UseBasicEffect(renderTargetBackground);
240 Quad.Render();
241
242 //Draw the rest of the structures, characters and front structures
243 spriteBatch.Begin(SpriteSortMode.BackToFront, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
244 Submarine.DrawBack(spriteBatch, false, e => !(e is Structure) || e.SpriteDepth < 0.9f);
245 foreach (Character c in Character.CharacterList)
246 {
247 if (!c.IsVisible || c.AnimController.Limbs.Any(l => l.DeformSprite != null)) { continue; }
248 c.Draw(spriteBatch, Cam);

250 spriteBatch.End();

252 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
253 DrawDeformed(firstPass: true);
254 DrawDeformed(firstPass: false);
255 spriteBatch.End();

257 void DrawDeformed(bool firstPass)
258 {
259 //backwards order to render the most recently spawned characters in front (characters spawned later have a larger sprite depth)
260 for (int i = Character.CharacterList.Count - 1; i >= 0; i--)
261 {
262 Character c = Character.CharacterList[i];
263 if (!c.IsVisible) { continue; }
264 if (c.Params.DrawLast == firstPass) { continue; }
265 if (c.AnimController.Limbs.All(l => l.DeformSprite == null)) { continue; }
266 c.Draw(spriteBatch, Cam);
267 }
268 }
269
270 Level.Loaded?.DrawFront(spriteBatch, cam);
271
272 //draw the rendertarget and particles that are only supposed to be drawn in water into renderTargetWater
273 graphics.SetRenderTarget(renderTargetWater);
274
275 graphics.BlendState = BlendState.Opaque;
276 graphics.SamplerStates[0] = SamplerState.LinearWrap;
277 Quad.UseBasicEffect(renderTarget);
278 Quad.Render();
279
280 //draw alpha blended particles that are inside a sub
281 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.DepthRead, null, null, cam.Transform);
282 GameMain.ParticleManager.Draw(spriteBatch, true, true, Particles.ParticleBlendState.AlphaBlend);
283 spriteBatch.End();
284
285 graphics.SetRenderTarget(renderTarget);
286
287 //draw alpha blended particles that are not in water
288 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, null, DepthStencilState.DepthRead, null, null, cam.Transform);
289 GameMain.ParticleManager.Draw(spriteBatch, false, null, Particles.ParticleBlendState.AlphaBlend);
290 spriteBatch.End();
291
292 //draw additive particles that are not in water
293 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, null, DepthStencilState.None, null, null, cam.Transform);
294 GameMain.ParticleManager.Draw(spriteBatch, false, null, Particles.ParticleBlendState.Additive);
295 spriteBatch.End();
296
297 graphics.DepthStencilState = DepthStencilState.DepthRead;
298 graphics.SetRenderTarget(renderTargetFinal);
299
300 WaterRenderer.Instance.ResetBuffers();
301 Hull.UpdateVertices(cam, WaterRenderer.Instance);
302 WaterRenderer.Instance.RenderWater(spriteBatch, renderTargetWater, cam);
303 WaterRenderer.Instance.RenderAir(graphics, cam, renderTarget, Cam.ShaderTransform);
304 graphics.DepthStencilState = DepthStencilState.None;
305
306 spriteBatch.Begin(SpriteSortMode.Immediate,
307 BlendState.NonPremultiplied, SamplerState.LinearWrap,
308 null, null,
309 damageEffect,
310 cam.Transform);
311 Submarine.DrawDamageable(spriteBatch, damageEffect, false);
312 spriteBatch.End();
313
314 spriteBatch.Begin(SpriteSortMode.BackToFront, BlendState.NonPremultiplied, null, DepthStencilState.None, null, null, cam.Transform);
315 Submarine.DrawFront(spriteBatch, false, null);
316 spriteBatch.End();
317
318 //draw additive particles that are inside a sub
319 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, null, DepthStencilState.Default, null, null, cam.Transform);
320 GameMain.ParticleManager.Draw(spriteBatch, true, true, Particles.ParticleBlendState.Additive);
321 foreach (var discharger in Items.Components.ElectricalDischarger.List)
322 {
323 discharger.DrawElectricity(spriteBatch);
324 }
325 spriteBatch.End();
326 if (GameMain.LightManager.LightingEnabled)
327 {
328 graphics.DepthStencilState = DepthStencilState.None;
329 graphics.SamplerStates[0] = SamplerState.LinearWrap;
330 graphics.BlendState = Lights.CustomBlendStates.Multiplicative;
331 Quad.UseBasicEffect(GameMain.LightManager.LightMap);
332 Quad.Render();
333 }
334
335 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, SamplerState.LinearWrap, DepthStencilState.None, null, null, cam.Transform);
336 foreach (Character c in Character.CharacterList)
337 {
338 c.DrawFront(spriteBatch, cam);
339 }
340
341 Level.Loaded?.DrawDebugOverlay(spriteBatch, cam);
342 if (GameMain.DebugDraw)
343 {
344 MapEntity.mapEntityList.ForEach(me => me.AiTarget?.Draw(spriteBatch));
345 Character.CharacterList.ForEach(c => c.AiTarget?.Draw(spriteBatch));
346 if (GameMain.GameSession?.EventManager != null)
347 {
348 GameMain.GameSession.EventManager.DebugDraw(spriteBatch);
349 }
350 }
351 spriteBatch.End();
352
353 if (GameMain.LightManager.LosEnabled && GameMain.LightManager.LosMode != LosMode.None && Lights.LightManager.ViewTarget != null)
354 {
355 GameMain.LightManager.LosEffect.CurrentTechnique = GameMain.LightManager.LosEffect.Techniques["LosShader"];
356
357 GameMain.LightManager.LosEffect.Parameters["xTexture"].SetValue(renderTargetBackground);
358 GameMain.LightManager.LosEffect.Parameters["xLosTexture"].SetValue(GameMain.LightManager.LosTexture);
359 GameMain.LightManager.LosEffect.Parameters["xLosAlpha"].SetValue(GameMain.LightManager.LosAlpha);
360
361 Color losColor;
362 if (GameMain.LightManager.LosMode == LosMode.Transparent)
363 {
364 //convert the los color to HLS and make sure the luminance of the color is always the same
365 //as the luminance of the ambient light color
366 float r = Character.Controlled?.CharacterHealth == null ?
367 0.0f : Math.Min(Character.Controlled.CharacterHealth.DamageOverlayTimer * 0.5f, 0.5f);
368 Vector3 ambientLightHls = GameMain.LightManager.AmbientLight.RgbToHLS();
369 Vector3 losColorHls = Color.Lerp(GameMain.LightManager.AmbientLight, Color.Red, r).RgbToHLS();
370 losColorHls.Y = ambientLightHls.Y;
371 losColor = ToolBox.HLSToRGB(losColorHls);
372 }
373 else
374 {
375 losColor = Color.Black;
376 }
377
378 GameMain.LightManager.LosEffect.Parameters["xColor"].SetValue(losColor.ToVector4());
379
380 graphics.BlendState = BlendState.NonPremultiplied;
381 graphics.SamplerStates[0] = SamplerState.PointClamp;
382 GameMain.LightManager.LosEffect.CurrentTechnique.Passes[0].Apply();
383 Quad.Render();
384 }
385
386 if (Character.Controlled is { } character)
387 {
388 float grainStrength = character.GrainStrength;
389 Rectangle screenRect = new Rectangle(0, 0, GameMain.GraphicsWidth, GameMain.GraphicsHeight);
390 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, effect: GrainEffect);
391 GUI.DrawRectangle(spriteBatch, screenRect, Color.White, isFilled: true);
392 GrainEffect.Parameters["seed"].SetValue(Rand.Range(0f, 1f, Rand.RandSync.Unsynced));
393 GrainEffect.Parameters["intensity"].SetValue(grainStrength);
394 GrainEffect.Parameters["grainColor"].SetValue(character.GrainColor.ToVector4());
395 spriteBatch.End();
396 }
397
398 graphics.SetRenderTarget(null);
399
400 float BlurStrength = 0.0f;
401 float DistortStrength = 0.0f;
402 Vector3 chromaticAberrationStrength = GameSettings.CurrentConfig.Graphics.ChromaticAberration ?
403 new Vector3(-0.02f, -0.01f, 0.0f) : Vector3.Zero;
404
405 if (Character.Controlled != null)
406 {
407 BlurStrength = Character.Controlled.BlurStrength * 0.005f;
408 DistortStrength = Character.Controlled.DistortStrength;
409 if (GameSettings.CurrentConfig.Graphics.RadialDistortion)
410 {
411 chromaticAberrationStrength -= Vector3.One * Character.Controlled.RadialDistortStrength;
412 }
413 chromaticAberrationStrength += new Vector3(-0.03f, -0.015f, 0.0f) * Character.Controlled.ChromaticAberrationStrength;
414 }
415 else
416 {
417 BlurStrength = 0.0f;
418 DistortStrength = 0.0f;
419 }
420
421 string postProcessTechnique = "";
422 if (BlurStrength > 0.0f)
423 {
424 postProcessTechnique += "Blur";
425 PostProcessEffect.Parameters["blurDistance"].SetValue(BlurStrength);
426 }
427 if (chromaticAberrationStrength != Vector3.Zero)
428 {
429 postProcessTechnique += "ChromaticAberration";
430 PostProcessEffect.Parameters["chromaticAberrationStrength"].SetValue(chromaticAberrationStrength);
431 }
432 if (DistortStrength > 0.0f)
433 {
434 postProcessTechnique += "Distort";
435 PostProcessEffect.Parameters["distortScale"].SetValue(Vector2.One * DistortStrength);
436 PostProcessEffect.Parameters["distortUvOffset"].SetValue(WaterRenderer.Instance.WavePos * 0.001f);
437 }
438
439 graphics.BlendState = BlendState.Opaque;
440 graphics.SamplerStates[0] = SamplerState.LinearClamp;
441 graphics.DepthStencilState = DepthStencilState.None;
442 if (string.IsNullOrEmpty(postProcessTechnique))
443 {
444 Quad.UseBasicEffect(renderTargetFinal);
445 }
446 else
447 {
448 PostProcessEffect.Parameters["MatrixTransform"].SetValue(Matrix.Identity);
449 PostProcessEffect.Parameters["xTexture"].SetValue(renderTargetFinal);
450 PostProcessEffect.CurrentTechnique = PostProcessEffect.Techniques[postProcessTechnique];
451 PostProcessEffect.CurrentTechnique.Passes[0].Apply();
452 }
453 Quad.Render();
454
455 if (fadeToBlackState > 0.0f)
456 {
457 spriteBatch.Begin(SpriteSortMode.Deferred);
458 GUI.DrawRectangle(spriteBatch, new Rectangle(0, 0, GameMain.GraphicsWidth, GameMain.GraphicsHeight), Color.Lerp(Color.TransparentBlack, Color.Black, fadeToBlackState), isFilled: true);
459 spriteBatch.End();
460 }
