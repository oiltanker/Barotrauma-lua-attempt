33 protected override float TargetEvaluation()
34 {
35 if (Targets.None()) { return 0; }
36 if (objectiveManager.IsOrder(this))
37 {
38 float prio = objectiveManager.GetOrderPriority(this);
39 if (subObjectives.All(so => so.SubObjectives.None()))
40 {
41 // If none of the subobjectives have subobjectives, no valid container was found. Don't allow running.
42 ForceWalk = true;
43 }
44 return prio;
45 }
46 return AIObjectiveManager.RunPriority - 0.5f;
47 }

49 protected override bool Filter(Item target)
50 {
51 // If the target was selected as a valid target, we'll have to accept it so that the objective can be completed.
52 // The validity changes when a character picks the item up.
53 if (!IsValidTarget(target, character, checkInventory: true)) { return Objectives.ContainsKey(target) && IsItemInsideValidSubmarine(target, character); }
54 if (target.CurrentHull.FireSources.Count > 0) { return false; }
55 // Don't clean up items in rooms that have enemies inside.
56 if (Character.CharacterList.Any(c => c.CurrentHull == target.CurrentHull && !HumanAIController.IsFriendly(c) && HumanAIController.IsActive(c))) { return false; }
57 return true;
58 }

71 public static bool IsItemInsideValidSubmarine(Item item, Character character)
72 {
73 if (item.CurrentHull == null) { return false; }
74 if (item.Submarine == null) { return false; }
75 if (item.Submarine.TeamID != character.TeamID) { return false; }
76 if (character.Submarine != null && !character.Submarine.IsConnectedTo(item.Submarine)) { return false; }
77 return true;
78 }

89 public static bool IsValidTarget(Item item, Character character, bool checkInventory, bool allowUnloading = true)
90 {
91 if (item == null) { return false; }
92 if (!item.HasAccess(character)) { return false; }
93 if ((item.SpawnedInCurrentOutpost && !item.AllowStealing) == character.IsOnPlayerTeam) { return false; }
94 if (item.ParentInventory != null)
95 {
96 if (item.Container == null)
97 {
98 // In a character inventory
99 return false;
100 }
101 if (!IsValidContainer(item.Container, character, allowUnloading)) { return false; }
102 }
103 if (character != null && !IsItemInsideValidSubmarine(item, character)) { return false; }
104 if (item.HasBallastFloraInHull) { return false; }
105 var pickable = item.GetComponent<Pickable>();
106 if (pickable == null) { return false; }
107 if (pickable is Holdable h && h.Attachable && h.Attached) { return false; }
108 var wire = item.GetComponent<Wire>();
109 if (wire != null)
110 {
111 if (wire.Connections.Any()) { return false; }
112 }
113 else
114 {
115 var connectionPanel = item.GetComponent<ConnectionPanel>();
116 if (connectionPanel != null && connectionPanel.Connections.Any(c => c.Wires.Any(w => w != null)))
117 {
118 return false;
119 }
120 }
121 if (item.Prefab.PreferredContainers.None())
122 {
123 return false;
124 }
125 if (!checkInventory)
126 {
127 return true;
128 }
129 return CanEquip(character, item);
130 }
