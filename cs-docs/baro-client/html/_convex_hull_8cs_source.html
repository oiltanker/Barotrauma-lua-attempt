221  private void MergeOverlappingSegments(ConvexHull ch)
222  {
223  if (ch == this) { return; }
224 
225  if (isHorizontal == ch.isHorizontal)
226  {
227  //hide segments that are roughly at the some position as some other segment (e.g. the ends of two adjacent wall pieces)
228  float mergeDist = 16;
229  float mergeDistSqr = mergeDist * mergeDist;
230 
231  Rectangle intersection = Rectangle.Intersect(BoundingBox, ch.BoundingBox);
232  int intersectionArea = intersection.Width * intersection.Height;
233  int bboxArea = BoundingBox.Width * BoundingBox.Height;
234  int otherBboxArea = ch.BoundingBox.Width * ch.BoundingBox.Height;
235  if (Math.Abs(intersectionArea - bboxArea) < mergeDistSqr) { return; }
236  if (Math.Abs(intersectionArea - otherBboxArea) < mergeDistSqr) { return; }
237 
238  for (int i = 0; i < segments.Length; i++)
239  {
240  for (int j = 0; j < ch.segments.Length; j++)
241  {
242  if (segments[i].IsHorizontal != ch.segments[j].IsHorizontal) { continue; }
243  if (ignoreEdge[i] || ch.ignoreEdge[j]) { continue; }
244 
245  //the segments must be at different sides of the convex hulls to be merged
246  //(e.g. the right edge of a wall piece and the left edge of another one)
247  var segment1Center = (segments[i].Start.Pos + segments[i].End.Pos) / 2.0f;
248  var segment2Center = (ch.segments[j].Start.Pos + ch.segments[j].End.Pos) / 2.0f;
249  if (Vector2.Dot(segment1Center - BoundingBox.Center.ToVector2(), segment2Center - ch.BoundingBox.Center.ToVector2()) > 0) { continue; }
250 
251  if (Vector2.DistanceSquared(segments[i].Start.Pos, ch.segments[j].Start.Pos) < mergeDistSqr &&
252  Vector2.DistanceSquared(segments[i].End.Pos, ch.segments[j].End.Pos) < mergeDistSqr)
253  {
254  ignoreEdge[i] = true;
255  ch.ignoreEdge[j] = true;
256  MergeSegments(segments[i], ch.segments[j], true);
257  }
258  else if (Vector2.DistanceSquared(segments[i].Start.Pos, ch.segments[j].End.Pos) < mergeDistSqr &&
259  Vector2.DistanceSquared(segments[i].End.Pos, ch.segments[j].Start.Pos) < mergeDistSqr)
260  {
261  ignoreEdge[i] = true;
262  ch.ignoreEdge[j] = true;
263  MergeSegments(segments[i], ch.segments[j], false);
264  }
265  }
266  }
267  }
268 
269  for (int i = 0; i < segments.Length; i++)
270  {
271  if (ignoreEdge[i]) { continue; }
272  if (Vector2.DistanceSquared(segments[i].Start.Pos, segments[i].End.Pos) < 1.0f) { continue; }
273  for (int j = 0; j < ch.segments.Length; j++)
274  {
275  if (ch.ignoreEdge[j]) { continue; }
276  if (Vector2.DistanceSquared(ch.segments[j].Start.Pos, ch.segments[j].End.Pos) < 1.0f) { continue; }
277  if (IsSegmentAInB(segments[i], ch.segments[j]))
278  {
279  ignoreEdge[i] = true;
280  if (Vector2.DistanceSquared(ch.segments[j].Start.Pos, segments[i].Start.Pos) < 4.0f)
281  {
282  ch.ShiftSegmentPoint(j, false, segments[i].End.Pos);
283  }
284  else if (Vector2.DistanceSquared(ch.segments[j].Start.Pos, segments[i].End.Pos) < 4.0f)
285  {
286  ch.ShiftSegmentPoint(j, false, segments[i].Start.Pos);
287  }
288 
289  if (Vector2.DistanceSquared(ch.segments[j].End.Pos, segments[i].Start.Pos) < 4.0f)
290  {
291  ch.ShiftSegmentPoint(j, true, segments[i].End.Pos);
292  }
293  else if (Vector2.DistanceSquared(ch.segments[j].End.Pos, segments[i].End.Pos) < 4.0f)
294  {
295  ch.ShiftSegmentPoint(j, true, segments[i].Start.Pos);
296  }
297  }
298  else if (IsSegmentAInB(ch.segments[j], segments[i]))
299  {
300  ch.ignoreEdge[j] = true;
301 
302  if (Vector2.DistanceSquared(segments[i].Start.Pos, ch.segments[j].Start.Pos) < 4.0f)
303  {
304  ShiftSegmentPoint(i, false, ch.segments[j].End.Pos);
305  }
306  else if (Vector2.DistanceSquared(segments[i].Start.Pos, ch.segments[j].End.Pos) < 4.0f)
307  {
308  ShiftSegmentPoint(i, false, ch.segments[j].Start.Pos);
309  }
310 
311  if (Vector2.DistanceSquared(segments[i].End.Pos, ch.segments[j].Start.Pos) < 4.0f)
312  {
313  ShiftSegmentPoint(i, true, ch.segments[j].End.Pos);
314  }
315  else if (Vector2.DistanceSquared(segments[i].End.Pos, ch.segments[j].End.Pos) < 4.0f)
316  {
317  ShiftSegmentPoint(i, true, ch.segments[j].Start.Pos);
318  }
319  }
320  }
321  }
322 
323  //ignore edges that are inside some other convex hull
324  for (int i = 0; i < vertices.Length; i++)
325  {
326  if (ch.IsPointInside(vertices[i].Pos))
327  {
328  if (ch.IsPointInside(vertices[(i + 1) % vertices.Length].Pos))
329  {
330  ignoreEdge[i] = true;
331  overlappingHulls.Add(ch);
332  }
333  }
334  }

357  public bool IsSegmentAInB(Segment a, Segment b)
358  {
359  if (Vector2.DistanceSquared(a.Start.Pos, a.End.Pos) > Vector2.DistanceSquared(b.Start.Pos, b.End.Pos))
360  {
361  return false;
362  }
363 
364  Vector2 min = new Vector2(Math.Min(b.Start.Pos.X, b.End.Pos.X), Math.Min(b.Start.Pos.Y, b.End.Pos.Y));
365  Vector2 max = new Vector2(Math.Max(b.Start.Pos.X, b.End.Pos.X), Math.Max(b.Start.Pos.Y, b.End.Pos.Y));
366  min.X -= 1.0f; min.Y -= 1.0f;
367  max.X += 1.0f; max.Y += 1.0f;
368 
369  if (a.Start.Pos.X < min.X) { return false; }
370  if (a.Start.Pos.Y < min.Y) { return false; }
371  if (a.End.Pos.X < min.X) { return false; }
372  if (a.End.Pos.Y < min.Y) { return false; }
373 
374  if (a.Start.Pos.X > max.X) { return false; }
375  if (a.Start.Pos.Y > max.Y) { return false; }
376  if (a.End.Pos.X > max.X) { return false; }
377  if (a.End.Pos.Y > max.Y) { return false; }
378 
379  float startDist = MathUtils.LineToPointDistanceSquared(b.Start.Pos, b.End.Pos, a.Start.Pos);
380  if (startDist > 1.0f) { return false; }
381  float endDist = MathUtils.LineToPointDistanceSquared(b.Start.Pos, b.End.Pos, a.End.Pos);
382  if (endDist > 1.0f) { return false; }
383  return true;
384  }

386  public bool IsPointInside(Vector2 point)
387  {
388  if (!BoundingBox.Contains(point)) { return false; }
389 
390  Vector2 center = (vertices[0].Pos + vertices[1].Pos + vertices[2].Pos + vertices[3].Pos) * 0.25f;
391  for (int i = 0; i < 4; i++)
392  {
393  Vector2 segmentVector = vertices[(i + 1) % 4].Pos - vertices[i].Pos;
394  Vector2 centerToVertex = center - vertices[i].Pos;
395  Vector2 pointToVertex = point - vertices[i].Pos;
396 
397  float dotCenter = Vector2.Dot(centerToVertex, segmentVector);
398  float dotPoint = Vector2.Dot(pointToVertex, segmentVector);
399 
400  if ((dotCenter > 0f && dotPoint < 0f) || (dotCenter < 0f && dotPoint > 0f)) { return false; }
401  }
402 
403  return true;
404  }

406  private void MergeSegments(Segment segment1, Segment segment2, bool startPointsMatch)
407  {
408  int startPointIndex = -1, endPointIndex = -1;
409  for (int i = 0; i < vertices.Length; i++)
410  {
411  if (vertices[i].Pos.NearlyEquals(segment1.Start.Pos))
412  startPointIndex = i;
413  else if (vertices[i].Pos.NearlyEquals(segment1.End.Pos))
414  endPointIndex = i;
415  }
416  if (startPointIndex == -1 || endPointIndex == -1) { return; }
417 
418  int startPoint2Index = -1, endPoint2Index = -1;
419  for (int i = 0; i < segment2.ConvexHull.vertices.Length; i++)
420  {
421  if (segment2.ConvexHull.vertices[i].Pos.NearlyEquals(segment2.Start.Pos))
422  startPoint2Index = i;
423  else if (segment2.ConvexHull.vertices[i].Pos.NearlyEquals(segment2.End.Pos))
424  endPoint2Index = i;
425  }
426  if (startPoint2Index == -1 || endPoint2Index == -1) { return; }
427 
428  if (startPointsMatch)
429  {
430  losVertices[startPointIndex].Pos = segment2.ConvexHull.losVertices[startPoint2Index].Pos =
431  (segment1.Start.Pos + segment2.Start.Pos) / 2.0f;
432  losVertices[endPointIndex].Pos = segment2.ConvexHull.losVertices[endPoint2Index].Pos =
433  (segment1.End.Pos + segment2.End.Pos) / 2.0f;
434  }
435  else
436  {
437  if (Vector2.DistanceSquared(losVertices[startPointIndex].Pos, segment1.Start.Pos) <
438  Vector2.DistanceSquared(losVertices[startPointIndex].Pos, segment1.End.Pos))
439  {
440  losVertices[startPointIndex].Pos = segment2.ConvexHull.losVertices[startPoint2Index].Pos =
441  (segment1.Start.Pos + segment2.End.Pos) / 2.0f;
442  losVertices[endPointIndex].Pos = segment2.ConvexHull.losVertices[endPoint2Index].Pos =
443  (segment1.End.Pos + segment2.Start.Pos) / 2.0f;
444  }
445  else
446  {
447  losVertices[startPointIndex].Pos = segment2.ConvexHull.losVertices[startPoint2Index].Pos =
448  (segment1.End.Pos + segment2.Start.Pos) / 2.0f;
449  losVertices[endPointIndex].Pos = segment2.ConvexHull.losVertices[endPoint2Index].Pos =
450  (segment1.Start.Pos + segment2.End.Pos) / 2.0f;
451  }
452  }
453 
454  overlappingHulls.Add(segment2.ConvexHull);
455  segment2.ConvexHull.overlappingHulls.Add(this);
456  }

544  public void SetVertices(Vector2[] points, Matrix? rotationMatrix = null)
545  {
546  Debug.Assert(points.Length == 4, "Only rectangular convex hulls are supported");
547 
548  LastVertexChangeTime = (float)Timing.TotalTime;
549 
550  for (int i = 0; i < 4; i++)
551  {
552  vertices[i] = new SegmentPoint(points[i], this);
553  losVertices[i] = new SegmentPoint(points[i], this);
554  losOffsets[i] = null;
555  }
556 
557  for (int i = 0; i < 4; i++)
558  {
559  ignoreEdge[i] = false;
560  }
561 
562  overlappingHulls.Clear();
563 
564  int margin = 0;
565  if (Math.Abs(points[0].X - points[2].X) < Math.Abs(points[0].Y - points[2].Y))
566  {
567  losVertices[0].Pos = new Vector2(points[0].X + margin, points[0].Y);
568  losVertices[1].Pos = new Vector2(points[1].X + margin, points[1].Y);
569  losVertices[2].Pos = new Vector2(points[2].X - margin, points[2].Y);
570  losVertices[3].Pos = new Vector2(points[3].X - margin, points[3].Y);
571  }
572  else
573  {
574  losVertices[0].Pos = new Vector2(points[0].X, points[0].Y + margin);
575  losVertices[1].Pos = new Vector2(points[1].X, points[1].Y - margin);
576  losVertices[2].Pos = new Vector2(points[2].X, points[2].Y - margin);
577  losVertices[3].Pos = new Vector2(points[3].X, points[3].Y + margin);
578  }
579 
580  if (rotationMatrix.HasValue)
581  {
582  for (int i = 0; i < vertices.Length; i++)
583  {
584  vertices[i].Pos = Vector2.Transform(vertices[i].Pos, rotationMatrix.Value);
585  losVertices[i].Pos = Vector2.Transform(losVertices[i].Pos, rotationMatrix.Value);
586  }
587  }
588  for (int i = 0; i < 4; i++)
589  {
590  segments[i] = new Segment(vertices[i], vertices[(i + 1) % 4], this);
591  }
592 
593  CalculateDimensions();
594 
595  if (ParentEntity == null) { return; }
596 
597  var chList = HullLists.Find(h => h.Submarine == ParentEntity.Submarine);
598  if (chList != null)
599  {
600  overlappingHulls.Clear();
601  foreach (ConvexHull ch in chList.List)
602  {
603  MergeOverlappingSegments(ch);
604  }
605  }

836  public static List<ConvexHull> GetHullsInRange(Vector2 position, float range, Submarine ParentSub)
837  {
838  List<ConvexHull> list = new List<ConvexHull>();
839 
840  foreach (ConvexHullList chList in HullLists)
841  {
842  Vector2 lightPos = position;
843  if (ParentSub == null)
844  {
845  //light and the convexhull are both outside
846  if (chList.Submarine == null)
847  {
848  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
849  }
850  //light is outside, convexhull inside a sub
851  else
852  {
853  Rectangle subBorders = chList.Submarine.Borders;
854  subBorders.Y -= chList.Submarine.Borders.Height;
855  if (!MathUtils.CircleIntersectsRectangle(lightPos - chList.Submarine.WorldPosition, range, subBorders)) { continue; }
856 
857  lightPos -= chList.Submarine.WorldPosition - chList.Submarine.HiddenSubPosition;
858 
859  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
860  }
861  }
862  else
863  {
864  //light is inside, convexhull outside
865  if (chList.Submarine == null)
866  {
867  continue;
868  }
869  //light and convexhull are both inside the same sub
870  if (chList.Submarine == ParentSub)
871  {
872  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
873  }
874  //light and convexhull are inside different subs
875  else
876  {
877  lightPos -= (chList.Submarine.Position - ParentSub.Position);
878 
879  Rectangle subBorders = chList.Submarine.Borders;
880  subBorders.Location += chList.Submarine.HiddenSubPosition.ToPoint() - new Point(0, chList.Submarine.Borders.Height);
881 
882  if (!MathUtils.CircleIntersectsRectangle(lightPos, range, subBorders)) continue;
883 
884  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
885  }
886  }
