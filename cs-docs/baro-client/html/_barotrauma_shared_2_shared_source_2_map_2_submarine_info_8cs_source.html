288 public void Reload()
289 {
290 XDocument doc = null;
291 int maxLoadRetries = 4;
292 for (int i = 0; i <= maxLoadRetries; i++)
293 {
294 doc = OpenFile(FilePath, out Exception e);
295 if (e != null && !(e is System.IO.IOException)) { break; }
296 if (doc != null || i == maxLoadRetries || !File.Exists(FilePath)) { break; }
297 DebugConsole.NewMessage("Opening submarine file \"" + FilePath + "\" failed, retrying in 250 ms...");
298 Thread.Sleep(250);
299 }
300 if (doc?.Root == null)
301 {
302 IsFileCorrupted = true;
303 return;
304 }
305 if (hash == null)
306 {
307 StartHashDocTask(doc);
308 }
309 SubmarineElement = doc.Root;
310 }

445 public void CheckSubsLeftBehind(XElement element = null)
446 {
447 if (element == null) { element = SubmarineElement; }
448
449 subsLeftBehind = false;
450 LeftBehindSubDockingPortOccupied = false;
451 LeftBehindDockingPortIDs.Clear();
452 BlockedDockingPortIDs.Clear();
453 foreach (var subElement in element.Elements())
454 {
455 if (!subElement.Name.ToString().Equals("linkedsubmarine", StringComparison.OrdinalIgnoreCase)) { continue; }
456 if (subElement.Attribute("location") == null) { continue; }
457
458 subsLeftBehind = true;
459 ushort targetDockingPortID = (ushort)subElement.GetAttributeInt("originallinkedto", 0);
460 LeftBehindDockingPortIDs.Add(targetDockingPortID);
461 XElement targetPortElement = targetDockingPortID == 0 ? null :
462 element.Elements().FirstOrDefault(e => e.GetAttributeInt("ID", 0) == targetDockingPortID);
463 if (targetPortElement != null && targetPortElement.GetAttributeIntArray("linked", Array.Empty<int>()).Length > 0)
464 {
465 BlockedDockingPortIDs.Add(targetDockingPortID);
466 LeftBehindSubDockingPortOccupied = true;
467 }
468 }

474 public float GetRealWorldCrushDepth()
475 {
476 if (SubmarineElement == null) { return Level.DefaultRealWorldCrushDepth; }
477 bool structureCrushDepthsDefined = false;
478 float realWorldCrushDepth = float.PositiveInfinity;
479 foreach (var structureElement in SubmarineElement.GetChildElements("structure"))
480 {
481 string name = structureElement.Attribute("name")?.Value ?? "";
482 Identifier identifier = structureElement.GetAttributeIdentifier("identifier", "");
483 var structurePrefab = Structure.FindPrefab(name, identifier);
484 if (structurePrefab == null || !structurePrefab.Body) { continue; }
485 if (!structureCrushDepthsDefined && structureElement.Attribute("crushdepth") != null)
486 {
487 structureCrushDepthsDefined = true;
488 }
489 float structureCrushDepth = structureElement.GetAttributeFloat("crushdepth", float.PositiveInfinity);
490 realWorldCrushDepth = Math.Min(structureCrushDepth, realWorldCrushDepth);
491 }
492 if (!structureCrushDepthsDefined)
493 {
494 realWorldCrushDepth = Level.DefaultRealWorldCrushDepth;
495 }
496 realWorldCrushDepth *= GetRealWorldCrushDepthMultiplier();
497 return realWorldCrushDepth;
498 }

574 public static void RefreshSavedSubs()
575 {
576 var contentPackageSubs = ContentPackageManager.EnabledPackages.All.SelectMany(c => c.GetFiles<BaseSubFile>());
577
578 for (int i = savedSubmarines.Count - 1; i >= 0; i--)
579 {
580 if (File.Exists(savedSubmarines[i].FilePath))
581 {
582 bool isDownloadedSub = Path.GetFullPath(Path.GetDirectoryName(savedSubmarines[i].FilePath)) == Path.GetFullPath(SaveUtil.SubmarineDownloadFolder);
583 bool isInContentPackage = contentPackageSubs.Any(f => f.Path == savedSubmarines[i].FilePath);
584 if (isDownloadedSub) { continue; }
585 if (savedSubmarines[i].LastModifiedTime == File.GetLastWriteTime(savedSubmarines[i].FilePath) && isInContentPackage) { continue; }
586 }
587 savedSubmarines[i].Dispose();
588 }
589
590 List<string> filePaths = new List<string>();
591 foreach (BaseSubFile subFile in contentPackageSubs)
592 {
593 if (!filePaths.Any(fp => fp == subFile.Path))
594 {
595 filePaths.Add(subFile.Path.Value);
596 }
597 }
598
599 filePaths.RemoveAll(p => savedSubmarines.Any(sub => sub.FilePath == p));
600
601 foreach (string path in filePaths)
602 {
603 var subInfo = new SubmarineInfo(path);
604 if (!subInfo.IsFileCorrupted)
605 {
606 savedSubmarines.Add(subInfo);
607 }
608 }

616 public static XDocument OpenFile(string file, out Exception exception)
617 {
618 XDocument doc = null;
619 string extension = "";
620 exception = null;
621
622 try
623 {
624 extension = System.IO.Path.GetExtension(file);
625 }
626 catch
627 {
628 //no file extension specified: try using the default one
629 file += ".sub";
630 }
631
632 if (string.IsNullOrWhiteSpace(extension))
633 {
634 extension = ".sub";
635 file += ".sub";
636 }
637
638 if (extension == ".sub")
639 {
640 System.IO.Stream stream;
641 try
642 {
643 stream = SaveUtil.DecompressFiletoStream(file);
644 }
645 catch (System.IO.FileNotFoundException e)
646 {
647 exception = e;
648 DebugConsole.ThrowError("Loading submarine \"" + file + "\" failed! (File not found) " + Environment.StackTrace.CleanupStackTrace(), e);
649 return null;
650 }
651 catch (Exception e)
652 {
653 exception = e;
654 DebugConsole.ThrowError("Loading submarine \"" + file + "\" failed!", e);
655 return null;
656 }
657
658 try
659 {
660 stream.Position = 0;
661 using (var reader = XMLExtensions.CreateReader(stream))
662 {
663 doc = XDocument.Load(reader);
664 }
665 stream.Close();
666 stream.Dispose();
667 }
668
669 catch (Exception e)
670 {
671 exception = e;
672 DebugConsole.ThrowError("Loading submarine \"" + file + "\" failed! (" + e.Message + ")");
673 return null;
674 }
675 }
676 else if (extension == ".xml")
677 {
678 try
679 {
680 ToolBox.IsProperFilenameCase(file);
681 using var stream = File.Open(file, System.IO.FileMode.Open, System.IO.FileAccess.Read);
682 using var reader = XMLExtensions.CreateReader(stream);
683 doc = XDocument.Load(reader);
684 }
685 catch (Exception e)
686 {
687 exception = e;
688 DebugConsole.ThrowError("Loading submarine \"" + file + "\" failed! (" + e.Message + ")");
689 return null;
690 }
691 }
692 else
693 {
694 DebugConsole.ThrowError("Couldn't load submarine \"" + file + "! (Unrecognized file extension)");
695 return null;
696 }
697
698 return doc;
699 }
