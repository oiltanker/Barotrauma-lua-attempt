101  WaterEffect.Parameters["xWaveWidth"].SetValue(distortionStrength.X);
102  WaterEffect.Parameters["xWaveHeight"].SetValue(distortionStrength.Y);
103  if (BlurAmount > 0.0f)
104  {
105  WaterEffect.CurrentTechnique = WaterEffect.Techniques["WaterShaderBlurred"];
106  WaterEffect.Parameters["xBlurDistance"].SetValue(BlurAmount / 100.0f);
107  }
108  else
109  {
110  WaterEffect.CurrentTechnique = WaterEffect.Techniques["WaterShader"];
111  }
112 
113  Vector2 offset = WavePos;
114  if (cam != null)
115  {
116  offset += (cam.Position - new Vector2(cam.WorldView.Width / 2.0f, -cam.WorldView.Height / 2.0f));
117  offset.Y += cam.WorldView.Height;
118  offset.X += cam.WorldView.Width;
119 #if LINUX || OSX
120  offset.X += cam.WorldView.Width;
121 #endif
122  offset *= DistortionScale;
123  }
124  offset.Y = -offset.Y;
125  WaterEffect.Parameters["xUvOffset"].SetValue(new Vector2((offset.X / GameMain.GraphicsWidth) % 1.0f, (offset.Y / GameMain.GraphicsHeight) % 1.0f));
126  WaterEffect.Parameters["xBumpPos"].SetValue(Vector2.Zero);
127 
128  if (cam != null)
129  {
130  WaterEffect.Parameters["xBumpScale"].SetValue(new Vector2(
131  (float)cam.WorldView.Width / GameMain.GraphicsWidth * DistortionScale.X,
132  (float)cam.WorldView.Height / GameMain.GraphicsHeight * DistortionScale.Y));
133  WaterEffect.Parameters["xTransform"].SetValue(cam.ShaderTransform
134  * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f);
135  WaterEffect.Parameters["xUvTransform"].SetValue(cam.ShaderTransform
136  * Matrix.CreateOrthographicOffCenter(0, spriteBatch.GraphicsDevice.Viewport.Width * 2, spriteBatch.GraphicsDevice.Viewport.Height * 2, 0, 0, 1) * Matrix.CreateTranslation(0.5f, 0.5f, 0.0f));
137  }
138  else
139  {
140  WaterEffect.Parameters["xBumpScale"].SetValue(new Vector2(1.0f, 1.0f));
141  WaterEffect.Parameters["xTransform"].SetValue(Matrix.Identity * Matrix.CreateTranslation(-1.0f, 1.0f, 0.0f));
142  WaterEffect.Parameters["xUvTransform"].SetValue(Matrix.CreateScale(0.5f, -0.5f, 0.0f));
143  }
144 
145  WaterEffect.CurrentTechnique.Passes[0].Apply();
146 
147  Rectangle view = cam != null ? cam.WorldView : spriteBatch.GraphicsDevice.Viewport.Bounds;
148 
149  tempCorners[0] = new Vector3(view.X, view.Y, 0.1f);
150  tempCorners[1] = new Vector3(view.Right, view.Y, 0.1f);
151  tempCorners[2] = new Vector3(view.Right, view.Y - view.Height, 0.1f);
152  tempCorners[3] = new Vector3(view.X, view.Y - view.Height, 0.1f);
153 
154  WaterVertexData backGroundColor = new WaterVertexData(0.1f, 0.1f, 0.5f, 1.0f);
155  tempVertices[0] = new VertexPositionColorTexture(tempCorners[0], backGroundColor, Vector2.Zero);
156  tempVertices[1] = new VertexPositionColorTexture(tempCorners[1], backGroundColor, Vector2.Zero);
157  tempVertices[2] = new VertexPositionColorTexture(tempCorners[2], backGroundColor, Vector2.Zero);
158  tempVertices[3] = new VertexPositionColorTexture(tempCorners[0], backGroundColor, Vector2.Zero);
159  tempVertices[4] = new VertexPositionColorTexture(tempCorners[2], backGroundColor, Vector2.Zero);
160  tempVertices[5] = new VertexPositionColorTexture(tempCorners[3], backGroundColor, Vector2.Zero);
161 
162  spriteBatch.GraphicsDevice.DrawUserPrimitives(PrimitiveType.TriangleList, tempVertices, 0, 2);
163 
164  foreach (KeyValuePair<EntityGrid, VertexPositionColorTexture[]> subVerts in IndoorsVertices)
165  {
166  if (!PositionInIndoorsBuffer.ContainsKey(subVerts.Key) || PositionInIndoorsBuffer[subVerts.Key] == 0) { continue; }
167 
168  offset = WavePos;
169  if (subVerts.Key.Submarine != null) { offset -= subVerts.Key.Submarine.WorldPosition; }
170  if (cam != null)
171  {
172  offset += cam.Position - new Vector2(cam.WorldView.Width / 2.0f, -cam.WorldView.Height / 2.0f);
173  offset.Y += cam.WorldView.Height;
174  offset.X += cam.WorldView.Width;
175  offset *= DistortionScale;
176  }
177  offset.Y = -offset.Y;
178  WaterEffect.Parameters["xUvOffset"].SetValue(new Vector2((offset.X / GameMain.GraphicsWidth) % 1.0f, (offset.Y / GameMain.GraphicsHeight) % 1.0f));
179 
180  WaterEffect.CurrentTechnique.Passes[0].Apply();
181

183  }
