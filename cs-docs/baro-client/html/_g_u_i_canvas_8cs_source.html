40 private static void OnChildrenChanged(RectTransform _)
41 {
42 //add weak reference if we don't have one yet
43 foreach (var child in _instance.Children)
44 {
45 if (!_instance.childrenWeakRef.Any(c => c.TryGetTarget(out var existingChild) && existingChild == child))
46 {
47 _instance.childrenWeakRef.Add(new WeakReference<RectTransform>(child));
48 }
49 }
50 //get rid of strong references
51 _instance.children.Clear();
52 //remove dead children
53 for (int i = _instance.childrenWeakRef.Count - 2; i >= 0; i--)
54 {
55 if (!_instance.childrenWeakRef[i].TryGetTarget(out var child) || child.Parent != _instance)
56 {
57 _instance.childrenWeakRef.RemoveAt(i);
58 }
59 }

63 private static void RecalculateSize()
64 {
65 Vector2 recalculatedSize = size;
66
67 // Scale children that are supposed to encompass the whole screen so that they are properly scaled on ultrawide as well
68 for (int i = 0; i < Instance.childrenWeakRef.Count; i++)
69 {
70 if (!_instance.childrenWeakRef[i].TryGetTarget(out RectTransform target) || target == null) { continue; };
71
72 _instance.children.Add(target);
73
74 if (target.RelativeSize.X < 1 && target.RelativeSize.Y < 1) { continue; }
75
76 ResizeAxis axis;
77
78 if (target.RelativeSize.X >= 1 && target.RelativeSize.Y >= 1)
79 {
80 axis = ResizeAxis.Both;
81 }
82 else if (target.RelativeSize.X >= 1)
83 {
84 axis = ResizeAxis.X;
85 }
86 else
87 {
88 axis = ResizeAxis.Y;
89 }
90
91 switch (axis)
92 {
93 case ResizeAxis.Both:
94 target.RelativeSize = recalculatedSize;
95 break;
96
97 case ResizeAxis.X:
98 target.RelativeSize = new Vector2(recalculatedSize.X, target.RelativeSize.Y);
99 break;
100
101 case ResizeAxis.Y:
102 target.RelativeSize = new Vector2(target.RelativeSize.X, recalculatedSize.Y);
103 break;
104 }
105 }
106
107 Instance.Resize(size, resizeChildren: true);
108 Instance.GetAllChildren().Select(c => c.GUIComponent as GUITextBlock).ForEach(t => t?.SetTextPos());
109 _instance.children.Clear();
110 }
