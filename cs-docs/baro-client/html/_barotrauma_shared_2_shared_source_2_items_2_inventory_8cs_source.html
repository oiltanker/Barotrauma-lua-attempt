23 public bool CanBePut(Item item, bool ignoreCondition = false)
24 {
25 if (item == null) { return false; }
26 if (items.Count > 0)
27 {
28 if (!ignoreCondition)
29 {
30 if (item.IsFullCondition)
31 {
32 if (items.Any(it => !it.IsFullCondition)) { return false; }
33 }
34 else if (MathUtils.NearlyEqual(item.Condition, 0.0f))
35 {
36 if (items.Any(it => !MathUtils.NearlyEqual(it.Condition, 0.0f))) { return false; }
37 }
38 else
39 {
40 return false;
41 }
42 }
43 if (items[0].Quality != item.Quality) { return false; }
44 if (items[0].Prefab.Identifier != item.Prefab.Identifier || items.Count + 1 > item.Prefab.MaxStackSize)
45 {
46 return false;
47 }

83 items.Count + 1 > itemPrefab.MaxStackSize)

470 public virtual bool TryPutItem(Item item, int i, bool allowSwapping, bool allowCombine, Character user, bool createNetworkEvent = true, bool ignoreCondition = false)
471 {
472 if (i < 0 || i >= slots.Length)
473 {
474 string thisItemStr = item?.Prefab.Identifier.Value ?? "null";
475 string ownerStr = "null";
476 if (Owner is Item ownerItem)
477 {
478 ownerStr = ownerItem.Prefab.Identifier.Value;
479 }
480 else if (Owner is Character ownerCharacter)
481 {
482 ownerStr = ownerCharacter.SpeciesName.Value;
483 }
484 string errorMsg = $"Inventory.TryPutItem failed: index was out of range (item: {thisItemStr}, inventory: {ownerStr}).";
485 GameAnalyticsManager.AddErrorEventOnce("Inventory.TryPutItem:IndexOutOfRange", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
486#if DEBUG
487 DebugConsole.ThrowError(errorMsg);
488#endif
489 return false;
490 }
491
492 if (Owner == null) return false;
493 //there's already an item in the slot
494 if (slots[i].Any() && allowCombine)
495 {
496 if (slots[i].First().Combine(item, user))
497 {
498 //item in the slot removed as a result of combining -> put this item in the now free slot
499 if (!slots[i].Any())
500 {
501 return TryPutItem(item, i, allowSwapping, allowCombine, user, createNetworkEvent, ignoreCondition);
502 }
503 return true;
504 }
505 }
506 if (CanBePutInSlot(item, i, ignoreCondition))
507 {
508 PutItem(item, i, user, true, createNetworkEvent);
509 return true;
510 }
511 else if (slots[i].Any() && item.ParentInventory != null && allowSwapping)
512 {
513 var itemInSlot = slots[i].First();
514 if (itemInSlot.OwnInventory != null &&
515 !itemInSlot.OwnInventory.Contains(item) &&
516 (itemInSlot.GetComponent<ItemContainer>()?.GetMaxStackSize(0) ?? 0) == 1 &&
517 itemInSlot.OwnInventory.TrySwapping(0, item, user, createNetworkEvent, swapWholeStack: false))
518 {
519 return true;
520 }
521 return
522 TrySwapping(i, item, user, createNetworkEvent, swapWholeStack: true) ||
523 TrySwapping(i, item, user, createNetworkEvent, swapWholeStack: false);
524 }
525 else
526 {
527#if CLIENT
528 if (visualSlots != null && createNetworkEvent) { visualSlots[i].ShowBorderHighlight(GUIStyle.Red, 0.1f, 0.9f); }
529#endif
530 return false;
531 }

534 protected virtual void PutItem(Item item, int i, Character user, bool removeItem = true, bool createNetworkEvent = true)
535 {
536 if (i < 0 || i >= slots.Length)
537 {
538 string errorMsg = "Inventory.PutItem failed: index was out of range(" + i + ").\n" + Environment.StackTrace.CleanupStackTrace();
539 GameAnalyticsManager.AddErrorEventOnce("Inventory.PutItem:IndexOutOfRange", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
540 return;
541 }
542
543 var should = GameMain.LuaCs.Hook.Call<bool?>("inventoryPutItem", this, item, user, i, removeItem);
544
545 if (should != null && should.Value)
546 return;
547
548 if (Owner == null) { return; }
549
550 Inventory prevInventory = item.ParentInventory;
551 Inventory prevOwnerInventory = item.FindParentInventory(inv => inv is CharacterInventory);
552
553 if (createNetworkEvent)
554 {
555 CreateNetworkEvent();
556 //also delay syncing the inventory the item was inside
557 if (prevInventory != null && prevInventory != this) { prevInventory.syncItemsDelay = 1.0f; }
558 }
559
560 if (removeItem)
561 {
562 item.Drop(user);
563 if (item.ParentInventory != null) { item.ParentInventory.RemoveItem(item); }
564 }
565
566 slots[i].Add(item);
567 item.ParentInventory = this;
568
569#if CLIENT
570 if (visualSlots != null)
571 {
572 visualSlots[i].ShowBorderHighlight(Color.White, 0.1f, 0.4f);
573 if (selectedSlot?.Inventory == this) { selectedSlot.ForceTooltipRefresh = true; }
574 }
575#endif
576
577 if (item.body != null)
578 {
579 item.body.Enabled = false;
580 item.body.BodyType = FarseerPhysics.BodyType.Dynamic;
581 }
582
583#if SERVER
584 if (prevOwnerInventory is CharacterInventory characterInventory && characterInventory != this && Owner == user)
585 {
586 var client = GameMain.Server?.ConnectedClients?.Find(cl => cl.Character == user);
587 GameMain.Server?.KarmaManager.OnItemTakenFromPlayer(characterInventory, client, item);
588 }
589#endif
590 if (this is CharacterInventory)
591 {
592 if (prevInventory != this && prevOwnerInventory != this)
593 {
594 HumanAIController.ItemTaken(item, user);
595 }
596 }
597 else
598 {
599 if (item.FindParentInventory(inv => inv is CharacterInventory) is CharacterInventory currentInventory)
600 {
601 if (currentInventory != prevInventory)
602 {
603 HumanAIController.ItemTaken(item, user);
604 }
605 }
606 }

645 protected bool TrySwapping(int index, Item item, Character user, bool createNetworkEvent, bool swapWholeStack)
646 {
647 if (item?.ParentInventory == null || !slots[index].Any()) { return false; }
648 if (slots[index].Items.Any(it => !it.IsInteractable(user))) { return false; }
649 if (!AllowSwappingContainedItems) { return false; }
650
651 var should = GameMain.LuaCs.Hook.Call<bool?>("inventoryItemSwap", this, item, user, index, swapWholeStack);
652
653 if (should != null)
654 return should.Value;
655
656 //swap to InvSlotType.Any if possible
657 Inventory otherInventory = item.ParentInventory;
658 bool otherIsEquipped = false;
659 int otherIndex = -1;
660 for (int i = 0; i < otherInventory.slots.Length; i++)
661 {
662 if (!otherInventory.slots[i].Contains(item)) { continue; }
663 if (otherInventory is CharacterInventory characterInventory)
664 {
665 if (characterInventory.SlotTypes[i] == InvSlotType.Any)
666 {
667 otherIndex = i;
668 break;
669 }
670 else
671 {
672 otherIsEquipped = true;
673 }
674 }
675 }
676 if (otherIndex == -1)
677 {
678 otherIndex = otherInventory.FindIndex(item);
679 if (otherIndex == -1)
680 {
681 DebugConsole.ThrowError("Something went wrong when trying to swap items between inventory slots: couldn't find the source item from it's inventory.\n" + Environment.StackTrace.CleanupStackTrace());
682 return false;
683 }
684 }
685
686 List<Item> existingItems = new List<Item>();
687 if (swapWholeStack)
688 {
689 existingItems.AddRange(slots[index].Items);
690 for (int j = 0; j < capacity; j++)
691 {
692 if (existingItems.Any(existingItem => slots[j].Contains(existingItem))) { slots[j].RemoveAllItems(); }
693 }
694 }
695 else
696 {
697 existingItems.Add(slots[index].FirstOrDefault());
698 for (int j = 0; j < capacity; j++)
699 {
700 if (existingItems.Any(existingItem => slots[j].Contains(existingItem))) { slots[j].RemoveItem(existingItems.First()); }
701 }
702 }
703
704 List<Item> stackedItems = new List<Item>();
705 if (swapWholeStack)
706 {
707 for (int j = 0; j < otherInventory.capacity; j++)
708 {
709 if (otherInventory.slots[j].Contains(item))
710 {
711 stackedItems.AddRange(otherInventory.slots[j].Items);
712 otherInventory.slots[j].RemoveAllItems();
713 }
714 }
715 }
716 else
717 {
718 stackedItems.Add(item);
719 otherInventory.slots[otherIndex].RemoveItem(item);
720 }
721
722
723 bool swapSuccessful = false;
724 if (otherIsEquipped)
725 {
726 swapSuccessful =
727 stackedItems.Distinct().All(stackedItem => TryPutItem(stackedItem, index, false, false, user, createNetworkEvent))
728 &&
729 (existingItems.All(existingItem => otherInventory.TryPutItem(existingItem, otherIndex, false, false, user, createNetworkEvent)) ||
730 existingItems.Count == 1 && otherInventory.TryPutItem(existingItems.First(), user, CharacterInventory.anySlot, createNetworkEvent));
731 }
732 else
733 {
734 swapSuccessful =
735 (existingItems.All(existingItem => otherInventory.TryPutItem(existingItem, otherIndex, false, false, user, createNetworkEvent)) ||
736 existingItems.Count == 1 && otherInventory.TryPutItem(existingItems.First(), user, CharacterInventory.anySlot, createNetworkEvent))
737 &&
738 stackedItems.Distinct().All(stackedItem => TryPutItem(stackedItem, index, false, false, user, createNetworkEvent));
739
740 if (!swapSuccessful && existingItems.Count == 1 && existingItems[0].AllowDroppingOnSwapWith(item))
741 {
742 if (!(existingItems[0].Container?.ParentInventory is CharacterInventory characterInv) ||
743 !characterInv.TryPutItem(existingItems[0], user, new List<InvSlotType>() { InvSlotType.Any }))
744 {
745 existingItems[0].Drop(user, createNetworkEvent);
746 }
747 swapSuccessful = stackedItems.Distinct().Any(stackedItem => TryPutItem(stackedItem, index, false, false, user, createNetworkEvent));
748#if CLIENT
749 if (swapSuccessful)
750 {
751 SoundPlayer.PlayUISound(GUISoundType.DropItem);
752 if (otherInventory.visualSlots != null && otherIndex > -1)
753 {
754 otherInventory.visualSlots[otherIndex].ShowBorderHighlight(Color.Transparent, 0.1f, 0.1f);
755 }
756 }
757#endif
758 }
759 }
760
761 //if the item in the slot can be moved to the slot of the moved item
762 if (swapSuccessful)
763 {
764 System.Diagnostics.Debug.Assert(slots[index].Contains(item), "Something when wrong when swapping items, item is not present in the inventory.");
765 System.Diagnostics.Debug.Assert(!existingItems.Any(it => !it.Prefab.AllowDroppingOnSwap && !otherInventory.Contains(it)), "Something when wrong when swapping items, item is not present in the other inventory.");
766#if CLIENT
767 if (visualSlots != null)
768 {
769 for (int j = 0; j < capacity; j++)
770 {
771 if (slots[j].Contains(item)) { visualSlots[j].ShowBorderHighlight(GUIStyle.Green, 0.1f, 0.9f); }
772 }
773 }
774 if (otherInventory.visualSlots != null)
775 {
776 for (int j = 0; j < otherInventory.capacity; j++)
777 {
778 if (otherInventory.slots[j].Contains(existingItems.FirstOrDefault())) { otherInventory.visualSlots[j].ShowBorderHighlight(GUIStyle.Green, 0.1f, 0.9f); }
779 }
780 }
781#endif
782 return true;
783 }
784 else //swapping the items failed -> move them back to where they were
785 {
786 if (swapWholeStack)
787 {
788 foreach (Item stackedItem in stackedItems)
789 {
790 for (int j = 0; j < capacity; j++)
791 {
792 if (slots[j].Contains(stackedItem)) { slots[j].RemoveItem(stackedItem); };
793 }
794 }
795 foreach (Item existingItem in existingItems)
796 {
797 for (int j = 0; j < otherInventory.capacity; j++)
798 {
799 if (otherInventory.slots[j].Contains(existingItem)) { otherInventory.slots[j].RemoveItem(existingItem); }
800 }
801 }
802 }
803 else
804 {
805 for (int j = 0; j < capacity; j++)
806 {
807 if (slots[j].Contains(item))
808 {
809 slots[j].RemoveWhere(it => existingItems.Contains(it) || stackedItems.Contains(it));
810 }
811 }
812 for (int j = 0; j < otherInventory.capacity; j++)
813 {
814 if (otherInventory.slots[j].Contains(existingItems.FirstOrDefault()))
815 {
816 otherInventory.slots[j].RemoveWhere(it => existingItems.Contains(it) || stackedItems.Contains(it));
817 }
818 }
819 }
820
821 if (otherIsEquipped)
822 {
823 existingItems.ForEach(existingItem => TryPutItem(existingItem, index, false, false, user, createNetworkEvent, ignoreCondition: true));
824 stackedItems.ForEach(stackedItem => otherInventory.TryPutItem(stackedItem, otherIndex, false, false, user, createNetworkEvent, ignoreCondition: true));
825 }
826 else
827 {
828 stackedItems.ForEach(stackedItem => otherInventory.TryPutItem(stackedItem, otherIndex, false, false, user, createNetworkEvent, ignoreCondition: true));
829 existingItems.ForEach(existingItem => TryPutItem(existingItem, index, false, false, user, createNetworkEvent, ignoreCondition: true));
830 }
831
