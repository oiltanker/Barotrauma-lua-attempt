207  public static T GetAnimParams<T>(Identifier speciesName, AnimationType animType, string fileName = null) where T : AnimationParams, new()
208  {
209  if (!allAnimations.TryGetValue(speciesName, out Dictionary<string, AnimationParams> anims))
210  {
211  anims = new Dictionary<string, AnimationParams>();
212  allAnimations.Add(speciesName, anims);
213  }
214  if (fileName == null || !anims.TryGetValue(fileName, out AnimationParams anim))
215  {
216  string selectedFile = null;
217  string folder = GetFolder(speciesName);
218  if (Directory.Exists(folder))
219  {
220  var files = Directory.GetFiles(folder);
221  if (files.None())
222  {
223  DebugConsole.ThrowError($"[AnimationParams] Could not find any animation files from the folder: {folder}. Using the default animation.");
224  selectedFile = GetDefaultFile(speciesName, animType);
225  }
226  var filteredFiles = FilterFilesByType(files, animType);
227  if (filteredFiles.None())
228  {
229  DebugConsole.ThrowError($"[AnimationParams] Could not find any animation files that match the animation type {animType} from the folder: {folder}. Using the default animation.");
230  selectedFile = GetDefaultFile(speciesName, animType);
231  }
232  else if (string.IsNullOrEmpty(fileName))
233  {
234  // Files found, but none specified.
235  selectedFile = GetDefaultFile(speciesName, animType);
236  }
237  else
238  {
239  selectedFile = filteredFiles.FirstOrDefault(f => IO.Path.GetFileNameWithoutExtension(f).Equals(fileName, StringComparison.OrdinalIgnoreCase));
240  if (selectedFile == null)
241  {
242  DebugConsole.ThrowError($"[AnimationParams] Could not find an animation file that matches the name {fileName} and the animation type {animType}. Using the default animations.");
243  selectedFile = GetDefaultFile(speciesName, animType);
244  }
245  }
246  }
247  else
248  {
249  DebugConsole.ThrowError($"[Animationparams] Invalid directory: {folder}. Using the default animation.");
250  selectedFile = GetDefaultFile(speciesName, animType);
251  }
252  if (selectedFile == null)
253  {
254  throw new Exception("[AnimationParams] Selected file null!");
255  }
256  DebugConsole.Log($"[AnimationParams] Loading animations from {selectedFile}.");
257  var characterPrefab = CharacterPrefab.Prefabs[speciesName];
258  T a = new T();
259  if (a.Load(ContentPath.FromRaw(characterPrefab.ContentPackage, selectedFile), speciesName))
260  {
261  fileName = IO.Path.GetFileNameWithoutExtension(selectedFile);
262  if (!anims.ContainsKey(fileName))
263  {
264  anims.Add(fileName, a);
265  }
266  }
267  else
268  {
269  DebugConsole.ThrowError($"[AnimationParams] Failed to load an animation {a} at {selectedFile} of type {animType} for the character {speciesName}");
270  }
271  return a;
272  }
273  return (T)anim;
274  }

322  public static T Create<T>(string fullPath, Identifier speciesName, AnimationType animationType) where T : AnimationParams, new()
323  {
324  if (animationType == AnimationType.NotDefined)
325  {
326  throw new Exception("Cannot create an animation file of type " + animationType.ToString());
327  }
328  if (!allAnimations.TryGetValue(speciesName, out Dictionary<string, AnimationParams> anims))
329  {
330  anims = new Dictionary<string, AnimationParams>();
331  allAnimations.Add(speciesName, anims);
332  }
333  var fileName = IO.Path.GetFileNameWithoutExtension(fullPath);
334  if (anims.ContainsKey(fileName))
335  {
336  DebugConsole.NewMessage($"[AnimationParams] Removing the old animation of type {animationType}.", Color.Red);
337  anims.Remove(fileName);
338  }
339  var instance = new T();
340  XElement animationElement = new XElement(GetDefaultFileName(speciesName, animationType), new XAttribute("animationtype", animationType.ToString()));
341  instance.doc = new XDocument(animationElement);
342  var characterPrefab = CharacterPrefab.Prefabs[speciesName];
343  var contentPath = ContentPath.FromRaw(characterPrefab.ContentPackage, fullPath);
344  instance.UpdatePath(contentPath);
345  instance.IsLoaded = instance.Deserialize(animationElement);
346  instance.Save();
347  instance.Load(contentPath, speciesName);
348  anims.Add(fileName, instance);
349  DebugConsole.NewMessage($"[AnimationParams] New animation file of type {animationType} created.", Color.GhostWhite);
350  return instance;
351  }
