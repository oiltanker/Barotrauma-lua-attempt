456  void CreatePathUnlockElement(GUIComponent reputationFrame, Faction faction, Location location)
457  {
458  if (GameMain.GameSession?.Campaign?.Map != null)
459  {
460  foreach (LocationConnection connection in GameMain.GameSession.Campaign.Map.Connections)
461  {
462  if (!connection.Locked || (!connection.Locations[0].Discovered && !connection.Locations[1].Discovered)) { continue; }
463 
464  var gateLocation = connection.Locations[0].IsGateBetweenBiomes ? connection.Locations[0] : connection.Locations[1];
465  var unlockEvent =
466  EventPrefab.Prefabs.FirstOrDefault(ep => ep.UnlockPathEvent && ep.BiomeIdentifier == gateLocation.LevelData.Biome.Identifier) ??
467  EventPrefab.Prefabs.FirstOrDefault(ep => ep.UnlockPathEvent && ep.BiomeIdentifier == Identifier.Empty);
468 
469  if (unlockEvent == null) { continue; }
470  if (string.IsNullOrEmpty(unlockEvent.UnlockPathFaction) || unlockEvent.UnlockPathFaction.Equals("location", StringComparison.OrdinalIgnoreCase))
471  {
472  if (location == null || gateLocation != location) { continue; }
473  }
474  else
475  {
476  if (faction == null || faction.Prefab.Identifier != unlockEvent.UnlockPathFaction) { continue; }
477  }
478 
479  if (unlockEvent != null)
480  {
481  Reputation unlockReputation = gateLocation.Reputation;
482  Faction unlockFaction = null;
483  if (!string.IsNullOrEmpty(unlockEvent.UnlockPathFaction))
484  {
485  unlockFaction = GameMain.GameSession.Campaign.Factions.Find(f => f.Prefab.Identifier == unlockEvent.UnlockPathFaction);
486  unlockReputation = unlockFaction?.Reputation;
487  }
488  float normalizedUnlockReputation = MathUtils.InverseLerp(unlockReputation.MinReputation, unlockReputation.MaxReputation, unlockEvent.UnlockPathReputation);
489  RichString unlockText = RichString.Rich(TextManager.GetWithVariables(
