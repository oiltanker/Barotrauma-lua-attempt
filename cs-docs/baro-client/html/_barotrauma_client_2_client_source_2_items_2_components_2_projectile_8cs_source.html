16 public void ClientEventRead(IReadMessage msg, float sendingTime)
17 {
18 bool launch = msg.ReadBoolean();
19 if (launch)
20 {
21 ushort userId = msg.ReadUInt16();
22 User = Entity.FindEntityByID(userId) as Character;
23 Vector2 simPosition = new Vector2(msg.ReadSingle(), msg.ReadSingle());
24 float rotation = msg.ReadSingle();
25 if (User != null)
26 {
27 Shoot(User, simPosition, simPosition, rotation, ignoredBodies: User.AnimController.Limbs.Where(l => !l.IsSevered).Select(l => l.body.FarseerBody).ToList(), createNetworkEvent: false);
28 }
29 else
30 {
31 Launch(User, simPosition, rotation);
32 }
33 }
34
35 bool isStuck = msg.ReadBoolean();
36 if (isStuck)
37 {
38 ushort submarineID = msg.ReadUInt16();
39 ushort hullID = msg.ReadUInt16();
40 Vector2 simPosition = new Vector2(
41 msg.ReadSingle(),
42 msg.ReadSingle());
43 Vector2 axis = new Vector2(
44 msg.ReadSingle(),
45 msg.ReadSingle());
46 UInt16 entityID = msg.ReadUInt16();
47
48 Entity entity = Entity.FindEntityByID(entityID);
49 Submarine submarine = Entity.FindEntityByID(submarineID) as Submarine;
50 Hull hull = Entity.FindEntityByID(hullID) as Hull;
51 item.Submarine = submarine;
52 item.CurrentHull = hull;
53 item.body.SetTransform(simPosition, item.body.Rotation);
54 if (entity is Character character)
55 {
56 byte limbIndex = msg.ReadByte();
57 if (limbIndex >= character.AnimController.Limbs.Length)
58 {
59 DebugConsole.ThrowError($"Failed to read a projectile update from the server. Limb index out of bounds ({limbIndex}, character: {character.ToString()})");
60 return;
61 }
62 if (character.Removed) { return; }
63 var limb = character.AnimController.Limbs[limbIndex];
64 StickToTarget(limb.body.FarseerBody, axis);
65 }
66 else if (entity is Structure structure)
67 {
68 byte bodyIndex = msg.ReadByte();
69 if (bodyIndex == 255) { bodyIndex = 0; }
70 if (bodyIndex >= structure.Bodies.Count)
71 {
72 DebugConsole.ThrowError($"Failed to read a projectile update from the server. Structure body index out of bounds ({bodyIndex}, structure: {structure.ToString()})");
73 return;
74 }
75 var body = structure.Bodies[bodyIndex];
76 StickToTarget(body, axis);
77 }
78 else if (entity is Item item)
79 {
80 if (item.Removed) { return; }
81 var door = item.GetComponent<Door>();
82 if (door != null)
83 {
84 StickToTarget(door.Body.FarseerBody, axis);
85 }
86 else if (item.body != null)
87 {
88 StickToTarget(item.body.FarseerBody, axis);
89 }
90 }
91 else if (entity is Submarine sub)
92 {
93 StickToTarget(sub.PhysicsBody.FarseerBody, axis);
94 }
95 else
96 {
97 DebugConsole.ThrowError($"Failed to read a projectile update from the server. Invalid stick target ({entity?.ToString() ?? "null"}, {entityID})");
98 }
99 }
100 else
101 {
102 Unstick();
103 }
104 }

107 {

110 if (item.body.Dir < 0.0f) { rotation += MathHelper.Pi; }
111 Tuple<Vector2, Vector2> tracerPoints = new Tuple<Vector2, Vector2>(startLocation, endLocation);

113 {
114 emitter.Emit(1.0f, particlePos, hullGuess: null, angle: rotation, particleRotation: rotation, colorMultiplier: emitter.Prefab.Properties.ColorMultiplier, tracerPoints: tracerPoints);
115 }
