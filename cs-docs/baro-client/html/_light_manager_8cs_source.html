164 public void RenderLightMap(GraphicsDevice graphics, SpriteBatch spriteBatch, Camera cam, RenderTarget2D backgroundObstructor = null)
165 {
166 if (!LightingEnabled) { return; }
167
168 if (Math.Abs(currLightMapScale - GameSettings.CurrentConfig.Graphics.LightMapScale) > 0.01f)
169 {
170 //lightmap scale has changed -> recreate render targets
171 CreateRenderTargets(graphics);
172 }
173
174 Matrix spriteBatchTransform = cam.Transform * Matrix.CreateScale(new Vector3(GameSettings.CurrentConfig.Graphics.LightMapScale, GameSettings.CurrentConfig.Graphics.LightMapScale, 1.0f));
175 Matrix transform = cam.ShaderTransform
176 * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
177
178 bool highlightsVisible = UpdateHighlights(graphics, spriteBatch, spriteBatchTransform, cam);
179
180 Rectangle viewRect = cam.WorldView;
181 viewRect.Y -= cam.WorldView.Height;
182 //check which lights need to be drawn
183 activeLights.Clear();
184 foreach (LightSource light in lights)
185 {
186 if (!light.Enabled) { continue; }
187 if ((light.Color.A < 1 || light.Range < 1.0f) && !light.LightSourceParams.OverrideLightSpriteAlpha.HasValue) { continue; }
188 if (light.ParentBody != null)
189 {
190 light.ParentBody.UpdateDrawPosition();
191
192 Vector2 pos = light.ParentBody.DrawPosition;
193 if (light.ParentSub != null) { pos -= light.ParentSub.DrawPosition; }
194 light.Position = pos;
195 }
196
197 float range = light.LightSourceParams.TextureRange;
198 if (light.LightSprite != null)
199 {
200 float spriteRange = Math.Max(
201 light.LightSprite.size.X * light.SpriteScale.X * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.X - 0.5f)),
202 light.LightSprite.size.Y * light.SpriteScale.Y * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.Y - 0.5f)));
203
204 float targetSize = Math.Max(light.LightTextureTargetSize.X, light.LightTextureTargetSize.Y);
205 range = Math.Max(Math.Max(spriteRange, targetSize), range);
206 }
207 if (!MathUtils.CircleIntersectsRectangle(light.WorldPosition, range, viewRect)) { continue; }
208 activeLights.Add(light);
209 }
210
211 //draw light sprites attached to characters
212 //render into a separate rendertarget using alpha blending (instead of on top of everything else with alpha blending)
213 //to prevent the lights from showing through other characters or other light sprites attached to the same character
214 //---------------------------------------------------------------------------------------------------
215 graphics.SetRenderTarget(LimbLightMap);
216 graphics.Clear(Color.Black);
217 graphics.BlendState = BlendState.NonPremultiplied;
218 spriteBatch.Begin(SpriteSortMode.BackToFront, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform);
219 foreach (LightSource light in activeLights)
220 {
221 if (light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
222 //draw limb lights at this point, because they were skipped over previously to prevent them from being obstructed
223 if (light.ParentBody?.UserData is Limb limb && !limb.Hide) { light.DrawSprite(spriteBatch, cam); }
224 }
225 spriteBatch.End();
226
227 //draw background lights
228 //---------------------------------------------------------------------------------------------------
229 graphics.SetRenderTarget(LightMap);
230 graphics.Clear(AmbientLight);
231 graphics.BlendState = BlendState.Additive;
232 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
233 Level.Loaded?.BackgroundCreatureManager?.DrawLights(spriteBatch, cam);
234 foreach (LightSource light in activeLights)
235 {
236 if (!light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
237 light.DrawSprite(spriteBatch, cam);
238 light.DrawLightVolume(spriteBatch, lightEffect, transform);
239 }
240 GameMain.ParticleManager.Draw(spriteBatch, true, null, Particles.ParticleBlendState.Additive);
241 spriteBatch.End();
242
243 //draw a black rectangle on hulls to hide background lights behind subs
244 //---------------------------------------------------------------------------------------------------
245
246 /*if (backgroundObstructor != null)
247 {
248 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied);
249 spriteBatch.Draw(backgroundObstructor, new Rectangle(0, 0,
250 (int)(GameMain.GraphicsWidth * currLightMapScale), (int)(GameMain.GraphicsHeight * currLightMapScale)), Color.Black);
251 spriteBatch.End();
252 }*/
253
254 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Opaque, transformMatrix: spriteBatchTransform);
255 Dictionary<Hull, Rectangle> visibleHulls = GetVisibleHulls(cam);
256 foreach (KeyValuePair<Hull, Rectangle> hull in visibleHulls)
257 {
258 GUI.DrawRectangle(spriteBatch,
259 new Vector2(hull.Value.X, -hull.Value.Y),
260 new Vector2(hull.Value.Width, hull.Value.Height),
261 hull.Key.AmbientLight == Color.TransparentBlack ? Color.Black : hull.Key.AmbientLight.Multiply(hull.Key.AmbientLight.A / 255.0f), true);
262 }
263 spriteBatch.End();
264
265 SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColor"];
266 SolidColorEffect.Parameters["color"].SetValue(AmbientLight.Opaque().ToVector4());
267 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform, effect: SolidColorEffect);
268 Submarine.DrawDamageable(spriteBatch, null);
269 spriteBatch.End();
270
271 graphics.BlendState = BlendState.Additive;
272
273
274 //draw the focused item and character to highlight them,
275 //and light sprites (done before drawing the actual light volumes so we can make characters obstruct the highlights and sprites)
276 //---------------------------------------------------------------------------------------------------
277 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
278 foreach (LightSource light in activeLights)
279 {
280 //don't draw limb lights at this point, they need to be drawn after lights have been obstructed by characters
281 if (light.IsBackground || light.ParentBody?.UserData is Limb || light.CurrentBrightness <= 0.0f) { continue; }
282 light.DrawSprite(spriteBatch, cam);
283 }
284 spriteBatch.End();
285
286 if (highlightsVisible)
287 {
288 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive);
289 spriteBatch.Draw(HighlightMap, Vector2.Zero, Color.White);
290 spriteBatch.End();
291 }
292
293 //draw characters to obstruct the highlighted items/characters and light sprites
294 //---------------------------------------------------------------------------------------------------
295
296 SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidVertexColor"];
297 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
298 foreach (Character character in Character.CharacterList)
299 {
300 if (character.CurrentHull == null || !character.Enabled || !character.IsVisible) { continue; }
301 if (Character.Controlled?.FocusedCharacter == character) { continue; }
302 Color lightColor = character.CurrentHull.AmbientLight == Color.TransparentBlack ?
303 Color.Black :
304 character.CurrentHull.AmbientLight.Multiply(character.CurrentHull.AmbientLight.A / 255.0f).Opaque();
305 foreach (Limb limb in character.AnimController.Limbs)
306 {
307 if (limb.DeformSprite != null) { continue; }
308 limb.Draw(spriteBatch, cam, lightColor);
309 }
310 }
311 spriteBatch.End();
312
313 DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShaderSolidVertexColor"];
314 DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
315 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform);
316 foreach (Character character in Character.CharacterList)
317 {
318 if (character.CurrentHull == null || !character.Enabled || !character.IsVisible) { continue; }
319 if (Character.Controlled?.FocusedCharacter == character) { continue; }
320 Color lightColor = character.CurrentHull.AmbientLight == Color.TransparentBlack ?
321 Color.Black :
322 character.CurrentHull.AmbientLight.Multiply(character.CurrentHull.AmbientLight.A / 255.0f).Opaque();
323 foreach (Limb limb in character.AnimController.Limbs)
324 {
325 if (limb.DeformSprite == null) { continue; }
326 limb.Draw(spriteBatch, cam, lightColor);
327 }
328 }
329 spriteBatch.End();
330 DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShader"];
331 graphics.BlendState = BlendState.Additive;
332
333 //draw the actual light volumes, additive particles, hull ambient lights and the halo around the player
334 //---------------------------------------------------------------------------------------------------
335 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
336
337 spriteBatch.Draw(LimbLightMap, new Rectangle(cam.WorldView.X, -cam.WorldView.Y, cam.WorldView.Width, cam.WorldView.Height), Color.White);
338
339 foreach (ElectricalDischarger discharger in ElectricalDischarger.List)
340 {
341 discharger.DrawElectricity(spriteBatch);
342 }
343
344 foreach (LightSource light in activeLights)
345 {
346 if (light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
347 light.DrawLightVolume(spriteBatch, lightEffect, transform);
348 }
349
350 lightEffect.World = transform;
351
352 GameMain.ParticleManager.Draw(spriteBatch, false, null, Particles.ParticleBlendState.Additive);
353
354 if (Character.Controlled != null)
355 {
356 DrawHalo(Character.Controlled);
357 }
358 else
359 {
360 foreach (Character character in Character.CharacterList)
361 {
362 if (character.Submarine == null || character.IsDead || !character.IsHuman) { continue; }
363 DrawHalo(character);
364 }

367 void DrawHalo(Character character)
368 {
369 if (character == null || character.Removed) { return; }
370 Vector2 haloDrawPos = character.DrawPosition;
371 haloDrawPos.Y = -haloDrawPos.Y;
372
373 //ambient light decreases the brightness of the halo (no need for a bright halo if the ambient light is bright enough)
374 float ambientBrightness = (AmbientLight.R + AmbientLight.B + AmbientLight.G) / 255.0f / 3.0f;
375 Color haloColor = Color.White.Multiply(0.3f - ambientBrightness);
376 if (haloColor.A > 0)
377 {
378 float scale = 512.0f / LightSource.LightTexture.Width;
379 spriteBatch.Draw(
380 LightSource.LightTexture, haloDrawPos, null, haloColor, 0.0f,
381 new Vector2(LightSource.LightTexture.Width, LightSource.LightTexture.Height) / 2, scale, SpriteEffects.None, 0.0f);
382 }
383 }
384
385 spriteBatch.End();
386
387 //draw the actual light volumes, additive particles, hull ambient lights and the halo around the player
388 //---------------------------------------------------------------------------------------------------
389
390 graphics.SetRenderTarget(null);
391 graphics.BlendState = BlendState.NonPremultiplied;
392 }

396 private bool UpdateHighlights(GraphicsDevice graphics, SpriteBatch spriteBatch, Matrix spriteBatchTransform, Camera cam)
397 {
398 if (GUI.DisableItemHighlights) { return false; }
399
400 highlightedEntities.Clear();
401 if (Character.Controlled != null && (!Character.Controlled.IsKeyDown(InputType.Aim) || Character.Controlled.HeldItems.Any(it => it.GetComponent<Sprayer>() == null)))
402 {
403 if (Character.Controlled.FocusedItem != null)
404 {
405 highlightedEntities.Add(Character.Controlled.FocusedItem);
406 }
407 if (Character.Controlled.FocusedCharacter != null)
408 {
409 highlightedEntities.Add(Character.Controlled.FocusedCharacter);
410 }
411 foreach (Item item in Item.ItemList)
412 {
413 if ((item.IsHighlighted || item.IconStyle != null) && !highlightedEntities.Contains(item))
414 {
415 highlightedEntities.Add(item);
416 }
417 }
418 }
419 if (highlightedEntities.Count == 0) { return false; }
420
421 //draw characters in light blue first
422 graphics.SetRenderTarget(HighlightMap);
423 SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColor"];
424 SolidColorEffect.Parameters["color"].SetValue(Color.LightBlue.ToVector4());
425 SolidColorEffect.CurrentTechnique.Passes[0].Apply();
426 DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShaderSolidColor"];
427 DeformableSprite.Effect.Parameters["solidColor"].SetValue(Color.LightBlue.ToVector4());
428 DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
429 spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
430 foreach (Entity highlighted in highlightedEntities)
431 {
432 if (highlighted is Item item)
433 {
434 if (item.IconStyle != null && (item != Character.Controlled.FocusedItem || Character.Controlled.FocusedItem == null))
435 {
436 //wait until next pass
437 }
438 else
439 {
440 item.Draw(spriteBatch, false, true);
441 }
442 }
443 else if (highlighted is Character character)
444 {
445 character.Draw(spriteBatch, cam);
446 }
447 }
448 spriteBatch.End();
449
450 //draw items with iconstyles in the style's color
451 spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.Additive, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
452 foreach (Entity highlighted in highlightedEntities)
453 {
454 if (highlighted is Item item)
455 {
456 if (item.IconStyle != null && (item != Character.Controlled.FocusedItem || Character.Controlled.FocusedItem == null))
457 {
458 SolidColorEffect.Parameters["color"].SetValue(item.IconStyle.Color.ToVector4());
459 SolidColorEffect.CurrentTechnique.Passes[0].Apply();
460 item.Draw(spriteBatch, false, true);
461 }
462 }
463 }
464 spriteBatch.End();
465
466 //draw characters in black with a bit of blur, leaving the white edges visible
467 float phase = (float)(Math.Sin(Timing.TotalTime * 3.0f) + 1.0f) / 2.0f; //phase oscillates between 0 and 1
468 Vector4 overlayColor = Color.Black.ToVector4() * MathHelper.Lerp(0.5f, 0.9f, phase);
469 SolidColorEffect.Parameters["color"].SetValue(overlayColor);
470 SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColorBlur"];
471 SolidColorEffect.CurrentTechnique.Passes[0].Apply();
472 DeformableSprite.Effect.Parameters["solidColor"].SetValue(overlayColor);
473 DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
474 spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
475 foreach (Entity highlighted in highlightedEntities)
476 {
477 if (highlighted is Item item)
478 {
479 SolidColorEffect.Parameters["blurDistance"].SetValue(0.02f);
480 item.Draw(spriteBatch, false, true);
481 }
482 else if (highlighted is Character character)
483 {
484 SolidColorEffect.Parameters["blurDistance"].SetValue(0.05f);
485 character.Draw(spriteBatch, cam);
486 }
487 }
488 spriteBatch.End();
489
490 //raster pattern on top of everything
491 spriteBatch.Begin(blendState: BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap);
492 spriteBatch.Draw(highlightRaster,
493 new Rectangle(0, 0, HighlightMap.Width, HighlightMap.Height),
494 new Rectangle(0, 0, (int)(HighlightMap.Width / currLightMapScale * 0.5f), (int)(HighlightMap.Height / currLightMapScale * 0.5f)),
495 Color.White * 0.5f);
496 spriteBatch.End();
497
498 DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShader"];
499
500 return true;
501 }

504 private Dictionary<Hull, Rectangle> GetVisibleHulls(Camera cam)
505 {
506 visibleHulls.Clear();
507 foreach (Hull hull in Hull.HullList)
508 {
509 if (hull.HiddenInGame) { continue; }
510 var drawRect =
511 hull.Submarine == null ?
512 hull.Rect :
513 new Rectangle((int)(hull.Submarine.DrawPosition.X + hull.Rect.X), (int)(hull.Submarine.DrawPosition.Y + hull.Rect.Y), hull.Rect.Width, hull.Rect.Height);
514
515 if (drawRect.Right < cam.WorldView.X || drawRect.X > cam.WorldView.Right ||
516 drawRect.Y - drawRect.Height > cam.WorldView.Y || drawRect.Y < cam.WorldView.Y - cam.WorldView.Height)
517 {
518 continue;
519 }
520 visibleHulls.Add(hull, drawRect);
521 }
522 return visibleHulls;
523 }

525 public void UpdateObstructVision(GraphicsDevice graphics, SpriteBatch spriteBatch, Camera cam, Vector2 lookAtPosition)
526 {
527 if ((!LosEnabled || LosMode == LosMode.None) && !ObstructVision) return;
528 if (ViewTarget == null) return;
529
530 graphics.SetRenderTarget(LosTexture);
531
532 if (ObstructVision)
533 {
534 graphics.Clear(Color.Black);
535 Vector2 diff = lookAtPosition - ViewTarget.WorldPosition;
536 diff.Y = -diff.Y;
537 if (diff.LengthSquared() > 20.0f * 20.0f) { losOffset = diff; }
538 float rotation = MathUtils.VectorToAngle(losOffset);
539
540 Vector2 scale = new Vector2(
541 MathHelper.Clamp(losOffset.Length() / 256.0f, 4.0f, 5.0f), 3.0f);
542
543 spriteBatch.Begin(SpriteSortMode.Deferred, transformMatrix: cam.Transform * Matrix.CreateScale(new Vector3(GameSettings.CurrentConfig.Graphics.LightMapScale, GameSettings.CurrentConfig.Graphics.LightMapScale, 1.0f)));
544 spriteBatch.Draw(visionCircle, new Vector2(ViewTarget.WorldPosition.X, -ViewTarget.WorldPosition.Y), null, Color.White, rotation,
545 new Vector2(visionCircle.Width * 0.2f, visionCircle.Height / 2), scale, SpriteEffects.None, 0.0f);
546 spriteBatch.End();
547 }
548 else
549 {
550 graphics.Clear(Color.White);
551 }
552
553
554 //--------------------------------------
555
556 if (LosEnabled && LosMode != LosMode.None && ViewTarget != null)
557 {
558 Vector2 pos = ViewTarget.DrawPosition;
559
560 Rectangle camView = new Rectangle(cam.WorldView.X, cam.WorldView.Y - cam.WorldView.Height, cam.WorldView.Width, cam.WorldView.Height);
561
562 Matrix shadowTransform = cam.ShaderTransform
563 * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
564
565 var convexHulls = ConvexHull.GetHullsInRange(ViewTarget.Position, cam.WorldView.Width*0.75f, ViewTarget.Submarine);
566 if (convexHulls != null)
567 {
568 List<VertexPositionColor> shadowVerts = new List<VertexPositionColor>();
569 List<VertexPositionTexture> penumbraVerts = new List<VertexPositionTexture>();
570 foreach (ConvexHull convexHull in convexHulls)
571 {
572 if (!convexHull.Enabled || !convexHull.Intersects(camView)) continue;
573
574 Vector2 relativeLightPos = pos;
575 if (convexHull.ParentEntity?.Submarine != null) relativeLightPos -= convexHull.ParentEntity.Submarine.Position;
576
577 convexHull.CalculateLosVertices(relativeLightPos);
578
579 for (int i = 0; i < convexHull.ShadowVertexCount; i++)
580 {
581 shadowVerts.Add(convexHull.ShadowVertices[i]);
582 }
583
584 for (int i = 0; i < convexHull.PenumbraVertexCount; i++)
585 {
586 penumbraVerts.Add(convexHull.PenumbraVertices[i]);
587 }
588 }
589
590 if (shadowVerts.Count > 0)
591 {
592 ConvexHull.shadowEffect.World = shadowTransform;
593 ConvexHull.shadowEffect.CurrentTechnique.Passes[0].Apply();
594 graphics.DrawUserPrimitives(PrimitiveType.TriangleList, shadowVerts.ToArray(), 0, shadowVerts.Count / 3, VertexPositionColor.VertexDeclaration);
595
596 if (penumbraVerts.Count > 0)
597 {
598 ConvexHull.penumbraEffect.World = shadowTransform;
599 ConvexHull.penumbraEffect.CurrentTechnique.Passes[0].Apply();
600 graphics.DrawUserPrimitives(PrimitiveType.TriangleList, penumbraVerts.ToArray(), 0, penumbraVerts.Count / 3, VertexPositionTexture.VertexDeclaration);
601 }
602 }
603 }
604 }
605 graphics.SetRenderTarget(null);
606 }
