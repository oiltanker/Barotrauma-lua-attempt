325 public KeyMapping(IEnumerable<XElement> elements, in KeyMapping? fallback)
326 {
327 var defaultBindings = GetDefault().Bindings;
328 Dictionary<InputType, KeyOrMouse> bindings = fallback?.Bindings?.ToMutable() ?? defaultBindings.ToMutable();
329 foreach (InputType inputType in (InputType[])Enum.GetValues(typeof(InputType)))
330 {
331 if (!bindings.ContainsKey(inputType)) { bindings.Add(inputType, defaultBindings[inputType]); }
332 }
333
334 foreach (XElement element in elements)
335 {
336 foreach (XAttribute attribute in element.Attributes())
337 {
338 if (Enum.TryParse(attribute.Name.LocalName, out InputType result))
339 {
340 bindings[result] = element.GetAttributeKeyOrMouse(attribute.Name.LocalName, bindings[result]);
341 }
342 }
343 }
344
345 Bindings = bindings.ToImmutableDictionary();
346 }

356 .ToImmutableDictionary();
357 return newMapping;

434 {

436#if CLIENT

443 CompletedTutorials.Init(currentConfigDoc.Root.GetChildElement("tutorials"));
444#endif
445 }

448 currentConfig = Config.GetDefault();
449 SaveCurrentConfig();

453 public static void SetCurrentConfig(in Config newConfig)
454 {
455 bool setGraphicsMode =
456 currentConfig.Graphics.Width != newConfig.Graphics.Width
457 || currentConfig.Graphics.Height != newConfig.Graphics.Height
458 || currentConfig.Graphics.VSync != newConfig.Graphics.VSync
459 || currentConfig.Graphics.DisplayMode != newConfig.Graphics.DisplayMode;
460
461 bool languageChanged = currentConfig.Language != newConfig.Language;
462
463 bool audioOutputChanged = currentConfig.Audio.AudioOutputDevice != newConfig.Audio.AudioOutputDevice;
464 bool voiceCaptureChanged = currentConfig.Audio.VoiceCaptureDevice != newConfig.Audio.VoiceCaptureDevice;
465
466 bool textScaleChanged = Math.Abs(currentConfig.Graphics.TextScale - newConfig.Graphics.TextScale) > MathF.Pow(2.0f, -7);
467
468 currentConfig = newConfig;
469
470#if CLIENT
471 if (setGraphicsMode)
472 {
473 GameMain.Instance.ApplyGraphicsSettings();
474 }
475
476 if (audioOutputChanged)
477 {
478 GameMain.SoundManager?.InitializeAlcDevice(currentConfig.Audio.AudioOutputDevice);
479 }
480
481 if (voiceCaptureChanged)
482 {
483 VoipCapture.ChangeCaptureDevice(currentConfig.Audio.VoiceCaptureDevice);
484 }
485
486 if (textScaleChanged)
487 {
488 foreach (var font in GUIStyle.Fonts.Values)
489 {
490 font.Prefabs.ForEach(p => p.LoadFont());
491 }
492 }
493
494 GameMain.SoundManager?.ApplySettings();
495#endif
496 if (languageChanged) { TextManager.ClearCache(); }
497 }

499 public static void SaveCurrentConfig()
500 {
501 XDocument configDoc = new XDocument();
502 XElement root = new XElement("config"); configDoc.Add(root);
503 currentConfig.SerializeElement(root);
504
505 XElement graphicsElement = new XElement("graphicssettings"); root.Add(graphicsElement);
506 currentConfig.Graphics.SerializeElement(graphicsElement);
507
508#region Backwards compatibility crap
509#warning TODO: remove once modding refactor ships in a stable release
510 XElement backwardsCompatibilityGraphicsMode = new XElement(graphicsElement); root.Add(backwardsCompatibilityGraphicsMode);
511 backwardsCompatibilityGraphicsMode.Name = "graphicsmode";
512#endregion
513
514 XElement audioElement = new XElement("audio"); root.Add(audioElement);
515 currentConfig.Audio.SerializeElement(audioElement);
516
517 XElement contentPackagesElement = new XElement("contentpackages"); root.Add(contentPackagesElement);
518#region More backwards compatibility crap
519 XComment backwardsCompatibleComment = new XComment("Backwards compatibility"); contentPackagesElement.Add(backwardsCompatibleComment);
520#warning TODO: remove once modding refactor ships in a stable release
521 XElement backwardsCompatibleCoreElement = new XElement("core"); contentPackagesElement.Add(backwardsCompatibleCoreElement);
522 backwardsCompatibleCoreElement.SetAttributeValue("name", "Vanilla 0.9");
523#endregion
524 XComment corePackageComment = new XComment(ContentPackageManager.EnabledPackages.Core?.Name ?? "Vanilla"); contentPackagesElement.Add(corePackageComment);
525 XElement corePackageElement = new XElement(ContentPackageManager.CorePackageElementName); contentPackagesElement.Add(corePackageElement);
526 corePackageElement.SetAttributeValue("path", ContentPackageManager.EnabledPackages.Core?.Path ?? ContentPackageManager.VanillaFileList);
527
528 XElement regularPackagesElement = new XElement(ContentPackageManager.RegularPackagesElementName); contentPackagesElement.Add(regularPackagesElement);
529 foreach (var regularPackage in ContentPackageManager.EnabledPackages.Regular)
530 {
531 XComment packageComment = new XComment(regularPackage.Name); regularPackagesElement.Add(packageComment);
532 XElement packageElement = new XElement(ContentPackageManager.RegularPackagesSubElementName); regularPackagesElement.Add(packageElement);
533 packageElement.SetAttributeValue("path", regularPackage.Path);
534 }
535
536#if CLIENT
537 XElement serverFiltersElement = new XElement("serverfilters"); root.Add(serverFiltersElement);
538 ServerListFilters.Instance.SaveTo(serverFiltersElement);
539
540 XElement characterElement = new XElement("player"); root.Add(characterElement);
541 MultiplayerPreferences.Instance.SaveTo(characterElement);
542
543 XElement ignoredHintsElement = new XElement("ignoredhints"); root.Add(ignoredHintsElement);
544 IgnoredHints.Instance.SaveTo(ignoredHintsElement);
545
546 XElement debugConsoleMappingElement = new XElement("debugconsolemapping"); root.Add(debugConsoleMappingElement);
547 DebugConsoleMapping.Instance.SaveTo(debugConsoleMappingElement);
548
549 XElement tutorialsElement = new XElement("tutorials"); root.Add(tutorialsElement);
550 CompletedTutorials.Instance.SaveTo(tutorialsElement);
551
552 XElement keyMappingElement = new XElement("keymapping",
553 currentConfig.KeyMap.Bindings.Select(kvp
554 => new XAttribute(kvp.Key.ToString(), kvp.Value.ToString())));
555 root.Add(keyMappingElement);
556
557 XElement inventoryKeyMappingElement = new XElement("inventorykeymapping",
558 Enumerable.Range(0, currentConfig.InventoryKeyMap.Bindings.Length)
559 .Zip(currentConfig.InventoryKeyMap.Bindings)
560 .Cast<(int Index, KeyOrMouse Bind)>()
561 .Select(kvp
562 => new XAttribute($"slot{kvp.Index.ToString(CultureInfo.InvariantCulture)}", kvp.Bind.ToString())));
563 root.Add(inventoryKeyMappingElement);
564#endif
565
566 configDoc.SaveSafe(PlayerConfigPath);
567
568 System.Xml.XmlWriterSettings settings = new System.Xml.XmlWriterSettings
569 {
570 Indent = true,
571 OmitXmlDeclaration = true,
572 NewLineOnAttributes = true
573 };
574
575 try
576 {
577 using (var writer = XmlWriter.Create(PlayerConfigPath, settings))
578 {
579 configDoc.WriteTo(writer);
580 writer.Flush();
581 }
582 }
583 catch (Exception e)
584 {
585 DebugConsole.ThrowError("Saving game settings failed.", e);
586 GameAnalyticsManager.AddErrorEventOnce("GameSettings.Save:SaveFailed", GameAnalyticsManager.ErrorSeverity.Error,
587 "Saving game settings failed.\n" + e.Message + "\n" + e.StackTrace.CleanupStackTrace());
588 }
589 }
