72  private void UpdateSubscribedModInstalls()
73  {
74  if (!SteamManager.IsInitialized) { return; }
75 
76  uint numSubscribedMods = SteamManager.GetNumSubscribedItems();
77  if (numSubscribedMods == memSubscribedModCount) { return; }
78  memSubscribedModCount = numSubscribedMods;
79 
80  var subscribedIds = SteamManager.GetSubscribedItems().ToHashSet();
81  var installedIds = ContentPackageManager.WorkshopPackages.Select(p => p.SteamWorkshopId).ToHashSet();
82  foreach (var id in subscribedIds.Where(id2 => !installedIds.Contains(id2)))
83  {
84  Steamworks.Ugc.Item item = new Steamworks.Ugc.Item(id);
85  if (!item.IsDownloading && !SteamManager.Workshop.IsInstalling(item))
86  {
87  SteamManager.Workshop.DownloadModThenEnqueueInstall(item);
88  }
89  }
90 
91  TaskPool.Add("RemoveUnsubscribedItems", SteamManager.Workshop.GetPublishedItems(), t =>
92  {
93  if (!t.TryGetResult(out ISet<Steamworks.Ugc.Item> publishedItems)) { return; }
94 
95  var allRequiredInstalled = subscribedIds.Union(publishedItems.Select(it => it.Id)).ToHashSet();
96  bool needsRefresh = false;
97  foreach (var id in installedIds.Where(id2 => !allRequiredInstalled.Contains(id2)))
98  {
99  Steamworks.Ugc.Item item = new Steamworks.Ugc.Item(id);
100  SteamManager.Workshop.Uninstall(item);
101  needsRefresh = true;
102  }
103 
104  if (needsRefresh)
105  {
106  PopulateInstalledModLists();
107  }
108  });
109  }

187  private void HandleDraggingAcrossModLists(GUIListBox from, GUIListBox to)
188  {
189  if (to.Rect.Contains(PlayerInput.MousePosition) && from.DraggedElement != null)
190  {
191  //move the dragged elements to the index determined previously
192  var draggedElement = from.DraggedElement;
193 
194  var selected = from.AllSelected.ToList();
195  selected.Sort((a, b) => from.Content.GetChildIndex(a) - from.Content.GetChildIndex(b));
196 
197  float oldCount = to.Content.CountChildren;
198  float newCount = oldCount + selected.Count;
199 
200  var offset = draggedElement.RectTransform.AbsoluteOffset;
201  offset += from.Content.Rect.Location;
202  offset -= to.Content.Rect.Location;
203 
204  for (int i = 0; i < selected.Count; i++)
205  {
206  var c = selected[i];
207  c.Parent.RemoveChild(c);
208  c.RectTransform.Parent = to.Content.RectTransform;
209  c.RectTransform.RepositionChildInHierarchy((int)oldCount+i);
210  }
211 
212  from.DraggedElement = null;
213  from.Deselect();
214  from.RecalculateChildren();
215  from.RectTransform.RecalculateScale(true);
216  to.RecalculateChildren();
217  to.RectTransform.RecalculateScale(true);
218  to.Select(selected);
219 
220  //recalculate the dragged element's offset so it doesn't jump around
221  draggedElement.RectTransform.AbsoluteOffset = offset;
222 
223  to.DraggedElement = draggedElement;
224 
225  to.BarScroll *= (oldCount / newCount);
226  }
227  }
