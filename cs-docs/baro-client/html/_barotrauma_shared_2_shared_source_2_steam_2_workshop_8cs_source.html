36  private static async Task<WorkshopItemSet> GetWorkshopItems(Steamworks.Ugc.Query query, int? maxPages = null)
37  {
38  if (!IsInitialized) { return new HashSet<Steamworks.Ugc.Item>(); }
39 
40  await Task.Yield();
41  query = query.WithKeyValueTags(true).WithLongDescription(true);
42  var set = new HashSet<Steamworks.Ugc.Item>(ItemEqualityComparer.Instance);
43  int prevSize = 0;
44  for (int i = 1; maxPages is null || i <= maxPages; i++)
45  {
46  Steamworks.Ugc.ResultPage? page = await query.GetPageAsync(i);
47  if (page is null || !page.Value.Entries.Any()) { break; }
48  set.UnionWith(page.Value.Entries);
49 
50  if (set.Count == prevSize) { break; }
51  prevSize = set.Count;
52  }
53  return set;
54  }

353  private static async Task CorrectPaths(string fileListDir, string modName, XElement element)
354  {
355  foreach (var attribute in element.Attributes())
356  {
357  await Task.Yield();
358 
359  string val = attribute.Value.CleanUpPathCrossPlatform(correctFilenameCase: false);
360 
361  //Handle mods that have been mangled by pre-modding-refactor
362  //copying of post-modding-refactor mods (what a clusterfuck)
363  int modDirStrIndex = val.IndexOf(ContentPath.ModDirStr, StringComparison.OrdinalIgnoreCase);
364  if (modDirStrIndex >= 0)
365  {
366  val = val[modDirStrIndex..];
367  }
368 
369  //Handle really old mods (0.9.0.4-era) that might be structured as
370  //%ModDir%/Mods/[NAME]/[RESOURCE]
371  string fullSrcPath = Path.Combine(fileListDir, val).CleanUpPath();
372  if (File.Exists(fullSrcPath))
373  {
374  val = $"{ContentPath.ModDirStr}/{val}";
375  }
376 
377  //Handle old mods that installed to the fixed Mods directory
378  //that no longer exists
379  string oldModDir = $"Mods/{modName}";
380  if (val.StartsWith(oldModDir, StringComparison.OrdinalIgnoreCase))
381  {
382  val = $"{ContentPath.ModDirStr}{val.Remove(0, oldModDir.Length)}";
383  }
384  //Handle old mods that depend on other mods
385  else if (val.StartsWith("Mods/", StringComparison.OrdinalIgnoreCase))
386  {
387  string otherModName = val.Substring(val.IndexOf('/')+1);
388  otherModName = otherModName.Substring(0, otherModName.IndexOf('/'));
389  val = $"{string.Format(ContentPath.OtherModDirFmt, otherModName)}{val.Remove(0, $"Mods/{otherModName}".Length)}";
390  }
391  //Handle really old mods that installed Submarines in the wrong place
392  else if (val.StartsWith("Submarines/", StringComparison.OrdinalIgnoreCase))
393  {
394  val = $"{ContentPath.ModDirStr}/{val}";
395  }
396  attribute.Value = val;
397  }
398  await Task.WhenAll(
399  element.Elements()
400  .Select(subElement => CorrectPaths(
401  fileListDir: fileListDir,
402  modName: modName,
403  element: subElement)));
404  }
