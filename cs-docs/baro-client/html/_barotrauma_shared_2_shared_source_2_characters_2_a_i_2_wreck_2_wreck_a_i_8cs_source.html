47 private WreckAI(Submarine wreck)
48 {
49 Wreck = wreck;
50 Config = WreckAIConfig.GetRandom();
51 if (Config == null)
52 {
53 DebugConsole.ThrowError("WreckAI: No wreck AI config found!");
54 return;
55 }
56 var thalamusPrefabs = ItemPrefab.Prefabs.Where(p => IsThalamus(p));
57 var brainPrefab = thalamusPrefabs.GetRandom(i => i.Tags.Contains(Config.Brain), Rand.RandSync.ServerAndClient);
58 if (brainPrefab == null)
59 {
60 DebugConsole.ThrowError($"WreckAI: Could not find any brain prefab with the tag {Config.Brain}! Cannot continue. Failed to create wreck AI.");
61 return;
62 }
63 allItems = Wreck.GetItems(false);
64 thalamusItems = allItems.FindAll(i => IsThalamus(((MapEntity)i).Prefab));
65 hulls.AddRange(Wreck.GetHulls(false));
66 var potentialBrainHulls = new List<(Hull hull, float weight)>();
67 brain = new Item(brainPrefab, Vector2.Zero, Wreck);
68 thalamusItems.Add(brain);
69 Point minSize = brain.Rect.Size.Multiply(brain.Scale);
70 // Bigger hulls are allowed, but not preferred more than what's sufficent.
71 Vector2 sufficentSize = new Vector2(minSize.X * 2, minSize.Y * 1.1f);
72 // Shrink the horizontal axis so that the brain is not placed in the left or right side, where we often have curved walls.
73 Rectangle shrinkedBounds = ToolBox.GetWorldBounds(Wreck.WorldPosition.ToPoint(), new Point(Wreck.Borders.Width - 500, Wreck.Borders.Height));
74 foreach (Hull hull in hulls)
75 {
76 float distanceFromCenter = Vector2.Distance(Wreck.WorldPosition, hull.WorldPosition);
77 float distanceFactor = MathHelper.Lerp(1.0f, 0.5f, MathUtils.InverseLerp(0, Math.Max(shrinkedBounds.Width, shrinkedBounds.Height) / 2, distanceFromCenter));
78 float horizontalSizeFactor = MathHelper.Lerp(0.5f, 1.0f, MathUtils.InverseLerp(minSize.X, sufficentSize.X, hull.Rect.Width));
79 float verticalSizeFactor = MathHelper.Lerp(0.5f, 1.0f, MathUtils.InverseLerp(minSize.Y, sufficentSize.Y, hull.Rect.Height));
80 float weight = verticalSizeFactor * horizontalSizeFactor * distanceFactor;
81 if (hull.GetLinkedEntities<Hull>().Any())
82 {
83 // Ignore hulls that have any linked hulls to keep the calculations simple.
84 continue;
85 }
86 else if (hull.ConnectedGaps.Any(g => g.Open > 0 && (!g.IsRoomToRoom || g.Position.Y < hull.Position.Y)))
87 {
88 // Ignore hulls that have open gaps to outside or below the center point, because we'll want the room to be full of water and not be accessible without breaking the wall.
89 continue;
90 }
91 else if (thalamusItems.Any(i => i.CurrentHull == hull))
92 {
93 // Don't create the brain in a room that already has thalamus items inside it.
94 continue;
95 }
96 else if (hull.Rect.Width < minSize.X || hull.Rect.Height < minSize.Y)
97 {
98 // Don't select too small rooms.
99 continue;
100 }
101 if (weight > 0)
102 {
103 potentialBrainHulls.Add((hull, weight));
104 }
105 }
106 Hull brainHull = ToolBox.SelectWeightedRandom(potentialBrainHulls.Select(pbh => pbh.hull).ToList(), potentialBrainHulls.Select(pbh => pbh.weight).ToList(), Rand.RandSync.ServerAndClient);
107 var thalamusStructurePrefabs = StructurePrefab.Prefabs.Where(IsThalamus);
108 if (brainHull == null)
109 {
110 DebugConsole.AddWarning("Wreck AI: Cannot find a proper room for the brain. Using a random room.");
111 brainHull = hulls.GetRandom(Rand.RandSync.ServerAndClient);
112 }
113 if (brainHull == null)
114 {
115 DebugConsole.ThrowError("Wreck AI: Cannot find any room for the brain! Failed to create the Thalamus.");
116 return;
117 }
118 brainHull.WaterVolume = brainHull.Volume;
119 brain.SetTransform(brainHull.SimPosition, rotation: 0, findNewHull: false);
120 brain.CurrentHull = brainHull;
121 var backgroundPrefab = thalamusStructurePrefabs.GetRandom(i => i.Tags.Contains(Config.BrainRoomBackground), Rand.RandSync.ServerAndClient);
122 if (backgroundPrefab != null)
123 {
124 new Structure(brainHull.Rect, backgroundPrefab, Wreck);
125 }
126 var horizontalWallPrefab = thalamusStructurePrefabs.GetRandom(p => p.Tags.Contains(Config.BrainRoomHorizontalWall), Rand.RandSync.ServerAndClient);
127 if (horizontalWallPrefab != null)
128 {
129 int height = (int)horizontalWallPrefab.Size.Y;
130 int halfHeight = height / 2;
131 int quarterHeight = halfHeight / 2;
132 new Structure(new Rectangle(brainHull.Rect.Left, brainHull.Rect.Top + quarterHeight, brainHull.Rect.Width, height), horizontalWallPrefab, Wreck);
133 new Structure(new Rectangle(brainHull.Rect.Left, brainHull.Rect.Top - brainHull.Rect.Height + halfHeight + quarterHeight, brainHull.Rect.Width, height), horizontalWallPrefab, Wreck);
134 }
135 var verticalWallPrefab = thalamusStructurePrefabs.GetRandom(p => p.Tags.Contains(Config.BrainRoomVerticalWall), Rand.RandSync.ServerAndClient);
136 if (verticalWallPrefab != null)
137 {
138 int width = (int)verticalWallPrefab.Size.X;
139 int halfWidth = width / 2;
140 int quarterWidth = halfWidth / 2;
141 new Structure(new Rectangle(brainHull.Rect.Left - quarterWidth, brainHull.Rect.Top, width, brainHull.Rect.Height), verticalWallPrefab, Wreck);
142 new Structure(new Rectangle(brainHull.Rect.Right - halfWidth - quarterWidth, brainHull.Rect.Top, width, brainHull.Rect.Height), verticalWallPrefab, Wreck);
143 }
144 foreach (Item item in allItems)
145 {
146 if (thalamusItems.Contains(item))
147 {
148 // Ensure that thalamus items are visible
149 item.HiddenInGame = false;
150 }
151 else
152 {
153 // Load regular turrets
154 var turret = item.GetComponent<Turret>();
155 if (turret != null)
156 {
157 foreach (var linkedItem in item.GetLinkedEntities<Item>())
158 {
159 var container = linkedItem.GetComponent<ItemContainer>();
160 if (container == null) { continue; }
161 for (int i = 0; i < container.Inventory.Capacity; i++)
162 {
163 if (container.Inventory.GetItemAt(i) != null) { continue; }
164 if (MapEntityPrefab.List.GetRandom(e => e is ItemPrefab ip && container.CanBeContained(ip, i) &&
165 Config.ForbiddenAmmunition.None(id => id == ip.Identifier), Rand.RandSync.ServerAndClient) is ItemPrefab ammoPrefab)
166 {
167 Item ammo = new Item(ammoPrefab, container.Item.WorldPosition, Wreck);
168 if (!container.Inventory.TryPutItem(ammo, i, allowSwapping: false, allowCombine: false, user: null, createNetworkEvent: false))
169 {
170 item.Remove();
171 }
172 }
173 }
174 }
175 }
176 }
177 }
178 foreach (var item in allItems)
179 {
180 var turret = item.GetComponent<Turret>();
181 if (turret != null)
182 {
183 turrets.Add(turret);
184 }
185 if (item.HasTag(Config.Spawner))
186 {
187 if (!spawnOrgans.Contains(item))
188 {
189 spawnOrgans.Add(item);
190 if (item.CurrentHull != null)
191 {
192 // Try to flood the hull so that the spawner won't die.
193 item.CurrentHull.WaterVolume = item.CurrentHull.Volume;
194 }
195 }
196 }
197 }
198 wayPoints.AddRange(Wreck.GetWaypoints(false));
199 IsAlive = true;
200 thalamusStructures = GetThalamusEntities<Structure>(Wreck, Config.Entity).ToList();
201 }

204 public void Update(float deltaTime)
205 {
206 if (!IsAlive) { return; }
207 if (Wreck == null || Wreck.Removed)
208 {
209 Remove();
210 return;
211 }
212 if (brain == null || brain.Removed || brain.Condition <= 0)
213 {
214 Kill();
215 return;
216 }
217 destroyedOrgans.Clear();
218 foreach (var organ in spawnOrgans)
219 {
220 if (organ.Condition <= 0)
221 {
222 destroyedOrgans.Add(organ);
223 }
224 }
225 destroyedOrgans.ForEach(o => spawnOrgans.Remove(o));
226 bool someoneNearby = false;
227 float minDist = Sonar.DefaultSonarRange * 2.0f;
228 foreach (Submarine submarine in Submarine.Loaded)
229 {
230 if (submarine.Info.Type != SubmarineType.Player) { continue; }
231 if (Vector2.DistanceSquared(submarine.WorldPosition, Wreck.WorldPosition) < minDist * minDist)
232 {
233 someoneNearby = true;
234 break;
235 }
236 }
237 foreach (Character c in Character.CharacterList)
238 {
239 if (c != Character.Controlled && !c.IsRemotePlayer) { continue; }
240 if (Vector2.DistanceSquared(c.WorldPosition, Wreck.WorldPosition) < minDist * minDist)
241 {
242 someoneNearby = true;
243 break;
244 }
245 }
246 if (!someoneNearby) { return; }
247 OperateTurrets(deltaTime);
248 if (!IsClient)
249 {
250 if (!initialCellsSpawned) { SpawnInitialCells(); }
251 UpdateReinforcements(deltaTime);
252 }
253 }

255 private void SpawnInitialCells()
256 {
257 int brainRoomCells = Rand.Range(MinCellsPerBrainRoom, MaxCellsPerRoom + 1);
258 if (brain.CurrentHull?.WaterPercentage >= MinWaterLevel)
259 {
260 for (int i = 0; i < brainRoomCells; i++)
261 {
262 if (!TrySpawnCell(out _, brain.CurrentHull)) { break; }
263 }
264 }
265 int cellsInside = Rand.Range(MinCellsInside, MaxCellsInside + 1);
266 for (int i = 0; i < cellsInside; i++)
267 {
268 if (!TrySpawnCell(out _)) { break; }
269 }
270 int cellsOutside = Rand.Range(MinCellsOutside, MaxCellsOutside + 1);
271 // If we failed to spawn some of the cells in the brainroom/inside, spawn some extra cells outside.
272 cellsOutside = Math.Clamp(cellsOutside + brainRoomCells + cellsInside - protectiveCells.Count, cellsOutside, MaxCellsOutside);
273 for (int i = 0; i < cellsOutside; i++)
274 {
275 ISpatialEntity targetEntity = wayPoints.GetRandomUnsynced(wp => wp.CurrentHull == null);
276 if (targetEntity == null) { break; }
277 if (!TrySpawnCell(out _, targetEntity)) { break; }
278 }
279 initialCellsSpawned = true;
280 }

282 public void Kill()
283 {
284 thalamusItems.ForEach(i => i.Condition = 0);
285 foreach (var turret in turrets)
286 {
287 // Snap all tendons
288 foreach (Item item in turret.ActiveProjectiles)
289 {
290 if (item.GetComponent<Projectile>()?.IsStuckToTarget ?? false)
291 {
292 item.Condition = 0;
293 }
294 }
295 }
296 FadeOutColors();
297 protectiveCells.ForEach(c => c.OnDeath -= OnCellDeath);
298 if (!IsClient)
299 {
300 if (Config != null)
301 {
302 if (Config.KillAgentsWhenEntityDies)
303 {
304 protectiveCells.ForEach(c => c.Kill(CauseOfDeathType.Unknown, null));
305 if (!string.IsNullOrWhiteSpace(Config.OffensiveAgent))
306 {
307 foreach (var character in Character.CharacterList)
308 {
309 // Kills ALL offensive agents that are near the thalamus. Not the ideal solution,
310 // but as long as spawning is handled via status effects, I don't know if there is any better way.
311 // In practice there shouldn't be terminal cells from different thalamus organisms at the same time.
312 // And if there was, the distance check should prevent killing the agents of a different organism.
313 if (character.SpeciesName == Config.OffensiveAgent)
314 {
315 // Sonar distance is used also for wreck positioning. No wreck should be closer to each other than this.
316 float maxDistance = Sonar.DefaultSonarRange;
317 if (Vector2.DistanceSquared(character.WorldPosition, Wreck.WorldPosition) < maxDistance * maxDistance)
318 {
319 character.Kill(CauseOfDeathType.Unknown, null);
320 }
321 }
322 }
323 }
324 }
325 }
326 }
327 protectiveCells.Clear();
328 IsAlive = false;

341 public static void RemoveThalamusItems(Submarine wreck)
342 {
343 List<MapEntity> thalamusItems = new List<MapEntity>();
344 foreach (var wreckAiConfig in WreckAIConfig.Prefabs)
345 {
346 thalamusItems.AddRange(GetThalamusEntities(wreck, wreckAiConfig.Entity));
347 }
348 thalamusItems = thalamusItems.Distinct().ToList();
349 foreach (MapEntity thalamusItem in thalamusItems)
350 {
351 thalamusItem.Remove();
352 wreck.PhysicsBody.FarseerBody.FixtureList.Where(f => f.UserData == thalamusItem).ForEachMod(f => wreck.PhysicsBody.FarseerBody.Remove(f));
353 }

371 private int CalculateCellCount(int minValue, int maxValue)
372 {
373 if (maxValue == 0) { return 0; }
374 float t = MathUtils.InverseLerp(0, 100, Level.Loaded.Difficulty * Config.AgentSpawnCountDifficultyMultiplier);
375 return (int)Math.Round(MathHelper.Lerp(minValue, maxValue, t));
376 }

378 private float GetSpawnTime()
379 {
380 float randomFactor = Config.AgentSpawnDelayRandomFactor;
381 float delay = Config.AgentSpawnDelay;
382 float min = delay;
383 float max = delay * 6;
384 float t = Level.Loaded.Difficulty * Config.AgentSpawnDelayDifficultyMultiplier * Rand.Range(1 - randomFactor, 1 + randomFactor);
385 return MathHelper.Lerp(max, min, MathUtils.InverseLerp(0, 100, t));
386 }

388 void UpdateReinforcements(float deltaTime)
389 {
390 if (spawnOrgans.Count == 0) { return; }
391 cellSpawnTimer -= deltaTime;
392 if (cellSpawnTimer < 0)
393 {
394 TrySpawnCell(out _, spawnOrgans.GetRandomUnsynced());
395 cellSpawnTimer = GetSpawnTime();
396 }
397 }
398
399 bool TrySpawnCell(out Character cell, ISpatialEntity targetEntity = null)
400 {
401 cell = null;
402 if (protectiveCells.Count >= MaxCellCount) { return false; }
403 if (targetEntity == null)
404 {
405 targetEntity =
406 wayPoints.GetRandomUnsynced(wp => wp.CurrentHull != null && populatedHulls.Count(h => h == wp.CurrentHull) < MaxCellsPerRoom && wp.CurrentHull.WaterPercentage >= MinWaterLevel) ??
407 hulls.GetRandomUnsynced(h => populatedHulls.Count(h2 => h2 == h) < MaxCellsPerRoom && h.WaterPercentage >= MinWaterLevel) as ISpatialEntity;
408 }
409 if (targetEntity == null) { return false; }
410 if (targetEntity is Hull h)
411 {
412 populatedHulls.Add(h);
413 }
414 else if (targetEntity is WayPoint wp && wp.CurrentHull != null)
415 {
416 populatedHulls.Add(wp.CurrentHull);
417 }
418 // Don't add items in the list, because we want to be able to ignore the restrictions for spawner organs.
419 cell = Character.Create(Config.DefensiveAgent, targetEntity.WorldPosition, ToolBox.RandomSeed(8), hasAi: true, createNetworkEvent: true);
420 protectiveCells.Add(cell);
421 cell.OnDeath += OnCellDeath;
422 cellSpawnTimer = GetSpawnTime();
423 return true;
424 }
