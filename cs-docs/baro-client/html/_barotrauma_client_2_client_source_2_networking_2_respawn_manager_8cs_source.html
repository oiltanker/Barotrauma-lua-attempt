74 public void ClientEventRead(IReadMessage msg, float sendingTime)
75 {
76 bool respawnPromptPending = false;
77 var newState = (State)msg.ReadRangedInteger(0, Enum.GetNames(typeof(State)).Length);
78 ForceSpawnInMainSub = false;
79 switch (newState)
80 {
81 case State.Transporting:
82 ReturnCountdownStarted = msg.ReadBoolean();
83 maxTransportTime = msg.ReadSingle();
84 float transportTimeLeft = msg.ReadSingle();
85
86 ReturnTime = DateTime.Now + new TimeSpan(0, 0, 0, 0, milliseconds: (int)(transportTimeLeft * 1000.0f));
87 RespawnCountdownStarted = false;
88 if (CurrentState != newState)
89 {
90 CoroutineManager.StopCoroutines("forcepos");
91 }
92 break;
93 case State.Waiting:
94 PendingRespawnCount = msg.ReadUInt16();
95 RequiredRespawnCount = msg.ReadUInt16();
96 respawnPromptPending = msg.ReadBoolean();
97 RespawnCountdownStarted = msg.ReadBoolean();
98 ForceSpawnInMainSub = msg.ReadBoolean();
99 ResetShuttle();
100 float newRespawnTime = msg.ReadSingle();
101 RespawnTime = DateTime.Now + new TimeSpan(0, 0, 0, 0, milliseconds: (int)(newRespawnTime * 1000.0f));
102 break;
103 case State.Returning:
104 RespawnCountdownStarted = false;
105 break;
106 }
107 CurrentState = newState;
108
109 if (respawnPromptPending)
110 {
111 GameMain.Client.HasSpawned = true;
112 ShowRespawnPromptIfNeeded(delay: 1.0f);
113 }
114
115 msg.ReadPadBits();
116 }
117 }
