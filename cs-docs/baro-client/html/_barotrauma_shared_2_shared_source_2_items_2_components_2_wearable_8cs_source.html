144 private ContentPath ParseSpritePath(ContentXElement element)
145 {
146 if (element.DoesAttributeReferenceFileNameAlone("texture"))
147 {
148 string textureName = element.GetAttributeString("texture", "");
149 return ContentPath.FromRaw(
150 element.ContentPackage,
151 $"{Path.GetDirectoryName(WearableComponent.Item.Prefab.FilePath)}/{textureName}");
152 }
153 else
154 {
155 return element.GetAttributeContentPath("texture") ?? ContentPath.Empty;
156 }
157 }

159 public void ParsePath(bool parseSpritePath)
160 {
161 SpritePath = UnassignedSpritePath.Value;
162 if (_picker?.Info != null)
163 {
164 SpritePath = _picker.Info.ReplaceVars(SpritePath);
165 }
166 SpritePath = SpritePath.Replace("[VARIANT]", Variant.ToString());
167 if (!File.Exists(SpritePath))
168 {
169 // If the variant does not exist, parse the path so that it uses first variant.
170 SpritePath = SpritePath.Replace("[VARIANT]", "1");
171 }
172 if (!File.Exists(SpritePath) && _picker?.Info == null)
173 {
174 // If there's no character info is defined, try to use first tagset from CharacterInfoPrefab
175 var charInfoPrefab = CharacterPrefab.HumanPrefab.CharacterInfoPrefab;
176 SpritePath = charInfoPrefab.ReplaceVars(SpritePath, charInfoPrefab.Heads.First());
177 }
178 if (parseSpritePath)
179 {
180 Sprite.ParseTexturePath(file: SpritePath);
181 }
182 }

186 {

189 _picker = picker;

193 Limb = (LimbType)Enum.Parse(typeof(LimbType), SourceElement.GetAttributeString("limb", "Head"), true);
194 HideLimb = SourceElement.GetAttributeBool("hidelimb", false);
195 HideOtherWearables = SourceElement.GetAttributeBool("hideotherwearables", false);
196 CanBeHiddenByOtherWearables = SourceElement.GetAttributeBool("canbehiddenbyotherwearables", true);
197 InheritLimbDepth = SourceElement.GetAttributeBool("inheritlimbdepth", true);
198 var scale = SourceElement.GetAttribute("inheritscale");
199 if (scale != null)
200 {
201 InheritScale = scale.GetAttributeBool(false);
202 }
203 else
204 {
205 InheritScale = SourceElement.GetAttributeBool("inherittexturescale", false);
206 }
207 IgnoreLimbScale = SourceElement.GetAttributeBool("ignorelimbscale", false);
208 IgnoreTextureScale = SourceElement.GetAttributeBool("ignoretexturescale", false);
209 IgnoreRagdollScale = SourceElement.GetAttributeBool("ignoreragdollscale", false);
210 SourceElement.GetAttributeBool("inherittexturescale", false);
211 InheritOrigin = SourceElement.GetAttributeBool("inheritorigin", false);
212 InheritSourceRect = SourceElement.GetAttributeBool("inheritsourcerect", false);
213 DepthLimb = (LimbType)Enum.Parse(typeof(LimbType), SourceElement.GetAttributeString("depthlimb", "None"), true);
214 Sound = SourceElement.GetAttributeString("sound", "");
215 Scale = SourceElement.GetAttributeFloat("scale", 1.0f);
216 Rotation = MathHelper.ToRadians(SourceElement.GetAttributeFloat("rotation", 0.0f));
217 var index = SourceElement.GetAttributePoint("sheetindex", new Point(-1, -1));
218 if (index.X > -1 && index.Y > -1)
219 {
220 SheetIndex = index;
221 }
222
223 HideWearablesOfType = new List<WearableType>();
224 var wearableTypes = SourceElement.GetAttributeStringArray("hidewearablesoftype", null);
225 if (wearableTypes != null && wearableTypes.Length > 0)
226 {
227 foreach (var value in wearableTypes)
228 {
229 if (Enum.TryParse(value, ignoreCase: true, out WearableType wearableType))
230 {
231 HideWearablesOfType.Add(wearableType);
232 }
233 }
234 }
235
236 IsInitialized = true;
237 }

304 public Wearable(Item item, ContentXElement element) : base(item, element)
305 {
306 this.item = item;
307
308 damageModifiers = new List<DamageModifier>();
309 SkillModifiers = new Dictionary<Identifier, float>();
310
311 int spriteCount = element.Elements().Count(x => x.Name.ToString() == "sprite");
312 Variants = element.GetAttributeInt("variants", 0);
313 variant = Rand.Range(1, Variants + 1, Rand.RandSync.ServerAndClient);
314 wearableSprites = new WearableSprite[spriteCount];
315 wearableElements = new ContentXElement[spriteCount];
316 limbType = new LimbType[spriteCount];
317 limb = new Limb[spriteCount];
318 AutoEquipWhenFull = element.GetAttributeBool("autoequipwhenfull", true);
319 DisplayContainedStatus = element.GetAttributeBool("displaycontainedstatus", false);
320 int i = 0;
321 foreach (var subElement in element.Elements())
322 {
323 switch (subElement.Name.ToString().ToLowerInvariant())
324 {
325 case "sprite":
326 if (subElement.GetAttribute("texture") == null)
327 {
328 DebugConsole.ThrowError("Item \"" + item.Name + "\" doesn't have a texture specified!");
329 return;
330 }
331
332 limbType[i] = (LimbType)Enum.Parse(typeof(LimbType),
333 subElement.GetAttributeString("limb", "Head"), true);
334
335 wearableSprites[i] = new WearableSprite(subElement, this, variant);
336 wearableElements[i] = subElement;
337
338 foreach (var lightElement in subElement.Elements())
339 {
340 if (!lightElement.Name.ToString().Equals("lightcomponent", StringComparison.OrdinalIgnoreCase)) { continue; }
341 wearableSprites[i].LightComponent = new LightComponent(item, lightElement)
342 {
343 Parent = this
344 };
345 item.AddComponent(wearableSprites[i].LightComponent);
346 }
347
348 i++;
349 break;
350 case "damagemodifier":
351 damageModifiers.Add(new DamageModifier(subElement, item.Name + ", Wearable"));
352 break;
353 case "skillmodifier":
354 Identifier skillIdentifier = subElement.GetAttributeIdentifier("skillidentifier", Identifier.Empty);
355 float skillValue = subElement.GetAttributeFloat("skillvalue", 0f);
356 if (SkillModifiers.ContainsKey(skillIdentifier))
357 {
358 SkillModifiers[skillIdentifier] += skillValue;
359 }
360 else
361 {
362 SkillModifiers.TryAdd(skillIdentifier, skillValue);
363 }
364 break;
365 case "statvalue":
366 StatTypes statType = CharacterAbilityGroup.ParseStatType(subElement.GetAttributeString("stattype", ""), Name);
367 float statValue = subElement.GetAttributeFloat("value", 0f);
368 if (WearableStatValues.ContainsKey(statType))
369 {
370 WearableStatValues[statType] += statValue;
371 }
372 else
373 {
374 WearableStatValues.TryAdd(statType, statValue);
375 }
376 break;
377 }
378 }

381 public override void Equip(Character character)
382 {
383 foreach (var allowedSlot in allowedSlots)
384 {
385 if (allowedSlot == InvSlotType.Any) { continue; }
386 foreach (Enum value in Enum.GetValues(typeof(InvSlotType)))
387 {
388 var slotType = (InvSlotType)value;
389 if (slotType == InvSlotType.Any || slotType == InvSlotType.None) { continue; }
390 if (allowedSlot.HasFlag(slotType) && !character.Inventory.IsInLimbSlot(item, slotType))
391 {
392 return;
393 }
394 }
395 }
396
397 picker = character;
398
399 for (int i = 0; i < wearableSprites.Length; i++ )
400 {
401 var wearableSprite = wearableSprites[i];
402 if (!wearableSprite.IsInitialized) { wearableSprite.Init(picker); }
403 // If the item is gender specific (it has a different textures for male and female), we have to change the gender here so that the texture is updated.
404 wearableSprite.Picker = picker;
405
406 Limb equipLimb = character.AnimController.GetLimb(limbType[i]);
407 if (equipLimb == null) { continue; }
408
409 if (item.body != null)
410 {
411 item.body.Enabled = false;
412 }
413 IsActive = true;
414 if (wearableSprite.LightComponent != null)
415 {
416 wearableSprite.LightComponent.ParentBody = equipLimb.body;
417 }
418
419 limb[i] = equipLimb;
420 if (!equipLimb.WearingItems.Contains(wearableSprite))
421 {
422 equipLimb.WearingItems.Add(wearableSprite);
423 equipLimb.WearingItems.Sort((i1, i2) => { return i2.Sprite.Depth.CompareTo(i1.Sprite.Depth); });
424 equipLimb.WearingItems.Sort((i1, i2) =>
425 {
426 if (i1?.WearableComponent == null && i2?.WearableComponent == null)
427 {
428 return 0;
429 }
430 else if (i1?.WearableComponent == null)
431 {
432 return -1;
433 }
434 else if (i2?.WearableComponent == null)
435 {
436 return 1;
437 }
438 return i1.WearableComponent.AllowedSlots.Contains(InvSlotType.OuterClothes).CompareTo(i2.WearableComponent.AllowedSlots.Contains(InvSlotType.OuterClothes));
439 });
440 }
441#if CLIENT
442 equipLimb.UpdateWearableTypesToHide();
443#endif
444 }
445 character.OnWearablesChanged();
446 }

458 public override void Unequip(Character character)
459 {
460 if (character == null || character.Removed) { return; }
461 if (picker == null) { return; }
462
463 for (int i = 0; i < wearableSprites.Length; i++)
464 {
465 Limb equipLimb = character.AnimController.GetLimb(limbType[i]);
466 if (equipLimb == null) { continue; }
467
468 if (wearableSprites[i].LightComponent != null)
469 {
470 wearableSprites[i].LightComponent.ParentBody = null;
471 }
472
473 equipLimb.WearingItems.RemoveAll(w => w != null && w == wearableSprites[i]);
474#if CLIENT
475 equipLimb.UpdateWearableTypesToHide();
476#endif
477 limb[i] = null;
478 }
479
480 IsActive = false;
481 }
