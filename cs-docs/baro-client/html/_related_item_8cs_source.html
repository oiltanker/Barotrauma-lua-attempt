96  public bool CheckRequirements(Character character, Item parentItem)
97  {
98  switch (type)
99  {
100  case RelationType.Contained:
101  if (parentItem == null) { return false; }
102  return CheckContained(parentItem);
103  case RelationType.Container:
104  if (parentItem == null || parentItem.Container == null) { return MatchOnEmpty; }
105  return (!ExcludeBroken || parentItem.Container.Condition > 0.0f) && MatchesItem(parentItem.Container);
106  case RelationType.Equipped:
107  if (character == null) { return false; }
108  if (MatchOnEmpty && !character.HeldItems.Any()) { return true; }
109  foreach (Item equippedItem in character.HeldItems)
110  {
111  if (equippedItem == null) { continue; }
112  if ((!ExcludeBroken || equippedItem.Condition > 0.0f) && MatchesItem(equippedItem)) { return true; }
113  }
114  break;
115  case RelationType.Picked:
116  if (character == null || character.Inventory == null) { return false; }
117  foreach (Item pickedItem in character.Inventory.AllItems)
118  {
119  if (MatchesItem(pickedItem)) { return true; }
120  }
121  break;
122  default:
123  return true;
124  }
125 
126  return false;
127  }
128 
129  private bool CheckContained(Item parentItem)
130  {
131  if (parentItem.OwnInventory == null) { return false; }
132 
133  if (MatchOnEmpty && parentItem.OwnInventory.IsEmpty())
134  {
135  return true;
136  }
137 
138  foreach (Item contained in parentItem.ContainedItems)
139  {
140  if (TargetSlot > -1 && parentItem.OwnInventory.FindIndex(contained) != TargetSlot) { continue; }
141  if ((!ExcludeBroken || contained.Condition > 0.0f) && MatchesItem(contained)) { return true; }
142 
143  if (CheckContained(contained)) { return true; }
144  }
145  return false;
146  }

167  public static RelatedItem Load(ContentXElement element, bool returnEmpty, string parentDebugName)
168  {
169  Identifier[] identifiers;
170  if (element.GetAttribute("name") != null)
171  {
172  //backwards compatibility + a console warning
173  DebugConsole.ThrowError("Error in RelatedItem config (" + (string.IsNullOrEmpty(parentDebugName) ? element.ToString() : parentDebugName) + ") - use item tags or identifiers instead of names.");
174  Identifier[] itemNames = element.GetAttributeIdentifierArray("name", Array.Empty<Identifier>());
175  //attempt to convert to identifiers and tags
176  List<Identifier> convertedIdentifiers = new List<Identifier>();
177  foreach (Identifier itemName in itemNames)
178  {
179  var matchingItem = ItemPrefab.Prefabs.Find(me => me.Name == itemName.Value);
180  if (matchingItem != null)
181  {
182  convertedIdentifiers.Add(matchingItem.Identifier);
183  }
184  else
185  {
186  //no matching item found, this must be a tag
187  convertedIdentifiers.Add(itemName);
188  }
189  }
190  identifiers = convertedIdentifiers.ToArray();
191  }
192  else
193  {
194  identifiers = element.GetAttributeIdentifierArray("items", null) ?? element.GetAttributeIdentifierArray("item", null);
195  if (identifiers == null)
196  {
197  identifiers = element.GetAttributeIdentifierArray("identifiers", null) ?? element.GetAttributeIdentifierArray("tags", null);
198  if (identifiers == null)
199  {
200  identifiers = element.GetAttributeIdentifierArray("identifier", null) ?? element.GetAttributeIdentifierArray("tag", Array.Empty<Identifier>());
201  }
202  }
203  }
204 
205  Identifier[] excludedIdentifiers = element.GetAttributeIdentifierArray("excludeditems", null) ?? element.GetAttributeIdentifierArray("excludeditem", null);
206  if (excludedIdentifiers == null)
207  {
208  excludedIdentifiers = element.GetAttributeIdentifierArray("excludedidentifiers", null) ?? element.GetAttributeIdentifierArray("excludedtags", null);
209  if (excludedIdentifiers == null)
210  {
211  excludedIdentifiers = element.GetAttributeIdentifierArray("excludedidentifier", null) ?? element.GetAttributeIdentifierArray("excludedtag", Array.Empty<Identifier>());
212  }
213  }
214 
215 
216  if (identifiers.Length == 0 && excludedIdentifiers.Length == 0 && !returnEmpty) { return null; }
217 
218  RelatedItem ri = new RelatedItem(identifiers, excludedIdentifiers)
219  {
220  ExcludeBroken = element.GetAttributeBool("excludebroken", true),
221  AllowVariants = element.GetAttributeBool("allowvariants", true)
222  };
223  string typeStr = element.GetAttributeString("type", "");
224  if (string.IsNullOrEmpty(typeStr))
225  {
226  switch (element.Name.ToString().ToLowerInvariant())
227  {
228  case "containable":
229  typeStr = "Contained";
230  break;
231  case "suitablefertilizer":
232  case "suitableseed":
233  typeStr = "None";
234  break;
235  }
236  }
237  if (!Enum.TryParse(typeStr, true, out ri.type))
238  {
239  DebugConsole.ThrowError("Error in RelatedItem config (" + parentDebugName + ") - \"" + typeStr + "\" is not a valid relation type.");
240  return null;
241  }
242 
243  ri.MsgTag = element.GetAttributeIdentifier("msg", Identifier.Empty);
244  LocalizedString msg = TextManager.Get(ri.MsgTag);
245  if (!msg.Loaded)
246  {
247  ri.Msg = ri.MsgTag.Value;
248  }
249  else
250  {
251 #if CLIENT
252  foreach (InputType inputType in Enum.GetValues(typeof(InputType)))
253  {
254  msg = msg.Replace("[" + inputType.ToString().ToLowerInvariant() + "]", GameSettings.CurrentConfig.KeyMap.KeyBindText(inputType));
255  }
256  ri.Msg = msg;
257 #endif
258  }
259 
260  foreach (var subElement in element.Elements())
261  {
262  if (!subElement.Name.ToString().Equals("statuseffect", StringComparison.OrdinalIgnoreCase)) { continue; }
263  ri.statusEffects.Add(StatusEffect.Load(subElement, parentDebugName));
264  }
265 
266  ri.IsOptional = element.GetAttributeBool("optional", false);
267  ri.IgnoreInEditor = element.GetAttributeBool("ignoreineditor", false);
268  ri.MatchOnEmpty = element.GetAttributeBool("matchonempty", false);
269  ri.TargetSlot = element.GetAttributeInt("targetslot", -1);
270 
271  return ri;
