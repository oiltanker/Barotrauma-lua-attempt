316  private void HighlightOrderOption(string option)
317  {
318  if (GameMain.GameSession.CrewManager.OrderOptionButtons.Count == 0) return;
319  var order = GameMain.GameSession.CrewManager.OrderOptionButtons[0].UserData as Order;
320 
321  int orderIndex = 0;
322  for (int i = 0; i < GameMain.GameSession.CrewManager.OrderOptionButtons.Count; i++)
323  {
324  if (orderIndex >= order.Options.Length)
325  {
326  orderIndex = 0;
327  }
328  if (order.Options[orderIndex] == option)
329  {
330  if (GameMain.GameSession.CrewManager.OrderOptionButtons[i].FlashTimer <= 0)
331  {
332  GameMain.GameSession.CrewManager.OrderOptionButtons[i].Flash(highlightColor);
333  }
334  }
335 
336  orderIndex++;
337  }
