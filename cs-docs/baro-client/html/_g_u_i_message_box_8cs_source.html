433  public static void AddActiveToGUIUpdateList()
434  {
435  messageBoxTypes ??= (Type[])Enum.GetValues(typeof(Type));
436 
437  foreach (var type in messageBoxTypes)
438  {
439  // Don't display hints when HUD is disabled
440  if (type == Type.Hint && GUI.DisableHUD) { continue; }
441 
442  for (int i = 0; i < MessageBoxes.Count; i++)
443  {
444  if (MessageBoxes[i] == null) { continue; }
445  if (!(MessageBoxes[i] is GUIMessageBox messageBox))
446  {
447  if (type == Type.Default)
448  {
449  // Message box not of type GUIMessageBox is likely the round summary
450  MessageBoxes[i].AddToGUIUpdateList();
451  break;
452  }
453  continue;
454  }
455  if (messageBox.type != type) { continue; }
456  if (!messageBox.DisplayInLoadingScreens && GameMain.Instance.LoadingScreenOpen)
457  {
458  continue;
459  }
460 
461  // These are handled separately in GUI.HandlePersistingElements()
462  if (MessageBoxes[i].UserData as string == "verificationprompt") { continue; }
463  if (MessageBoxes[i].UserData as string == "bugreporter") { continue; }
464 
465  messageBox.AddToGUIUpdateList();
466  break;
467  }

488  protected override void Update(float deltaTime)
489  {
490  if (Draggable)
491  {
492  GUIComponent parent = GUI.MouseOn?.Parent?.Parent;
493  if ((GUI.MouseOn == InnerFrame || InnerFrame.IsParentOf(GUI.MouseOn)) && !(GUI.MouseOn is GUIButton || GUI.MouseOn is GUIColorPicker || GUI.MouseOn is GUITextBox || parent is GUITextBox))
494  {
495  GUI.MouseCursor = CursorState.Move;
496  if (PlayerInput.PrimaryMouseButtonDown())
497  {
498  DraggingPosition = RectTransform.ScreenSpaceOffset.ToVector2() - PlayerInput.MousePosition;
499  }
500  }
501 
502  if (PlayerInput.PrimaryMouseButtonHeld() && DraggingPosition != Vector2.Zero)
503  {
504  GUI.MouseCursor = CursorState.Dragging;
505  RectTransform.ScreenSpaceOffset = (PlayerInput.MousePosition + DraggingPosition).ToPoint();
506  }
507  else
508  {
509  DraggingPosition = Vector2.Zero;
510  }
511  }
512 
513  if (type == Type.InGame || type == Type.Hint)
514  {
515  Vector2 initialPos, defaultPos, endPos;
516  if (type == Type.InGame)
517  {
518  initialPos = new Vector2(0.0f, GameMain.GraphicsHeight);
519  defaultPos = new Vector2(0.0f, HUDLayoutSettings.InventoryAreaLower.Y - InnerFrame.Rect.Height - 20 * GUI.Scale);
520  endPos = new Vector2(GameMain.GraphicsWidth, defaultPos.Y);
521  }
522  else
523  {
524  initialPos = new Vector2(GUI.IntScale(64), -InnerFrame.Rect.Height);
525  defaultPos = new Vector2(initialPos.X, HUDLayoutSettings.ButtonAreaTop.Height + GUI.IntScale(64));
526  endPos = new Vector2(-InnerFrame.Rect.Width, defaultPos.Y);
527  }
528 
529  if (!closing)
530  {
531  Point step = Vector2.SmoothStep(initialPos, defaultPos, openState).ToPoint();
532  InnerFrame.RectTransform.AbsoluteOffset = step;
533  if (BackgroundIcon != null)
534  {
535  BackgroundIcon.RectTransform.AbsoluteOffset = new Point(InnerFrame.Rect.Location.X - (int)(BackgroundIcon.Rect.Size.X / 1.25f), (int)defaultPos.Y - BackgroundIcon.Rect.Size.Y / 2);
536  if (!MathUtils.NearlyEqual(openState, 1.0f))
537  {
538  BackgroundIcon.Color = ToolBox.GradientLerp(openState, Color.Transparent, Color.White);
539  }
540  }
541  if (!(Screen.Selected is RoundSummaryScreen) && !MessageBoxes.Any(mb => mb.UserData is RoundSummary))
542  {
543  openState = Math.Min(openState + deltaTime * 2.0f, 1.0f);
544  }
545 
546  if (GUI.MouseOn != InnerFrame && !InnerFrame.IsParentOf(GUI.MouseOn) && AutoClose)
547  {
548  inGameCloseTimer += deltaTime;
549  }
550 
551  if (inGameCloseTimer >= inGameCloseTime)
552  {
553  Close();
554  }
555  }
556  else
557  {
558  openState += deltaTime * 2.0f;
559  Point step = Vector2.SmoothStep(defaultPos, endPos, openState - 1.0f).ToPoint();
560  InnerFrame.RectTransform.AbsoluteOffset = step;
561  if (BackgroundIcon != null)
562  {
563  BackgroundIcon.Color *= 0.9f;
564  }
565  if (openState >= 2.0f)
566  {
567  Parent?.RemoveChild(this);
568  if (MessageBoxes.Contains(this)) { MessageBoxes.Remove(this); }
569  }
570  }
571 
572  if (newBackgroundIcon != null)
573  {
574  if (!iconSwitching)
575  {
576  if (BackgroundIcon != null)
577  {
578  BackgroundIcon.Color *= 0.9f;
579  if (BackgroundIcon.Color.A == 0)
580  {
581  BackgroundIcon = null;
582  iconSwitching = true;
583  RemoveChild(BackgroundIcon);
584  }
585  }
586  else
587  {
588  iconSwitching = true;
589  }
590  iconState = 0;
591  }
592  else
593  {
594  newBackgroundIcon.SetAsFirstChild();
595  newBackgroundIcon.RectTransform.AbsoluteOffset = new Point(InnerFrame.Rect.Location.X - (int)(newBackgroundIcon.Rect.Size.X / 1.25f), (int)defaultPos.Y - newBackgroundIcon.Rect.Size.Y / 2);
596  newBackgroundIcon.Color = ToolBox.GradientLerp(iconState, Color.Transparent, Color.White);
597  if (newBackgroundIcon.Color.A == 255)
598  {
599  BackgroundIcon = newBackgroundIcon;
600  BackgroundIcon.SetAsFirstChild();
601  newBackgroundIcon = null;
602  iconSwitching = false;
603  }
604 
605  iconState = Math.Min(iconState + deltaTime * 2.0f, 1.0f);
606  }
607  }
608  }
