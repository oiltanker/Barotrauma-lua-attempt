79 public Particle CreateParticle(ParticlePrefab prefab, Vector2 position, Vector2 velocity, float rotation = 0.0f, Hull hullGuess = null, bool drawOnTop = false, float collisionIgnoreTimer = 0f, Tuple<Vector2, Vector2> tracerPoints = null)
80 {
81 if (prefab == null || prefab.Sprites.Count == 0) { return null; }
82
83 if (particleCount >= MaxParticles)
84 {
85 for (int i = 0; i < particleCount; i++)
86 {
87 if (particles[i].Prefab.Priority < prefab.Priority)
88 {
89 RemoveParticle(i);
90 break;
91 }
92 }
93 if (particleCount >= MaxParticles) { return null; }
94 }
95
96 Vector2 particleEndPos = prefab.CalculateEndPosition(position, velocity);
97
98 Vector2 minPos = new Vector2(Math.Min(position.X, particleEndPos.X), Math.Min(position.Y, particleEndPos.Y));
99 Vector2 maxPos = new Vector2(Math.Max(position.X, particleEndPos.X), Math.Max(position.Y, particleEndPos.Y));
100
101 if (tracerPoints != null)
102 {
103 minPos = new Vector2(
104 Math.Min(Math.Min(minPos.X, tracerPoints.Item1.X), tracerPoints.Item2.X),
105 Math.Min(Math.Min(minPos.Y, tracerPoints.Item1.Y), tracerPoints.Item2.Y));
106 maxPos = new Vector2(
107 Math.Max(Math.Max(maxPos.X, tracerPoints.Item1.X), tracerPoints.Item2.X),
108 Math.Max(Math.Max(maxPos.Y, tracerPoints.Item1.Y), tracerPoints.Item2.Y));
109 }
110
111 Rectangle expandedViewRect = MathUtils.ExpandRect(cam.WorldView, MaxOutOfViewDist);
112
113 if (minPos.X > expandedViewRect.Right || maxPos.X < expandedViewRect.X) { return null; }
114 if (minPos.Y > expandedViewRect.Y || maxPos.Y < expandedViewRect.Y - expandedViewRect.Height) { return null; }
115
116 if (particles[particleCount] == null) { particles[particleCount] = new Particle(); }
117
118 particles[particleCount].Init(prefab, position, velocity, rotation, hullGuess, drawOnTop, collisionIgnoreTimer, tracerPoints: tracerPoints);
119
120 particleCount++;
121
122 return particles[particleCount - 1];
123 }

158 {

160
161 for (int i = 0; i < particleCount; i++)
162 {
163 bool remove;
164 try
165 {

168 catch (Exception e)

197 public void Draw(SpriteBatch spriteBatch, bool inWater, bool? inSub, ParticleBlendState blendState)
198 {
199 ParticlePrefab.DrawTargetType drawTarget = inWater ? ParticlePrefab.DrawTargetType.Water : ParticlePrefab.DrawTargetType.Air;
200
201 for (int i = 0; i < particleCount; i++)
202 {
203 var particle = particles[i];
204 if (particle.BlendState != blendState) { continue; }
205 //equivalent to !particles[i].DrawTarget.HasFlag(drawTarget) but garbage free and faster
206 if ((particle.DrawTarget & drawTarget) == 0) { continue; }
207 if (inSub.HasValue)
208 {
209 bool isOutside = particle.CurrentHull == null;
210 if (!particle.DrawOnTop && isOutside == inSub.Value)
211 {
212 continue;
213 }
214 }
215
216 particles[i].Draw(spriteBatch);
217 }

220 public void RemoveByPrefab(ParticlePrefab prefab)
221 {
222 if (particles == null) { return; }
223 for (int i = particles.Length - 1; i >= 0; i--)
224 {
225 if (particles[i]?.Prefab == prefab)
226 {
227 if (i < particleCount) { particleCount--; }
228
229 Particle swap = particles[particleCount];
230 particles[particleCount] = null;
231 particles[i] = swap;
232 }
233 }
