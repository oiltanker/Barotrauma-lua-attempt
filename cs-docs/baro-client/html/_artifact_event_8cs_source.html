67 private void SpawnItem()
68 {
69 item = new Item(itemPrefab, spawnPos, null);
70 item.body.FarseerBody.BodyType = FarseerPhysics.BodyType.Kinematic;
71
72 //try to find an artifact holder and place the artifact inside it
73 foreach (Item it in Item.ItemList)
74 {
75 if (it.Submarine != null || !it.HasTag("artifactholder")) continue;
76
77 var itemContainer = it.GetComponent<Items.Components.ItemContainer>();
78 if (itemContainer == null) continue;
79 if (itemContainer.Combine(item, user: null)) break; // Placement successful
80 }
