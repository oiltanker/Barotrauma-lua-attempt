134 partial void UpdateProjSpecific(float deltaTime)
135 {
136 float rotationRad = MathHelper.ToRadians(item.Rotation);
137 if (FlowPercentage < 0.0f)
138 {
139 foreach (var (position, emitter) in pumpOutEmitters)
140 {
141 if (item.CurrentHull != null && item.CurrentHull.Surface < item.Rect.Location.Y + position.Y) { continue; }
142
143 //only emit "pump out" particles when underwater
144 Vector2 relativeParticlePos = (item.WorldRect.Location.ToVector2() + position * item.Scale) - item.WorldPosition;
145 relativeParticlePos = MathUtils.RotatePoint(relativeParticlePos, item.FlippedX ? rotationRad : -rotationRad);
146 float angle = -rotationRad;
147 if (item.FlippedX)
148 {
149 relativeParticlePos.X = -relativeParticlePos.X;
150 angle += MathHelper.Pi;
151 }
152 if (item.FlippedY)
153 {
154 relativeParticlePos.Y = -relativeParticlePos.Y;
155 }
156
157 emitter.Emit(deltaTime, item.WorldPosition + relativeParticlePos, item.CurrentHull, angle,
158 velocityMultiplier: MathHelper.Lerp(0.5f, 1.0f, -FlowPercentage / 100.0f));
159 }
160 }
161 else if (FlowPercentage > 0.0f)
162 {
163 foreach (var (position, emitter) in pumpInEmitters)
164 {
165 Vector2 relativeParticlePos = (item.WorldRect.Location.ToVector2() + position * item.Scale) - item.WorldPosition;
166 relativeParticlePos = MathUtils.RotatePoint(relativeParticlePos, item.FlippedX ? rotationRad : -rotationRad);
167 float angle = -rotationRad;
168 if (item.FlippedX)
169 {
170 relativeParticlePos.X = -relativeParticlePos.X;
171 angle += MathHelper.Pi;
172 }
173 if (item.FlippedY)
174 {
175 relativeParticlePos.Y = -relativeParticlePos.Y;
176 }
177 emitter.Emit(deltaTime, item.WorldPosition + relativeParticlePos, item.CurrentHull, angle,
178 velocityMultiplier: MathHelper.Lerp(0.5f, 1.0f, FlowPercentage / 100.0f));
179 }
180 }

226 public void ClientEventRead(IReadMessage msg, float sendingTime)
227 {
228 int msgStartPos = msg.BitPosition;
229
230 float flowPercentage = msg.ReadRangedInteger(-10, 10) * 10.0f;
231 bool isActive = msg.ReadBoolean();
232 bool hijacked = msg.ReadBoolean();
233 float? targetLevel;
234 if (msg.ReadBoolean())
235 {
236 targetLevel = msg.ReadSingle();
237 }
238 else
239 {
240 targetLevel = null;
241 }
242
243 if (correctionTimer > 0.0f)
244 {
245 int msgLength = msg.BitPosition - msgStartPos;
246 msg.BitPosition = msgStartPos;
247 StartDelayedCorrection(msg.ExtractBits(msgLength), sendingTime);
248 return;
249 }
250
251 FlowPercentage = flowPercentage;
252 IsActive = isActive;
253 Hijacked = hijacked;
254 TargetLevel = targetLevel;
255 }
