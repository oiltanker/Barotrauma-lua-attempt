584 SettingsMenu.Instance?.Close();
585 #warning TODO: reimplement settings confirmation dialog

588 {
589 case Tab.NewGame:

592 selectedTab = Tab.Empty;

606 campaignSetupUI.UpdateSubList(SubmarineInfo.SavedSubmarines);
607 break;
608 case Tab.LoadGame:
609 campaignSetupUI.UpdateLoadMenu();
610 break;
611 case Tab.Settings:
612 SettingsMenu.Create(menuTabs[Tab.Settings].RectTransform);
613 break;
614 case Tab.JoinServer:
