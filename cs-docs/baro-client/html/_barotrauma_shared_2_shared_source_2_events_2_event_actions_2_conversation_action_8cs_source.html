82 public ConversationAction(ScriptedEvent parentEvent, ContentXElement element) : base(parentEvent, element)
83 {
84 actionCount++;
85 Identifier = actionCount;
86 Options = new List<SubactionGroup>();
87 foreach (var elem in element.Elements())
88 {
89 if (elem.Name.LocalName.Equals("option", StringComparison.InvariantCultureIgnoreCase))
90 {
91 Options.Add(new SubactionGroup(ParentEvent, elem));
92 }
93 else if (elem.Name.LocalName.Equals("interrupt", StringComparison.InvariantCultureIgnoreCase))
94 {
95 Interrupted = new SubactionGroup(ParentEvent, elem);
96 }

105 public override bool IsFinished(ref string goTo)
106 {
107 if (interrupt)
108 {
109 if (dialogOpened)
110 {
111#if CLIENT
112 dialogBox?.Close();
113 GUIMessageBox.MessageBoxes.ForEachMod(mb =>
114 {
115 if (mb.UserData as string == "ConversationAction")
116 {
117 (mb as GUIMessageBox)?.Close();

120#else
121 foreach (Client c in GameMain.Server.ConnectedClients)
122 {
123 if (c.InGame && c.Character != null) { ServerWrite(speaker, c); }
124 }
125#endif
126 ResetSpeaker();
127 dialogOpened = false;
128 }
129
130 if (Interrupted == null)
131 {
132 if (EndEventIfInterrupted) { goTo = "_end"; }
133 return true;
134 }
135 else
136 {
137 return Interrupted.IsFinished(ref goTo);
138 }
139 }
140
141 if (selectedOption >= 0)
142 {
143 if (!Options.Any() || Options[selectedOption].IsFinished(ref goTo))
144 {
145 ResetSpeaker();
146 return true;
147 }
148 }
149 return false;
150 }

178 private void ResetSpeaker()
179 {
180 if (speaker == null) { return; }
181 speaker.CampaignInteractionType = CampaignMode.InteractionType.None;
182 speaker.ActiveConversation = null;
183 speaker.SetCustomInteract(null, null);
184#if SERVER
185 GameMain.NetworkMember.CreateEntityEvent(speaker, new Character.AssignCampaignInteractionEventData());
186#endif
187 var humanAI = speaker.AIController as HumanAIController;
188 if (humanAI != null && !speaker.IsDead && !speaker.Removed)
189 {
190 humanAI.ClearForcedOrder();
191 if (prevIdleObjective != null) { humanAI.ObjectiveManager.AddObjective(prevIdleObjective); }
192 if (prevGotoObjective != null) { humanAI.ObjectiveManager.AddObjective(prevGotoObjective); }
193 humanAI.ObjectiveManager.SortObjectives();
194 }

204 public override void Update(float deltaTime)
205 {
206 lastActiveTime = Timing.TotalTime;
207 if (interrupt)
208 {
209 Interrupted?.Update(deltaTime);
210 }
211 else if (selectedOption < 0)
212 {
213 if (dialogOpened)
214 {
215#if CLIENT
216 if (GUIMessageBox.MessageBoxes.Any(mb => mb.UserData as string == "ConversationAction"))
217 {
218 Character.DisableControls = true;
219 }
220 else
221 {
222 Reset();
223 }
224#endif
225 if (ShouldInterrupt())
226 {
227 ResetSpeaker();
228 interrupt = true;
229 }
230 return;
231 }
232
233 if (!SpeakerTag.IsEmpty)
234 {
235 if (speaker != null && !speaker.Removed && speaker.CampaignInteractionType == CampaignMode.InteractionType.Talk && speaker.ActiveConversation?.ParentEvent != this.ParentEvent) { return; }
236 speaker = ParentEvent.GetTargets(SpeakerTag).FirstOrDefault(e => e is Character) as Character;
237 if (speaker == null || speaker.Removed)
238 {
239 return;
240 }
241 //some conversation already assigned to the speaker, wait for it to be removed
242 if (speaker.CampaignInteractionType == CampaignMode.InteractionType.Talk && speaker.ActiveConversation?.ParentEvent != this.ParentEvent)
243 {
244 return;
245 }
246 else if (!WaitForInteraction)
247 {
248 TryStartConversation(speaker);
249 }
250 else if (speaker.ActiveConversation != this)
251 {
252 speaker.CampaignInteractionType = CampaignMode.InteractionType.Talk;
253 speaker.ActiveConversation = this;
254#if CLIENT
255 speaker.SetCustomInteract(
256 TryStartConversation,
257 TextManager.GetWithVariable("CampaignInteraction.Talk", "[key]", GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Use)));
258#else
259 speaker.SetCustomInteract(
260 TryStartConversation,
261 TextManager.Get("CampaignInteraction.Talk"));
262 GameMain.NetworkMember.CreateEntityEvent(speaker, new Character.AssignCampaignInteractionEventData());
263#endif
264 }
265 return;
266 }
267 else
268 {
269 TryStartConversation(null);
270 }
271 }
272 else
273 {
274 if (ShouldInterrupt())
275 {
276 ResetSpeaker();
277 interrupt = true;
278 }
279 else if (Options.Any())
280 {
281 Options[selectedOption].Update(deltaTime);
282 }
283 }

286 private bool ShouldInterrupt()
287 {
288 IEnumerable<Entity> targets = Enumerable.Empty<Entity>();
289 if (!TargetTag.IsEmpty)
290 {
291 targets = ParentEvent.GetTargets(TargetTag).Where(e => IsValidTarget(e));
292 if (!targets.Any()) { return true; }
293 }
294
295 if (speaker != null)
296 {
297 if (!TargetTag.IsEmpty)
298 {
299 if (targets.All(t => Vector2.DistanceSquared(t.WorldPosition, speaker.WorldPosition) > InterruptDistance * InterruptDistance)) { return true; }
300 }
301 if (speaker.AIController is HumanAIController humanAI && !humanAI.AllowCampaignInteraction())
302 {
303 return true;
304 }
305 return speaker.Removed || speaker.IsDead || speaker.IsIncapacitated;
306 }
307
308 return false;
309 }

311 private bool IsValidTarget(Entity e)
312 {
313 bool isValid = e is Character character && !character.Removed && !character.IsDead && !character.IsIncapacitated &&
314 (e == Character.Controlled || character.IsRemotePlayer);
315#if SERVER
316 UpdateIgnoredClients();
317 isValid &= !ignoredClients.Keys.Any(c => c.Character == e);
318#elif CLIENT
319 isValid &= (e != Character.Controlled || !GUI.InputBlockingMenuOpen);
320#endif
321 return isValid;
322 }

324 private void TryStartConversation(Character speaker, Character targetCharacter = null)
325 {
326 IEnumerable<Entity> targets = Enumerable.Empty<Entity>();
327 if (!TargetTag.IsEmpty)
328 {
329 targets = ParentEvent.GetTargets(TargetTag).Where(e => IsValidTarget(e));
330 if (!targets.Any() || IsBlockedByAnotherConversation(targets)) { return; }
331 }
332
333 if (speaker?.AIController is HumanAIController humanAI)
334 {
335 prevIdleObjective = humanAI.ObjectiveManager.GetObjective<AIObjectiveIdle>();
336 prevGotoObjective = humanAI.ObjectiveManager.GetObjective<AIObjectiveGoTo>();
337 humanAI.SetForcedOrder(
338 new Order(OrderPrefab.Prefabs["wait"], Barotrauma.Identifier.Empty, null, orderGiver: null));
339 if (targets.Any())
340 {
341 Entity closestTarget = null;
342 float closestDist = float.MaxValue;
343 foreach (Entity entity in targets)
344 {
345 float dist = Vector2.DistanceSquared(entity.WorldPosition, speaker.WorldPosition);
346 if (dist < closestDist)
347 {
348 closestTarget = entity;
349 closestDist = dist;
350 }
351 }
352 if (closestTarget != null)
353 {
354 humanAI.FaceTarget(closestTarget);
355 }
356 }
357 }
358
359 if (targetCharacter != null && !InvokerTag.IsEmpty)
360 {
361 ParentEvent.AddTarget(InvokerTag, targetCharacter);
362 }
363
364 ShowDialog(speaker, targetCharacter);
365
366 dialogOpened = true;
367 if (speaker != null)
368 {
369 speaker.CampaignInteractionType = CampaignMode.InteractionType.None;
370 speaker.SetCustomInteract(null, null);
371#if SERVER
372 GameMain.NetworkMember.CreateEntityEvent(speaker, new Character.AssignCampaignInteractionEventData());
373#endif
374 }
