286  private bool ShouldInterrupt()
287  {
288  IEnumerable<Entity> targets = Enumerable.Empty<Entity>();
289  if (!TargetTag.IsEmpty)
290  {
291  targets = ParentEvent.GetTargets(TargetTag).Where(e => IsValidTarget(e));
292  if (!targets.Any()) { return true; }
293  }
294 
295  if (speaker != null)
296  {
297  if (!TargetTag.IsEmpty)
298  {
299  if (targets.All(t => Vector2.DistanceSquared(t.WorldPosition, speaker.WorldPosition) > InterruptDistance * InterruptDistance)) { return true; }
300  }
301  if (speaker.AIController is HumanAIController humanAI && !humanAI.AllowCampaignInteraction())
302  {
303  return true;
304  }
305  return speaker.Removed || speaker.IsDead || speaker.IsIncapacitated;
306  }
307 
308  return false;
309  }

324  private void TryStartConversation(Character speaker, Character targetCharacter = null)
325  {
326  IEnumerable<Entity> targets = Enumerable.Empty<Entity>();
327  if (!TargetTag.IsEmpty)
328  {
329  targets = ParentEvent.GetTargets(TargetTag).Where(e => IsValidTarget(e));
330  if (!targets.Any() || IsBlockedByAnotherConversation(targets)) { return; }
331  }
332 
333  if (speaker?.AIController is HumanAIController humanAI)
334  {
335  prevIdleObjective = humanAI.ObjectiveManager.GetObjective<AIObjectiveIdle>();
336  prevGotoObjective = humanAI.ObjectiveManager.GetObjective<AIObjectiveGoTo>();
337  humanAI.SetForcedOrder(
338  new Order(OrderPrefab.Prefabs["wait"], Barotrauma.Identifier.Empty, null, orderGiver: null));
339  if (targets.Any())
340  {
341  Entity closestTarget = null;
342  float closestDist = float.MaxValue;
343  foreach (Entity entity in targets)
344  {
345  float dist = Vector2.DistanceSquared(entity.WorldPosition, speaker.WorldPosition);
346  if (dist < closestDist)
347  {
348  closestTarget = entity;
349  closestDist = dist;
350  }
351  }
352  if (closestTarget != null)
353  {
354  humanAI.FaceTarget(closestTarget);
355  }
356  }
357  }
358 
359  if (targetCharacter != null && !InvokerTag.IsEmpty)
360  {
361  ParentEvent.AddTarget(InvokerTag, targetCharacter);
362  }
363 
364  ShowDialog(speaker, targetCharacter);
365 
366  dialogOpened = true;
367  if (speaker != null)
368  {
369  speaker.CampaignInteractionType = CampaignMode.InteractionType.None;
370  speaker.SetCustomInteract(null, null);
371 #if SERVER
372  GameMain.NetworkMember.CreateEntityEvent(speaker, new Character.AssignCampaignInteractionEventData());
373 #endif
374  }
375  }
