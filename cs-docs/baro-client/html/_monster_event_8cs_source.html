171  position.PositionType != Level.PositionType.SidePath)

173  continue;
174  }

188  private void FindSpawnPosition(bool affectSubImmediately)
189  {
190  if (disallowed) { return; }
191 
192  spawnPos = Vector2.Zero;
193  var availablePositions = GetAvailableSpawnPositions();
194  var chosenPosition = new Level.InterestingPosition(Point.Zero, Level.PositionType.MainPath, isValid: false);
195  bool isRuinOrWreck = SpawnPosType.HasFlag(Level.PositionType.Ruin) || SpawnPosType.HasFlag(Level.PositionType.Wreck);
196  if (affectSubImmediately && !isRuinOrWreck && !SpawnPosType.HasFlag(Level.PositionType.Abyss))
197  {
198  if (availablePositions.None())
199  {
200  //no suitable position found, disable the event
201  spawnPos = null;
202  Finished();
203  return;
204  }
205  Submarine refSub = GetReferenceSub();
206  if (Submarine.MainSubs.Length == 2 && Submarine.MainSubs[1] != null)
207  {
208  refSub = Submarine.MainSubs.GetRandom(Rand.RandSync.Unsynced);
209  }
210  float closestDist = float.PositiveInfinity;
211  //find the closest spawnposition that isn't too close to any of the subs
212  foreach (var position in availablePositions)
213  {
214  Vector2 pos = position.Position.ToVector2();
215  float dist = Vector2.DistanceSquared(pos, refSub.WorldPosition);
216  foreach (Submarine sub in Submarine.Loaded)
217  {
218  if (sub.Info.Type != SubmarineType.Player && sub != GameMain.NetworkMember?.RespawnManager?.RespawnShuttle) { continue; }
219 
220  float minDistToSub = GetMinDistanceToSub(sub);
221  if (dist < minDistToSub * minDistToSub) { continue; }
222 
223  if (closestDist == float.PositiveInfinity)
224  {
225  closestDist = dist;
226  chosenPosition = position;
227  continue;
228  }
229 
230  //chosen position behind the sub -> override with anything that's closer or to the right
231  if (chosenPosition.Position.X < refSub.WorldPosition.X)
232  {
233  if (dist < closestDist || pos.X > refSub.WorldPosition.X)
234  {
235  closestDist = dist;
236  chosenPosition = position;
237  }
238  }
239  //chosen position ahead of the sub -> only override with a position that's also ahead
240  else if (chosenPosition.Position.X > refSub.WorldPosition.X)
241  {
242  if (dist < closestDist && pos.X > refSub.WorldPosition.X)
243  {
244  closestDist = dist;
245  chosenPosition = position;
246  }
247  }
248  }
249  }
250  //only found a spawnpos that's very far from the sub, pick one that's closer
251  //and wait for the sub to move further before spawning
252  if (closestDist > 15000.0f * 15000.0f)
253  {
254  foreach (var position in availablePositions)
255  {
256  float dist = Vector2.DistanceSquared(position.Position.ToVector2(), refSub.WorldPosition);
257  if (dist < closestDist)
258  {
259  closestDist = dist;
260  chosenPosition = position;
261  }
262  }
263  }
264  }
265  else
266  {
267  if (!isRuinOrWreck)
268  {
269  float minDistance = 20000;
270  var refSub = GetReferenceSub();
271  availablePositions.RemoveAll(p => Vector2.DistanceSquared(refSub.WorldPosition, p.Position.ToVector2()) < minDistance * minDistance);
272  if (Submarine.MainSubs.Length > 1)
273  {
274  for (int i = 1; i < Submarine.MainSubs.Length; i++)
275  {
276  if (Submarine.MainSubs[i] == null) { continue; }
277  availablePositions.RemoveAll(p => Vector2.DistanceSquared(Submarine.MainSubs[i].WorldPosition, p.Position.ToVector2()) < minDistance * minDistance);
278  }
279  }
280  }
281  if (availablePositions.None())
282  {
283  //no suitable position found, disable the event
284  spawnPos = null;
285  Finished();
286  return;
287  }
288  chosenPosition = availablePositions.GetRandomUnsynced();
289  }
290  if (chosenPosition.IsValid)
291  {
292  spawnPos = chosenPosition.Position.ToVector2();
293  if (chosenPosition.Submarine != null || chosenPosition.Ruin != null)
294  {
295  bool ignoreSubmarine = chosenPosition.Ruin != null;
296  var spawnPoint = WayPoint.GetRandom(SpawnType.Enemy, sub: chosenPosition.Submarine, useSyncedRand: false, spawnPointTag: spawnPointTag, ignoreSubmarine: ignoreSubmarine);
297  if (spawnPoint != null)
298  {
299  if (!ignoreSubmarine)
300  {
301  System.Diagnostics.Debug.Assert(spawnPoint.Submarine == chosenPosition.Submarine);
302  }
303  spawnPos = spawnPoint.WorldPosition;
304  }
305  else
306  {
307  //no suitable position found, disable the event
308  spawnPos = null;
309  Finished();
310  return;
311  }
312  }
313  else if (chosenPosition.PositionType == Level.PositionType.MainPath || chosenPosition.PositionType == Level.PositionType.SidePath)
314  {
315  if (offset > 0)
316  {
317  Vector2 dir;
318  var waypoints = WayPoint.WayPointList.FindAll(wp => wp.Submarine == null && wp.Ruin == null);
319  var nearestWaypoint = waypoints.OrderBy(wp => Vector2.DistanceSquared(wp.WorldPosition, spawnPos.Value)).FirstOrDefault();
320  if (nearestWaypoint != null)
321  {
322  int currentIndex = waypoints.IndexOf(nearestWaypoint);
323  var nextWaypoint = waypoints[Math.Min(currentIndex + 20, waypoints.Count - 1)];
324  dir = Vector2.Normalize(nextWaypoint.WorldPosition - nearestWaypoint.WorldPosition);
325  // Ensure that the spawn position is not offset to the left.
326  if (dir.X < 0)
327  {
328  dir.X = 0;
329  }
330  }
331  else
332  {
333  dir = new Vector2(1, Rand.Range(-1f, 1f));
334  }
335  Vector2 targetPos = spawnPos.Value + dir * offset;
336  var targetWaypoint = waypoints.OrderBy(wp => Vector2.DistanceSquared(wp.WorldPosition, targetPos)).FirstOrDefault();
337  if (targetWaypoint != null)
338  {
339  spawnPos = targetWaypoint.WorldPosition;
340  }
341  }
342  // Ensure that the position is not inside a submarine (in practice wrecks).
343  if (Submarine.Loaded.Any(s => ToolBox.GetWorldBounds(s.Borders.Center, s.Borders.Size).ContainsWorld(spawnPos.Value)))
344  {
345  //no suitable position found, disable the event
346  spawnPos = null;
347  Finished();
348  return;
349  }
350  }
351  spawnPending = true;
352  }

388  if (spawnPending)

390  //wait until there are no submarines at the spawnpos
391  if (SpawnPosType.HasFlag(Level.PositionType.MainPath) || SpawnPosType.HasFlag(Level.PositionType.SidePath) || SpawnPosType.HasFlag(Level.PositionType.Abyss))
392  {
393  foreach (Submarine submarine in Submarine.Loaded)
394  {
395  if (submarine.Info.Type != SubmarineType.Player) { continue; }
396  float minDist = GetMinDistanceToSub(submarine);
397  if (Vector2.DistanceSquared(submarine.WorldPosition, spawnPos.Value) < minDist * minDist) { return; }
398  }
399  }
400  float minDistance = Prefab.SpawnDistance;
401  if (minDistance <= 0)
402  {
403  if (SpawnPosType.HasFlag(Level.PositionType.Cave))
404  {
405  minDistance = 8000;
406  }
407  else if (SpawnPosType.HasFlag(Level.PositionType.Ruin))
408  {
409  minDistance = 5000;
410  }
411  else if (SpawnPosType.HasFlag(Level.PositionType.Wreck))
412  {
413  minDistance = 3000;
414  }
415  }
416  if (minDistance > 0)
417  {
418  bool someoneNearby = false;
419  foreach (Submarine submarine in Submarine.Loaded)
420  {
421  if (submarine.Info.Type != SubmarineType.Player) { continue; }
422  if (Vector2.DistanceSquared(submarine.WorldPosition, spawnPos.Value) < MathUtils.Pow2(minDistance))
423  {
424  someoneNearby = true;
425  break;
426  }
427  }
428  foreach (Character c in Character.CharacterList)
429  {
430  if (c == Character.Controlled || c.IsRemotePlayer)
431  {
432  if (Vector2.DistanceSquared(c.WorldPosition, spawnPos.Value) < MathUtils.Pow2(minDistance))
433  {
434  someoneNearby = true;
435  break;
436  }
437  }
438  }
439  if (!someoneNearby) { return; }
440  }
441 
442 
443  if (SpawnPosType.HasFlag(Level.PositionType.Abyss) || SpawnPosType.HasFlag(Level.PositionType.AbyssCave))
444  {
445  bool anyInAbyss = false;
446  foreach (Submarine submarine in Submarine.Loaded)
447  {
448  if (submarine.Info.Type != SubmarineType.Player || submarine == GameMain.NetworkMember?.RespawnManager?.RespawnShuttle) { continue; }
449  if (submarine.WorldPosition.Y < 0)
450  {
451  anyInAbyss = true;
452  break;
453  }
454  }
455  if (!anyInAbyss) { return; }
456  }
457 
458  spawnPending = false;
459 
460  //+1 because Range returns an integer less than the max value
461  int amount = Rand.Range(MinAmount, MaxAmount + 1);
462  monsters = new List<Character>();
463  float scatterAmount = scatter;
464  if (SpawnPosType.HasFlag(Level.PositionType.SidePath))
465  {
466  var sidePaths = Level.Loaded.Tunnels.Where(t => t.Type == Level.TunnelType.SidePath);
467  if (sidePaths.Any())
468  {
469  scatterAmount = Math.Min(scatter, sidePaths.Min(t => t.MinWidth) / 2);
470  }
471  else
472  {
473  scatterAmount = scatter;
474  }
475  }
476  else if (!SpawnPosType.HasFlag(Level.PositionType.MainPath))
477  {
478  scatterAmount = 0;
479  }
480 
481  for (int i = 0; i < amount; i++)
482  {
483  string seed = Level.Loaded.Seed + i.ToString();
484  CoroutineManager.Invoke(() =>
485  {
486  //round ended before the coroutine finished
487  if (GameMain.GameSession == null || Level.Loaded == null) { return; }
488 
489  System.Diagnostics.Debug.Assert(GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer, "Clients should not create monster events.");
490 
491  Vector2 pos = spawnPos.Value + Rand.Vector(scatterAmount);
492  if (scatterAmount > 0)
493  {
494  if (Submarine.Loaded.Any(s => ToolBox.GetWorldBounds(s.Borders.Center, s.Borders.Size).ContainsWorld(pos)))
495  {
496  // Can't use the offset position, let's use the exact spawn position.
497  pos = spawnPos.Value;
498  }
499  else if (Level.Loaded.Ruins.Any(r => ToolBox.GetWorldBounds(r.Area.Center, r.Area.Size).ContainsWorld(pos)))
500  {
501  // Can't use the offset position, let's use the exact spawn position.
502  pos = spawnPos.Value;
503  }
504  }
505 
506  Character createdCharacter = Character.Create(SpeciesName, pos, seed, characterInfo: null, isRemotePlayer: false, hasAi: true, createNetworkEvent: true);
507  var eventManager = GameMain.GameSession.EventManager;
508  if (eventManager != null)
509  {

512  eventManager.CumulativeMonsterStrengthMain += createdCharacter.Params.AI.CombatStrength;
513  eventManager.AddTimeStamp(this);
514  }
515  else if (SpawnPosType.HasFlag(Level.PositionType.Ruin))
516  {
517  eventManager.CumulativeMonsterStrengthRuins += createdCharacter.Params.AI.CombatStrength;
518  }
519  else if (SpawnPosType.HasFlag(Level.PositionType.Wreck))
520  {
521  eventManager.CumulativeMonsterStrengthWrecks += createdCharacter.Params.AI.CombatStrength;
522  }
523  else if (SpawnPosType.HasFlag(Level.PositionType.Cave))
524  {
525  eventManager.CumulativeMonsterStrengthCaves += createdCharacter.Params.AI.CombatStrength;
526  }
527  }
528  if (GameMain.GameSession.IsCurrentLocationRadiated())
529  {
530  AfflictionPrefab radiationPrefab = AfflictionPrefab.RadiationSickness;
531  Affliction affliction = new Affliction(radiationPrefab, radiationPrefab.MaxStrength);
532  createdCharacter?.CharacterHealth.ApplyAffliction(null, affliction);
533  // TODO test multiplayer
534  createdCharacter?.Kill(CauseOfDeathType.Affliction, affliction, log: false);
535  }

537 
538  if (monsters.Count == amount)
539  {
540  spawnReady = true;
541  //this will do nothing if the monsters have no swarm behavior defined,
542  //otherwise it'll make the spawned characters act as a swarm
543  SwarmBehavior.CreateSwarm(monsters.Cast<AICharacter>());
544  DebugConsole.NewMessage($"Spawned: {ToString()}. Strength: {StringFormatter.FormatZeroDecimal(monsters.Sum(m => m.Params.AI.CombatStrength))}.", Color.LightBlue, debugOnly: true);
545  }
546 
547  if (GameMain.GameSession != null)
548  {

550  $"MonsterSpawn:{GameMain.GameSession.GameMode?.Preset?.Identifier.Value ?? "none"}:{Level.Loaded?.LevelData?.Biome?.Identifier.Value ?? "none"}:{SpawnPosType}:{SpeciesName}",
551  value: Timing.TotalTime - GameMain.GameSession.RoundStartTime);
552  }
553  }, delayBetweenSpawns * i);
554  }
555  }
556

562 #endif
563

566  {

568  {
569  monstersDead = false;
570

579  }
