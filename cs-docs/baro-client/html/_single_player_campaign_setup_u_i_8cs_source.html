270 public void RandomizeCrew()
271 {
272 var characterInfos = new List<(CharacterInfo Info, JobPrefab Job)>();
273 foreach (JobPrefab jobPrefab in JobPrefab.Prefabs)
274 {
275 for (int i = 0; i < jobPrefab.InitialCount; i++)
276 {
277 var variant = Rand.Range(0, jobPrefab.Variants);
278 characterInfos.Add((new CharacterInfo(CharacterPrefab.HumanSpeciesName, jobOrJobPrefab: jobPrefab, variant: variant), jobPrefab));
279 }
280 }
281 if (characterInfos.Count == 0)
282 {
283 DebugConsole.ThrowError($"No starting crew found! If you're using mods, it may be that the mods have overridden the vanilla jobs without specifying which types of characters the starting crew should consist of. If you're the developer of the mod, ensure that you've set the {nameof(JobPrefab.InitialCount)} properties for the custom jobs.");
284 DebugConsole.AddWarning("Choosing the first available jobs as the starting crew...");
285 foreach (JobPrefab jobPrefab in JobPrefab.Prefabs)
286 {
287 var variant = Rand.Range(0, jobPrefab.Variants);
288 characterInfos.Add((new CharacterInfo(CharacterPrefab.HumanSpeciesName, jobOrJobPrefab: jobPrefab, variant: variant), jobPrefab));
289 if (characterInfos.Count >= 3) { break; }
290 }
291 }
292 characterInfos.Sort((a, b) => Math.Sign(b.Job.MinKarma - a.Job.MinKarma));
293
294 characterInfoColumns.ClearChildren();
295 CharacterMenus?.ForEach(m => m.Dispose());
296 CharacterMenus = new CharacterInfo.AppearanceCustomizationMenu[characterInfos.Count];
297
298 for (int i = 0; i < characterInfos.Count; i++)
299 {
300 var subLayout = new GUILayoutGroup(new RectTransform(new Vector2(1.0f / characterInfos.Count, 1.0f),
301 characterInfoColumns.RectTransform));
302
303 var (characterInfo, job) = characterInfos[i];

305 characterInfo.CreateIcon(new RectTransform(new Vector2(1.0f, 0.275f), subLayout.RectTransform));
306
307 var jobTextContainer =
308 new GUIFrame(new RectTransform(new Vector2(1.0f, 0.05f), subLayout.RectTransform), style: null);

322 sender.UserData = "random";

326 characterInfo.Name = sender.Text;

339 new CharacterInfo.AppearanceCustomizationMenu(characterInfo, customizationFrame, hasIcon: false)

416 private bool FinishSetup(GUIButton btn, object userdata)
417 {
418 if (string.IsNullOrWhiteSpace(saveNameBox.Text))
419 {
420 saveNameBox.Flash(GUIStyle.Red);
421 return false;
422 }
423
424 SubmarineInfo selectedSub = null;
425
426 if (!(subList.SelectedData is SubmarineInfo)) { return false; }
427 selectedSub = subList.SelectedData as SubmarineInfo;
428
429 if (selectedSub.SubmarineClass == SubmarineClass.Undefined)
430 {
431 new GUIMessageBox(TextManager.Get("error"), TextManager.Get("undefinedsubmarineselected"));
432 return false;
433 }
434
435 if (string.IsNullOrEmpty(selectedSub.MD5Hash.StringRepresentation))
436 {
437 ((GUITextBlock)subList.SelectedComponent).TextColor = Color.DarkRed * 0.8f;
438 subList.SelectedComponent.CanBeFocused = false;
439 subList.Deselect();
440 return false;
441 }
442
443 string savePath = SaveUtil.CreateSavePath(SaveUtil.SaveType.Singleplayer, saveNameBox.Text);
444 bool hasRequiredContentPackages = selectedSub.RequiredContentPackagesInstalled;
445
446 CampaignSettings settings = new CampaignSettings();
447 settings.RadiationEnabled = EnableRadiationToggle?.Selected ?? false;
448 if (MaxMissionCountText != null && Int32.TryParse(MaxMissionCountText.Text.SanitizedValue, out int missionCount))
449 {
450 settings.MaxMissionCount = missionCount;
451 }
452 else
453 {
454 settings.MaxMissionCount = CampaignSettings.DefaultMaxMissionCount;
455 }
456
457 if (selectedSub.HasTag(SubmarineTag.Shuttle) || !hasRequiredContentPackages)
458 {
459 if (!hasRequiredContentPackages)
460 {
461 var msgBox = new GUIMessageBox(TextManager.Get("ContentPackageMismatch"),
462 TextManager.GetWithVariable("ContentPackageMismatchWarning", "[requiredcontentpackages]", string.Join(", ", selectedSub.RequiredContentPackages)),
463 new LocalizedString[] { TextManager.Get("Yes"), TextManager.Get("No") });
464
465 msgBox.Buttons[0].OnClicked = msgBox.Close;
466 msgBox.Buttons[0].OnClicked += (button, obj) =>
467 {
468 if (GUIMessageBox.MessageBoxes.Count == 0)
469 {
470 StartNewGame?.Invoke(selectedSub, savePath, seedBox.Text, settings);
471 }
472 return true;
473 };
474
475 msgBox.Buttons[1].OnClicked = msgBox.Close;
476 }
477
478 if (selectedSub.HasTag(SubmarineTag.Shuttle))
479 {
480 var msgBox = new GUIMessageBox(TextManager.Get("ShuttleSelected"),
481 TextManager.Get("ShuttleWarning"),
482 new LocalizedString[] { TextManager.Get("Yes"), TextManager.Get("No") });
483
484 msgBox.Buttons[0].OnClicked = (button, obj) =>
485 {
486 StartNewGame?.Invoke(selectedSub, savePath, seedBox.Text, settings);
487 return true;
488 };
489 msgBox.Buttons[0].OnClicked += msgBox.Close;
490
491 msgBox.Buttons[1].OnClicked = msgBox.Close;
492 return false;
493 }
494 }
495 else
496 {
497 StartNewGame?.Invoke(selectedSub, savePath, seedBox.Text, settings);
498 }
499
500 return true;
501 }

544 public void UpdateSubList(IEnumerable<SubmarineInfo> submarines)
545 {
546 List<SubmarineInfo> subsToShow;
547 if (subFilter != CategoryFilter.All)
548 {
549 subsToShow = submarines.Where(s => s.IsCampaignCompatibleIgnoreClass && s.IsVanillaSubmarine() == (subFilter == CategoryFilter.Vanilla)).ToList();
550 }
551 else
552 {
553 string downloadFolder = Path.GetFullPath(SaveUtil.SubmarineDownloadFolder);
554 subsToShow = submarines.Where(s => s.IsCampaignCompatibleIgnoreClass && Path.GetDirectoryName(Path.GetFullPath(s.FilePath)) != downloadFolder).ToList();
555 }
556
557 subsToShow.Sort((s1, s2) =>
558 {
559 int p1 = s1.Price > CampaignMode.InitialMoney ? 10 : 0;
560 int p2 = s2.Price > CampaignMode.InitialMoney ? 10 : 0;
561 return p1.CompareTo(p2) * 100 + s1.Name.CompareTo(s2.Name);
562 });
563
564 subList.ClearChildren();
565
566 foreach (SubmarineInfo sub in subsToShow)
567 {
568 var textBlock = new GUITextBlock(
569 new RectTransform(new Vector2(1, 0.1f), subList.Content.RectTransform) { MinSize = new Point(0, 30) },
570 ToolBox.LimitString(sub.DisplayName.Value, GUIStyle.Font, subList.Rect.Width - 65), style: "ListBoxElement")
571 {
572 ToolTip = sub.Description,
573 UserData = sub
574 };
575
576 if (!sub.RequiredContentPackagesInstalled)
577 {
578 textBlock.TextColor = Color.Lerp(textBlock.TextColor, Color.DarkRed, .5f);
579 textBlock.ToolTip = TextManager.Get("ContentPackageMismatch") + "\n\n" + textBlock.ToolTip.SanitizedString;
580 }
581
582 var priceText = new GUITextBlock(new RectTransform(new Vector2(0.5f, 1.0f), textBlock.RectTransform, Anchor.CenterRight),
583 TextManager.GetWithVariable("currencyformat", "[credits]", string.Format(CultureInfo.InvariantCulture, "{0:N0}", sub.Price)), textAlignment: Alignment.CenterRight, font: GUIStyle.SmallFont)
584 {
585 TextColor = sub.Price > CampaignMode.InitialMoney ? GUIStyle.Red : textBlock.TextColor * 0.8f,
586 ToolTip = textBlock.ToolTip
587 };
588#if !DEBUG
589 if (!GameMain.DebugDraw)
590 {
591 if (sub.Price > CampaignMode.InitialMoney || !sub.IsCampaignCompatible)
592 {
593 textBlock.CanBeFocused = false;
594 textBlock.TextColor *= 0.5f;
595 }
596 }
597#endif
598 }
599 if (SubmarineInfo.SavedSubmarines.Any())
600 {
601 var validSubs = subsToShow.Where(s => s.IsCampaignCompatible && s.Price <= CampaignMode.InitialMoney).ToList();
602 if (validSubs.Count > 0)
603 {
604 subList.Select(validSubs[Rand.Int(validSubs.Count)]);
605 }
606 }
