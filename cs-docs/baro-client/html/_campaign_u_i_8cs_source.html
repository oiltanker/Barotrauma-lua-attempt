324  private void UpdateMap(float deltaTime, GUICustomComponent mapContainer)
325  {
326  var map = GameMain.GameSession?.Map;
327  if (map == null) { return; }
328  if (selectedLocation != null && selectedLocation == GameMain.GameSession.Campaign.GetCurrentDisplayLocation())
329  {
330  map.SelectLocation(-1);
331  }
332  map.Update(deltaTime, mapContainer);
333  foreach (GUITickBox tickBox in missionTickBoxes)
334  {
335  bool disable = hasMaxMissions && !tickBox.Selected;
336  tickBox.Enabled = Campaign.AllowedToManageCampaign(ClientPermissions.ManageMap) && !disable;
337  tickBox.Box.DisabledColor = disable ? tickBox.Box.Color * 0.5f : tickBox.Box.Color * 0.8f;
338  foreach (GUIComponent child in tickBox.Parent.Parent.Children)
339  {
340  if (child is GUITextBlock textBlock)
341  {
342  textBlock.SelectedTextColor = textBlock.HoverTextColor = textBlock.TextColor =
343  disable ? new Color(textBlock.TextColor, 0.5f) : new Color(textBlock.TextColor, 1.0f);
344  }
345  }
