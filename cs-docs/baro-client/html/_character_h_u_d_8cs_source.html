599 public static void ShowBossHealthBar(Character character)
600 {
601 if (character == null || character.IsDead || character.Removed) { return; }
602
603 var existingBar = bossHealthBars.Find(b => b.Character == character);
604 if (existingBar != null)
605 {
606 existingBar.FadeTimer = BossHealthBarDuration;
607 return;
608 }
609
610 if (bossHealthBars.Count > 5)
611 {
612 BossHealthBar oldestHealthBar = bossHealthBars.First();
613 foreach (var bar in bossHealthBars)
614 {
615 if (bar.TopHealthBar.BarSize < oldestHealthBar.TopHealthBar.BarSize)
616 {
617 oldestHealthBar = bar;
618 }
619 }
620 oldestHealthBar.FadeTimer = Math.Min(oldestHealthBar.FadeTimer, 1.0f);
621 }
622
623 bossHealthBars.Add(new BossHealthBar(character));
624 }

626 public static void UpdateBossHealthBars(float deltaTime)
627 {
628 for (int i = 0; i < bossHealthBars.Count; i++)
629 {
630 var bossHealthBar = bossHealthBars[i];
631
632 bool showTopBar = i == 0;
633 if (showTopBar != bossHealthBar.TopContainer.Visible)
634 {
635 bossHealthContainer.Recalculate();
636 }
637
638 bossHealthBar.TopContainer.Visible = showTopBar;
639 bossHealthBar.SideContainer.Visible = !bossHealthBar.TopContainer.Visible;
640
641 float health = bossHealthBar.Character.Vitality / bossHealthBar.Character.MaxVitality;
642
643 float alpha = Math.Min(bossHealthBar.FadeTimer, 1.0f);
644 foreach (var c in bossHealthBar.SideContainer.GetAllChildren().Concat(bossHealthBar.TopContainer.GetAllChildren()))
645 {
646 c.Color = new Color(c.Color, (byte)(alpha * 255));
647 if (c is GUITextBlock textBlock)
648 {
649 textBlock.TextColor = new Color(bossHealthBar.Character.IsDead ? Color.Gray : textBlock.TextColor, (byte)(alpha * 255));
650 }
651 }
652
653 bossHealthBar.TopHealthBar.BarSize = bossHealthBar.SideHealthBar.BarSize = health;
654
655 if (bossHealthBar.Character.Removed || !bossHealthBar.Character.Enabled)
656 {
657 bossHealthBar.FadeTimer = Math.Min(bossHealthBar.FadeTimer, 1.0f);
658 }
659 else if (bossHealthBar.Character.IsDead)
660 {
661 bossHealthBar.FadeTimer = Math.Min(bossHealthBar.FadeTimer, 5.0f);
662 }
663 bossHealthBar.FadeTimer -= deltaTime;
664 }
665
666 for (int i = bossHealthBars.Count - 1; i >= 0 ; i--)
667 {
668 var bossHealthBar = bossHealthBars[i];
669 if (bossHealthBar.FadeTimer <= 0)
670 {
671 bossHealthBar.SideContainer.Parent?.RemoveChild(bossHealthBar.SideContainer);
672 bossHealthBar.TopContainer.Parent?.RemoveChild(bossHealthBar.TopContainer);
673 bossHealthBars.RemoveAt(i);
674 bossHealthContainer.Recalculate();
675 }
676 }

687 float iconAlpha = 1.0f, bool createOffset = true, float scaleMultiplier = 1.0f, bool overrideAlpha = false)
688 {
689 if (order?.SymbolSprite == null) { return; }
690 if (order.IsReport && order.OrderGiver != character && !order.HasAppropriateJob(character)) { return; }

692 ISpatialEntity target = order.ConnectedController?.Item ?? order.TargetSpatialEntity;

696 //prevents exploiting the indicators in locating the sub

698 Vector2.DistanceSquared(character.WorldPosition, target.WorldPosition) > 1000.0f * 1000.0f)
699 {
700 return;
701 }
702
703 if (!orderIndicatorCount.ContainsKey(target)) { orderIndicatorCount.Add(target, 0); }
704
705 Vector2 drawPos = target is Entity ? (target as Entity).DrawPosition :
706 target.Submarine == null ? target.Position : target.Position + target.Submarine.DrawPosition;
707 drawPos += Vector2.UnitX * order.SymbolSprite.size.X * 1.5f * orderIndicatorCount[target];
708 GUI.DrawIndicator(spriteBatch, drawPos, cam, 100.0f, order.SymbolSprite, order.Color * iconAlpha,
709 createOffset: createOffset, scaleMultiplier: scaleMultiplier, overrideAlpha: overrideAlpha ? (float?)iconAlpha : null);
710
711 orderIndicatorCount[target] = orderIndicatorCount[target] + 1;
712 }
