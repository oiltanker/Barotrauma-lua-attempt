626  public static void UpdateBossHealthBars(float deltaTime)
627  {
628  for (int i = 0; i < bossHealthBars.Count; i++)
629  {
630  var bossHealthBar = bossHealthBars[i];
631 
632  bool showTopBar = i == 0;
633  if (showTopBar != bossHealthBar.TopContainer.Visible)
634  {
635  bossHealthContainer.Recalculate();
636  }
637 
638  bossHealthBar.TopContainer.Visible = showTopBar;
639  bossHealthBar.SideContainer.Visible = !bossHealthBar.TopContainer.Visible;
640 
641  float health = bossHealthBar.Character.Vitality / bossHealthBar.Character.MaxVitality;
642 
643  float alpha = Math.Min(bossHealthBar.FadeTimer, 1.0f);
644  foreach (var c in bossHealthBar.SideContainer.GetAllChildren().Concat(bossHealthBar.TopContainer.GetAllChildren()))
645  {
646  c.Color = new Color(c.Color, (byte)(alpha * 255));
647  if (c is GUITextBlock textBlock)
648  {
649  textBlock.TextColor = new Color(bossHealthBar.Character.IsDead ? Color.Gray : textBlock.TextColor, (byte)(alpha * 255));
650  }
651  }
652 
653  bossHealthBar.TopHealthBar.BarSize = bossHealthBar.SideHealthBar.BarSize = health;
654 
655  if (bossHealthBar.Character.Removed || !bossHealthBar.Character.Enabled)
656  {
657  bossHealthBar.FadeTimer = Math.Min(bossHealthBar.FadeTimer, 1.0f);
658  }
659  else if (bossHealthBar.Character.IsDead)
660  {
661  bossHealthBar.FadeTimer = Math.Min(bossHealthBar.FadeTimer, 5.0f);
662  }
663  bossHealthBar.FadeTimer -= deltaTime;
664  }
665 
666  for (int i = bossHealthBars.Count - 1; i >= 0 ; i--)
667  {
668  var bossHealthBar = bossHealthBars[i];
669  if (bossHealthBar.FadeTimer <= 0)
670  {
671  bossHealthBar.SideContainer.Parent?.RemoveChild(bossHealthBar.SideContainer);
672  bossHealthBar.TopContainer.Parent?.RemoveChild(bossHealthBar.TopContainer);
673  bossHealthBars.RemoveAt(i);
674  bossHealthContainer.Recalculate();
675  }
