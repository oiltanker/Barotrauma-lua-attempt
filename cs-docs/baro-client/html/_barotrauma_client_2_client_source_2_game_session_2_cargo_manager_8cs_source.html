16 public IEnumerable<Item> GetSellableItems(Character character)
17 {
18 if (character == null) { return new List<Item>(); }
19 var confirmedSoldEntities = GetConfirmedSoldEntities();
20 return character.Inventory.FindAllItems(item =>
21 {
22 if (!IsItemSellable(item, confirmedSoldEntities)) { return false; }
23 // Item must be in a non-equipment slot if possible
24 if (!item.AllowedSlots.All(s => equipmentSlots.Contains(s)) && IsInEquipmentSlot(item)) { return false; }
25 // Item must not be contained inside an item in an equipment slot
26 if (item.GetRootContainer() is Item rootContainer && IsInEquipmentSlot(rootContainer)) { return false; }
27 return true;
28 }, recursive: true).Distinct();
29
30 bool IsInEquipmentSlot(Item item)
31 {
32 foreach (InvSlotType slot in equipmentSlots)
33 {
34 if (character.Inventory.IsInLimbSlot(item, slot)) { return true; }
35 }
36 return false;
37 }
38 }

68 public void SetSoldItems(Dictionary<Identifier, List<SoldItem>> items)
69 {
70 if (SoldItems.Count == 0 && items.Count == 0) { return; }
71
72 SoldItems.Clear();
73 foreach (var entry in items)
74 {
75 SoldItems.Add(entry.Key, entry.Value);
76 }
77 foreach (var se in SoldEntities)
78 {
79 if (se.Status == SoldEntity.SellStatus.Confirmed) { continue; }
80 if (SoldItems.Any(si => si.Value.Any(si => Match(si, se, true))))
81 {
82 se.Status = SoldEntity.SellStatus.Confirmed;
83 }
84 else
85 {
86 se.Status = SoldEntity.SellStatus.Unconfirmed;
87 }
88 }
89 foreach (var soldItems in SoldItems.Values)
90 {
91 foreach (var si in soldItems)
92 {
93 if (si.Origin != SoldItem.SellOrigin.Submarine) { continue; }
94 if (!(SoldEntities.FirstOrDefault(se => se.Item == null && Match(si, se, false)) is SoldEntity soldEntityMatch)) { continue; }
95 if (!(Entity.FindEntityByID(si.ID) is Item item)) { continue; }
96 soldEntityMatch.SetItem(item);
97 soldEntityMatch.Status = SoldEntity.SellStatus.Confirmed;
98 }
99 }
100 OnSoldItemsChanged?.Invoke();
101
102 static bool Match(SoldItem soldItem, SoldEntity soldEntity, bool matchId)
103 {
104 if (soldItem.ItemPrefab != soldEntity.ItemPrefab) { return false; }
105 if (matchId && (soldEntity.Item == null || soldItem.ID != soldEntity.Item.ID)) { return false; }
106 if (soldItem.Origin == SoldItem.SellOrigin.Character && GameMain.Client != null && soldItem.SellerID != GameMain.Client.ID) { return false; }
107 return true;
108 }

128 public void SellItems(Identifier storeIdentifier, List<PurchasedItem> itemsToSell, Store.StoreTab sellingMode)
129 {
130 IEnumerable<Item> sellableItems;
131 try
132 {
133 sellableItems = sellingMode switch
134 {
135 Store.StoreTab.Sell => GetSellableItems(Character.Controlled),
136 Store.StoreTab.SellSub => GetSellableItemsFromSub(),
137 _ => throw new NotImplementedException()
138 };
139 }
140 catch (NotImplementedException e)
141 {
142 DebugConsole.ShowError($"Error selling items: uknown store tab type \"{sellingMode}\".\n{e.StackTrace.CleanupStackTrace()}");
143 return;
144 }
145 bool canAddToRemoveQueue = campaign.IsSinglePlayer && Entity.Spawner != null;
146 byte sellerId = GameMain.Client?.ID ?? 0;
147 // Check all the prices before starting the transaction to make sure the modifiers stay the same for the whole transaction
148 var sellValues = GetSellValuesAtCurrentLocation(storeIdentifier, itemsToSell.Select(i => i.ItemPrefab));
149 if (!(Location.GetStore(storeIdentifier) is { } store))
150 {
151 DebugConsole.ShowError($"Error selling items at {Location}: no store with identifier \"{storeIdentifier}\" exists.\n{Environment.StackTrace.CleanupStackTrace()}");
152 return;
153 }
154 var storeSpecificSoldItems = GetSoldItems(storeIdentifier, create: true);
155 foreach (var item in itemsToSell)
156 {
157 int itemValue = item.Quantity * sellValues[item.ItemPrefab];
158 // check if the store can afford the item
159 if (store.Balance < itemValue) { continue; }
160 // TODO: Write logic for prioritizing certain items over others (e.g. lone Battery Cell should be preferred over one inside a Stun Baton)
161 var matchingItems = sellableItems.Where(i => i.Prefab == item.ItemPrefab);
162 int count = Math.Min(item.Quantity, matchingItems.Count());
163 SoldItem.SellOrigin origin = sellingMode == Store.StoreTab.Sell ? SoldItem.SellOrigin.Character : SoldItem.SellOrigin.Submarine;
164 if (origin == SoldItem.SellOrigin.Character || GameMain.IsSingleplayer)
165 {
166 for (int i = 0; i < count; i++)
167 {
168 var matchingItem = matchingItems.ElementAt(i);
169 storeSpecificSoldItems.Add(new SoldItem(matchingItem.Prefab, matchingItem.ID, canAddToRemoveQueue, sellerId, origin));
170 SoldEntities.Add(new SoldEntity(matchingItem, campaign.IsSinglePlayer ? SoldEntity.SellStatus.Confirmed : SoldEntity.SellStatus.Local));
171 if (canAddToRemoveQueue) { Entity.Spawner.AddItemToRemoveQueue(matchingItem); }
172 }
173 }
174 else
175 {
176 // When selling from the sub in multiplayer, the server will determine the items that are sold
177 for (int i = 0; i < count; i++)
178 {
179 storeSpecificSoldItems.Add(new SoldItem(item.ItemPrefab, Entity.NullEntityID, canAddToRemoveQueue, sellerId, origin));
180 SoldEntities.Add(new SoldEntity(item.ItemPrefab, SoldEntity.SellStatus.Local));
181 }
182 }
183 // Exchange money
184 store.Balance -= itemValue;
185 if (GameMain.IsSingleplayer)
186 {
187 campaign.Bank.Give(itemValue);
188 }
189 GameAnalyticsManager.AddMoneyGainedEvent(itemValue, GameAnalyticsManager.MoneySource.Store, item.ItemPrefab.Identifier.Value);
190
191 // Remove from the sell crate
192 var sellCrate = (sellingMode == Store.StoreTab.Sell ? GetSellCrateItems(storeIdentifier) : GetSubCrateItems(storeIdentifier));
193 if (sellCrate?.Find(pi => pi.ItemPrefab == item.ItemPrefab) is { } itemToSell)
194 {
195 itemToSell.Quantity -= item.Quantity;
196 if (itemToSell.Quantity < 1)
197 {
198 sellCrate.Remove(itemToSell);
199 }
200 }
201 }
202 OnSoldItemsChanged?.Invoke();
203 }
