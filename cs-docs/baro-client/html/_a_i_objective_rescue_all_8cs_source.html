16  private const float vitalityThreshold = 75;
17  private const float vitalityThresholdForOrders = 90;
18  public static float GetVitalityThreshold(AIObjectiveManager manager, Character character, Character target)
19  {
20  if (manager == null)
21  {
22  return vitalityThreshold;
23  }
24  else
25  {
26  // When targeting player characters, always treat them when ordered, else use the threshold so that minor/non-severe damage is ignored.
27  // If we ignore any damage when the player orders a bot to do healings, it's observed to cause confusion among the players.
28  // On the other hand, if the bots too eagerly heal characters when it's not necessary, it's inefficient and can feel frustrating, because it can't be controlled.
29  return character == target || manager.HasOrder<AIObjectiveRescueAll>() ? (target.IsPlayer ? 100 : vitalityThresholdForOrders) : vitalityThreshold;
30  }
31  }

40  protected override float TargetEvaluation()
41  {
42  if (Targets.None()) { return 100; }
43  if (!objectiveManager.IsOrder(this))
44  {
45  if (!character.IsMedic && HumanAIController.IsTrueForAnyCrewMember(c => c != HumanAIController && c.Character.IsMedic && !c.Character.IsUnconscious))
46  {
47  // Don't do anything if there's a medic on board and we are not a medic
48  return 100;
49  }
50  }
51  float worstCondition = Targets.Min(t => GetVitalityFactor(t));
52  if (Targets.Contains(character))
53  {
54  if (character.Bleeding > 10)
55  {
56  // Enforce the highest priority when bleeding out.
57  worstCondition = 0;
58  }
59  // Boost the priority when wounded.
60  worstCondition /= 2;
61  }
62  return worstCondition;
63  }

65  public static float GetVitalityFactor(Character character)
66  {
67  float vitality = 100;
68  vitality -= character.Bleeding * 2;
69  vitality += Math.Min(character.Oxygen, 0);
70  vitality -= character.CharacterHealth.GetAfflictionStrength("paralysis");
71  foreach (Affliction affliction in AIObjectiveRescue.GetTreatableAfflictions(character))
72  {
73  float strength = character.CharacterHealth.GetPredictedStrength(affliction, predictFutureDuration: 10.0f);
74  vitality -= affliction.GetVitalityDecrease(character.CharacterHealth, strength) / character.MaxVitality * 100;
75  }
76  return Math.Clamp(vitality, 0, 100);
77  }

85  public static bool IsValidTarget(Character target, Character character)
86  {
87  if (target == null || target.IsDead || target.Removed) { return false; }
88  if (target.IsInstigator) { return false; }
89  if (target.IsPet) { return false; }
90  if (!HumanAIController.IsFriendly(character, target, onlySameTeam: true)) { return false; }
91  if (character.AIController is HumanAIController humanAI)
92  {
93  if (GetVitalityFactor(target) >= GetVitalityThreshold(humanAI.ObjectiveManager, character, target))
94  {
95  return false;
96  }
97  if (!humanAI.ObjectiveManager.HasOrder<AIObjectiveRescueAll>())
98  {
99  if (!character.IsMedic && target != character)
100  {
101  // Don't allow to treat others autonomously
102  return false;
103  }
104  // Ignore unsafe hulls, unless ordered
105  if (humanAI.UnsafeHulls.Contains(target.CurrentHull))
106  {
107  return false;
108  }
109  }
110  }
111  else
112  {
113  if (GetVitalityFactor(target) >= vitalityThreshold) { return false; }
114  }
115  if (character.Submarine != null)
116  {
117  // Don't allow going into another sub, unless it's connected and of the same team and type.
118  if (!character.Submarine.IsEntityFoundOnThisSub(target.CurrentHull, includingConnectedSubs: true)) { return false; }
119  }
120  else
121  {
122  return target.Submarine == null;
123  }
124  if (target != character && target.IsBot && HumanAIController.IsActive(target) && target.AIController is HumanAIController targetAI)
125  {
126  // Ignore all concious targets that are currently fighting, fleeing, fixing, or treating characters
127  if (targetAI.ObjectiveManager.HasActiveObjective<AIObjectiveCombat>() ||
128  targetAI.ObjectiveManager.HasActiveObjective<AIObjectiveFindSafety>() ||
129  targetAI.ObjectiveManager.HasActiveObjective<AIObjectiveRescue>() ||
130  targetAI.ObjectiveManager.HasActiveObjective<AIObjectiveFixLeak>())
131  {
132  return false;
133  }
134  }
135  if (target.CurrentHull != null)
136  {
137  // Don't go into rooms that have enemies
138  if (Character.CharacterList.Any(c => c.CurrentHull == target.CurrentHull && !HumanAIController.IsFriendly(character, c) && HumanAIController.IsActive(c))) { return false; }
139  }
140  return character.GetDamageDoneByAttacker(target) <= 0;
141  }
142  }
