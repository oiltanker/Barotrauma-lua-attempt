118  private void SetupVertexBuffers()
119  {
120  Vector2 textureSize = new Vector2(Sprite.Texture.Width, Sprite.Texture.Height);
121  var pos = Sprite.SourceRect.Location;
122  var size = Sprite.SourceRect.Size;
123 
124  uvTopLeft = Vector2.Divide(pos.ToVector2(), textureSize);
125  uvBottomRight = Vector2.Divide((pos + size).ToVector2(), textureSize);
126  uvTopLeftFlipped = Vector2.Divide(new Vector2(pos.X + size.X, pos.Y), textureSize);
127  uvBottomRightFlipped = Vector2.Divide(new Vector2(pos.X, pos.Y + size.Y), textureSize);
128  if (Invert)
129  {
130  var temp = uvBottomRightFlipped;
131  uvBottomRightFlipped = uvTopLeftFlipped;
132  uvTopLeftFlipped = temp;
133  }
134 
135  for (int i = 0; i < 2; i++)
136  {
137  bool flip = i == 1;
138 
139  var vertices = new VertexPositionColorTexture[(subDivX + 1) * (subDivY + 1)];
140  for (int x = 0; x <= subDivX; x++)
141  {
142  for (int y = 0; y <= subDivY; y++)
143  {
144  //{0,0} -> {1,1}
145  Vector2 relativePos = new Vector2(x / (float)subDivX, y / (float)subDivY);
146 
147  Vector2 uvCoord = flip ?
148  uvTopLeftFlipped + (uvBottomRightFlipped - uvTopLeftFlipped) * relativePos :
149  uvTopLeft + (uvBottomRight - uvTopLeft) * relativePos;
150 
151  vertices[x + y * (subDivX + 1)] = new VertexPositionColorTexture(
152  position: new Vector3(relativePos.X * Sprite.SourceRect.Width, relativePos.Y * Sprite.SourceRect.Height, 0.0f),
153  color: Color.White,
154  textureCoordinate: uvCoord);
155  }
156  }
157 
158  if (flip)
159  {
160  if (flippedVertexBuffer != null && flippedVertexBuffer.VertexCount != vertices.Length)
161  {
162  flippedVertexBuffer.Dispose();
163  flippedVertexBuffer = null;
164  }
165  if (flippedVertexBuffer == null)
166  {
167  flippedVertexBuffer = new VertexBuffer(GameMain.Instance.GraphicsDevice, VertexPositionColorTexture.VertexDeclaration, vertices.Length, BufferUsage.None);
168  }
169  flippedVertexBuffer.SetData(vertices);
170  }
171  else
172  {
173  if (vertexBuffer != null && vertexBuffer.VertexCount != vertices.Length)
174  {
175  vertexBuffer.Dispose();
176  vertexBuffer = null;
177  }
178  if (vertexBuffer == null)
179  {
180  vertexBuffer = new VertexBuffer(GameMain.Instance.GraphicsDevice, VertexPositionColorTexture.VertexDeclaration, vertices.Length, BufferUsage.None);
181  }
182  vertexBuffer.SetData(vertices);
183  }
184  }
185 
186  spritePos = Sprite.SourceRect.Location;
187  spriteSize = Sprite.SourceRect.Size;
188  }

190  private void SetupIndexBuffer()
191  {
192  triangleCount = subDivX * subDivY * 2;
193  var indices = new ushort[triangleCount * 3];
194  int offset = 0;
195  for (int i = 0; i < triangleCount / 2; i++)
196  {
197  indices[i * 6] = (ushort)(i + offset + 1);
198  indices[i * 6 + 1] = (ushort)(i + offset + (subDivX + 1) + 1);
199  indices[i * 6 + 2] = (ushort)(i + offset + (subDivX + 1));
200 
201  indices[i * 6 + 3] = (ushort)(i + offset);
202  indices[i * 6 + 4] = (ushort)(i + offset + 1);
203  indices[i * 6 + 5] = (ushort)(i + offset + (subDivX + 1));
204 
205  if ((i + 1) % subDivX == 0) offset++;
206  }
207 
208  indexBuffer?.Dispose();
209  indexBuffer = null;
210 
211  indexBuffer = new IndexBuffer(GameMain.Instance.GraphicsDevice, IndexElementSize.SixteenBits, indices.Length, BufferUsage.None);
212  indexBuffer.SetData(indices);
