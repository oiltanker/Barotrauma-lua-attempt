356  partial void SetupDrawOrder()
357  {
358  //make sure every character gets drawn at a distinct "layer"
359  //(instead of having some of the limbs appear behind and some in front of other characters)
360  float startDepth = 0.1f;
361  float increment = 0.001f;
362  foreach (Character otherCharacter in Character.CharacterList)
363  {
364  if (otherCharacter == character) { continue; }
365  startDepth += increment;
366  }
367  //make sure each limb has a distinct depth value
368  List<Limb> depthSortedLimbs = Limbs.OrderBy(l => l.DefaultSpriteDepth).ToList();
369  foreach (Limb limb in Limbs)
370  {
371  var sprite = limb.GetActiveSprite();
372  if (sprite == null) { continue; }
373  sprite.Depth = startDepth + depthSortedLimbs.IndexOf(limb) * 0.00001f;
374  foreach (var conditionalSprite in limb.ConditionalSprites)
375  {
376  if (conditionalSprite.Exclusive)
377  {
378  conditionalSprite.ActiveSprite.Depth = sprite.Depth;
379  }
380  }
381  }
382  foreach (Limb limb in Limbs)
383  {
384  if (limb.ActiveSprite == null) { continue; }
385  if (limb.Params.InheritLimbDepth == LimbType.None) { continue; }
386  var matchingLimb = GetLimb(limb.Params.InheritLimbDepth);
387  if (matchingLimb != null)
388  {
389  limb.ActiveSprite.Depth = matchingLimb.ActiveSprite.Depth - 0.0000001f;
390  }
391  }
392 
393  depthSortedLimbs.Reverse();
394  inversedLimbDrawOrder = depthSortedLimbs.ToArray();
395  }

494  public float GetDepthOffset()
495  {
496  float maxDepth = 0.0f;
497  float minDepth = 1.0f;
498  float depthOffset = 0.0f;
499  var ladder = character.SelectedConstruction?.GetComponent<Ladder>();
500 
501  if (ladder != null)
502  {
503  CalculateLimbDepths();
504  if (character.WorldPosition.X < character.SelectedConstruction.WorldPosition.X)
505  {
506  //at the left side of the ladder, needs to be drawn in front of the rungs
507  if (maxDepth > ladder.BackgroundSpriteDepth)
508  {
509  depthOffset = Math.Max(ladder.BackgroundSpriteDepth - 0.01f - maxDepth, 0.0f);
510  }
511  else
512  {
513  depthOffset = Math.Max(ladder.Item.GetDrawDepth() + 0.0001f - minDepth, -minDepth);
514  }
515  }
516  else
517  {
518  //at the right side of the ladder, needs to be drawn behind the rungs
519  depthOffset = Math.Max(ladder.BackgroundSpriteDepth + 0.01f - minDepth, 0.0f);
520  }
521  }
522  else
523  {
524  CalculateLimbDepths();
525  var controller = character.SelectedConstruction?.GetComponent<Controller>();
526  if (controller != null && controller.ControlCharacterPose && controller.User == character && controller.UserInCorrectPosition)
527  {
528  if (controller.Item.SpriteDepth <= maxDepth || controller.DrawUserBehind)
529  {
530  depthOffset = Math.Max(controller.Item.GetDrawDepth() + 0.0001f - minDepth, -minDepth);
531  }
532  else
533  {
534  depthOffset = Math.Max(controller.Item.GetDrawDepth() - 0.0001f - maxDepth, 0.0f);
535  }
536  }
