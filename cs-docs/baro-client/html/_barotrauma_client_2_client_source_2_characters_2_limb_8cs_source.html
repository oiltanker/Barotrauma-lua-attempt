378 SpriteDeformation deformation = null;

380 {
381 // if the element is marked with the sync attribute, use a deformation of the same type with the same sync value, if there is one already.
382 string typeName = animationElement.GetAttributeString("type", "").ToLowerInvariant();
383 deformation = ragdoll.Limbs
384 .Where(l => l != null)

387 }
388 if (deformation == null)

390 deformation = SpriteDeformation.Load(animationElement, character.SpeciesName.Value);
391 if (deformation != null)
392 {
393 ragdoll.SpriteDeformations.Add(deformation);

395 }

398 deformations.Add(deformation);

402 }

587 }

591 if (damageParticleAmount > 0.001f)
592 {
593 foreach (ParticleEmitter emitter in character.DamageEmitters)

620 }
621 }

689 public void Draw(SpriteBatch spriteBatch, Camera cam, Color? overrideColor = null, bool disableDeformations = false)
690 {
691 float brightness = 1.0f - (burnOverLayStrength / 100.0f) * 0.5f;
692 var spriteParams = Params.GetSprite();
693 if (spriteParams == null) { return; }
694
695 Color clr = spriteParams.Color;
696 if (!spriteParams.IgnoreTint)
697 {
698 clr = clr.Multiply(ragdoll.RagdollParams.Color);
699 if (character.Info != null)
700 {
701 clr = clr.Multiply(character.Info.Head.SkinColor);
702 }
703 if (character.CharacterHealth.FaceTint.A > 0 && type == LimbType.Head)
704 {
705 clr = Color.Lerp(clr, character.CharacterHealth.FaceTint.Opaque(), character.CharacterHealth.FaceTint.A / 255.0f);
706 }
707 if (character.CharacterHealth.BodyTint.A > 0)
708 {
709 clr = Color.Lerp(clr, character.CharacterHealth.BodyTint.Opaque(), character.CharacterHealth.BodyTint.A / 255.0f);
710 }
