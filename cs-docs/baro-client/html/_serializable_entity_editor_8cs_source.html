339 public GUIComponent CreateNewField(SerializableProperty property, ISerializableEntity entity)
340 {
341 object value = property.GetValue(entity);
342 if (property.PropertyType == typeof(string) && value == null)
343 {
344 value = "";
345 }
346
347 Identifier propertyTag = $"{property.PropertyInfo.DeclaringType.Name}.{property.PropertyInfo.Name}".ToIdentifier();
348 Identifier fallbackTag = property.PropertyInfo.Name.ToIdentifier();
349 LocalizedString displayName =
350 TextManager.Get(propertyTag, $"sp.{propertyTag}.name".ToIdentifier());
351 if (displayName.IsNullOrEmpty())
352 {
353 Editable editable = property.GetAttribute<Editable>();
354 if (editable != null && !string.IsNullOrEmpty(editable.FallBackTextTag))
355 {
356 displayName = TextManager.Get(editable.FallBackTextTag);
357 }
358 else
359 {
360 displayName = TextManager.Get(fallbackTag, $"sp.{fallbackTag}.name".ToIdentifier());
361 }
362 }
363
364 if (displayName.IsNullOrEmpty())
365 {
366 displayName = property.Name.FormatCamelCaseWithSpaces();
367#if DEBUG
368 InGameEditable editable = property.GetAttribute<InGameEditable>();
369 if (editable != null)
370 {
371 if (!MissingLocalizations.Contains($"sp.{propertyTag}.name|{displayName}"))
372 {
373 DebugConsole.NewMessage("Missing Localization for property: " + propertyTag);
374 MissingLocalizations.Add($"sp.{propertyTag}.name|{displayName}");
375 MissingLocalizations.Add($"sp.{propertyTag}.description|{property.GetAttribute<Serialize>().Description}");
376 }
377 }
378#endif
379 }
380
381 LocalizedString toolTip = TextManager.Get($"sp.{propertyTag}.description");
382 if (toolTip.IsNullOrEmpty())
383 {
384 toolTip = TextManager.Get($"{propertyTag}.description", $"sp.{fallbackTag}.description");
385 }
386
387 if (toolTip == null)
388 {
389 toolTip = property.GetAttribute<Serialize>().Description;
390 }
391
392 GUIComponent propertyField = null;
393 if (value is bool boolVal)
394 {
395 propertyField = CreateBoolField(entity, property, boolVal, displayName, toolTip);
396 }
397 else if (value is string stringVal)
398 {
399 propertyField = CreateStringField(entity, property, stringVal, displayName, toolTip);
400 }
401 else if (value.GetType().IsEnum)
402 {
403 if (value.GetType().IsDefined(typeof(FlagsAttribute), inherit: false))
404 {
405 propertyField = CreateEnumFlagField(entity, property, value, displayName, toolTip);
406 }
407 else
408 {
409 propertyField = CreateEnumField(entity, property, value, displayName, toolTip);
410 }
411 }
412 else if (value is int i)
413 {
414 propertyField = CreateIntField(entity, property, i, displayName, toolTip);
415 }
416 else if (value is float f)
417 {
418 propertyField = CreateFloatField(entity, property, f, displayName, toolTip);
419 }
420 else if (value is Point p)
421 {
422 propertyField = CreatePointField(entity, property, p, displayName, toolTip);
423 }
424 else if (value is Vector2 v2)
425 {
426 propertyField = CreateVector2Field(entity, property, v2, displayName, toolTip);
427 }
428 else if (value is Vector3 v3)
429 {
430 propertyField = CreateVector3Field(entity, property, v3, displayName, toolTip);
431 }
432 else if (value is Vector4 v4)
433 {
434 propertyField = CreateVector4Field(entity, property, v4, displayName, toolTip);
435 }
436 else if (value is Color c)
437 {
438 propertyField = CreateColorField(entity, property, c, displayName, toolTip);
439 }
440 else if (value is Rectangle r)
441 {
442 propertyField = CreateRectangleField(entity, property, r, displayName, toolTip);
443 }
444 else if(value is string[] a)
445 {
446 propertyField = CreateStringArrayField(entity, property, a, displayName, toolTip);
447 }
448 return propertyField;
449 }

656 {
657 var frame = new GUILayoutGroup(new RectTransform(new Point(Rect.Width, elementHeight), layoutGroup.RectTransform, isFixedSize: true), isHorizontal: true, childAnchor: Anchor.CenterLeft)

661 var label = new GUITextBlock(new RectTransform(new Vector2(1.0f - inputFieldWidth, 1), frame.RectTransform), displayName, font: GUIStyle.SmallFont, textAlignment: Alignment.Left)
662 {

664 };
665 Identifier translationTextTag = property.GetAttribute<Serialize>()?.TranslationTextTag ?? Identifier.Empty;
666 float browseButtonWidth = 0.1f;
667 var editableAttribute = property.GetAttribute<Editable>();
668 float textBoxWidth = inputFieldWidth;
669 if (!translationTextTag.IsEmpty) { textBoxWidth -= browseButtonWidth; }
670 GUITextBox propertyBox = new GUITextBox(new RectTransform(new Vector2(textBoxWidth, 1), frame.RectTransform))
671 {
672 Enabled = editableAttribute != null && !editableAttribute.ReadOnly,
673 ToolTip = toolTip,
674 Font = GUIStyle.SmallFont,
675 Text = value,
676 OverflowClip = true
677 };
678
679 HashSet<MapEntity> editedEntities = new HashSet<MapEntity>();
680 propertyBox.OnTextChanged += (textBox, text) =>
681 {
682 foreach (var entity in MapEntity.SelectedList)
683 {
684 editedEntities.Add(entity);
685 }
686 return true;
687 };
688 propertyBox.OnDeselected += (textBox, keys) => OnApply(textBox);
689 propertyBox.OnEnterPressed += (box, text) => OnApply(box);
690 refresh += () =>
691 {
692 if (!propertyBox.Selected) { propertyBox.Text = (string)property.GetValue(entity); }
693 };
694
695 bool OnApply(GUITextBox textBox)
696 {
697 List<MapEntity> prevSelected = MapEntity.SelectedList.ToList();
698 //reselect the entities that were selected during editing
699 //otherwise multi-editing won't work when we deselect the entities with unapplied changes in the textbox
700 foreach (var entity in editedEntities)
701 {
702 MapEntity.SelectedList.Add(entity);
703 }
704 if (SetPropertyValue(property, entity, textBox.Text))
705 {
706 TrySendNetworkUpdate(entity, property);
707 textBox.Text = (string) property.GetValue(entity);
708 textBox.Flash(GUIStyle.Green, flashDuration: 1f);
709 }
710 //restore the entities that were selected before applying
711 MapEntity.SelectedList.Clear();
712 foreach (var entity in prevSelected)
713 {
714 MapEntity.SelectedList.Add(entity);
715 }
716 return true;
717 }
718
719 if (!translationTextTag.IsEmpty)
720 {
721 new GUIButton(new RectTransform(new Vector2(browseButtonWidth, 1), frame.RectTransform, Anchor.TopRight), "...", style: "GUIButtonSmall")
722 {
723 OnClicked = (bt, userData) => { CreateTextPicker(translationTextTag.Value, entity, property, propertyBox); return true; }
724 };
725 propertyBox.OnTextChanged += (tb, text) =>
726 {
727 LocalizedString translatedText = TextManager.Get(text);

732 }

737 }
