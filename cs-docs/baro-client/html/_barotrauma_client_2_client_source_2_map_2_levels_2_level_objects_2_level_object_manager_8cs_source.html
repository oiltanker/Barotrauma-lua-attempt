50 private void RefreshVisibleObjects(Rectangle currentIndices, float zoom)
51 {
52 visibleObjectsBack.Clear();
53 visibleObjectsMid.Clear();
54 visibleObjectsFront.Clear();
55
56 float minSizeToDraw = MathHelper.Lerp(10.0f, 5.0f, Math.Min(zoom * 20.0f, 1.0f));
57
58 for (int x = currentIndices.X; x <= currentIndices.Width; x++)
59 {
60 for (int y = currentIndices.Y; y <= currentIndices.Height; y++)
61 {
62 if (objectGrid[x, y] == null) { continue; }
63 foreach (LevelObject obj in objectGrid[x, y])
64 {
65 if (obj.Prefab.HideWhenBroken && obj.Health <= 0.0f) { continue; }
66
67 if (zoom < 0.05f)
68 {
69 //hide if the sprite is very small when zoomed this far out
70 if ((obj.Sprite != null && Math.Min(obj.Sprite.size.X * zoom, obj.Sprite.size.Y * zoom) < 5.0f) ||
71 (obj.ActivePrefab?.DeformableSprite != null && Math.Min(obj.ActivePrefab.DeformableSprite.Sprite.size.X * zoom, obj.ActivePrefab.DeformableSprite.Sprite.size.Y * zoom) < minSizeToDraw))
72 {
73 continue;
74 }
75
76 float zCutoff = MathHelper.Lerp(5000.0f, 500.0f, (0.05f - zoom) * 20.0f);
77 if (obj.Position.Z > zCutoff)
78 {
79 continue;
80 }
81 }
82
83 var objectList =
84 obj.Position.Z >= 0 ?
85 visibleObjectsBack :
86 (obj.Position.Z < -1 ? visibleObjectsFront : visibleObjectsMid);
87 int drawOrderIndex = 0;
88 for (int i = 0; i < objectList.Count; i++)
89 {
90 if (objectList[i] == obj)
91 {
92 drawOrderIndex = -1;
93 break;
94 }
95
96 if (objectList[i].Position.Z < obj.Position.Z)
97 {
98 break;
99 }
100 else
101 {
102 drawOrderIndex = i + 1;
103 }
104 }
105
106 if (drawOrderIndex >= 0)
107 {
108 objectList.Insert(drawOrderIndex, obj);
109 if (objectList.Count >= MaxVisibleObjects) { break; }
110 }
111 }
112 }
113 }
114
115 currentGridIndices = currentIndices;
116 }

142 private void DrawObjects(SpriteBatch spriteBatch, Camera cam, List<LevelObject> objectList)
143 {
144 Rectangle indices = Rectangle.Empty;
145 indices.X = (int)Math.Floor(cam.WorldView.X / (float)GridSize);
146 if (indices.X >= objectGrid.GetLength(0)) return;
147 indices.Y = (int)Math.Floor((cam.WorldView.Y - cam.WorldView.Height - Level.Loaded.BottomPos) / (float)GridSize);
148 if (indices.Y >= objectGrid.GetLength(1)) return;
149
150 indices.Width = (int)Math.Floor(cam.WorldView.Right / (float)GridSize) + 1;
151 if (indices.Width < 0) return;
152 indices.Height = (int)Math.Floor((cam.WorldView.Y - Level.Loaded.BottomPos) / (float)GridSize) + 1;
153 if (indices.Height < 0) return;
154
155 indices.X = Math.Max(indices.X, 0);
156 indices.Y = Math.Max(indices.Y, 0);
157 indices.Width = Math.Min(indices.Width, objectGrid.GetLength(0) - 1);
158 indices.Height = Math.Min(indices.Height, objectGrid.GetLength(1) - 1);
159
160 float z = 0.0f;
161 if (ForceRefreshVisibleObjects || (currentGridIndices != indices && Timing.TotalTime > NextRefreshTime))
162 {
163 RefreshVisibleObjects(indices, cam.Zoom);
164 ForceRefreshVisibleObjects = false;
165 if (cam.Zoom < 0.1f)
166 {
167 //when zoomed very far out, refresh a little less often
168 NextRefreshTime = Timing.TotalTime + MathHelper.Lerp(1.0f, 0.0f, cam.Zoom * 10.0f);
169 }
170 }
171
172 foreach (LevelObject obj in objectList)
173 {
174 Vector2 camDiff = new Vector2(obj.Position.X, obj.Position.Y) - cam.WorldViewCenter;
175 camDiff.Y = -camDiff.Y;
176
177 Sprite activeSprite = obj.Sprite;
178 activeSprite?.Draw(
179 spriteBatch,
180 new Vector2(obj.Position.X, -obj.Position.Y) - camDiff * obj.Position.Z / 10000.0f,
181 Color.Lerp(Color.White, Level.Loaded.BackgroundTextureColor, obj.Position.Z / 3000.0f),
182 activeSprite.Origin,
183 obj.CurrentRotation,
184 obj.CurrentScale,
185 SpriteEffects.None,
186 z);
187
188 if (obj.ActivePrefab.DeformableSprite != null)
189 {
190 if (obj.CurrentSpriteDeformation != null)
191 {
192 obj.ActivePrefab.DeformableSprite.Deform(obj.CurrentSpriteDeformation);
193 }
194 else
195 {
196 obj.ActivePrefab.DeformableSprite.Reset();
197 }
198 obj.ActivePrefab.DeformableSprite?.Draw(cam,
199 new Vector3(new Vector2(obj.Position.X, obj.Position.Y) - camDiff * obj.Position.Z / 10000.0f, z * 10.0f),
200 obj.ActivePrefab.DeformableSprite.Origin,
201 obj.CurrentRotation,
202 obj.CurrentScale,
203 Color.Lerp(Color.White, Level.Loaded.BackgroundTextureColor, obj.Position.Z / 5000.0f));
204 }
205
206
207 if (GameMain.DebugDraw)
208 {
209 GUI.DrawRectangle(spriteBatch, new Vector2(obj.Position.X, -obj.Position.Y), new Vector2(10.0f, 10.0f), GUIStyle.Red, true);
210
211 if (obj.Triggers == null) { continue; }
212 foreach (LevelTrigger trigger in obj.Triggers)
213 {
214 if (trigger.PhysicsBody == null) continue;
215 GUI.DrawLine(spriteBatch, new Vector2(obj.Position.X, -obj.Position.Y), new Vector2(trigger.WorldPosition.X, -trigger.WorldPosition.Y), Color.Cyan, 0, 3);
216
217 Vector2 flowForce = trigger.GetWaterFlowVelocity();
218 if (flowForce.LengthSquared() > 1)
219 {
220 flowForce.Y = -flowForce.Y;
221 GUI.DrawLine(spriteBatch, new Vector2(trigger.WorldPosition.X, -trigger.WorldPosition.Y), new Vector2(trigger.WorldPosition.X, -trigger.WorldPosition.Y) + flowForce * 10, GUIStyle.Orange, 0, 5);
222 }
223 trigger.PhysicsBody.UpdateDrawPosition();
224 trigger.PhysicsBody.DebugDraw(spriteBatch, trigger.IsTriggered ? Color.Cyan : Color.DarkCyan);
225 }
226 }
227
228 z += 0.0001f;
229 }
