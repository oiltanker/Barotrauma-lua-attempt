50  private void RefreshVisibleObjects(Rectangle currentIndices, float zoom)
51  {
52  visibleObjectsBack.Clear();
53  visibleObjectsMid.Clear();
54  visibleObjectsFront.Clear();
55 
56  float minSizeToDraw = MathHelper.Lerp(10.0f, 5.0f, Math.Min(zoom * 20.0f, 1.0f));
57 
58  for (int x = currentIndices.X; x <= currentIndices.Width; x++)
59  {
60  for (int y = currentIndices.Y; y <= currentIndices.Height; y++)
61  {
62  if (objectGrid[x, y] == null) { continue; }
63  foreach (LevelObject obj in objectGrid[x, y])
64  {
65  if (obj.Prefab.HideWhenBroken && obj.Health <= 0.0f) { continue; }
66 
67  if (zoom < 0.05f)
68  {
69  //hide if the sprite is very small when zoomed this far out
70  if ((obj.Sprite != null && Math.Min(obj.Sprite.size.X * zoom, obj.Sprite.size.Y * zoom) < 5.0f) ||
71  (obj.ActivePrefab?.DeformableSprite != null && Math.Min(obj.ActivePrefab.DeformableSprite.Sprite.size.X * zoom, obj.ActivePrefab.DeformableSprite.Sprite.size.Y * zoom) < minSizeToDraw))
72  {
73  continue;
74  }
75 
76  float zCutoff = MathHelper.Lerp(5000.0f, 500.0f, (0.05f - zoom) * 20.0f);
77  if (obj.Position.Z > zCutoff)
78  {
79  continue;
80  }
81  }
82 
83  var objectList =
84  obj.Position.Z >= 0 ?
85  visibleObjectsBack :
86  (obj.Position.Z < -1 ? visibleObjectsFront : visibleObjectsMid);
87  int drawOrderIndex = 0;
88  for (int i = 0; i < objectList.Count; i++)
89  {
90  if (objectList[i] == obj)
91  {
92  drawOrderIndex = -1;
93  break;
94  }
95 
96  if (objectList[i].Position.Z < obj.Position.Z)
97  {
98  break;
99  }
100  else
101  {
102  drawOrderIndex = i + 1;
103  }
104  }
105 
106  if (drawOrderIndex >= 0)
107  {
108  objectList.Insert(drawOrderIndex, obj);
109  if (objectList.Count >= MaxVisibleObjects) { break; }
110  }
111  }
112  }
113  }
114 
115  currentGridIndices = currentIndices;
116  }
