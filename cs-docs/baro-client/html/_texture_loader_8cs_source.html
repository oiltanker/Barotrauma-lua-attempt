46 private static byte[] CompressDxt5(byte[] data, int width, int height)
47 {
48 using (System.IO.MemoryStream mstream = new System.IO.MemoryStream())
49 {
50 for (int y = 0; y < height; y += 4)
51 {
52 for (int x = 0; x < width; x += 4)
53 {
54 int offset = x * 4 + y * 4 * width;
55 CompressDxt5Block(data, offset, width, mstream);
56 }
57 }
58 return mstream.ToArray();
59 }

62 private static void CompressDxt5Block(byte[] data, int offset, int width, System.IO.Stream output)
63 {
64 int r1 = 255, g1 = 255, b1 = 255, a1 = 255;
65 int r2 = 0, g2 = 0, b2 = 0, a2 = 0;
66
67 //determine the two colors to interpolate between:
68 //color 1 represents lowest luma, color 2 represents highest luma
69 for (int i = 0; i < 16; i++)
70 {
71 int pixelOffset = offset + (4 * ((i % 4) + (width * (i >> 2))));
72 int r, g, b, a;
73 r = data[pixelOffset + 0];
74 g = data[pixelOffset + 1];
75 b = data[pixelOffset + 2];
76 a = data[pixelOffset + 3];
77 if (r * 299 + g * 587 + b * 114 < r1 * 299 + g1 * 587 + b1 * 114)
78 {
79 r1 = r; g1 = g; b1 = b;
80 }
81 if (r * 299 + g * 587 + b * 114 > r2 * 299 + g2 * 587 + b2 * 114)
82 {
83 r2 = r; g2 = g; b2 = b;
84 }
85 if (a < a1) { a1 = a; }
86 if (a > a2) { a2 = a; }
87 }
88
89 //convert the colors to rgb565 (16-bit rgb)
90 int r1_565 = (r1 * 0x1f) / 0xff; if (r1_565 > 0x1f) { r1_565 = 0x1f; }
91 int g1_565 = (g1 * 0x3f) / 0xff; if (g1_565 > 0x3f) { g1_565 = 0x3f; }
92 int b1_565 = (b1 * 0x1f) / 0xff; if (b1_565 > 0x1f) { b1_565 = 0x1f; }
93
94 int r2_565 = (r2 * 0x1f) / 0xff; if (r2_565 > 0x1f) { r2_565 = 0x1f; }
95 int g2_565 = (g2 * 0x3f) / 0xff; if (g2_565 > 0x3f) { g2_565 = 0x3f; }
96 int b2_565 = (b2 * 0x1f) / 0xff; if (b2_565 > 0x1f) { b2_565 = 0x1f; }
97
98 //luma is also used to determine which color on the palette
99 //most closely resembles each pixel to compress, so we
100 //calculate this here
101 int y1 = r1 * 299 + g1 * 587 + b1 * 114;
102 int y2 = r2 * 299 + g2 * 587 + b2 * 114;
103
104 byte[] newData = new byte[16];
105 for (int i = 0; i < 16; i++)
106 {
107 int pixelOffset = offset + (4 * ((i % 4) + (width * (i >> 2))));
108 int r, g, b, a;
109 r = data[pixelOffset + 0];
110 g = data[pixelOffset + 1];
111 b = data[pixelOffset + 2];
112 a = data[pixelOffset + 3];
113
114 if (a1 < a2)
115 {
116 a -= a1;
117 a = (a * 0x7) / (a2 - a1);
118 if (a > 0x7) { a = 0x7; }
119
120 switch (a)
121 {
122 case 0:
123 a = 1;
124 break;
125 case 1:
126 a = 7;
127 break;
128 case 2:
129 a = 6;
130 break;
131 case 3:
132 a = 5;
133 break;
134 case 4:
135 a = 4;
136 break;
137 case 5:
138 a = 3;
139 break;
140 case 6:
141 a = 2;
142 break;
143 case 7:
144 a = 0;
145 break;
146 }
147 }
148 else
149 {
150 a = 0;
151 }
152
153 NetBitWriter.WriteUInt32((uint)a, 3, newData, 16 + (i * 3));
154
155 int y = r * 299 + g * 587 + b * 114;
156
157 int max = y2 - y1;
158 int diffY = y - y1;
159
160 int paletteIndex;
161 if (diffY < max / 4)
162 {
163 paletteIndex = 0;
164 }
165 else if (diffY < max / 2)
166 {
167 paletteIndex = 2;
168 }
169 else if (diffY < max * 3 / 4)
170 {
171 paletteIndex = 3;
172 }
173 else
174 {
175 paletteIndex = 1;
176 }
177 newData[12 + (i / 4)] |= (byte)(paletteIndex << (2 * (i % 4)));
178 }
179
180 newData[0] = (byte)a2;
181 newData[1] = (byte)a1;
182
183 newData[9] = (byte)((r1_565 << 3) | (g1_565 >> 3));
184 newData[8] = (byte)((g1_565 << 5) | b1_565);
185 newData[11] = (byte)((r2_565 << 3) | (g2_565 >> 3));
186 newData[10] = (byte)((g2_565 << 5) | b2_565);
187
188 output.Write(newData, 0, 16);
189 }

199 public static Texture2D FromStream(System.IO.Stream stream, string path = null, bool compress = true, bool mipmap = false)
200 {
201 try
202 {
203 path = path.CleanUpPath();
204 byte[] textureData = null;
205 textureData = Texture2D.TextureDataFromStream(stream, out int width, out int height, out int channels);
206
207 SurfaceFormat format = SurfaceFormat.Color;
208 if (GameSettings.CurrentConfig.Graphics.CompressTextures && compress)
209 {
210 if (((width & 0x03) == 0) && ((height & 0x03) == 0))
211 {
212 textureData = CompressDxt5(textureData, width, height);
213 format = SurfaceFormat.Dxt5;
214 mipmap = false;
215 }
216 else
217 {
218 DebugConsole.AddWarning($"Could not compress a texture because the dimensions aren't a multiple of 4 (path: {path ?? "null"}, size: {width}x{height})");
219 }
220 }
221
222 Texture2D tex = null;
223 CrossThread.RequestExecutionOnMainThread(() =>
224 {
225 if (cancelAll) { return; }
226 tex = new Texture2D(_graphicsDevice, width, height, mipmap, format);
227 tex.SetData(textureData);
228 });
229 return tex;
230 }
231 catch (Exception e)
232 {
233#if WINDOWS
234 if (e is SharpDX.SharpDXException) { throw; }
235#endif
236
237 DebugConsole.ThrowError(string.IsNullOrEmpty(path) ? "Loading texture from stream failed!" :
238 "Loading texture \"" + path + "\" failed!", e);
239 return null;
240 }
241 }
