256 public override IEnumerable<CoroutineStatus> UpdateState()
257 {
258 while (GameMain.Instance.LoadingScreenOpen) yield return null;
259
260 yield return new WaitForSeconds(0.01f);
261
262 // Room 1
263 SoundPlayer.PlayDamageSound("StructureBlunt", 10, Character.Controlled.WorldPosition);
264 while (shakeTimer > 0.0f) // Wake up, shake
265 {
266 shakeTimer -= 0.1f;
267 GameMain.GameScreen.Cam.Shake = shakeAmount;
268 yield return new WaitForSeconds(0.1f, false);
269 }
270
271 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Officer.Radio.WakeUp"), ChatMessageType.Radio, null);
272
273 // Room 2
274 do { yield return null; } while (!officer_equipmentObjectiveSensor.MotionDetected);
275 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Officer.Radio.Equipment"), ChatMessageType.Radio, null);
276 yield return new WaitForSeconds(3f, false);
277 //TriggerTutorialSegment(0, GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Select], GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Deselect]); // Retrieve equipment
278 SetHighlight(officer_equipmentCabinet.Item, true);
279 bool firstSlotRemoved = false;
280 bool secondSlotRemoved = false;
281 bool thirdSlotRemoved = false;
282 do
283 {
284 if (IsSelectedItem(officer_equipmentCabinet.Item))
285 {
286 if (!firstSlotRemoved)
287 {
288 HighlightInventorySlot(officer_equipmentCabinet.Inventory, 0, highlightColor, .5f, .5f, 0f);
289 if (officer_equipmentCabinet.Inventory.GetItemAt(0) == null) { firstSlotRemoved = true; }
290 }
291
292 if (!secondSlotRemoved)
293 {
294 HighlightInventorySlot(officer_equipmentCabinet.Inventory, 1, highlightColor, .5f, .5f, 0f);
295 if (officer_equipmentCabinet.Inventory.GetItemAt(1) == null) { secondSlotRemoved = true; }
296 }
297
298 if (!thirdSlotRemoved)
299 {
300 HighlightInventorySlot(officer_equipmentCabinet.Inventory, 2, highlightColor, .5f, .5f, 0f);
301 if (officer_equipmentCabinet.Inventory.GetItemAt(2) == null) { thirdSlotRemoved = true; }
302 }
303
304 for (int i = 0; i < officer.Inventory.visualSlots.Length; i++)
305 {
306 if (officer.Inventory.GetItemAt(i) == null) { HighlightInventorySlot(officer.Inventory, i, highlightColor, .5f, .5f, 0f); }
307 }
308 }
309
310 yield return null;
311 } while (!officer_equipmentCabinet.Inventory.IsEmpty()); // Wait until looted
312 //RemoveCompletedObjective(segments[0]);
313 SetHighlight(officer_equipmentCabinet.Item, false);
314 do { yield return null; } while (IsSelectedItem(officer_equipmentCabinet.Item));
315 TriggerTutorialSegment(1, GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Aim), GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Shoot)); // Equip melee weapon & armor
316 do
317 {
318 if (!officer.HasEquippedItem("stunbaton".ToIdentifier()))
319 {
320 HighlightInventorySlot(officer.Inventory, "stunbaton".ToIdentifier(), highlightColor, .5f, .5f, 0f);
321 }
322 if (!officer.HasEquippedItem("bodyarmor".ToIdentifier()))
323 {
324 HighlightInventorySlot(officer.Inventory, "bodyarmor".ToIdentifier(), highlightColor, .5f, .5f, 0f);
325 }
326 if (!officer.HasEquippedItem("ballistichelmet1".ToIdentifier()))
327 {
328 HighlightInventorySlot(officer.Inventory, "ballistichelmet1".ToIdentifier(), highlightColor, .5f, .5f, 0f);
329 }
330 yield return new WaitForSeconds(1f, false);
331 } while (!officer.HasEquippedItem("stunbaton".ToIdentifier()) || !officer.HasEquippedItem("bodyarmor".ToIdentifier()) || !officer.HasEquippedItem("ballistichelmet1".ToIdentifier()));
332 RemoveCompletedObjective(1);
333 SetDoorAccess(officer_firstDoor, officer_firstDoorLight, true);
334
335 // Room 3
336 do { yield return null; } while (!officer_crawlerSensor.MotionDetected);

342 yield return new WaitForSeconds(1f, false);
343 GameMain.GameSession?.CrewManager.AddSinglePlayerChatMessage(radioSpeakerName, TextManager.Get("Officer.Radio.CrawlerDead"), ChatMessageType.Radio, null);
344 SetDoorAccess(officer_secondDoor, officer_secondDoorLight, true);
345
346 // Room 4
347 do { yield return null; } while (!officer_somethingBigSensor.MotionDetected);

349 yield return new WaitForSeconds(2f, false);

364 SetHighlight(officer_coilgunLoader.Item, false);
365 SetHighlight(officer_superCapacitor.Item, false);
