510 public IEnumerable<CoroutineStatus> KeepPatientAlive(Character patient)
511 {
512 while (patient != null && !patient.Removed)
513 {
514 patient.Oxygen = Math.Max(patient.Oxygen, -50);
515 yield return null;
516 }
517 }
