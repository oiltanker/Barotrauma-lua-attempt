330  private void DrawInputOverLay(SpriteBatch spriteBatch, GUICustomComponent overlayComponent)
331  {
332  overlayComponent.RectTransform.SetAsLastChild();
333 
334  FabricationRecipe targetItem = fabricatedItem ?? selectedItem;
335  if (targetItem != null)
336  {
337  int slotIndex = 0;
338 
339  var missingItems = new List<FabricationRecipe.RequiredItem>();
340 
341  foreach (FabricationRecipe.RequiredItem requiredItem in targetItem.RequiredItems)
342  {
343  for (int i = 0; i < requiredItem.Amount; i++)
344  {
345  missingItems.Add(requiredItem);
346  }
347  }
348  foreach (Item item in inputContainer.Inventory.AllItems)
349  {
350  missingItems.Remove(missingItems.FirstOrDefault(mi => mi.ItemPrefabs.Contains(item.Prefab)));
351  }
352  var missingCounts = missingItems.GroupBy(missingItem => missingItem).ToDictionary(x => x.Key, x => x.Count());
353  missingItems = missingItems.Distinct().ToList();
354 
355  foreach (FabricationRecipe.RequiredItem requiredItem in missingItems)
356  {
357  while (slotIndex < inputContainer.Capacity && inputContainer.Inventory.GetItemAt(slotIndex) != null)
358  {
359  slotIndex++;
360  }
361 
362  requiredItem.ItemPrefabs
363  .Where(requiredPrefab => availableIngredients.ContainsKey(requiredPrefab.Identifier))
364  .ForEach(requiredPrefab => {
365  var availableItems = availableIngredients[requiredPrefab.Identifier];
366  foreach (Item it in availableItems)
367  {
368  if (it.ParentInventory == inputContainer.Inventory) { continue; }
369  var rootInventoryOwner = it.GetRootInventoryOwner();
370  Inventory rootInventory = (rootInventoryOwner as Item)?.OwnInventory as Inventory ?? (rootInventoryOwner as Character)?.Inventory;
371  if (rootInventory?.visualSlots == null) { continue; }
372  int availableSlotIndex = rootInventory.FindIndex((it.Container != rootInventoryOwner ? it.Container : it) ?? it);
373  if (availableSlotIndex < 0) { continue; }
374  if (rootInventory.visualSlots[availableSlotIndex].HighlightTimer <= 0.0f)
375  {
376  rootInventory.visualSlots[availableSlotIndex].ShowBorderHighlight(GUIStyle.Green, 0.5f, 0.5f, 0.2f);
377  if (slotIndex < inputContainer.Capacity)
378  {
379  inputContainer.Inventory.visualSlots[slotIndex].ShowBorderHighlight(GUIStyle.Green, 0.5f, 0.5f, 0.2f);
380  }
381  }
382  }
383  });
384 
385  if (slotIndex >= inputContainer.Capacity) { break; }
386 
387  var itemIcon = requiredItem.ItemPrefabs.First().InventoryIcon ?? requiredItem.ItemPrefabs.First().Sprite;
388  Rectangle slotRect = inputContainer.Inventory.visualSlots[slotIndex].Rect;
389  itemIcon.Draw(
390  spriteBatch,
391  slotRect.Center.ToVector2(),
392  color: requiredItem.ItemPrefabs.First().InventoryIconColor * 0.3f,
393  scale: Math.Min(slotRect.Width / itemIcon.size.X, slotRect.Height / itemIcon.size.Y));
394 
395 
396  if (missingCounts[requiredItem] > 1)
397  {
398  Vector2 stackCountPos = new Vector2(slotRect.Right, slotRect.Bottom);
399  string stackCountText = "x" + missingCounts[requiredItem];
400  stackCountPos -= GUIStyle.SmallFont.MeasureString(stackCountText) + new Vector2(4, 2);
401  GUIStyle.SmallFont.DrawString(spriteBatch, stackCountText, stackCountPos + Vector2.One, Color.Black);
402  GUIStyle.SmallFont.DrawString(spriteBatch, stackCountText, stackCountPos, Color.White);
403  }
404 
405  if (requiredItem.UseCondition && requiredItem.MinCondition < 1.0f)
406  {
407  DrawConditionBar(spriteBatch, requiredItem.MinCondition);
408  }
409  else if (requiredItem.MaxCondition < 1.0f)
