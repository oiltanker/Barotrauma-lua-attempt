142  public void ClientEventRead(IReadMessage msg, float sendingTime)
143  {
144  if (GameMain.Client.MidRoundSyncing)
145  {
146  //delay reading the state until midround syncing is done
147  //because some of the wires connected to the panel may not exist yet
148  long msgStartPos = msg.BitPosition;
149  msg.ReadUInt16(); //user ID
150  foreach (Connection connection in Connections)
151  {
152  for (int i = 0; i < connection.MaxWires; i++)
153  {
154  msg.ReadUInt16();
155  }
156  }
157  ushort disconnectedWireCount = msg.ReadUInt16();
158  for (int i = 0; i < disconnectedWireCount; i++)
159  {
160  msg.ReadUInt16();
161  }
162  int msgLength = (int)(msg.BitPosition - msgStartPos);
163  msg.BitPosition = (int)msgStartPos;
164  StartDelayedCorrection(msg.ExtractBits(msgLength), sendingTime, waitForMidRoundSync: true);
165  }
166  else
167  {
168  //don't trigger rewiring sounds if the rewiring is being done by the local user (in that case we'll trigger it locally)
169  if (Character.Controlled == null || user != Character.Controlled) { TriggerRewiringSound(); }
170  ApplyRemoteState(msg);
171  }

174  private void ApplyRemoteState(IReadMessage msg)
175  {
176  List<Wire> prevWires = Connections.SelectMany(c => c.Wires.Where(w => w != null)).ToList();
177  List<Wire> newWires = new List<Wire>();
178 
179  ushort userID = msg.ReadUInt16();
180 
181  if (userID == 0)
182  {
183  user = null;
184  }
185  else
186  {
187  user = Entity.FindEntityByID(userID) as Character;
188  base.IsActive = true;
189  }
190 
191  foreach (Connection connection in Connections)
192  {
193  connection.ClearConnections();
194  }
195 
196  foreach (Connection connection in Connections)
197  {
198  for (int i = 0; i < connection.MaxWires; i++)
199  {
200  ushort wireId = msg.ReadUInt16();
201 
202  if (!(Entity.FindEntityByID(wireId) is Item wireItem)) { continue; }
203  Wire wireComponent = wireItem.GetComponent<Wire>();
204  if (wireComponent == null) { continue; }
205 
206  newWires.Add(wireComponent);
207 
208  connection.SetWire(i, wireComponent);
209  wireComponent.Connect(connection, false);
210  }
211  }
212 
213  List<Wire> previousDisconnectedWires = new List<Wire>(DisconnectedWires);
214  DisconnectedWires.Clear();
215  ushort disconnectedWireCount = msg.ReadUInt16();
216  for (int i = 0; i < disconnectedWireCount; i++)
217  {
218  ushort wireId = msg.ReadUInt16();
219  if (!(Entity.FindEntityByID(wireId) is Item wireItem)) { continue; }
220  Wire wireComponent = wireItem.GetComponent<Wire>();
221  if (wireComponent == null) { continue; }
222  DisconnectedWires.Add(wireComponent);
223  base.IsActive = true;
224  }
225 
226  foreach (Wire wire in prevWires)
227  {
228  bool connected = wire.Connections[0] != null || wire.Connections[1] != null;
229  if (!connected)
230  {
231  foreach (Item item in Item.ItemList)
232  {
233  var connectionPanel = item.GetComponent<ConnectionPanel>();
234  if (connectionPanel != null && connectionPanel.DisconnectedWires.Contains(wire))
235  {
236  connected = true;
237  break;
238  }
239  }
240  }
241  if (wire.Item.ParentInventory == null && !connected)
242  {
243  wire.Item.Drop(null);
244  }
245  }
246 
247  foreach (Wire disconnectedWire in previousDisconnectedWires)
248  {
249  if (disconnectedWire.Connections[0] == null &&
250  disconnectedWire.Connections[1] == null &&
251  !DisconnectedWires.Contains(disconnectedWire))
252  {
253  disconnectedWire.Item.Drop(dropper: null);
254  }
255  }
256  }
257  }
