604 private float GetRotationAngle(Vector2 drawPosition)
605 {
606 Vector2 mouseVector = Screen.Selected.Cam.ScreenToWorld(PlayerInput.MousePosition);
607 mouseVector.Y = -mouseVector.Y;
608 Vector2 rotationVector = mouseVector - drawPosition;
609 rotationVector.Normalize();
610 double angle = Math.Atan2(MathHelper.ToRadians(rotationVector.Y), MathHelper.ToRadians(rotationVector.X));
611 if (angle < 0)
612 {// calculates which coterminal angle is closer to previous angle
613 angle = Math.Abs(angle - prevAngle) < Math.Abs((angle + Math.PI * 2) - prevAngle) ? angle : angle + Math.PI * 2;
614 }
615 else if (angle > 0)
616 {
617 angle = Math.Abs(angle - prevAngle) < Math.Abs((angle - Math.PI * 2) - prevAngle) ? angle : angle - Math.PI * 2;
618 }
619 angle = MathHelper.Clamp((float)angle, -((float)Math.PI * 2), (float)Math.PI * 2);
620 prevAngle = (float)angle;
621 return (float)angle;
622 }

632 GetAvailablePower(out float batteryCharge, out float batteryCapacity);
633
634 List<Item> availableAmmo = new List<Item>();
635 foreach (MapEntity e in item.linkedTo)
636 {
637 if (!(e is Item linkedItem)) { continue; }
638 var itemContainer = linkedItem.GetComponent<ItemContainer>();
639 if (itemContainer == null) { continue; }
640 availableAmmo.AddRange(itemContainer.Inventory.AllItems);
641 for (int i = 0; i < itemContainer.Inventory.Capacity - itemContainer.Inventory.AllItems.Count(); i++)
642 {
643 availableAmmo.Add(null);
644 }
645 }
646
647 float chargeRate =
648 powerConsumption <= 0.0f ?
649 1.0f :
650 batteryCapacity > 0.0f ? batteryCharge / batteryCapacity : 0.0f;
651 bool charged = batteryCharge * 3600.0f > powerConsumption;
652 bool readyToFire = reload <= 0.0f && charged && availableAmmo.Any(p => p != null);
653 if (ShowChargeIndicator && PowerConsumption > 0.0f)
654 {
655 powerIndicator.Color = charged ?
656 (HasPowerToShoot() ? GUIStyle.Green : GUIStyle.Orange) :
657 GUIStyle.Red;
658 if (flashLowPower)
659 {
660 powerIndicator.BarSize = 1;
661 powerIndicator.Color *= (float)Math.Sin(flashTimer * 12);
662 powerIndicator.RectTransform.ChangeScale(Vector2.Lerp(Vector2.One, Vector2.One * 1.01f, 2 * (float)Math.Sin(flashTimer * 15)));
663 }
664 else
665 {
666 powerIndicator.BarSize = chargeRate;
667 }
668 powerIndicator.DrawManually(spriteBatch, true);
669
670 Rectangle sliderRect = powerIndicator.GetSliderRect(1.0f);
671 int requiredChargeIndicatorPos = (int)(powerConsumption / (batteryCapacity * 3600.0f) * sliderRect.Width);
672 GUI.DrawRectangle(spriteBatch,
673 new Rectangle(sliderRect.X + requiredChargeIndicatorPos, sliderRect.Y, 2, sliderRect.Height),
674 Color.White * 0.5f, true);
675 }
676
677 if (ShowProjectileIndicator)
678 {
679 Point slotSize = (Inventory.SlotSpriteSmall.size * Inventory.UIScale).ToPoint();
680 Point spacing = new Point(GUI.IntScale(5), GUI.IntScale(20));
681 int slotsPerRow = Math.Min(availableAmmo.Count, 6);
