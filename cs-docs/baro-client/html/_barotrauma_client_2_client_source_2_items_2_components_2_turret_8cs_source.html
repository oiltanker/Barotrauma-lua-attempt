604  private float GetRotationAngle(Vector2 drawPosition)
605  {
606  Vector2 mouseVector = Screen.Selected.Cam.ScreenToWorld(PlayerInput.MousePosition);
607  mouseVector.Y = -mouseVector.Y;
608  Vector2 rotationVector = mouseVector - drawPosition;
609  rotationVector.Normalize();
610  double angle = Math.Atan2(MathHelper.ToRadians(rotationVector.Y), MathHelper.ToRadians(rotationVector.X));
611  if (angle < 0)
612  {// calculates which coterminal angle is closer to previous angle
613  angle = Math.Abs(angle - prevAngle) < Math.Abs((angle + Math.PI * 2) - prevAngle) ? angle : angle + Math.PI * 2;
614  }
615  else if (angle > 0)
616  {
617  angle = Math.Abs(angle - prevAngle) < Math.Abs((angle - Math.PI * 2) - prevAngle) ? angle : angle - Math.PI * 2;
618  }
619  angle = MathHelper.Clamp((float)angle, -((float)Math.PI * 2), (float)Math.PI * 2);
620  prevAngle = (float)angle;
621  return (float)angle;
622  }
