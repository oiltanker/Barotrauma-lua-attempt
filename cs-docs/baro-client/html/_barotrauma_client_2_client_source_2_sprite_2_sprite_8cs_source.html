137  public static Texture2D LoadTexture(string file, bool compress = true)
138  {
139  if (string.IsNullOrWhiteSpace(file))
140  {
141  Texture2D t = null;
142  CrossThread.RequestExecutionOnMainThread(() =>
143  {
144  t = new Texture2D(GameMain.GraphicsDeviceManager.GraphicsDevice, 1, 1);
145  });
146  return t;
147  }
148  Identifier fullPath = Path.GetFullPath(file).CleanUpPathCrossPlatform(correctFilenameCase: false).ToIdentifier();
149  lock (list)
150  {
151  if (textureRefCounts.ContainsKey(fullPath))
152  {
153  textureRefCounts[fullPath].RefCount++;
154  return textureRefCounts[fullPath].Texture;
155  }
156  }
157 
158  if (File.Exists(file))
159  {
160  if (!ToolBox.IsProperFilenameCase(file))
161  {
162 #if DEBUG
163  DebugConsole.ThrowError("Texture file \"" + file + "\" has incorrect case!");
164 #endif
165  }
166 
167  Texture2D newTexture = TextureLoader.FromFile(file, compress);
168  lock (list)
169  {
170  if (!textureRefCounts.TryAdd(fullPath,
171  new TextureRefCounter { RefCount = 1, Texture = newTexture }))
172  {
173  CrossThread.RequestExecutionOnMainThread(() => newTexture.Dispose());
174  textureRefCounts[fullPath].RefCount++;
175  return textureRefCounts[fullPath].Texture;
176  }
177  }
178  return newTexture;
179  }
180  else
181  {
182  DebugConsole.ThrowError($"Sprite \"{file}\" not found!");
183  DebugConsole.Log(Environment.StackTrace.CleanupStackTrace());
184  }
185 
186  return null;
187  }

226  public void DrawTiled(ISpriteBatch spriteBatch, Vector2 position, Vector2 targetSize,
227  Color? color = null, Vector2? startOffset = null, Vector2? textureScale = null, float? depth = null)
228  {
229  if (Texture == null) { return; }
230  //Init optional values
231  Vector2 drawOffset = startOffset ?? Vector2.Zero;
232  Vector2 scale = textureScale ?? Vector2.One;
233  Color drawColor = color ?? Color.White;
234 
235  bool flipHorizontal = (effects & SpriteEffects.FlipHorizontally) != 0;
236  bool flipVertical = (effects & SpriteEffects.FlipVertically) != 0;
237 
238  //wrap the drawOffset inside the sourceRect
239  drawOffset.X = (drawOffset.X / scale.X) % sourceRect.Width;
240  drawOffset.Y = (drawOffset.Y / scale.Y) % sourceRect.Height;
241 
242  Vector2 flippedDrawOffset = Vector2.Zero;
243  if (flipHorizontal)
244  {
245  float diff = targetSize.X % (sourceRect.Width * scale.X);
246  flippedDrawOffset.X = (int)((sourceRect.Width * scale.X - diff) / scale.X);
247  }
248  if (flipVertical)
249  {
250  float diff = targetSize.Y % (sourceRect.Height * scale.Y);
251  flippedDrawOffset.Y = (int)((sourceRect.Height * scale.Y - diff) / scale.Y);
252  }
253  drawOffset += flippedDrawOffset;
254 
255  //how many times the texture needs to be drawn on the x-axis
256  int xTiles = (int)Math.Ceiling((targetSize.X + drawOffset.X * scale.X) / (sourceRect.Width * scale.X));
257  //how many times the texture needs to be drawn on the y-axis
258  int yTiles = (int)Math.Ceiling((targetSize.Y + drawOffset.Y * scale.Y) / (sourceRect.Height * scale.Y));
259 
260  //where the current tile is being drawn;
261  Vector2 currDrawPosition = position - drawOffset;
262  //which part of the texture we are currently drawing
263  Rectangle texPerspective = sourceRect;
264 
265 
266  for (int x = 0; x < xTiles; x++)
267  {
268  texPerspective.X = sourceRect.X;
269  texPerspective.Width = sourceRect.Width;
270  texPerspective.Height = sourceRect.Height;
271 
272  //offset to the left, draw a partial slice
273  if (currDrawPosition.X < position.X)
274  {
275  float diff = (position.X - currDrawPosition.X);
276  currDrawPosition.X += diff;
277  texPerspective.Width -= (int)diff;
278  if (!flipHorizontal)
279  {
280  texPerspective.X += (int)diff;
281  }
282  if (!flipVertical)
283  {
284  texPerspective.Y += (int)diff;
285  }
286  }
287  //drawing an offset flipped sprite, need to draw an extra slice to the left side
288  if (currDrawPosition.X > position.X && x == 0)
289  {
290  if (flipHorizontal)
291  {
292  int sliceWidth = (int)((currDrawPosition.X - position.X) * scale.X);
293 
294  Vector2 slicePos = currDrawPosition;
295  slicePos.X = position.X;
296  Rectangle sliceRect = texPerspective;
297  sliceRect.X = SourceRect.X;
298  sliceRect.Width = (int)(sliceWidth / scale.X);
299 
300  if (flipVertical)
301  {
302  slicePos.Y += flippedDrawOffset.Y;
303  }
304 
305  spriteBatch.Draw(texture, slicePos, sliceRect, drawColor, rotation, Vector2.Zero, scale, effects, depth ?? this.depth);
306  currDrawPosition.X = slicePos.X + sliceWidth;
307  }
308  }
309  //make sure the rightmost tiles don't go over the right side
310  if (x == xTiles - 1)
311  {
312  int diff = (int)(((currDrawPosition.X + texPerspective.Width * scale.X) - (position.X + targetSize.X)) / scale.X);
313  texPerspective.Width -= diff;
314  if (flipHorizontal)
315  {
316  texPerspective.X += diff;
317  }
318  }
319 
320  currDrawPosition.Y = position.Y - drawOffset.Y;
321 
322  for (int y = 0; y < yTiles; y++)
323  {
324  texPerspective.Y = sourceRect.Y;
325  texPerspective.Height = sourceRect.Height;
326 
327  //offset above the top, draw a partial slice
328  if (currDrawPosition.Y < position.Y)
329  {
330  float diff = (position.Y - currDrawPosition.Y);
331  currDrawPosition.Y += diff;
332  texPerspective.Height -= (int)diff;
333  if (!flipVertical)
334  {
335  texPerspective.Y += (int)diff;
336  }
337  }
338 
339  //drawing an offset flipped sprite, need to draw an extra slice to the top
340  if (currDrawPosition.Y > position.Y && y == 0)
341  {
342  if (flipVertical)
343  {
344  int sliceHeight = (int)((currDrawPosition.Y - position.Y) * scale.Y);
345 
346  Vector2 slicePos = currDrawPosition;
347  slicePos.Y = position.Y;
348  Rectangle sliceRect = texPerspective;
349  sliceRect.Y = SourceRect.Y;
350  sliceRect.Height = (int)(sliceHeight / scale.Y);
351 
352  spriteBatch.Draw(texture, slicePos, sliceRect, drawColor, rotation, Vector2.Zero, scale, effects, depth ?? this.depth);
353 
354  currDrawPosition.Y = slicePos.Y + sliceHeight;
355  }
356  }
357 
358  //make sure the bottommost tiles don't go over the bottom
359  if (y == yTiles - 1)
360  {
361  int diff = (int)(((currDrawPosition.Y + texPerspective.Height * scale.Y) - (position.Y + targetSize.Y)) / scale.Y);
362  texPerspective.Height -= diff;
363  if (flipVertical)
364  {
365  texPerspective.Y += diff;
366  }
367  }
368 
369  spriteBatch.Draw(texture, currDrawPosition,
370  texPerspective, drawColor, rotation, Vector2.Zero, scale, effects, depth ?? this.depth);
371 
372  currDrawPosition.Y += texPerspective.Height * scale.Y;
373  }
374 
375  currDrawPosition.X += texPerspective.Width * scale.X;
376  }
377  }
