492 public void DrawHUD(SpriteBatch spriteBatch, Rectangle rect)
493 {
494 int width = rect.Width, height = rect.Height;
495 int x = rect.X;
496 int y = rect.Y;
497
498 if (Voltage < MinVoltage) { return; }
499
500 Rectangle velRect = new Rectangle(x + 20, y + 20, width - 40, height - 40);
501 Vector2 steeringOrigin = steerArea.Rect.Center.ToVector2();
502
503 if (!AutoPilot)
504 {
505 Vector2 unitSteeringInput = steeringInput / 100.0f;
506 //map input from rectangle to circle
507 Vector2 steeringInputPos = new Vector2(
508 steeringInput.X * (float)Math.Sqrt(1.0f - 0.5f * unitSteeringInput.Y * unitSteeringInput.Y),
509 -steeringInput.Y * (float)Math.Sqrt(1.0f - 0.5f * unitSteeringInput.X * unitSteeringInput.X));
510 steeringInputPos += steeringOrigin;
511
512 if (steeringIndicator != null)
513 {
514 Vector2 dir = steeringInputPos - steeringOrigin;
515 float angle = (float)Math.Atan2(dir.Y, dir.X);
516 steeringIndicator.Draw(spriteBatch, steeringOrigin, Color.White, origin: steeringIndicator.Origin, rotate: angle,
517 scale: new Vector2(dir.Length() / steeringIndicator.size.X, 1.0f));
518 }
519 else
520 {
521 GUI.DrawLine(spriteBatch, steeringOrigin, steeringInputPos, Color.LightGray);
522 GUI.DrawRectangle(spriteBatch, new Rectangle((int)steeringInputPos.X - 5, (int)steeringInputPos.Y - 5, 10, 10), Color.White);
523 }
524
525 if (velRect.Contains(PlayerInput.MousePosition))
526 {
527 GUI.DrawRectangle(spriteBatch, new Rectangle((int)steeringInputPos.X - 4, (int)steeringInputPos.Y - 4, 8, 8), GUIStyle.Red, thickness: 2);
528 }
529 }
530 else if (posToMaintain.HasValue && !LevelStartSelected && !LevelEndSelected)
531 {
532 Sonar sonar = item.GetComponent<Sonar>();
533 if (sonar != null && controlledSub != null)
534 {
535 Vector2 displayPosToMaintain = ((posToMaintain.Value - controlledSub.WorldPosition)) / sonar.Range * sonar.DisplayRadius * sonar.Zoom;
536 displayPosToMaintain.Y = -displayPosToMaintain.Y;
537 displayPosToMaintain = displayPosToMaintain.ClampLength(velRect.Width / 2);
538 displayPosToMaintain = steerArea.Rect.Center.ToVector2() + displayPosToMaintain;
539
540 Color crosshairColor = GUIStyle.Orange * (0.5f + ((float)Math.Sin(Timing.TotalTime * 5.0f) + 1.0f) / 4.0f);
541 if (maintainPosIndicator != null)
542 {
543 maintainPosIndicator.Draw(spriteBatch, displayPosToMaintain, crosshairColor, scale: 0.5f * sonar.Zoom);
544 }
545 else
546 {
547 float crossHairSize = 8.0f;
548 GUI.DrawLine(spriteBatch, displayPosToMaintain + Vector2.UnitY * crossHairSize, displayPosToMaintain - Vector2.UnitY * crossHairSize, crosshairColor, width: 3);
549 GUI.DrawLine(spriteBatch, displayPosToMaintain + Vector2.UnitX * crossHairSize, displayPosToMaintain - Vector2.UnitX * crossHairSize, crosshairColor, width: 3);
550 }
551
552 if (maintainPosOriginIndicator != null)
553 {
554 maintainPosOriginIndicator.Draw(spriteBatch, steeringOrigin, GUIStyle.Orange, scale: 0.5f * sonar.Zoom);
555 }
556 else
557 {
558 GUI.DrawRectangle(spriteBatch, new Rectangle((int)steeringOrigin.X - 5, (int)steeringOrigin.Y - 5, 10, 10), GUIStyle.Orange);
559 }
560 }
561 }
562
563 //map velocity from rectangle to circle
564 Vector2 unitTargetVel = targetVelocity / 100.0f;
565 Vector2 steeringPos = new Vector2(
566 targetVelocity.X * 0.9f * (float)Math.Sqrt(1.0f - 0.5f * unitTargetVel.Y * unitTargetVel.Y),
567 -targetVelocity.Y * 0.9f * (float)Math.Sqrt(1.0f - 0.5f * unitTargetVel.X * unitTargetVel.X));
568 steeringPos += steeringOrigin;
569
570 if (steeringIndicator != null)
571 {
572 Vector2 dir = steeringPos - steeringOrigin;
573 float angle = (float)Math.Atan2(dir.Y, dir.X);
574 steeringIndicator.Draw(spriteBatch, steeringOrigin, Color.Gray, origin: steeringIndicator.Origin, rotate: angle,
575 scale: new Vector2(dir.Length() / steeringIndicator.size.X, 0.7f));
576 }
577 else
578 {
579 GUI.DrawLine(spriteBatch,
580 steeringOrigin,
581 steeringPos,
582 Color.CadetBlue, 0, 2);
583 }
584 }

586 public void DebugDrawHUD(SpriteBatch spriteBatch, Vector2 transducerCenter, float displayScale, float displayRadius, Vector2 center)
587 {
588 if (SteeringPath == null) return;
589
590 Vector2 prevPos = Vector2.Zero;
591 foreach (WayPoint wp in SteeringPath.Nodes)
592 {
593 Vector2 pos = (wp.Position - transducerCenter) * displayScale;

795 float maxCumulation = 1;
