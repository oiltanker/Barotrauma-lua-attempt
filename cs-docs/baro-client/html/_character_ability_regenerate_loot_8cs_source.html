18
19 public CharacterAbilityRegenerateLoot(CharacterAbilityGroup characterAbilityGroup, ContentXElement abilityElement) : base(characterAbilityGroup, abilityElement)

22 }
23
24 protected override void ApplyEffect(AbilityObject abilityObject)
25 {
26 if ((abilityObject as IAbilityItem)?.Item is Item item)
27 {
28 if (openedContainers.Contains(item)) { return; }
29 openedContainers.Add(item);
30 if (randomChance < Rand.Range(0f, 1f, Rand.RandSync.Unsynced)) { return; }
31
32 if (item.GetComponent<ItemContainer>() is ItemContainer itemContainer)

34 AutoItemPlacer.RegenerateLoot(item.Submarine, itemContainer);
