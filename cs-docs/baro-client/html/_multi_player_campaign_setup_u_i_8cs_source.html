195 private IEnumerable<CoroutineStatus> WaitForCampaignSetup()
196 {
197 GUI.SetCursorWaiting();
198 var headerText = TextManager.Get("CampaignStartingPleaseWait");
199 var msgBox = new GUIMessageBox(headerText, TextManager.Get("CampaignStarting"), new LocalizedString[] { TextManager.Get("Cancel") });
200
201 msgBox.Buttons[0].OnClicked = (btn, userdata) =>
202 {
203 GameMain.NetLobbyScreen.HighlightMode(GameMain.NetLobbyScreen.SelectedModeIndex);
204 GameMain.NetLobbyScreen.SelectMode(GameMain.NetLobbyScreen.SelectedModeIndex);
205 GUI.ClearCursorWait();
206 CoroutineManager.StopCoroutines("WaitForCampaignSetup");
207 return true;
208 };
209 msgBox.Buttons[0].OnClicked += msgBox.Close;
210
211 DateTime timeOut = DateTime.Now + new TimeSpan(0, 0, 20);
212 while (Screen.Selected != GameMain.GameScreen && DateTime.Now < timeOut)
213 {
214 msgBox.Header.Text = headerText + new string('.', (int)Timing.TotalTime % 3 + 1);
215 yield return CoroutineStatus.Running;
216 }
217 msgBox.Close();
218 GUI.ClearCursorWait();
219 yield return CoroutineStatus.Success;
220 }

249 saveList.Content.RectTransform.SortChildren((c1, c2) =>
250 {
251 string file1 = c1.GUIComponent.UserData as string;
252 string file2 = c2.GUIComponent.UserData as string;
253 DateTime file1WriteTime = DateTime.MinValue;
254 DateTime file2WriteTime = DateTime.MinValue;
255 try
256 {
257 file1WriteTime = File.GetLastWriteTime(file1);
258 }
259 catch
260 {
261 //do nothing - DateTime.MinValue will be used and the element will get sorted at the bottom of the list
262 };
263 try
264 {
265 file2WriteTime = File.GetLastWriteTime(file2);
266 }
267 catch
268 {
269 //do nothing - DateTime.MinValue will be used and the element will get sorted at the bottom of the list
270 };
271 return file2WriteTime.CompareTo(file1WriteTime);
272 });

307 private bool DeleteSave(GUIButton button, object obj)
308 {
309 string saveFile = obj as string;
310 if (obj == null) { return false; }
311
312 var header = TextManager.Get("deletedialoglabel");
313 var body = TextManager.GetWithVariable("deletedialogquestion", "[file]", Path.GetFileNameWithoutExtension(saveFile));
314
315 EventEditorScreen.AskForConfirmation(header, body, () =>
316 {
317 SaveUtil.DeleteSave(saveFile);
318 prevSaveFiles?.RemoveAll(s => s.FilePath == saveFile);
319 UpdateLoadMenu(prevSaveFiles.ToList());
320 return true;
321 });
322
323 return true;
