406  private void UpdateAutoPilot(float deltaTime)
407  {
408  if (controlledSub == null) { return; }
409  if (posToMaintain != null)
410  {
411  Vector2 steeringVel = GetSteeringVelocity((Vector2)posToMaintain, 10.0f);
412  TargetVelocity = Vector2.Lerp(TargetVelocity, steeringVel, AutoPilotSteeringLerp);
413  showIceSpireWarning = false;
414  return;
415  }
416 
417  autopilotRayCastTimer -= deltaTime;
418  autopilotRecalculatePathTimer -= deltaTime;
419  if (autopilotRecalculatePathTimer <= 0.0f)
420  {
421  //periodically recalculate the path in case the sub ends up to a position
422  //where it can't keep traversing the initially calculated path
423  UpdatePath();
424  autopilotRecalculatePathTimer = RecalculatePathInterval;
425  }
426 
427  if (steeringPath == null)
428  {
429  showIceSpireWarning = false;
430  return;
431  }
432  steeringPath.CheckProgress(ConvertUnits.ToSimUnits(controlledSub.WorldPosition), 10.0f);
433 
434  connectedSubUpdateTimer -= deltaTime;
435  if (connectedSubUpdateTimer <= 0.0f)
436  {
437  connectedSubs.Clear();
438  connectedSubs = controlledSub?.GetConnectedSubs();
439  connectedSubUpdateTimer = ConnectedSubUpdateInterval;
440  }
441 
442  if (autopilotRayCastTimer <= 0.0f && steeringPath.NextNode != null)
443  {
444  Vector2 diff = ConvertUnits.ToSimUnits(steeringPath.NextNode.Position - controlledSub.WorldPosition);
445 
446  //if the node is close enough, check if it's visible
447  float lengthSqr = diff.LengthSquared();
448  if (lengthSqr > 0.001f && lengthSqr < AutopilotMinDistToPathNode * AutopilotMinDistToPathNode)
449  {
450  diff = Vector2.Normalize(diff);
451 
452  //check if the next waypoint is visible from all corners of the sub
453  //(i.e. if we can navigate directly towards it or if there's obstacles in the way)
454  bool nextVisible = true;
455  for (int x = -1; x < 2; x += 2)
456  {
457  for (int y = -1; y < 2; y += 2)
458  {
459  Vector2 cornerPos =
460  new Vector2(controlledSub.Borders.Width * x, controlledSub.Borders.Height * y) / 2.0f;
461 
462  cornerPos = ConvertUnits.ToSimUnits(cornerPos * 1.1f + controlledSub.WorldPosition);
463 
464  float dist = Vector2.Distance(cornerPos, steeringPath.NextNode.SimPosition);
465 
466  if (Submarine.PickBody(cornerPos, cornerPos + diff * dist, null, Physics.CollisionLevel) == null) { continue; }
467 
468  nextVisible = false;
469  x = 2;
470  y = 2;
471  }
472  }
473 
474  if (nextVisible) steeringPath.SkipToNextNode();
475  }
476 
477  autopilotRayCastTimer = AutopilotRayCastInterval;
478  }
479 
480  Vector2 newVelocity = Vector2.Zero;
481  if (steeringPath.CurrentNode != null)
482  {
483  newVelocity = GetSteeringVelocity(steeringPath.CurrentNode.WorldPosition, 2.0f);
484  }
485 
486  Vector2 avoidDist = new Vector2(
487  Math.Max(1000.0f * Math.Abs(controlledSub.Velocity.X), controlledSub.Borders.Width * 0.75f),
488  Math.Max(1000.0f * Math.Abs(controlledSub.Velocity.Y), controlledSub.Borders.Height * 0.75f));
489 
490  float avoidRadius = avoidDist.Length();
491  float damagingWallAvoidRadius = MathHelper.Clamp(avoidRadius * 1.5f, 5000.0f, 10000.0f);
492 
493  Vector2 newAvoidStrength = Vector2.Zero;
494 
495  debugDrawObstacles.Clear();
496 
497  //steer away from nearby walls
498  showIceSpireWarning = false;
499  var closeCells = Level.Loaded.GetCells(controlledSub.WorldPosition, 4);
500  foreach (VoronoiCell cell in closeCells)
501  {
502  if (cell.DoesDamage)
503  {
504  foreach (GraphEdge edge in cell.Edges)
505  {
506  Vector2 closestPoint = MathUtils.GetClosestPointOnLineSegment(edge.Point1 + cell.Translation, edge.Point2 + cell.Translation, controlledSub.WorldPosition);
507  Vector2 diff = closestPoint - controlledSub.WorldPosition;
508  float dist = diff.Length() - Math.Max(controlledSub.Borders.Width, controlledSub.Borders.Height) / 2;
509  if (dist > damagingWallAvoidRadius) { continue; }
510 
511  Vector2 normalizedDiff = Vector2.Normalize(diff);
512  float dot = Vector2.Dot(normalizedDiff, controlledSub.Velocity);
513 
514  float avoidStrength = MathHelper.Clamp(MathHelper.Lerp(1.0f, 0.0f, dist / damagingWallAvoidRadius - dot), 0.0f, 1.0f);
515  Vector2 avoid = -normalizedDiff * avoidStrength;
516  newAvoidStrength += avoid;
517  debugDrawObstacles.Add(new ObstacleDebugInfo(edge, edge.Center, 1.0f, avoid, cell.Translation));
518 
519  if (dot > 0.0f)
520  {
521  showIceSpireWarning = true;
522  }
523  }
524  continue;
525  }
526 
527  foreach (GraphEdge edge in cell.Edges)
528  {
529  if (MathUtils.GetLineIntersection(edge.Point1 + cell.Translation, edge.Point2 + cell.Translation, controlledSub.WorldPosition, cell.Center, out Vector2 intersection))
530  {
531  Vector2 diff = controlledSub.WorldPosition - intersection;
532  //far enough -> ignore
533  if (Math.Abs(diff.X) > avoidDist.X && Math.Abs(diff.Y) > avoidDist.Y)
534  {
535  debugDrawObstacles.Add(new ObstacleDebugInfo(edge, intersection, 0.0f, Vector2.Zero, Vector2.Zero));
536  continue;
537  }
538  if (diff.LengthSquared() < 1.0f) { diff = Vector2.UnitY; }
539 
540  Vector2 normalizedDiff = Vector2.Normalize(diff);
541  float dot = controlledSub.Velocity == Vector2.Zero ?
542  0.0f : Vector2.Dot(controlledSub.Velocity, -normalizedDiff);
543 
544  //not heading towards the wall -> ignore
545  if (dot < 1.0)
546  {
547  debugDrawObstacles.Add(new ObstacleDebugInfo(edge, intersection, dot, Vector2.Zero, cell.Translation));
548  continue;
549  }
550 
551  Vector2 change = (normalizedDiff * Math.Max((avoidRadius - diff.Length()), 0.0f)) / avoidRadius;
552  if (change.LengthSquared() < 0.001f) { continue; }
553  newAvoidStrength += change * (dot - 1.0f);
554  debugDrawObstacles.Add(new ObstacleDebugInfo(edge, intersection, dot - 1.0f, change * (dot - 1.0f), cell.Translation));
555  }
556  }
557  }
558 
559  avoidStrength = Vector2.Lerp(avoidStrength, newAvoidStrength, deltaTime * 10.0f);
560  TargetVelocity = Vector2.Lerp(TargetVelocity, newVelocity + avoidStrength * 100.0f, AutoPilotSteeringLerp);
561 
562  //steer away from other subs
563  foreach (Submarine sub in Submarine.Loaded)
564  {
565  if (sub == controlledSub || connectedSubs.Contains(sub)) { continue; }
566  Point sizeSum = controlledSub.Borders.Size + sub.Borders.Size;
567  Vector2 minDist = sizeSum.ToVector2() / 2;
568  Vector2 diff = controlledSub.WorldPosition - sub.WorldPosition;
569  float xDist = Math.Abs(diff.X);
570  float yDist = Math.Abs(diff.Y);
571  Vector2 maxAvoidDistance = minDist * 2;
572  if (xDist > maxAvoidDistance.X || yDist > maxAvoidDistance.Y)
573  {
574  //far enough -> ignore
575  continue;
576  }
577  float dot = controlledSub.Velocity == Vector2.Zero ? 0.0f : Vector2.Dot(Vector2.Normalize(controlledSub.Velocity), -diff);
578  if (dot < 0.0f)
579  {
580  //heading away -> ignore
581  continue;
582  }
583  float distanceFactor = MathHelper.Lerp(0, 1, MathUtils.InverseLerp(maxAvoidDistance.X + maxAvoidDistance.Y, minDist.X + minDist.Y, xDist + yDist));
584  float velocityFactor = MathHelper.Lerp(0, 1, MathUtils.InverseLerp(0, 3, controlledSub.Velocity.Length()));
585  TargetVelocity += 100 * Vector2.Normalize(diff) * distanceFactor * velocityFactor;
586  }
587 
588  //clamp velocity magnitude to 100.0f (Is this required? The X and Y components are clamped in the property setter)
589  float velMagnitude = TargetVelocity.Length();
590  if (velMagnitude > 100.0f)
591  {
592  TargetVelocity *= 100.0f / velMagnitude;
593  }
594 
595 #if CLIENT
596  HintManager.OnAutoPilotPathUpdated(this);
597 #endif
598  }

600  private float? GetNodePenalty(PathNode node, PathNode nextNode)
601  {
602  if (node.Waypoint?.Tunnel == null || controlledSub == null || node.Waypoint.Tunnel.Type == Level.TunnelType.MainPath) { return 0.0f; }
603  //never navigate from the main path to another type of path
604  if (node.Waypoint.Tunnel.Type == Level.TunnelType.MainPath && nextNode.Waypoint?.Tunnel?.Type != Level.TunnelType.MainPath) { return null; }
605  //higher cost for side paths (= autopilot prefers the main path, but can still navigate side paths if it ends up on one)
606  return 1000.0f;
