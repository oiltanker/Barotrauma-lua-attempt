59 public static RoundSound? Load(ContentXElement element, bool stream = false)
60 {
61 if (GameMain.SoundManager?.Disabled ?? true) { return null; }
62
63 var filename = element.GetAttributeContentPath("file") ?? element.GetAttributeContentPath("sound");
64
65 if (filename is null)
66 {
67 string errorMsg = "Error when loading round sound (" + element + ") - file path not set";
68 DebugConsole.ThrowError(errorMsg);
69 GameAnalyticsManager.AddErrorEventOnce("RoundSound.LoadRoundSound:FilePathEmpty" + element.ToString(), GameAnalyticsManager.ErrorSeverity.Error, errorMsg + "\n" + Environment.StackTrace.CleanupStackTrace());
70 return null;
71 }
72
73 Sound? existingSound = roundSounds.Find(s => s.Filename == filename?.FullPath && s.Stream == stream && s.Sound is { Disposed: false })?.Sound;
74
75 if (existingSound is null)
76 {
77 try
78 {
79 existingSound = GameMain.SoundManager.LoadSound(filename?.FullPath, stream);
80 if (existingSound == null) { return null; }
81 }
82 catch (System.IO.FileNotFoundException e)
83 {
84 string errorMsg = "Failed to load sound file \"" + filename + "\".";
85 DebugConsole.ThrowError(errorMsg, e);
86 GameAnalyticsManager.AddErrorEventOnce("RoundSound.LoadRoundSound:FileNotFound" + filename, GameAnalyticsManager.ErrorSeverity.Error, errorMsg + "\n" + Environment.StackTrace.CleanupStackTrace());
87 return null;
88 }
89 }
90
91 RoundSound newSound = new RoundSound(element, existingSound);
92
93 roundSounds.Add(newSound);
94 return newSound;
95 }
