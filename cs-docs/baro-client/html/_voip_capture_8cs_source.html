179 private void UpdateCapture()
180 {
181 Array.Copy(uncompressedBuffer, 0, prevUncompressedBuffer, 0, VoipConfig.BUFFER_SIZE);
182 Array.Clear(uncompressedBuffer, 0, VoipConfig.BUFFER_SIZE);
183 nativeBuffer = Marshal.AllocHGlobal(VoipConfig.BUFFER_SIZE * 2);
184 try
185 {
186 while (capturing)
187 {
188 int alcError;
189
190 if (CanDetectDisconnect)
191 {
192 Alc.GetInteger(captureDevice, Alc.EnumConnected, out int isConnected);
193 alcError = Alc.GetError(captureDevice);
194 if (alcError != Alc.NoError)
195 {
196 throw new Exception("Failed to determine if capture device is connected: " + alcError.ToString());
197 }
198
199 if (isConnected == 0)
200 {
201 DebugConsole.ThrowError("Capture device has been disconnected. You can select another available device in the settings.");
202 Disconnected = true;
203 break;
204 }
205 }
206
207 FillBuffer();
208
209 alcError = Alc.GetError(captureDevice);
210 if (alcError != Alc.NoError)
211 {
212 throw new Exception("Failed to capture samples: " + alcError.ToString());
213 }
214
215 double maxAmplitude = 0.0f;
216 for (int i = 0; i < VoipConfig.BUFFER_SIZE; i++)
217 {
218 uncompressedBuffer[i] = (short)MathHelper.Clamp((uncompressedBuffer[i] * Gain), -short.MaxValue, short.MaxValue);
219 double sampleVal = uncompressedBuffer[i] / (double)short.MaxValue;
220 maxAmplitude = Math.Max(maxAmplitude, Math.Abs(sampleVal));
221 }
222 double dB = Math.Min(20 * Math.Log10(maxAmplitude), 0.0);
223
224 LastdB = dB;
225 LastAmplitude = maxAmplitude;
226
227 bool allowEnqueue = overrideSound != null;
228 if (GameMain.WindowActive && SettingsMenu.Instance is null)
229 {
230 ForceLocal = captureTimer > 0 ? ForceLocal : GameSettings.CurrentConfig.Audio.UseLocalVoiceByDefault;
231 bool pttDown = false;
232 if ((PlayerInput.KeyDown(InputType.Voice) || PlayerInput.KeyDown(InputType.LocalVoice)) &&
233 GUI.KeyboardDispatcher.Subscriber == null)
234 {
235 pttDown = true;
236 if (PlayerInput.KeyDown(InputType.LocalVoice))
237 {
238 ForceLocal = true;
239 }
240 else
241 {
242 ForceLocal = false;
243 }
244 }
245 if (GameSettings.CurrentConfig.Audio.VoiceSetting == VoiceMode.Activity)
246 {
247 if (dB > GameSettings.CurrentConfig.Audio.NoiseGateThreshold)
248 {
249 allowEnqueue = true;
250 }
251 }
252 else if (GameSettings.CurrentConfig.Audio.VoiceSetting == VoiceMode.PushToTalk)
253 {
254 if (pttDown)
255 {
256 allowEnqueue = true;
257 }
258 }
259 }
260
261 if (allowEnqueue || captureTimer > 0)
262 {
263 LastEnqueueAudio = DateTime.Now;
264 if (GameMain.Client?.Character != null)
265 {
266 var messageType = !ForceLocal && ChatMessage.CanUseRadio(GameMain.Client.Character, out _) ? ChatMessageType.Radio : ChatMessageType.Default;
267 GameMain.Client.Character.ShowSpeechBubble(1.25f, ChatMessage.MessageColor[(int)messageType]);
268 }
269 //encode audio and enqueue it
270 lock (buffers)
271 {
272 if (!prevCaptured) //enqueue the previous buffer if not sent to avoid cutoff
273 {
274 int compressedCountPrev = encoder.Encode(prevUncompressedBuffer, 0, VoipConfig.BUFFER_SIZE, BufferToQueue, 0, VoipConfig.MAX_COMPRESSED_SIZE);
275 EnqueueBuffer(compressedCountPrev);
276 }
277 int compressedCount = encoder.Encode(uncompressedBuffer, 0, VoipConfig.BUFFER_SIZE, BufferToQueue, 0, VoipConfig.MAX_COMPRESSED_SIZE);
278 EnqueueBuffer(compressedCount);
279 }
280 captureTimer -= (VoipConfig.BUFFER_SIZE * 1000) / VoipConfig.FREQUENCY;
281 if (allowEnqueue)
282 {
283 captureTimer = GameSettings.CurrentConfig.Audio.VoiceChatCutoffPrevention;
284 }
285 prevCaptured = true;
286 }
287 else
288 {
289 captureTimer = 0;
290 prevCaptured = false;
291 //enqueue silence
292 lock (buffers)
293 {
294 EnqueueBuffer(0);
295 }
296 }
297 }
298 }

300 {
301 DebugConsole.ThrowError($"VoipCapture threw an exception. Disabling capture...", e);
302 capturing = false;

314 private void FillBuffer()
315 {
316 if (overrideSound != null)
317 {
318 int totalSampleCount = 0;
319 while (totalSampleCount < VoipConfig.BUFFER_SIZE)
320 {
321 int sampleCount = overrideSound.FillStreamBuffer(overridePos, overrideBuf);
322 overridePos += sampleCount * 2;
323 Array.Copy(overrideBuf, 0, uncompressedBuffer, totalSampleCount, sampleCount);
324 totalSampleCount += sampleCount;
325
326 if (sampleCount == 0)
327 {
328 overridePos = 0;
329 }
330 }
331 int sleepMs = VoipConfig.BUFFER_SIZE * 800 / VoipConfig.FREQUENCY;
332 Thread.Sleep(sleepMs - 1);
333 }
334 else
335 {
336 int sampleCount = 0;
337
338 while (sampleCount < VoipConfig.BUFFER_SIZE)
339 {
340 Alc.GetInteger(captureDevice, Alc.EnumCaptureSamples, out sampleCount);
341
342 int alcError = Alc.GetError(captureDevice);
343 if (alcError != Alc.NoError)
344 {
345 throw new Exception("Failed to determine sample count: " + alcError.ToString());
346 }
347
348 if (sampleCount < VoipConfig.BUFFER_SIZE)
349 {
350 int sleepMs = (VoipConfig.BUFFER_SIZE - sampleCount) * 800 / VoipConfig.FREQUENCY;
351 if (sleepMs >= 1)
352 {
353 Thread.Sleep(sleepMs);
354 }
355 }
356
357 if (!capturing) { return; }
358 }
359
360 Alc.CaptureSamples(captureDevice, nativeBuffer, VoipConfig.BUFFER_SIZE);
361 Marshal.Copy(nativeBuffer, uncompressedBuffer, 0, uncompressedBuffer.Length);
362 }
