124 public void ClientEventRead(IReadMessage msg, float sendingTime)
125 {
126 EventType eventType = (EventType)msg.ReadByte();
127
128 switch (eventType)
129 {
130 case EventType.GlobalDestructibleWall:
131 {
132 foreach (LevelWall levelWall in ExtraWalls)
133 {
134 if (levelWall.Body.BodyType == BodyType.Static) { continue; }
135
136 Vector2 bodyPos = new Vector2(
137 msg.ReadSingle(),
138 msg.ReadSingle());
139 levelWall.MoveState = msg.ReadRangedSingle(0.0f, MathHelper.TwoPi, 16);
140 if (Vector2.DistanceSquared(bodyPos, levelWall.Body.Position) > 0.5f
141 && !(levelWall is DestructibleLevelWall { Destroyed: true }))
142 {
143 levelWall.Body.SetTransformIgnoreContacts(ref bodyPos, levelWall.Body.Rotation);
144 }
145 }
146 }
147 break;
148 case EventType.SingleDestructibleWall:
149 {
150 int index = msg.ReadUInt16();
151 float damageByte = msg.ReadByte();
152 if (index < ExtraWalls.Count && ExtraWalls[index] is DestructibleLevelWall destructibleWall)
153 {
154 destructibleWall.SetDamage(destructibleWall.MaxHealth * damageByte / 255.0f);
155 }
156 }
157 break;
158 default:
159 throw new Exception($"Malformed incoming level event: {eventType} is not a supported event type");
160 }
161 }
