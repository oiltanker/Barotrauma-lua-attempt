797  public Item(Rectangle newRect, ItemPrefab itemPrefab, Submarine submarine, bool callOnItemLoaded = true, ushort id = Entity.NullEntityID)
798  : base(itemPrefab, submarine, id)
799  {
800  spriteColor = base.Prefab.SpriteColor;
801 
802  components = new List<ItemComponent>();
803  drawableComponents = new List<IDrawableComponent>(); hasComponentsToDraw = false;
804  tags = new HashSet<Identifier>();
805  repairables = new List<Repairable>();
806 
807  defaultRect = newRect;
808  rect = newRect;
809 
810  condition = MaxCondition;
811  lastSentCondition = condition;
812 
813  AllowDeconstruct = itemPrefab.AllowDeconstruct;
814 
815  allPropertyObjects.Add(this);
816 
817  ContentXElement element = itemPrefab.ConfigElement;
818  if (element == null) return;
819 
820  SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
821 
822  if (submarine == null || !submarine.Loading) { FindHull(); }
823 
824  SetActiveSprite();
825 
826  foreach (var subElement in element.Elements())
827  {
828  switch (subElement.Name.ToString().ToLowerInvariant())
829  {
830  case "body":
831  float density = subElement.GetAttributeFloat("density", 10.0f);
832  float minDensity = subElement.GetAttributeFloat("mindensity", density);
833  float maxDensity = subElement.GetAttributeFloat("maxdensity", density);
834  if (minDensity < maxDensity)
835  {
836  var rand = new Random(ID);
837  density = MathHelper.Lerp(minDensity, maxDensity, (float)rand.NextDouble());
838  }
839  body = new PhysicsBody(subElement, ConvertUnits.ToSimUnits(Position), Scale, density);
840 
841  string collisionCategory = subElement.GetAttributeString("collisioncategory", null);
842  if ((Prefab.DamagedByProjectiles || Prefab.DamagedByMeleeWeapons) && Condition > 0)
843  {
844  //force collision category to Character to allow projectiles and weapons to hit
845  //(we could also do this by making the projectiles and weapons hit CollisionItem
846  //and check if the collision should be ignored in the OnCollision callback, but
847  //that'd make the hit detection more expensive because every item would be included)
848  body.CollisionCategories = Physics.CollisionCharacter;
849  body.CollidesWith = Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionPlatform | Physics.CollisionProjectile;
850  }
851  if (collisionCategory != null)
852  {
853  if (!Physics.TryParseCollisionCategory(collisionCategory, out Category cat))
854  {
855  DebugConsole.ThrowError("Invalid collision category in item \"" + Name+"\" (" + collisionCategory + ")");
856  }
857  else
858  {
859  body.CollisionCategories = cat;
860  if (cat.HasFlag(Physics.CollisionCharacter))
861  {
862  body.CollidesWith = Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionPlatform | Physics.CollisionProjectile;
863  }
864  }
865  }
866  body.FarseerBody.AngularDamping = subElement.GetAttributeFloat("angulardamping", 0.2f);
867  body.FarseerBody.LinearDamping = subElement.GetAttributeFloat("lineardamping", 0.1f);
868  body.UserData = this;
869  break;
