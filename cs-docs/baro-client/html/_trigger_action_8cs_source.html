61 public override void Update(float deltaTime)
62 {
63 if (isFinished) { return; }
64
65 isRunning = true;
66
67 var targets1 = ParentEvent.GetTargets(Target1Tag);
68 if (!targets1.Any()) { return; }
69
70 foreach (Entity e1 in targets1)
71 {
72 if (DisableInCombat && IsInCombat(e1)) { continue; }
73 if (DisableIfTargetIncapacitated && e1 is Character character1 && (character1.IsDead || character1.IsIncapacitated)) { continue; }
74 if (!TargetModuleType.IsEmpty)
75 {
76 if (IsCloseEnoughToHull(e1, out Hull hull))
77 {
78 Trigger(e1, hull);
79 return;
80 }
81 continue;
82 }
83
84 var targets2 = ParentEvent.GetTargets(Target2Tag);
85
86 foreach (Entity e2 in targets2)
87 {
88 if (e1 == e2) { continue; }
89 if (DisableInCombat && IsInCombat(e2)) { continue; }
90 if (DisableIfTargetIncapacitated && e2 is Character character2 && (character2.IsDead || character2.IsIncapacitated)) { continue; }
91
92 if (WaitForInteraction)
93 {
94 Character player = null;
95 Character npc = null;
96 Item item = null;
97 if (e1 is Character char1)
98 {
99 if (char1.IsBot)
100 {
101 npc ??= char1;
102 }
103 else
104 {
105 player = char1;
106 }
107 }
108 else
109 {
110 item ??= e1 as Item;
111 }
112 if (e2 is Character char2)
113 {
114 if (char2.IsBot)
115 {
116 npc ??= char2;
117 }
118 else
119 {
120 player = char2;
121 }
122 }
123 else
124 {
125 item ??= e2 as Item;
126 }
127
128 if (player != null)
129 {
130 if (npc != null)
131 {
132 if (npc.CampaignInteractionType != CampaignMode.InteractionType.Examine)
133 {
134 if (!npcsOrItems.Any(n => n.TryGet(out Character npc2) && npc2 == npc))
135 {
136 npcsOrItems.Add(npc);
137 }
138 npc.CampaignInteractionType = CampaignMode.InteractionType.Examine;
139 npc.RequireConsciousnessForCustomInteract = DisableIfTargetIncapacitated;
140#if CLIENT
141 npc.SetCustomInteract(
142 (speaker, player) => { if (e1 == speaker) { Trigger(speaker, player); } else { Trigger(player, speaker); } },
143 TextManager.GetWithVariable("CampaignInteraction.Examine", "[key]", GameSettings.CurrentConfig.KeyMap.KeyBindText(InputType.Use)));
144#else
145 npc.SetCustomInteract(
146 (speaker, player) => { if (e1 == speaker) { Trigger(speaker, player); } else { Trigger(player, speaker); } },
147 TextManager.Get("CampaignInteraction.Talk"));
148 GameMain.NetworkMember.CreateEntityEvent(npc, new Character.AssignCampaignInteractionEventData());
149#endif
150 }
151 if (!AllowMultipleTargets) { return; }
152 }
153 else if (item != null)
154 {
155 if (!npcsOrItems.Any(n => n.TryGet(out Item item2) && item2 == item))
156 {
157 npcsOrItems.Add(item);
158 }
159 item.CampaignInteractionType = CampaignMode.InteractionType.Examine;
160 if (player.SelectedConstruction == item ||
161 player.Inventory != null && player.Inventory.Contains(item) ||
162 (player.FocusedItem == item && player.IsKeyHit(InputType.Use)))
163 {
164 Trigger(e1, e2);
165 return;
166 }
167 }
168 }
169 }
170 else
171 {
172 Vector2 pos1 = e1.WorldPosition;
173 Vector2 pos2 = e2.WorldPosition;
174 distance = Vector2.Distance(pos1, pos2);
175 if (((e1 is MapEntity m1) && Submarine.RectContains(m1.WorldRect, pos2)) ||
176 ((e2 is MapEntity m2) && Submarine.RectContains(m2.WorldRect, pos1)) ||
177 Vector2.DistanceSquared(pos1, pos2) < Radius * Radius)
178 {
179 Trigger(e1, e2);
180 return;
181 }
182 }
183 }
184 }

207 private bool IsCloseEnoughToHull(Entity e, out Hull hull)
208 {
209 hull = null;
210 if (Radius <= 0)
211 {
212 if (e is Character character && character.CurrentHull != null && character.CurrentHull.OutpostModuleTags.Contains(TargetModuleType))
213 {
214 hull = character.CurrentHull;
215 return true;
216 }
217 else if (e is Item item && item.CurrentHull != null && item.CurrentHull.OutpostModuleTags.Contains(TargetModuleType))
218 {
219 hull = item.CurrentHull;
220 return true;
221 }
222 return false;
223 }
224 else
225 {
226 foreach (Hull potentialHull in Hull.HullList)
227 {
228 if (!potentialHull.OutpostModuleTags.Contains(TargetModuleType)) { continue; }
229
230 Rectangle hullRect = potentialHull.WorldRect;
231 hullRect.Inflate(Radius, Radius);
232 if (Submarine.RectContains(hullRect, e.WorldPosition))
233 {
234 hull = potentialHull;
235 return true;
236 }
237 }
238 return false;
239 }

242 private bool IsInCombat(Entity entity)
243 {
244 if (!(entity is Character character)) { return false; }
245 foreach (Character c in Character.CharacterList)
246 {
247 if (c.IsDead || c.Removed || c.IsIncapacitated || !c.Enabled) { continue; }
248 if (c.IsBot && c.AIController is HumanAIController humanAi)
249 {
250 if (humanAi.ObjectiveManager.CurrentObjective is AIObjectiveCombat combatObjective &&
251 combatObjective.Enemy == character)
252 {
253 return true;
254 }
255 }
256 else if (c.AIController is EnemyAIController enemyAI && (enemyAI.State == AIState.Aggressive || enemyAI.State == AIState.Attack))
257 {
258 if (enemyAI.SelectedAiTarget?.Entity == character || c.CurrentHull == character.CurrentHull)
259 {
260 return true;
261 }
262 }
263 }
264 return false;
265 }
