43 public static XElement Save(List<MapEntity> entities, string name, string description, bool hideInMenus = false)
44 {
45 XElement element = new XElement("ItemAssembly",
46 new XAttribute("name", name),
47 new XAttribute("description", description),
48 new XAttribute("hideinmenus", hideInMenus));
49
50
51 //move the entities so that their "center of mass" is at {0,0}
52 var assemblyEntities = MapEntity.CopyEntities(entities);
53
54 //find wires and items that are contained inside another item
55 //place them at {0,0} to prevent them from messing up the origin of the prefab and to hide them in preview
56 List<MapEntity> disabledEntities = new List<MapEntity>();
57 foreach (MapEntity mapEntity in assemblyEntities)
58 {
59 if (mapEntity is Item item)
60 {
61 var wire = item.GetComponent<Wire>();
62 if (item.ParentInventory != null ||
63 (wire != null && wire.Connections.Any(c => c != null)))
64 {
65 item.SetTransform(Vector2.Zero, 0.0f);
66 disabledEntities.Add(mapEntity);
67 }
68 }
69 }
70
71 float minX = int.MaxValue, maxX = int.MinValue;
72 float minY = int.MaxValue, maxY = int.MinValue;
73 foreach (MapEntity mapEntity in assemblyEntities)
74 {
75 if (disabledEntities.Contains(mapEntity)) { continue; }
76 minX = Math.Min(minX, mapEntity.WorldRect.X);
77 maxX = Math.Max(maxX, mapEntity.WorldRect.Right);
78 minY = Math.Min(minY, mapEntity.WorldRect.Y - mapEntity.WorldRect.Height);
79 maxY = Math.Max(maxY, mapEntity.WorldRect.Y);
80 }
81 Vector2 center = new Vector2((minX + maxX) / 2.0f, (minY + maxY) / 2.0f);
82 if (Submarine.MainSub != null) { center -= Submarine.MainSub.HiddenSubPosition; }
83 center.X -= MathUtils.RoundTowardsClosest(center.X, Submarine.GridSize.X);
84 center.Y -= MathUtils.RoundTowardsClosest(center.Y, Submarine.GridSize.Y);
85
86 MapEntity.SelectedList.Clear();
87 assemblyEntities.ForEach(e => MapEntity.AddSelection(e));
88
89 foreach (MapEntity mapEntity in assemblyEntities)
90 {
91 mapEntity.Move(-center);
92 mapEntity.Submarine = Submarine.MainSub;
93 var entityElement = mapEntity.Save(element);
94 if (disabledEntities.Contains(mapEntity))
95 {
96 entityElement.Add(new XAttribute("hideinassemblypreview", "true"));
97 }
98 }
99
100 return element;
101 }
