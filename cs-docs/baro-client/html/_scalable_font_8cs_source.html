134  private void RenderAtlas(GraphicsDevice gd, uint[] charRanges = null, int texDims = 1024, uint baseChar = 0x54)
135  {
136  if (DynamicLoading) { return; }
137 
138  if (charRanges == null)
139  {
140  charRanges = new uint[] { 0x20, 0xFFFF };
141  }
142  this.charRanges = charRanges;
143  this.texDims = texDims;
144  this.baseChar = baseChar;
145 
146  textures.ForEach(t => t.Dispose());
147  textures.Clear();
148  texCoords.Clear();
149 
150  uint[] pixelBuffer = new uint[texDims * texDims];
151  for (int i = 0; i < texDims * texDims; i++)
152  {
153  pixelBuffer[i] = 0;
154  }
155 
156  CrossThread.RequestExecutionOnMainThread(() =>
157  {
158  textures.Add(new Texture2D(gd, texDims, texDims, false, SurfaceFormat.Color));
159  });
160  int texIndex = 0;
161 
162  Vector2 currentCoords = Vector2.Zero;
163  int nextY = 0;
164 
165  lock (mutex)
166  {
167  face.SetPixelSizes(0, size);
168  face.LoadGlyph(face.GetCharIndex(baseChar), LoadFlags.Default, LoadTarget.Normal);
169  baseHeight = face.Glyph.Metrics.Height.ToInt32();
170 
171  for (int i = 0; i < charRanges.Length; i += 2)
172  {
173  uint start = charRanges[i];
174  uint end = charRanges[i + 1];
175  for (uint j = start; j <= end; j++)
176  {
177  uint glyphIndex = face.GetCharIndex(j);
178  if (glyphIndex == 0) continue;
179  face.LoadGlyph(glyphIndex, LoadFlags.Default, LoadTarget.Normal);
180  if (face.Glyph.Metrics.Width == 0 || face.Glyph.Metrics.Height == 0)
181  {
182  if (face.Glyph.Metrics.HorizontalAdvance > 0)
183  {
184  //glyph is empty, but char still applies advance
185  GlyphData blankData = new GlyphData(
186  advance: (float)face.Glyph.Metrics.HorizontalAdvance,
187  texIndex: -1); //indicates no texture because the glyph is empty
188 
189  texCoords.Add(j, blankData);
190  }
191  continue;
192  }
193  //stacktrace doesn't really work that well when RenderGlyph throws an exception
194  face.Glyph.RenderGlyph(RenderMode.Normal);
195  byte[] bitmap = face.Glyph.Bitmap.BufferData;
196  int glyphWidth = face.Glyph.Bitmap.Width;
197  int glyphHeight = bitmap.Length / glyphWidth;
198 
199  //if (glyphHeight>lineHeight) lineHeight=glyphHeight;
200 
201  if (glyphWidth > texDims - 1 || glyphHeight > texDims - 1)
202  {
203  throw new Exception(filename + ", " + size.ToString() + ", " + (char)j + "; Glyph dimensions exceed texture atlas dimensions");
204  }
205 
206  nextY = Math.Max(nextY, glyphHeight + 2);
207 
208  if (currentCoords.X + glyphWidth + 2 > texDims - 1)
209  {
210  currentCoords.X = 0;
211  currentCoords.Y += nextY;
212  nextY = 0;
213  }
214  if (currentCoords.Y + glyphHeight + 2 > texDims - 1)
215  {
216  currentCoords.X = 0;
217  currentCoords.Y = 0;
218  CrossThread.RequestExecutionOnMainThread(() =>
219  {
220  textures[texIndex].SetData<uint>(pixelBuffer);
221  textures.Add(new Texture2D(gd, texDims, texDims, false, SurfaceFormat.Color));
222  });
223  texIndex++;
224  for (int k = 0; k < texDims * texDims; k++)
225  {
226  pixelBuffer[k] = 0;
227  }
228  }
229 
230  GlyphData newData = new GlyphData(
231  advance: (float)face.Glyph.Metrics.HorizontalAdvance,
232  texIndex: texIndex,
233  texCoords: new Rectangle((int)currentCoords.X, (int)currentCoords.Y, glyphWidth, glyphHeight),
234  drawOffset: new Vector2(face.Glyph.BitmapLeft, baseHeight * 14 / 10 - face.Glyph.BitmapTop)
235  );
236  texCoords.Add(j, newData);
237 
238  for (int y = 0; y < glyphHeight; y++)
239  {
240  for (int x = 0; x < glyphWidth; x++)
241  {
242  byte byteColor = bitmap[x + y * glyphWidth];
243  pixelBuffer[((int)currentCoords.X + x) + ((int)currentCoords.Y + y) * texDims] = (uint)(byteColor << 24 | 0x00ffffff);
244  }
245  }
246 
247  currentCoords.X += glyphWidth + 2;
248  }
249  CrossThread.RequestExecutionOnMainThread(() =>
250  {
251  textures[texIndex].SetData<uint>(pixelBuffer);
252  });
253  }
254  }

367  }
368

370  advance: (float)horizontalAdvance,
371  texIndex: textures.Count - 1,
372  texCoords: new Rectangle((int)currentDynamicAtlasCoords.X, (int)currentDynamicAtlasCoords.Y, glyphWidth, glyphHeight),
373  drawOffset: drawOffset
374  );
375  texCoords.Add(character, newData);
376 
377  if (currentDynamicPixelBuffer == null)
378  {
379  currentDynamicPixelBuffer = new uint[texDims * texDims];

381  }
382 
383  for (int y = 0; y < glyphHeight; y++)
384  {
385  for (int x = 0; x < glyphWidth; x++)
386  {

388  currentDynamicPixelBuffer[((int)currentDynamicAtlasCoords.X + x) + ((int)currentDynamicAtlasCoords.Y + y) * texDims] = (uint)(byteColor << 24 | 0x00ffffff);

402  private void HandleNewLineAndAlignment(
403  string text,
404  in Vector2 advanceUnit,
405  in Vector2 position,
406  in Vector2 scale,
407  Alignment alignment,
408  int i,
409  ref float lineWidth,
410  ref Vector2 currentLineOffset,
411  ref int lineNum,
412  ref Vector2 currentPos,
413  out uint charIndex,
414  out bool shouldContinue)
415  {
416  if ((alignment.HasFlag(Alignment.CenterX) || alignment.HasFlag(Alignment.Right)) && (lineWidth < 0.0f || text[i] == '\n'))
417  {
418  int startIndex = lineWidth < 0.0f ? i : (i + 1);
419  lineWidth = 0.0f;
420  for (int j = startIndex; j < text.Length; j++)
421  {
422  if (text[j] == '\n') { break; }
423  uint chrIndex = text[j];
424 
425  var gd2 = GetGlyphData(chrIndex);
426  lineWidth += gd2.Advance;
427  }
428  currentLineOffset = -lineWidth * advanceUnit * scale.X;
429  if (alignment.HasFlag(Alignment.CenterX)) { currentLineOffset *= 0.5f; }
430 
431  currentLineOffset.X = MathF.Round(currentLineOffset.X);
432  currentLineOffset.Y = MathF.Round(currentLineOffset.Y);
433  }
434  if (text[i] == '\n')
435  {
436  lineNum++;
437  currentPos = position;
438  currentPos.X -= LineHeight * lineNum * advanceUnit.Y * scale.Y;
439  currentPos.Y += LineHeight * lineNum * advanceUnit.X * scale.Y;
440  shouldContinue = true; charIndex = 0; return;
441  }
442 
443  shouldContinue = false;
444  charIndex = text[i];
445  }

511  {

514  if (DynamicLoading)
515  {
516  DynamicRenderAtlas(graphicsDevice, text);
517  }
518

520  for (int i = 0; i < text.Length; i++)
521  {
522  if (text[i] == '\n')
523  {
524  currentPos.X = position.X;
525  currentPos.Y += LineHeight;
526  continue;
527  }
528 
529  uint charIndex = text[i];
530 
531  GlyphData gd = GetGlyphData(charIndex);
532  if (gd.TexIndex >= 0)
533  {
534  float halfCharHeight = gd.TexCoords.Height * 0.5f;
535  float slantStrength = 0.35f;
536  float topItalicOffset = italics ? ((halfCharHeight - gd.DrawOffset.Y) * slantStrength) + baseHeight * 0.18f : 0.0f;
537  float bottomItalicOffset = italics ? ((-halfCharHeight - gd.DrawOffset.Y) * slantStrength) + baseHeight * 0.18f : 0.0f;
538 
539  Texture2D tex = textures[gd.TexIndex];
540  quadVertices[0].Position = new Vector3(currentPos + gd.DrawOffset + (bottomItalicOffset, gd.TexCoords.Height), 0.0f);
541  quadVertices[0].TextureCoordinate = ((float)gd.TexCoords.Left / tex.Width, (float)gd.TexCoords.Bottom / tex.Height);
542  quadVertices[0].Color = color;
543 
544  quadVertices[1].Position = new Vector3(currentPos + gd.DrawOffset + (topItalicOffset, 0.0f), 0.0f);
545  quadVertices[1].TextureCoordinate = ((float)gd.TexCoords.Left / tex.Width, (float)gd.TexCoords.Top / tex.Height);
546  quadVertices[1].Color = color;
547 
548  quadVertices[2].Position = new Vector3(currentPos + gd.DrawOffset + (gd.TexCoords.Width + bottomItalicOffset, gd.TexCoords.Height), 0.0f);
549  quadVertices[2].TextureCoordinate = ((float)gd.TexCoords.Right / tex.Width, (float)gd.TexCoords.Bottom / tex.Height);
550  quadVertices[2].Color = color;
551 
552  quadVertices[3].Position = new Vector3(currentPos + gd.DrawOffset + (gd.TexCoords.Width + topItalicOffset, 0.0f), 0.0f);
553  quadVertices[3].TextureCoordinate = ((float)gd.TexCoords.Right / tex.Width, (float)gd.TexCoords.Top / tex.Height);
554  quadVertices[3].Color = color;
555

557  }

644  private string WrapText(string text,
645  float width,
646  int requestCharPos,
647  out Vector2 requestedCharPos,
648  bool returnAllCharPositions,
649  out Vector2[] allCharPositions)
650  {
651  int currLineStart = 0;
652  Vector2 currentPos = Vector2.Zero;
653  Vector2 foundCharPos = Vector2.Zero;
654  int? lastBreakerIndex = null;
655  string result = "";
656  var allCharPos = returnAllCharPositions ? new Vector2[text.Length+1] : null;
657  for (int i = 0; i < text.Length; i++)
658  {
659  //Records the caret position of the current character
660  void recordCurrentPos()
661  {
662  if (i == requestCharPos) { foundCharPos = currentPos; }
663 
664  if (allCharPos != null) { allCharPos[i] = currentPos; }
665  }
666  recordCurrentPos();
667 
668  //Appends a newline to the result and resets the caret position's X value
669  void nextLine()
670  {
671  result += text[currLineStart..i].Remove("\n") + "\n";
672  lastBreakerIndex = null;
673  currentPos.X = 0.0f;
674  currentPos.Y += LineHeight;
675  currLineStart = i;
676  }
677 
678  //If a newline is found in the source, split immediately
679  if (text[i] == '\n')
680  {
681  nextLine();
682  continue;
683  }
684 
685  //Otherwise, advance based on the width of the current character
686  GlyphData gd = GetGlyphData(text[i]);
687  float advance = gd.Advance;
688  if (currentPos.X + advance >= width)
689  {
690  //Advancing based on the last character
691  //would put us past the max width!
692  if (i > 0 && char.IsWhiteSpace(text[i]) && !char.IsWhiteSpace(text[i - 1]))
693  {
694  //Whitespace immediately after a visible
695  //character can be shrunk down to fit
696  advance = width - currentPos.X;
697  }
698  else
699  {
700  if (lastBreakerIndex.HasValue)
701  {
702  //A breaker (whitespace or CJK) was found earlier
703  //in this line, so let's break the line there
704  i = lastBreakerIndex.Value + 1;
705  gd = GetGlyphData(text[i]);
706  advance = gd.Advance;
707  }
708 
709  nextLine();
710  recordCurrentPos(); //must re-record current caret position since we are on a new line now
711  }
712  }
713  currentPos.X += advance;
714 
715  if (char.IsWhiteSpace(text[i]) || TextManager.IsCJK($"{text[i]}"))
716  {
717  lastBreakerIndex = i;
718  }
719  }
720  if (requestCharPos >= text.Length) { foundCharPos = currentPos; }
721  if (allCharPos != null) { allCharPos[text.Length] = currentPos; }
722  allCharPositions = allCharPos;
723  result += text[currLineStart..].Remove("\n");
724  requestedCharPos = foundCharPos;
725  return result;
726  }
