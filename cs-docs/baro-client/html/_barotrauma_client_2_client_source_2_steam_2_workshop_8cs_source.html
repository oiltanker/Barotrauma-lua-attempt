76 {
77 ItemId = item.Id;
78 lock (TextureRefs)
79 {
80 if (TextureRefs.TryGetValue(ItemId, out var refCounter))
81 {
82 TextureRefs[ItemId] = new RefCounter { Texture = refCounter.Texture, Count = refCounter.Count + 1, Loading = refCounter.Loading };

97 public void Dispose()
98 {
99 if (ItemId == 0) { return; }
100 lock (TextureRefs)
101 {
102 var refCounter = TextureRefs[ItemId];
103 TextureRefs[ItemId] = new RefCounter { Texture = refCounter.Texture, Count = refCounter.Count - 1 };
104 if (TextureRefs[ItemId].Count <= 0)
105 {
106 TextureRefs[ItemId].Texture?.Dispose();
107 TextureRefs.Remove(ItemId);
108 }

113 private static async Task<Texture2D?> GetTexture(Steamworks.Ugc.Item item, CancellationToken cancellationToken)
114 {
115 await Task.Yield();
116
117 string thumbnailUrl = item.PreviewImageUrl;
118 if (thumbnailUrl.IsNullOrWhiteSpace()) { return null; }
119 var client = new RestClient(thumbnailUrl);
120 var request = new RestRequest(".", Method.GET);
121 IRestResponse response = await client.ExecuteAsync(request, cancellationToken);
122 if (response is { StatusCode: System.Net.HttpStatusCode.OK, ResponseStatus: ResponseStatus.Completed })
123 {
124 using var dataStream = new System.IO.MemoryStream();
125 await dataStream.WriteAsync(response.RawBytes, cancellationToken);
126 dataStream.Seek(0, System.IO.SeekOrigin.Begin);
127 return TextureLoader.FromStream(dataStream, compress: false);
128 }
129 return null;
130 }

190 public static async Task<ContentPackage?> CreateLocalCopy(ContentPackage contentPackage)
191 {
192 await Task.Yield();
193
194 if (!ContentPackageManager.WorkshopPackages.Contains(contentPackage))
195 {
196 throw new Exception("Expected Workshop package");
197 }
198
199 if (contentPackage.SteamWorkshopId == 0)
200 {
201 throw new Exception($"Steam Workshop ID not set for {contentPackage.Name}");
202 }
203
204 string sanitizedName = ToolBox.RemoveInvalidFileNameChars(contentPackage.Name).Trim();
205 if (sanitizedName.IsNullOrWhiteSpace())
206 {
207 throw new Exception($"Sanitized name for {contentPackage.Name} is empty");
208 }
209
210 string newPath = $"{ContentPackage.LocalModsDir}/{sanitizedName}";
211 if (File.Exists(newPath) || Directory.Exists(newPath))
212 {
213 newPath += $"_{contentPackage.SteamWorkshopId}";
214 }
215
216 if (File.Exists(newPath) || Directory.Exists(newPath))
217 {
218 throw new Exception($"{newPath} already exists");
219 }
220
221 await CopyDirectory(contentPackage.Dir, contentPackage.Name, Path.GetDirectoryName(contentPackage.Path)!, newPath);
222
223 ModProject modProject = new ModProject(contentPackage);
224 modProject.DiscardHashAndInstallTime();
225 modProject.Save(Path.Combine(newPath, ContentPackage.FileListFileName));
226
227 RefreshLocalMods();
228
229 return ContentPackageManager.LocalPackages.FirstOrDefault(p => p.SteamWorkshopId == contentPackage.SteamWorkshopId);
230 }
