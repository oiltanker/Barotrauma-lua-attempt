147  public bool Combine(GeneticMaterial otherGeneticMaterial, Character user)
148  {
149  if (!CanBeCombinedWith(otherGeneticMaterial)) { return false; }
150 
151  float conditionIncrease = Rand.Range(ConditionIncreaseOnCombineMin, ConditionIncreaseOnCombineMax);
152  conditionIncrease += user?.GetStatValue(StatTypes.GeneticMaterialRefineBonus) ?? 0.0f;
153  if (item.Prefab == otherGeneticMaterial.item.Prefab)
154  {
155  item.Condition = Math.Max(item.Condition, otherGeneticMaterial.item.Condition) + conditionIncrease;
156  float taintedProbability = GetTaintedProbabilityOnRefine(user);
157  if (taintedProbability >= Rand.Range(0.0f, 1.0f))
158  {
159  MakeTainted();
160  }
161  return true;
162  }
163  else
164  {
165  item.Condition = otherGeneticMaterial.Item.Condition =
166  (item.Condition + otherGeneticMaterial.Item.Condition) / 2.0f + conditionIncrease;
167  item.OwnInventory?.TryPutItem(otherGeneticMaterial.Item, user: null);
168  item.AllowDeconstruct = false;
169  otherGeneticMaterial.Item.AllowDeconstruct = false;
170  if (GetTaintedProbabilityOnCombine(user) >= Rand.Range(0.0f, 1.0f))
171  {
172  MakeTainted();
173  }
174  return false;
175  }
176  }

178  private float GetCombinedEffectStrength()
179  {
180  float effectStrength = 0.0f;
181  foreach (Item otherItem in targetCharacter.Inventory.FindAllItems(recursive: true))
182  {
183  var geneticMaterial = otherItem.GetComponent<GeneticMaterial>();
184  if (geneticMaterial == null || !geneticMaterial.IsActive) { continue; }
185  if (geneticMaterial.selectedEffect == selectedEffect)
186  {
187  effectStrength += otherItem.ConditionPercentage / 100.0f * selectedEffect.MaxStrength;
188  }
189  }
190  return effectStrength;
191  }

208  private float GetTaintedProbabilityOnRefine(Character user)
209  {
210  if (user == null) { return 1.0f; }
211  float probability = MathHelper.Lerp(0.0f, 0.99f, item.Condition / 100.0f);
212  probability *= MathHelper.Lerp(1.0f, 0.25f, DegreeOfSuccess(user));
213  return MathHelper.Clamp(probability, 0.0f, 1.0f);
214  }

216  private float GetTaintedProbabilityOnCombine(Character user)

220  return MathHelper.Clamp(probability, 0.0f, 1.0f);
221  }
