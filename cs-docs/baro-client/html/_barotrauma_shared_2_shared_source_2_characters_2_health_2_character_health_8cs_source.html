32 public LimbHealth(ContentXElement element, CharacterHealth characterHealth)
33 {
34 string limbName = element.GetAttributeString("name", null) ?? "generic";
35 if (limbName != "generic")
36 {
37 Name = TextManager.Get("HealthLimbName." + limbName);
38 }
39 foreach (var subElement in element.Elements())
40 {
41 switch (subElement.Name.ToString().ToLowerInvariant())
42 {
43 case "sprite":
44 IndicatorSprite = new Sprite(subElement);
45 HighlightArea = subElement.GetAttributeRect("highlightarea", new Rectangle(0, 0, (int)IndicatorSprite.size.X, (int)IndicatorSprite.size.Y));
46 break;
47 case "highlightsprite":
48 HighlightSprite = new Sprite(subElement);
49 break;
50 case "vitalitymultiplier":
51 if (subElement.GetAttribute("name") != null)
52 {
53 DebugConsole.ThrowError("Error in character health config (" + characterHealth.Character.Name + ") - define vitality multipliers using affliction identifiers or types instead of names.");
54 continue;
55 }
56 var vitalityMultipliers = subElement.GetAttributeIdentifierArray("identifier", null) ?? subElement.GetAttributeIdentifierArray("identifiers", null);
57 if (vitalityMultipliers != null)
58 {
59 float multiplier = subElement.GetAttributeFloat("multiplier", 1.0f);
60 vitalityMultipliers.ForEach(i => VitalityMultipliers.Add(i, multiplier));
61 }
62 var vitalityTypeMultipliers = subElement.GetAttributeIdentifierArray("type", null) ?? subElement.GetAttributeIdentifierArray("types", null);
63 if (vitalityTypeMultipliers != null)
64 {
65 float multiplier = subElement.GetAttributeFloat("multiplier", 1.0f);
66 vitalityTypeMultipliers.ForEach(i => VitalityTypeMultipliers.Add(i, multiplier));
67 }
68 if (vitalityMultipliers == null && VitalityTypeMultipliers == null)
69 {
70 DebugConsole.ThrowError($"Error in character health config {characterHealth.Character.Name}: affliction identifier(s) or type(s) not defined in the \"VitalityMultiplier\" elements!");
71 }
72 break;
73 }

495 private void ReduceMatchingAfflictions(float amount, ActionType? treatmentAction)
496 {
497 if (matchingAfflictions.Count == 0) { return; }
498
499 float reduceAmount = amount / matchingAfflictions.Count;
500 for (int i = matchingAfflictions.Count - 1; i >= 0; i--)
501 {
502 var matchingAffliction = matchingAfflictions[i];
503
504 if (matchingAffliction.Strength < reduceAmount)
505 {
506 float surplus = reduceAmount - matchingAffliction.Strength;
507 amount -= matchingAffliction.Strength;
508 matchingAffliction.Strength = 0.0f;
509 matchingAfflictions.RemoveAt(i);
510 if (i == 0) { i = matchingAfflictions.Count; }
511 if (i > 0) { reduceAmount += surplus / i; }
512 SteamAchievementManager.OnAfflictionRemoved(matchingAffliction, Character);
513 }
514 else
515 {
516 matchingAffliction.Strength -= reduceAmount;
517 amount -= reduceAmount;
518 if (treatmentAction != null)
519 {
520 if (treatmentAction.Value == ActionType.OnUse)
521 {
522 matchingAffliction.AppliedAsSuccessfulTreatmentTime = Timing.TotalTime;
523 }
524 else if (treatmentAction.Value == ActionType.OnFailure)
525 {
526 matchingAffliction.AppliedAsFailedTreatmentTime = Timing.TotalTime;
527 }
528 }
529 }

589 public float GetLimbDamage(Limb limb, string afflictionType = null)
590 {
591 float damageStrength;
592 if (limb.IsSevered)
593 {
594 return 1;
595 }
596 else
597 {
598 // Instead of using the limbhealth count here, I think it's best to define the max vitality per limb roughly with a constant value.
599 // Therefore with e.g. 80 health, the max damage per limb would be 40.
600 // Having at least 40 damage on both legs would cause maximum limping.
601 float max = MaxVitality / 2;
602 if (string.IsNullOrEmpty(afflictionType))
603 {
604 float damage = GetAfflictionStrength("damage", limb, true);
605 float bleeding = GetAfflictionStrength("bleeding", limb, true);
606 float burn = GetAfflictionStrength("burn", limb, true);
607 damageStrength = Math.Min(damage + bleeding + burn, max);
608 }
609 else
610 {
611 damageStrength = Math.Min(GetAfflictionStrength("damage", limb, true), max);
612 }
613 return damageStrength / max;
614 }

663 private void AddLimbAffliction(LimbHealth limbHealth, Affliction newAffliction, bool allowStacking = true)
664 {
665 if (!DoesBleed && newAffliction is AfflictionBleeding) { return; }
666 if (!Character.NeedsOxygen && newAffliction.Prefab == AfflictionPrefab.OxygenLow) { return; }
667 if (Character.Params.Health.StunImmunity && newAffliction.Prefab.AfflictionType == "stun") { return; }
668 if (Character.Params.Health.PoisonImmunity && newAffliction.Prefab.AfflictionType == "poison") { return; }
669 if (newAffliction.Prefab is AfflictionPrefabHusk huskPrefab)
670 {
671 if (huskPrefab.TargetSpecies.None(s => s == Character.SpeciesName))
672 {
673 return;
674 }
675 }
676
677 var should = GameMain.LuaCs.Hook.Call<bool?>("character.applyAffliction", this, limbHealth, newAffliction, allowStacking);
678
679 if (should != null && should.Value)
680 return;
681
682 Affliction existingAffliction = null;
683 foreach (KeyValuePair<Affliction, LimbHealth> kvp in afflictions)
684 {
685 var affliction = kvp.Key;
686 if (kvp.Value == limbHealth && kvp.Key.Prefab == newAffliction.Prefab)
687 {
688 existingAffliction = kvp.Key;
689 break;
690 }
691 }
692
693 if (existingAffliction != null)
694 {
695 float newStrength = newAffliction.Strength * (100.0f / MaxVitality) * (1f - GetResistance(existingAffliction.Prefab));
696 if (allowStacking)
697 {
698 // Add the existing strength
699 newStrength += existingAffliction.Strength;
700 }
701 newStrength = Math.Min(existingAffliction.Prefab.MaxStrength, newStrength);
702 if (existingAffliction == stunAffliction) { Character.SetStun(newStrength, true, true); }
703 existingAffliction.Strength = newStrength;
704 if (newAffliction.Source != null) { existingAffliction.Source = newAffliction.Source; }
705 CalculateVitality();
706 if (Vitality <= MinVitality)
707 {
708 Kill();
709 }
710 return;
711 }

1003 public void GetSuitableTreatments(Dictionary<Identifier, float> treatmentSuitability, bool normalize, Limb limb = null, bool ignoreHiddenAfflictions = false, float predictFutureDuration = 0.0f)
1004 {
1005 //key = item identifier
1006 //float = suitability
1007 treatmentSuitability.Clear();
1008 float minSuitability = -10, maxSuitability = 10;
1009 foreach (KeyValuePair<Affliction, LimbHealth> kvp in afflictions)
1010 {
1011 var affliction = kvp.Key;
1012 var limbHealth = kvp.Value;
1013 if (limb != null && affliction.Prefab.IndicatorLimb != limb.type)

1019
1020 float strength = affliction.Strength;
1021 if (predictFutureDuration > 0.0f)
1022 {
1023 strength = GetPredictedStrength(affliction, predictFutureDuration, limb);

1025
1026 if (strength <= affliction.Prefab.TreatmentThreshold) { continue; }
1027 if (ignoreHiddenAfflictions && strength < affliction.Prefab.ShowIconThreshold) { continue; }

1030 {
1031 if (!treatmentSuitability.ContainsKey(treatment.Key))
1032 {
1033 treatmentSuitability[treatment.Key] = treatment.Value * strength;
1034 }
1035 else
1036 {
1037 treatmentSuitability[treatment.Key] += treatment.Value * strength;
1038 }
1039 minSuitability = Math.Min(treatmentSuitability[treatment.Key], minSuitability);
1040 maxSuitability = Math.Max(treatmentSuitability[treatment.Key], maxSuitability);
1041 }
1042 }
1043 //normalize the suitabilities to a range of 0 to 1
1044 if (normalize)
1045 {
1046 foreach (Identifier treatment in treatmentSuitability.Keys.ToList())
1047 {
1048 treatmentSuitability[treatment] = (treatmentSuitability[treatment] - minSuitability) / (maxSuitability - minSuitability);
1049 }
