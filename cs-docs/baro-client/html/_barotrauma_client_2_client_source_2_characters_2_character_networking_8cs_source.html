115 public void ClientWriteInput(IWriteMessage msg)
116 {
117 msg.Write((byte)ClientNetObject.CHARACTER_INPUT);
118
119 if (memInput.Count > 60)
120 {
121 memInput.RemoveRange(60, memInput.Count - 60);
122 }
123
124 msg.Write(LastNetworkUpdateID);
125 byte inputCount = Math.Min((byte)memInput.Count, (byte)60);
126 msg.Write(inputCount);
127 for (int i = 0; i < inputCount; i++)
128 {
129 msg.WriteRangedInteger((int)memInput[i].states, 0, (int)InputNetFlags.MaxVal);
130 msg.Write(memInput[i].intAim);
131 if (memInput[i].states.HasFlag(InputNetFlags.Select) ||

133 memInput[i].states.HasFlag(InputNetFlags.Use) ||

135 memInput[i].states.HasFlag(InputNetFlags.Grab))
136 {
137 msg.Write(memInput[i].interact);
138 }

142 public virtual void ClientEventWrite(IWriteMessage msg, NetEntityEvent.IData extraData = null)
143 {
144 if (!(extraData is IEventData eventData)) { throw new Exception($"Malformed character event: expected {nameof(Character)}.{nameof(IEventData)}"); }
145
146 msg.WriteRangedInteger((int)eventData.EventType, (int)EventType.MinValue, (int)EventType.MaxValue);
147 switch (eventData)
148 {
149 case InventoryStateEventData inventoryStateEventData:
150 Inventory.ClientEventWrite(msg, inventoryStateEventData);
151 break;
152 case TreatmentEventData _:
153 msg.Write(AnimController.Anim == AnimController.Animation.CPR);
154 break;
155 case CharacterStatusEventData _:
156 //do nothing
157 break;
158 case UpdateTalentsEventData _:
159 msg.Write((ushort)characterTalents.Count);
160 foreach (var unlockedTalent in characterTalents)
161 {
162 msg.Write(unlockedTalent.Prefab.UintIdentifier);
163 }
164 break;
165 default:
166 throw new Exception($"Malformed character event: did not expect {eventData.GetType().Name}");
167 }

297 public virtual void ClientEventRead(IReadMessage msg, float sendingTime)
298 {
299 EventType eventType = (EventType)msg.ReadRangedInteger((int)EventType.MinValue, (int)EventType.MaxValue);
300 switch (eventType)
301 {
302 case EventType.InventoryState:
303 if (Inventory == null)
304 {
305 string errorMsg = "Received an inventory update message for an entity with no inventory ([name], removed: " + Removed + ")";
306 DebugConsole.ThrowError(errorMsg.Replace("[name]", Name));
307 GameAnalyticsManager.AddErrorEventOnce("CharacterNetworking.ClientRead:NoInventory" + ID, GameAnalyticsManager.ErrorSeverity.Error, errorMsg.Replace("[name]", SpeciesName.Value));
308
309 //read anyway to prevent messing up reading the rest of the message
310 _ = msg.ReadUInt16();
311 byte inventoryItemCount = msg.ReadByte();
312 for (int i = 0; i < inventoryItemCount; i++)
313 {
314 msg.ReadUInt16();
315 }
316 }
317 else
318 {
319 Inventory.ClientEventRead(msg, sendingTime);
320 }
321 break;
322 case EventType.Control:
323 bool myCharacter = msg.ReadBoolean();
324 byte ownerID = msg.ReadByte();
325 ResetNetState();
326 if (myCharacter)
327 {
328 if (controlled != null)
329 {
330 LastNetworkUpdateID = controlled.LastNetworkUpdateID;
331 }
332
333 if (!IsDead) { Controlled = this; }
334 IsRemotePlayer = false;
335 GameMain.Client.HasSpawned = true;
336 GameMain.Client.Character = this;
337 GameMain.LightManager.LosEnabled = true;
338 GameMain.LightManager.LosAlpha = 1f;
339 GameMain.Client.WaitForNextRoundRespawn = null;
340 }
341 else
342 {
343 if (controlled == this)
344 {
345 Controlled = null;
346 }
347 if (GameMain.Client?.Character == this)
348 {
349 GameMain.Client.Character = null;
350 }
351 IsRemotePlayer = ownerID > 0;
352 }
353 break;
354 case EventType.Status:
355 ReadStatus(msg);
356 break;
357 case EventType.UpdateSkills:
358 int skillCount = msg.ReadByte();
359 for (int i = 0; i < skillCount; i++)
360 {
361 Identifier skillIdentifier = msg.ReadIdentifier();
362 float skillLevel = msg.ReadSingle();
363 info?.SetSkillLevel(skillIdentifier, skillLevel);
364 }
365 break;
366 case EventType.SetAttackTarget:
367 case EventType.ExecuteAttack:
368 int attackLimbIndex = msg.ReadByte();
369 UInt16 targetEntityID = msg.ReadUInt16();
370 int targetLimbIndex = msg.ReadByte();
371 float targetX = msg.ReadSingle();
372 float targetY = msg.ReadSingle();
373 Vector2 targetSimPos = new Vector2(targetX, targetY);
374 //255 = entity already removed, no need to do anything
375 if (attackLimbIndex == 255 || Removed) { break; }
376 if (attackLimbIndex >= AnimController.Limbs.Length)
377 {
378 string errorMsg = $"Received invalid {(eventType == EventType.SetAttackTarget ? "SetAttackTarget" : "ExecuteAttack")} message. Limb index out of bounds (character: {Name}, limb index: {attackLimbIndex}, limb count: {AnimController.Limbs.Length})";
379 DebugConsole.ThrowError(errorMsg);
380 GameAnalyticsManager.AddErrorEventOnce("Character.ClientEventRead:AttackLimbOutOfBounds", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
381 break;
382 }
383 Limb attackLimb = AnimController.Limbs[attackLimbIndex];
384 Limb targetLimb = null;
385 IDamageable targetEntity = FindEntityByID(targetEntityID) as IDamageable;
386 if (targetEntity == null && eventType == EventType.SetAttackTarget)
387 {
388 DebugConsole.ThrowError($"Received invalid SetAttackTarget message. Target entity not found (ID {targetEntityID})");
389 GameAnalyticsManager.AddErrorEventOnce("Character.ClientEventRead:TargetNotFound", GameAnalyticsManager.ErrorSeverity.Error, "Received invalid SetAttackTarget message. Target entity not found.");
390 break;
391 }
392 if (targetEntity is Character targetCharacter && targetLimbIndex != 255)
393 {
394 if (targetLimbIndex >= targetCharacter.AnimController.Limbs.Length)
395 {
396 DebugConsole.ThrowError($"Received invalid {(eventType == EventType.SetAttackTarget ? "SetAttackTarget" : "ExecuteAttack")} message. Target limb index out of bounds (target character: {targetCharacter.Name}, limb index: {targetLimbIndex}, limb count: {targetCharacter.AnimController.Limbs.Length})");
397 string errorMsgWithoutName = $"Received invalid {(eventType == EventType.SetAttackTarget ? "SetAttackTarget" : "ExecuteAttack")} message. Target limb index out of bounds (target character: {targetCharacter.SpeciesName}, limb index: {targetLimbIndex}, limb count: {targetCharacter.AnimController.Limbs.Length})";
398 GameAnalyticsManager.AddErrorEventOnce("Character.ClientEventRead:TargetLimbOutOfBounds", GameAnalyticsManager.ErrorSeverity.Error, errorMsgWithoutName);
399 break;
400 }
401 targetLimb = targetCharacter.AnimController.Limbs[targetLimbIndex];
402 }
403 if (attackLimb?.attack != null && Controlled != this)
404 {
405 if (eventType == EventType.SetAttackTarget)
406 {
407 SetAttackTarget(attackLimb, targetEntity, targetSimPos);
408 PlaySound(CharacterSound.SoundType.Attack, maxInterval: 3);
409 }
410 else
411 {
412 attackLimb.ExecuteAttack(targetEntity, targetLimb, out _);
413 }
414 }
415 break;
416 case EventType.AssignCampaignInteraction:
417 byte campaignInteractionType = msg.ReadByte();
418 bool requireConsciousness = msg.ReadBoolean();
419 (GameMain.GameSession?.GameMode as CampaignMode)?.AssignNPCMenuInteraction(this, (CampaignMode.InteractionType)campaignInteractionType);
420 RequireConsciousnessForCustomInteract = requireConsciousness;
421 break;
422 case EventType.ObjectiveManagerState:
423 // 1 = order, 2 = objective
424 AIObjectiveManager.ObjectiveType msgType
425 = (AIObjectiveManager.ObjectiveType)msg.ReadRangedInteger(
426 (int)AIObjectiveManager.ObjectiveType.MinValue,
427 (int)AIObjectiveManager.ObjectiveType.MaxValue);
428 if (msgType == 0) { break; }
429 bool validData = msg.ReadBoolean();
430 if (!validData) { break; }
431 if (msgType == AIObjectiveManager.ObjectiveType.Order)
432 {
433 UInt32 orderPrefabUintIdentifier = msg.ReadUInt32();
434 var orderPrefab = OrderPrefab.Prefabs.Find(p => p.UintIdentifier == orderPrefabUintIdentifier);
435 Identifier option = Identifier.Empty;
436 if (orderPrefab.HasOptions)
437 {
438 int optionIndex = msg.ReadRangedInteger(-1, orderPrefab.AllOptions.Length);
439 if (optionIndex > -1)
440 {
441 option = orderPrefab.AllOptions[optionIndex];
442 }
443 }
444 GameMain.GameSession?.CrewManager?.SetOrderHighlight(this, orderPrefab.Identifier, option);
445 }
446 else if (msgType == AIObjectiveManager.ObjectiveType.Objective)
447 {
448 Identifier identifier = msg.ReadIdentifier();
449 Identifier option = msg.ReadIdentifier();
450 ushort objectiveTargetEntityId = msg.ReadUInt16();
451 var objectiveTargetEntity = FindEntityByID(objectiveTargetEntityId);
452 GameMain.GameSession?.CrewManager?.CreateObjectiveIcon(this, identifier, option, objectiveTargetEntity);
453 }
454 break;
455 case EventType.TeamChange:
456 byte newTeamId = msg.ReadByte();
457 ChangeTeam((CharacterTeamType)newTeamId);
458 break;
459 case EventType.AddToCrew:
460 GameMain.GameSession.CrewManager.AddCharacter(this);
461 CharacterTeamType teamID = (CharacterTeamType)msg.ReadByte();
462 ushort itemCount = msg.ReadUInt16();
463 for (int i = 0; i < itemCount; i++)
464 {
465 ushort itemID = msg.ReadUInt16();
466 if (!(Entity.FindEntityByID(itemID) is Item item)) { continue; }
467 item.AllowStealing = true;
468 var wifiComponent = item.GetComponent<WifiComponent>();
469 if (wifiComponent != null)
470 {
471 wifiComponent.TeamID = teamID;
472 }
473 var idCard = item.GetComponent<IdCard>();
474 if (idCard != null)
475 {
476 idCard.TeamID = teamID;
477 idCard.SubmarineSpecificID = 0;
478 }
479 }
480 break;
481 case EventType.UpdateExperience:
482 int experienceAmount = msg.ReadInt32();
483 info?.SetExperience(experienceAmount);
484 break;
485 case EventType.UpdateTalents:
486 ushort talentCount = msg.ReadUInt16();
487 for (int i = 0; i < talentCount; i++)
488 {
489 bool addedThisRound = msg.ReadBoolean();
490 UInt32 talentIdentifier = msg.ReadUInt32();
491 GiveTalent(talentIdentifier, addedThisRound);
492 }
493 break;
494 case EventType.UpdateMoney:
495 int moneyAmount = msg.ReadInt32();
496 SetMoney(moneyAmount);
497 break;
498 case EventType.UpdatePermanentStats:
499 byte savedStatValueCount = msg.ReadByte();
500 StatTypes statType = (StatTypes)msg.ReadByte();
501 info?.ClearSavedStatValues(statType);
502 for (int i = 0; i < savedStatValueCount; i++)
503 {
504 string statIdentifier = msg.ReadString();
505 float statValue = msg.ReadSingle();
506 bool removeOnDeath = msg.ReadBoolean();
507 info?.ChangeSavedStatValue(statType, statValue, statIdentifier, removeOnDeath, setValue: true);
508 }
509 break;
510
511 }
512 msg.ReadPadBits();
513 }

515 public static Character ReadSpawnData(IReadMessage inc)
516 {
517 DebugConsole.Log("Reading character spawn data");
518
519 if (GameMain.Client == null) { return null; }
520
521 bool noInfo = inc.ReadBoolean();
522 ushort id = inc.ReadUInt16();
523 string speciesName = inc.ReadString();
524 string seed = inc.ReadString();
525
526 Vector2 position = new Vector2(inc.ReadSingle(), inc.ReadSingle());
527
528 bool enabled = inc.ReadBoolean();
529
530 DebugConsole.Log("Received spawn data for " + speciesName);
531
532 Character character = null;
533 if (noInfo)
534 {
535 try
536 {
537 character = Create(speciesName, position, seed, characterInfo: null, id: id, isRemotePlayer: false);
538 }
539 catch (Exception e)
540 {
541 DebugConsole.ThrowError($"Failed to spawn character {speciesName}", e);
542 throw;
543 }
544 bool containsStatusData = inc.ReadBoolean();
545 if (containsStatusData)
546 {
547 character.ReadStatus(inc);
548 }
549 }
550 else
551 {
552 bool hasOwner = inc.ReadBoolean();
553 int ownerId = hasOwner ? inc.ReadByte() : -1;
554 int balance = inc.ReadInt32();
555 int rewardDistribution = inc.ReadRangedInteger(0, 100);
556 byte teamID = inc.ReadByte();
557 bool hasAi = inc.ReadBoolean();
558 Identifier infoSpeciesName = inc.ReadIdentifier();
559
560 CharacterInfo info = CharacterInfo.ClientRead(infoSpeciesName, inc);
561 try
562 {
563 character = Create(speciesName, position, seed, characterInfo: info, id: id, isRemotePlayer: ownerId > 0 && GameMain.Client.ID != ownerId, hasAi: hasAi);
564 }
565 catch (Exception e)
566 {
567 DebugConsole.ThrowError($"Failed to spawn character {speciesName}", e);
568 throw;
569 }
570 character.TeamID = (CharacterTeamType)teamID;
571 character.CampaignInteractionType = (CampaignMode.InteractionType)inc.ReadByte();
572 if (character.CampaignInteractionType == CampaignMode.InteractionType.Store)
573 {
574 character.MerchantIdentifier = inc.ReadIdentifier();
575 }
576 character.Wallet.Balance = balance;
577 character.Wallet.RewardDistribution = rewardDistribution;
578 if (character.CampaignInteractionType != CampaignMode.InteractionType.None)
579 {
580 (GameMain.GameSession.GameMode as CampaignMode)?.AssignNPCMenuInteraction(character, character.CampaignInteractionType);
581 }
582
583 // Check if the character has current orders
584 int orderCount = inc.ReadByte();
585 for (int i = 0; i < orderCount; i++)
586 {
587 UInt32 orderPrefabUintIdentifier = inc.ReadUInt32();
588 Entity targetEntity = FindEntityByID(inc.ReadUInt16());
589 Character orderGiver = inc.ReadBoolean() ? FindEntityByID(inc.ReadUInt16()) as Character : null;
590 int orderOptionIndex = inc.ReadByte();
591 int orderPriority = inc.ReadByte();
592 OrderTarget targetPosition = null;
593 if (inc.ReadBoolean())
594 {
595 var x = inc.ReadSingle();
596 var y = inc.ReadSingle();
597 var hull = FindEntityByID(inc.ReadUInt16()) as Hull;
598 targetPosition = new OrderTarget(new Vector2(x, y), hull, creatingFromExistingData: true);
599 }
600
601 OrderPrefab orderPrefab =
602 OrderPrefab.Prefabs.Find(p => p.UintIdentifier == orderPrefabUintIdentifier);
603 if (orderPrefab != null)
604 {
605 var component = orderPrefab.GetTargetItemComponent(targetEntity as Item);
606 if (!orderPrefab.MustSetTarget || (targetEntity != null && component != null) || targetPosition != null)
607 {
608 var order = targetPosition == null ?
609 new Order(orderPrefab, targetEntity, component, orderGiver: orderGiver) :
610 new Order(orderPrefab, targetPosition, orderGiver: orderGiver);
611 order = order.WithOption(
612 orderOptionIndex >= 0 && orderOptionIndex < orderPrefab.Options.Length
613 ? orderPrefab.Options[orderOptionIndex]
614 : Identifier.Empty)
615 .WithManualPriority(orderPriority)
616 .WithOrderGiver(orderGiver);
617 character.SetOrder(order, isNewOrder: true, speak: false, force: true);
618 }
619 else
620 {
621 DebugConsole.ThrowError("Could not set order \"" + orderPrefab.Identifier + "\" for character \"" + character.Name + "\" because required target entity was not found.");
622 }
623 }
624 else
625 {
626 DebugConsole.ThrowError("Invalid order prefab index - index (" + orderPrefabUintIdentifier + ") out of bounds.");
627 }
628 }
629
630 bool containsStatusData = inc.ReadBoolean();
631 if (containsStatusData)
632 {
633 character.ReadStatus(inc);
634 }
635
636 if (character.IsHuman && character.TeamID != CharacterTeamType.FriendlyNPC && character.TeamID != CharacterTeamType.None && !character.IsDead)
637 {
638 CharacterInfo duplicateCharacterInfo = GameMain.GameSession.CrewManager.GetCharacterInfos().FirstOrDefault(c => c.ID == info.ID);
639 GameMain.GameSession.CrewManager.RemoveCharacterInfo(duplicateCharacterInfo);
640 GameMain.GameSession.CrewManager.AddCharacter(character);
641 }
642
643 if (GameMain.Client.ID == ownerId)
644 {
645 GameMain.Client.HasSpawned = true;
646 GameMain.Client.Character = character;
647 if (!character.IsDead) { Controlled = character; }
648
649 GameMain.LightManager.LosEnabled = true;
650 GameMain.LightManager.LosAlpha = 1f;
651
652 character.memInput.Clear();
653 character.memState.Clear();
654 character.memLocalState.Clear();
655 }
656 }
657
658 character.Enabled = Controlled == character || enabled;
659
660 return character;
661 }

663 private void ReadStatus(IReadMessage msg)
664 {
665 bool isDead = msg.ReadBoolean();
666 if (isDead)
667 {
668 CauseOfDeathType causeOfDeathType = (CauseOfDeathType)msg.ReadRangedInteger(0, Enum.GetValues(typeof(CauseOfDeathType)).Length - 1);
669 AfflictionPrefab causeOfDeathAffliction = null;
670 if (causeOfDeathType == CauseOfDeathType.Affliction)
671 {
672 string afflictionName = msg.ReadString();
673 if (!AfflictionPrefab.Prefabs.ContainsKey(afflictionName))
674 {
675 string errorMsg = $"Error in CharacterNetworking.ReadStatus: affliction not found ({afflictionName})";
676 causeOfDeathType = CauseOfDeathType.Unknown;
677 GameAnalyticsManager.AddErrorEventOnce("CharacterNetworking.ReadStatus:AfflictionIndexOutOfBounts", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
678 }
679 else
680 {
681 causeOfDeathAffliction = AfflictionPrefab.Prefabs[afflictionName];
682 }

709 }
710 else
711 {

713 }
