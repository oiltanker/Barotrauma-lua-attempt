25 public CharacterTalent(TalentPrefab talentPrefab, Character character)
26 {
27 Character = character;
28
29 Prefab = talentPrefab;
30 var element = talentPrefab.ConfigElement;
31 DebugIdentifier = talentPrefab.OriginalName;
32
33 foreach (var subElement in element.Elements())
34 {
35 switch (subElement.Name.ToString().ToLowerInvariant())
36 {
37 case "abilitygroupeffect":
38 LoadAbilityGroupEffect(subElement);
39 break;
40 case "abilitygroupinterval":
41 LoadAbilityGroupInterval(subElement);
42 break;
43 case "addedrecipe":
44 if (subElement.GetAttributeIdentifier("itemidentifier", Identifier.Empty) is { IsEmpty: false } recipeIdentifier)
45 {
46 UnlockedRecipes.Add(recipeIdentifier);
47 }
48 else
49 {
50 DebugConsole.ThrowError("No recipe identifier defined for talent " + DebugIdentifier);
51 }
52 break;
53 }
54 }

99 public void AddAbilityGroupEffect(CharacterAbilityGroupEffect characterAbilityGroup, AbilityEffectType abilityEffectType = AbilityEffectType.None)
100 {
101 if (characterAbilityGroupEffectDictionary.TryGetValue(abilityEffectType, out var characterAbilityList))
102 {
103 characterAbilityList.Add(characterAbilityGroup);
104 }
105 else
106 {
107 List<CharacterAbilityGroupEffect> characterAbilityGroups = new List<CharacterAbilityGroupEffect>();
108 characterAbilityGroups.Add(characterAbilityGroup);
109 characterAbilityGroupEffectDictionary.Add(abilityEffectType, characterAbilityGroups);
110 }

113 public static AbilityEffectType ParseAbilityEffectType(CharacterTalent characterTalent, string abilityEffectTypeString)
114 {
115 if (!Enum.TryParse(abilityEffectTypeString, true, out AbilityEffectType abilityEffectType))
116 {
117 DebugConsole.ThrowError("Invalid ability effect type \"" + abilityEffectTypeString + "\" in CharacterTalent (" + characterTalent.DebugIdentifier + ")");
118 }
119 if (abilityEffectType == AbilityEffectType.Undefined)
120 {
121 DebugConsole.ThrowError("Ability effect type not defined in CharacterTalent (" + characterTalent.DebugIdentifier + ")");
122 }
123
124 return abilityEffectType;
125 }
