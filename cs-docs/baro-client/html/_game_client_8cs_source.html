690  private void ReadDataMessage(IReadMessage inc)
691  {
692  ServerPacketHeader header = (ServerPacketHeader)inc.ReadByte();
693 
694  GameMain.LuaCs.Networking.NetMessageReceived(inc, header);
695 
696  if (roundInitStatus == RoundInitStatus.WaitingForStartGameFinalize &&
697  roundInitStatus == RoundInitStatus.Started &&
698  header != ServerPacketHeader.ENDGAME &&
699  header != ServerPacketHeader.PING_REQUEST &&
700  header != ServerPacketHeader.FILE_TRANSFER)
701  {
702  //rewind the header byte we just read
703  inc.BitPosition -= 8;
704  pendingIncomingMessages.Add(inc);
705  return;
706  }
707 
708  MultiPlayerCampaign campaign = GameMain.NetLobbyScreen.SelectedMode == GameMain.GameSession?.GameMode.Preset ?
709  GameMain.GameSession?.GameMode as MultiPlayerCampaign : null;
710 
711  if (Screen.Selected is ModDownloadScreen)
712  {
713  switch (header)
714  {
715  case ServerPacketHeader.UPDATE_LOBBY:
716  case ServerPacketHeader.PING_REQUEST:
717  case ServerPacketHeader.FILE_TRANSFER:
718  case ServerPacketHeader.PERMISSIONS:
719  case ServerPacketHeader.CHEATS_ENABLED:
720  //allow interpreting this packet
721  break;
722  case ServerPacketHeader.STARTGAME:
723  gameStarted = true;
724  return;
725  case ServerPacketHeader.ENDGAME:
726  gameStarted = false;
727  return;
728  default:
729  return; //ignore any other packets
730  }
731  }
732 
733  switch (header)
734  {
735  case ServerPacketHeader.PING_REQUEST:
736  IWriteMessage response = new WriteOnlyMessage();
737  response.Write((byte)ClientPacketHeader.PING_RESPONSE);
738  byte requestLen = inc.ReadByte();
739  response.Write(requestLen);
740  for (int i = 0; i < requestLen; i++)
741  {
742  byte b = inc.ReadByte();
743  response.Write(b);
744  }
745  clientPeer.Send(response, DeliveryMethod.Unreliable);
746  break;
747  case ServerPacketHeader.CLIENT_PINGS:
748  byte clientCount = inc.ReadByte();
749  for (int i = 0; i < clientCount; i++)
750  {
751  byte clientId = inc.ReadByte();
752  UInt16 clientPing = inc.ReadUInt16();
753  Client client = ConnectedClients.Find(c => c.ID == clientId);
754  if (client != null)
755  {
756  client.Ping = clientPing;
757  }
758  }
759  break;
760  case ServerPacketHeader.UPDATE_LOBBY:
761  ReadLobbyUpdate(inc);
