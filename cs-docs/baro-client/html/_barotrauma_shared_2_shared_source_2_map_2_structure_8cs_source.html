564  private void CreateSections()
565  {
566  int xsections = 1, ysections = 1;
567  int width = rect.Width, height = rect.Height;
568 
569  if (!HasBody)
570  {
571  if (FlippedX && IsHorizontal)
572  {
573  xsections = (int)Math.Ceiling((float)rect.Width / base.Prefab.Sprite.SourceRect.Width);
574  width = base.Prefab.Sprite.SourceRect.Width;
575  }
576  else if (FlippedY && !IsHorizontal)
577  {
578  ysections = (int)Math.Ceiling((float)rect.Height / base.Prefab.Sprite.SourceRect.Height);
579  width = base.Prefab.Sprite.SourceRect.Height;
580  }
581  else
582  {
583  xsections = 1;
584  ysections = 1;
585  }
586  Sections = new WallSection[xsections];
587  }
588  else
589  {
590  if (IsHorizontal)
591  {
592  //equivalent to (int)Math.Ceiling((double)rect.Width / WallSectionSize) without the potential for floating point indeterminism
593  xsections = (rect.Width + WallSectionSize - 1) / WallSectionSize;
594  Sections = new WallSection[xsections];
595  width = WallSectionSize;
596  }
597  else
598  {
599  ysections = (rect.Height + WallSectionSize - 1) / WallSectionSize;
600  Sections = new WallSection[ysections];
601  height = WallSectionSize;
602  }
603  }
604 
605  for (int x = 0; x < xsections; x++)
606  {
607  for (int y = 0; y < ysections; y++)
608  {
609  if (FlippedX || FlippedY)
610  {
611  Rectangle sectionRect = new Rectangle(
612  FlippedX ? rect.Right - (x + 1) * width : rect.X + x * width,
613  FlippedY ? rect.Y - rect.Height + (y + 1) * height : rect.Y - y * height,
614  width, height);
615 
616  if (FlippedX)
617  {
618  int over = Math.Max(rect.X - sectionRect.X, 0);
619  sectionRect.X += over;
620  sectionRect.Width -= over;
621  }
622  else
623  {
624  sectionRect.Width -= (int)Math.Max(sectionRect.Right - rect.Right, 0.0f);
625  }
626  if (FlippedY)
627  {
628  int over = Math.Max(sectionRect.Y - rect.Y, 0);
629  sectionRect.Y -= over;
630  sectionRect.Height -= over;
631  }
632  else
633  {
634  sectionRect.Height -= (int)Math.Max((rect.Y - rect.Height) - (sectionRect.Y - sectionRect.Height), 0.0f);
635  }
636 
637  //sectionRect.Height -= (int)Math.Max((rect.Y - rect.Height) - (sectionRect.Y - sectionRect.Height), 0.0f);
638  int xIndex = FlippedX && IsHorizontal ? (xsections - 1 - x) : x;
639  int yIndex = FlippedY && !IsHorizontal ? (ysections - 1 - y) : y;
640  Sections[xIndex + yIndex] = new WallSection(sectionRect, this);
641  }
642  else
643  {

645  sectionRect.Width -= (int)Math.Max(sectionRect.Right - rect.Right, 0.0f);
646  sectionRect.Height -= (int)Math.Max((rect.Y - rect.Height) - (sectionRect.Y - sectionRect.Height), 0.0f);
