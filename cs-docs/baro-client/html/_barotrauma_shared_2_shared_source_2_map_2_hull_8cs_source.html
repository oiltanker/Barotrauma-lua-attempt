941  waveY[waveY.Length - 1] += rightDelta[waveY.Length - 1];
942

953  }
954  }
955 
956 
957  //apply spread (two iterations)
958  for (int j = 0; j < 2; j++)
959  {
960  for (int i = 1; i < waveY.Length - 1; i++)
961  {
962  waveY[i - 1] += leftDelta[i];
963  waveY[i + 1] += rightDelta[i];
964  }
965  }
966
