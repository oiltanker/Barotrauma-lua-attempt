204  DateTime getEditTime(ContentPackage p)
205  {
206  DateTime writeTime = File.GetLastWriteTime(p.Dir);
207 
208  //File.GetLastWriteTime on the directory is not good enough;
209  //it's possible to update a file in a directory without
210  //updating its parent directories' write time, so let's
211  //look at all of those files
212  var files = Directory.GetFiles(p.Dir, "*", System.IO.SearchOption.AllDirectories);
213  foreach (var file in files)
214  {
215  DateTime newTime = File.GetLastWriteTime(file);
216  if (newTime > writeTime) { writeTime = newTime; }
217  }
218 
219  return writeTime;
220  }
221 
222  //Find local packages associated with the Workshop items if available
223  (Steamworks.Ugc.Item WorkshopItem, ContentPackage? LocalPackage)[] publishedItems = workshopItems
224  .Select(item => (item,
225  (ContentPackage?)ContentPackageManager.LocalPackages.FirstOrDefault(p
226  => p.SteamWorkshopId != 0 && p.SteamWorkshopId == item.Id)))
227  //Sort the pairs by last local edit time if available
228  .OrderBy(t => t.Item2 == null)
229  .ThenByDescending(t => t.Item2 is { } p ? getEditTime(p) : t.Item1.LatestUpdateTime)

479  private GUICustomComponent CreateAsyncThumbnailComponent(GUIFrame thumbnailContainer, Func<Texture2D?> textureGetter, Func<bool> throbberEnabled)
480  {
481  int randomThrobberOffset = Rand.Range(0, 10, Rand.RandSync.Unsynced);
482  return new GUICustomComponent(

484  onDraw: (spriteBatch, component) =>
485  {
486  Rectangle rect = component.Rect;
487  Texture2D? texture = textureGetter();
488  if (texture != null)
489  {
490  rect.Location += (4, 4);
491  rect.Size -= (8, 8);
492  Point destinationSizeMaxWidth = (rect.Width, rect.Width * texture.Height / texture.Width);
493  Point destinationSizeMaxHeight = (rect.Height * texture.Width / texture.Height, rect.Height);
494  Point destinationSize = destinationSizeMaxHeight.X > rect.Width
495  ? destinationSizeMaxWidth
496  : destinationSizeMaxHeight;
497  Rectangle destinationRectangle = new Rectangle(
498  rect.Center.X - destinationSize.X / 2,
499  rect.Center.Y - destinationSize.Y / 2,
500  destinationSize.X,
501  destinationSize.Y);
502  spriteBatch.Draw(texture, destinationRectangle, Color.White);
503  }
504  else if (throbberEnabled())
505  {
506  var sheet = GUIStyle.GenericThrobber;
507  Vector2 pos = rect.Center.ToVector2() - Vector2.One * rect.Height * 0.4f;
508  sheet.Draw(spriteBatch, ((int)Math.Floor(Timing.TotalTime * 24.0f) + randomThrobberOffset) % sheet.FrameCount, pos, Color.White,
509  origin: Vector2.Zero, rotate: 0.0f,
