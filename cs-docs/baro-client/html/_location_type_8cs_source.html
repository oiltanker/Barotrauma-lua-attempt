176  var hireableJobs = new List<(Identifier, float)>();

178  {
179  switch (subElement.Name.ToString().ToLowerInvariant())

263  public static LocationType Random(Random rand, int? zone = null, bool requireOutpost = false)
264  {
265  Debug.Assert(Prefabs.Any(), "LocationType.list.Count == 0, you probably need to initialize LocationTypes");
266 
267  LocationType[] allowedLocationTypes =
268  Prefabs.Where(lt => (!zone.HasValue || lt.CommonnessPerZone.ContainsKey(zone.Value)) && (!requireOutpost || lt.HasOutpost))
269  .OrderBy(p => p.UintIdentifier).ToArray();
270 
271  if (allowedLocationTypes.Length == 0)
272  {
273  DebugConsole.ThrowError("Could not generate a random location type - no location types for the zone " + zone + " found!");
274  }
275 
276  if (zone.HasValue)
277  {
278  return ToolBox.SelectWeightedRandom(
279  allowedLocationTypes,
280  allowedLocationTypes.Select(a => a.CommonnessPerZone[zone.Value]).ToArray(),
281  rand);
282  }
283  else
284  {
285  return allowedLocationTypes[rand.Next() % allowedLocationTypes.Length];
286  }
