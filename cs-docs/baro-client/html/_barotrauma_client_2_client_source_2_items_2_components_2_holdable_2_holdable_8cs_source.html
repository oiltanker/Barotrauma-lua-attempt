77 {

80 bool readAttachData = msg.ReadBoolean();
81 if (!readAttachData) { return; }
82
83 bool shouldBeAttached = msg.ReadBoolean();
84 Vector2 simPosition = new Vector2(msg.ReadSingle(), msg.ReadSingle());
85 UInt16 submarineID = msg.ReadUInt16();
86 Submarine sub = Entity.FindEntityByID(submarineID) as Submarine;

88 if (shouldBeAttached)
