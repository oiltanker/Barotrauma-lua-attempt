85 partial void ShowOnDisplay(string input, bool addToHistory, Color color)
86 {
87 if (addToHistory)
88 {
89 messageHistory.Add(new TerminalMessage(input, color));
90 while (messageHistory.Count > MaxMessages)
91 {
92 messageHistory.RemoveAt(0);
93 }
94 while (historyBox.Content.CountChildren > MaxMessages)
95 {
96 historyBox.RemoveChild(historyBox.Content.Children.First());
97 }
98 }
99
100 GUITextBlock newBlock = new GUITextBlock(
101 new RectTransform(new Vector2(1, 0), historyBox.Content.RectTransform, anchor: Anchor.TopCenter),
102 "> " + input,
103 textColor: color, wrap: true, font: UseMonospaceFont ? GUIStyle.MonospacedFont : GUIStyle.GlobalFont)
104 {
105 CanBeFocused = false
106 };
107
108 if (fillerBlock != null)
109 {
110 float y = fillerBlock.RectTransform.RelativeSize.Y - newBlock.RectTransform.RelativeSize.Y;
111 if (y > 0)
112 {
113 fillerBlock.RectTransform.RelativeSize = new Vector2(1, y);
114 }
115 else
116 {
117 historyBox.RemoveChild(fillerBlock);
118 fillerBlock = null;
119 }
120 }
121
122 historyBox.RecalculateChildren();
123 historyBox.UpdateScrollBarSize();
124 historyBox.ScrollBar.BarScrollValue = 1;
