386  public static bool CheckConnectionLabelOverlap(ConnectionPanel panel, out Point newRectSize)
387  {
388  Rectangle panelRect = panel.GuiFrame.Rect;
389  int x = panelRect.X, y = panelRect.Y;
390  Vector2 scale = GetScale(panel.GuiFrame.RectTransform.MaxSize, panel.GuiFrame.Rect.Size);
391  Vector2 rightPos = GetRightPos(x, y, panelRect.Width, scale);
392  Vector2 leftPos = GetLeftPos(x, y, scale);
393  int connectorIntervalLeft = GetConnectorIntervalLeft(panelRect.Height, scale, panel);
394  int connectorIntervalRight = GetConnectorIntervalRight(panelRect.Height, scale, panel);
395  newRectSize = panelRect.Size;
396  var labelAreas = new List<Rectangle>();
397  for (int i = 0; i < 100; i++)
398  {
399  labelAreas.Clear();
400  foreach (var c in panel.Connections)
401  {
402  if (c.IsOutput)
403  {
404  var labelArea = GetLabelArea(GetOutputLabelPosition(rightPos, panel, c), c.DisplayName.Value.ToUpperInvariant(), scale);
405  labelAreas.Add(labelArea);
406  rightPos.Y += connectorIntervalLeft;
407  }
408  else
409  {
410  var labelArea = GetLabelArea(GetInputLabelPosition(leftPos, panel, c), c.DisplayName.Value.ToUpperInvariant(), scale);
411  labelAreas.Add(labelArea);
412  leftPos.Y += connectorIntervalRight;
413  }
414  }
415  bool foundOverlap = false;
416  for (int j = 0; j < labelAreas.Count; j++)
417  {
418  for (int k = 0; k < labelAreas.Count; k++)
419  {
420  if (k == j) { continue; }
421  if (!labelAreas[j].Intersects(labelAreas[k])) { continue; }
422  newRectSize += new Point(10);
423  Point maxSize = new Point(
424  Math.Max(panel.GuiFrame.RectTransform.MaxSize.X, newRectSize.X),
425  Math.Max(panel.GuiFrame.RectTransform.MaxSize.Y, newRectSize.Y));
426  scale = GetScale(maxSize, newRectSize);
427  rightPos = GetRightPos(x, y, newRectSize.X, scale);
428  leftPos = GetLeftPos(x, y, scale);
429  connectorIntervalLeft = GetConnectorIntervalLeft(newRectSize.Y, scale, panel);
430  connectorIntervalRight = GetConnectorIntervalRight(newRectSize.Y, scale, panel);
431  foundOverlap = true;
432  break;
433  }
434  }
435  if (!foundOverlap) { break; }
436  }
437  return newRectSize.X != panel.GuiFrame.Rect.Width || newRectSize.Y > panel.GuiFrame.Rect.Height;
438  }
