74  float rotation = (item.body.Dir == 1.0f) ? item.body.Rotation : item.body.Rotation - MathHelper.Pi;

76 
77  Vector2 mouseDiff = itemPos - PlayerInput.MousePosition;
78  crosshairPos = new Vector2(
79  MathHelper.Clamp(itemPos.X + barrelDir.X * mouseDiff.Length(), 0, GameMain.GraphicsWidth),

81 
82  float spread = GetSpread(character);

88

90  float spreadDist = (float)Math.Sin(spread) * crossHairDist;

98  partial void UpdateProjSpecific(float deltaTime)
99  {

101 
102  switch (currentChargingState)

106  Vector2 particlePos = item.WorldPosition + ConvertUnits.ToDisplayUnits(TransformedBarrelPos);

108  foreach (ParticleEmitter emitter in particleEmitterCharges)
109  {
110  emitter.Emit(deltaTime, particlePos, hullGuess: item.CurrentHull, sizeMultiplier: sizeMultiplier, colorMultiplier: emitter.Prefab.Properties.ColorMultiplier);
111  }
112 
113  if (chargeSoundChannel == null || !chargeSoundChannel.IsPlaying)
114  {
115  if (chargeSound != null)
116  {
117  chargeSoundChannel = SoundPlayer.PlaySound(chargeSound.Sound, item.WorldPosition, chargeSound.Volume, chargeSound.Range, ignoreMuffling: chargeSound.IgnoreMuffling);
118  if (chargeSoundChannel != null) chargeSoundChannel.Looping = true;
119  }
120  }
121  else if (chargeSoundChannel != null)
122  {
123  chargeSoundChannel.FrequencyMultiplier = MathHelper.Lerp(0.5f, 1.5f, chargeRatio);
124  }
125  break;
126  default:
127  if (chargeSoundChannel != null)
128  {
129  if (chargeSoundChannel.IsPlaying)
130  {
131  chargeSoundChannel.FadeOutAndDispose();
132  chargeSoundChannel.Looping = false;
133  }
134  else
135  {
136  chargeSoundChannel = null;
137  }
138  }
139  break;
