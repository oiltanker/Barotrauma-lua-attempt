36 partial void InitProjSpecific(ContentXElement element)
37 {
38 foreach (var subElement in element.Elements())
39 {
40 string textureDir = GetTextureDirectory(subElement);
41 switch (subElement.Name.ToString().ToLowerInvariant())
42 {
43 case "crosshair":
44 {
45 crosshairSprite = new Sprite(subElement, path: textureDir);
46 }
47 break;
48 case "crosshairpointer":
49 {
50 crosshairPointerSprite = new Sprite(subElement, path: textureDir);
51 }
52 break;
53 case "particleemitter":
54 particleEmitters.Add(new ParticleEmitter(subElement));
55 break;
56 case "particleemittercharge":
57 particleEmitterCharges.Add(new ParticleEmitter(subElement));
58 break;
59 case "chargesound":
60 chargeSound = RoundSound.Load(subElement, false);
61 break;
62 }
63 }

66 public override void UpdateHUD(Character character, float deltaTime, Camera cam)
67 {
68 currentCrossHairScale = currentCrossHairPointerScale = cam == null ? 1.0f : cam.Zoom;
69 if (crosshairSprite != null)
70 {
71 Vector2 aimRefWorldPos = character.AimRefPosition;
72 if (character.Submarine != null) { aimRefWorldPos += character.Submarine.Position; }
73 Vector2 itemPos = cam.WorldToScreen(aimRefWorldPos);
74 float rotation = (item.body.Dir == 1.0f) ? item.body.Rotation : item.body.Rotation - MathHelper.Pi;
75 Vector2 barrelDir = new Vector2((float)Math.Cos(rotation), -(float)Math.Sin(rotation));
76
77 Vector2 mouseDiff = itemPos - PlayerInput.MousePosition;
78 crosshairPos = new Vector2(
79 MathHelper.Clamp(itemPos.X + barrelDir.X * mouseDiff.Length(), 0, GameMain.GraphicsWidth),
80 MathHelper.Clamp(itemPos.Y + barrelDir.Y * mouseDiff.Length(), 0, GameMain.GraphicsHeight));
81
82 float spread = GetSpread(character);
83 Projectile projectile = FindProjectile();
84 if (projectile != null)
85 {
86 spread += MathHelper.ToRadians(projectile.Spread);
87 }
88
89 float crossHairDist = Vector2.Distance(item.WorldPosition, cam.ScreenToWorld(crosshairPos));
90 float spreadDist = (float)Math.Sin(spread) * crossHairDist;
91
92 currentCrossHairPointerScale = MathHelper.Clamp(spreadDist / Math.Min(crosshairSprite.size.X, crosshairSprite.size.Y), 0.1f, 10.0f);
93 }
94 currentCrossHairScale *= CrossHairScale;
95 crosshairPointerPos = PlayerInput.MousePosition;
96 }

98 partial void UpdateProjSpecific(float deltaTime)
99 {
100 float chargeRatio = currentChargeTime / MaxChargeTime;
101
102 switch (currentChargingState)
103 {
104 case ChargingState.WindingUp:
105 case ChargingState.WindingDown:
106 Vector2 particlePos = item.WorldPosition + ConvertUnits.ToDisplayUnits(TransformedBarrelPos);
107 float sizeMultiplier = Math.Clamp(chargeRatio, 0.1f, 1f);
108 foreach (ParticleEmitter emitter in particleEmitterCharges)
109 {
110 emitter.Emit(deltaTime, particlePos, hullGuess: item.CurrentHull, sizeMultiplier: sizeMultiplier, colorMultiplier: emitter.Prefab.Properties.ColorMultiplier);
111 }
112
113 if (chargeSoundChannel == null || !chargeSoundChannel.IsPlaying)
114 {
115 if (chargeSound != null)
116 {
117 chargeSoundChannel = SoundPlayer.PlaySound(chargeSound.Sound, item.WorldPosition, chargeSound.Volume, chargeSound.Range, ignoreMuffling: chargeSound.IgnoreMuffling);
118 if (chargeSoundChannel != null) chargeSoundChannel.Looping = true;
119 }
120 }
121 else if (chargeSoundChannel != null)
122 {
123 chargeSoundChannel.FrequencyMultiplier = MathHelper.Lerp(0.5f, 1.5f, chargeRatio);
124 }
125 break;
126 default:
127 if (chargeSoundChannel != null)
128 {
129 if (chargeSoundChannel.IsPlaying)
130 {
131 chargeSoundChannel.FadeOutAndDispose();
132 chargeSoundChannel.Looping = false;
133 }
134 else
135 {
136 chargeSoundChannel = null;
137 }
138 }
139 break;
140 }

143 public override void DrawHUD(SpriteBatch spriteBatch, Character character)
144 {
145 if (character == null || !character.IsKeyDown(InputType.Aim)) { return; }
146
147 //camera focused on some other item/device, don't draw the crosshair
148 if (character.ViewTarget != null && (character.ViewTarget is Item viewTargetItem) && viewTargetItem.Prefab.FocusOnSelected) { return; }
149 //don't draw the crosshair if the item is in some other type of equip slot than hands (e.g. assault rifle in the bag slot)
150 if (!character.HeldItems.Contains(item)) { return; }
151
152 GUI.HideCursor = (crosshairSprite != null || crosshairPointerSprite != null) &&
153 GUI.MouseOn == null && !Inventory.IsMouseOnInventory && !GameMain.Instance.Paused;
154 if (GUI.HideCursor)
155 {
156 crosshairSprite?.Draw(spriteBatch, crosshairPos, Color.White, 0, currentCrossHairScale);
157 crosshairPointerSprite?.Draw(spriteBatch, crosshairPointerPos, 0, currentCrossHairPointerScale);
158 }
159
160 if (GameMain.DebugDraw)
161 {
162 Vector2 barrelPos = item.DrawPosition + ConvertUnits.ToDisplayUnits(TransformedBarrelPos);
163 barrelPos = Screen.Selected.Cam.WorldToScreen(barrelPos);
164 GUI.DrawLine(spriteBatch, barrelPos - Vector2.UnitY * 3, barrelPos + Vector2.UnitY * 3, Color.Red);
165 GUI.DrawLine(spriteBatch, barrelPos - Vector2.UnitX * 3, barrelPos + Vector2.UnitX * 3, Color.Red);
166 }

169 partial void LaunchProjSpecific()
170 {
171 Vector2 particlePos = item.WorldPosition + ConvertUnits.ToDisplayUnits(TransformedBarrelPos);
172 float rotation = -item.body.Rotation;
173 if (item.body.Dir < 0.0f) { rotation += MathHelper.Pi; }
174 foreach (ParticleEmitter emitter in particleEmitters)
175 {
176 emitter.Emit(1.0f, particlePos, hullGuess: item.CurrentHull, angle: rotation, particleRotation: rotation);
177 }
