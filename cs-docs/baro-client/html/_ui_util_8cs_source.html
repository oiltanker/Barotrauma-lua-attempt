70 protected static void SwapDropdownValues<T>(
71 GUIDropDown dropdown, Func<T, LocalizedString> textFunc, IReadOnlyList<T> values, T currentValue,
72 Action<T> setter)
73 {
74 if (dropdown.ListBox.Content.Children.Any(c => !(c.UserData is T)))
75 {
76 throw new Exception("SwapValues must preserve the type of the dropdown's userdata");
77 }
78
79 dropdown.OnSelected = null;
80 dropdown.ClearChildren();
81
82 values.ForEach(v => dropdown.AddItem(text: textFunc(v), userData: v));
83 dropdown.Select(values.IndexOf(currentValue));
84 dropdown.OnSelected = (dd, obj) =>
85 {
86 setter((T)obj);
87 return true;
88 };
89 }
