236 {

252 private bool ChangeSpawnType(GUIButton button, object obj)
253 {
254 GUITextBlock spawnTypeText = button.Parent.GetChildByUserData("spawntypetext") as GUITextBlock;
255 var values = (SpawnType[])Enum.GetValues(typeof(SpawnType));
256 int currIndex = values.IndexOf(spawnType);
257 currIndex += (int)button.UserData;
258 int firstIndex = 1;
259 int lastIndex = values.Length - 1;
260 if (currIndex > lastIndex)
261 {
262 currIndex = firstIndex;
263 }
264 if (currIndex < firstIndex)
265 {
266 currIndex = lastIndex;
267 }
268 spawnType = values[currIndex];
269 spawnTypeText.Text = spawnType.ToString();
270 return true;
271 }
