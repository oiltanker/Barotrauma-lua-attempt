64  protected override void StartMissionSpecific(Level level)
65  {
66  Reset();
67 
68  if (IsClient) { return; }
69 
70  if (itemConfig == null)
71  {
72  DebugConsole.ThrowError("Failed to initialize a Scan mission: item config is not set");
73  return;
74  }
75 
76  foreach (var element in itemConfig.Elements())
77  {
78  LoadItem(element, null);
79  }
80  GetScanners();
81 
82  TargetRuin = Level.Loaded?.Ruins?.GetRandom(randSync: Rand.RandSync.ServerAndClient);
83  if (TargetRuin == null)
84  {
85  DebugConsole.ThrowError("Failed to initialize a Scan mission: level contains no alien ruins");
86  return;
87  }
88 
89  var ruinWaypoints = TargetRuin.Submarine.GetWaypoints(false);
90  ruinWaypoints.RemoveAll(wp => wp.CurrentHull == null);
91  if (ruinWaypoints.Count < targetsToScan)
92  {
93  DebugConsole.ThrowError($"Failed to initialize a Scan mission: target ruin has less waypoints than required as scan targets ({ruinWaypoints.Count} < {targetsToScan})");
94  return;
95  }
96  var availableWaypoints = new List<WayPoint>();
97  float minTargetDistanceSquared = minTargetDistance * minTargetDistance;
98  for (int tries = 0; tries < 15; tries++)
99  {
100  scanTargets.Clear();
101  availableWaypoints.Clear();
102  availableWaypoints.AddRange(ruinWaypoints);
103  for (int i = 0; i < targetsToScan; i++)
104  {
105  var selectedWaypoint = availableWaypoints.GetRandom(randSync: Rand.RandSync.ServerAndClient);
106  scanTargets.Add(selectedWaypoint, false);
107  availableWaypoints.Remove(selectedWaypoint);
108  if (i < (targetsToScan - 1))
109  {
110  availableWaypoints.RemoveAll(wp => wp.CurrentHull == selectedWaypoint.CurrentHull);
111  availableWaypoints.RemoveAll(wp => Vector2.DistanceSquared(wp.WorldPosition, selectedWaypoint.WorldPosition) < minTargetDistanceSquared);
112  if (availableWaypoints.None())
113  {
114 #if DEBUG
115  DebugConsole.ThrowError($"Error initializing a Scan mission: not enough targets available on try #{tries + 1} to reach the required scan target count (current targets: {scanTargets.Count}, required targets: {targetsToScan})");
116 #endif
117  break;
118  }
119  }
120  }
121  if (scanTargets.Count >= targetsToScan)
122  {
123 #if DEBUG
124  DebugConsole.NewMessage($"Successfully initialized a Scan mission: targets set on try #{tries + 1}", Color.Green);
125 #endif
126  break;
127  }
128  if ((tries + 1) % 5 == 0)
129  {
130  float reducedMinTargetDistance = (1.0f - (((tries + 1) / 5) * 0.1f)) * minTargetDistance;
131  minTargetDistanceSquared = reducedMinTargetDistance * reducedMinTargetDistance;
132 #if DEBUG
133  DebugConsole.NewMessage($"Reducing minimum distance between Scan mission targets (new min: {reducedMinTargetDistance}) to reach the required target count", Color.Yellow);
134 #endif
135  }
136  }
137  if (scanTargets.Count < targetsToScan)
138  {
139  DebugConsole.ThrowError($"Error initializing a Scan mission: not enough targets (current targets: {scanTargets.Count}, required targets: {targetsToScan})");
140  }
141  }

206  private void OnScanCompleted(Scanner scanner)
207  {
208  if (IsClient) { return; }
209  newTargetsScanned.Clear();
210  float scanRadiusSquared = scanner.ScanRadius * scanner.ScanRadius;
211  foreach (var kvp in scanTargets)
212  {
213  if (!IsValidScanPosition(scanner, kvp, scanRadiusSquared)) { continue; }
214  newTargetsScanned.Add(kvp.Key);
215  }
216  foreach (var wp in newTargetsScanned)

218  scanTargets[wp] = true;

226  private bool IsValidScanPosition(Scanner scanner, KeyValuePair<WayPoint, bool> scanStatus, float scanRadiusSquared)
227  {
228  if (scanStatus.Value) { return false; }
229  if (scanStatus.Key.Submarine != scanner.Item.Submarine) { return false; }
230  if (Vector2.DistanceSquared(scanStatus.Key.WorldPosition, scanner.Item.WorldPosition) > scanRadiusSquared) { return false; }
231  return true;

250  public override void End()
251  {
252  if (State == 2 && AllScannersReturned())
253  {
254  GiveReward();
255  completed = true;
256  }
257  foreach (var scanner in scanners)
258  {
259  if (scanner.Item != null && !scanner.Item.Removed)
260  {
261  scanner.OnScanStarted -= OnScanStarted;
262  scanner.OnScanCompleted -= OnScanCompleted;
263  scanner.Item.Remove();

269  bool AllScannersReturned()
270  {
271  foreach (var scanner in scanners)
272  {
273  if (scanner?.Item == null || scanner.Item.Removed) { return false; }
274  var owner = scanner.Item.GetRootInventoryOwner();
275  if (owner.Submarine != null && owner.Submarine.Info.Type == SubmarineType.Player)
276  {
277  continue;
278  }
279  else if (owner is Character c && c.Info != null && GameMain.GameSession.CrewManager.CharacterInfos.Contains(c.Info))
280  {
281  continue;
282  }
283  return false;
284  }
285  return true;
286  }
