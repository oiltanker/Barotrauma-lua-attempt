94 private void GenerateOutpost(Submarine submarine)
95 {
96 Submarine outpost = OutpostGenerator.Generate(OutpostParams ?? OutpostGenerationParams.OutpostParams.GetRandomUnsynced(), OutpostType ?? LocationType.Prefabs.GetRandomUnsynced());
97 outpost.SetPosition(Vector2.Zero);
98
99 float closestDistance = 0.0f;
100 DockingPort myPort = null, outPostPort = null;
101 foreach (DockingPort port in DockingPort.List)
102 {
103 if (port.IsHorizontal || port.Docked) { continue; }
104 if (port.Item.Submarine == outpost)
105 {
106 outPostPort = port;
107 continue;
108 }
109 if (port.Item.Submarine != submarine) { continue; }
110
111 //the submarine port has to be at the top of the sub
112 if (port.Item.WorldPosition.Y < submarine.WorldPosition.Y) { continue; }
113
114 float dist = Vector2.DistanceSquared(port.Item.WorldPosition, outpost.WorldPosition);
115 if ((myPort == null || dist < closestDistance || port.MainDockingPort) && !(myPort?.MainDockingPort ?? false))
116 {
117 myPort = port;
118 closestDistance = dist;
119 }
120 }
121
122 if (myPort != null && outPostPort != null)
123 {
124 Vector2 portDiff = myPort.Item.WorldPosition - submarine.WorldPosition;
125 Vector2 spawnPos = (outPostPort.Item.WorldPosition - portDiff) - Vector2.UnitY * outPostPort.DockedDistance;
126
127 submarine.SetPosition(spawnPos);
128 myPort.Dock(outPostPort);
129 myPort.Lock(true);
130 }
131
132 if (Character.Controlled != null)
133 {
134 Character.Controlled.TeleportTo(outpost.GetWaypoints(false).GetRandomUnsynced(point => point.SpawnType == SpawnType.Human).WorldPosition);
135 }
136 }
