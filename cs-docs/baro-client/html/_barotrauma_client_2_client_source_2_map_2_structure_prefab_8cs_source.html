24 Vector2 position = Submarine.MouseToWorldGrid(cam, Submarine.MainSub);

27
28 if (placePosition == Vector2.Zero)
29 {
30 if (PlayerInput.PrimaryMouseButtonHeld())
31 placePosition = Submarine.MouseToWorldGrid(cam, Submarine.MainSub);
32
33 newRect.X = (int)position.X;
34 newRect.Y = (int)position.Y;
35 }
36 else
37 {
38 Vector2 placeSize = size;

40 if (ResizeVertical) placeSize.Y = placePosition.Y - position.Y;
41
42 //don't allow resizing width/height to less than the grid size
43 if (ResizeHorizontal && Math.Abs(placeSize.X) < Submarine.GridSize.X)
44 {
45 placeSize.X = Submarine.GridSize.X;
46 }
47 if (ResizeVertical && Math.Abs(placeSize.Y) < Submarine.GridSize.Y)
48 {
49 placeSize.Y = Submarine.GridSize.Y;
50 }
51
52 newRect = Submarine.AbsRect(placePosition, placeSize);
53 if (PlayerInput.PrimaryMouseButtonReleased())
54 {
55 newRect.Location -= MathUtils.ToPoint(Submarine.MainSub.Position);
56 var structure = new Structure(newRect, this, Submarine.MainSub)
57 {
58 Submarine = Submarine.MainSub
59 };
60
61 SubEditorScreen.StoreCommand(new AddOrDeleteCommand(new List<MapEntity> { structure }, false));
62 placePosition = Vector2.Zero;
63 if (!PlayerInput.IsShiftDown())

65 Selected = null;
66 }
