104 public float GetSliceBorderScale(Point drawSize)
105 {
106 if (!Slice) { return 1.0f; }
107
108 Vector2 scale = new Vector2(
109 MathHelper.Clamp((float)drawSize.X / (Slices[0].Height + Slices[6].Height), 0, 1),
110 MathHelper.Clamp((float)drawSize.Y / (Slices[0].Width + Slices[2].Width), 0, 1));
111 return MathHelper.Clamp(Math.Min(Math.Min(scale.X, scale.Y), GUI.SlicedSpriteScale), minBorderScale, maxBorderScale);
112 }

114 public void Draw(SpriteBatch spriteBatch, Rectangle rect, Color color, SpriteEffects spriteEffects = SpriteEffects.None, Vector2? uvOffset = null)
115 {
116 uvOffset ??= Vector2.Zero;
117 if (Sprite.Texture == null)
118 {
119 GUI.DrawRectangle(spriteBatch, rect, Color.Magenta);
120 return;
121 }
122
123 if (Slice)
124 {
125 Vector2 pos = new Vector2(rect.X, rect.Y);
126
127 float scale = MaintainBorderAspectRatio ? 1.0f : GetSliceBorderScale(rect.Size);
128 float aspectScale = MaintainBorderAspectRatio ? Math.Min((float)rect.Width / Sprite.SourceRect.Width, (float)rect.Height / Sprite.SourceRect.Height) : 1.0f;
129
130 int centerHeight = rect.Height - (int)((Slices[0].Height + Slices[6].Height) * scale);
131 int centerWidth = rect.Width - (int)((Slices[0].Width + Slices[2].Width) * scale * aspectScale);
132
133 for (int x = 0; x < 3; x++)
134 {
135
136 int width = (int)(x == 1 ? centerWidth : Slices[x].Width * scale * aspectScale);
137 if (width <= 0) { continue; }
138 for (int y = 0; y < 3; y++)
139 {
140 int height = (int)(y == 1 ? centerHeight : Slices[x + y * 3].Height * scale);
141 if (height <= 0) { continue; }
142
143 spriteBatch.Draw(Sprite.Texture,
144 new Rectangle((int)pos.X, (int)pos.Y, width, height),
145 Slices[x + y * 3],
146 color);
147
148 pos.Y += height;
149 }
150 pos.X += width;
151 pos.Y = rect.Y;
152 }
153 }
154 else if (Tile)
155 {
156 Vector2 startPos = new Vector2(rect.X, rect.Y);
157 Sprite.DrawTiled(spriteBatch, startPos, new Vector2(rect.Width, rect.Height), color, startOffset: uvOffset);
158 }
159 else
160 {
161 if (MaintainAspectRatio)
162 {
163 float scale = Math.Min((float)rect.Width / Sprite.SourceRect.Width, (float)rect.Height / Sprite.SourceRect.Height);
164
165 spriteBatch.Draw(Sprite.Texture, rect.Center.ToVector2(),
166 Sprite.SourceRect,
167 color,
168 rotation: 0.0f,
169 origin: Sprite.size / 2.0f,
170 scale: scale,
171 effects: spriteEffects, layerDepth: 0.0f);
172 }
173 else
174 {
175 spriteBatch.Draw(Sprite.Texture, rect, Sprite.SourceRect, color, 0, Vector2.Zero, spriteEffects, 0);
176 }
177 }
178 }
