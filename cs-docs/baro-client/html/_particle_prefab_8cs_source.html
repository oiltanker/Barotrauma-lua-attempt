214  public ParticlePrefab(ContentXElement element, ContentFile file) : base(file, element.NameAsIdentifier())
215  {
216  Sprites = new List<Sprite>();
217 
218  SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
219 
220  foreach (var subElement in element.Elements())
221  {
222  switch (subElement.Name.ToString().ToLowerInvariant())
223  {
224  case "sprite":
225  Sprites.Add(new Sprite(subElement));
226  break;
227  case "spritesheet":
228  case "animatedsprite":
229  Sprites.Add(new SpriteSheet(subElement));
230  break;
231  case "particleemitter":
232  case "emitter":
233  case "subemitter":
234  SubEmitters.Add(new ParticleEmitterPrefab(subElement));
235  break;
236  }
237  }
238 
239  if (Sprites.Count == 0)
240  {
241  DebugConsole.ThrowError($"Particle prefab \"{Name}\" in the file \"{file}\" has no sprites defined!");
242  }
243 
244  //if velocity change in water is not given, it defaults to the normal velocity change
245  if (element.GetAttribute("velocitychangewater") == null)
246  {
247  VelocityChangeWater = VelocityChange;
248  }
249 
250  if (element.GetAttribute("angularvelocity") != null)
251  {
252  AngularVelocityMin = element.GetAttributeFloat("angularvelocity", 0.0f);
253  AngularVelocityMax = AngularVelocityMin;
254  }
255 
256  if (element.GetAttribute("startsize") != null)
257  {
258  StartSizeMin = element.GetAttributeVector2("startsize", Vector2.One);
259  StartSizeMax = StartSizeMin;
260  }
261 
262  if (element.GetAttribute("sizechange") != null)
263  {
264  SizeChangeMin = element.GetAttributeVector2("sizechange", Vector2.Zero);
265  SizeChangeMax = SizeChangeMin;
266  }
267 
268  if (element.GetAttribute("startrotation") != null)
269  {
270  StartRotationMin = element.GetAttributeFloat("startrotation", 0.0f);
271  StartRotationMax = StartRotationMin;
272  }
273 
274  if (CollisionRadius <= 0.0f && UseCollision)
275  {
276  CollisionRadius = Sprites.Count > 0 ? Sprites[0].SourceRect.Width / 2.0f : 1;
277  }
278  }

280  public Vector2 CalculateEndPosition(Vector2 startPosition, Vector2 velocity)
281  {
282  //endPos = x + vt + 1/2 * at^2
283  return startPosition + velocity * LifeTime + 0.5f * VelocityChangeDisplay * LifeTime * LifeTime;
284  }

286  public Vector2 CalculateEndSize()
287  {
288  //endPos = x + vt + 1/2 * at^2
