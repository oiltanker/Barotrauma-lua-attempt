322  private readonly HashSet<Sprite> loadedSprites = new HashSet<Sprite>();
323  private void LoadSprites()
324  {
325  loadedSprites.ForEach(s => s.Remove());
326  loadedSprites.Clear();
327  var contentPackages = ContentPackageManager.EnabledPackages.All.ToList();
328 
329 #if !DEBUG
330  var vanilla = GameMain.VanillaContent;
331  if (vanilla != null)
332  {
333  contentPackages.Remove(vanilla);
334  }
335 #endif
336  foreach (var contentPackage in contentPackages)
337  {
338  foreach (var file in contentPackage.Files)
339  {
340  if (file.Path.EndsWith(".xml"))
341  {
342  XDocument doc = XMLExtensions.TryLoadXml(file.Path);
343  if (doc != null)
344  {
345  LoadSprites(doc.Root.FromPackage(file.Path.ContentPackage));
346  }
347  }
348  }
349  }

352  {

378  void CreateSprite(ContentXElement element)
379  {
380  string spriteFolder = "";
381  ContentPath texturePath = null;
382 
383  if (element.GetAttribute("texture") != null)
384  {
385  texturePath = element.GetAttributeContentPath("texture");
386  }
387  else
388  {
389  if (element.Name.ToString().ToLower() == "vinesprite")
390  {
391  texturePath = element.Parent.GetAttributeContentPath("vineatlas");
392  }
393  }
394  if (texturePath.IsNullOrEmpty()) { return; }
395 
396  // TODO: parse and create?
397  if (texturePath.Value.Contains("[GENDER]") || texturePath.Value.Contains("[HEADID]") || texturePath.Value.Contains("[RACE]") || texturePath.Value.Contains("[VARIANT]")) { return; }
398  if (!texturePath.Value.Contains("/"))
399  {
400  var parsedPath = element.ParseContentPathFromUri();
401  spriteFolder = Path.GetDirectoryName(parsedPath);
402  }
403  // Uncomment if we do multiple passes -> there can be duplicates
404  //string identifier = Sprite.GetID(element);
405  //if (loadedSprites.None(s => s.ID == identifier))
406  //{
407  // loadedSprites.Add(new Sprite(element, spriteFolder));
408  //}
409  loadedSprites.Add(new Sprite(element, spriteFolder, texturePath.Value, lazyLoad: true));
410  }

413  private bool SaveSprites(IEnumerable<Sprite> sprites)
414  {
415  if (SelectedTexture == null) { return false; }
416  if (sprites.None()) { return false; }
417  HashSet<XDocument> docsToSave = new HashSet<XDocument>();
418  foreach (Sprite sprite in sprites)
419  {
420  if (sprite.FullPath != selectedTexturePath) { continue; }
421  var element = sprite.SourceElement;
422  if (element == null) { continue; }
423  element.SetAttributeValue("sourcerect", XMLExtensions.RectToString(sprite.SourceRect));
424  element.SetAttributeValue("origin", XMLExtensions.Vector2ToString(sprite.RelativeOrigin));
425 
426  /*if (element.Attribute("slice") != null)
427  {
428  Rectangle slice = new Rectangle(
429  sprite.SourceRect.X + 5,
430  sprite.SourceRect.Y + 5,
431  sprite.SourceRect.Right - 5,
432  sprite.SourceRect.Bottom - 5);
433  element.SetAttributeValue("slice", XMLExtensions.RectToString(slice));
434  }*/
435  docsToSave.Add(element.Document);
436  }
437  xmlPathText.Text = TextManager.Get("spriteeditor.allchangessavedto");
438  foreach (XDocument doc in docsToSave)
439  {
440  string xmlPath = doc.ParseContentPathFromUri();
441  xmlPathText.Text += "\n" + xmlPath;
442 #if DEBUG
443  doc.Save(xmlPath);
444 #else
445  doc.SaveSafe(xmlPath);
446 #endif
447  }
448  xmlPathText.TextColor = GUIStyle.Green;
449  return true;
450  }

466  public override void Update(double deltaTime)
467  {
468  base.Update(deltaTime);
469  Widget.EnableMultiSelect = PlayerInput.KeyDown(Keys.LeftControl);
470  spriteList.SelectMultiple = Widget.EnableMultiSelect;
471  // Select rects with the mouse
472  if (Widget.selectedWidgets.None() || Widget.EnableMultiSelect)
473  {
474  if (SelectedTexture != null && GUI.MouseOn == null)
475  {
476  foreach (Sprite sprite in loadedSprites)
477  {
478  if (sprite.FullPath != selectedTexturePath) { continue; }
479  if (PlayerInput.PrimaryMouseButtonClicked())
480  {
481  var scaledRect = new Rectangle(textureRect.Location + sprite.SourceRect.Location.Multiply(zoom), sprite.SourceRect.Size.Multiply(zoom));
482  if (scaledRect.Contains(PlayerInput.MousePosition))
483  {
484  spriteList.Select(sprite, autoScroll: false);
485  UpdateScrollBar(spriteList);
486  UpdateScrollBar(textureList);
487  // Release the keyboard so that we can nudge the source rects
488  GUI.KeyboardDispatcher.Subscriber = null;
489  }
490  }
491  }
492  }
493  }
494  if (GUI.MouseOn == null)
495  {
496  if (PlayerInput.ScrollWheelSpeed != 0)
497  {
498  zoom = MathHelper.Clamp(zoom + PlayerInput.ScrollWheelSpeed * (float)deltaTime * 0.05f * zoom, MinZoom, MaxZoom);
499  zoomBar.BarScroll = GetBarScrollValue();
500  }
501  widgets.Values.ForEach(w => w.Update((float)deltaTime));
502  if (PlayerInput.MidButtonHeld())
503  {
504  // "Camera" Pan
505  Vector2 moveSpeed = PlayerInput.MouseSpeed * (float)deltaTime * 100.0f;
506  viewAreaOffset += moveSpeed.ToPoint();
507  }
508  }
509  if (GUI.KeyboardDispatcher.Subscriber == null)
510  {
511  if (PlayerInput.KeyHit(Keys.Left))
512  {
513  Nudge(Keys.Left);
514  }
515  if (PlayerInput.KeyHit(Keys.Right))
516  {
517  Nudge(Keys.Right);
518  }
519  if (PlayerInput.KeyHit(Keys.Down))
520  {
521  Nudge(Keys.Down);
522  }
523  if (PlayerInput.KeyHit(Keys.Up))
524  {
525  Nudge(Keys.Up);
526  }
527  if (PlayerInput.KeyDown(Keys.Left))
528  {
529  holdTimer += deltaTime;
530  if (holdTimer > holdTime)
531  {
532  Nudge(Keys.Left);
533  }
534  }
535  else if (PlayerInput.KeyDown(Keys.Right))
536  {
537  holdTimer += deltaTime;
538  if (holdTimer > holdTime)
539  {
540  Nudge(Keys.Right);
541  }
542  }
543  else if (PlayerInput.KeyDown(Keys.Down))
544  {
545  holdTimer += deltaTime;
546  if (holdTimer > holdTime)
547  {
548  Nudge(Keys.Down);
549  }
550  }
551  else if (PlayerInput.KeyDown(Keys.Up))
552  {
553  holdTimer += deltaTime;
554  if (holdTimer > holdTime)
555  {
556  Nudge(Keys.Up);
557  }
558  }
559  else
560  {
561  holdTimer = 0;
562  }
563  }
564  }

566  private double holdTimer;

568  private void Nudge(Keys key)
569  {
570  switch (key)
571  {
572  case Keys.Left:
573  foreach (var sprite in selectedSprites)
574  {
575  var newRect = sprite.SourceRect;
576  if (PlayerInput.KeyDown(Keys.LeftControl))
577  {
578  newRect.Width--;
579  }
580  else
581  {
582  newRect.X--;
583  }
584  UpdateSourceRect(sprite, newRect);
585  }
586  break;
587  case Keys.Right:
588  foreach (var sprite in selectedSprites)
589  {
590  var newRect = sprite.SourceRect;
591  if (PlayerInput.KeyDown(Keys.LeftControl))
592  {
593  newRect.Width++;
594  }
595  else
596  {
597  newRect.X++;
598  }
599  UpdateSourceRect(sprite, newRect);
600  }
601  break;
602  case Keys.Down:
603  foreach (var sprite in selectedSprites)
604  {
605  var newRect = sprite.SourceRect;
606  if (PlayerInput.KeyDown(Keys.LeftControl))
607  {
608  newRect.Height++;
609  }
610  else
611  {
612  newRect.Y++;
613  }
614  UpdateSourceRect(sprite, newRect);
615  }
616  break;
617  case Keys.Up:
618  foreach (var sprite in selectedSprites)
619  {
620  var newRect = sprite.SourceRect;
621  if (PlayerInput.KeyDown(Keys.LeftControl))
622  {
623  newRect.Height--;
624  }
625  else
626  {
627  newRect.Y--;
628  }
629  UpdateSourceRect(sprite, newRect);
630  }

784  private Vector2 SnapToGrid(Vector2 position, Rectangle gridArea, float zoom, Vector2 gridSize, float tolerance)
785  {
786  gridSize *= zoom;
787  if (gridSize.X < 1.0f) { return position; }
788  if (gridSize.Y < 1.0f) { return position; }
789 
790  Vector2 snappedPos = position;
791  snappedPos.X -= gridArea.X;
792  snappedPos.Y -= gridArea.Y;
793 
794  Vector2 gridPos = new Vector2(
795  MathUtils.RoundTowardsClosest(snappedPos.X, gridSize.X),
796  MathUtils.RoundTowardsClosest(snappedPos.Y, gridSize.Y));
797 
798  if (Math.Abs(gridPos.X - snappedPos.X) < tolerance)
799  {
800  snappedPos.X = gridPos.X;
801  }
802  if (Math.Abs(gridPos.Y - snappedPos.Y) < tolerance)
803  {
804  snappedPos.Y = gridPos.Y;
805  }
806 
807  snappedPos.X += gridArea.X;
