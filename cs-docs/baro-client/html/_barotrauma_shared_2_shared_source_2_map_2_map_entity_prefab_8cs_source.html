94 [Obsolete("Prefer MapEntityPrefab.FindByIdentifier or MapEntityPrefab.FindByName")]
95 public static MapEntityPrefab Find(string name, Identifier identifier, bool showErrorMessages = true)
96 {
97 //try to search based on identifier first
98 if (string.IsNullOrEmpty(name) && !identifier.IsEmpty)
99 {
100 if (CoreEntityPrefab.Prefabs.ContainsKey(identifier)) { return CoreEntityPrefab.Prefabs[identifier]; }
101 if (StructurePrefab.Prefabs.ContainsKey(identifier)) { return StructurePrefab.Prefabs[identifier]; }
102 if (ItemPrefab.Prefabs.ContainsKey(identifier)) { return ItemPrefab.Prefabs[identifier]; }
103 if (ItemAssemblyPrefab.Prefabs.ContainsKey(identifier)) { return ItemAssemblyPrefab.Prefabs[identifier]; }
104 }
105
106 foreach (MapEntityPrefab prefab in List)
107 {
108 if (!identifier.IsEmpty)
109 {
110 if (prefab.Identifier != identifier)
111 {
112 if (prefab.Aliases != null && prefab.Aliases.Any(a => a == identifier))
113 {
114 return prefab;
115 }
116 continue;
117 }
118 else
119 {
120 if (string.IsNullOrEmpty(name)) { return prefab; }
121 }
122 }
123 if (!string.IsNullOrEmpty(name))
124 {
125 if (prefab.Name.Equals(name, StringComparison.OrdinalIgnoreCase) ||
126 prefab.OriginalName.Equals(name, StringComparison.OrdinalIgnoreCase) ||
127 (prefab.Aliases != null && prefab.Aliases.Any(a => a.Equals(name, StringComparison.OrdinalIgnoreCase))))
128 {
129 return prefab;
130 }
131 }
132 }
133
134 if (showErrorMessages)
135 {
136 DebugConsole.ThrowError("Failed to find a matching MapEntityPrefab (name: \"" + name + "\", identifier: \"" + identifier + "\").\n" + Environment.StackTrace.CleanupStackTrace());
137 }
138 return null;
139 }

273 public bool IsLinkAllowed(MapEntityPrefab target)
274 {
275 if (target == null) { return false; }
276 if (target is StructurePrefab && AllowedLinks.Contains("structure".ToIdentifier())) { return true; }
277 if (target is ItemPrefab && AllowedLinks.Contains("item".ToIdentifier())) { return true; }
278 if (target is LinkedSubmarinePrefab && Tags.Contains("dock".ToIdentifier())) { return true; }
279 if (this is LinkedSubmarinePrefab && target.Tags.Contains("dock".ToIdentifier())) { return true; }
280 return AllowedLinks.Contains(target.Identifier) || target.AllowedLinks.Contains(Identifier)
281 || target.Tags.Any(t => AllowedLinks.Contains(t)) || Tags.Any(t => target.AllowedLinks.Contains(t));
282 }
