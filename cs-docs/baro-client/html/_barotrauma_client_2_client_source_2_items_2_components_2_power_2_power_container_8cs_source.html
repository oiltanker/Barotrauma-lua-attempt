133 public void Draw(SpriteBatch spriteBatch, bool editing = false, float itemDepth = -1)
134 {
135 if (indicatorSize.X <= 1.0f || indicatorSize.Y <= 1.0f) { return; }
136
137 Vector2 itemSize = new Vector2(item.Sprite.SourceRect.Width, item.Sprite.SourceRect.Height) * item.Scale;
138 Vector2 indicatorPos = -itemSize / 2 + indicatorPosition * item.Scale;
139 if (item.FlippedX && item.Prefab.CanSpriteFlipX) { indicatorPos.X = -indicatorPos.X - indicatorSize.X * item.Scale; }
140 if (item.FlippedY && item.Prefab.CanSpriteFlipY) { indicatorPos.Y = -indicatorPos.Y - indicatorSize.Y * item.Scale; }
141
142 if (charge > 0 && capacity > 0)
143 {
144 float chargeRatio = MathHelper.Clamp(charge / capacity, 0.0f, 1.0f);
145 Color indicatorColor = ToolBox.GradientLerp(chargeRatio, Color.Red, Color.Orange, Color.Green);
146 if (!isHorizontal)
147 {
148 GUI.DrawRectangle(spriteBatch,
149 new Vector2(item.DrawPosition.X, -item.DrawPosition.Y + ((indicatorSize.Y * item.Scale) * (1.0f - chargeRatio))) + indicatorPos,
150 new Vector2(indicatorSize.X * item.Scale, (indicatorSize.Y * item.Scale) * chargeRatio), indicatorColor, true,
151 depth: item.SpriteDepth - 0.00001f);
152 }
153 else
154 {
155 GUI.DrawRectangle(spriteBatch,
156 new Vector2(item.DrawPosition.X, -item.DrawPosition.Y) + indicatorPos,
157 new Vector2((indicatorSize.X * item.Scale) * chargeRatio, indicatorSize.Y * item.Scale), indicatorColor, true,
158 depth: item.SpriteDepth - 0.00001f);
159 }
160 }
161 GUI.DrawRectangle(spriteBatch,
162 new Vector2(item.DrawPosition.X, -item.DrawPosition.Y) + indicatorPos,
163 indicatorSize * item.Scale, Color.Black, depth: item.SpriteDepth - 0.000015f);
164 }

171 public void ClientEventRead(IReadMessage msg, float sendingTime)
172 {
173 if (correctionTimer > 0.0f)
174 {
175 StartDelayedCorrection(msg.ExtractBits(4 + 8), sendingTime);
176 return;
177 }
178
179 float rechargeRate = msg.ReadRangedInteger(0, 10) / 10.0f;
180 RechargeSpeed = rechargeRate * MaxRechargeSpeed;
181#if CLIENT
182 if (rechargeSpeedSlider != null)
183 {
184 rechargeSpeedSlider.BarScroll = rechargeRate;
185 }
186#endif
187 Charge = msg.ReadRangedSingle(0.0f, 1.0f, 8) * capacity;
188 }
