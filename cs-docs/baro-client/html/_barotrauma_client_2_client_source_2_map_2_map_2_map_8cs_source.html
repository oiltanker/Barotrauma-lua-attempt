210 private void RemoveFogOfWar(Location location, bool removeFromAdjacentLocations = true)
211 {
212 if (location == null) { return; }
213
214 var mapTile = generationParams.MapTiles.Values.FirstOrDefault().FirstOrDefault();
215 if (mapTile == null) { return; }
216
217 Vector2 mapTileSize = mapTile.size * generationParams.MapTileScale;
218 int startX = (int)Math.Max(Math.Floor(location.MapPosition.X / mapTileSize.X - 0.25f), 0);
219 int startY = (int)Math.Max(Math.Floor(location.MapPosition.Y / mapTileSize.Y - 0.25f), 0);
220 int endX = (int)Math.Min(Math.Floor(location.MapPosition.X / mapTileSize.X + 0.25f), mapTiles.GetLength(0) - 1);
221 int endY = (int)Math.Min(Math.Floor(location.MapPosition.Y / mapTileSize.Y + 0.25f), mapTiles.GetLength(1) - 1);
222 for (int x = startX; x <= endX; x++)
223 {
224 for (int y = startY; y <= endY; y++)
225 {
226 tileDiscovered[x, y] = true;
227 }
228 }
229 if (removeFromAdjacentLocations)
230 {
231 foreach (LocationConnection c in location.Connections)
232 {
233 var otherLocation = c.OtherLocation(location);
234 RemoveFogOfWar(otherLocation, removeFromAdjacentLocations: false);
235 }
