43  protected override void StartMissionSpecific(Level level)
44  {
45  existingTargets.Clear();
46  spawnedTargets.Clear();
47  allTargets.Clear();
48  if (IsClient) { return; }
49  TargetRuin = Level.Loaded?.Ruins?.GetRandom(randSync: Rand.RandSync.ServerAndClient);
50  if (TargetRuin == null)
51  {
52  DebugConsole.ThrowError($"Failed to initialize an Alien Ruin mission (\"{Prefab.Identifier}\"): level contains no alien ruins");
53  return;
54  }
55  if (targetItemIdentifiers.Length < 1 && targetEnemyIdentifiers.Length < 1)
56  {
57  DebugConsole.ThrowError($"Failed to initialize an Alien Ruin mission (\"{Prefab.Identifier}\"): no target identifiers set in the mission definition");
58  return;
59  }
60  foreach (var item in Item.ItemList)
61  {
62  if (!targetItemIdentifiers.Contains(item.Prefab.Identifier)) { continue; }
63  if (item.Submarine != TargetRuin.Submarine) { continue; }
64  existingTargets.Add(item);
65  allTargets.Add(item);
66  }
67  int existingEnemyCount = 0;
68  foreach (var character in Character.CharacterList)
69  {
70  if (character.SpeciesName.IsEmpty) { continue; }
71  if (!targetEnemyIdentifiers.Contains(character.SpeciesName)) { continue; }
72  if (character.Submarine != TargetRuin.Submarine) { continue; }
73  existingTargets.Add(character);
74  allTargets.Add(character);
75  existingEnemyCount++;
76  }
77  if (existingEnemyCount < minEnemyCount)
78  {
79  var enemyPrefabs = new HashSet<CharacterPrefab>();
80  foreach (Identifier identifier in targetEnemyIdentifiers)
81  {
82  var prefab = CharacterPrefab.FindBySpeciesName(identifier);
83  if (prefab != null)
84  {
85  enemyPrefabs.Add(prefab);
86  }
87  else
88  {
89  DebugConsole.ThrowError($"Error in an Alien Ruin mission (\"{Prefab.Identifier}\"): could not find a character prefab with the species \"{identifier}\"");
90  }
91  }
92  if (enemyPrefabs.None())
93  {
94  DebugConsole.ThrowError($"Error in an Alien Ruin mission (\"{Prefab.Identifier}\"): no enemy species defined that could be used to spawn more ({minEnemyCount - existingEnemyCount}) enemies");
95  return;
96  }
97  for (int i = 0; i < (minEnemyCount - existingEnemyCount); i++)
98  {
99  var prefab = enemyPrefabs.GetRandomUnsynced();
100  var spawnPos = TargetRuin.Submarine.GetWaypoints(false).GetRandomUnsynced(w => w.CurrentHull != null)?.WorldPosition;
101  if (!spawnPos.HasValue)
102  {
103  DebugConsole.ThrowError($"Error in an Alien Ruin mission (\"{Prefab.Identifier}\"): no valid spawn positions could be found for the additional ({minEnemyCount - existingEnemyCount}) enemies to be spawned");
104  return;
105  }
106  var newEnemy = Character.Create(prefab.Identifier, spawnPos.Value, ToolBox.RandomSeed(8), createNetworkEvent: false);
107  spawnedTargets.Add(newEnemy);
108  allTargets.Add(newEnemy);
109  }
110  }
111 #if DEBUG
112  DebugConsole.NewMessage("********** CLEAR RUIN MISSION INFO **********");
113  DebugConsole.NewMessage($"Existing item targets: {existingTargets.Count - existingEnemyCount}");
114  DebugConsole.NewMessage($"Existing enemy targets: {existingEnemyCount}");
115  DebugConsole.NewMessage($"Spawned enemy targets: {spawnedTargets.Count}");
116 #endif
117  }
