554  public Effect GetActiveEffect(float currentStrength)
555  {
556  foreach (Effect effect in effects)
557  {
558  if (currentStrength > effect.MinStrength && currentStrength <= effect.MaxStrength)
559  {
560  return effect;
561  }
562  }

564  //if above the strength range of all effects, use the highest strength effect
565  Effect strongestEffect = null;
566  float largestStrength = currentStrength;
567  foreach (Effect effect in effects)
568  {
569  if (currentStrength > effect.MaxStrength &&
570  (strongestEffect == null || effect.MaxStrength > largestStrength))
