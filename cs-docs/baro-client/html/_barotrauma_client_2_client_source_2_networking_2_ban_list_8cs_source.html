144 public void ClientAdminRead(IReadMessage incMsg)
145 {
146 bool hasPermission = incMsg.ReadBoolean();
147 if (!hasPermission)
148 {
149 incMsg.ReadPadBits();
150 return;
151 }
152
153 bool isOwner = incMsg.ReadBoolean();
154 incMsg.ReadPadBits();
155
156 bannedPlayers.Clear();
157 UInt32 bannedPlayerCount = incMsg.ReadVariableUInt32();
158
159 for (int i = 0; i < (int)bannedPlayerCount; i++)
160 {
161 string name = incMsg.ReadString();
162 UInt16 uniqueIdentifier = incMsg.ReadUInt16();
163 bool isRangeBan = incMsg.ReadBoolean();
164 bool includesExpiration = incMsg.ReadBoolean();
165 incMsg.ReadPadBits();
166
167 DateTime? expiration = null;
168 if (includesExpiration)
169 {
170 double hoursFromNow = incMsg.ReadDouble();
171 expiration = DateTime.Now + TimeSpan.FromHours(hoursFromNow);
172 }
173
174 string reason = incMsg.ReadString();
175
176 string endPoint = "";
177 UInt64 steamID = 0;
178 if (isOwner)
179 {
180 endPoint = incMsg.ReadString();
181 steamID = incMsg.ReadUInt64();
182 }
183 else
184 {
185 endPoint = "Endpoint concealed by host";
186 steamID = 0;
187 }
188 bannedPlayers.Add(new BannedPlayer(name, uniqueIdentifier, isRangeBan, endPoint, steamID, reason, expiration));
189 }
190
191 if (banFrame != null)
192 {
193 var parent = banFrame.Parent;
194 parent.RemoveChild(banFrame);
195 CreateBanFrame(parent);
196 }
197 }

199 public void ClientAdminWrite(IWriteMessage outMsg)
200 {
201 outMsg.Write((UInt16)localRemovedBans.Count);
202 foreach (UInt16 uniqueId in localRemovedBans)
203 {
204 outMsg.Write(uniqueId);
205 }
206
207 outMsg.Write((UInt16)localRangeBans.Count);
208 foreach (UInt16 uniqueId in localRangeBans)
209 {
210 outMsg.Write(uniqueId);
211 }
212
213 localRemovedBans.Clear();
214 localRangeBans.Clear();
215 }
216 }
