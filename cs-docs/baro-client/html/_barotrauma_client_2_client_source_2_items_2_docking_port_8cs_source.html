110 public void ClientEventRead(IReadMessage msg, float sendingTime)
111 {
112 bool isDocked = msg.ReadBoolean();
113
114 for (int i = 0; i < 2; i++)
115 {
116 if (hulls[i] == null) continue;
117 item.linkedTo.Remove(hulls[i]);
118 hulls[i].Remove();
119 hulls[i] = null;
120 }
121
122 if (gap != null)
123 {
124 item.linkedTo.Remove(gap);
125 gap.Remove();
126 gap = null;
127 }
128
129 if (isDocked)
130 {
131 ushort dockingTargetID = msg.ReadUInt16();
132
133 bool isLocked = msg.ReadBoolean();
134
135 Entity targetEntity = Entity.FindEntityByID(dockingTargetID);
136 if (targetEntity == null || !(targetEntity is Item))
137 {
138 DebugConsole.ThrowError("Invalid docking port network event (can't dock to " + (targetEntity?.ToString() ?? "null") + ")");
139 return;
140 }
141
142 DockingTarget = (targetEntity as Item).GetComponent<DockingPort>();
143 if (DockingTarget == null)
144 {
145 DebugConsole.ThrowError("Invalid docking port network event (" + targetEntity + " doesn't have a docking port component)");
146 return;
147 }
148
149 Dock(DockingTarget);
150 if (joint == null)
151 {
152 string errorMsg = "Error while reading a docking port network event (Dock method did not create a joint between the ports)." +
153 " Submarine: " + (item.Submarine?.Info.Name ?? "null") +
154 ", target submarine: " + (DockingTarget.item.Submarine?.Info.Name ?? "null");
155 if (item.Submarine?.ConnectedDockingPorts.ContainsKey(DockingTarget.item.Submarine) ?? false)
156 {
157 errorMsg += "\nAlready docked.";
158 }
159 if (item.Submarine == DockingTarget.item.Submarine)
160 {
161 errorMsg += "\nTrying to dock the submarine to itself.";
162 }
163 GameAnalyticsManager.AddErrorEventOnce("DockingPort.ClientRead:JointNotCreated", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
164 }
165
166 if (isLocked)
167 {
168 if (DockingTarget.joint != null)
169 {
170 DockingTarget.Lock(isNetworkMessage: true);
171 }
172 else
173 {
174 Lock(isNetworkMessage: true);
175 }
176 }
177 }
178 else
179 {
180 Undock();
181 }
182 }
