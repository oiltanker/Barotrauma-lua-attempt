39 {
40 if (isFinished) { return; }
41
42 affectedNpcs = ParentEvent.GetTargets(NPCTag).Where(e => e is Character).Select(e => e as Character);
43
44 foreach (var npc in affectedNpcs)
45 {
46 if (!(npc.AIController is HumanAIController humanAiController)) { continue; }
47
48 Character enemy = null;
49 float closestDist = float.MaxValue;
50 foreach (Entity target in ParentEvent.GetTargets(EnemyTag))
51 {
52 if (!(target is Character character)) { continue; }
53 float dist = Vector2.DistanceSquared(npc.WorldPosition, target.WorldPosition);
54 if (dist < closestDist)
55 {
56 enemy = character;
57 closestDist = dist;
58 }
59 }
60 if (enemy == null) { continue; }
61
62 npc.CombatAction = this;
63
64 var objectiveManager = humanAiController.ObjectiveManager;
65 foreach (var goToObjective in objectiveManager.GetActiveObjectives<AIObjectiveGoTo>())
66 {
67 goToObjective.Abandon = true;
68 }
69 objectiveManager.AddObjective(new AIObjectiveCombat(npc, enemy, CombatMode, objectiveManager, coolDown: CoolDown));
70 }
71 isFinished = true;
