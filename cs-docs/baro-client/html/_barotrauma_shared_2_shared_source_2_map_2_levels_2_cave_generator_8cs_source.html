38  int y = (int)(Math.Floor((site.Coord.Y - borders.Y) / gridCellSize));

41  y = MathHelper.Clamp(y, 0, cellGrid.GetLength(1) - 1);

124  }

158  public static List<VoronoiCell> GeneratePath(List<VoronoiCell> targetCells, List<VoronoiCell> cells)
159  {
160  Stopwatch sw2 = new Stopwatch();
161  sw2.Start();
162 
163  List<VoronoiCell> pathCells = new List<VoronoiCell>();
164 
165  if (targetCells.Count == 0) { return pathCells; }
166 
167  VoronoiCell currentCell = targetCells[0];
168  currentCell.CellType = CellType.Path;
169  pathCells.Add(currentCell);
170 
171  int currentTargetIndex = 0;
172 
173  int iterationsLeft = cells.Count / 2;
174 
175  do
176  {
177  int edgeIndex = 0;
178 
179  double smallestDist = double.PositiveInfinity;
180  for (int i = 0; i < currentCell.Edges.Count; i++)
181  {
182  var adjacentCell = currentCell.Edges[i].AdjacentCell(currentCell);
183  if (adjacentCell == null) { continue; }
184  double dist = MathUtils.Distance(adjacentCell.Site.Coord.X, adjacentCell.Site.Coord.Y, targetCells[currentTargetIndex].Site.Coord.X, targetCells[currentTargetIndex].Site.Coord.Y);
185  dist += MathUtils.Distance(adjacentCell.Site.Coord.X, adjacentCell.Site.Coord.Y, currentCell.Site.Coord.X, currentCell.Site.Coord.Y) * 0.5f;
186 
187  //disfavor short edges to prevent generating a very small passage
188  if (Vector2.DistanceSquared(currentCell.Edges[i].Point1, currentCell.Edges[i].Point2) < 150.0f * 150.0f)
189  {
190  //divide by the number of times the current cell has been used
191  // prevents the path from getting "stuck" (jumping back and forth between adjacent cells)
192  // if there's no other way to the destination than going through a short edge
193  dist *= 10.0f / Math.Max(pathCells.Count(c => c == currentCell), 1.0f);
194  }
195  if (dist < smallestDist)
196  {
197  edgeIndex = i;
198  smallestDist = dist;
199  }
200  }
201 
202  currentCell = currentCell.Edges[edgeIndex].AdjacentCell(currentCell);
203  currentCell.CellType = CellType.Path;
204  pathCells.Add(currentCell);
205 
206  iterationsLeft--;
207 
208  if (currentCell == targetCells[currentTargetIndex])
209  {
210  currentTargetIndex++;
211  if (currentTargetIndex >= targetCells.Count) { break; }
212  }
213 
214  } while (currentCell != targetCells[targetCells.Count - 1] && iterationsLeft > 0);
215 
216 
217  Debug.WriteLine("gettooclose: " + sw2.ElapsedMilliseconds + " ms");
218  sw2.Restart();
219 
220  return pathCells;
221  }

227  public static void RoundCell(VoronoiCell cell, float minEdgeLength = 500.0f, float roundingAmount = 0.5f, float irregularity = 0.1f)
228  {
229  List<GraphEdge> tempEdges = new List<GraphEdge>();
230  foreach (GraphEdge edge in cell.Edges)
231  {
232  if (!edge.IsSolid || edge.OutsideLevel)
233  {
234  tempEdges.Add(edge);
235  continue;
236  }
237 
238  //If the edge is next to an empty cell and there's another solid cell at the other side of the empty one,
239  //don't touch this edge. Otherwise we may end up closing off small passages between cells.
240  var adjacentEmptyCell = edge.AdjacentCell(cell);
241  if (adjacentEmptyCell?.CellType == CellType.Solid) { adjacentEmptyCell = null; }
242  if (adjacentEmptyCell != null)
243  {
244  GraphEdge adjacentEdge = null;
245  //find the edge at the opposite side of the adjacent cell
246  foreach (GraphEdge otherEdge in adjacentEmptyCell.Edges)
247  {
248  if (Vector2.Dot(adjacentEmptyCell.Center - edge.Center, adjacentEmptyCell.Center - otherEdge.Center) < 0 &&
249  otherEdge.AdjacentCell(adjacentEmptyCell)?.CellType == CellType.Solid)
250  {
251  adjacentEdge = otherEdge;
252  break;
253  }
254  }
255  if (adjacentEdge != null)
256  {
257  tempEdges.Add(edge);
258  continue;
259  }
260  }
261 
262  List<Vector2> edgePoints = new List<Vector2>();
263  Vector2 edgeNormal = edge.GetNormal(cell);
264 
265  float edgeLength = Vector2.Distance(edge.Point1, edge.Point2);
266  int pointCount = (int)Math.Max(Math.Ceiling(edgeLength / minEdgeLength), 1);
267  Vector2 edgeDir = edge.Point2 - edge.Point1;
268  for (int i = 0; i <= pointCount; i++)
269  {
270  if (i == 0)
271  {
272  edgePoints.Add(edge.Point1);
273  }
274  else if (i == pointCount)
275  {
276  edgePoints.Add(edge.Point2);
277  }
278  else
279  {
280 
281  float centerF = 0.5f - Math.Abs(0.5f - (i / (float)pointCount));
282  float randomVariance = Rand.Range(0, irregularity, Rand.RandSync.ServerAndClient);
283  Vector2 extrudedPoint =
284  edge.Point1 +
285  edgeDir * (i / (float)pointCount) +
286  edgeNormal * edgeLength * (roundingAmount + randomVariance) * centerF;
287 
288  var nearbyCells = Level.Loaded.GetCells(extrudedPoint, searchDepth: 2);
289  bool isInside = false;
290  foreach (var nearbyCell in nearbyCells)
291  {
292  if (nearbyCell == cell || nearbyCell.CellType != CellType.Solid) { continue; }
293  //check if extruding the edge causes it to go inside another one
294  if (nearbyCell.IsPointInside(extrudedPoint))
295  {
296  isInside = true;
297  break;
298  }
299  //check if another edge will be inside this cell after the extrusion
300  Vector2 triangleCenter = (edge.Point1 + edge.Point2 + extrudedPoint) / 3;
301  foreach (GraphEdge nearbyEdge in nearbyCell.Edges)
302  {
303  if (!MathUtils.LinesIntersect(nearbyEdge.Point1, triangleCenter, edge.Point1, extrudedPoint) &&
304  !MathUtils.LinesIntersect(nearbyEdge.Point1, triangleCenter, edge.Point2, extrudedPoint) &&
305  !MathUtils.LinesIntersect(nearbyEdge.Point1, triangleCenter, edge.Point1, edge.Point2))
306  {
307  isInside = true;
308  break;
309  }
310  }
311  if (isInside) { break; }
312  }
313 
314  if (!isInside)
315  {
316  edgePoints.Add(extrudedPoint);
317  }
318  }
319  }
320 
321  for (int i = 0; i < edgePoints.Count - 1; i++)
322  {
323  tempEdges.Add(new GraphEdge(edgePoints[i], edgePoints[i + 1])
324  {
325  Cell1 = edge.Cell1,
326  Cell2 = edge.Cell2,
327  IsSolid = edge.IsSolid,
328  Site1 = edge.Site1,
329  Site2 = edge.Site2,
330  OutsideLevel = edge.OutsideLevel,
331  NextToCave = edge.NextToCave,
332  NextToMainPath = edge.NextToMainPath,
333  NextToSidePath = edge.NextToSidePath
334  });
335  }
336  }
337 
338  cell.Edges = tempEdges;
339  }

341  public static Body GeneratePolygons(List<VoronoiCell> cells, Level level, out List<Vector2[]> renderTriangles)
342  {
343  renderTriangles = new List<Vector2[]>();
344 
345  List<Vector2> tempVertices = new List<Vector2>();
346  List<Vector2> bodyPoints = new List<Vector2>();
347 
348  Body cellBody = new Body()
349  {
350  SleepingAllowed = false,
351  BodyType = BodyType.Static,
352  CollisionCategories = Physics.CollisionLevel
353  };
354  GameMain.World.Add(cellBody);
355 
356  for (int n = cells.Count - 1; n >= 0; n-- )
357  {
358  VoronoiCell cell = cells[n];
359 
360  bodyPoints.Clear();
361  tempVertices.Clear();
362  foreach (GraphEdge ge in cell.Edges)
363  {
364  if (Vector2.DistanceSquared(ge.Point1, ge.Point2) < 0.01f) continue;
365  if (!tempVertices.Any(v => Vector2.DistanceSquared(ge.Point1, v) < 1.0f))
366  {
367  tempVertices.Add(ge.Point1);
368  bodyPoints.Add(ge.Point1);
369  }
370  if (!tempVertices.Any(v => Vector2.DistanceSquared(ge.Point2, v) < 1.0f))
371  {
372  tempVertices.Add(ge.Point2);
373  bodyPoints.Add(ge.Point2);
374  }
375  }
376 
377  if (tempVertices.Count < 3 || bodyPoints.Count < 2)
378  {
379  cells.RemoveAt(n);
380  continue;
381  }
382 
383  Vector2 minVert = tempVertices[0];
384  Vector2 maxVert = tempVertices[0];
385  foreach (var vert in tempVertices)
386  {
387  minVert = new Vector2(
388  Math.Min(minVert.X, vert.X),
389  Math.Min(minVert.Y, vert.Y));
390  maxVert = new Vector2(
391  Math.Max(maxVert.X, vert.X),
392  Math.Max(maxVert.Y, vert.Y));
393  }
394  Vector2 center = (minVert + maxVert) / 2;
395  renderTriangles.AddRange(MathUtils.TriangulateConvexHull(tempVertices, center));
396 
397  if (bodyPoints.Count < 2) { continue; }
398 
399  if (bodyPoints.Count < 3)
400  {
401  foreach (Vector2 vertex in tempVertices)
402  {
403  if (bodyPoints.Contains(vertex)) { continue; }
404  bodyPoints.Add(vertex);
405  break;
406  }
407  }
408 
409  for (int i = 0; i < bodyPoints.Count; i++)
410  {
411  cell.BodyVertices.Add(bodyPoints[i]);
412  bodyPoints[i] = ConvertUnits.ToSimUnits(bodyPoints[i]);
413  }
414 
415  if (cell.CellType == CellType.Empty) { continue; }
416 
417  cellBody.UserData = cell;
418  var triangles = MathUtils.TriangulateConvexHull(bodyPoints, ConvertUnits.ToSimUnits(center));
419 
420  for (int i = 0; i < triangles.Count; i++)
421  {
422  //don't create a triangle if the area of the triangle is too small
423  //(apparently Farseer doesn't like polygons with a very small area, see Shape.ComputeProperties)
424  Vector2 a = triangles[i][0];
425  Vector2 b = triangles[i][1];
426  Vector2 c = triangles[i][2];
427  float area = Math.Abs(a.X * (b.Y - c.Y) + b.X * (c.Y - a.Y) + c.X * (a.Y - b.Y)) / 2.0f;
428  if (area < 1.0f) { continue; }
429 
430  Vertices bodyVertices = new Vertices(triangles[i]);
431  PolygonShape polygon = new PolygonShape(bodyVertices, 5.0f);
432  Fixture fixture = new Fixture(polygon)
433  {
434  UserData = cell
435  };
436  cellBody.Add(fixture, resetMassData: false);
437 
438  if (fixture.Shape.MassData.Area < FarseerPhysics.Settings.Epsilon)
439  {
440  DebugConsole.ThrowError("Invalid triangle created by CaveGenerator (" + triangles[i][0] + ", " + triangles[i][1] + ", " + triangles[i][2] + ")");
441  GameAnalyticsManager.AddErrorEventOnce(
442  "CaveGenerator.GeneratePolygons:InvalidTriangle",
443  GameAnalyticsManager.ErrorSeverity.Warning,
444  "Invalid triangle created by CaveGenerator (" + triangles[i][0] + ", " + triangles[i][1] + ", " + triangles[i][2] + "). Seed: " + level.Seed);
445  }
446  }
447  cell.Body = cellBody;
448  }
449 
450  cellBody.CollisionCategories = Physics.CollisionLevel;
451  cellBody.ResetMassData();
452 
453  return cellBody;
454  }

461  public static List<Vector2> CreateRandomChunk(float width, float height, int vertexCount, float radiusVariance)
462  {
463  Debug.Assert(radiusVariance < Math.Min(width, height));
464  Debug.Assert(vertexCount >= 3);
465 
466  List<Vector2> verts = new List<Vector2>();
467  float angleStep = MathHelper.TwoPi / vertexCount;
