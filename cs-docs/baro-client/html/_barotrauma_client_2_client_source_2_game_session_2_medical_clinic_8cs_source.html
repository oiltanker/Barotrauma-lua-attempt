124 {

126 UpdateQueue(afflictionRequests, now, onTimeout: callback => { callback(new AfflictionRequest(RequestResult.Timeout, ImmutableArray<NetAffliction>.Empty)); });

128 UpdateQueue(healAllRequests, now, onTimeout: callback => { callback(new HealRequest(RequestResult.Timeout, HealRequestResult.Unknown)); });

148 private static bool TryDequeue<T>(List<RequestAction<T>> requestQueue, out Action<T> result)
149 {
150 RequestAction<T>? first = requestQueue.FirstOrNull();
151 if (!(first is { } action))
152 {
153 result = _ => { };
154 return false;
155 }
156
157 requestQueue.Remove(action);
158 result = action.Callback;
159 return true;
160 }
161
162 private static void UpdateQueue<T>(List<RequestAction<T>> requestQueue, DateTimeOffset now, Action<Action<T>> onTimeout)
163 {
164 HashSet<RequestAction<T>>? removals = null;
165 foreach (RequestAction<T> action in requestQueue)
166 {
167 if (action.Timeout < now)
168 {
169 onTimeout.Invoke(action.Callback);
170
171 removals ??= new HashSet<RequestAction<T>>();
172 removals.Add(action);
173 }
174 }
175
176 if (removals is null) { return; }
177
178 foreach (RequestAction<T> action in removals)
179 {
180 requestQueue.Remove(action);
181 }
182 }

369 public void ClientRead(IReadMessage inc)
370 {
371 NetworkHeader header = (NetworkHeader)inc.ReadByte();
372 MessageFlag flag = (MessageFlag)inc.ReadByte();
373
374 switch (header)
375 {
376 case NetworkHeader.REQUEST_AFFLICTIONS:
377 AfflictionRequestReceived(inc);
378 break;
379 case NetworkHeader.REQUEST_PENDING:
380 PendingRequestReceived(inc);
381 break;
382 case NetworkHeader.ADD_PENDING:
383 NewAdditonReceived(inc, flag);
384 break;
385 case NetworkHeader.REMOVE_PENDING:
386 NewRemovalReceived(inc, flag);
387 break;
388 case NetworkHeader.HEAL_PENDING:
389 HealRequestReceived(inc);
390 break;
391 case NetworkHeader.CLEAR_PENDING:
392 ClearRequstReceived();
393 break;
394 }
395 }
