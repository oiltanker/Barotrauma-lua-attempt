65 public bool AddOrder(Order order, float? fadeOutTime)
66 {
67 if (order.TargetEntity == null)
68 {
69 string message = $"Attempted to add a \"{order.Name}\" order with no target entity to CrewManager!\n{Environment.StackTrace.CleanupStackTrace()}";
70 DebugConsole.AddWarning(message);
71 GameAnalyticsManager.AddErrorEventOnce("CrewManager.AddOrder:OrderTargetEntityNull", GameAnalyticsManager.ErrorSeverity.Error, message);
72 return false;
73 }
74
75 // Ignore orders work a bit differently since the "unignore" order counters the "ignore" order
76 var isUnignoreOrder = order.Identifier == "unignorethis";
77 var orderPrefab = !isUnignoreOrder ? order.Prefab : OrderPrefab.Prefabs["ignorethis"];
78 ActiveOrder existingOrder = ActiveOrders.Find(o =>
79 o.Order.Prefab == orderPrefab && MatchesTarget(o.Order.TargetEntity, order.TargetEntity) &&
80 (o.Order.TargetType != Order.OrderTargetType.WallSection || o.Order.WallSectionIndex == order.WallSectionIndex));
81
82 if (existingOrder != null)
83 {
84 if (!isUnignoreOrder)
85 {
86 existingOrder.FadeOutTime = fadeOutTime;
87 return false;
88 }
89 else
90 {
91 ActiveOrders.Remove(existingOrder);
92 return true;
93 }
94 }
95 else if (!isUnignoreOrder)
96 {
97 ActiveOrders.Add(new ActiveOrder(order, fadeOutTime));
98#if CLIENT
99 HintManager.OnActiveOrderAdded(order);
100#endif
101 return true;
102 }

104 static bool MatchesTarget(Entity existingTarget, Entity newTarget)
105 {
106 if (existingTarget == newTarget) { return true; }
107 if (existingTarget is Hull existingHullTarget && newTarget is Hull newHullTarget)
108 {
109 return existingHullTarget.linkedTo.Contains(newHullTarget);
110 }
111 return false;
112 }
113
114 return false;
115 }

211 public void InitRound()
212 {
213#if CLIENT
214 GUIContextMenu.CurrentContextMenu = null;
215#endif
216
217 characters.Clear();
218
219 List<WayPoint> spawnWaypoints = null;
220 List<WayPoint> mainSubWaypoints = WayPoint.SelectCrewSpawnPoints(characterInfos, Submarine.MainSub).ToList();
221
222 if (Level.IsLoadedOutpost && Submarine.Loaded.Any(s => s.Info.Type == SubmarineType.Outpost && (s.Info.OutpostGenerationParams?.SpawnCrewInsideOutpost ?? false)))
223 {
224 spawnWaypoints = WayPoint.WayPointList.FindAll(wp =>
225 wp.SpawnType == SpawnType.Human &&
226 wp.Submarine == Level.Loaded.StartOutpost &&
227 wp.CurrentHull != null &&
228 wp.CurrentHull.OutpostModuleTags.Contains("airlock".ToIdentifier()));
229 while (spawnWaypoints.Count > characterInfos.Count)
230 {
231 spawnWaypoints.RemoveAt(Rand.Int(spawnWaypoints.Count));
232 }
233 while (spawnWaypoints.Any() && spawnWaypoints.Count < characterInfos.Count)
234 {
235 spawnWaypoints.Add(spawnWaypoints[Rand.Int(spawnWaypoints.Count)]);
236 }
237 }
238
239 if (spawnWaypoints == null || !spawnWaypoints.Any())
240 {
241 spawnWaypoints = mainSubWaypoints;
242 }
243
244 System.Diagnostics.Debug.Assert(spawnWaypoints.Count == mainSubWaypoints.Count);
245
246 for (int i = 0; i < spawnWaypoints.Count; i++)
247 {
248 var info = characterInfos[i];
249 info.TeamID = CharacterTeamType.Team1;
250 Character character = Character.Create(info, spawnWaypoints[i].WorldPosition, info.Name);
251 if (character.Info != null)
252 {
253 if (!character.Info.StartItemsGiven && character.Info.InventoryData != null)
254 {
255 DebugConsole.AddWarning($"Error when initializing a round: character \"{character.Name}\" has not been given their initial items but has saved inventory data. Using the saved inventory data instead of giving the character new items.");
256 }
257 if (character.Info.InventoryData != null)
258 {
259 character.SpawnInventoryItems(character.Inventory, character.Info.InventoryData.FromPackage(null));
260 }
261 else if (!character.Info.StartItemsGiven)
262 {
263 character.GiveJobItems(mainSubWaypoints[i]);
264 }
265 if (character.Info.HealthData != null)
266 {
267 CharacterInfo.ApplyHealthData(character, character.Info.HealthData);
268 }
269
270 character.LoadTalents();
271
272 character.GiveIdCardTags(spawnWaypoints[i]);
273 character.Info.StartItemsGiven = true;
274 if (character.Info.OrderData != null)
275 {
276 character.Info.ApplyOrderData();
277 }
278 }
279
280 AddCharacter(character, sortCrewList: false);
281#if CLIENT
282 if (IsSinglePlayer && (Character.Controlled == null || character.Info.LastControlled)) { Character.Controlled = character; }
283#endif
284 }
285
286#if CLIENT
287 if (IsSinglePlayer) { SortCrewList(); }
288#endif

350 private void UpdateConversations(float deltaTime)
351 {
352 if (GameMain.GameSession?.GameMode?.Preset == GameModePreset.TestMode) { return; }
353 if (GameMain.NetworkMember != null && GameMain.NetworkMember.ServerSettings.DisableBotConversations) { return; }
354
355 conversationTimer -= deltaTime;
356 if (conversationTimer <= 0.0f)
357 {
358 CreateRandomConversation();
359 conversationTimer = Rand.Range(ConversationIntervalMin, ConversationIntervalMax);
360 if (GameMain.NetworkMember != null)
361 {
362 conversationTimer *= ConversationIntervalMultiplierMultiplayer;
363 }
364 }
365
366 if (welcomeMessageNPC == null)
367 {
368 foreach (Character npc in Character.CharacterList)
369 {
370 if ((npc.TeamID != CharacterTeamType.FriendlyNPC && npc.TeamID != CharacterTeamType.None) || npc.CurrentHull == null || npc.IsIncapacitated) { continue; }
371 if (npc.AIController is HumanAIController humanAI && (humanAI.ObjectiveManager.IsCurrentObjective<AIObjectiveFindSafety>() || humanAI.ObjectiveManager.IsCurrentObjective<AIObjectiveCombat>()))
372 {
373 continue;
374 }
375 foreach (Character player in Character.CharacterList)
376 {
377 if (player.TeamID != npc.TeamID && !player.IsIncapacitated && player.CurrentHull == npc.CurrentHull)
378 {
379 List<Character> availableSpeakers = new List<Character>() { npc, player };
380 List<Identifier> dialogFlags = new List<Identifier>() { "OutpostNPC".ToIdentifier(), "EnterOutpost".ToIdentifier() };
381 if (GameMain.GameSession?.GameMode is CampaignMode campaignMode)
382 {
383 if (campaignMode.Map?.CurrentLocation?.Type?.Identifier == "abandoned")
384 {
385 if (npc.TeamID == CharacterTeamType.None)
386 {
387 dialogFlags.Remove("OutpostNPC".ToIdentifier());
388 dialogFlags.Add("Bandit".ToIdentifier());
389 }
390 else if (npc.TeamID == CharacterTeamType.FriendlyNPC)
391 {
392 dialogFlags.Remove("OutpostNPC".ToIdentifier());
393 dialogFlags.Add("Hostage".ToIdentifier());
394 }
395 }
396 else if (campaignMode.Map?.CurrentLocation?.Reputation != null)
397 {
398 float normalizedReputation = MathUtils.InverseLerp(
399 campaignMode.Map.CurrentLocation.Reputation.MinReputation,
400 campaignMode.Map.CurrentLocation.Reputation.MaxReputation,
401 campaignMode.Map.CurrentLocation.Reputation.Value);
402 if (normalizedReputation < 0.2f)
403 {
404 dialogFlags.Add("LowReputation".ToIdentifier());
405 }
406 else if (normalizedReputation > 0.8f)
407 {
408 dialogFlags.Add("HighReputation".ToIdentifier());
409 }
410 }
411 }
412 pendingConversationLines.AddRange(NPCConversation.CreateRandom(availableSpeakers, dialogFlags));
413 welcomeMessageNPC = npc;
414 break;
415 }
416 }
417 if (welcomeMessageNPC != null) { break; }
418 }
419 }
420 else if (welcomeMessageNPC.Removed)
421 {
422 welcomeMessageNPC = null;
423 }
424
425 if (pendingConversationLines.Count > 0)
426 {
427 conversationLineTimer -= deltaTime;
428 if (conversationLineTimer <= 0.0f)
429 {
430 //speaker of the next line can't speak, interrupt the conversation
431 if (pendingConversationLines[0].speaker.SpeechImpediment >= 100.0f)
432 {
433 pendingConversationLines.Clear();
434 return;
435 }
436
437 pendingConversationLines[0].speaker.Speak(pendingConversationLines[0].line, null);
438 if (pendingConversationLines.Count > 1)
439 {
440 conversationLineTimer = MathHelper.Clamp(pendingConversationLines[0].line.Length * 0.1f, 1.0f, 5.0f);
441 }
442 pendingConversationLines.RemoveAt(0);
443 }
444 }

449 public static Character GetCharacterForQuickAssignment(Order order, Character controlledCharacter, IEnumerable<Character> characters, bool includeSelf = false)
450 {
451 bool isControlledCharacterNull = controlledCharacter == null;
452#if !DEBUG
453 if (isControlledCharacterNull) { return null; }
454#endif
455 if (order.Category == OrderCategory.Operate && HumanAIController.IsItemTargetedBySomeone(order.TargetItemComponent, controlledCharacter != null ? controlledCharacter.TeamID : CharacterTeamType.Team1, out Character operatingCharacter) &&
456 (isControlledCharacterNull || operatingCharacter.CanHearCharacter(controlledCharacter)))
457 {
458 return operatingCharacter;
459 }
460 return GetCharactersSortedForOrder(order, characters, controlledCharacter, includeSelf).FirstOrDefault(c => isControlledCharacterNull || c.CanHearCharacter(controlledCharacter)) ?? controlledCharacter;
461 }

463 public static IEnumerable<Character> GetCharactersSortedForOrder(Order order, IEnumerable<Character> characters, Character controlledCharacter, bool includeSelf, IEnumerable<Character> extraCharacters = null)
464 {
465 var filteredCharacters = characters.Where(c => controlledCharacter == null || ((includeSelf || c != controlledCharacter) && c.TeamID == controlledCharacter.TeamID));
466 if (extraCharacters != null)
467 {
468 filteredCharacters = filteredCharacters.Union(extraCharacters);
469 }
470 return filteredCharacters
471 // Prioritize those who are on the same submarine as the controlled character
472 .OrderByDescending(c => Character.Controlled == null || c.Submarine == Character.Controlled.Submarine)
473 // Prioritize those who are already ordered to operate the device
474 .ThenByDescending(c
475 => order.Category == OrderCategory.Operate
476 && c.CurrentOrders.Any(o
477 => o != null
478 && o.Identifier == order.Identifier
479 && o.TargetEntity == order.TargetEntity))
480 // Prioritize those with the appropriate job for the order
481 .ThenByDescending(order.HasAppropriateJob)
482 // Prioritize those who don't yet have the same order (which allows quick-assigning the order to different characters)
483 .ThenByDescending(c => c.CurrentOrders.None(o => o != null && o.Identifier == order.Identifier))
484 // Prioritize those with the preferred job for the order
485 .ThenByDescending(order.HasPreferredJob)
486 // Prioritize bots over player-controlled characters
487 .ThenByDescending(c => c.IsBot)
488 // Prioritize those with a lower current objective priority
489 .ThenBy(c => c.AIController is HumanAIController humanAI ? humanAI.ObjectiveManager.CurrentObjective?.Priority : 0)
490 // Prioritize those with a higher order skill level
491 .ThenByDescending(c => c.GetSkillLevel(order.AppropriateSkill));
492 }

496 private void SaveActiveOrders(XElement parentElement)
497 {
498 ActiveOrdersElement = new XElement("activeorders");
499 // Only save orders with no fade out time (e.g. ignore orders)
500 var ordersToSave = new List<Order>();
501 foreach (var activeOrder in ActiveOrders)
502 {
503 var order = activeOrder?.Order;
504 if (order == null || activeOrder.FadeOutTime.HasValue) { continue; }
505 ordersToSave.Add(order.WithManualPriority(CharacterInfo.HighestManualOrderPriority));
506 }
507 CharacterInfo.SaveOrders(ActiveOrdersElement, ordersToSave.ToArray());
508 parentElement?.Add(ActiveOrdersElement);
509 }

511 public void LoadActiveOrders()
512 {
513 if (ActiveOrdersElement == null) { return; }
514 foreach (var orderInfo in CharacterInfo.LoadOrders(ActiveOrdersElement))
515 {
516 IIgnorable ignoreTarget = null;
517 if (orderInfo.IsIgnoreOrder)
518 {
519 switch (orderInfo.TargetType)
520 {
521 case Order.OrderTargetType.Entity:
522 ignoreTarget = orderInfo.TargetEntity as IIgnorable;
523 break;
524 case Order.OrderTargetType.WallSection when orderInfo.TargetEntity is Structure s && orderInfo.WallSectionIndex.HasValue:
525 ignoreTarget = s.GetSection(orderInfo.WallSectionIndex.Value);
526 break;
527 default:
528 DebugConsole.ThrowError("Error loading an ignore order - can't find a proper ignore target");
529 continue;
530 }
531 }
532 if (orderInfo.TargetEntity == null || (orderInfo.IsIgnoreOrder && ignoreTarget == null))
533 {
534 // The order target doesn't exist anymore, just discard the loaded order
535 continue;
536 }
537 if (ignoreTarget != null)
538 {
539 ignoreTarget.OrderedToBeIgnored = true;
540 }
541 AddOrder(orderInfo, null);
542 }
