350  private void UpdateConversations(float deltaTime)
351  {
352  if (GameMain.GameSession?.GameMode?.Preset == GameModePreset.TestMode) { return; }
353  if (GameMain.NetworkMember != null && GameMain.NetworkMember.ServerSettings.DisableBotConversations) { return; }
354 
355  conversationTimer -= deltaTime;
356  if (conversationTimer <= 0.0f)
357  {
358  CreateRandomConversation();
359  conversationTimer = Rand.Range(ConversationIntervalMin, ConversationIntervalMax);
360  if (GameMain.NetworkMember != null)
361  {
362  conversationTimer *= ConversationIntervalMultiplierMultiplayer;
363  }
364  }
365 
366  if (welcomeMessageNPC == null)
367  {
368  foreach (Character npc in Character.CharacterList)
369  {
370  if ((npc.TeamID != CharacterTeamType.FriendlyNPC && npc.TeamID != CharacterTeamType.None) || npc.CurrentHull == null || npc.IsIncapacitated) { continue; }
371  if (npc.AIController is HumanAIController humanAI && (humanAI.ObjectiveManager.IsCurrentObjective<AIObjectiveFindSafety>() || humanAI.ObjectiveManager.IsCurrentObjective<AIObjectiveCombat>()))
372  {
373  continue;
374  }
375  foreach (Character player in Character.CharacterList)
376  {
377  if (player.TeamID != npc.TeamID && !player.IsIncapacitated && player.CurrentHull == npc.CurrentHull)
378  {
379  List<Character> availableSpeakers = new List<Character>() { npc, player };
380  List<Identifier> dialogFlags = new List<Identifier>() { "OutpostNPC".ToIdentifier(), "EnterOutpost".ToIdentifier() };
381  if (GameMain.GameSession?.GameMode is CampaignMode campaignMode)
382  {
383  if (campaignMode.Map?.CurrentLocation?.Type?.Identifier == "abandoned")
384  {
385  if (npc.TeamID == CharacterTeamType.None)
386  {
387  dialogFlags.Remove("OutpostNPC".ToIdentifier());
388  dialogFlags.Add("Bandit".ToIdentifier());
389  }
390  else if (npc.TeamID == CharacterTeamType.FriendlyNPC)
391  {
392  dialogFlags.Remove("OutpostNPC".ToIdentifier());
393  dialogFlags.Add("Hostage".ToIdentifier());
394  }
395  }
396  else if (campaignMode.Map?.CurrentLocation?.Reputation != null)
397  {
398  float normalizedReputation = MathUtils.InverseLerp(
399  campaignMode.Map.CurrentLocation.Reputation.MinReputation,
400  campaignMode.Map.CurrentLocation.Reputation.MaxReputation,
401  campaignMode.Map.CurrentLocation.Reputation.Value);
402  if (normalizedReputation < 0.2f)
403  {
404  dialogFlags.Add("LowReputation".ToIdentifier());
405  }
406  else if (normalizedReputation > 0.8f)
407  {
408  dialogFlags.Add("HighReputation".ToIdentifier());
409  }
410  }
411  }
412  pendingConversationLines.AddRange(NPCConversation.CreateRandom(availableSpeakers, dialogFlags));
413  welcomeMessageNPC = npc;
414  break;
415  }
416  }
417  if (welcomeMessageNPC != null) { break; }
418  }
419  }
420  else if (welcomeMessageNPC.Removed)
421  {
422  welcomeMessageNPC = null;
423  }
424 
425  if (pendingConversationLines.Count > 0)
426  {
427  conversationLineTimer -= deltaTime;
428  if (conversationLineTimer <= 0.0f)
429  {
430  //speaker of the next line can't speak, interrupt the conversation
431  if (pendingConversationLines[0].speaker.SpeechImpediment >= 100.0f)
432  {
433  pendingConversationLines.Clear();
434  return;
435  }
436 
437  pendingConversationLines[0].speaker.Speak(pendingConversationLines[0].line, null);
438  if (pendingConversationLines.Count > 1)
439  {
440  conversationLineTimer = MathHelper.Clamp(pendingConversationLines[0].line.Length * 0.1f, 1.0f, 5.0f);
441  }
442  pendingConversationLines.RemoveAt(0);
443  }
444  }

463  public static IEnumerable<Character> GetCharactersSortedForOrder(Order order, IEnumerable<Character> characters, Character controlledCharacter, bool includeSelf, IEnumerable<Character> extraCharacters = null)
464  {
465  var filteredCharacters = characters.Where(c => controlledCharacter == null || ((includeSelf || c != controlledCharacter) && c.TeamID == controlledCharacter.TeamID));
466  if (extraCharacters != null)
467  {
468  filteredCharacters = filteredCharacters.Union(extraCharacters);
469  }
470  return filteredCharacters
471  // Prioritize those who are on the same submarine as the controlled character
472  .OrderByDescending(c => Character.Controlled == null || c.Submarine == Character.Controlled.Submarine)
473  // Prioritize those who are already ordered to operate the device
474  .ThenByDescending(c
475  => order.Category == OrderCategory.Operate
476  && c.CurrentOrders.Any(o
477  => o != null
478  && o.Identifier == order.Identifier
479  && o.TargetEntity == order.TargetEntity))
480  // Prioritize those with the appropriate job for the order
481  .ThenByDescending(order.HasAppropriateJob)
482  // Prioritize those who don't yet have the same order (which allows quick-assigning the order to different characters)
483  .ThenByDescending(c => c.CurrentOrders.None(o => o != null && o.Identifier == order.Identifier))
484  // Prioritize those with the preferred job for the order
485  .ThenByDescending(order.HasPreferredJob)
486  // Prioritize bots over player-controlled characters
487  .ThenByDescending(c => c.IsBot)
488  // Prioritize those with a lower current objective priority
489  .ThenBy(c => c.AIController is HumanAIController humanAI ? humanAI.ObjectiveManager.CurrentObjective?.Priority : 0)
490  // Prioritize those with a higher order skill level
491  .ThenByDescending(c => c.GetSkillLevel(order.AppropriateSkill));
492  }

511  public void LoadActiveOrders()
512  {
513  if (ActiveOrdersElement == null) { return; }
514  foreach (var orderInfo in CharacterInfo.LoadOrders(ActiveOrdersElement))
515  {
516  IIgnorable ignoreTarget = null;
517  if (orderInfo.IsIgnoreOrder)
518  {
519  switch (orderInfo.TargetType)
520  {
521  case Order.OrderTargetType.Entity:
522  ignoreTarget = orderInfo.TargetEntity as IIgnorable;
523  break;
524  case Order.OrderTargetType.WallSection when orderInfo.TargetEntity is Structure s && orderInfo.WallSectionIndex.HasValue:
525  ignoreTarget = s.GetSection(orderInfo.WallSectionIndex.Value);
526  break;
527  default:
528  DebugConsole.ThrowError("Error loading an ignore order - can't find a proper ignore target");
529  continue;
530  }
531  }
532  if (orderInfo.TargetEntity == null || (orderInfo.IsIgnoreOrder && ignoreTarget == null))
533  {
534  // The order target doesn't exist anymore, just discard the loaded order
535  continue;
536  }
