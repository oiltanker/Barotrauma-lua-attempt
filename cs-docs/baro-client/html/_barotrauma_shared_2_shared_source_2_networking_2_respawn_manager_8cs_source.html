75 public RespawnManager(NetworkMember networkMember, SubmarineInfo shuttleInfo)
76 : base(null, Entity.RespawnManagerID)
77 {
78 this.networkMember = networkMember;
79
80 if (shuttleInfo != null)
81 {
82 RespawnShuttle = new Submarine(shuttleInfo, true);
83 RespawnShuttle.PhysicsBody.FarseerBody.OnCollision += OnShuttleCollision;
84 //set crush depth slightly deeper than the main sub's
85 RespawnShuttle.RealWorldCrushDepth = Math.Max(RespawnShuttle.RealWorldCrushDepth, Submarine.MainSub.RealWorldCrushDepth * 1.2f);
86
87 //prevent wifi components from communicating between the respawn shuttle and other subs
88 List<WifiComponent> wifiComponents = new List<WifiComponent>();
89 foreach (Item item in Item.ItemList)
90 {
91 if (item.Submarine == RespawnShuttle) { wifiComponents.AddRange(item.GetComponents<WifiComponent>()); }
92 }
93 foreach (WifiComponent wifiComponent in wifiComponents)
94 {
95 wifiComponent.TeamID = CharacterTeamType.FriendlyNPC;
96 }
97
98 ResetShuttle();
99
100 shuttleDoors = new List<Door>();
101 foreach (Item item in Item.ItemList)
102 {
103 if (item.Submarine != RespawnShuttle) continue;
104
105 var steering = item.GetComponent<Steering>();
106 if (steering != null) shuttleSteering = steering;
107
108 var door = item.GetComponent<Door>();
109 if (door != null) shuttleDoors.Add(door);
110
111 //lock all wires to prevent the players from messing up the electronics
112 var connectionPanel = item.GetComponent<ConnectionPanel>();
113 if (connectionPanel != null)
114 {
115 foreach (Connection connection in connectionPanel.Connections)
116 {
117 foreach (Wire wire in connection.Wires)
118 {
119#if SERVER
120 if (GameMain.LuaCs.Game.overrideRespawnSub == false)
121 {
122 if (wire != null) wire.Locked = true;
123 }
124#endif
125 }
126 }
127 }
128 }
129 }
130 else
131 {
132 RespawnShuttle = null;
133 }
134
135#if SERVER
136 if (networkMember is GameServer server)
137 {
138 maxTransportTime = server.ServerSettings.MaxTransportTime;
139 }
140#endif

149 public void Update(float deltaTime)
150 {
151 if (RespawnShuttle == null)
152 {
153 if (CurrentState != State.Waiting)
154 {
155 CurrentState = State.Waiting;

157 }

164 case State.Transporting:

204 private IEnumerable<CoroutineStatus> ForceShuttleToPos(Vector2 position, float speed)
205 {
206
207#if SERVER
208 if (GameMain.LuaCs.Game.overrideRespawnSub)
209 {
210 yield return CoroutineStatus.Success;
211 }
212#endif
213
214 if (RespawnShuttle == null)
215 {
216 yield return CoroutineStatus.Success;
217 }
218
219 while (Math.Abs(position.Y - RespawnShuttle.WorldPosition.Y) > 100.0f)
220 {
221 Vector2 diff = position - RespawnShuttle.WorldPosition;
222 if (diff.LengthSquared() > 0.01f)
223 {
224 Vector2 displayVel = Vector2.Normalize(diff) * speed;
225 RespawnShuttle.SubBody.Body.LinearVelocity = ConvertUnits.ToSimUnits(displayVel);
226 }
227 yield return CoroutineStatus.Running;
228
229 if (RespawnShuttle.SubBody == null) yield return CoroutineStatus.Success;
230 }
231
232 yield return CoroutineStatus.Success;
233 }

235 private void ResetShuttle()
236 {
237 ReturnTime = DateTime.Now + new TimeSpan(0, 0, 0, 0, milliseconds: (int)(maxTransportTime * 1000));
238
239#if SERVER
240 despawnTime = ReturnTime + new TimeSpan(0, 0, seconds: 30);
241#endif
242
243 if (RespawnShuttle == null) return;
244
245 foreach (Item item in Item.ItemList)
246 {
247 if (item.Submarine != RespawnShuttle) { continue; }
248
249 //remove respawn items that have been left in the shuttle
250 if (respawnItems.Contains(item))
251 {
252 Spawner.AddItemToRemoveQueue(item);
253 continue;
254 }
255
256 //restore other items to full condition and recharge batteries
257 item.Condition = item.MaxCondition;
258 item.GetComponent<Repairable>()?.ResetDeterioration();
259 var powerContainer = item.GetComponent<PowerContainer>();
260 if (powerContainer != null)
261 {
262 powerContainer.Charge = powerContainer.Capacity;
263 }
264
265 var door = item.GetComponent<Door>();
266 if (door != null) { door.Stuck = 0.0f; }
267
268 var steering = item.GetComponent<Steering>();
269 if (steering != null)
270 {
271 steering.MaintainPos = true;
272 steering.AutoPilot = true;
273#if SERVER
274 steering.UnsentChanges = true;
275#endif
276 }
277 }
278
279 foreach (Structure wall in Structure.WallList)
280 {
281 if (wall.Submarine != RespawnShuttle) { continue; }
282 for (int i = 0; i < wall.SectionCount; i++)
283 {
284 wall.AddDamage(i, -100000.0f);
285 }
286 }
287
288 foreach (Hull hull in Hull.HullList)
289 {
290 if (hull.Submarine != RespawnShuttle) { continue; }
291 hull.OxygenPercentage = 100.0f;
292 hull.WaterVolume = 0.0f;
293 hull.BallastFlora?.Kill();
294 }
295
296 Dictionary<Character, Vector2> characterPositions = new Dictionary<Character, Vector2>();
297 foreach (Character c in Character.CharacterList)
298 {
299 if (c.Submarine != RespawnShuttle) { continue; }
300 if (!respawnedCharacters.Contains(c))
301 {
302 characterPositions.Add(c, c.WorldPosition);
303 continue;
304 }
305#if CLIENT
306 if (Character.Controlled == c) { Character.Controlled = null; }
307#endif
308 c.Kill(CauseOfDeathType.Unknown, null, true);
309 c.Enabled = false;
310
311 Spawner.AddEntityToRemoveQueue(c);
312 if (c.Inventory != null)
313 {
314 foreach (Item item in c.Inventory.AllItems)
315 {
316 Spawner.AddItemToRemoveQueue(item);
317 }
318 }
319 }
320
321 RespawnShuttle.SetPosition(new Vector2(Level.Loaded.StartPosition.X, Level.Loaded.Size.Y + RespawnShuttle.Borders.Height));
322 RespawnShuttle.Velocity = Vector2.Zero;
323
324 foreach (var characterPosition in characterPositions)
325 {
326 characterPosition.Key.TeleportTo(characterPosition.Value);
327 }

351 public Vector2 FindSpawnPos()
352 {
353 if (Level.Loaded == null || Submarine.MainSub == null) { return Vector2.Zero; }
354
355 Rectangle dockedBorders = RespawnShuttle.GetDockedBorders();
356 Vector2 diffFromDockedBorders =
357 new Vector2(dockedBorders.Center.X, dockedBorders.Y - dockedBorders.Height / 2)
358 - new Vector2(RespawnShuttle.Borders.Center.X, RespawnShuttle.Borders.Y - RespawnShuttle.Borders.Height / 2);
359
360 int minWidth = Math.Max(dockedBorders.Width, 1000);
361 int minHeight = Math.Max(dockedBorders.Height, 1000);
362
363 List<Level.InterestingPosition> potentialSpawnPositions = new List<Level.InterestingPosition>();
364 foreach (Level.InterestingPosition potentialSpawnPos in Level.Loaded.PositionsOfInterest.Where(p => p.PositionType == Level.PositionType.MainPath))
365 {
366 bool invalid = false;
367 //make sure the shuttle won't overlap with any ruins
368 foreach (var ruin in Level.Loaded.Ruins)
369 {
370 if (Math.Abs(ruin.Area.Center.X - potentialSpawnPos.Position.X) < (minWidth + ruin.Area.Width) / 2) { invalid = true; break; }
371 if (Math.Abs(ruin.Area.Center.Y - potentialSpawnPos.Position.Y) < (minHeight + ruin.Area.Height) / 2) { invalid = true; break; }
372 }
373 if (invalid) { continue; }
374
375 //make sure there aren't any walls too close
376 var tooCloseCells = Level.Loaded.GetTooCloseCells(potentialSpawnPos.Position.ToVector2(), Math.Max(minWidth, minHeight));
377 if (tooCloseCells.Any()) { continue; }
378
379 //make sure the spawnpoint is far enough from other subs
380 foreach (Submarine sub in Submarine.Loaded)
381 {
382 if (sub == RespawnShuttle || RespawnShuttle.DockedTo.Contains(sub)) { continue; }
383 float minDist = Math.Max(Math.Max(minWidth, minHeight) + Math.Max(sub.Borders.Width, sub.Borders.Height), 10000.0f);
384 if (Vector2.DistanceSquared(sub.WorldPosition, potentialSpawnPos.Position.ToVector2()) < minDist * minDist)
385 {
386 invalid = true;
387 break;
388 }
389 }
390 if (invalid) { continue; }
391
392 foreach (Character character in Character.CharacterList)
393 {
394 if (character.IsDead)
395 {
396 //cannot spawn directly over dead bodies
397 if (Math.Abs(character.WorldPosition.X - potentialSpawnPos.Position.X) < minWidth) { invalid = true; break; }
398 if (Math.Abs(character.WorldPosition.Y - potentialSpawnPos.Position.Y) < minHeight) { invalid = true; break; }
399 }
400 else
401 {
402 //cannot spawn near alive characters (to prevent other players from seeing the shuttle
403 //appear out of nowhere, or monsters from immediatelly wrecking the shuttle)
404 if (Vector2.DistanceSquared(character.WorldPosition, potentialSpawnPos.Position.ToVector2()) < 5000.0f * 5000.0f)
405 {
406 invalid = true;
407 break;
408 }
409 }
410 }
411 if (invalid) { continue; }
412
413 potentialSpawnPositions.Add(potentialSpawnPos);
414 }
415 Vector2 bestSpawnPos = new Vector2(Level.Loaded.StartPosition.X, Level.Loaded.Size.Y + RespawnShuttle.Borders.Height);
416 float bestSpawnPosValue = 0.0f;
417 foreach (var potentialSpawnPos in potentialSpawnPositions)
418 {
419 //the closer the spawnpos is to the main sub, the better
420 float spawnPosValue = 100000.0f / Math.Max(Vector2.Distance(potentialSpawnPos.Position.ToVector2(), Submarine.MainSub.WorldPosition), 1.0f);
421
422 //prefer spawnpoints that are at the left side of the sub (so the shuttle doesn't have to go backwards)
423 if (potentialSpawnPos.Position.X > Submarine.MainSub.WorldPosition.X)
424 {
425 spawnPosValue *= 0.1f;
426 }
427
428 if (spawnPosValue > bestSpawnPosValue)
429 {
430 bestSpawnPos = potentialSpawnPos.Position.ToVector2();
431 bestSpawnPosValue = spawnPosValue;
432 }
433 }
434
435 return bestSpawnPos;
436 }
