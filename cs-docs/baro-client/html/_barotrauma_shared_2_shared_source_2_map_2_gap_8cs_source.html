239 private void FindHulls()
240 {
241 Hull[] hulls = new Hull[2];
242
243 foreach (var linked in linkedTo)
244 {
245 if (linked is Hull hull)
246 {
247 hull.ConnectedGaps.Remove(this);
248 }
249 }
250 linkedTo.Clear();
251
252 int tolerance = 1;
253 Vector2[] searchPos = new Vector2[2];
254 if (IsHorizontal)
255 {
256 searchPos[0] = new Vector2(rect.X - tolerance, rect.Y - rect.Height / 2);
257 searchPos[1] = new Vector2(rect.Right + tolerance, rect.Y - rect.Height / 2);
258 }
259 else
260 {
261 searchPos[0] = new Vector2(rect.Center.X, rect.Y + tolerance);
262 searchPos[1] = new Vector2(rect.Center.X, rect.Y - rect.Height - tolerance);
263 }
264
265 for (int i = 0; i < 2; i++)
266 {
267 hulls[i] = Hull.FindHullUnoptimized(searchPos[i], null, false);
268 if (hulls[i] == null) hulls[i] = Hull.FindHullUnoptimized(searchPos[i], null, false, true);
269 }
270
271 if (hulls[0] == null && hulls[1] == null) { return; }
272
273 if (hulls[0] == null && hulls[1] != null)
274 {
275 Hull temp = hulls[0];
276 hulls[0] = hulls[1];
277 hulls[1] = temp;
278 }
279
280 flowTargetHull = hulls[0];
281
282 for (int i = 0; i < 2; i++)
283 {
284 if (hulls[i] == null) { continue; }
285 linkedTo.Add(hulls[i]);
286 if (!hulls[i].ConnectedGaps.Contains(this)) hulls[i].ConnectedGaps.Add(this);
287 }
288 }

290 public override void Update(float deltaTime, Camera cam)
291 {
292 flowForce = Vector2.Zero;
293
294 outsideColliderRaycastTimer -= deltaTime;
295
296 if (open == 0.0f || linkedTo.Count == 0)
297 {
298 lerpedFlowForce = Vector2.Zero;
299 return;
300 }
301
302 Hull hull1 = (Hull)linkedTo[0];
303 Hull hull2 = linkedTo.Count < 2 ? null : (Hull)linkedTo[1];
304 if (hull1 == hull2) { return; }
305
306 UpdateOxygen(hull1, hull2, deltaTime);
307
308 if (linkedTo.Count == 1)
309 {
310 //gap leading from a room to outside
311 UpdateRoomToOut(deltaTime, hull1);
312 }
313 else if (linkedTo.Count == 2)
314 {
315 //gap leading from a room to another
316 UpdateRoomToRoom(deltaTime, hull1, hull2);
317 }
318
319 flowForce.X = MathHelper.Clamp(flowForce.X, -MaxFlowForce, MaxFlowForce);
320 flowForce.Y = MathHelper.Clamp(flowForce.Y, -MaxFlowForce, MaxFlowForce);
321 if (openedTimer > 0.0f && flowForce.LengthSquared() > lerpedFlowForce.LengthSquared())
322 {
323 //if the gap has just been opened/created, allow it to exert a large force instantly without any smoothing
324 lerpedFlowForce = flowForce;
325 }
326 else
327 {
328 lerpedFlowForce = Vector2.Lerp(lerpedFlowForce, flowForce, deltaTime * 5.0f);
329 }
330 if (FlowTargetHull != null && IsRoomToRoom)
331 {
332 var otherRoom = linkedTo[1] == FlowTargetHull ? linkedTo[0] : linkedTo[1];
333 if ((otherRoom as Hull).Volume < FlowTargetHull.Volume)
334 {
335 lerpedFlowForce = Vector2.Zero;
336 }
337 }
338
339 openedTimer -= deltaTime;
340
341 EmitParticles(deltaTime);
342 }

346 void UpdateRoomToRoom(float deltaTime, Hull hull1, Hull hull2)
347 {
348 Vector2 subOffset = Vector2.Zero;
349 if (hull1.Submarine != Submarine)
350 {
351 subOffset = Submarine.Position - hull1.Submarine.Position;
352 }
353 else if (hull2.Submarine != Submarine)
354 {
355 subOffset = hull2.Submarine.Position - Submarine.Position;
356 }
357
358 if (hull1.WaterVolume <= 0.0 && hull2.WaterVolume <= 0.0) { return; }
359
360 float size = IsHorizontal ? rect.Height : rect.Width;
361
362 //a variable affecting the water flow through the gap
363 //the larger the gap is, the faster the water flows
364 float sizeModifier = size / 100.0f * open;
365
366 //horizontal gap (such as a regular door)
367 if (IsHorizontal)
368 {
369 higherSurface = Math.Max(hull1.Surface, hull2.Surface + subOffset.Y);
370 float delta = 0.0f;
371
372 //water level is above the lower boundary of the gap
373 if (Math.Max(hull1.Surface + hull1.WaveY[hull1.WaveY.Length - 1], hull2.Surface + subOffset.Y + hull2.WaveY[0]) > rect.Y - size)
374 {
375 int dir = (hull1.Pressure > hull2.Pressure + subOffset.Y) ? 1 : -1;
376
377 //water flowing from the righthand room to the lefthand room
378 if (dir == -1)
379 {
380 if (!(hull2.WaterVolume > 0.0f)) { return; }
381 lowerSurface = hull1.Surface - hull1.WaveY[hull1.WaveY.Length - 1];
382 //delta = Math.Min((room2.water.pressure - room1.water.pressure) * sizeModifier, Math.Min(room2.water.Volume, room2.Volume));
383 //delta = Math.Min(delta, room1.Volume - room1.water.Volume + Water.MaxCompress);
384
385 flowTargetHull = hull1;
386
387 //make sure not to move more than what the room contains
388 delta = Math.Min(((hull2.Pressure + subOffset.Y) - hull1.Pressure) * 300.0f * sizeModifier * deltaTime, Math.Min(hull2.WaterVolume, hull2.Volume));
389
390 //make sure not to place more water to the target room than it can hold
391 delta = Math.Min(delta, hull1.Volume * Hull.MaxCompress - hull1.WaterVolume);
392 hull1.WaterVolume += delta;
393 hull2.WaterVolume -= delta;
394 if (hull1.WaterVolume > hull1.Volume)
395 {
396 hull1.Pressure = Math.Max(hull1.Pressure, (hull1.Pressure + hull2.Pressure+subOffset.Y) / 2);
397 }
398
399 flowForce = new Vector2(-delta * (float)(Timing.Step / deltaTime), 0.0f);
400 }
401 else if (dir == 1)
402 {
403 if (!(hull1.WaterVolume > 0.0f)) { return; }
404 lowerSurface = hull2.Surface - hull2.WaveY[hull2.WaveY.Length - 1];
405
406 flowTargetHull = hull2;
407
408 //make sure not to move more than what the room contains
409 delta = Math.Min((hull1.Pressure - (hull2.Pressure + subOffset.Y)) * 300.0f * sizeModifier * deltaTime, Math.Min(hull1.WaterVolume, hull1.Volume));
410
411 //make sure not to place more water to the target room than it can hold
412 delta = Math.Min(delta, hull2.Volume * Hull.MaxCompress - hull2.WaterVolume);
413 hull1.WaterVolume -= delta;
414 hull2.WaterVolume += delta;
415 if (hull2.WaterVolume > hull2.Volume)
416 {
417 hull2.Pressure = Math.Max(hull2.Pressure, ((hull1.Pressure-subOffset.Y) + hull2.Pressure) / 2);
418 }
419
420 flowForce = new Vector2(delta * (float)(Timing.Step / deltaTime), 0.0f);
421 }
422
423 if (delta > 1.5f && subOffset == Vector2.Zero)
424 {
425 float avg = (hull1.Surface + hull2.Surface) / 2.0f;
426
427 if (hull1.WaterVolume < hull1.Volume / Hull.MaxCompress &&
428 hull1.Surface + hull1.WaveY[hull1.WaveY.Length - 1] < rect.Y)
429 {
430 hull1.WaveVel[hull1.WaveY.Length - 1] = (avg - (hull1.Surface + hull1.WaveY[hull1.WaveY.Length - 1])) * 0.1f;
431 hull1.WaveVel[hull1.WaveY.Length - 2] = hull1.WaveVel[hull1.WaveY.Length - 1];
432 }
433
434 if (hull2.WaterVolume < hull2.Volume / Hull.MaxCompress &&
435 hull2.Surface + hull2.WaveY[0] < rect.Y)
436 {
437 hull2.WaveVel[0] = (avg - (hull2.Surface + hull2.WaveY[0])) * 0.1f;
438 hull2.WaveVel[1] = hull2.WaveVel[0];
439 }
440 }
441 }
442
443 }
444 else
445 {
446 //lower room is full of water
447 if (hull2.Pressure + subOffset.Y > hull1.Pressure && hull2.WaterVolume > 0.0f)
448 {
449 float delta = Math.Min(hull2.WaterVolume - hull2.Volume + (hull2.Volume * Hull.MaxCompress), deltaTime * 8000.0f * sizeModifier);
450
451 //make sure not to place more water to the target room than it can hold
452 if (hull1.WaterVolume + delta > hull1.Volume * Hull.MaxCompress)
453 {
454 delta -= (hull1.WaterVolume + delta) - (hull1.Volume * Hull.MaxCompress);
455 }
456
457 delta = Math.Max(delta, 0.0f);
458 hull1.WaterVolume += delta;
459 hull2.WaterVolume -= delta;
460
461 flowForce = new Vector2(
462 0.0f,
463 Math.Min(Math.Min((hull2.Pressure + subOffset.Y) - hull1.Pressure, 200.0f), delta * (float)(Timing.Step / deltaTime)));
464
465 flowTargetHull = hull1;
466
467 if (hull1.WaterVolume > hull1.Volume)
468 {
469 hull1.Pressure = Math.Max(hull1.Pressure, (hull1.Pressure + (hull2.Pressure + subOffset.Y)) / 2);
470 }
471
472 }
473 //there's water in the upper room, drop to lower
474 else if (hull1.WaterVolume > 0)
475 {
476 flowTargetHull = hull2;
477
478 //make sure the amount of water moved isn't more than what the room contains
479 float delta = Math.Min(hull1.WaterVolume, deltaTime * 25000f * sizeModifier);
480
481 //make sure not to place more water to the target room than it can hold
482 if (hull2.WaterVolume + delta > hull2.Volume * Hull.MaxCompress)
483 {
484 delta -= (hull2.WaterVolume + delta) - (hull2.Volume * Hull.MaxCompress);
485 }
486 hull1.WaterVolume -= delta;
487 hull2.WaterVolume += delta;
488
489 flowForce = new Vector2(
490 hull1.WaveY[hull1.GetWaveIndex(rect.X)] - hull1.WaveY[hull1.GetWaveIndex(rect.Right)],
491 MathHelper.Clamp(-delta * (float)(Timing.Step / deltaTime), -200.0f, 0.0f));
492
493 if (hull2.WaterVolume > hull2.Volume)
494 {
495 hull2.Pressure = Math.Max(hull2.Pressure, ((hull1.Pressure - subOffset.Y) + hull2.Pressure) / 2);
496 }
497 }
498 }
499
500 if (open > 0.0f)
501 {
502 if (hull1.WaterVolume > hull1.Volume / Hull.MaxCompress && hull2.WaterVolume > hull2.Volume / Hull.MaxCompress)
503 {
504 float avgLethality = (hull1.LethalPressure + hull2.LethalPressure) / 2.0f;
505 hull1.LethalPressure = avgLethality;
506 hull2.LethalPressure = avgLethality;
507 }
508 else
509 {
510 hull1.LethalPressure = 0.0f;
511 hull2.LethalPressure = 0.0f;
512 }
513 }

516 void UpdateRoomToOut(float deltaTime, Hull hull1)
517 {
518 float size = IsHorizontal ? rect.Height : rect.Width;
519
520 //a variable affecting the water flow through the gap
521 //the larger the gap is, the faster the water flows
522 float sizeModifier = size * open * open;
523
524 float delta = 500.0f * sizeModifier * deltaTime;
525
526 //make sure not to place more water to the target room than it can hold
527 delta = Math.Min(delta, hull1.Volume * Hull.MaxCompress - hull1.WaterVolume);
528 hull1.WaterVolume += delta;
529
530 if (hull1.WaterVolume > hull1.Volume) { hull1.Pressure += 30.0f * deltaTime; }
531
532 flowTargetHull = hull1;
533
534 if (IsHorizontal)
535 {
536 //water flowing from right to left
537 if (rect.X > hull1.Rect.X + hull1.Rect.Width / 2.0f)
538 {
539 flowForce = new Vector2(-delta * (float)(Timing.Step / deltaTime), 0.0f);
540
541 }
542 else
543 {
544 flowForce = new Vector2(delta * (float)(Timing.Step / deltaTime), 0.0f);
545 }
546
547 higherSurface = hull1.Surface;
548 lowerSurface = rect.Y;
549
550 if (hull1.WaterVolume < hull1.Volume / Hull.MaxCompress &&
551 hull1.Surface < rect.Y)
552 {
553 if (rect.X > hull1.Rect.X + hull1.Rect.Width / 2.0f)
554 {
555 float vel = ((rect.Y - rect.Height / 2) - (hull1.Surface + hull1.WaveY[hull1.WaveY.Length - 1])) * 6.0f;
556 vel *= Math.Min(Math.Abs(flowForce.X) / 200.0f, 1.0f);
557
558 hull1.WaveVel[hull1.WaveY.Length - 1] += vel * deltaTime;
559 hull1.WaveVel[hull1.WaveY.Length - 2] += vel * deltaTime;
560 }
561 else
562 {
563 float vel = ((rect.Y - rect.Height / 2) - (hull1.Surface + hull1.WaveY[0])) * 6.0f;
564 vel *= Math.Min(Math.Abs(flowForce.X) / 200.0f, 1.0f);
565
566 hull1.WaveVel[0] += vel * deltaTime;
567 hull1.WaveVel[1] += vel * deltaTime;
568 }
569 }
570 else
571 {
572 hull1.LethalPressure += ((Submarine != null && Submarine.AtDamageDepth) ? 100.0f : 10.0f) * deltaTime;
573 }
574 }
575 else
576 {
577 if (rect.Y > hull1.Rect.Y - hull1.Rect.Height / 2.0f)
578 {
579 flowForce = new Vector2(0.0f, -delta * (float)(Timing.Step / deltaTime));
580 }
581 else
582 {
583 flowForce = new Vector2(0.0f, delta * (float)(Timing.Step / deltaTime));
584 }
585 if (hull1.WaterVolume >= hull1.Volume / Hull.MaxCompress)
586 {
587 hull1.LethalPressure += ((Submarine != null && Submarine.AtDamageDepth) ? 100.0f : 10.0f) * deltaTime;
588 }
589 }

592 public bool RefreshOutsideCollider()
593 {
594 if (IsRoomToRoom || Submarine == null || open <= 0.0f || linkedTo.Count == 0 || !(linkedTo[0] is Hull)) return false;
595
596 if (outsideColliderRaycastTimer <= 0.0f)

598 UpdateOutsideColliderPos((Hull)linkedTo[0]);
599 outsideColliderRaycastTimer = outsideCollisionBlocker.Enabled ?
600 OutsideColliderRaycastIntervalHighPrio :
601 OutsideColliderRaycastIntervalLowPrio;
602 }
603
604 return outsideCollisionBlocker.Enabled;

607 private void UpdateOutsideColliderPos(Hull hull)
608 {
609 if (Submarine == null || IsRoomToRoom || Level.Loaded == null) { return; }
610
611 Vector2 rayDir;
612 if (IsHorizontal)
613 {
614 rayDir = new Vector2(Math.Sign(rect.Center.X - hull.Rect.Center.X), 0);
615 }
616 else
617 {
618 rayDir = new Vector2(0, Math.Sign((rect.Y - rect.Height / 2) - (hull.Rect.Y - hull.Rect.Height / 2)));
619 }
620
621 Vector2 rayStart = ConvertUnits.ToSimUnits(WorldPosition);
622 Vector2 rayEnd = rayStart + rayDir * 5.0f;
623
624 var levelCells = Level.Loaded.GetCells(WorldPosition, searchDepth: 1);
625 foreach (var cell in levelCells)
626 {
627 if (cell.IsPointInside(WorldPosition))
628 {
629 outsideCollisionBlocker.Enabled = true;
630 Vector2 colliderPos = rayStart - Submarine.SimPosition;
631 float colliderRotation = MathUtils.VectorToAngle(rayDir) - MathHelper.PiOver2;
632 outsideCollisionBlocker.SetTransformIgnoreContacts(ref colliderPos, colliderRotation);
633 return;
634 }
635 }
636
637 var blockingBody = Submarine.CheckVisibility(rayStart, rayEnd);
638 if (blockingBody != null)
639 {
640 //if the ray hit the body of the submarine itself (for example, if there's 2 layers of walls) we can ignore it
641 if (blockingBody.UserData == Submarine) { return; }
642 outsideCollisionBlocker.Enabled = true;
643 Vector2 colliderPos = Submarine.LastPickedPosition - Submarine.SimPosition;
644 float colliderRotation = MathUtils.VectorToAngle(rayDir) - MathHelper.PiOver2;
645 outsideCollisionBlocker.SetTransformIgnoreContacts(ref colliderPos, colliderRotation);
646 }
647 else
648 {
649 outsideCollisionBlocker.Enabled = false;
650 }
651 }

653 private void UpdateOxygen(Hull hull1, Hull hull2, float deltaTime)
654 {
655 if (hull1 == null || hull2 == null) { return; }
656
657 if (IsHorizontal)
658 {
659 //if the water level is above the gap, oxygen doesn't circulate
660 if (Math.Max(hull1.WorldSurface + hull1.WaveY[hull1.WaveY.Length - 1], hull2.WorldSurface + hull2.WaveY[0]) > WorldRect.Y) { return; }
661 }
662
663 var should = GameMain.LuaCs.Hook.Call<bool?>("gapOxygenUpdate", this, hull1, hull2);
664
665 if (should != null && should.Value) return;
666
667 float totalOxygen = hull1.Oxygen + hull2.Oxygen;
668 float totalVolume = hull1.Volume + hull2.Volume;
669
670 float deltaOxygen = (totalOxygen * hull1.Volume / totalVolume) - hull1.Oxygen;
671 deltaOxygen = MathHelper.Clamp(deltaOxygen, -Hull.OxygenDistributionSpeed * deltaTime, Hull.OxygenDistributionSpeed * deltaTime);
672
673 hull1.Oxygen += deltaOxygen;
674 hull2.Oxygen -= deltaOxygen;
675 }

677 public static Gap FindAdjacent(IEnumerable<Gap> gaps, Vector2 worldPos, float allowedOrthogonalDist)
678 {
679 foreach (Gap gap in gaps)
680 {
681 if (gap.Open == 0.0f || gap.IsRoomToRoom) continue;
682
683 if (gap.ConnectedWall != null)
684 {
685 int sectionIndex = gap.ConnectedWall.FindSectionIndex(gap.Position);
686 if (sectionIndex > -1 && !gap.ConnectedWall.SectionBodyDisabled(sectionIndex)) continue;
687 }
688
689 if (gap.IsHorizontal)
690 {
691 if (worldPos.Y < gap.WorldRect.Y && worldPos.Y > gap.WorldRect.Y - gap.WorldRect.Height &&
692 Math.Abs(gap.WorldRect.Center.X - worldPos.X) < allowedOrthogonalDist)
693 {
694 return gap;
695 }
696 }
697 else
698 {
699 if (worldPos.X > gap.WorldRect.X && worldPos.X < gap.WorldRect.Right &&
700 Math.Abs(gap.WorldRect.Y - gap.WorldRect.Height / 2 - worldPos.Y) < allowedOrthogonalDist)
701 {
