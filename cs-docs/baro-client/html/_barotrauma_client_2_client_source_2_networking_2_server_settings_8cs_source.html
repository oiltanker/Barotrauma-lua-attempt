23 public object GUIComponentValue
24 {
25 get
26 {
27 if (GUIComponent == null) return null;
28 else if (GUIComponent is GUITickBox tickBox) return tickBox.Selected;
29 else if (GUIComponent is GUITextBox textBox) return textBox.Text;
30 else if (GUIComponent is GUIScrollBar scrollBar) return scrollBar.BarScrollValue;
31 else if (GUIComponent is GUIRadioButtonGroup radioButtonGroup) return radioButtonGroup.Selected;
32 else if (GUIComponent is GUIDropDown dropdown) return dropdown.SelectedData;
33 else if (GUIComponent is GUINumberInput numInput)
34 {
35 if (numInput.InputType == GUINumberInput.NumberType.Int) { return numInput.IntValue; } else { return numInput.FloatValue; }
36 }
37 return null;
38 }
39 set
40 {
41 if (GUIComponent == null) return;
42 else if (GUIComponent is GUITickBox tickBox) tickBox.Selected = (bool)value;
43 else if (GUIComponent is GUITextBox textBox) textBox.Text = (string)value;
44 else if (GUIComponent is GUIScrollBar scrollBar)
45 {
46 if (value.GetType() == typeof(int))
47 {
48 scrollBar.BarScrollValue = (int)value;
49 }
50 else
51 {
52 scrollBar.BarScrollValue = (float)value;
53 }
54 }
55 else if (GUIComponent is GUIRadioButtonGroup radioButtonGroup) radioButtonGroup.Selected = (int)value;
56 else if (GUIComponent is GUIDropDown dropdown) dropdown.SelectItem(value);
57 else if (GUIComponent is GUINumberInput numInput)
58 {
59 if (numInput.InputType == GUINumberInput.NumberType.Int)
60 {
61 numInput.IntValue = (int)value;
62 }
63 else
64 {
65 numInput.FloatValue = (float)value;
66 }
67 }
68 }

95 public void ClientAdminRead(IReadMessage incMsg)
96 {
97 while (true)
98 {
99 UInt32 key = incMsg.ReadUInt32();
100 if (key == 0) { break; }
101 if (netProperties.ContainsKey(key))
102 {
103 bool changedLocally = netProperties[key].ChangedLocally;
104 netProperties[key].Read(incMsg);
105 netProperties[key].TempValue = netProperties[key].Value;
106
107 if (netProperties[key].GUIComponent != null)
108 {
109 if (!changedLocally)
110 {
111 netProperties[key].GUIComponentValue = netProperties[key].Value;
112 }
113 }
114 }
115 else
116 {
117 UInt32 size = incMsg.ReadVariableUInt32();
118 incMsg.BitPosition += (int)(8 * size);
119 }
120 }
121
122 ReadMonsterEnabled(incMsg);
123 BanList.ClientAdminRead(incMsg);
124 Whitelist.ClientAdminRead(incMsg);
125 }

127 public void ClientRead(IReadMessage incMsg)
128 {
129 cachedServerListInfo = null;
130
131 NetFlags requiredFlags = (NetFlags)incMsg.ReadByte();
132
133 if (requiredFlags.HasFlag(NetFlags.Name))
134 {
135 ServerName = incMsg.ReadString();
136 }
137
138 if (requiredFlags.HasFlag(NetFlags.Message))
139 {
140 ServerMessageText = incMsg.ReadString();
141 }
142 PlayStyle = (PlayStyle)incMsg.ReadByte();
143 MaxPlayers = incMsg.ReadByte();
144 HasPassword = incMsg.ReadBoolean();
145 IsPublic = incMsg.ReadBoolean();
146 GameMain.NetLobbyScreen.SetPublic(IsPublic);
147 AllowFileTransfers = incMsg.ReadBoolean();
148 incMsg.ReadPadBits();
149 TickRate = incMsg.ReadRangedInteger(1, 60);
150 GameMain.NetworkMember.TickRate = TickRate;
151
152 if (requiredFlags.HasFlag(NetFlags.Properties))
153 {
154 ReadExtraCargo(incMsg);
155 }
156
157 if (requiredFlags.HasFlag(NetFlags.HiddenSubs))
158 {
159 ReadHiddenSubs(incMsg);
160 }
161 GameMain.NetLobbyScreen.UpdateSubVisibility();
162
163 bool isAdmin = incMsg.ReadBoolean();
164 incMsg.ReadPadBits();
165 if (isAdmin)
166 {
167 ClientAdminRead(incMsg);
168 }

171 public void ClientAdminWrite(
172 NetFlags dataToSend,
173 int? missionTypeOr = null,
174 int? missionTypeAnd = null,
175 float? levelDifficulty = null,
176 bool? autoRestart = null,
177 int traitorSetting = 0,
178 int botCount = 0,
179 int botSpawnMode = 0,
180 bool? useRespawnShuttle = null)
181 {
182 if (!GameMain.Client.HasPermission(Networking.ClientPermissions.ManageSettings)) return;
183
184 IWriteMessage outMsg = new WriteOnlyMessage();
185
186 outMsg.Write((byte)ClientPacketHeader.SERVER_SETTINGS);
187
188 outMsg.Write((byte)dataToSend);
189
190 if (dataToSend.HasFlag(NetFlags.Name))
191 {
192 if (GameMain.NetLobbyScreen.ServerName.Text != ServerName)
193 {
194 ServerName = GameMain.NetLobbyScreen.ServerName.Text;
195 }
196 outMsg.Write(ServerName);
197 }
198
199 if (dataToSend.HasFlag(NetFlags.Message))
200 {
201 if (GameMain.NetLobbyScreen.ServerMessage.Text != ServerMessageText)
202 {
203 ServerMessageText = GameMain.NetLobbyScreen.ServerMessage.Text;
204 }
205 outMsg.Write(ServerMessageText);
206 }
207
208 if (dataToSend.HasFlag(NetFlags.Properties))
209 {
210 //TODO: split this up?
211 WriteExtraCargo(outMsg);
212
213 IEnumerable<KeyValuePair<UInt32, NetPropertyData>> changedProperties = netProperties.Where(kvp => kvp.Value.ChangedLocally);
214 UInt32 count = (UInt32)changedProperties.Count();
215 bool changedMonsterSettings = tempMonsterEnabled != null && tempMonsterEnabled.Any(p => p.Value != MonsterEnabled[p.Key]);
216
217 outMsg.Write(count);
218 foreach (KeyValuePair<UInt32, NetPropertyData> prop in changedProperties)
219 {
220 DebugConsole.NewMessage(prop.Value.Name.Value, Color.Lime);
221 outMsg.Write(prop.Key);
222 prop.Value.Write(outMsg, prop.Value.GUIComponentValue);
223 }
224
225 outMsg.Write(changedMonsterSettings); outMsg.WritePadBits();
226 if (changedMonsterSettings) WriteMonsterEnabled(outMsg, tempMonsterEnabled);
227 BanList.ClientAdminWrite(outMsg);
228 Whitelist.ClientAdminWrite(outMsg);
229 }
230
231 if (dataToSend.HasFlag(NetFlags.HiddenSubs))
232 {
233 WriteHiddenSubs(outMsg);
234 }
235
236 if (dataToSend.HasFlag(NetFlags.Misc))
237 {
238 outMsg.WriteRangedInteger(missionTypeOr ?? (int)Barotrauma.MissionType.None, 0, (int)Barotrauma.MissionType.All);
239 outMsg.WriteRangedInteger(missionTypeAnd ?? (int)Barotrauma.MissionType.All, 0, (int)Barotrauma.MissionType.All);
240 outMsg.Write((byte)(traitorSetting + 1));
241 outMsg.Write((byte)(botCount + 1));
242 outMsg.Write((byte)(botSpawnMode + 1));
243
244 outMsg.Write(levelDifficulty ?? -1000.0f);
245
246 outMsg.Write(useRespawnShuttle ?? UseRespawnShuttle);
247
248 outMsg.Write(autoRestart != null);
249 outMsg.Write(autoRestart ?? false);
250
251 outMsg.WritePadBits();
252 }
253
254 if (dataToSend.HasFlag(NetFlags.LevelSeed))
255 {
256 outMsg.Write(GameMain.NetLobbyScreen.SeedBox.Text);
257 }
258
259 GameMain.Client.ClientPeer.Send(outMsg, DeliveryMethod.Reliable);
260 }
