130  public void Update(float deltaTime, Camera cam)
131  {
132  //calculate the sum of the forces of nearby level triggers
133  //and use it to move the dust texture and water distortion effect
134  Vector2 currentDustVel = defaultDustVelocity;
135  foreach (LevelObject levelObject in level.LevelObjectManager.GetVisibleObjects())
136  {
137  if (levelObject.Triggers == null) { continue; }
138  //use the largest water flow velocity of all the triggers
139  Vector2 objectMaxFlow = Vector2.Zero;
140  foreach (LevelTrigger trigger in levelObject.Triggers)
141  {
142  Vector2 vel = trigger.GetWaterFlowVelocity(cam.WorldViewCenter);
143  if (vel.LengthSquared() > objectMaxFlow.LengthSquared())
144  {
145  objectMaxFlow = vel;
146  }
147  }
148  currentDustVel += objectMaxFlow;
149  }
150 
151  dustVelocity = Vector2.Lerp(dustVelocity, currentDustVel, deltaTime);
152 
153  WaterRenderer.Instance?.ScrollWater(dustVelocity, deltaTime);
154 
155  if (level.GenerationParams.WaterParticles != null)
156  {
157  Vector2 waterTextureSize = level.GenerationParams.WaterParticles.size * level.GenerationParams.WaterParticleScale;
158  dustOffset += new Vector2(dustVelocity.X, -dustVelocity.Y) * level.GenerationParams.WaterParticleScale * deltaTime;
159  while (dustOffset.X <= -waterTextureSize.X) dustOffset.X += waterTextureSize.X;
160  while (dustOffset.X >= waterTextureSize.X) dustOffset.X -= waterTextureSize.X;
161  while (dustOffset.Y <= -waterTextureSize.Y) dustOffset.Y += waterTextureSize.Y;
162  while (dustOffset.Y >= waterTextureSize.Y) dustOffset.Y -= waterTextureSize.Y;
163  }
164  }

404  //render destructible walls
