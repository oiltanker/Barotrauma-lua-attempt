351 public static ServerInfo FromXElement(XElement element)
352 {
353 ServerInfo info = new ServerInfo()
354 {
355 ServerName = element.GetAttributeString("ServerName", ""),
356 ServerMessage = element.GetAttributeString("ServerMessage", ""),
357 IP = element.GetAttributeString("IP", ""),

365 info.GameMode = element.GetAttributeIdentifier("GameMode", Identifier.Empty);
366 info.GameVersion = element.GetAttributeString("GameVersion", "");
367
368 int maxPlayersElement = element.GetAttributeInt("MaxPlayers", 0);
369
370 if (maxPlayersElement > NetConfig.MaxPlayers)

372 /*DebugConsole.IsOpen = true;
373 DebugConsole.NewMessage($"Setting the maximum amount of players to {maxPlayersElement} failed due to exceeding the limit of {NetConfig.MaxPlayers} players per server. Using the maximum of {NetConfig.MaxPlayers} instead.", Color.Red);*/

376
377 info.MaxPlayers = maxPlayersElement;
378
379 if (Enum.TryParse(element.GetAttributeString("PlayStyle", ""), out PlayStyle playStyleTemp)) { info.PlayStyle = playStyleTemp; }
380 if (bool.TryParse(element.GetAttributeString("UsingWhiteList", ""), out bool whitelistTemp)) { info.UsingWhiteList = whitelistTemp; }
381 if (Enum.TryParse(element.GetAttributeString("TraitorsEnabled", ""), out YesNoMaybe traitorsTemp)) { info.TraitorsEnabled = traitorsTemp; }
382 if (Enum.TryParse(element.GetAttributeString("SubSelectionMode", ""), out SelectionMode subSelectionTemp)) { info.SubSelectionMode = subSelectionTemp; }
383 if (Enum.TryParse(element.GetAttributeString("ModeSelectionMode", ""), out SelectionMode modeSelectionTemp)) { info.ModeSelectionMode = modeSelectionTemp; }
384 if (bool.TryParse(element.GetAttributeString("VoipEnabled", ""), out bool voipTemp)) { info.VoipEnabled = voipTemp; }

393 public void QueryLiveInfo(Action<Networking.ServerInfo> onServerRulesReceived, Action<Networking.ServerInfo> onQueryDone)
394 {
395 if (!SteamManager.IsInitialized) { return; }
396
397 if (int.TryParse(QueryPort, out int parsedPort) && IPAddress.TryParse(IP, out IPAddress parsedIP))
398 {
399 if (MatchmakingPingResponse?.QueryActive ?? false)
400 {
401 MatchmakingPingResponse.Cancel();
402 }
403
404 MatchmakingPingResponse = new Steamworks.SteamMatchmakingPingResponse(
405 (server) =>
406 {
407 ServerName = server.Name;
408 RespondedToSteamQuery = true;
409 PlayerCount = server.Players;
410 MaxPlayers = server.MaxPlayers;
411 HasPassword = server.Passworded;
412 PingChecked = true;
413 Ping = server.Ping;
414 LobbyID = 0;
415 TaskPool.Add("QueryServerRules (QueryLiveInfo)", server.QueryRulesAsync(),
416 (t) =>
417 {
418 onQueryDone(this);
419 if (t.Status == TaskStatus.Faulted)
420 {
421 TaskPool.PrintTaskExceptions(t, "Failed to retrieve rules for " + ServerName);
422 return;
423 }
424
425 t.TryGetResult(out Dictionary<string, string> rules);
426 SteamManager.AssignServerRulesToServerInfo(rules, this);
427
428 onServerRulesReceived(this);
429 });
430 },
431 () =>
432 {
433 RespondedToSteamQuery = false;
434 });
435
436 MatchmakingPingResponse.HQueryPing(parsedIP, parsedPort);
437 }
438 else if (OwnerID != 0)
439 {
440 if (SteamFriend == null)
441 {
442 SteamFriend = new Steamworks.Friend(OwnerID);
443 }
444 if (LobbyID == 0)
445 {
446 TaskPool.Add("RequestSteamP2POwnerInfo", SteamFriend?.RequestInfoAsync(),
447 (t) =>
448 {
449 onQueryDone(this);
450 if ((SteamFriend?.IsPlayingThisGame ?? false) && ((SteamFriend?.GameInfo?.Lobby?.Id ?? 0) != 0))
451 {
452 LobbyID = SteamFriend?.GameInfo?.Lobby?.Id.Value ?? 0;
453 Steamworks.SteamMatchmaking.OnLobbyDataChanged += UpdateInfoFromSteamworksLobby;
454 SteamFriend?.GameInfo?.Lobby?.Refresh();
455 }
456 else
457 {
458 RespondedToSteamQuery = false;
459 }
460 });
461 }
462 else
463 {
464 onQueryDone(this);
465 }
466 }

469 private void UpdateInfoFromSteamworksLobby(Steamworks.Data.Lobby lobby)
470 {
471 if (lobby.Id != LobbyID) { return; }
472 Steamworks.SteamMatchmaking.OnLobbyDataChanged -= UpdateInfoFromSteamworksLobby;
473 if (string.IsNullOrWhiteSpace(lobby.GetData("haspassword"))) { return; }
474 bool.TryParse(lobby.GetData("haspassword"), out bool hasPassword);
475 int.TryParse(lobby.GetData("playercount"), out int currPlayers);
476 int.TryParse(lobby.GetData("maxplayernum"), out int maxPlayers);
477 UInt64 ownerId = SteamManager.SteamIDStringToUInt64(lobby.GetData("lobbyowner"));
478
479 if (OwnerID != ownerId) { return; }
480
481 ServerName = lobby.GetData("name");
482 IP = "";
483 Port = "";
484 QueryPort = "";
485 PlayerCount = currPlayers;
486 MaxPlayers = maxPlayers;
487 HasPassword = hasPassword;
488 RespondedToSteamQuery = true;
489 LobbyID = lobby.Id;
490 OwnerID = ownerId;
491 PingChecked = false;
492 OwnerVerified = true;
493
494 SteamManager.AssignLobbyDataToServerInfo(lobby, this);
495 }
