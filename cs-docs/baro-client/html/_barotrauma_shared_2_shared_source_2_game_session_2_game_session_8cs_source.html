182  private GameMode InstantiateGameMode(GameModePreset gameModePreset, string? seed, SubmarineInfo selectedSub, CampaignSettings settings, IEnumerable<MissionPrefab>? missionPrefabs = null, MissionType missionType = MissionType.None)
183  {
184  if (gameModePreset.GameModeType == typeof(CoOpMode) || gameModePreset.GameModeType == typeof(PvPMode))
185  {
186  //don't allow hidden mission types (e.g. GoTo) in single mission modes
187  var missionTypes = (MissionType[])Enum.GetValues(typeof(MissionType));
188  for (int i = 0; i < missionTypes.Length; i++)
189  {
190  if (MissionPrefab.HiddenMissionClasses.Contains(missionTypes[i]))
191  {
192  missionType &= ~missionTypes[i];
193  }
194  }
195  }
196  if (gameModePreset.GameModeType == typeof(CoOpMode))
197  {
198  return missionPrefabs != null ?
199  new CoOpMode(gameModePreset, missionPrefabs) :
200  new CoOpMode(gameModePreset, missionType, seed ?? ToolBox.RandomSeed(8));
201  }
202  else if (gameModePreset.GameModeType == typeof(PvPMode))
203  {
204  return missionPrefabs != null ?
205  new PvPMode(gameModePreset, missionPrefabs) :
206  new PvPMode(gameModePreset, missionType, seed ?? ToolBox.RandomSeed(8));
207  }
208  else if (gameModePreset.GameModeType == typeof(MultiPlayerCampaign))
209  {
210  var campaign = MultiPlayerCampaign.StartNew(seed ?? ToolBox.RandomSeed(8), selectedSub, settings);
211  if (selectedSub != null)
212  {
213  campaign.Bank.Deduct(selectedSub.Price);
214  campaign.Bank.Balance = Math.Max(campaign.Bank.Balance, MultiPlayerCampaign.MinimumInitialMoney);
215  }
216  return campaign;
217  }
218 #if CLIENT
219  else if (gameModePreset.GameModeType == typeof(SinglePlayerCampaign))
220  {
221  var campaign = SinglePlayerCampaign.StartNew(seed ?? ToolBox.RandomSeed(8), selectedSub, settings);
222  if (selectedSub != null)
223  {
224  campaign.Bank.TryDeduct(selectedSub.Price);
225  campaign.Bank.Balance = Math.Max(campaign.Bank.Balance, MultiPlayerCampaign.MinimumInitialMoney);
226  }
227  return campaign;
228  }
229  else if (gameModePreset.GameModeType == typeof(TutorialMode))
230  {
231  return new TutorialMode(gameModePreset);
232  }
233  else if (gameModePreset.GameModeType == typeof(TestGameMode))
234  {
235  return new TestGameMode(gameModePreset);
236  }
237 #endif
238  else if (gameModePreset.GameModeType == typeof(GameMode))
239  {
240  return new GameMode(gameModePreset);
241  }
242  else
243  {
244  throw new Exception($"Could not find a game mode of the type \"{gameModePreset.GameModeType}\"");
245  }

638  Rectangle outpostBorders = Level.Loaded.StartOutpost.GetDockedBorders();
639  Rectangle subBorders = Submarine.GetDockedBorders();
640 
641  Submarine.SetPosition(
642  Level.Loaded.StartOutpost.WorldPosition -
643  new Vector2(0.0f, outpostBorders.Height / 2 + subBorders.Height / 2));
644 
645  //find the port that's the nearest to the outpost and dock if one is found
646  float closestDistance = 0.0f;
647  DockingPort? myPort = null, outPostPort = null;
648  foreach (DockingPort port in DockingPort.List)
649  {
650  if (port.IsHorizontal || port.Docked) { continue; }
651  if (port.Item.Submarine == level.StartOutpost)
652  {
653  if (port.DockingTarget == null)
654  {
655  outPostPort = port;
656  }
657  continue;
658  }
