45  public void CreateEvent(IClientSerializable entity, NetEntityEvent.IData extraData = null)
46  {
47  if (GameMain.Client?.Character == null) { return; }
48 
49  if (!ValidateEntity(entity)) { return; }
50 
51  var newEvent = new ClientEntityEvent(
52  entity,
53  eventId: (UInt16)(ID + 1),
54  characterStateId: GameMain.Client.Character.LastNetworkUpdateID);
55  if (extraData != null) { newEvent.SetData(extraData); }
56 
57  for (int i = events.Count - 1; i >= 0; i--)
58  {
59  //we already have an identical event that's waiting to be sent
60  // -> no need to add a new one
61  if (!events[i].Sent && events[i].IsDuplicate(newEvent)) return;
62  }
63 
64  ID++;
65 
66  events.Add(newEvent);
67  }

69  public void Write(IWriteMessage msg, NetworkConnection serverConnection)
70  {
71  if (events.Count == 0 || serverConnection == null) return;
72 
73  List<NetEntityEvent> eventsToSync = new List<NetEntityEvent>();
74 
75  //find the index of the first event the server hasn't received
76  int startIndex = events.Count;
77  while (startIndex > 0 &&
78  NetIdUtils.IdMoreRecent(events[startIndex - 1].ID, thisClient.LastSentEntityEventID))
79  {
80  startIndex--;
81  }
82 
83  //remove events the server has already received
84  events.RemoveRange(0, startIndex);
85 
86  for (int i = 0; i < events.Count; i++)
87  {
88  //find the first event that hasn't been sent in roundtriptime or at all
89  eventLastSent.TryGetValue(events[i].ID, out float lastSent);
90 
91  if (lastSent > Lidgren.Network.NetTime.Now - 0.2) //TODO: reimplement serverConnection.AverageRoundtripTime
92  {
93  continue;
94  }
95 
96  eventsToSync.AddRange(events.GetRange(i, events.Count - i));
97  break;
98  }
99  if (eventsToSync.Count == 0) { return; }
100 
101  foreach (NetEntityEvent entityEvent in eventsToSync)
102  {
103  eventLastSent[entityEvent.ID] = (float)Lidgren.Network.NetTime.Now;
104  }
105 
106  msg.Write((byte)ClientNetObject.ENTITY_STATE);
107  Write(msg, eventsToSync, out _);
108  }

115  public bool Read(ServerNetObject type, IReadMessage msg, float sendingTime, List<IServerSerializable> entities)
116  {
117  UInt16 unreceivedEntityEventCount = 0;
118 
119  if (type == ServerNetObject.ENTITY_EVENT_INITIAL)
120  {
121  unreceivedEntityEventCount = msg.ReadUInt16();
122  firstNewID = msg.ReadUInt16();
123 
124  if (GameSettings.CurrentConfig.VerboseLogging)
125  {
126  DebugConsole.NewMessage(
127  "received midround syncing msg, unreceived: " + unreceivedEntityEventCount +
128  ", first new ID: " + firstNewID, Microsoft.Xna.Framework.Color.Yellow);
129  }
130  }
131  else
132  {
133  MidRoundSyncingDone = true;
134  if (firstNewID != null)
135  {
136  if (GameSettings.CurrentConfig.VerboseLogging)
137  {
138  DebugConsole.NewMessage("midround syncing complete, switching to ID " + (UInt16) (firstNewID - 1),
139  Microsoft.Xna.Framework.Color.Yellow);
140  }
141  lastReceivedID = (UInt16)(firstNewID - 1);
142  firstNewID = null;
143  }
144  }
145 
146  entities.Clear();
147 
148  msg.ReadPadBits();
149  UInt16 firstEventID = msg.ReadUInt16();
150  int eventCount = msg.ReadByte();
151 
152  for (int i = 0; i < eventCount; i++)
153  {
154  //16 = entity ID, 8 = msg length
155  if (msg.BitPosition + 16 + 8 > msg.LengthBits)
156  {
157  string errorMsg = $"Error while reading a message from the server. Entity event data exceeds the size of the buffer (current position: {msg.BitPosition}, length: {msg.LengthBits}).";
158  errorMsg += "\nPrevious entities:";
159  for (int j = entities.Count - 1; j >= 0; j--)
160  {
161  errorMsg += "\n" + (entities[j] == null ? "NULL" : entities[j].ToString());
162  }
163  DebugConsole.ThrowError(errorMsg);
164  return false;
165  }
166 
167  UInt16 thisEventID = (UInt16)(firstEventID + (UInt16)i);
168  UInt16 entityID = msg.ReadUInt16();
169 
170  if (entityID == Entity.NullEntityID)
171  {
172  if (GameSettings.CurrentConfig.VerboseLogging)
173  {
174  DebugConsole.NewMessage("received msg " + thisEventID + " (null entity)",
175  Microsoft.Xna.Framework.Color.Orange);
176  }
177  entities.Add(null);
178  if (thisEventID == (UInt16)(lastReceivedID + 1)) { lastReceivedID++; }
179  continue;
180  }
181 
182  int msgLength = (int)msg.ReadVariableUInt32();
183 
184  IServerSerializable entity = Entity.FindEntityByID(entityID) as IServerSerializable;
185  entities.Add(entity);
186 
187  //skip the event if we've already received it or if the entity isn't found
188  if (thisEventID != (UInt16)(lastReceivedID + 1) || entity == null)
189  {
190  if (thisEventID != (UInt16) (lastReceivedID + 1))
191  {
192  if (GameSettings.CurrentConfig.VerboseLogging)
193  {
194  DebugConsole.NewMessage(
195  "Received msg " + thisEventID + " (waiting for " + (lastReceivedID + 1) + ")",
196  NetIdUtils.IdMoreRecent(thisEventID, (UInt16)(lastReceivedID + 1))
197  ? GUIStyle.Red
198  : Microsoft.Xna.Framework.Color.Yellow);
199  }
200  }
201  else if (entity == null)
202  {
203  DebugConsole.NewMessage(
204  "Received msg " + thisEventID + ", entity " + entityID + " not found",
205  GUIStyle.Red);
206  GameMain.Client.ReportError(ClientNetError.MISSING_ENTITY, eventID: thisEventID, entityID: entityID);
207  return false;
208  }
209 
210  msg.BitPosition += msgLength * 8;
211  }
212  else
213  {
214  int msgPosition = msg.BitPosition;
215  if (GameSettings.CurrentConfig.VerboseLogging)
216  {
217  DebugConsole.NewMessage("received msg " + thisEventID + " (" + entity.ToString() + ")",
218  Microsoft.Xna.Framework.Color.Green);
219  }
220  lastReceivedID++;
221  try
222  {
223  ReadEvent(msg, entity, sendingTime);
224  msg.ReadPadBits();
225 
226  if (msg.BitPosition != msgPosition + msgLength * 8)
227  {
228  var prevEntity = entities.Count >= 2 ? entities[entities.Count - 2] : null;
229  ushort prevId = prevEntity is Entity p ? p.ID : (ushort)0;
230  string errorMsg = $"Message byte position incorrect after reading an event for the entity \"{entity}\" (ID {(entity is Entity e ? e.ID : 0)}). "
231  +$"The previous entity was \"{prevEntity}\" (ID {prevId}) "
232  +$"Read {msg.BitPosition - msgPosition} bits, expected message length was {msgLength * 8} bits.";
233 
234  DebugConsole.ThrowError(errorMsg);
235 
236  GameAnalyticsManager.AddErrorEventOnce("ClientEntityEventManager.Read:BitPosMismatch", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
237 
238  //TODO: force the BitPosition to correct place? Having some entity in a potentially incorrect state is not as bad as a desync kick
239  //msg.BitPosition = (int)(msgPosition + msgLength * 8);
240  }
241  }
242  catch (Exception e)
243  {
244  string errorMsg = $"Failed to read event {thisEventID} for entity \"{entity}\"" +
245  $"{(entity is Entity { ID: var entityId } ? $", id {entityId}" : "")} ";
246  DebugConsole.ThrowError(errorMsg, e);
247 
248  errorMsg += $"({e.Message})! (MidRoundSyncing: {thisClient.MidRoundSyncing})\n{e.StackTrace.CleanupStackTrace()}";
249  errorMsg += "\nPrevious entities:";
250  for (int j = entities.Count - 2; j >= 0; j--)
251  {
252  errorMsg += "\n" + (entities[j] == null ? "NULL" : entities[j].ToString());
253  }
254 
255  GameAnalyticsManager.AddErrorEventOnce("ClientEntityEventManager.Read:ReadFailed" + entity.ToString(),
256  GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
257  msg.BitPosition = (int)(msgPosition + msgLength * 8);
258  }
259  }
260  }
261  return true;
262  }
