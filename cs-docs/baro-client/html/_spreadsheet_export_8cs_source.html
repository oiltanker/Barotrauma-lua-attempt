81  private static XElement ParseMedical(ItemPrefab prefab)
82  {
83  ContentXElement? itemMeleeWeapon = prefab.ConfigElement.GetChildElement(nameof(MeleeWeapon));
84  // affliction, amount, duration
85  List<(LocalizedString Name, float Amount, float Duration)> onSuccessAfflictions = new List<(LocalizedString Name, float Amount, float Duration)>();
86  List<(LocalizedString Name, float Amount, float Duration)> onFailureAfflictions = new List<(LocalizedString Name, float Amount, float Duration)>();
87  int medicalRequiredSkill = 0;
88  if (itemMeleeWeapon != null)
89  {
90  List<StatusEffect> statusEffects = new List<StatusEffect>();
91  foreach (var subElement in itemMeleeWeapon.Elements())
92  {
93  string name = subElement.Name.ToString();
94  if (name.Equals(nameof(StatusEffect), StringComparison.OrdinalIgnoreCase))
95  {
96  StatusEffect statusEffect = StatusEffect.Load(subElement, debugIdentifier);
97  if (statusEffect == null || !statusEffect.HasTag("medical")) { continue; }
98 
99  statusEffects.Add(statusEffect);
100  }
101  else if (IsRequiredSkill(subElement, out Skill? skill) && skill != null)
102  {
103  medicalRequiredSkill = (int) skill.Level;
104  }
105  }
106 
107  List<StatusEffect> successEffects = statusEffects.Where(se => se.type == ActionType.OnUse).ToList();
108  List<StatusEffect> failureEffects = statusEffects.Where(se => se.type == ActionType.OnFailure).ToList();
109 
110  foreach (StatusEffect statusEffect in successEffects)
111  {
112  float duration = statusEffect.Duration;
113  onSuccessAfflictions.AddRange(statusEffect.ReduceAffliction.Select(ra => (GetAfflictionName(ra.AfflictionIdentifier), -ra.ReduceAmount, duration)));
114  onSuccessAfflictions.AddRange(statusEffect.Afflictions.Select(affliction => (affliction.Prefab.Name, affliction.NonClampedStrength, duration)));
115  }
116 
117  foreach (StatusEffect statusEffect in failureEffects)
118  {
119  float duration = statusEffect.Duration;
120  onFailureAfflictions.AddRange(statusEffect.ReduceAffliction.Select(ra => (GetAfflictionName(ra.AfflictionIdentifier), -ra.ReduceAmount, duration)));
121  onFailureAfflictions.AddRange(statusEffect.Afflictions.Select(affliction => (affliction.Prefab.Name, affliction.NonClampedStrength, duration)));
122  }
123  }
124 
125  return new XElement("Medical",
126  new XAttribute("skillamount", medicalRequiredSkill),
127  new XAttribute("successafflictions", FormatArray(onSuccessAfflictions.Select(tpl => tpl.Item1))),
128  new XAttribute("successamounts", FormatArray(onSuccessAfflictions.Select(tpl => FormatFloat(tpl.Item2)))),
129  new XAttribute("successdurations", FormatArray(onSuccessAfflictions.Select(tpl => FormatFloat(tpl.Item3)))),
130  new XAttribute("failureafflictions", FormatArray(onFailureAfflictions.Select(tpl => tpl.Item1))),
131  new XAttribute("failureamounts", FormatArray(onFailureAfflictions.Select(tpl => FormatFloat(tpl.Item2)))),
132  new XAttribute("failuredurations", FormatArray(onFailureAfflictions.Select(tpl => FormatFloat(tpl.Item3))))
133  );
134  }

136  private static XElement ParseWeapon(ItemPrefab prefab)
137  {
138  float stun = 0;
139  bool isAoE = false;
140  float? structDamage = null;
141  int skillRequirement = 0;
142 
143  // affliction, amount
144  List<(LocalizedString Name, float Amount)> damages = new List<(LocalizedString Name, float Amount)>();
145 
146  string[] validNames = { nameof(Projectile), nameof(MeleeWeapon), nameof(RepairTool), nameof(ItemComponent), nameof(RangedWeapon) };
147  foreach (var icElement in prefab.ConfigElement.Elements())
148  {
149  string icName = icElement.Name.ToString();
150  if (!validNames.Any(name => icName.Equals(name, StringComparison.OrdinalIgnoreCase))) { continue; }
151 
152  foreach (var icChildElement in icElement.Elements())
153  {
154  string name = icChildElement.Name.ToString();
155  if (IsRequiredSkill(icChildElement, out Skill? skill) && skill != null)
156  {
157  skillRequirement = (int) skill.Level;
158  }
159  else if (name.Equals(nameof(Attack), StringComparison.OrdinalIgnoreCase))
160  {
161  ParseAttack(new Attack(icChildElement, debugIdentifier));
162  }
163  else if (name.Equals(nameof(Explosion), StringComparison.OrdinalIgnoreCase))
164  {
165  ParseExplosion(new[] { new Explosion(icChildElement, debugIdentifier) });
166  }
167  else if (name.Equals(nameof(StatusEffect), StringComparison.OrdinalIgnoreCase))
168  {
169  ParseStatusEffect(new[] { StatusEffect.Load(icChildElement, debugIdentifier) });
170  }

174  foreach (StatusEffect effect in statusEffects)

200  void ParseAfflictions(IEnumerable<Affliction> afflictions)
201  {
202  foreach (Affliction affliction in afflictions)
203  {
204  // Exclude stuns
205  if (affliction.Prefab == AfflictionPrefab.Stun)
206  {
207  stun += affliction.NonClampedStrength;
208  continue;
209  }
210 
211  damages.Add((affliction.Prefab.Name, affliction.NonClampedStrength));
212  }
213  }
214  }
215  }
216 
217  return new XElement("Weapon",
218  new XAttribute("damagenames", FormatArray(damages.Select(tpl => tpl.Item1))),
219  new XAttribute("damageamounts", FormatArray(damages.Select(tpl => FormatFloat(tpl.Item2)))),
220  new XAttribute("isaoe", isAoE),
221  new XAttribute("structuredamage", structDamage ?? 0),
222  new XAttribute("stun", FormatFloat(stun)),
223  new XAttribute("skillrequirement", skillRequirement)
224  );
225  }

242  private static bool IsRequiredSkill(ContentXElement element, out Skill? skill)
243  {
244  string name = element.Name.ToString();
245  bool isSkill = name.Equals("RequiredSkill", StringComparison.OrdinalIgnoreCase) ||
246  name.Equals("RequiredSkills", StringComparison.OrdinalIgnoreCase);
247 
248  if (isSkill)
249  {
250  Identifier identifier = element.GetAttributeIdentifier(nameof(Skill.Identifier), Identifier.Empty);
251  float level = element.GetAttributeFloat(nameof(Skill.Level).ToLowerInvariant(), 0f);
252  skill = new Skill(identifier, level);
253  }
254  else
255  {
256  skill = null;
257  }
258 
259  return isSkill;
260  }
