91  public void Draw(SpriteBatch spriteBatch, bool editing, float itemDepth = -1)
92  {

95 
96  Vector2 startPos = GetSourcePos();

98  if (source is Item sourceItem && !sourceItem.Removed)

101  var weapon = sourceItem.GetComponent<RangedWeapon>();
102  if (turret != null)
103  {
104  startPos = new Vector2(sourceItem.WorldRect.X + turret.TransformedBarrelPos.X, -(sourceItem.WorldRect.Y - turret.TransformedBarrelPos.Y));
105  if (turret.BarrelSprite != null)
106  {
107  startPos += new Vector2((float)Math.Cos(turret.Rotation), (float)Math.Sin(turret.Rotation)) * turret.BarrelSprite.size.Y * turret.BarrelSprite.RelativeOrigin.Y * item.Scale * 0.9f;
108  }
109  }
110  else if (weapon != null)
111  {
112  Vector2 barrelPos = FarseerPhysics.ConvertUnits.ToDisplayUnits(weapon.TransformedBarrelPos);
113  barrelPos.Y = -barrelPos.Y;
114  startPos += barrelPos;
115  }
116  }
117  Vector2 endPos = new Vector2(target.DrawPosition.X, target.DrawPosition.Y);
118  Vector2 flippedPos = target.Sprite.size * target.Scale * (Origin - new Vector2(0.5f));
119  if (target.body.Dir < 0.0f)
120  {
121  flippedPos.X = -flippedPos.X;
122  }
123  endPos += Vector2.Transform(flippedPos, Matrix.CreateRotationZ(target.body.Rotation));
124  endPos.Y = -endPos.Y;
125 
126  if (Snapped)

128  float snapState = 1.0f - snapTimer / SnapAnimDuration;

142  }

147  }
148
