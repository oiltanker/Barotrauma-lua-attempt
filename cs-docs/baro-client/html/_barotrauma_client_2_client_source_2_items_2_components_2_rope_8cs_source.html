91 public void Draw(SpriteBatch spriteBatch, bool editing, float itemDepth = -1)
92 {
93 if (target == null || target.Removed) { return; }
94 if (target.ParentInventory != null) { return; }
95
96 Vector2 startPos = GetSourcePos();
97 startPos.Y = -startPos.Y;
98 if (source is Item sourceItem && !sourceItem.Removed)
99 {
100 var turret = sourceItem.GetComponent<Turret>();
101 var weapon = sourceItem.GetComponent<RangedWeapon>();
102 if (turret != null)
103 {
104 startPos = new Vector2(sourceItem.WorldRect.X + turret.TransformedBarrelPos.X, -(sourceItem.WorldRect.Y - turret.TransformedBarrelPos.Y));
105 if (turret.BarrelSprite != null)
106 {
107 startPos += new Vector2((float)Math.Cos(turret.Rotation), (float)Math.Sin(turret.Rotation)) * turret.BarrelSprite.size.Y * turret.BarrelSprite.RelativeOrigin.Y * item.Scale * 0.9f;
108 }
109 }
110 else if (weapon != null)
111 {
112 Vector2 barrelPos = FarseerPhysics.ConvertUnits.ToDisplayUnits(weapon.TransformedBarrelPos);
113 barrelPos.Y = -barrelPos.Y;
114 startPos += barrelPos;
115 }
116 }
117 Vector2 endPos = new Vector2(target.DrawPosition.X, target.DrawPosition.Y);
118 Vector2 flippedPos = target.Sprite.size * target.Scale * (Origin - new Vector2(0.5f));
119 if (target.body.Dir < 0.0f)
120 {
121 flippedPos.X = -flippedPos.X;
122 }
123 endPos += Vector2.Transform(flippedPos, Matrix.CreateRotationZ(target.body.Rotation));
124 endPos.Y = -endPos.Y;
125
126 if (Snapped)
127 {
128 float snapState = 1.0f - snapTimer / SnapAnimDuration;
129 Vector2 diff = target.DrawPosition - new Vector2(startPos.X, -startPos.Y);
130 diff.Y = -diff.Y;
131
132 int width = (int)(SpriteWidth * snapState);
133 if (width > 0.0f)
134 {
135 float positionMultiplier = snapState;
136 if (BreakFromMiddle)
137 {
138 positionMultiplier /= 2;
139 DrawRope(spriteBatch, endPos - diff * positionMultiplier, endPos, width);
140 }
141 DrawRope(spriteBatch, startPos, startPos + diff * positionMultiplier, width);
142 }
143 }
144 else
145 {
146 DrawRope(spriteBatch, startPos, endPos, SpriteWidth);
147 }
148
149 if (startSprite != null || endSprite != null)
150 {
151 Vector2 dir = endPos - startPos;
152 float angle = (float)Math.Atan2(dir.Y, dir.X);
153 if (startSprite != null)
154 {
155 float depth = Math.Min(item.GetDrawDepth() + (startSprite.Depth - item.Sprite.Depth), 0.999f);
156 startSprite?.Draw(spriteBatch, startPos, SpriteColor, angle, depth: depth);
157 }
158 if (endSprite != null && (!Snapped || BreakFromMiddle))
159 {
160 float depth = Math.Min(item.GetDrawDepth() + (endSprite.Depth - item.Sprite.Depth), 0.999f);
161 endSprite?.Draw(spriteBatch, endPos, SpriteColor, angle, depth: depth);
162 }
163 }
164 }

166 private void DrawRope(SpriteBatch spriteBatch, Vector2 startPos, Vector2 endPos, int width)
167 {
168 float depth = sprite == null ?
169 item.Sprite.Depth + 0.001f :
170 Math.Min(item.GetDrawDepth() + (sprite.Depth - item.Sprite.Depth), 0.999f);
171
172 if (sprite?.Texture == null)
173 {
174 GUI.DrawLine(spriteBatch,
175 startPos,
176 endPos,
177 SpriteColor, depth: depth, width: width);
178 return;
179 }
180
181 if (Tile)
182 {
183 float length = Vector2.Distance(startPos, endPos);
184 Vector2 dir = (endPos - startPos) / length;
185 float x;
186 for (x = 0.0f; x <= length - sprite.size.X; x += sprite.size.X)
187 {
188 GUI.DrawLine(spriteBatch, sprite,
189 startPos + dir * (x - 5.0f),
190 startPos + dir * (x + sprite.size.X),
191 SpriteColor, depth: depth, width: width);
192 }
193 float leftOver = length - x;
194 if (leftOver > 0.0f)
195 {
196 GUI.DrawLine(spriteBatch, sprite,
197 startPos + dir * (x - 5.0f),
198 endPos,
199 SpriteColor, depth: depth, width: width);
200 }
201 }
202 else
203 {
204 GUI.DrawLine(spriteBatch, sprite,
205 startPos,
206 endPos,
207 SpriteColor, depth: depth, width: width);
208 }

211 public void ClientEventRead(IReadMessage msg, float sendingTime)
212 {
213 snapped = msg.ReadBoolean();
214
215 if (!snapped)
216 {
217 UInt16 targetId = msg.ReadUInt16();
218 UInt16 sourceId = msg.ReadUInt16();
219 byte limbIndex = msg.ReadByte();
220
221 Item target = Entity.FindEntityByID(targetId) as Item;
222 if (target == null) { return; }
223 var source = Entity.FindEntityByID(sourceId);
224 if (source is Character sourceCharacter && limbIndex >= 0 && limbIndex < sourceCharacter.AnimController.Limbs.Length)
225 {
226 Limb sourceLimb = sourceCharacter.AnimController.Limbs[limbIndex];
227 Attach(sourceLimb, target);
228 }
229 else if (source is ISpatialEntity spatialEntity)
230 {
231 Attach(spatialEntity, target);
232 }
233 }
