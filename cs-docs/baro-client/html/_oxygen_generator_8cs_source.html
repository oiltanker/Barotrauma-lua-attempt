88  private void GetVents()
89  {

91  ventList ??= new List<(Vent vent, float hullVolume)>();
92  ventList.Clear();
93  foreach (MapEntity entity in item.linkedTo)

97  Vent vent = linkedItem.GetComponent<Vent>();
98  if (vent?.Item.CurrentHull == null) { continue; }

100  totalHullVolume += vent.Item.CurrentHull.Volume;
101  ventList.Add((vent, vent.Item.CurrentHull.Volume));
102  }
103 
104  for (int i = 0; i < ventList.Count; i++)
105  {
106  Vent vent = ventList[i].vent;
107  foreach (Hull connectedHull in vent.Item.CurrentHull.GetConnectedHulls(includingThis: false, searchDepth: 3, ignoreClosedGaps: true))
108  {
109  //another vent in the connected hull -> don't add it to this vent's total hull volume
110  if (ventList.Any(v => v.vent != vent && v.vent.Item.CurrentHull == connectedHull)) { continue; }
111  totalHullVolume += connectedHull.Volume;
112  ventList[i] = (ventList[i].vent, ventList[i].hullVolume + connectedHull.Volume);
113  }
114  }
