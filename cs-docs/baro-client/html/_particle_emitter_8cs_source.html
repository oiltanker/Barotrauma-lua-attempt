139 public void Emit(float deltaTime, Vector2 position, Hull hullGuess = null, float angle = 0.0f, float particleRotation = 0.0f, float velocityMultiplier = 1.0f, float sizeMultiplier = 1.0f, float amountMultiplier = 1.0f, Color? colorMultiplier = null, ParticlePrefab overrideParticle = null, bool mirrorAngle = false, Tuple<Vector2, Vector2> tracerPoints = null)
140 {
141 if (GameMain.Client?.MidRoundSyncing ?? false) { return; }
142
143 if (initialDelay < Prefab.Properties.InitialDelay)
144 {
145 initialDelay += deltaTime;
146 return;
147 }
148
149 emitTimer += deltaTime * amountMultiplier;
150 burstEmitTimer -= deltaTime;
151
152 if (Prefab.Properties.EmitAcrossRayInterval > 0.0f && tracerPoints != null)
153 {
154 Vector2 dir = tracerPoints.Item2 - tracerPoints.Item1;
155 if (dir.LengthSquared() > 0.001f)
156 {
157 float dist = dir.Length();
158 dir /= dist;
159 for (float z = 0.0f; z < dist; z += Prefab.Properties.EmitAcrossRayInterval)
160 {
161 Vector2 pos = tracerPoints.Item1 + dir * z;
162 Emit(pos, hullGuess, angle, particleRotation, velocityMultiplier, sizeMultiplier, colorMultiplier, overrideParticle, mirrorAngle, tracerPoints: null);
163 }
164 }
165 }
166
167 if (Prefab.Properties.ParticlesPerSecond > 0)
168 {
169 float emitInterval = 1.0f / Prefab.Properties.ParticlesPerSecond;
170 while (emitTimer > emitInterval)
171 {
172 Emit(position, hullGuess, angle, particleRotation, velocityMultiplier, sizeMultiplier, colorMultiplier, overrideParticle, mirrorAngle, tracerPoints: tracerPoints);
173 emitTimer -= emitInterval;
174 }
175 }
176
177 if (burstEmitTimer > 0.0f) { return; }
178
179 burstEmitTimer = Prefab.Properties.EmitInterval;
180 for (int i = 0; i < Prefab.Properties.ParticleAmount * amountMultiplier; i++)
181 {
182 Emit(position, hullGuess, angle, particleRotation, velocityMultiplier, sizeMultiplier, colorMultiplier, overrideParticle, tracerPoints: tracerPoints);
183 }
184 }

186 private void Emit(Vector2 position, Hull hullGuess, float angle, float particleRotation, float velocityMultiplier, float sizeMultiplier, Color? colorMultiplier = null, ParticlePrefab overrideParticle = null, bool mirrorAngle = false, Tuple<Vector2, Vector2> tracerPoints = null)
187 {
188 var particlePrefab = overrideParticle ?? Prefab.ParticlePrefab;
189 if (particlePrefab == null) { return; }
190
191 Vector2 velocity = Vector2.Zero;
192 if (!MathUtils.NearlyEqual(Prefab.Properties.VelocityMax * velocityMultiplier, 0.0f) || !MathUtils.NearlyEqual(Prefab.Properties.DistanceMax, 0.0f))
193 {
194 angle += Rand.Range(Prefab.Properties.AngleMinRad, Prefab.Properties.AngleMaxRad) * (mirrorAngle ? -1 : 1);
195 Vector2 dir = new Vector2((float)Math.Cos(angle), (float)Math.Sin(angle));
196 velocity = dir * Rand.Range(Prefab.Properties.VelocityMin, Prefab.Properties.VelocityMax) * velocityMultiplier;
197 position += dir * Rand.Range(Prefab.Properties.DistanceMin, Prefab.Properties.DistanceMax);
198 }
199
200 var particle = GameMain.ParticleManager.CreateParticle(particlePrefab, position, velocity, particleRotation, hullGuess, Prefab.DrawOnTop, tracerPoints: tracerPoints);
201
202 if (particle != null)
203 {
204 particle.Size *= Rand.Range(Prefab.Properties.ScaleMin, Prefab.Properties.ScaleMax) * sizeMultiplier;
205 particle.Size *= Prefab.Properties.ScaleMultiplier;
206 particle.HighQualityCollisionDetection = Prefab.Properties.HighQualityCollisionDetection;
207 if (colorMultiplier.HasValue)
208 {
209 particle.ColorMultiplier = colorMultiplier.Value.ToVector4();
210 }
211 else if (Prefab.Properties.ColorMultiplier != Color.White)
212 {
213 particle.ColorMultiplier = Prefab.Properties.ColorMultiplier.ToVector4();
214 }
215 }

218 public Rectangle CalculateParticleBounds(Vector2 startPosition)
219 {
220 Rectangle bounds = new Rectangle((int)startPosition.X, (int)startPosition.Y, (int)startPosition.X, (int)startPosition.Y);
221
222 for (float angle = Prefab.Properties.AngleMinRad; angle <= Prefab.Properties.AngleMaxRad; angle += 0.1f)
223 {
224 Vector2 velocity = new Vector2((float)Math.Cos(angle), (float)Math.Sin(angle)) * Prefab.Properties.VelocityMax;
225 Vector2 endPosition = Prefab.ParticlePrefab.CalculateEndPosition(startPosition, velocity);
226
227 Vector2 endSize = Prefab.ParticlePrefab.CalculateEndSize();
228 float spriteExtent = 0.0f;
229 foreach (Sprite sprite in Prefab.ParticlePrefab.Sprites)
230 {
231 if (sprite is SpriteSheet spriteSheet)
232 {
233 spriteExtent = Math.Max(spriteExtent, Math.Max(spriteSheet.FrameSize.X * endSize.X, spriteSheet.FrameSize.Y * endSize.Y));
234 }
235 else
236 {
237 spriteExtent = Math.Max(spriteExtent, Math.Max(sprite.size.X * endSize.X, sprite.size.Y * endSize.Y));
238 }
239 }
240
241 bounds = new Rectangle(
242 (int)Math.Min(bounds.X, endPosition.X - Prefab.Properties.DistanceMax - spriteExtent / 2),
243 (int)Math.Min(bounds.Y, endPosition.Y - Prefab.Properties.DistanceMax - spriteExtent / 2),
244 (int)Math.Max(bounds.X, endPosition.X + Prefab.Properties.DistanceMax + spriteExtent / 2),
245 (int)Math.Max(bounds.Y, endPosition.Y + Prefab.Properties.DistanceMax + spriteExtent / 2));
246 }
247
248 bounds = new Rectangle(bounds.X, bounds.Y, bounds.Width - bounds.X, bounds.Height - bounds.Y);
249
250 return bounds;
251 }
