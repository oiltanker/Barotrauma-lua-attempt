409  private void DrawTempMeter(SpriteBatch spriteBatch, GUICustomComponent container)
410  {
411  Vector2 meterPos = new Vector2(container.Rect.X, container.Rect.Y);
412  Vector2 meterScale = new Vector2(container.Rect.Width / (float)tempMeterFrame.SourceRect.Width, container.Rect.Height / (float)tempMeterFrame.SourceRect.Height);
413  tempMeterFrame.Draw(spriteBatch, meterPos, Color.White, tempMeterFrame.Origin, 0.0f, scale: meterScale);
414 
415  float tempFill = temperature / 100.0f;
416  float meterBarScale = container.Rect.Width / (float)tempMeterBar.SourceRect.Width;
417  Vector2 meterBarPos = new Vector2(container.Center.X, container.Rect.Bottom - tempMeterBar.size.Y * meterBarScale - (int)(5 * GUI.yScale));
418  while (meterBarPos.Y > container.Rect.Bottom + (int)(5 * GUI.yScale) - container.Rect.Height * tempFill)
419  {
420  float tempRatio = 1.0f - ((meterBarPos.Y - container.Rect.Y) / container.Rect.Height);
421  Color color = ToolBox.GradientLerp(tempRatio, coldColor, optimalRangeColor, warmColor, hotColor);
422  tempMeterBar.Draw(spriteBatch, meterBarPos, color: color, scale: meterBarScale);
423  int spacing = 2;
424  meterBarPos.Y -= tempMeterBar.size.Y * meterBarScale + spacing;
425  }
426 
427  if (temperature > optimalTemperature.Y)
428  {
429  GUI.DrawRectangle(spriteBatch,
430  meterPos,
431  new Vector2(container.Rect.Width, (container.Rect.Bottom - container.Rect.Height * optimalTemperature.Y / 100.0f) - container.Rect.Y),
432  warningColor * (float)Math.Sin(Timing.TotalTime * 5.0f) * 0.7f, isFilled: true);
433  }
434  if (temperature < optimalTemperature.X)
435  {
436  GUI.DrawRectangle(spriteBatch,
437  new Vector2(meterPos.X, container.Rect.Bottom - container.Rect.Height * optimalTemperature.X / 100.0f),
438  new Vector2(container.Rect.Width, container.Rect.Bottom - (container.Rect.Bottom - container.Rect.Height * optimalTemperature.X / 100.0f)),
439  warningColor * (float)Math.Sin(Timing.TotalTime * 5.0f) * 0.7f, isFilled: true);
440  }
441 
442  float tempRangeIndicatorScale = container.Rect.Width / (float)tempRangeIndicator.SourceRect.Width;
443  tempRangeIndicator.Draw(spriteBatch, new Vector2(container.Center.X, container.Rect.Bottom - container.Rect.Height * optimalTemperature.X / 100.0f), Color.White, tempRangeIndicator.Origin, 0, scale: tempRangeIndicatorScale);
444  tempRangeIndicator.Draw(spriteBatch, new Vector2(container.Center.X, container.Rect.Bottom - container.Rect.Height * optimalTemperature.Y / 100.0f), Color.White, tempRangeIndicator.Origin, 0, scale: tempRangeIndicatorScale);
445  }

477  private void DrawFissionRateMeter(SpriteBatch spriteBatch, GUICustomComponent container)
478  {
479  if (item.Removed) { return; }
480 
481  Rectangle prevScissorRect = spriteBatch.GraphicsDevice.ScissorRectangle;
482  spriteBatch.End();
483  spriteBatch.GraphicsDevice.ScissorRectangle = container.Rect;
484  spriteBatch.Begin(SpriteSortMode.Deferred, rasterizerState: GameMain.ScissorTestEnable);
485 
486  //make the pointer jitter a bit if it's at the upper limit of the fission rate
487  float jitter = 0.0f;
488  if (FissionRate > allowedFissionRate.Y - 5.0f)
489  {
490  float jitterAmount = Math.Min(TargetFissionRate - allowedFissionRate.Y, 10.0f);
491  float t = graphTimer / updateGraphInterval;
492 
493  jitter = (PerlinNoise.GetPerlin(t * 0.5f, t * 0.1f) - 0.5f) * jitterAmount;
494  }
495 
496  DrawMeter(spriteBatch, container.Rect,
497  fissionRateMeter, FissionRate + jitter, new Vector2(0.0f, 100.0f), optimalFissionRate, allowedFissionRate);
498 
499  spriteBatch.End();
500  spriteBatch.GraphicsDevice.ScissorRectangle = prevScissorRect;
501  spriteBatch.Begin(SpriteSortMode.Deferred);
502  }
