36 public override void OnItemLoaded()
37 {
38 base.OnItemLoaded();
39
40 if (Signals == null)
41 {
42 Signals = new string[RequiredSignalCount];
43 for (int i = 0; i < RequiredSignalCount; i++)
44 {
45 Signals[i] = string.Empty;
46 }
47 }
48 else if (Signals.Length != RequiredSignalCount)
49 {
50 string[] newSignals = new string[RequiredSignalCount];
51 if (Signals.Length < RequiredSignalCount)
52 {
53 Signals.CopyTo(newSignals, 0);
54 for (int i = Signals.Length; i < RequiredSignalCount; i++)
55 {
56 newSignals[i] = string.Empty;
57 }
58 }
59 else
60 {
61 for (int i = 0; i < RequiredSignalCount; i++)
62 {
63 newSignals[i] = Signals[i];
64 }
65 }
66 Signals = newSignals;
67 }
68
69 ActivatingItemPrefabs.Clear();
70 if (!string.IsNullOrEmpty(ActivatingItems))
71 {
72 foreach (var activatingItem in ActivatingItems.Split(','))
73 {
74 if (MapEntityPrefab.Find(null, identifier: activatingItem, showErrorMessages: false) is ItemPrefab prefab)
75 {
76 ActivatingItemPrefabs.Add(prefab);
77 }
78 else
79 {
80 ItemPrefab.Prefabs.Where(p => p.Tags.Any(t => t == activatingItem))
81 .ForEach(p => ActivatingItemPrefabs.Add(p));
82 }
83 }
84 if (ActivatingItemPrefabs.None())
85 {
86 DebugConsole.ThrowError($"Error in item \"{item.Name}\": no activating item prefabs found with identifiers or tags \"{ActivatingItems}\"");
87 }
88 }
89
90 var containers = item.GetComponents<ItemContainer>();
91 if (containers.Count() != 1)
92 {
93 DebugConsole.ThrowError($"Error in item \"{item.Name}\": the ButtonTerminal component requires exactly one ItemContainer component!");
94 return;
95 }
