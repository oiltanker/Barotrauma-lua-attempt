27 public static List<VertexPositionTexture> GenerateWallEdgeVertices(List<VoronoiCell> cells, Level level, float zCoord)
28 {
29 float outWardThickness = level.GenerationParams.WallEdgeExpandOutwardsAmount;
30
31 List<VertexPositionTexture> vertices = new List<VertexPositionTexture>();
32 foreach (VoronoiCell cell in cells)
33 {
34 Vector2 minVert = cell.Edges[0].Point1;
35 Vector2 maxVert = cell.Edges[0].Point1;
36 float circumference = 0.0f;
37 foreach (GraphEdge edge in cell.Edges)
38 {
39 circumference += Vector2.Distance(edge.Point1, edge.Point2);
40 minVert = new Vector2(
41 Math.Min(minVert.X, edge.Point1.X),
42 Math.Min(minVert.Y, edge.Point1.Y));
43 maxVert = new Vector2(
44 Math.Max(maxVert.X, edge.Point1.X),
45 Math.Max(maxVert.Y, edge.Point1.Y));
46 }
47 Vector2 center = (minVert + maxVert) / 2;
48 foreach (GraphEdge edge in cell.Edges)
49 {
50 if (!edge.IsSolid) { continue; }
51
52 GraphEdge leftEdge = cell.Edges.Find(e => e != edge && (edge.Point1.NearlyEquals(e.Point1) || edge.Point1.NearlyEquals(e.Point2)));
53 var leftAdjacentCell = leftEdge?.AdjacentCell(cell);
54 if (leftAdjacentCell != null)
55 {
56 var adjEdge = leftAdjacentCell.Edges.Find(e => e != leftEdge && e.IsSolid && (edge.Point1.NearlyEquals(e.Point1) || edge.Point1.NearlyEquals(e.Point2)));
57 if (adjEdge != null) { leftEdge = adjEdge; }
58 }
59
60 GraphEdge rightEdge = cell.Edges.Find(e => e != edge && (edge.Point2.NearlyEquals(e.Point1) || edge.Point2.NearlyEquals(e.Point2)));
61 var rightAdjacentCell = rightEdge?.AdjacentCell(cell);
62 if (rightAdjacentCell != null)
63 {
64 var adjEdge = rightAdjacentCell.Edges.Find(e => e != rightEdge && e.IsSolid && (edge.Point2.NearlyEquals(e.Point1) || edge.Point2.NearlyEquals(e.Point2)));
65 if (adjEdge != null) { rightEdge = adjEdge; }
66 }
67
68 Vector2 leftNormal = Vector2.Zero, rightNormal = Vector2.Zero;
69
70 float inwardThickness1 = level.GenerationParams.WallEdgeExpandInwardsAmount;
71 float inwardThickness2 = level.GenerationParams.WallEdgeExpandInwardsAmount;
72 if (leftEdge != null && !leftEdge.IsSolid)
73 {
74 leftNormal = edge.Point1.NearlyEquals(leftEdge.Point1) ?
75 Vector2.Normalize(leftEdge.Point2 - leftEdge.Point1) :
76 Vector2.Normalize(leftEdge.Point1 - leftEdge.Point2);
77 }
78 else if (leftEdge != null)
79 {
80 leftNormal = -Vector2.Normalize(edge.GetNormal(cell) + leftEdge.GetNormal(leftAdjacentCell ?? cell));
81 if (!MathUtils.IsValid(leftNormal)) { leftNormal = -edge.GetNormal(cell); }
82 }
83 else
84 {
85 leftNormal = Vector2.Normalize(cell.Center - edge.Point1);
86 }
87 inwardThickness1 = Math.Min(Vector2.Distance(edge.Point1, cell.Center), inwardThickness1);
88
89 if (!MathUtils.IsValid(leftNormal))
90 {
91#if DEBUG
92 DebugConsole.ThrowError("Invalid left normal");
93#endif
94 GameAnalyticsManager.AddErrorEventOnce("CaveGenerator.GenerateWallShapes:InvalidLeftNormal:" + level.Seed,
95 GameAnalyticsManager.ErrorSeverity.Warning,
96 "Invalid left normal (leftedge: " + leftEdge + ", rightedge: " + rightEdge + ", normal: " + leftNormal + ", seed: " + level.Seed + ")");
97
98 if (cell.Body != null)
99 {
100 if (GameMain.World.BodyList.Contains(cell.Body)) { GameMain.World.Remove(cell.Body); }
101 cell.Body = null;
102 }
103 leftNormal = Vector2.UnitX;
104 break;
105 }
106
107 if (rightEdge != null && !rightEdge.IsSolid)
108 {
109 rightNormal = edge.Point2.NearlyEquals(rightEdge.Point1) ?
110 Vector2.Normalize(rightEdge.Point2 - rightEdge.Point1) :
111 Vector2.Normalize(rightEdge.Point1 - rightEdge.Point2);
112 }
113 else if (rightEdge != null)
114 {
115 rightNormal = -Vector2.Normalize(edge.GetNormal(cell) + rightEdge.GetNormal(rightAdjacentCell ?? cell));
116 if (!MathUtils.IsValid(rightNormal)) { rightNormal = -edge.GetNormal(cell); }
117 }
118 else
119 {
120 rightNormal = Vector2.Normalize(cell.Center - edge.Point2);
121 }
122 inwardThickness2 = Math.Min(Vector2.Distance(edge.Point2, cell.Center), inwardThickness2);
123
124 if (!MathUtils.IsValid(rightNormal))
125 {
126#if DEBUG
127 DebugConsole.ThrowError("Invalid right normal");
128#endif
129 GameAnalyticsManager.AddErrorEventOnce("CaveGenerator.GenerateWallShapes:InvalidRightNormal:" + level.Seed,
130 GameAnalyticsManager.ErrorSeverity.Warning,
131 "Invalid right normal (leftedge: " + leftEdge + ", rightedge: " + rightEdge + ", normal: " + rightNormal + ", seed: " + level.Seed + ")");
132
133 if (cell.Body != null)
134 {
135 if (GameMain.World.BodyList.Contains(cell.Body)) { GameMain.World.Remove(cell.Body); }
136 cell.Body = null;
137 }
138 rightNormal = Vector2.UnitX;
139 break;
140 }
141
142 float point1UV = MathUtils.WrapAngleTwoPi(MathUtils.VectorToAngle(edge.Point1 - center));
143 float point2UV = MathUtils.WrapAngleTwoPi(MathUtils.VectorToAngle(edge.Point2 - center));
144 //handle wrapping around 0/360
145 if (point1UV - point2UV > MathHelper.Pi)
146 {
147 point1UV -= MathHelper.TwoPi;
148 }
149 int textureRepeatCount = (int)Math.Max(circumference / 2 / level.GenerationParams.WallEdgeTextureWidth, 1);
150 point1UV = point1UV / MathHelper.TwoPi * textureRepeatCount;
151 point2UV = point2UV / MathHelper.TwoPi * textureRepeatCount;
152
153 for (int i = 0; i < 2; i++)
154 {
155 Vector2[] verts = new Vector2[3];
156 VertexPositionTexture[] vertPos = new VertexPositionTexture[3];
157
158 if (i == 0)
159 {
160 verts[0] = edge.Point1 - leftNormal * outWardThickness;
161 verts[1] = edge.Point2 - rightNormal * outWardThickness;
162 verts[2] = edge.Point1 + leftNormal * inwardThickness1;
163
164 vertPos[0] = new VertexPositionTexture(new Vector3(verts[0], zCoord), new Vector2(point1UV, 0.0f));
165 vertPos[1] = new VertexPositionTexture(new Vector3(verts[1], zCoord), new Vector2(point2UV, 0.0f));
166 vertPos[2] = new VertexPositionTexture(new Vector3(verts[2], zCoord), new Vector2(point1UV, 1.0f));
167 }
168 else
169 {
170
171 verts[0] = edge.Point1 + leftNormal * inwardThickness1;
172 verts[1] = edge.Point2 - rightNormal * outWardThickness;
173 verts[2] = edge.Point2 + rightNormal * inwardThickness2;
174
175 vertPos[0] = new VertexPositionTexture(new Vector3(verts[0], zCoord), new Vector2(point1UV, 1.0f));
176 vertPos[1] = new VertexPositionTexture(new Vector3(verts[1], zCoord), new Vector2(point2UV, 0.0f));
177 vertPos[2] = new VertexPositionTexture(new Vector3(verts[2], zCoord), new Vector2(point2UV, 1.0f));
178 }
179 vertices.AddRange(vertPos);
180 }
181 }
182 }
183
184 return vertices;
185 }
