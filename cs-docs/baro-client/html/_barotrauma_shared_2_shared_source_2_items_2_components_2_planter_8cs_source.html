158 public override bool OnPicked(Character character)
159 {
160 if (container?.Inventory == null) { return false; }
161
162 SuitablePlantItem plantItem = GetSuitableItem(character);
163 if (plantItem.IsNull())
164 {
165 return TryHarvest(character);
166 }
167
168 switch (plantItem.Type)
169 {
170 case PlantItemType.Seed:
171 ApplyStatusEffects(ActionType.OnPicked, 1.0f, character);
172 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
173 {
174 return container.Inventory.TryPutItem(plantItem.Item, character);
175 }
176 else
177 {
178 //let the server handle moving the item
179 return false;
180 }
181 case PlantItemType.Fertilizer when plantItem.Item != null:
182 float canAdd = FertilizerCapacity - Fertilizer;
183 float maxAvailable = plantItem.Item.Condition;
184 float toAdd = Math.Min(canAdd, maxAvailable);
185 plantItem.Item.Condition -= toAdd;
186 fertilizer += toAdd;
187#if CLIENT
188 character.UpdateHUDProgressBar(this, Item.DrawPosition, Fertilizer / FertilizerCapacity, Color.SaddleBrown, Color.SaddleBrown, "entityname.fertilizer");
189#endif
190 ApplyStatusEffects(ActionType.OnPicked, 1.0f, character);
191 return false;
192 }
193
194 return false;
195 }

202 private bool TryHarvest(Character? character)
203 {
204 Debug.Assert(container != null, "Tried to harvest a planter without an item container.");
205
206 bool anyDecayed = GrowableSeeds.Any(s => s is { } seed && (seed.Decayed || seed.FullyGrown));
207 for (var i = 0; i < GrowableSeeds.Length; i++)
208 {
209 Growable? seed = GrowableSeeds[i];
210 if (seed == null) { continue; }
211
212 if (!anyDecayed || seed.Decayed || seed.FullyGrown)
213 {
214 container?.Inventory.RemoveItem(seed.Item);
215 Entity.Spawner?.AddItemToRemoveQueue(seed.Item);
216 GrowableSeeds[i] = null;
217 ApplyStatusEffects(ActionType.OnPicked, 1.0f, character);
218 return true;
219 }
220 }
221
222 return false;
223 }

225 public override void Update(float deltaTime, Camera cam)
226 {
227 base.Update(deltaTime, cam);
228
229#if CLIENT
230 if (lightComponent != null)
231 {
232 bool hasSeed = false;
233 foreach (Growable? seed in GrowableSeeds) { hasSeed |= seed != null; }
234
235 lightComponent.Light.Enabled = hasSeed;
236 }
237#endif
238
239 if (container?.Inventory == null) { return; }
240
241 for (var i = 0; i < container.Inventory.Capacity; i++)
242 {
243 if (i < 0 || GrowableSeeds.Length <= i) { continue; }
244
245 Item containedItem = container.Inventory.GetItemAt(i);
246 Growable? growable = containedItem?.GetComponent<Growable>();
247
248 if (growable != null)
249 {
250 GrowableSeeds[i] = growable;
251 growable.IsActive = true;
252 }
253 else
254 {
255 if (GrowableSeeds[i] is { } oldGrowable)
256 {
257 // Kill the plant if it's somehow removed
258 oldGrowable.Decayed = true;
259 oldGrowable.IsActive = false;
260 }
261
262 GrowableSeeds[i] = null;
263 }
264 }
265
266 // server handles this
267 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
268
269 float delay = GrowthTickDelay;
270 if (Fertilizer > 0)
271 {
272 delay /= 2f;
273 Fertilizer -= deltaTime / 10f;
274 }
275
276 if (growthTickTimer > delay)
277 {
278 for (var i = 0; i < GrowableSeeds.Length; i++)
279 {
280 PlantSlot slot = PlantSlots.ContainsKey(i) ? PlantSlots[i] : NullSlot;
281 Growable? seed = GrowableSeeds[i];
282 seed?.OnGrowthTick(this, slot);
283 }
284
285 growthTickTimer = 0;
286 }
287 else if (Item.ParentInventory == null)
288 {
289 if (item.GetComponent<Holdable>() is { } holdable)
290 {
291 if (holdable.Attachable && !holdable.Attached)
292 {
293 return;
294 }
295 }
296
297 growthTickTimer += deltaTime;
298 }
299 }

301 private SuitablePlantItem GetSuitableItem(Character character)
302 {
303 foreach (Item heldItem in character.HeldItems)
304 {
305 if (container?.Inventory != null && container.Inventory.CanBePut(heldItem))
306 {
307 if (heldItem.GetComponent<Growable>() != null && SuitableSeeds.Any(ri => ri.MatchesItem(heldItem)))
308 {
309 return new SuitablePlantItem(heldItem, PlantItemType.Seed, MsgPlanting);
310 }
311 }
312
313 if (SuitableFertilizer.Any(ri => ri.MatchesItem(heldItem)))
314 {
315 return new SuitablePlantItem(heldItem, PlantItemType.Fertilizer, MsgFertilizing);
316 }
317 }
318
319 return NullItem;
320 }
