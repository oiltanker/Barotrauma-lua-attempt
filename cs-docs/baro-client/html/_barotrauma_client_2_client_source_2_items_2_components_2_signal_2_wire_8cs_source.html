35 private void RecalculateVertices(Wire wire, float width)
36 {
37 if (MathUtils.NearlyEqual(cachedWidth, width)) { return; }
38 cachedWidth = width;
39
40 vertices = new VertexPositionColorTexture[4];
41
42 Vector2 expandDir = start-end;
43 expandDir.Normalize();
44 float temp = expandDir.X;
45 expandDir.X = -expandDir.Y;
46 expandDir.Y = -temp;
47
48 Rectangle srcRect = wire.wireSprite.SourceRect;
49
50 expandDir *= width * srcRect.Height * 0.5f;
51
52 Vector2 rectLocation = srcRect.Location.ToVector2();
53 Vector2 rectSize = srcRect.Size.ToVector2();
54 Vector2 textureSize = new Vector2(wire.wireSprite.Texture.Width, wire.wireSprite.Texture.Height);
55
56 Vector2 topLeftUv = rectLocation / textureSize;
57 Vector2 bottomRightUv = (rectLocation + rectSize) / textureSize;
58
59 Vector2 invStart = new Vector2(start.X, -start.Y);
60 Vector2 invEnd = new Vector2(end.X, -end.Y);
61
62 vertices[0] = new VertexPositionColorTexture(new Vector3(invStart + expandDir, 0f), Color.White, topLeftUv);
63 vertices[2] = new VertexPositionColorTexture(new Vector3(invEnd + expandDir, 0f), Color.White, new Vector2(bottomRightUv.X, topLeftUv.Y));
64 vertices[1] = new VertexPositionColorTexture(new Vector3(invStart - expandDir, 0f), Color.White, new Vector2(topLeftUv.X, bottomRightUv.Y));
65 vertices[3] = new VertexPositionColorTexture(new Vector3(invEnd - expandDir, 0f), Color.White, bottomRightUv);
66
67 shiftedVertices = (VertexPositionColorTexture[])vertices.Clone();
68 }

72 if (width <= 0f) { return; }

80 shiftedVertices[i].Position.Y -= offset.Y;

83 shiftedVertices,
84 depth);

311 Wire equippedWire = Character.Controlled.HeldItems.FirstOrDefault(it => it.GetComponent<Wire>() != null)?.GetComponent<Wire>();

321 //dragging a node of some wire
322 if (draggingWire != null && !doubleClicked)
323 {
324 if (Character.Controlled != null)
325 {
326 Character.Controlled.FocusedItem = null;
327 Character.Controlled.ResetInteract = true;
328 Character.Controlled.ClearInputs();
329 }

331 if (!PlayerInput.PrimaryMouseButtonHeld())
332 {
333 draggingWire = null;
334 selectedNodeIndex = null;
335 }
336 //update dragging
337 else
338 {
339 MapEntity.DisableSelect = true;
340
341 Submarine sub = draggingWire.item.Submarine;
342 if (draggingWire.connections[0] != null && draggingWire.connections[0].Item.Submarine != null) sub = draggingWire.connections[0].Item.Submarine;
343 if (draggingWire.connections[1] != null && draggingWire.connections[1].Item.Submarine != null) sub = draggingWire.connections[1].Item.Submarine;
344
345 Vector2 nodeWorldPos = GameMain.SubEditorScreen.Cam.ScreenToWorld(PlayerInput.MousePosition);
346 if (sub != null)
347 {
348 nodeWorldPos = nodeWorldPos - sub.HiddenSubPosition - sub.Position;
349 }
350
351 if (selectedNodeIndex.HasValue && selectedNodeIndex.Value >= draggingWire.nodes.Count) { selectedNodeIndex = null; }
352 if (highlightedNodeIndex.HasValue && highlightedNodeIndex.Value >= draggingWire.nodes.Count) { highlightedNodeIndex = null; }
353
354 if (selectedNodeIndex.HasValue)
355 {
356 if (!PlayerInput.IsShiftDown())
357 {

359 nodeWorldPos.Y = MathUtils.Round(nodeWorldPos.Y, Submarine.GridSize.Y / 2.0f);
360 }
361
362 draggingWire.nodes[(int)selectedNodeIndex] = nodeWorldPos;
363 draggingWire.UpdateSections();
364 }
365 else
366 {
367 float dragDistance = Submarine.GridSize.X * Submarine.GridSize.Y;
368 dragDistance *= 0.5f;
369 if ((highlightedNodeIndex.HasValue && Vector2.DistanceSquared(nodeWorldPos, draggingWire.nodes[(int)highlightedNodeIndex]) >= dragDistance) ||
370 PlayerInput.IsShiftDown())
371 {
372 selectedNodeIndex = highlightedNodeIndex;
373 }
374 }
375
376 MapEntity.SelectEntity(draggingWire.item);
377 }
378
379 return;
380 }

382 bool updateHighlight = true;
383
384 //a wire has been selected -> check if we should start dragging one of the nodes
385 float nodeSelectDist = 10, sectionSelectDist = 5;
386 highlightedNodeIndex = null;
387 if (MapEntity.SelectedList.Count == 1 && MapEntity.SelectedList.FirstOrDefault() is Item selectedItem)

410 selectedWire.nodes.Insert(closestSectionIndex + 1, mousePos);
411 selectedWire.UpdateSections();
412 }
413 }
414 }
415 else
416 {
417 //check if close enough to a node
418 int closestIndex = selectedWire.GetClosestNodeIndex(mousePos, nodeSelectDist, out _);
419 if (closestIndex > -1)
420 {
421 highlightedNodeIndex = closestIndex;
422

429 //start dragging the node

431 {
432 if (Character.Controlled != null)

444 selectedWire.nodes.RemoveAt(closestIndex);

446 }
447 // if only one end of the wire is disconnect pick it back up with double click
448 else if (doubleClicked && equippedWire == null && Character.Controlled != null && selectedWire.connections.Any(conn => conn != null))
449 {
450 if (selectedWire.connections[0] == null && closestIndex == 0 || selectedWire.connections[1] == null && closestIndex == selectedWire.nodes.Count - 1)
451 {
452 selectedWire.IsActive = true;
453 selectedWire.nodes.RemoveAt(closestIndex);
454 selectedWire.UpdateSections();
455

457 if (closestIndex == 0)
458 {
459 selectedWire.nodes.Reverse();
460 selectedWire.connections[0] = selectedWire.connections[1];
461 selectedWire.connections[1] = null;
462 }
463
464 selectedWire.shouldClearConnections = false;
465 Character.Controlled.Inventory.TryPutItem(selectedWire.item, Character.Controlled, new List<InvSlotType> { InvSlotType.LeftHand, InvSlotType.RightHand });
466 foreach (var entity in MapEntity.mapEntityList)
467 {
468 if (entity is Item item)

470 item.GetComponent<ConnectionPanel>()?.DisconnectedWires.Remove(selectedWire);
471 }
472 }
473 MapEntity.SelectedList.Clear();
474 selectedWire.shouldClearConnections = true;
475 updateHighlight = false;
476 }
477 }
478 }
479 }
480 }
481 }
482
483 Wire highlighted = null;
484
485 //check which wire is highlighted with the cursor
486 if (GUI.MouseOn == null)
487 {
488 float closestDist = float.PositiveInfinity;
489 foreach (Wire w in wires)
490 {
491 Vector2 mousePos = GameMain.SubEditorScreen.Cam.ScreenToWorld(PlayerInput.MousePosition);
492 if (w.item.Submarine != null) { mousePos -= (w.item.Submarine.Position + w.item.Submarine.HiddenSubPosition); }
493
494 int highlightedNode = w.GetClosestNodeIndex(mousePos, highlighted == null ? nodeSelectDist : closestDist, out float dist);
495 if (highlightedNode > -1)
496 {
497 if (dist < closestDist)
498 {
499 highlightedNodeIndex = highlightedNode;
500 highlighted = w;
501 closestDist = dist;
502 }
503 }
504
505 if (w.GetClosestSectionIndex(mousePos, highlighted == null ? sectionSelectDist : closestDist, out dist) > -1)
506 {
507 //prefer nodes over sections
508 if (dist + nodeSelectDist * 0.5f < closestDist)
509 {
510 highlightedNodeIndex = null;
511 highlighted = w;
512 closestDist = dist + nodeSelectDist * 0.5f;
513 }
514 }
515 }
516 }
517
518 if (highlighted != null && updateHighlight)

520 highlighted.item.IsHighlighted = true;

529 public override void Move(Vector2 amount)
530 {
531 //only used in the sub editor, hence only in the client project
532 if (!item.IsSelected) { return; }
533
534 Vector2 wireNodeOffset = item.Submarine == null ? Vector2.Zero : item.Submarine.HiddenSubPosition + amount;
535 for (int i = 0; i < nodes.Count; i++)
536 {
537 if (i == 0 || i == nodes.Count - 1)
538 {
539 if (connections[0]?.Item != null && !connections[0].Item.IsSelected &&
540 (Submarine.RectContains(connections[0].Item.Rect, nodes[i] + wireNodeOffset) || Submarine.RectContains(connections[0].Item.Rect, nodes[i] + wireNodeOffset - amount)))
541 {
542 continue;
543 }
544 else if (connections[1]?.Item != null && !connections[1].Item.IsSelected &&
545 (Submarine.RectContains(connections[1].Item.Rect, nodes[i] + wireNodeOffset) || Submarine.RectContains(connections[1].Item.Rect, nodes[i] + wireNodeOffset - amount)))
546 {
547 continue;

549 }

551 }

568 public void ClientEventRead(IReadMessage msg, float sendingTime)
569 {
570 int eventIndex = msg.ReadRangedInteger(0, (int)Math.Ceiling(MaxNodeCount / (float)MaxNodesPerNetworkEvent));
571 int nodeCount = msg.ReadRangedInteger(0, MaxNodesPerNetworkEvent);
572 int nodeStartIndex = eventIndex * MaxNodesPerNetworkEvent;
573
574 Vector2[] nodePositions = new Vector2[nodeStartIndex + nodeCount];
575 for (int i = 0; i < nodes.Count && i < nodePositions.Length; i++)
576 {
577 nodePositions[i] = nodes[i];
578 }
579
580 for (int i = 0; i < nodeCount; i++)
581 {
582 nodePositions[nodeStartIndex + i] = new Vector2(msg.ReadSingle(), msg.ReadSingle());
583 }
584
585 if (nodePositions.Any(n => !MathUtils.IsValid(n)))
586 {
587 nodes.Clear();
588 return;
589 }
590
591 nodes = nodePositions.ToList();
592 UpdateSections();
593 Drawable = nodes.Any();
594 IsActive =
595 (connections[0] == null ^ connections[1] == null) &&
596 (item.ParentInventory is CharacterInventory characterInventory && ((characterInventory.Owner as Character)?.HasEquippedItem(item) ?? false));
597 }
