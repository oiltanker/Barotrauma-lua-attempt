175 private static Vector2[] CreateSector(double radius, int sides, float radians, float offset = 0)
176 {
177 //circle sectors need one extra point at the center
178 var points = new Vector2[radians < MathHelper.TwoPi ? sides + 1 : sides];
179 var step = radians / sides;
180
181 double theta = offset;
182 for (var i = 0; i < sides; i++)
183 {
184 points[i] = new Vector2((float)Math.Cos(theta), (float)Math.Sin(theta)) * (float)radius;
185 theta += step;
186 }
187
188 return points;
189 }

271 private Vector2[] GetTransformedVertices()
272 {
273 var newVertices = new Vector2[_localVertices.Length];
274 var isScaled = _scale != Vector2.One;
275
276 for (var i = 0; i < _localVertices.Length; i++)
277 {
278 var p = _localVertices[i];
279
280 if (isScaled)
281 p *= _scale;
282
283 // ReSharper disable once CompareOfFloatsByEqualityOperator
284 if (_rotation != 0)
285 {
286 var cos = (float)Math.Cos(_rotation);
287 var sin = (float)Math.Sin(_rotation);
288 p = new Vector2(cos * p.X - sin * p.Y, sin * p.X + cos * p.Y);
289 }
290
291 newVertices[i] = p + _offset;
292 }
293
294 return newVertices;

311 public bool Contains(float x, float y)
312 {
313 var intersects = 0;
314 var vertices = Vertices;
315
316 for (var i = 0; i < vertices.Length; i++)
317 {
318 var x1 = vertices[i].X;
319 var y1 = vertices[i].Y;
320 var x2 = vertices[(i + 1) % vertices.Length].X;
321 var y2 = vertices[(i + 1) % vertices.Length].Y;
322
323 if ((((y1 <= y) && (y < y2)) || ((y2 <= y) && (y < y1))) && (x < (x2 - x1) / (y2 - y1) * (y - y1) + x1))
324 intersects++;
325 }
326
327 return (intersects & 1) == 1;
328 }
