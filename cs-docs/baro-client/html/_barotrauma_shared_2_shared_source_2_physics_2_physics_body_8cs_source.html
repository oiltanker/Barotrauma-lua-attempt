620  public void ApplyLinearImpulse(Vector2 impulse, float maxVelocity)
621  {
622  if (!IsValidValue(impulse, "impulse", -1e10f, 1e10f)) return;
623  if (!IsValidValue(maxVelocity, "max velocity")) return;
624 
625  Vector2 velocityAddition = impulse / Mass;
626  Vector2 newVelocity = FarseerBody.LinearVelocity + velocityAddition;
627  float newSpeedSqr = newVelocity.LengthSquared();
628  if (newSpeedSqr > maxVelocity * maxVelocity)
629  {
630  newVelocity = newVelocity.ClampLength(maxVelocity);
631  }
632 
633  if (!IsValidValue((newVelocity - FarseerBody.LinearVelocity), "new velocity", -1000.0f, 1000.0f)) return;
634 
635  FarseerBody.ApplyLinearImpulse((newVelocity - FarseerBody.LinearVelocity) * Mass);
636  }

672  public void ApplyForce(Vector2 force, float maxVelocity = NetConfig.MaxPhysicsBodyVelocity)
673  {
674  if (!IsValidValue(maxVelocity, "max velocity")) { return; }
675 
676  Vector2 velocityAddition = force / Mass * (float)Timing.Step;
677  Vector2 newVelocity = FarseerBody.LinearVelocity + velocityAddition;
678 
679  float newSpeedSqr = newVelocity.LengthSquared();
680  if (newSpeedSqr > maxVelocity * maxVelocity && Vector2.Dot(FarseerBody.LinearVelocity, force) > 0.0f)
681  {
682  float newSpeed = (float)Math.Sqrt(newSpeedSqr);
683  float maxVelAddition = maxVelocity - newSpeed;
684  if (maxVelAddition <= 0.0f) { return; }
685  force = velocityAddition.ClampLength(maxVelAddition) * Mass / (float)Timing.Step;
686  }
687 
688  if (!IsValidValue(force, "clamped force", -1e10f, 1e10f)) { return; }
689  FarseerBody.ApplyForce(force);
690  }
