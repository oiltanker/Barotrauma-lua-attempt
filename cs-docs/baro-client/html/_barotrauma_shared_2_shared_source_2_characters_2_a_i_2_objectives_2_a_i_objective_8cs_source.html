171 public void TryComplete(float deltaTime)
172 {
173 if (isCompleted) { return; }
174 if (CheckState()) { return; }
175 // Not ready -> act (can't do foreach because it's possible that the collection is modified in event callbacks.
176 for (int i = 0; i < subObjectives.Count; i++)
177 {
178 subObjectives[i].TryComplete(deltaTime);
179 if (!ConcurrentObjectives) { return; }
180 }
181 Act(deltaTime);
182 }

221 public void SortSubObjectives()
222 {
223 if (!AllowSubObjectiveSorting) { return; }
224 if (subObjectives.None()) { return; }
225 var previousSubObjective = subObjectives.First();
226 subObjectives.ForEach(so => so.GetPriority());
227 subObjectives.Sort((x, y) => y.Priority.CompareTo(x.Priority));
228 if (ConcurrentObjectives)
229 {
230 subObjectives.ForEach(so => so.SortSubObjectives());
231 }
232 else
233 {
234 var currentSubObjective = subObjectives.First();
235 if (previousSubObjective != currentSubObjective)
236 {
237 previousSubObjective.OnDeselected();
238 currentSubObjective.OnSelected();
239 }
240 currentSubObjective.SortSubObjectives();
241 }

343 protected bool TryAddSubObjective<T>(ref T objective, Func<T> constructor, Action onCompleted = null, Action onAbandon = null) where T : AIObjective
344 {
345 if (objective != null)
346 {
347 // Sub objective already found, no need to do anything if it remains in the subobjectives
348 // If the sub objective is removed -> it's either completed or impossible to complete.
349 if (!subObjectives.Contains(objective))
350 {
351 objective = null;
352 }
353 return false;
354 }
355 else
356 {
357 objective = constructor();
358 if (!subObjectives.Contains(objective))
359 {
360 if (objective.AllowMultipleInstances)
361 {
362 subObjectives.Add(objective);
363 }
364 else
365 {
366 AddSubObjective(objective);
367 }
368 if (onCompleted != null)
369 {
370 objective.Completed += onCompleted;
371 }
372 if (onAbandon != null)
373 {
374 objective.Abandoned += onAbandon;
375 }
376 return true;
377 }
378#if DEBUG
379 DebugConsole.ThrowError("Attempted to add a duplicate subobjective!\n" + Environment.StackTrace.CleanupStackTrace());
380#endif
381 return false;
382 }

458 private bool CheckState()
459 {
460 hasBeenChecked = true;
461 CheckSubObjectives();
462 if (subObjectives.None() || ConcurrentObjectives && subObjectives.All(so => so is AIObjectiveGoTo))
463 {
464 if (Check())
465 {
466 IsCompleted = true;
467 }
468 }

472 private void CheckSubObjectives()
473 {
474 for (int i = 0; i < subObjectives.Count; i++)
475 {
476 var subObjective = subObjectives[i];
477 subObjective.CheckState();
478 if (subObjective.IsCompleted)
479 {
480#if DEBUG
481 DebugConsole.NewMessage($"{character.Name}: Removing SUBobjective {subObjective.DebugTag} of {DebugTag}, because it is completed.", Color.LightGreen);
482#endif
483 subObjectives.Remove(subObjective);
484 }
485 else if (!subObjective.CanBeCompleted)
486 {
487#if DEBUG
488 DebugConsole.NewMessage($"{character.Name}: Removing SUBobjective {subObjective.DebugTag} of {DebugTag}, because it cannot be completed.", Color.Red);
489#endif
490 subObjectives.Remove(subObjective);
491 if (AbandonWhenCannotCompleteSubjectives)
492 {
493 if (objectiveManager.IsOrder(this))
494 {
495 Reset();
496 }
497 else
498 {
499 Abandon = true;
500 }
501 }
502 }

506 protected static bool CanEquip(Character character, Item item)
507 {
508 bool canEquip = item != null;
509 if (canEquip && !item.AllowedSlots.Contains(InvSlotType.Any))
510 {
511 canEquip = false;
512 var inv = character.Inventory;
513 foreach (var allowedSlot in item.AllowedSlots)
514 {
515 foreach (var slotType in inv.SlotTypes)
516 {
517 if (!allowedSlot.HasFlag(slotType)) { continue; }
518 for (int i = 0; i < inv.Capacity; i++)
519 {
520 canEquip = true;
521 if (allowedSlot.HasFlag(inv.SlotTypes[i]) && inv.GetItemAt(i) != null)
522 {
523 canEquip = false;
524 break;
525 }
526 }
527 }
528 }
529 }
530 return canEquip;
531 }
