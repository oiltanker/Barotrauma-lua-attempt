27 protected override bool Filter(Pump pump)
28 {
29 if (pump == null) { return false; }
30 if (pump.Item.IgnoreByAI(character)) { return false; }
31 if (!pump.Item.IsInteractable(character)) { return false; }
32 if (pump.Item.HasTag("ballast")) { return false; }
33 if (pump.Item.Submarine == null) { return false; }
34 if (pump.Item.CurrentHull == null) { return false; }
35 if (pump.Item.Submarine.TeamID != character.TeamID) { return false; }
36 if (pump.IsAutoControlled) { return false; }
37 if (pump.Item.ConditionPercentage <= 0) { return false; }
38 if (pump.Item.CurrentHull.FireSources.Count > 0) { return false; }
39 if (character.Submarine != null)
40 {
41 if (!character.Submarine.IsConnectedTo(pump.Item.Submarine)) { return false; }
42 }
43 if (Character.CharacterList.Any(c => c.CurrentHull == pump.Item.CurrentHull && !HumanAIController.IsFriendly(c) && HumanAIController.IsActive(c))) { return false; }
44 if (pump.Item.IsClaimedByBallastFlora) { return false; }
45 if (IsReady(pump)) { return false; }
46 return true;
47 }
48 protected override IEnumerable<Pump> GetList()
49 {
50 if (pumpList == null)
51 {
52 if (character == null || character.Submarine == null) { return Array.Empty<Pump>(); }
53 pumpList = character.Submarine.GetItems(true).Select(i => i.GetComponent<Pump>()).Where(p => p != null);
54 }
55 return pumpList;

58 protected override float TargetEvaluation()
59 {
60 if (Targets.None()) { return 0; }
61 if (Option == "stoppumping")
62 {
63 return Targets.Max(t => MathHelper.Lerp(0, 100, Math.Abs(t.FlowPercentage / 100)));
64 }
65 else
66 {
67 return Targets.Max(t => MathHelper.Lerp(100, 0, Math.Abs(-t.FlowPercentage / 100)));
68 }
69 }

71 private bool IsReady(Pump pump)
72 {
73 if (Option == "stoppumping")
74 {
75 return !pump.IsActive || MathUtils.NearlyEqual(pump.FlowPercentage, 0);
76 }
77 else
78 {
79 return !pump.Item.InWater || pump.IsActive && pump.FlowPercentage <= -99.9f;
80 }
81 }
