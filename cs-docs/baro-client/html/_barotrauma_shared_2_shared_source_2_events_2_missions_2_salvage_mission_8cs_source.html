46  public SalvageMission(MissionPrefab prefab, Location[] locations, Submarine sub)
47  : base(prefab, locations, sub)
48  {
49  containerTag = prefab.ConfigElement.GetAttributeString("containertag", "");
50 
51  if (prefab.ConfigElement.GetAttribute("itemname") != null)
52  {
53  DebugConsole.ThrowError("Error in SalvageMission - use item identifier instead of the name of the item.");
54  string itemName = prefab.ConfigElement.GetAttributeString("itemname", "");
55  itemPrefab = MapEntityPrefab.Find(itemName) as ItemPrefab;
56  if (itemPrefab == null)
57  {
58  DebugConsole.ThrowError("Error in SalvageMission: couldn't find an item prefab with the name " + itemName);
59  }
60  }
61  else
62  {
63  string itemIdentifier = prefab.ConfigElement.GetAttributeString("itemidentifier", null);
64  if (itemIdentifier != null)
65  {
66  itemPrefab = MapEntityPrefab.FindByIdentifier(itemIdentifier.ToIdentifier()) as ItemPrefab;
67  }
68  if (itemPrefab == null)
69  {
70  string itemTag = prefab.ConfigElement.GetAttributeString("itemtag", "");
71  itemPrefab = MapEntityPrefab.GetRandom(p => p.Tags.Contains(itemTag), Rand.RandSync.Unsynced) as ItemPrefab;
72  }
73  if (itemPrefab == null)
74  {
75  DebugConsole.ThrowError("Error in SalvageMission - couldn't find an item prefab with the identifier " + itemIdentifier);
76  }
77  }
78 
79  existingItemTag = prefab.ConfigElement.GetAttributeString("existingitemtag", "");
80  showMessageWhenPickedUp = prefab.ConfigElement.GetAttributeBool("showmessagewhenpickedup", false);
81 
82  string spawnPositionTypeStr = prefab.ConfigElement.GetAttributeString("spawntype", "");
83  if (string.IsNullOrWhiteSpace(spawnPositionTypeStr) ||
84  !Enum.TryParse(spawnPositionTypeStr, true, out spawnPositionType))
85  {
86  spawnPositionType = Level.PositionType.Cave | Level.PositionType.Ruin;
87  }
88 
89  foreach (var element in prefab.ConfigElement.Elements())
90  {
91  switch (element.Name.ToString().ToLowerInvariant())
92  {
93  case "statuseffect":
94  {
95  var newEffect = StatusEffect.Load(element, parentDebugName: prefab.Name.Value);
96  if (newEffect == null) { continue; }
97  statusEffects.Add(new List<StatusEffect> { newEffect });
98  break;
99  }
100  case "chooserandom":
101  statusEffects.Add(new List<StatusEffect>());
102  foreach (var subElement in element.Elements())
103  {
104  var newEffect = StatusEffect.Load(subElement, parentDebugName: prefab.Name.Value);
105  if (newEffect == null) { continue; }
106  statusEffects.Last().Add(newEffect);
107  }
108  break;
109  }
110  }

113  protected override void StartMissionSpecific(Level level)
114  {
115 #if SERVER
116  originalInventoryID = Entity.NullEntityID;
117 #endif
118  item = null;
119  if (!IsClient)
120  {
121  //ruin/cave/wreck items are allowed to spawn close to the sub
122  float minDistance = spawnPositionType == Level.PositionType.Ruin || spawnPositionType == Level.PositionType.Cave || spawnPositionType == Level.PositionType.Wreck ?
123  0.0f : Level.Loaded.Size.X * 0.3f;
124  Vector2 position = Level.Loaded.GetRandomItemPos(spawnPositionType, 100.0f, minDistance, 30.0f);
125 
126  if (!string.IsNullOrEmpty(existingItemTag))
127  {
128  var suitableItems = Item.ItemList.Where(it => it.HasTag(existingItemTag));
129  switch (spawnPositionType)
130  {
131  case Level.PositionType.Cave:
132  case Level.PositionType.MainPath:
133  case Level.PositionType.SidePath:
134  item = suitableItems.FirstOrDefault(it => Vector2.DistanceSquared(it.WorldPosition, position) < 1000.0f);
135  break;
136  case Level.PositionType.Ruin:
137  case Level.PositionType.Wreck:
138  foreach (Item it in suitableItems)
139  {
140  if (it.Submarine?.Info == null) { continue; }
141  if (spawnPositionType == Level.PositionType.Ruin && it.Submarine.Info.Type != SubmarineType.Ruin) { continue; }
142  if (spawnPositionType == Level.PositionType.Wreck && it.Submarine.Info.Type != SubmarineType.Wreck) { continue; }
143  Rectangle worldBorders = it.Submarine.Borders;
144  worldBorders.Location += it.Submarine.WorldPosition.ToPoint();
145  if (Submarine.RectContains(worldBorders, it.WorldPosition))
146  {
147  item = it;
148 #if SERVER
149  usedExistingItem = true;
150 #endif
151  break;
152  }
153  }
154  break;
155  }
156  }
157 
158  if (item == null)
159  {
160  item = new Item(itemPrefab, position, null);
161  item.body.SetTransformIgnoreContacts(item.body.SimPosition, item.body.Rotation);
162  item.body.FarseerBody.BodyType = BodyType.Kinematic;
163  }
164 
165  for (int i = 0; i < statusEffects.Count; i++)
166  {
167  List<StatusEffect> effectList = statusEffects[i];
168  if (effectList.Count == 0) { continue; }
169  int effectIndex = Rand.Int(effectList.Count);
170  var selectedEffect = effectList[effectIndex];
171  item.ApplyStatusEffect(selectedEffect, selectedEffect.type, deltaTime: 1.0f, worldPosition: item.Position);
172 #if SERVER
173  executedEffectIndices.Add(new Pair<int, int>(i, effectIndex));
174 #endif
175  }
176 
177  //try to find a container and place the item inside it
178  if (!string.IsNullOrEmpty(containerTag) && item.ParentInventory == null)
179  {
180  List<ItemContainer> validContainers = new List<ItemContainer>();
181  foreach (Item it in Item.ItemList)
182  {
183  if (!it.HasTag(containerTag)) { continue; }
184  if (!it.IsPlayerTeamInteractable) { continue; }
185  switch (spawnPositionType)
186  {
187  case Level.PositionType.Cave:
188  case Level.PositionType.MainPath:
189  if (it.Submarine != null) { continue; }
190  break;
191  case Level.PositionType.Ruin:
192  if (it.Submarine?.Info == null || !it.Submarine.Info.IsRuin) { continue; }
193  break;
194  case Level.PositionType.Wreck:
195  if (it.Submarine == null || it.Submarine.Info.Type != SubmarineType.Wreck) { continue; }
196  break;
197  }
198  var itemContainer = it.GetComponent<ItemContainer>();
199  if (itemContainer != null && itemContainer.Inventory.CanBePut(item)) { validContainers.Add(itemContainer); }
200  }
201  if (validContainers.Any())
202  {
203  var selectedContainer = validContainers.GetRandomUnsynced();
204  if (selectedContainer.Combine(item, user: null))
205  {
206 #if SERVER
207  originalInventoryID = selectedContainer.Item.ID;
208  originalItemContainerIndex = (byte)selectedContainer.Item.GetComponentIndex(selectedContainer);
209  originalSlotIndex = item.ParentInventory?.FindIndex(item) ?? -1;
210 #endif
211  } // Placement successful
212  }
213  }
214  }

217  protected override void UpdateMissionSpecific(float deltaTime)
218  {
219  if (item == null)
220  {
221 #if DEBUG
222  DebugConsole.ThrowError("Error in salvage mission " + Prefab.Identifier + " (item was null)");
223 #endif
224  return;
225  }
226 
227  if (IsClient)
228  {
229  if (item.ParentInventory != null && item.body != null) { item.body.FarseerBody.BodyType = BodyType.Dynamic; }
230  return;
231  }
232  switch (State)
233  {
234  case 0:
235  if (item.ParentInventory != null && item.body != null) { item.body.FarseerBody.BodyType = BodyType.Dynamic; }
236  if (showMessageWhenPickedUp)
237  {
238  if (!(item.GetRootInventoryOwner() is Character)) { return; }
239  }
240  else
241  {
242  Submarine parentSub = item.CurrentHull?.Submarine ?? item.GetRootInventoryOwner()?.Submarine;
243  if (parentSub == null || parentSub.Info.Type != SubmarineType.Player)
244  {
245  return;
246  }
247  }
248  State = 1;
249  break;
250  case 1:
251  if (!Submarine.MainSub.AtEndExit && !Submarine.MainSub.AtStartExit) { return; }
252  State = 2;
253  break;
254  }

257  public override void End()
258  {
259  var root = item?.GetRootContainer() ?? item;
260  if (root?.CurrentHull?.Submarine == null || (!root.CurrentHull.Submarine.AtEndExit && !root.CurrentHull.Submarine.AtStartExit) || item.Removed)
261  {
262  return;
263  }
264 
265  if (Prefab.LocationTypeChangeOnCompleted != null)
266  {
267  ChangeLocationType(Prefab.LocationTypeChangeOnCompleted);
268  }
269 
270  item?.Remove();
271  item = null;
272  GiveReward();
273  completed = true;
274  failed = !completed && state > 0;
275  }
