417  private void UpdateVertices(Camera cam, EntityGrid entityGrid, WaterRenderer renderer)
418  {
419  Vector2 submarinePos = Submarine == null ? Vector2.Zero : Submarine.DrawPosition;
420 
421  //if there's no more space in the buffer, don't render the water in the hull
422  //not an ideal solution, but this seems to only happen in cases where the missing
423  //water is not very noticeable (e.g. zoomed very far out so that multiple subs and ruins are visible)
424  if (renderer.PositionInBuffer > renderer.vertices.Length - 6)
425  {
426  return;
427  }
428 
429  //calculate where the surface should be based on the water volume
430  float top = rect.Y + submarinePos.Y;
431  float bottom = top - rect.Height;
432  float renderSurface = drawSurface + submarinePos.Y;
433 
434  if (bottom > cam.WorldView.Y || top < cam.WorldView.Y - cam.WorldView.Height) { return; }
435  if (rect.X + submarinePos.X > cam.WorldView.Right || rect.Right + submarinePos.X < cam.WorldView.X) { return; }
436 
437  Matrix transform = cam.Transform * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
438  if (!update)
439  {
440  // create the four corners of our triangle.
441 
442  corners[0] = new Vector3(rect.X, rect.Y, 0.0f);
443  corners[1] = new Vector3(rect.X + rect.Width, rect.Y, 0.0f);
444 
445  corners[2] = new Vector3(corners[1].X, rect.Y - rect.Height, 0.0f);
446  corners[3] = new Vector3(corners[0].X, corners[2].Y, 0.0f);
447 
448  for (int i = 0; i < 4; i++)
449  {
450  corners[i] += new Vector3(submarinePos, 0.0f);
451  uvCoords[i] = Vector2.Transform(new Vector2(corners[i].X, -corners[i].Y), transform);
452  }
453 
454  renderer.vertices[renderer.PositionInBuffer] = new VertexPositionTexture(corners[0], uvCoords[0]);
455  renderer.vertices[renderer.PositionInBuffer + 1] = new VertexPositionTexture(corners[1], uvCoords[1]);
456  renderer.vertices[renderer.PositionInBuffer + 2] = new VertexPositionTexture(corners[2], uvCoords[2]);
457 
458  renderer.vertices[renderer.PositionInBuffer + 3] = new VertexPositionTexture(corners[0], uvCoords[0]);
459  renderer.vertices[renderer.PositionInBuffer + 4] = new VertexPositionTexture(corners[2], uvCoords[2]);
460  renderer.vertices[renderer.PositionInBuffer + 5] = new VertexPositionTexture(corners[3], uvCoords[3]);
461 
462  renderer.PositionInBuffer += 6;
463 
464  return;
465  }
466 
467  if (!renderer.IndoorsVertices.ContainsKey(entityGrid))
468  {
469  renderer.IndoorsVertices[entityGrid] = new VertexPositionColorTexture[WaterRenderer.DefaultIndoorsBufferSize];
470  }
471  if (!renderer.PositionInIndoorsBuffer.ContainsKey(entityGrid))
472  {
473  renderer.PositionInIndoorsBuffer[entityGrid] = 0;
474  }
475 
476  float x = rect.X;
477  if (Submarine != null) { x += Submarine.DrawPosition.X; }
478 
479  int start = (int)Math.Floor((cam.WorldView.X - x) / WaveWidth);
480  start = Math.Max(start, 0);
481 
482  int end = (waveY.Length - 1) - (int)Math.Floor(((x + rect.Width) - (cam.WorldView.Right)) / WaveWidth);
483  end = Math.Min(end, waveY.Length - 1);
484 
485  x += start * WaveWidth;
486 
487  int width = WaveWidth;
488 
489  for (int i = start; i < end; i++)
490  {
491  //top left
492  corners[0] = new Vector3(x, top, 0.0f);
493  //watersurface left
494  corners[3] = new Vector3(corners[0].X, renderSurface + waveY[i], 0.0f);
495 
496  //top right
497  corners[1] = new Vector3(x + width, top, 0.0f);
498  //watersurface right
499  corners[2] = new Vector3(corners[1].X, renderSurface + waveY[i + 1], 0.0f);
500 
501  //bottom left
502  corners[4] = new Vector3(x, bottom, 0.0f);
503  //bottom right
504  corners[5] = new Vector3(x + width, bottom, 0.0f);
505 
506  Vector2[] uvCoords = new Vector2[4];
507  for (int n = 0; n < 4; n++)
508  {
509  uvCoords[n] = Vector2.Transform(new Vector2(corners[n].X, -corners[n].Y), transform);
510  }
511 
512  if (renderer.PositionInBuffer <= renderer.vertices.Length - 6)
513  {
514  if (i == start)
515  {
516  prevCorners[0] = corners[0];
517  prevCorners[1] = corners[3];
518  prevUVs[0] = uvCoords[0];
519  prevUVs[1] = uvCoords[3];
520  }
521 
522  //we only create a new quad if this is the first or the last one, of if there's a wave large enough that we need more geometry
523  if (i == end - 1 || i == start || Math.Abs(prevCorners[1].Y - corners[2].Y) > 0.01f)
524  {
525  renderer.vertices[renderer.PositionInBuffer] = new VertexPositionTexture(prevCorners[0], prevUVs[0]);
526  renderer.vertices[renderer.PositionInBuffer + 1] = new VertexPositionTexture(corners[1], uvCoords[1]);
527  renderer.vertices[renderer.PositionInBuffer + 2] = new VertexPositionTexture(corners[2], uvCoords[2]);
528 
529  renderer.vertices[renderer.PositionInBuffer + 3] = new VertexPositionTexture(prevCorners[0], prevUVs[0]);
530  renderer.vertices[renderer.PositionInBuffer + 4] = new VertexPositionTexture(corners[2], uvCoords[2]);
531  renderer.vertices[renderer.PositionInBuffer + 5] = new VertexPositionTexture(prevCorners[1], prevUVs[1]);
532 
533  prevCorners[0] = corners[1];
534  prevCorners[1] = corners[2];
535  prevUVs[0] = uvCoords[1];
536  prevUVs[1] = uvCoords[2];
537 
538  renderer.PositionInBuffer += 6;
539  }
540  }
541 
542  if (renderer.PositionInIndoorsBuffer[entityGrid] <= renderer.IndoorsVertices[entityGrid].Length - 12 &&
543  cam.Zoom > 0.6f)
544  {
545  const float SurfaceSize = 10.0f;
546  const float SineFrequency1 = 0.01f;
547  const float SineFrequency2 = 0.05f;
548 
549  //surface shrinks and finally disappears when the water level starts to reach the top of the hull
550  float surfaceScale = 1.0f - MathHelper.Clamp(corners[3].Y - (top - SurfaceSize), 0.0f, 1.0f);
551 
552  Vector3 surfaceOffset = new Vector3(0.0f, -SurfaceSize, 0.0f);
553  surfaceOffset.Y += (float)Math.Sin((rect.X + i * WaveWidth) * SineFrequency1 + renderer.WavePos.X * 0.25f) * 2;
554  surfaceOffset.Y += (float)Math.Sin((rect.X + i * WaveWidth) * SineFrequency2 - renderer.WavePos.X) * 2;
555  surfaceOffset *= surfaceScale;
556

622  public void ClientEventRead(IReadMessage msg, float sendingTime)
623  {
624  EventType eventType = (EventType)msg.ReadRangedInteger((int)EventType.MinValue, (int)EventType.MaxValue);
625  switch (eventType)
626  {
627  case EventType.Status:
628  remoteOxygenPercentage = msg.ReadRangedSingle(0.0f, 100.0f, 8);
629 
630  SharedStatusRead(
631  msg,
632  out float newWaterVolume,
633  out NetworkFireSource[] newFireSources);
634 
635  remoteWaterVolume = newWaterVolume;
636  remoteFireSources = newFireSources;
637  break;
638  case EventType.BackgroundSections:
639  SharedBackgroundSectionRead(
640  msg,
641  bsnu =>
642  {
643  int i = bsnu.SectionIndex;
644  Color color = bsnu.Color;
645  float colorStrength = bsnu.ColorStrength;
646 
647  var remoteBackgroundSection = remoteBackgroundSections.Find(s => s.Index == i);
648  if (remoteBackgroundSection != null)
649  {
650  remoteBackgroundSection.SetColorStrength(colorStrength);
651  remoteBackgroundSection.SetColor(color);
652  }
653  else
654  {
655  remoteBackgroundSections.Add(new BackgroundSection(new Rectangle(0, 0, 1, 1), (ushort)i, colorStrength, color, 0));
656  }
657  }, out _);
658  paintAmount = BackgroundSections.Sum(s => s.ColorStrength);
659  break;
660  case EventType.Decal:
661  int decalCount = msg.ReadRangedInteger(0, MaxDecalsPerHull);
662  if (decalCount == 0) { decals.Clear(); }
663  remoteDecals.Clear();
664  for (int i = 0; i < decalCount; i++)
665  {
666  UInt32 decalId = msg.ReadUInt32();
667  int spriteIndex = msg.ReadByte();

669  float normalizedYPos = msg.ReadRangedSingle(0.0f, 1.0f, 8);
670  float decalScale = msg.ReadRangedSingle(0.0f, 2.0f, 12);
671  remoteDecals.Add(new RemoteDecal(decalId, spriteIndex, new Vector2(normalizedXPos, normalizedYPos), decalScale));
672  }
673  break;
674  case EventType.BallastFlora:
675  BallastFloraBehavior.NetworkHeader header = (BallastFloraBehavior.NetworkHeader) msg.ReadByte();
676  if (header == BallastFloraBehavior.NetworkHeader.Spawn)
677  {
678  Identifier identifier = msg.ReadIdentifier();
679  float x = msg.ReadSingle();
680  float y = msg.ReadSingle();

682  {
683  PowerConsumptionTimer = msg.ReadSingle()
684  };
685  }
686  else
687  {
688  BallastFlora?.ClientRead(msg, header);
689  }
690  break;
691  default:
692  throw new Exception($"Malformed incoming hull event: {eventType} is not a supported event type");
693  }
694 
695  if (serverUpdateDelay > 0.0f) { return; }
696 
697  ApplyRemoteState();
698  }

700  private void ApplyRemoteState()
701  {
702  foreach (BackgroundSection remoteBackgroundSection in remoteBackgroundSections)
703  {
704  float prevColorStrength = BackgroundSections[remoteBackgroundSection.Index].ColorStrength;
705  BackgroundSections[remoteBackgroundSection.Index].SetColor(remoteBackgroundSection.Color);
706  BackgroundSections[remoteBackgroundSection.Index].SetColorStrength(remoteBackgroundSection.ColorStrength);
707  paintAmount = Math.Max(0, paintAmount + (BackgroundSections[remoteBackgroundSection.Index].ColorStrength - prevColorStrength) / BackgroundSections.Count);
708  }
709  remoteBackgroundSections.Clear();
710 
711  if (remoteDecals.Count > 0)
712  {
713  decals.Clear();
714  foreach (RemoteDecal remoteDecal in remoteDecals)
715  {
716  float decalPosX = MathHelper.Lerp(rect.X, rect.Right, remoteDecal.NormalizedPos.X);
717  float decalPosY = MathHelper.Lerp(rect.Y - rect.Height, rect.Y, remoteDecal.NormalizedPos.Y);
718  if (Submarine != null)
719  {
720  decalPosX += Submarine.Position.X;
721  decalPosY += Submarine.Position.Y;
722  }
723  AddDecal(remoteDecal.DecalId, new Vector2(decalPosX, decalPosY), remoteDecal.Scale, isNetworkEvent: true, spriteIndex: remoteDecal.SpriteIndex);
724  }
725  remoteDecals.Clear();
726  }
727 
728  if (remoteFireSources is null) { return; }
729 
730  WaterVolume = remoteWaterVolume;
731  OxygenPercentage = remoteOxygenPercentage;
732 
733  for (int i = 0; i < remoteFireSources.Length; i++)
734  {
735  Vector2 pos = remoteFireSources[i].Position;
736  float size = remoteFireSources[i].Size;
737 
738  var newFire = i < FireSources.Count ?
739  FireSources[i] :
740  new FireSource(Submarine == null ? pos : pos + Submarine.Position, null, true);
741  newFire.Position = pos;
742  newFire.Size = new Vector2(size, newFire.Size.Y);
743 
744  //ignore if the fire wasn't added to this room (invalid position)?
745  if (!FireSources.Contains(newFire))
746  {
747  newFire.Remove();
748  continue;
749  }
750  }
751 
752  for (int i = FireSources.Count - 1; i >= remoteFireSources.Length; i--)
753  {
754  FireSources[i].Remove();
755  if (i < FireSources.Count)
756  {
757  FireSources.RemoveAt(i);
758  }
759  }
760  remoteFireSources = null;
761  }
762  }
