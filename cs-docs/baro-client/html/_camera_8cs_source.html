254  public void MoveCamera(float deltaTime, bool allowMove = true, bool allowZoom = true, bool? followSub = null)
255  {
256  prevPosition = position;
257  prevZoom = zoom;
258 
259  float moveSpeed = 20.0f / zoom;
260 
261  Vector2 moveCam = Vector2.Zero;
262  if (TargetPos == Vector2.Zero)
263  {
264  Vector2 moveInput = Vector2.Zero;
265  if (allowMove && !Freeze)
266  {
267  if (GUI.KeyboardDispatcher.Subscriber == null)
268  {
269  if (PlayerInput.KeyDown(Keys.LeftShift)) { moveSpeed *= 2.0f; }
270  if (PlayerInput.KeyDown(Keys.LeftControl)) { moveSpeed *= 0.5f; }
271 
272  if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Left].IsDown()) { moveInput.X -= 1.0f; }
273  if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Right].IsDown()) { moveInput.X += 1.0f; }
274  if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Down].IsDown()) { moveInput.Y -= 1.0f; }
275  if (GameSettings.CurrentConfig.KeyMap.Bindings[InputType.Up].IsDown()) { moveInput.Y += 1.0f; }
276  }
277 
278  velocity = Vector2.Lerp(velocity, moveInput, deltaTime * 10.0f);
279  moveCam = velocity * moveSpeed * deltaTime * FreeCamMoveSpeed * 60.0f;
280 
281  if (Screen.Selected == GameMain.GameScreen && (followSub ?? FollowSub))
282  {
283  var closestSub = Submarine.FindClosest(WorldViewCenter);
284  if (closestSub != null)
285  {
286  moveCam += FarseerPhysics.ConvertUnits.ToDisplayUnits(closestSub.Velocity * deltaTime);
287  }
288  }
289  }
290 
291  if (allowZoom)
292  {
293  Vector2 mouseInWorld = ScreenToWorld(PlayerInput.MousePosition);
294  Vector2 diffViewCenter;
295  diffViewCenter = (mouseInWorld - Position) * Zoom;
296  targetZoom = MathHelper.Clamp(
297  targetZoom + PlayerInput.ScrollWheelSpeed / 1000.0f * zoom,
298  GameMain.DebugDraw ? MinZoom * 0.1f : MinZoom,
299  MaxZoom);
300 
301  if (PlayerInput.KeyDown(Keys.LeftControl))
302  {
303  Zoom += (targetZoom - zoom) / (ZoomSmoothness * 10.0f);
304  }
305  else
306  {
307  Zoom = MathHelper.Lerp(Zoom, targetZoom, deltaTime * 10.0f);
308  }
309  if (!PlayerInput.KeyDown(Keys.F)) { Position = mouseInWorld - (diffViewCenter / Zoom); }
310  }
311  }
312  else if (allowMove)
313  {
314  Vector2 mousePos = PlayerInput.MousePosition;
315  Vector2 offset = mousePos - Resolution.ToVector2() / 2;
316  offset.X = offset.X / (Resolution.X * 0.4f);
317  offset.Y = -offset.Y / (Resolution.Y * 0.3f);
318  if (offset.LengthSquared() > 1.0f) offset.Normalize();
319  float offsetUnscaledLen = offset.Length();
320  offset *= OffsetAmount;
321  // Freeze the camera movement by default, when the cursor is on top of an ui element.
322  // Setting a positive value to the OffsetAmount, will override this behaviour.
323  if (GUI.MouseOn != null && OffsetAmount > 0)
324  {
325  Freeze = true;
326  }
327  if (CharacterHealth.OpenHealthWindow != null || CrewManager.IsCommandInterfaceOpen || ConversationAction.IsDialogOpen)
328  {
329  offset *= 0;
330  Freeze = false;
331  }
332  if (Freeze)
333  {
334  if (offset.LengthSquared() > 0.001f) { offset = previousOffset; }
335  }
336  else
337  {
338  previousOffset = offset;
339  }
340 
341  if (allowZoom)
342  {
343  //how much to zoom out (zoom completely out when offset is 1000)
344  float zoomOutAmount = GetZoomAmount(offset);
345  //scaled zoom amount
346  float scaledZoom = MathHelper.Lerp(DefaultZoom, MinZoom, zoomOutAmount) * globalZoomScale;
347  //zoom in further if zoomOutAmount is low and resolution is lower than reference
348  float newZoom = scaledZoom * (MathHelper.Lerp(0.3f * (1f - Math.Min(globalZoomScale, 1f)), 0f,
349  (GameSettings.CurrentConfig.EnableMouseLook) ? (float)Math.Sqrt(offsetUnscaledLen) : 0.3f) + 1f);
350 
351  Zoom += (newZoom - zoom) / ZoomSmoothness;
352  }
353 
354  //force targetzoom to the current zoom value, so the camera stays at the same zoom when switching to freecam
355  targetZoom = Zoom;
356 
357  Vector2 diff = (TargetPos + offset) - position;
358 
359  moveCam = diff / MoveSmoothness;
360  }
361  rotation += angularVelocity * deltaTime;
362  angularVelocity *= (1.0f - angularDamping);
363  angularVelocity += -rotation * angularSpring;
364 
365  angularDamping = 0.05f;
366  angularSpring = 0.2f;
367 
368  if (Shake < 0.01f)
369  {
370  shakePosition = Vector2.Zero;
371  shakeTimer = 0.0f;
372  }
373  else
374  {
375  shakeTimer += deltaTime * 5.0f;
376  Vector2 noisePos = new Vector2((float)PerlinNoise.CalculatePerlin(shakeTimer, shakeTimer, 0) - 0.5f, (float)PerlinNoise.CalculatePerlin(shakeTimer, shakeTimer, 0.5f) - 0.5f);
377 
378  shakePosition = noisePos * Shake * 2.0f;
379  Shake = MathHelper.Lerp(Shake, 0.0f, deltaTime * 2.0f);
380  }
381 
382  Translate(moveCam + shakePosition);
383  Freeze = false;
384  }
