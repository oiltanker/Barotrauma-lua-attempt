44 public void SendToServer()
45 {
46 if (GameSettings.CurrentConfig.Audio.VoiceSetting == VoiceMode.Disabled)
47 {
48 if (VoipCapture.Instance != null)
49 {
50 storedBufferID = VoipCapture.Instance.LatestBufferID;
51 VoipCapture.Instance.Dispose();
52 }
53 return;
54 }
55 else
56 {
57 try
58 {
59 if (VoipCapture.Instance == null) { VoipCapture.Create(GameSettings.CurrentConfig.Audio.VoiceCaptureDevice, storedBufferID); }
60 }
61 catch (Exception e)
62 {
63 DebugConsole.ThrowError($"VoipCature.Create failed: {e.Message} {e.StackTrace.CleanupStackTrace()}");
64 var config = GameSettings.CurrentConfig;
65 config.Audio.VoiceSetting = VoiceMode.Disabled;
66 GameSettings.SetCurrentConfig(config);
67 }
68 if (VoipCapture.Instance == null || VoipCapture.Instance.EnqueuedTotalLength <= 0) { return; }
69 }
70
71 if (DateTime.Now >= lastSendTime + VoipConfig.SEND_INTERVAL)
72 {
73 IWriteMessage msg = new WriteOnlyMessage();
74
75 msg.Write((byte)ClientPacketHeader.VOICE);
76 msg.Write((byte)VoipCapture.Instance.QueueID);
77 VoipCapture.Instance.Write(msg);
78
79 netClient.Send(msg, DeliveryMethod.Unreliable);
80
81 lastSendTime = DateTime.Now;
82 }

85 public void Read(IReadMessage msg)
86 {
87 byte queueId = msg.ReadByte();
88 VoipQueue queue = queues.Find(q => q.QueueID == queueId);
89
90 if (queue == null)
91 {
92#if DEBUG
93 DebugConsole.NewMessage("Couldn't find VoipQueue with id " + queueId.ToString() + "!", GUIStyle.Red);
94#endif
95 return;
96 }
97
98 Client client = gameClient.ConnectedClients.Find(c => c.VoipQueue == queue);
99 if (queue.Read(msg, discardData: client.Muted || client.MutedLocally))
100 {
101 if (client.Muted || client.MutedLocally) { return; }
102 if (client.VoipSound == null)
103 {
104 DebugConsole.Log("Recreating voipsound " + queueId);
105 client.VoipSound = new VoipSound(client.Name, GameMain.SoundManager, client.VoipQueue);
106 }
107 GameMain.SoundManager.ForceStreamUpdate();
108
109 if (client.Character != null && !client.Character.IsDead && !client.Character.Removed && client.Character.SpeechImpediment <= 100.0f)
110 {
111 WifiComponent radio = null;
112 var messageType = !client.VoipQueue.ForceLocal && ChatMessage.CanUseRadio(client.Character, out radio) ? ChatMessageType.Radio : ChatMessageType.Default;
113 client.Character.ShowSpeechBubble(1.25f, ChatMessage.MessageColor[(int)messageType]);
114
115 client.VoipSound.UseRadioFilter = messageType == ChatMessageType.Radio && !GameSettings.CurrentConfig.Audio.DisableVoiceChatFilters;
116 if (messageType == ChatMessageType.Radio)
117 {
118 client.VoipSound.SetRange(radio.Range * 0.8f, radio.Range);
119 }
120 else
121 {
122 client.VoipSound.SetRange(ChatMessage.SpeakRange * 0.4f, ChatMessage.SpeakRange);
123 }
124 if (messageType != ChatMessageType.Radio && Character.Controlled != null && !GameSettings.CurrentConfig.Audio.DisableVoiceChatFilters)
125 {
126 client.VoipSound.UseMuffleFilter = SoundPlayer.ShouldMuffleSound(Character.Controlled, client.Character.WorldPosition, ChatMessage.SpeakRange, client.Character.CurrentHull);
127 }
128 }
129
130 GameMain.NetLobbyScreen?.SetPlayerSpeaking(client);
131 GameMain.GameSession?.CrewManager?.SetClientSpeaking(client);
132
133 if ((client.VoipSound.CurrentAmplitude * client.VoipSound.Gain * GameMain.SoundManager.GetCategoryGainMultiplier("voip")) > 0.1f) //TODO: might need to tweak
134 {
135 if (client.Character != null && !client.Character.Removed)
136 {
137 Vector3 clientPos = new Vector3(client.Character.WorldPosition.X, client.Character.WorldPosition.Y, 0.0f);
138 Vector3 listenerPos = GameMain.SoundManager.ListenerPosition;
139 float attenuationDist = client.VoipSound.Near * 1.125f;
140 if (Vector3.DistanceSquared(clientPos, listenerPos) < attenuationDist * attenuationDist)
141 {
142 GameMain.SoundManager.VoipAttenuatedGain = 0.5f;
143 }
144 }
145 else
146 {
147 GameMain.SoundManager.VoipAttenuatedGain = 0.5f;
148 }
149 }
150 }

153 public static void UpdateVoiceIndicator(GUIImage soundIcon, float voipAmplitude, float deltaTime)
154 {
155 if (voiceIconSheetRects == null)
156 {
157 var soundIconStyle = GUIStyle.GetComponentStyle("GUISoundIcon");
158 Rectangle sourceRect = soundIconStyle.Sprites.First().Value.First().Sprite.SourceRect;
159 var indexPieces = soundIconStyle.Element.GetAttribute("sheetindices").Value.Split(';');
160 voiceIconSheetRects = new Rectangle[indexPieces.Length];
161 for (int i = 0; i < indexPieces.Length; i++)
162 {
163 Point location = sourceRect.Location + XMLExtensions.ParsePoint(indexPieces[i].Trim()) * sourceRect.Size;
164 voiceIconSheetRects[i] = new Rectangle(location, sourceRect.Size);
165 }
166 }
167
168 Pair<string, float> userdata = soundIcon.UserData as Pair<string, float>;
169 userdata.Second = Math.Max(voipAmplitude, userdata.Second - deltaTime);
170
171 if (userdata.Second <= 0.0f)
172 {
173 soundIcon.Visible = false;
174 }
175 else
176 {
177 soundIcon.Visible = true;
178 int sheetIndex = (int)Math.Floor(userdata.Second * voiceIconSheetRects.Length);
179 sheetIndex = MathHelper.Clamp(sheetIndex, 0, voiceIconSheetRects.Length - 1);
180 soundIcon.SourceRect = voiceIconSheetRects[sheetIndex];
181 soundIcon.OverrideState = GUIComponent.ComponentState.None;
182 soundIcon.HoverColor = Color.White;
183 }
