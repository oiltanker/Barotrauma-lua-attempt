142  {
143  if (!string.IsNullOrWhiteSpace(extra))
144  if (exceptionType == ExceptionType.Lua) PrintError(extra);
145  else if (exceptionType == ExceptionType.CSharp) PrintCsError(extra);
146  else PrintBothError(extra);

148  if (ex is NetRuntimeException netRuntimeException)

150  if (netRuntimeException.DecoratedMessage == null)
151  {
152  PrintError(netRuntimeException);
153  }
154  else
155  {
156  PrintError(netRuntimeException.DecoratedMessage + ": " + netRuntimeException.ToString());
157  }
158  }
159  else if (ex is InterpreterException interpreterException)
160  {
161  if (interpreterException.DecoratedMessage == null)
162  {
163  PrintError(interpreterException);
164  }
165  else
166  {
167  PrintError(interpreterException.DecoratedMessage);
168  }
169  }
170  else
171  {
172  string msg = ex.StackTrace != null
173  ? ex.ToString()
174  : $"{ex}\n{Environment.StackTrace}";
175 
176  if (exceptionType == ExceptionType.Lua) { PrintError(msg); }
177  else if (exceptionType == ExceptionType.CSharp) { PrintCsError(msg); }
178  else { PrintBothError(msg); }
179  }
