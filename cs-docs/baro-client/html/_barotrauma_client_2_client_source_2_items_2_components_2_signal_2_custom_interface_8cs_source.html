201 public override void UpdateHUD(Character character, float deltaTime, Camera cam)
202 {
203 bool elementVisibilityChanged = false;
204 int visibleElementCount = 0;
205 foreach (var uiElement in uiElements)
206 {
207 if (!(uiElement.UserData is CustomInterfaceElement element)) { continue; }
208 bool visible = Screen.Selected == GameMain.SubEditorScreen || element.StatusEffects.Any() || element.HasPropertyName || (element.Connection != null && element.Connection.Wires.Any(w => w != null));
209 if (visible) { visibleElementCount++; }
210 if (uiElement.Visible != visible)
211 {
212 uiElement.Visible = visible;
213 uiElement.IgnoreLayoutGroups = !uiElement.Visible;
214 elementVisibilityChanged = true;
215 }
216 }
217
218 if (elementVisibilityChanged)
219 {
220 uiElementContainer.Stretch = visibleElementCount > 2;
221 uiElementContainer.ChildAnchor = visibleElementCount > 1 ? Anchor.TopCenter : Anchor.Center;
222 float elementSize = Math.Min(1.0f / visibleElementCount, 1);
223 foreach (var uiElement in uiElements)
224 {
225 uiElement.RectTransform.RelativeSize = new Vector2(1.0f, elementSize);
226 }
227 GuiFrame.Visible = visibleElementCount > 0;
228 uiElementContainer.Recalculate();
229 }

305 public void ClientEventWrite(IWriteMessage msg, NetEntityEvent.IData extraData = null)
306 {
307 //extradata contains an array of buttons clicked by the player (or nothing if the player didn't click anything)
308 for (int i = 0; i < customInterfaceElementList.Count; i++)

332 public void ClientEventRead(IReadMessage msg, float sendingTime)
333 {
334 for (int i = 0; i < customInterfaceElementList.Count; i++)
335 {
336 if (customInterfaceElementList[i].HasPropertyName)
337 {
338 if (!customInterfaceElementList[i].IsIntegerInput)
339 {
340 TextChanged(customInterfaceElementList[i], msg.ReadString());
341 }
342 else
343 {
344 int.TryParse(msg.ReadString(), out int value);
345 ValueChanged(customInterfaceElementList[i], value);
346 }
347 }
348 else
349 {
350 bool elementState = msg.ReadBoolean();
351 if (customInterfaceElementList[i].ContinuousSignal)
352 {
353 ((GUITickBox)uiElements[i]).Selected = elementState;
354 TickBoxToggled(customInterfaceElementList[i], elementState);
355 }
356 else if (elementState)
357 {
358 ButtonClicked(customInterfaceElementList[i]);
359 }
360 }
361 }
362
363 UpdateSignalsProjSpecific();
364 }
