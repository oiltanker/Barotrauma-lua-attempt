558 private IEnumerable<CoroutineStatus> KillCharacterAnim(GUIComponent component)
559 {
560 List<GUIComponent> components = component.GetAllChildren().ToList();
561 components.Add(component);
562 components.RemoveAll(c =>
563 c.UserData is Pair<string, float> pair && pair.First == "soundicon" ||
564 c.UserData as string == "soundicondisabled");
565 components.ForEach(c => c.Color = Color.DarkRed);
566
567 yield return new WaitForSeconds(1.0f);
568
569 float timer = 0.0f;
570 float hideDuration = 1.0f;
571 while (timer < hideDuration)
572 {
573 foreach (GUIComponent comp in components)
574 {
575 comp.Color = Color.Lerp(Color.DarkRed, Color.Transparent, timer / hideDuration);
576 comp.RectTransform.LocalScale = new Vector2(comp.RectTransform.LocalScale.X, 1.0f - (timer / hideDuration));
577 }
578 timer += CoroutineManager.DeltaTime;
579 yield return CoroutineStatus.Running;
580 }
581
582 crewList.Content.RemoveChild(component);
583 // GUITextBlock.AutoScaleAndNormalize(list.Content.GetAllChildren<GUITextBlock>(), defaultScale: 1.0f);
584 crewList.UpdateScrollBarSize();
585
586 yield return CoroutineStatus.Success;
587 }

634 crewList.Content.RectTransform.SortChildren((x, y) =>
635 {
636 var infoX = (x.GUIComponent.UserData as Character)?.Info?.CrewListIndex;
637 var infoY = (y.GUIComponent.UserData as Character)?.Info?.CrewListIndex;
638 if (infoX.HasValue)
639 {

644 return infoY.HasValue ? 1 : 0;
645 }
646 });
