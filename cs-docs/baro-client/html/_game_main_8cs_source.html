622 protected override void Update(GameTime gameTime)
623 {
624 Timing.Accumulator += gameTime.ElapsedGameTime.TotalSeconds;
625 if (Timing.Accumulator > Timing.AccumulatorMax)
626 {
627 //prevent spiral of death:
628 //if the game's running too slowly then we have no choice but to skip a bunch of steps
629 //otherwise it snowballs and becomes unplayable
630 Timing.Accumulator = Timing.Step;
631 }
632
633 CrossThread.ProcessTasks();
634
635 PlayerInput.UpdateVariable();
636
637 if (SoundManager != null)
638 {
639 if (WindowActive || !GameSettings.CurrentConfig.Audio.MuteOnFocusLost)
640 {
641 SoundManager.ListenerGain = SoundManager.CompressionDynamicRangeGain;
642 }
643 else
644 {
645 SoundManager.ListenerGain = 0.0f;
646 }
647 }
648
649 while (Timing.Accumulator >= Timing.Step)
650 {
651 Timing.TotalTime += Timing.Step;
652
653 Stopwatch sw = new Stopwatch();
654 sw.Start();
655
656 fixedTime.IsRunningSlowly = gameTime.IsRunningSlowly;
657 TimeSpan addTime = new TimeSpan(0, 0, 0, 0, 16);
658 fixedTime.ElapsedGameTime = addTime;
659 fixedTime.TotalGameTime.Add(addTime);
660 base.Update(fixedTime);
661
662 PlayerInput.Update(Timing.Step);
663
664 if (loadingScreenOpen)
665 {
666 //reset accumulator if loading
667 // -> less choppy loading screens because the screen is rendered after each update
668 // -> no pause caused by leftover time in the accumulator when starting a new shift
669 ResetFrameTime();
670
671 if (!TitleScreen.PlayingSplashScreen)
672 {
673 SoundPlayer.Update((float)Timing.Step);
674 GUI.ClearUpdateList();
675 GUI.UpdateGUIMessageBoxesOnly((float)Timing.Step);
676 }
677
678 if (TitleScreen.LoadState >= 100.0f && !TitleScreen.PlayingSplashScreen &&
679 (!waitForKeyHit || ((PlayerInput.GetKeyboardState.GetPressedKeys().Length > 0 || PlayerInput.PrimaryMouseButtonClicked()) && WindowActive)))
680 {
681 loadingScreenOpen = false;
682 }
683
684#if DEBUG
685 if (PlayerInput.KeyHit(Keys.LeftShift))
686 {
687 CancelQuickStart = !CancelQuickStart;
688 }
689
690 if (TitleScreen.LoadState >= 100.0f && !TitleScreen.PlayingSplashScreen &&
691 (GameSettings.CurrentConfig.AutomaticQuickStartEnabled ||
692 GameSettings.CurrentConfig.AutomaticCampaignLoadEnabled ||
693 GameSettings.CurrentConfig.TestScreenEnabled) && FirstLoad && !CancelQuickStart)
694 {
695 loadingScreenOpen = false;
696 FirstLoad = false;
697
698 if (GameSettings.CurrentConfig.TestScreenEnabled)
699 {
700 TestScreen.Select();
701 }
702 else if (GameSettings.CurrentConfig.AutomaticQuickStartEnabled)
703 {
704 MainMenuScreen.QuickStart();
705 }
706 else if (GameSettings.CurrentConfig.AutomaticCampaignLoadEnabled)
707 {
708 var saveFiles = SaveUtil.GetSaveFiles(SaveUtil.SaveType.Singleplayer);
709 if (saveFiles.Count() > 0)
710 {
711 try
712 {
713 SaveUtil.LoadGame(saveFiles.OrderBy(file => file.SaveTime).Last().FilePath);
714 }
715 catch (Exception e)
716 {
717 DebugConsole.ThrowError("Loading save \"" + saveFiles.Last() + "\" failed", e);
718 return;
719 }
720 }
721 }
722 }
723#endif
724
725 NetworkMember?.Update((float)Timing.Step);
726
727 if (!HasLoaded && !CoroutineManager.IsCoroutineRunning(loadingCoroutine))
728 {
729 throw new LoadingException(loadingCoroutine.Exception);
730 }
731 }
732 else if (HasLoaded)
733 {
734 if (ConnectLobby != 0)
735 {
736 if (Client != null)
737 {
738 Client.Disconnect();
739 Client = null;
740
741 GameMain.MainMenuScreen.Select();
742 }
743 Steam.SteamManager.JoinLobby(ConnectLobby, true);
744
745 ConnectLobby = 0;
746 ConnectEndpoint = null;
747 ConnectName = null;
748 }
749 else if (!string.IsNullOrWhiteSpace(ConnectEndpoint))
750 {
751 if (Client != null)
752 {
753 Client.Disconnect();
754 Client = null;
755
756 GameMain.MainMenuScreen.Select();
757 }
758 UInt64 serverSteamId = SteamManager.SteamIDStringToUInt64(ConnectEndpoint);
759 Client = new GameClient(MultiplayerPreferences.Instance.PlayerName.FallbackNullOrEmpty(SteamManager.GetUsername()),
760 serverSteamId != 0 ? null : ConnectEndpoint,
761 serverSteamId,
762 string.IsNullOrWhiteSpace(ConnectName) ? ConnectEndpoint : ConnectName);
763 ConnectLobby = 0;
764 ConnectEndpoint = null;
765 ConnectName = null;
766 }
767
768 SoundPlayer.Update((float)Timing.Step);
769
770 if (PlayerInput.KeyHit(Keys.Escape) && WindowActive)
771 {
772 // Check if a text input is selected.
773 if (GUI.KeyboardDispatcher.Subscriber != null)
774 {
775 if (GUI.KeyboardDispatcher.Subscriber is GUITextBox textBox)
776 {
777 textBox.Deselect();
778 }
779 GUI.KeyboardDispatcher.Subscriber = null;
780 }
781 //if a verification prompt (are you sure you want to x) is open, close it
782 else if (GUIMessageBox.VisibleBox as GUIMessageBox != null &&
783 GUIMessageBox.VisibleBox.UserData as string == "verificationprompt")
784 {
785 ((GUIMessageBox)GUIMessageBox.VisibleBox).Close();
786 }
787 else if (GUIMessageBox.VisibleBox?.UserData is RoundSummary roundSummary &&
788 roundSummary.ContinueButton != null &&
789 roundSummary.ContinueButton.Visible)
790 {
791 GUIMessageBox.MessageBoxes.Remove(GUIMessageBox.VisibleBox);
792 }
793 else if (GameSession?.GameMode is TutorialMode tutorialMode && tutorialMode.Tutorial.ContentRunning)
794 {
795 tutorialMode.Tutorial.CloseActiveContentGUI();
796 }
797 else if (GameSession.IsTabMenuOpen)
798 {
799 gameSession.ToggleTabMenu();
800 }
801 else if (GUIMessageBox.VisibleBox as GUIMessageBox != null &&
802 GUIMessageBox.VisibleBox.UserData as string == "bugreporter")
803 {
804 ((GUIMessageBox)GUIMessageBox.VisibleBox).Close();
805 }
806 else if (GUI.PauseMenuOpen)
807 {
808 GUI.TogglePauseMenu();
809 }
810 //open the pause menu if not controlling a character OR if the character has no UIs active that can be closed with ESC
811 else if ((Character.Controlled == null || !itemHudActive())
812 && CharacterHealth.OpenHealthWindow == null
813 && !CrewManager.IsCommandInterfaceOpen
814 && !(Screen.Selected is SubEditorScreen editor && !editor.WiringMode && Character.Controlled?.SelectedConstruction != null))
