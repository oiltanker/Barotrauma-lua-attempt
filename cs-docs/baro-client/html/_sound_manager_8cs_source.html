111  public float PlaybackAmplitude
112  {
113  get
114  {
115  if (Disabled) { return 0.0f; }
116  float aggregateAmplitude = 0.0f;
117  //NOTE: this is obviously not entirely accurate;
118  //It assumes a linear falloff model, and assumes that audio
119  //is simply added together to produce the final result.
120  //Adjustments may be needed under certain scenarios.
121  for (int i = 0; i < 2; i++)
122  {
123  foreach (SoundChannel soundChannel in playingChannels[i].Where(ch => ch != null))
124  {
125  float amplitude = soundChannel.CurrentAmplitude;
126  amplitude *= soundChannel.Gain;
127  float dist = Vector3.Distance(ListenerPosition, soundChannel.Position ?? ListenerPosition);
128  if (dist > soundChannel.Near)
129  {
130  amplitude *= 1.0f - Math.Min(1.0f, (dist - soundChannel.Near) / (soundChannel.Far - soundChannel.Near));
131  }
132  aggregateAmplitude += amplitude;
133  }
134  }
135  return aggregateAmplitude;
136  }

625  public void Update()
626  {
627  if (Disconnected || Disabled) { return; }
628 
629  if (CanDetectDisconnect)
630  {
631  Alc.GetInteger(alcDevice, Alc.EnumConnected, out int isConnected);
632  int alcError = Alc.GetError(alcDevice);
633  if (alcError != Alc.NoError)
634  {
635  throw new Exception("Failed to determine if device is connected: " + alcError.ToString());
636  }
637 
638  if (isConnected == 0)
639  {
640  DebugConsole.ThrowError("Playback device has been disconnected. You can select another available device in the settings.");
641  SetAudioOutputDevice("<disconnected>");
642  Disconnected = true;
643  return;
644  }
645  }
646 
647  if (GameMain.Client != null && GameSettings.CurrentConfig.Audio.VoipAttenuationEnabled)
648  {
649  if (Timing.TotalTime > lastAttenuationTime+0.2)
650  {
651  voipAttenuatedGain = voipAttenuatedGain * 0.9f + 0.1f;
652  }
653  }
654  else
655  {
656  voipAttenuatedGain = 1.0f;
657  }
658  SetCategoryGainMultiplier("default", VoipAttenuatedGain, 1);
659  SetCategoryGainMultiplier("ui", VoipAttenuatedGain, 1);
660  SetCategoryGainMultiplier("waterambience", VoipAttenuatedGain, 1);
661  SetCategoryGainMultiplier("music", VoipAttenuatedGain, 1);
662 
663  if (GameSettings.CurrentConfig.Audio.DynamicRangeCompressionEnabled)
664  {
665  float targetGain = (Math.Min(1.0f, 1.0f / PlaybackAmplitude) - 1.0f) * 0.5f + 1.0f;
666  if (targetGain < CompressionDynamicRangeGain)
667  {
668  //if the target gain is lower than the current gain, lower the current gain immediately to prevent clipping
669  CompressionDynamicRangeGain = targetGain;
670  }
671  else
672  {
673  //otherwise, let it rise back smoothly
674  CompressionDynamicRangeGain = (targetGain) * 0.05f + CompressionDynamicRangeGain * 0.95f;
675  }
676  }
677  else
678  {
679  CompressionDynamicRangeGain = 1.0f;
680  }
681 
682  if (streamingThread == null || streamingThread.ThreadState.HasFlag(ThreadState.Stopped))
683  {
684  bool startedStreamThread = false;
685  for (int i = 0; i < playingChannels.Length; i++)
686  {
687  lock (playingChannels[i])
688  {
689  for (int j = 0; j < playingChannels[i].Length; j++)
690  {
691  if (playingChannels[i][j] == null) { continue; }
692  if (playingChannels[i][j].IsStream && playingChannels[i][j].IsPlaying)
693  {
694  InitStreamThread();
695  startedStreamThread = true;
696  }
697  if (startedStreamThread) { break; }
698  }
699  }
700  if (startedStreamThread) { break; }
701  }
702  }
703  }

714  public void InitStreamThread()
715  {
716  if (Disabled) { return; }
717  bool isStreamThreadDying;
718  lock (threadDeathMutex)
719  {
720  isStreamThreadDying = !areStreamsPlaying;
721  }
722  if (streamingThread == null || streamingThread.ThreadState.HasFlag(ThreadState.Stopped) || isStreamThreadDying)
723  {
724  if (streamingThread != null && !streamingThread.Join(1000))
725  {
726  DebugConsole.ThrowError("Sound stream thread join timed out!");
727  }
728  areStreamsPlaying = true;
729  streamingThread = new Thread(UpdateStreaming)
