78 
79  string[] requiredHappinessStr = element.GetAttributeString("requiredhappiness", "0-100").Split('-');
80  string[] requiredHungerStr = element.GetAttributeString("requiredhunger", "0-100").Split('-');
81  HappinessRange = new Vector2(0, 100);
82  HungerRange = new Vector2(0, 100);
83  float tempF;
84  if (requiredHappinessStr.Length >= 2)
85  {
86  if (float.TryParse(requiredHappinessStr[0], NumberStyles.Any, CultureInfo.InvariantCulture, out tempF)) { HappinessRange.X = tempF; }
87  if (float.TryParse(requiredHappinessStr[1], NumberStyles.Any, CultureInfo.InvariantCulture, out tempF)) { HappinessRange.Y = tempF; }
88  }
89  if (requiredHungerStr.Length >= 2)
90  {
91  if (float.TryParse(requiredHungerStr[0], NumberStyles.Any, CultureInfo.InvariantCulture, out tempF)) { HungerRange.X = tempF; }
92  if (float.TryParse(requiredHungerStr[1], NumberStyles.Any, CultureInfo.InvariantCulture, out tempF)) { HungerRange.Y = tempF; }
93  }
94  Rate = element.GetAttributeFloat("rate", 0.016f);
95  totalCommonness = 0.0f;
96  foreach (var subElement in element.Elements())
97  {
98  switch (subElement.Name.LocalName.ToLowerInvariant())
99  {
100  case "item":
101  Identifier identifier = subElement.GetAttributeIdentifier("identifier", Identifier.Empty);
102  Item newItemToProduce = new Item
103  {
104  Prefab = identifier.IsEmpty ? null : ItemPrefab.Find("", subElement.GetAttributeIdentifier("identifier", Identifier.Empty)),
105  Commonness = subElement.GetAttributeFloat("commonness", 0.0f)
106  };
107  totalCommonness += newItemToProduce.Commonness;
108  Items.Add(newItemToProduce);
109  break;
110  }
111  }
112 
113  timer = 1.0f;

116  public void Update(PetBehavior pet, float deltaTime)
117  {
118  if (pet.Happiness < HappinessRange.X || pet.Happiness > HappinessRange.Y) { return; }
119  if (pet.Hunger < HungerRange.X || pet.Hunger > HungerRange.Y) { return; }
120 
121  float currentRate = Rate;
122  currentRate += HappinessRate * (pet.Happiness - HappinessRange.X) / (HappinessRange.Y - HappinessRange.X);
123  currentRate += InvHappinessRate * (1.0f - ((pet.Happiness - HappinessRange.X) / (HappinessRange.Y - HappinessRange.X)));
124  currentRate += HungerRate * (pet.Hunger - HungerRange.X) / (HungerRange.Y - HungerRange.X);
125  currentRate += InvHungerRate * (1.0f - ((pet.Hunger - HungerRange.X) / (HungerRange.Y - HungerRange.X)));
126  timer -= currentRate * deltaTime;
127  if (timer <= 0.0f)
128  {
129  timer = 1.0f;
130  float r = Rand.Range(0.0f, totalCommonness);
131  float aggregate = 0.0f;
132  for (int i = 0; i < Items.Count; i++)
133  {
134  aggregate += Items[i].Commonness;
135  if (aggregate >= r && Items[i].Prefab != null)
136  {
137  GameAnalyticsManager.AddDesignEvent("MicroInteraction:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "null") + ":PetProducedItem:" + pet.AiController.Character.SpeciesName + ":" + Items[i].Prefab.Identifier);
138  Entity.Spawner.AddItemToSpawnQueue(Items[i].Prefab, pet.AiController.Character.WorldPosition);
139  break;
140  }
141  }
142  }

294  public void Update(float deltaTime)

298 
299  if (unstunY.HasValue)

304  if (character.SimPosition.Y < (unstunY.Value + extent * 3.0f) &&
305  character.AnimController.MainLimb.body.LinearVelocity.Y < 0.0f)
306  {
307  character.IsRagdolled = false;
308  unstunY = null;
309  }
310  else
311  {
312  character.IsRagdolled = true;
313  }
314  }
315  else
316  {
317  character.IsRagdolled = false;
318  unstunY = null;
319  }
320  }
321 
322  PlayTimer -= deltaTime;
323 
324  if (GameMain.NetworkMember?.IsClient ?? false) { return; }
325  if (Owner != null && (Owner.Removed || Owner.IsDead)) { Owner = null; }
326 
327  Hunger += HungerIncreaseRate * deltaTime;
328  Happiness -= HappinessDecreaseRate * deltaTime;
329 
330  for (int i = 0; i < foods.Count; i++)
331  {
332  Food food = foods[i];
333  if (Hunger >= food.HungerRange.X && Hunger <= food.HungerRange.Y)
334  {
335  if (food.TargetParams == null &&
336  AiController.AIParams.TryAddNewTarget(food.Tag, AIState.Eat, food.Priority, out CharacterParams.TargetParams targetParams))
337  {
338  targetParams.IgnoreContained = food.IgnoreContained;
339  food.TargetParams = targetParams;
340  }
341  }
342  else if (food.TargetParams != null)
343  {
344  AiController.AIParams.RemoveTarget(food.TargetParams);
345  food.TargetParams = null;
346  }
347  }
348 
349  if (Hunger >= MaxHunger * 0.99f)
350  {
351  character.CharacterHealth.ApplyAffliction(character.AnimController.MainLimb, new Affliction(AfflictionPrefab.InternalDamage, 8.0f * deltaTime));

361  unstunY = character.SimPosition.Y;
362  }

422  //if not found, try any player sub (shuttle/drone etc)
