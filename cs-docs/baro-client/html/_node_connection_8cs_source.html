280  private void DrawSquareLine(SpriteBatch spriteBatch, Vector2 position, int yOffset, float width = 2, Color? overrideColor = null)
281  {
282  // draw a line between 2 nodes using points
283  // the order of this array is messed up, I know
284  // order of points is from start node to end node: 0, 4, 1, 2, 5, 3
285  Vector2[] points = new Vector2[6];
286  int xOffset = 24 * (yOffset + 1);
287  points[0] = new Vector2(DrawRectangle.Right, DrawRectangle.Center.Y);
288  points[3] = position;
289  points[1] = points[0];
290  points[2] = points[3];
291 
292  points[4] = points[1];
293  points[5] = points[2];
294 
295  points[1].X += (points[2].X - points[1].X) / 2;
296  points[1].X = Math.Max(points[1].X, points[0].X + xOffset);
297  points[2].X = points[1].X;
298 
299  // if the node is "behind" us do some magic to make the line curve to prevent overlapping
300  if (points[1].X <= points[0].X + xOffset)
301  {
302  points[4].X += xOffset;
303  points[1].X = points[4].X;
304  points[1].Y += (points[2].Y - points[1].Y) / 2;
305  }
306 
307  if (points[2].X >= points[3].X - xOffset)
308  {
309  points[5].X -= xOffset;
310  points[2].X = points[5].X;
311  points[2].Y -= points[2].Y - points[1].Y;
312  }
313 
314  Color drawColor = Parent is ValueNode ? GetPropertyColor(ValueType) : GUIStyle.Red;
315 
316  if (overrideColor != null)
317  {
318  drawColor = overrideColor.Value;
319  }
320 
321  GUI.DrawLine(spriteBatch, points[0], points[4], drawColor, width: (int)width);
322  GUI.DrawLine(spriteBatch, points[4], points[1], drawColor, width: (int)width);
323  GUI.DrawLine(spriteBatch, points[1], points[2], drawColor, width: (int)width);
324  GUI.DrawLine(spriteBatch, points[2], points[5], drawColor, width: (int)width);
325  GUI.DrawLine(spriteBatch, points[5], points[3], drawColor, width: (int)width);
326  }
