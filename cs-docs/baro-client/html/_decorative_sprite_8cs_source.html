126 public DecorativeSprite(ContentXElement element, string path = "", string file = "", bool lazyLoad = false)
127 {
128 Sprite = new Sprite(element, path, file, lazyLoad: lazyLoad);
129 SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
130 // load property conditionals
131 foreach (var subElement in element.Elements())
132 {
133 //choose which list the new conditional should be placed to
134 List<PropertyConditional> conditionalList = null;
135 switch (subElement.Name.ToString().ToLowerInvariant())
136 {
137 case "conditional":
138 case "isactiveconditional":
139 conditionalList = IsActiveConditionals;
140 break;
141 case "animationconditional":
142 conditionalList = AnimationConditionals;
143 break;
144 default:
145 continue;
146 }
147 foreach (XAttribute attribute in subElement.Attributes())
148 {
149 if (PropertyConditional.IsValid(attribute))
150 {
151 conditionalList.Add(new PropertyConditional(attribute));
152 }
153 }
154 }

157 public Vector2 GetOffset(ref float offsetState, Vector2 randomOffsetMultiplier, float rotation = 0.0f)
158 {
159 Vector2 offset = Offset;
160 if (OffsetAnimSpeed > 0.0f)
161 {
162 switch (OffsetAnim)
163 {
164 case AnimationType.Sine:
165 offsetState %= (MathHelper.TwoPi / OffsetAnimSpeed);
166 offset *= (float)Math.Sin(offsetState * OffsetAnimSpeed);
167 break;
168 case AnimationType.Noise:
169 offsetState %= 1.0f / (OffsetAnimSpeed * 0.1f);
170
171 float t = offsetState * 0.1f * OffsetAnimSpeed;
172 offset = new Vector2(
173 offset.X * (PerlinNoise.GetPerlin(t, t) - 0.5f),
174 offset.Y * (PerlinNoise.GetPerlin(t + 0.5f, t + 0.5f) - 0.5f));
175 break;
176 }
177 }
178 offset += new Vector2(
179 RandomOffset.X * randomOffsetMultiplier.X,
180 RandomOffset.Y * randomOffsetMultiplier.Y);
181 if (Math.Abs(rotation) > 0.01f)
182 {
183 Matrix transform = Matrix.CreateRotationZ(rotation);
184 offset = Vector2.Transform(offset, transform);
185 }
186 return offset;
187 }

189 public float GetRotation(ref float rotationState, float randomRotationFactor)
190 {
191 RotationSpeed = -Math.Abs(RotationSpeed);
192 switch (RotationAnim)
193 {
194 case AnimationType.Sine:
195 rotationState %= MathHelper.TwoPi / absRotationSpeedRadians;
196 return
197 rotationRadians * (float)Math.Sin(rotationState * rotationSpeedRadians)
198 + MathHelper.Lerp(randomRotationRadians.X, randomRotationRadians.Y, randomRotationFactor);
199 case AnimationType.Noise:
200 rotationState %= 1.0f / absRotationSpeedRadians;
201 return
202 rotationRadians * (PerlinNoise.GetPerlin(rotationState * absRotationSpeedRadians, rotationState * absRotationSpeedRadians) - 0.5f)
203 + MathHelper.Lerp(randomRotationRadians.X, randomRotationRadians.Y, randomRotationFactor);
204 default:
205 return
206 rotationRadians +
207 rotationState * rotationSpeedRadians
208 + MathHelper.Lerp(randomRotationRadians.X, randomRotationRadians.Y, randomRotationFactor);
209 }
210 }

221 public static void UpdateSpriteStates(ImmutableDictionary<int, ImmutableArray<DecorativeSprite>> spriteGroups, Dictionary<DecorativeSprite, State> animStates,
222 int entityID, float deltaTime, Func<PropertyConditional, bool> checkConditional)
223 {
224 foreach (int spriteGroup in spriteGroups.Keys)
225 {
226 for (int i = 0; i < spriteGroups[spriteGroup].Length; i++)
227 {
228 var decorativeSprite = spriteGroups[spriteGroup][i];
229 if (decorativeSprite == null) { continue; }
230 if (spriteGroup > 0)
231 {
232 int activeSpriteIndex = entityID % spriteGroups[spriteGroup].Length;
233 if (i != activeSpriteIndex)
234 {
235 animStates[decorativeSprite].IsActive = false;
236 continue;
237 }
238 }
239
240 //check if the sprite is active (whether it should be drawn or not)
241 var spriteState = animStates[decorativeSprite];
242 spriteState.IsActive = true;
243 foreach (PropertyConditional conditional in decorativeSprite.IsActiveConditionals)
244 {
245 if (!checkConditional(conditional))
246 {
247 spriteState.IsActive = false;
248 break;
249 }
250 }
251 if (!spriteState.IsActive) { continue; }
252
253 //check if the sprite should be animated
254 bool animate = true;
255 foreach (PropertyConditional conditional in decorativeSprite.AnimationConditionals)
256 {
257 if (!checkConditional(conditional)) { animate = false; break; }
258 }
259 if (!animate) { continue; }
260 spriteState.OffsetState += deltaTime;
261 spriteState.RotationState += deltaTime;
262 }
263 }
