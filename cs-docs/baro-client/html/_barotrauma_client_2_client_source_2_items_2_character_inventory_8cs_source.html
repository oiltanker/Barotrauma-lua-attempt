1049  void Equip()
1050  {
1051  //attempt to put in a free slot first
1052  for (int i = capacity - 1; i >= 0; i--)
1053  {
1054  if (!slots[i].Empty()) { continue; }
1055  if (SlotTypes[i] == InvSlotType.Any || !item.AllowedSlots.Any(a => a.HasFlag(SlotTypes[i]))) { continue; }
1056  success = TryPutItem(item, i, true, false, Character.Controlled, true);
1057  if (success) { break; }
1058  }
1059 
1060  if (!success)
1061  {
1062  for (int i = capacity - 1; i >= 0; i--)
1063  {
1064  if (SlotTypes[i] == InvSlotType.Any || !item.AllowedSlots.Any(a => a.HasFlag(SlotTypes[i]))) { continue; }
1065  // something else already equipped in a hand slot, attempt to unequip it so items aren't unnecessarily swapped to it
1066  if (!slots[i].Empty() && slots[i].First().AllowedSlots.Contains(InvSlotType.Any) &&
1067  (SlotTypes[i] == InvSlotType.LeftHand || SlotTypes[i] == InvSlotType.RightHand))
1068  {
1069  TryPutItem(slots[i].First(), Character.Controlled, new List<InvSlotType>() { InvSlotType.Any }, true);
1070  }
1071  success = TryPutItem(item, i, true, false, Character.Controlled, true);
1072  if (success) { break; }
1073  }
1074  }
1075  }
1076  break;
1077  case QuickUseAction.Unequip:
1078  if (item.AllowedSlots.Contains(InvSlotType.Any))
1079  {
1080  success = TryPutItem(item, Character.Controlled, new List<InvSlotType>() { InvSlotType.Any }, true);
1081  }
1082  break;
1083  case QuickUseAction.UseTreatment:
1084  CharacterHealth.OpenHealthWindow?.OnItemDropped(item, ignoreMousePos: true);
1085  return;
1086  case QuickUseAction.Drop:
1087  //do nothing, the item is dropped after a delay
1088  return;
1089  case QuickUseAction.PutToCharacter:
1090  if (character.SelectedCharacter != null && character.SelectedCharacter.Inventory != null)
1091  {
1092  //player has selected the inventory of another character -> attempt to move the item there
1093  success = character.SelectedCharacter.Inventory.TryPutItem(item, Character.Controlled, item.AllowedSlots, true);
1094  }
1095  break;
1096  case QuickUseAction.PutToContainer:
1097  var selectedContainer = character.SelectedConstruction?.GetComponent<ItemContainer>();
1098  if (selectedContainer != null && selectedContainer.Inventory != null)
1099  {
1100  //player has selected the inventory of another item -> attempt to move the item there
1101  success = selectedContainer.Inventory.TryPutItem(item, Character.Controlled, item.AllowedSlots, true);
1102  }
1103  break;
1104  case QuickUseAction.TakeFromCharacter:
1105  if (character.SelectedBy != null && Character.Controlled == character.SelectedBy &&
1106  character.SelectedBy.Inventory != null)
1107  {
1108  //item is in the inventory of another character -> attempt to get the item from there
1109  success = character.SelectedBy.Inventory.TryPutItemWithAutoEquipCheck(item, Character.Controlled, item.AllowedSlots, true);
1110  }
1111  break;
1112  case QuickUseAction.TakeFromContainer:
1113  // Check open subinventories and put the item in it if equipped
1114  ItemInventory activeSubInventory = null;
1115  for (int i = 0; i < capacity; i++)
1116  {
1117  activeSubInventory = GetActiveEquippedSubInventory(i);
1118  if (activeSubInventory != null)
1119  {
1120  success = activeSubInventory.TryPutItem(item, Character.Controlled, item.AllowedSlots, true);
1121  break;
1122  }
1123  }
1124 
1125  // No subinventory found or placing unsuccessful -> attempt to put in the character's inventory
1126  if (!success)
1127  {
1128  success = TryPutItemWithAutoEquipCheck(item, Character.Controlled, item.AllowedSlots, true);
1129  }
1130  break;
1131  case QuickUseAction.PutToEquippedItem:
1132  foreach (Item heldItem in character.HeldItems)
1133  {
1134  if (heldItem.OwnInventory == null) { continue; }
1135  if (heldItem.OwnInventory.TryPutItem(item, Character.Controlled) ||
1136  (heldItem.OwnInventory.Capacity == 1 && heldItem.OwnInventory.TryPutItem(item, 0, allowSwapping: true, allowCombine: false, user: Character.Controlled)))
1137  {
1138  success = true;
1139  for (int j = 0; j < capacity; j++)
1140  {
1141  if (slots[j].Contains(heldItem)) { visualSlots[j].ShowBorderHighlight(GUIStyle.Green, 0.1f, 0.4f); }
1142  }
1143  break;
1144  }
1145  }
1146  break;
1147  }
1148 
1149  if (success)
1150  {
1151  for (int i = 0; i < capacity; i++)
1152  {
1153  if (slots[i].Contains(item)) { visualSlots[i].ShowBorderHighlight(GUIStyle.Green, 0.1f, 0.4f); }
1154  }
1155  }
1156 
1157  DraggingItems.Clear();
1158  if (playSound)
1159  {
1160  SoundPlayer.PlayUISound(success ? GUISoundType.PickItem : GUISoundType.PickItemFail);
1161  }
1162  }
