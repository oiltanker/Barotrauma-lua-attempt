151 public static float CurveAngle(float from, float to, float step)
152 {
153
154 from = WrapAngleTwoPi(from);
155 to = WrapAngleTwoPi(to);
156
157 if (Math.Abs(from - to) < MathHelper.Pi)
158 {
159 // The simple case - a straight lerp will do.
160 return MathHelper.Lerp(from, to, step);
161 }
162
163 // If we get here we have the more complex case.
164 // First, increment the lesser value to be greater.
165 if (from < to)
166 from += MathHelper.TwoPi;
167 else
168 to += MathHelper.TwoPi;
169
170 float retVal = MathHelper.Lerp(from, to, step);
171
172 // Now ensure the return value is between 0 and 2pi
173 if (retVal >= MathHelper.TwoPi)
174 retVal -= MathHelper.TwoPi;
175 return retVal;
176 }

182 {
183 if (float.IsInfinity(angle) || float.IsNegativeInfinity(angle) || float.IsNaN(angle))
184 {
185 return 0.0f;
186 }
187
188 while (angle < 0)
189 angle += MathHelper.TwoPi;
190 while (angle >= MathHelper.TwoPi)
191 angle -= MathHelper.TwoPi;
192
193 return angle;

284 public static bool LinesIntersect(Vector2 a, Vector2 b, Vector2 c, Vector2 d)
285 {
286 float denominator = ((b.X - a.X) * (d.Y - c.Y)) - ((b.Y - a.Y) * (d.X - c.X));
287 float numerator1 = ((a.Y - c.Y) * (d.X - c.X)) - ((a.X - c.X) * (d.Y - c.Y));
288 float numerator2 = ((a.Y - c.Y) * (b.X - a.X)) - ((a.X - c.X) * (b.Y - a.Y));
289
290 if (denominator == 0) return numerator1 == 0 && numerator2 == 0;
291
292 float r = numerator1 / denominator;
293 float s = numerator2 / denominator;
294
295 return (r >= 0 && r <= 1) && (s >= 0 && s <= 1);
296 }

304 public static bool GetLineIntersection(Vector2 a1, Vector2 a2, Vector2 b1, Vector2 b2, bool ignoreSegments, out Vector2 intersection)
305 {
306 intersection = Vector2.Zero;
307
308 Vector2 b = a2 - a1;
309 Vector2 d = b2 - b1;
310 float bDotDPerp = b.X * d.Y - b.Y * d.X;
311
312 // if b dot d == 0, it means the lines are parallel so have infinite intersection points
313 if (bDotDPerp == 0) return false;
314
315 Vector2 c = b1 - a1;
316 float t = (c.X * d.Y - c.Y * d.X) / bDotDPerp;
317 if ((t < 0 || t > 1) && !ignoreSegments) return false;
318
319 float u = (c.X * b.Y - c.Y * b.X) / bDotDPerp;
320 if ((u < 0 || u > 1) && !ignoreSegments) return false;
321
322 intersection = a1 + t * b;
323 return true;
324 }

326 public static bool GetAxisAlignedLineIntersection(Vector2 a1, Vector2 a2, Vector2 axisAligned1, Vector2 axisAligned2, bool isHorizontal, out Vector2 intersection)
327 {
328 intersection = Vector2.Zero;
329
330 if (!isHorizontal)
331 {
332 float xDiff = axisAligned1.X - a1.X;
333 if (Math.Sign(xDiff) == Math.Sign(axisAligned1.X - a2.X)) { return false; }
334
335 float s = (a2.Y - a1.Y) / (a2.X - a1.X);
336 float y = a1.Y + xDiff * s;
337
338 if (axisAligned1.Y < axisAligned2.Y)
339 {
340 if (y < axisAligned1.Y) return false;
341 if (y > axisAligned2.Y) return false;
342 }
343 else
344 {
345 if (y > axisAligned1.Y) return false;
346 if (y < axisAligned2.Y) return false;
347 }
348
349 intersection = new Vector2(axisAligned1.X, y);
350 return true;
351 }
352 else //horizontal line
353 {
354 float yDiff = axisAligned1.Y - a1.Y;
355 if (Math.Sign(yDiff) == Math.Sign(axisAligned1.Y - a2.Y)) { return false; }
356
357 float s = (a2.X - a1.X) / (a2.Y - a1.Y);
358 float x = a1.X + yDiff * s;
359
360 if (axisAligned1.X < axisAligned2.X)
361 {
362 if (x < axisAligned1.X) return false;
363 if (x > axisAligned2.X) return false;
364 }
365 else
366 {
367 if (x > axisAligned1.X) return false;
368 if (x < axisAligned2.X) return false;
369 }
370
371 intersection = new Vector2(x, axisAligned1.Y);
372 return true;
373 }

458 public static int GetLineCircleIntersections(Vector2 circlePos, float radius,
459 Vector2 point1, Vector2 point2, bool isLineSegment, out Vector2? intersection1, out Vector2? intersection2)
460 {
461 float dx, dy, A, B, C, det;
462
463 dx = point2.X - point1.X;
464 dy = point2.Y - point1.Y;
465
466 A = dx * dx + dy * dy;
467 B = 2 * (dx * (point1.X - circlePos.X) + dy * (point1.Y - circlePos.Y));
468 C = (point1.X - circlePos.X) * (point1.X - circlePos.X) + (point1.Y - circlePos.Y) * (point1.Y - circlePos.Y) - radius * radius;
469
470 det = B * B - 4 * A * C;
471 if ((A <= 0.0000001) || (det < 0))
472 {
473 // No real solutions.
474 intersection1 = null;
475 intersection2 = null;
476 return 0;
477 }
478 else if (det == 0)
479 {
480 // One solution.
481 float t = -B / (2 * A);
482 intersection1 = new Vector2(point1.X + t * dx, point1.Y + t * dy);
483 intersection2 = null;
484 return 1;
485 }
486 else
487 {
488 // Two solutions.
489 float t1 = (float)((-B + Math.Sqrt(det)) / (2 * A));
490 float t2 = (float)((-B - Math.Sqrt(det)) / (2 * A));
491
492 //if the line is not infinite, we need to check if the intersections are on the segment
493 if (isLineSegment)
494 {
495 if (t1 >= 0 && t1 <= 1.0f)
496 {
497 intersection1 = point1 + new Vector2(dx, dy) * t1;
498 if (t2 >= 0 && t2 <= 1.0f)
499 {
500 //both intersections on the segment
501 intersection2 = point1 + new Vector2(dx, dy) * t2;
502 return 2;
503
504 }
505 //only the first intersection is on the segment
506 intersection2 = null;
507 return 1;
508 }
509 else if (t2 >= 0 && t2 <= 1.0f)
510 {
511 //only the second intersection is on the segment
512 intersection1 = point1 + new Vector2(dx, dy) * t2;
513 intersection2 = null;
514 return 1;
515 }
516 else
517 {
518 //neither is on the segment
519 intersection1 = null;
520 intersection2 = null;
521 return 0;
522 }
523 }
524 else
525 {
526 intersection1 = point1 + new Vector2(dx, dy) * t1;
527 intersection2 = point1 + new Vector2(dx, dy) * t2;
528 return 2;
529 }
530 }

534 {

561 public static double LineSegmentToPointDistanceSquared(Point lineA, Point lineB, Point point)
562 {
563 double xDiff = lineB.X - lineA.X;
564 double yDiff = lineB.Y - lineA.Y;
565
566 if (xDiff == 0 && yDiff == 0)
567 {
568 double v1 = lineA.X - point.X;
569 double v2 = lineA.Y - point.Y;
570 return (v1 * v1) + (v2 * v2);
571 }
572
573 // Calculate the t that minimizes the distance.
574 double t = ((point.X - lineA.X) * xDiff + (point.Y - lineA.Y) * yDiff) / (xDiff * xDiff + yDiff * yDiff);
575
576 // See if this represents one of the segment's
577 // end points or a point in the middle.
578 if (t < 0)
579 {
580 xDiff = point.X - lineA.X;
581 yDiff = point.Y - lineA.Y;
582 }
583 else if (t > 1)
584 {
585 xDiff = point.X - lineB.X;
586 yDiff = point.Y - lineB.Y;
587 }
588 else
589 {
590 xDiff = point.X - (lineA.X + t * xDiff);
591 yDiff = point.Y - (lineA.Y + t * yDiff);
592 }
593
594 return xDiff * xDiff + yDiff * yDiff;
595 }

626 public static bool CircleIntersectsRectangle(Vector2 circlePos, float radius, Rectangle rect)
627 {
628 int halfWidth = rect.Width / 2;
629 float xDist = Math.Abs(circlePos.X - (rect.X + halfWidth));
630 if (xDist > halfWidth + radius) { return false; }
631
632 int halfHeight = rect.Height / 2;
633 float yDist = Math.Abs(circlePos.Y - (rect.Y + halfHeight));
634 if (yDist > halfHeight + radius) { return false; }
635
636 if (xDist <= halfWidth || yDist <= halfHeight) { return true; }
637
638 float distSqX = xDist - halfWidth;
639 float distSqY = yDist - halfHeight;
640
641 return distSqX * distSqX + distSqY * distSqY <= radius * radius;
642 }

695 public static List<Vector2> GiftWrap(List<Vector2> points)
696 {
697 if (points.Count == 0) return points;
698
699 Vector2 leftMost = points[0];
700 foreach (Vector2 point in points)
701 {
702 if (point.X < leftMost.X) leftMost = point;
703 }
704
705 List<Vector2> wrappedPoints = new List<Vector2>();
706
707 Vector2 currPoint = leftMost;
708 Vector2 endPoint;
709 do
710 {
711 wrappedPoints.Add(currPoint);
712 endPoint = points[0];
713
714 for (int i = 1; i < points.Count; i++)
715 {
716 if (points[i].NearlyEquals(currPoint)) continue;
717 if (currPoint == endPoint ||
718 MathUtils.VectorOrientation(currPoint, endPoint, points[i]) == -1)
719 {
720 endPoint = points[i];
721 }
722 }
723
724 currPoint = endPoint;
725
726 }
727 while (endPoint != leftMost);
728
729 return wrappedPoints;
730 }

732 public static List<Vector2[]> GenerateJaggedLine(Vector2 start, Vector2 end, int iterations, float offsetAmount, Rectangle? bounds = null)
733 {
734 List<Vector2[]> segments = new List<Vector2[]>
735 {
736 new Vector2[] { start, end }
737 };
738
739 for (int n = 0; n < iterations; n++)
740 {
741 for (int i = 0; i < segments.Count; i++)
742 {
743 Vector2 startSegment = segments[i][0];
744 Vector2 endSegment = segments[i][1];
745
746 segments.RemoveAt(i);
747
748 Vector2 midPoint = (startSegment + endSegment) / 2.0f;
749
750 Vector2 normal = Vector2.Normalize(endSegment - startSegment);
751 normal = new Vector2(-normal.Y, normal.X);
752 midPoint += normal * Rand.Range(-offsetAmount, offsetAmount, Rand.RandSync.ServerAndClient);
753
754 if (bounds.HasValue)
755 {
756 if (midPoint.X < bounds.Value.X)
757 {
758 midPoint.X = bounds.Value.X + (bounds.Value.X - midPoint.X);
759 }
760 else if (midPoint.X > bounds.Value.Right)
761 {
762 midPoint.X = bounds.Value.Right - (midPoint.X - bounds.Value.Right);
763 }
764 if (midPoint.Y < bounds.Value.Y)
765 {
766 midPoint.Y = bounds.Value.Y + (bounds.Value.Y - midPoint.Y);
767 }
768 else if (midPoint.Y > bounds.Value.Bottom)
769 {
770 midPoint.Y = bounds.Value.Bottom - (midPoint.Y - bounds.Value.Bottom);
771 }
772 }
773
774 segments.Insert(i, new Vector2[] { startSegment, midPoint });
775 segments.Insert(i + 1, new Vector2[] { midPoint, endSegment });
776
777 i++;
778 }
779
780 offsetAmount *= 0.5f;
781 }
782
783 return segments;
784 }

786 // Returns the human-readable file size for an arbitrary, 64-bit file size

888 {

1042 public static bool TriangleContainsPoint(Vector2 c1, Vector2 c2, Vector2 c3, Vector2 point)
1043 {
1044 // Compute vectors
1045 Vector2 v0 = c3 - c1;
1046 Vector2 v1 = c2 - c1;
1047 Vector2 v2 = point - c1;
1048
1049 // Compute dot products
1050 float dot00 = Vector2.Dot(v0, v0);
1051 float dot01 = Vector2.Dot(v0, v1);
1052 float dot02 = Vector2.Dot(v0, v2);
1053 float dot11 = Vector2.Dot(v1, v1);
1054 float dot12 = Vector2.Dot(v1, v2);
1055
1056 // Compute barycentric coordinates
1057 float invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
1058 float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
1059 float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
1060
1061 // Check if the point is in triangle
1062 return u >= 0 && v >= 0 && (u + v) < 1;
1063 }

1100 public static int Compare(Vector2 a, Vector2 b, Vector2 center)
1101 {
1102 if (a == b) return 0;
1103 if (a.X - center.X >= 0 && b.X - center.X < 0) return -1;
1104 if (a.X - center.X < 0 && b.X - center.X >= 0) return 1;
1105 if (a.X - center.X == 0 && b.X - center.X == 0)
1106 {
1107 if (a.Y - center.Y >= 0 || b.Y - center.Y >= 0) return Math.Sign(b.Y - a.Y);
1108 return Math.Sign(a.Y - b.Y);
1109 }
1110
1111 // compute the cross product of vectors (center -> a) x (center -> b)
1112 float det = (a.X - center.X) * (b.Y - center.Y) - (b.X - center.X) * (a.Y - center.Y);
1113 if (det < 0) return -1;
1114 if (det > 0) return 1;
1115
1116 // points a and b are on the same line from the center
1117 // check which point is closer to the center
1118 float d1 = (a.X - center.X) * (a.X - center.X) + (a.Y - center.Y) * (a.Y - center.Y);
1119 float d2 = (b.X - center.X) * (b.X - center.X) + (b.Y - center.Y) * (b.Y - center.Y);
1120 return Math.Sign(d2 - d1);
1121 }
