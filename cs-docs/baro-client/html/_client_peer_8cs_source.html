88  {
89  ConnectionInitialization step = (ConnectionInitialization)inc.ReadByte();
90 
91  IWriteMessage outMsg;
92 
93  switch (step)
94  {
95  case ConnectionInitialization.SteamTicketAndVersion:
96  if (initializationStep != ConnectionInitialization.SteamTicketAndVersion) { return; }
97  outMsg = new WriteOnlyMessage();
98  outMsg.Write((byte)PacketHeader.IsConnectionInitializationStep);
99  outMsg.Write((byte)ConnectionInitialization.SteamTicketAndVersion);
100  outMsg.Write(Name);
101  outMsg.Write(ownerKey);
102  outMsg.Write(SteamManager.GetSteamID());
103  if (steamAuthTicket == null)
104  {
105  outMsg.Write((UInt16)0);
106  }
107  else
108  {
109  outMsg.Write((UInt16)steamAuthTicket.Data.Length);
110  outMsg.Write(steamAuthTicket.Data, 0, steamAuthTicket.Data.Length);
111  }
112  outMsg.Write(GameMain.Version.ToString());
113  outMsg.Write(GameSettings.CurrentConfig.Language.Value);
114 
115  SendMsgInternal(DeliveryMethod.Reliable, outMsg);
116  break;
117  case ConnectionInitialization.ContentPackageOrder:
118  if (initializationStep == ConnectionInitialization.SteamTicketAndVersion ||
119  initializationStep == ConnectionInitialization.Password) { initializationStep = ConnectionInitialization.ContentPackageOrder; }
120  if (initializationStep != ConnectionInitialization.ContentPackageOrder) { return; }
121  outMsg = new WriteOnlyMessage();
122  outMsg.Write((byte)PacketHeader.IsConnectionInitializationStep);
123  outMsg.Write((byte)ConnectionInitialization.ContentPackageOrder);
124 
125  string serverName = inc.ReadString();
126 
127  UInt32 packageCount = inc.ReadVariableUInt32();
128  List<ServerContentPackage> serverPackages = new List<ServerContentPackage>();
129  for (int i = 0; i < packageCount; i++)
130  {
131  string name = inc.ReadString();
132  UInt32 hashByteCount = inc.ReadVariableUInt32();
133  byte[] hashBytes = inc.ReadBytes((int)hashByteCount);
134  UInt64 workshopId = inc.ReadUInt64();
135  UInt32 installTimeDiffSeconds = inc.ReadUInt32();
136  DateTime installTime = DateTime.UtcNow + TimeSpan.FromSeconds(installTimeDiffSeconds);
137 
138  var pkg = new ServerContentPackage(name, Md5Hash.BytesAsHash(hashBytes), workshopId, installTime);
139  serverPackages.Add(pkg);
140  }
141 
142  if (!contentPackageOrderReceived)
143  {
144  ServerContentPackages = serverPackages.ToImmutableArray();
145  SendMsgInternal(DeliveryMethod.Reliable, outMsg);
146  }
147  break;
148  case ConnectionInitialization.Password:
149  if (initializationStep == ConnectionInitialization.SteamTicketAndVersion) { initializationStep = ConnectionInitialization.Password; }
150  if (initializationStep != ConnectionInitialization.Password) { return; }
151  bool incomingSalt = inc.ReadBoolean(); inc.ReadPadBits();
152  int retries = 0;
153  if (incomingSalt)
154  {
155  passwordSalt = inc.ReadInt32();
156  }
157  else
158  {
159  retries = inc.ReadInt32();
160  }
161  OnRequestPassword?.Invoke(passwordSalt, retries);
162  break;
163  }
