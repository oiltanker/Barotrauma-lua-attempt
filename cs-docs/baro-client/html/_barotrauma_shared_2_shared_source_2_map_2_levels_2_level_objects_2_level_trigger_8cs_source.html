301 public static Category GetCollisionCategories(TriggererType triggeredBy)
302 {
303 var collidesWith = Physics.CollisionNone;

310 private void CalculateDirectionalForce()
311 {
312 var ca = (float)Math.Cos(-Rotation);
313 var sa = (float)Math.Sin(-Rotation);
314
315 Force = new Vector2(
316 ca * unrotatedForce.X + sa * unrotatedForce.Y,
317 -sa * unrotatedForce.X + ca * unrotatedForce.Y);

340 private bool PhysicsBody_OnCollision(Fixture fixtureA, Fixture fixtureB, Contact contact)
341 {
342 Entity entity = GetEntity(fixtureB);
343 if (entity == null) { return false; }
344 if (!IsTriggeredByEntity(entity, triggeredBy, mustBeOutside: true)) { return false; }
345 if (!triggerers.Contains(entity))
346 {
347 if (!IsTriggered)
348 {
349 OnTriggered?.Invoke(this, entity);
350 }
351 TriggererPosition[entity] = entity.WorldPosition;
352 triggerers.Add(entity);
353 }
354 return true;

357 public static bool IsTriggeredByEntity(Entity entity, TriggererType triggeredBy, bool mustBeOutside = false, (bool mustBe, Submarine sub) mustBeOnSpecificSub = default)
358 {
359 if (entity is Character character)
360 {
361 if (mustBeOutside && character.CurrentHull != null) { return false; }
362 if (mustBeOnSpecificSub.mustBe && character.Submarine != mustBeOnSpecificSub.sub) { return false; }
363 if (character.IsHuman)
364 {
365 if (!triggeredBy.HasFlag(TriggererType.Human)) { return false; }
366 }
367 else
368 {
369 if (!triggeredBy.HasFlag(TriggererType.Creature)) { return false; }
370 }
371 }
372 else if (entity is Item item)
373 {
374 if (mustBeOutside && item.CurrentHull != null) { return false; }
375 if (mustBeOnSpecificSub.mustBe && item.Submarine != mustBeOnSpecificSub.sub) { return false; }
376 if (!triggeredBy.HasFlag(TriggererType.Item)) { return false; }
377 }
378 else if (entity is Submarine)
379 {
380 if (!triggeredBy.HasFlag(TriggererType.Submarine)) { return false; }
381 }
382 return true;

408 public static bool CheckContactsForOtherFixtures(PhysicsBody triggerBody, Fixture otherFixture, Entity separatingEntity)
409 {
410 //check if there are contacts with any other fixture of the trigger
411 //(the OnSeparation callback happens when two fixtures separate,
412 //e.g. if a body stops touching the circular fixture at the end of a capsule-shaped body)
413 foreach (Fixture fixture in triggerBody.FarseerBody.FixtureList)
414 {
415 ContactEdge contactEdge = fixture.Body.ContactList;
416 while (contactEdge != null)
417 {
418 if (contactEdge.Contact != null &&
419 contactEdge.Contact.Enabled &&
420 contactEdge.Contact.IsTouching)
421 {
422 if (contactEdge.Contact.FixtureA != fixture && contactEdge.Contact.FixtureB != fixture)
423 {
424 var otherEntity = GetEntity(contactEdge.Contact.FixtureB == otherFixture ?
425 contactEdge.Contact.FixtureB :
426 contactEdge.Contact.FixtureA);
427 if (otherEntity == separatingEntity) { return true; }
428 }
429 }
430 contactEdge = contactEdge.Next;
431 }
432 }
433 return false;
434 }

439 public static bool CheckContactsForEntity(PhysicsBody triggerBody, Entity targetEntity)
440 {
441 foreach (Fixture fixture in triggerBody.FarseerBody.FixtureList)
442 {
443 ContactEdge contactEdge = fixture.Body.ContactList;
444 while (contactEdge != null)
445 {
446 if (contactEdge.Contact != null &&
447 contactEdge.Contact.Enabled &&
448 contactEdge.Contact.IsTouching)
449 {
450 if ((contactEdge.Contact.FixtureA.Body == triggerBody.FarseerBody && GetEntity(contactEdge.Contact.FixtureB) == targetEntity) ||
451 (contactEdge.Contact.FixtureB.Body == triggerBody.FarseerBody && GetEntity(contactEdge.Contact.FixtureA) == targetEntity))
452 {
453 return true;
454 }
455 }
456 contactEdge = contactEdge.Next;
457 }
458 }
459 return false;

475 {

497 public void Update(float deltaTime)
498 {
499 if (ParentTrigger != null && !ParentTrigger.IsTriggered) { return; }
500
501
502 bool isNotClient = true;
503#if CLIENT
504 isNotClient = GameMain.Client == null;
505#endif
506
507 if (!UseNetworkSyncing || isNotClient)
508 {
509 if (GlobalForceDecreaseInterval > 0.0f && Level.Loaded?.LevelObjectManager != null &&
510 Level.Loaded.LevelObjectManager.GlobalForceDecreaseTimer % (GlobalForceDecreaseInterval * 2) < GlobalForceDecreaseInterval)
511 {
512 NeedsNetworkSyncing |= currentForceFluctuation > 0.0f;
513 currentForceFluctuation = 0.0f;
514 }
515 else if (ForceFluctuationStrength > 0.0f)
516 {
517 //no need for force fluctuation (or network updates) if the trigger limits velocity and there are no triggerers
518 if (forceMode != TriggerForceMode.LimitVelocity || triggerers.Any())
519 {
520 forceFluctuationTimer += deltaTime;
521 if (forceFluctuationTimer > ForceFluctuationInterval)
522 {
523 NeedsNetworkSyncing = true;
524 currentForceFluctuation = Rand.Range(1.0f - ForceFluctuationStrength, 1.0f);
525 forceFluctuationTimer = 0.0f;
526 }
527 }
528 }
529
530 if (randomTriggerProbability > 0.0f)
531 {
532 randomTriggerTimer += deltaTime;
533 if (randomTriggerTimer > randomTriggerInterval)
534 {
535 if (Rand.Range(0.0f, 1.0f) < randomTriggerProbability)
536 {
537 NeedsNetworkSyncing = true;
538 triggeredTimer = stayTriggeredDelay;
539 }
540 randomTriggerTimer = 0.0f;
541 }
542 }
543 }
544
545 RemoveInActiveTriggerers(PhysicsBody, triggerers);
546
547 if (stayTriggeredDelay > 0.0f)
548 {
549 if (triggerers.Count == 0)
550 {
551 triggeredTimer -= deltaTime;
552 }
553 else
554 {
555 triggeredTimer = stayTriggeredDelay;
556 }
557 }
558
559 if (triggerOnce && triggeredOnce)
560 {
561 return;
562 }
563
564 foreach (Entity triggerer in triggerers)
565 {
566 if (triggerer.Removed) { continue; }
567
568 ApplyStatusEffects(statusEffects, worldPosition, triggerer, deltaTime, targets);
569
570 if (triggerer is IDamageable damageable)
571 {
572 ApplyAttacks(attacks, damageable, worldPosition, deltaTime);
573 }
574 else if (triggerer is Submarine submarine)
575 {
576 ApplyAttacks(attacks, worldPosition, deltaTime);
577 if (!InfectIdentifier.IsEmpty)
578 {
579 submarine.AttemptBallastFloraInfection(InfectIdentifier, deltaTime, InfectionChance);
580 }
581 }
582
583 if (Force.LengthSquared() > 0.01f)
584 {
585 if (triggerer is Character character)
586 {
587 ApplyForce(character.AnimController.Collider);
588 foreach (Limb limb in character.AnimController.Limbs)
589 {
590 if (limb.IsSevered) { continue; }
591 ApplyForce(limb.body);
592 }
593 }
594 else if (triggerer is Submarine submarine)
595 {
596 ApplyForce(submarine.SubBody.Body);
597 }
598 }
599
600 if (triggerer == Character.Controlled || triggerer == Character.Controlled?.Submarine)
601 {
602 GameMain.GameScreen.Cam.Shake = Math.Max(GameMain.GameScreen.Cam.Shake, cameraShake);
603 }
604 }
605
606 if (triggerOnce && triggerers.Count > 0)
607 {
608 PhysicsBody.Enabled = false;
609 triggeredOnce = true;
610 }
611 }

637 {
638 foreach (StatusEffect effect in statusEffects)
639 {
640 if (effect.type == ActionType.OnBroken) { return; }

642 if (effect.HasTargetType(StatusEffect.TargetType.This)) { position = worldPosition; }
643 if (triggerer is Character character)
644 {
645 effect.Apply(effect.type, deltaTime, triggerer, character, position);
646 if (effect.HasTargetType(StatusEffect.TargetType.Contained) && character.Inventory != null)
647 {

649 {

651 foreach (Item containedItem in item.ContainedItems)

653 effect.Apply(effect.type, deltaTime, triggerer, containedItem.AllPropertyObjects, position);
654 }
655 }
656 }
657 }

659 {
660 effect.Apply(effect.type, deltaTime, triggerer, item.AllPropertyObjects, position);

662 if (effect.HasTargetType(StatusEffect.TargetType.NearbyItems) || effect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))

697 private void ApplyForce(PhysicsBody body)
698 {
699 if (body == null) { return; }
700
701 float distFactor = 1.0f;
702 if (ForceFalloff)
703 {
704 distFactor = GetDistanceFactor(body, PhysicsBody, ColliderRadius);
705 if (distFactor < 0.0f) return;
706 }
707
708 switch (ForceMode)
709 {
710 case TriggerForceMode.Force:
711 if (ForceVelocityLimit < 1000.0f)
712 body.ApplyForce(Force * currentForceFluctuation * distFactor, ForceVelocityLimit);
713 else
714 body.ApplyForce(Force * currentForceFluctuation * distFactor);
715 break;
716 case TriggerForceMode.Acceleration:
717 if (ForceVelocityLimit < 1000.0f)
718 body.ApplyForce(Force * body.Mass * currentForceFluctuation * distFactor, ForceVelocityLimit);
719 else
720 body.ApplyForce(Force * body.Mass * currentForceFluctuation * distFactor);
721 break;
722 case TriggerForceMode.Impulse:
723 if (ForceVelocityLimit < 1000.0f)
724 body.ApplyLinearImpulse(Force * currentForceFluctuation * distFactor, maxVelocity: ForceVelocityLimit);
725 else
726 body.ApplyLinearImpulse(Force * currentForceFluctuation * distFactor);
727 break;
728 case TriggerForceMode.LimitVelocity:
729 float maxVel = ForceVelocityLimit * currentForceFluctuation * distFactor;
730 if (body.LinearVelocity.LengthSquared() > maxVel * maxVel)
731 {
732 body.ApplyForce(
733 Vector2.Normalize(-body.LinearVelocity) *
734 Force.Length() * body.Mass * currentForceFluctuation * distFactor,
735 maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
736 }
737 break;
738 }
739 }

746 public Vector2 GetWaterFlowVelocity(Vector2 viewPosition)
747 {
748 Vector2 baseVel = GetWaterFlowVelocity();
749 if (baseVel.LengthSquared() < 0.1f) return Vector2.Zero;
750
751 float triggerSize = ConvertUnits.ToDisplayUnits(Math.Max(Math.Max(PhysicsBody.radius, PhysicsBody.width / 2.0f), PhysicsBody.height / 2.0f));
752 float dist = Vector2.Distance(viewPosition, WorldPosition);
753 if (dist > triggerSize) return Vector2.Zero;
754
755 return baseVel * (1.0f - dist / triggerSize);
756 }

758 public Vector2 GetWaterFlowVelocity()
759 {
760 if (Force == Vector2.Zero || ForceMode == TriggerForceMode.LimitVelocity) { return Vector2.Zero; }
761
762 Vector2 vel = Force;
763 if (ForceMode == TriggerForceMode.Acceleration)
764 {
765 vel *= 1000.0f;
766 }
767 else if (ForceMode == TriggerForceMode.Impulse)
768 {
769 vel /= (float)Timing.Step;
770 }
771 return vel.ClampLength(ConvertUnits.ToDisplayUnits(ForceVelocityLimit)) * currentForceFluctuation;
772 }
