470 public void UpdateSpriteStates(float deltaTime)
471 {
472 if (Prefab.DecorativeSpriteGroups.Count == 0) { return; }
473 DecorativeSprite.UpdateSpriteStates(Prefab.DecorativeSpriteGroups, spriteAnimState, ID, deltaTime, ConditionalMatches);
474 foreach (int spriteGroup in Prefab.DecorativeSpriteGroups.Keys)
475 {
476 for (int i = 0; i < Prefab.DecorativeSpriteGroups[spriteGroup].Length; i++)
477 {
478 var decorativeSprite = Prefab.DecorativeSpriteGroups[spriteGroup][i];
479 if (decorativeSprite == null) { continue; }
480 if (spriteGroup > 0)
481 {
482 int activeSpriteIndex = ID % Prefab.DecorativeSpriteGroups[spriteGroup].Length;
483 if (i != activeSpriteIndex)
484 {
485 spriteAnimState[decorativeSprite].IsActive = false;
486 continue;
487 }
488 }
489
490 //check if the sprite is active (whether it should be drawn or not)
491 var spriteState = spriteAnimState[decorativeSprite];
492 spriteState.IsActive = true;
493 foreach (PropertyConditional conditional in decorativeSprite.IsActiveConditionals)
494 {
495 if (!ConditionalMatches(conditional))
496 {
497 spriteState.IsActive = false;
498 break;
499 }
500 }
501 if (!spriteState.IsActive) { continue; }
502
503 //check if the sprite should be animated
504 bool animate = true;
505 foreach (PropertyConditional conditional in decorativeSprite.AnimationConditionals)
506 {
507 if (!ConditionalMatches(conditional)) { animate = false; break; }
508 }
509 if (!animate) { continue; }
510 spriteState.OffsetState += deltaTime;
511 spriteState.RotationState += deltaTime;
512 }
513 }
514 }

516 private bool ConditionalMatches(PropertyConditional conditional)
517 {
518 if (!string.IsNullOrEmpty(conditional.TargetItemComponentName))
519 {
520 return false;
521 }
522 else
523 {
524 if (!conditional.Matches(this)) { return false; }
525 }
526 return true;
