61 protected override void Act(float deltaTime)
62 {
63 if (character.LockHands || targetCharacter == null || targetCharacter.Removed || targetCharacter.IsDead)
64 {
65 Abandon = true;
66 return;
67 }
68 var otherRescuer = targetCharacter.SelectedBy;
69 if (otherRescuer != null && otherRescuer != character)
70 {
71 // Someone else is rescuing/holding the target.
72 Abandon = otherRescuer.IsPlayer || character.GetSkillLevel("medical") < otherRescuer.GetSkillLevel("medical");
73 return;
74 }
75 if (targetCharacter != character)
76 {
77 if (targetCharacter.IsIncapacitated)
78 {
79 // Check if the character needs more oxygen
80 if (!ignoreOxygen && character.SelectedCharacter == targetCharacter || character.CanInteractWith(targetCharacter))
81 {
82 // Replace empty oxygen and welding fuel.
83 if (HumanAIController.HasItem(targetCharacter, AIObjectiveFindDivingGear.HEAVY_DIVING_GEAR, out IEnumerable<Item> suits, requireEquipped: true))
84 {
85 Item suit = suits.FirstOrDefault();
86 if (suit != null)
87 {
88 AIController.UnequipEmptyItems(character, suit);
89 AIController.UnequipContainedItems(character, suit, it => it.HasTag("weldingfuel"));
90 }
91 }
92 else if (HumanAIController.HasItem(targetCharacter, AIObjectiveFindDivingGear.LIGHT_DIVING_GEAR, out IEnumerable<Item> masks, requireEquipped: true))
93 {
94 Item mask = masks.FirstOrDefault();
95 if (mask != null)
96 {
97 AIController.UnequipEmptyItems(character, mask);
98 AIController.UnequipContainedItems(character, mask, it => it.HasTag("weldingfuel"));
99 }
100 }
101 bool ShouldRemoveDivingSuit() => targetCharacter.OxygenAvailable < CharacterHealth.InsufficientOxygenThreshold && targetCharacter.CurrentHull?.LethalPressure <= 0;
102 if (ShouldRemoveDivingSuit())
103 {
104 suits.ForEach(suit => suit.Drop(character));
105 }
106 else if (suits.Any() && suits.None(s => s.OwnInventory?.AllItems != null && s.OwnInventory.AllItems.Any(it => it.HasTag(AIObjectiveFindDivingGear.OXYGEN_SOURCE) && it.ConditionPercentage > 0)))
107 {
108 // The target has a suit equipped with an empty oxygen tank.
109 // Can't remove the suit, because the target needs it.
110 // If we happen to have an extra oxygen tank in the inventory, let's swap it.
111 Item spareOxygenTank = FindOxygenTank(targetCharacter) ?? FindOxygenTank(character);
112 if (spareOxygenTank != null)
113 {
114 Item suit = suits.FirstOrDefault();
115 if (suit != null)
116 {
117 // Insert the new oxygen tank
118 TryAddSubObjective(ref replaceOxygenObjective, () => new AIObjectiveContainItem(character, spareOxygenTank, suit.GetComponent<ItemContainer>(), objectiveManager),
119 onCompleted: () => RemoveSubObjective(ref replaceOxygenObjective),
120 onAbandon: () =>
121 {
122 RemoveSubObjective(ref replaceOxygenObjective);
123 ignoreOxygen = true;
124 if (ShouldRemoveDivingSuit())
125 {
126 suits.ForEach(suit => suit.Drop(character));
127 }
128 });
129 return;
130 }

133 Item FindOxygenTank(Character c) =>
134 c.Inventory.FindItem(i =>
135 i.HasTag(AIObjectiveFindDivingGear.OXYGEN_SOURCE) &&
136 i.ConditionPercentage > 1 &&
137 i.FindParentInventory(inv => inv.Owner is Item otherItem && otherItem.HasTag("diving")) == null,
138 recursive: true);
139 }
140 }
141 if (character.Submarine != null)
142 {
143 if (HumanAIController.GetHullSafety(targetCharacter.CurrentHull, targetCharacter) < HumanAIController.HULL_SAFETY_THRESHOLD)
144 {
145 // Incapacitated target is not in a safe place -> Move to a safe place first
146 if (character.SelectedCharacter != targetCharacter)
147 {
148 if (targetCharacter.CurrentHull != null && HumanAIController.VisibleHulls.Contains(targetCharacter.CurrentHull) && targetCharacter.CurrentHull.DisplayName != null)

150 character.Speak(TextManager.GetWithVariables("DialogFoundUnconsciousTarget",

153 null, 1.0f, $"foundunconscioustarget{targetCharacter.Name}".ToIdentifier(), 60.0f);
154 }
155 // Go to the target and select it
156 if (!character.CanInteractWith(targetCharacter))
157 {
158 RemoveSubObjective(ref replaceOxygenObjective);
159 RemoveSubObjective(ref goToObjective);
160 TryAddSubObjective(ref goToObjective, () => new AIObjectiveGoTo(targetCharacter, character, objectiveManager)
161 {
162 CloseEnough = CloseEnoughToTreat,
163 DialogueIdentifier = "dialogcannotreachpatient".ToIdentifier(),
164 TargetName = targetCharacter.DisplayName

166 onCompleted: () => RemoveSubObjective(ref goToObjective),
167 onAbandon: () =>
168 {
169 RemoveSubObjective(ref goToObjective);
170 Abandon = true;
171 });
172 }
173 else
174 {
175 character.SelectCharacter(targetCharacter);
176 }
177 }
178 else
179 {
180 // Drag the character into safety
181 if (safeHull == null)
182 {
183 if (findHullTimer > 0)
184 {
185 findHullTimer -= deltaTime;

188 {

194 {
195 RemoveSubObjective(ref replaceOxygenObjective);
196 RemoveSubObjective(ref goToObjective);

211 if (subObjectives.Any()) { return; }
212
213 if (targetCharacter != character && !character.CanInteractWith(targetCharacter))
214 {
215 RemoveSubObjective(ref replaceOxygenObjective);
216 RemoveSubObjective(ref goToObjective);

230 }

235 {
236 if (targetCharacter.CurrentHull?.DisplayName != null)

251 private void GiveTreatment(float deltaTime)
252 {
253 if (targetCharacter == null)
254 {
255 string errorMsg = $"{character.Name}: Attempted to update a Rescue objective with no target!";
256 DebugConsole.ThrowError(errorMsg);
257 Abandon = true;
258 return;
259 }
260
261 SteeringManager.Reset();
262
263 if (!targetCharacter.IsPlayer)
264 {
265 // If the target is a bot, don't let it move
266 targetCharacter.AIController?.SteeringManager?.Reset();
267 }
268 if (treatmentTimer > 0.0f)
269 {
270 treatmentTimer -= deltaTime;
271 return;
272 }
273 treatmentTimer = TreatmentDelay;
274
275 float cprSuitability = targetCharacter.Oxygen < 0.0f ? -targetCharacter.Oxygen * 100.0f : 0.0f;
276
277 //find which treatments are the most suitable to treat the character's current condition
278 targetCharacter.CharacterHealth.GetSuitableTreatments(currentTreatmentSuitabilities, normalize: false, predictFutureDuration: 10.0f);
279
280 //check if we already have a suitable treatment for any of the afflictions
281 foreach (Affliction affliction in GetSortedAfflictions(targetCharacter))
282 {
283 if (affliction == null) { throw new Exception("Affliction was null"); }
284 if (affliction.Prefab == null) { throw new Exception("Affliction prefab was null"); }
285 float bestSuitability = 0.0f;
286 Item bestItem = null;
287 foreach (KeyValuePair<Identifier, float> treatmentSuitability in affliction.Prefab.TreatmentSuitability)
288 {
289 if (currentTreatmentSuitabilities.ContainsKey(treatmentSuitability.Key) &&
290 currentTreatmentSuitabilities[treatmentSuitability.Key] > bestSuitability)
291 {
292 Item matchingItem = character.Inventory.FindItemByIdentifier(treatmentSuitability.Key, true);
293 if (matchingItem != null)
294 {
295 bestItem = matchingItem;
296 bestSuitability = currentTreatmentSuitabilities[treatmentSuitability.Key];
297 }
298 }
299 }
300 if (bestItem != null)
301 {
302 if (targetCharacter != character) { character.SelectCharacter(targetCharacter); }
303 ApplyTreatment(affliction, bestItem);
304 //wait a bit longer after applying a treatment to wait for potential side-effects to manifest
305 treatmentTimer = TreatmentDelay * 4;
306 return;
307 }
308 }
309 // Find treatments outside of own inventory only if inside the own sub.
310 if (character.Submarine != null && character.Submarine.TeamID == character.TeamID)
311 {
312 //didn't have any suitable treatments available, try to find some medical items
313 if (currentTreatmentSuitabilities.Any(s => s.Value > cprSuitability))
314 {
315 itemNameList.Clear();
316 suitableItemIdentifiers.Clear();
317 foreach (KeyValuePair<Identifier, float> treatmentSuitability in currentTreatmentSuitabilities)
318 {
319 if (treatmentSuitability.Value <= cprSuitability) { continue; }
320 if (MapEntityPrefab.Find(null, treatmentSuitability.Key, showErrorMessages: false) is ItemPrefab itemPrefab)
321 {
322 if (!Item.ItemList.Any(it => ((MapEntity)it).Prefab.Identifier == treatmentSuitability.Key)) { continue; }
323 suitableItemIdentifiers.Add(treatmentSuitability.Key);
324 //only list the first 4 items
325 if (itemNameList.Count < 4)
326 {
327 itemNameList.Add(itemPrefab.Name);
328 }
329 }
330 }
331 if (itemNameList.Any())
332 {
333 LocalizedString itemListStr = "";
334 if (itemNameList.Count == 1)
335 {
336 itemListStr = itemNameList[0];
337 }
338 else if (itemNameList.Count == 2)
339 {
340 //[treatment1] or [treatment2]
341 itemListStr = TextManager.GetWithVariables(
342 "DialogRequiredTreatmentOptionsLast",
343 ("[treatment1]", itemNameList[0]),
344 ("[treatment2]", itemNameList[1]));
345 }
346 else
347 {
348 //[treatment1], [treatment2], [treatment3] ... or [treatmentx]
349 itemListStr = TextManager.GetWithVariables(
350 "DialogRequiredTreatmentOptionsFirst",
351 ("[treatment1]", itemNameList[0]),
352 ("[treatment2]", itemNameList[1]));
353 for (int i = 2; i < itemNameList.Count - 1; i++)
354 {
355 itemListStr = TextManager.GetWithVariables(
356 "DialogRequiredTreatmentOptionsFirst",
357 ("[treatment1]", itemListStr),
358 ("[treatment2]", itemNameList[i]));
359 }
360 itemListStr = TextManager.GetWithVariables(
361 "DialogRequiredTreatmentOptionsLast",
362 ("[treatment1]", itemListStr),
363 ("[treatment2]", itemNameList.Last()));
364 }
365 if (targetCharacter != character && character.IsOnPlayerTeam)
366 {
367 character.Speak(TextManager.GetWithVariables("DialogListRequiredTreatments",
368 ("[targetname]", targetCharacter.Name, FormatCapitals.No),
369 ("[treatmentlist]", itemListStr, FormatCapitals.Yes)).Value,
370 null, 2.0f, $"listrequiredtreatments{targetCharacter.Name}".ToIdentifier(), 60.0f);
371 }
372 RemoveSubObjective(ref getItemObjective);
373 TryAddSubObjective(ref getItemObjective,
374 constructor: () => new AIObjectiveGetItem(character, suitableItemIdentifiers.ToArray(), objectiveManager, equip: true, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC),
375 onCompleted: () => RemoveSubObjective(ref getItemObjective),
376 onAbandon: () =>
377 {
378 Abandon = true;
379 if (character != targetCharacter && character.IsOnPlayerTeam)
380 {
381 character.Speak(TextManager.GetWithVariable("dialogcannottreatpatient", "[name]", targetCharacter.DisplayName, FormatCapitals.No).Value, identifier: "cannottreatpatient".ToIdentifier(), minDurationBetweenSimilar: 20.0f);
382 }
383 });
384 }
385 else if (cprSuitability <= 0)
386 {
387 character.Speak(TextManager.GetWithVariable("dialogcannottreatpatient", "[name]", targetCharacter.DisplayName, formatCapitals: FormatCapitals.No).Value, identifier: "cannottreatpatient".ToIdentifier(), minDurationBetweenSimilar: 20.0f);
388 Abandon = true;
389 }
390 }
391 }
392 else if (!targetCharacter.IsUnconscious)
393 {
394 //no suitable treatments found, not inside our own sub (= can't search for more treatments), the target isn't unconscious (= can't give CPR)
395 character.Speak(TextManager.GetWithVariable("dialogcannottreatpatient", "[name]", targetCharacter.DisplayName, formatCapitals: FormatCapitals.No).Value, identifier: "cannottreatpatient".ToIdentifier(), minDurationBetweenSimilar: 20.0f);
396 Abandon = true;
397 return;
398 }
399 if (character != targetCharacter)
400 {
401 if (cprSuitability > 0.0f)
402 {
403 character.SelectCharacter(targetCharacter);
404 character.AnimController.Anim = AnimController.Animation.CPR;
405 performedCpr = true;
406 }
407 else
408 {
409 character.DeselectCharacter();
410 }
411 }

414 private void ApplyTreatment(Affliction affliction, Item item)
415 {
416 var targetLimb = targetCharacter.CharacterHealth.GetAfflictionLimb(affliction);
417 bool remove = false;
418 foreach (ItemComponent ic in item.Components)
419 {
420 if (!ic.HasRequiredContainedItems(user: character, addMessage: false)) { continue; }
421#if CLIENT
422 ic.PlaySound(ActionType.OnUse, character);
423#endif
424 ic.WasUsed = true;
425 ic.ApplyStatusEffects(ActionType.OnUse, 1.0f, targetCharacter, targetLimb, user: character);
426 if (ic.DeleteOnUse)
427 {
428 remove = true;
429 }
430 }
431 if (remove)
432 {
433 Entity.Spawner?.AddItemToRemoveQueue(item);
434 }

449 protected override float GetPriority()
450 {
451 if (!IsAllowed)
452 {
453 Priority = 0;
454 Abandon = true;
455 return Priority;
456 }
457 if (character.CurrentHull == null)
458 {
459 if (!objectiveManager.HasOrder<AIObjectiveRescueAll>())
460 {
461 Priority = 0;
462 Abandon = true;
463 return Priority;
464 }
465 }
466 else if (Character.CharacterList.Any(c => c.CurrentHull == targetCharacter.CurrentHull && !HumanAIController.IsFriendly(character, c) && HumanAIController.IsActive(c)))
467 {
468 // Don't go into rooms that have enemies
469 Priority = 0;
470 Abandon = true;
471 return Priority;
472 }
473 if (targetCharacter == null)
474 {
475 Priority = 0;
476 Abandon = true;
477 }
478 else
479 {
480 float horizontalDistance = Math.Abs(character.WorldPosition.X - targetCharacter.WorldPosition.X);
481 float verticalDistance = Math.Abs(character.WorldPosition.Y - targetCharacter.WorldPosition.Y);
482 if (character.Submarine?.Info is { IsRuin: false })
483 {
484 verticalDistance *= 2;
485 }
486 float distanceFactor = MathHelper.Lerp(1, 0.1f, MathUtils.InverseLerp(0, 5000, horizontalDistance + verticalDistance));
487 if (targetCharacter.CurrentHull == character.CurrentHull)
488 {
489 distanceFactor = 1;
490 }
491 float vitalityFactor = 1 - AIObjectiveRescueAll.GetVitalityFactor(targetCharacter) / 100;
492 float devotion = CumulatedDevotion / 100;
493 Priority = MathHelper.Lerp(0, 100, MathHelper.Clamp(devotion + (vitalityFactor * distanceFactor * PriorityModifier), 0, 1));
494 }
495 return Priority;
496 }
