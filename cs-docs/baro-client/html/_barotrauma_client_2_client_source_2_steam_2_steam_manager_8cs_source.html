103 public static async Task InitRelayNetworkAccess()
104 {
105 if (!IsInitialized) { return; }
106
107 await Task.Yield();
108 Steamworks.SteamNetworkingUtils.InitRelayNetworkAccess();
109
110 //SetSteamworksNetworkingDebugLog(true);
111 var status = Steamworks.SteamNetworkingUtils.Status;
112 while (status.Avail != Steamworks.SteamNetworkingAvailability.Current)
113 {
114 if (status.Avail == Steamworks.SteamNetworkingAvailability.CannotTry ||
115 status.Avail == Steamworks.SteamNetworkingAvailability.Previously ||
116 status.Avail == Steamworks.SteamNetworkingAvailability.Failed)
117 {
118 DebugConsole.ThrowError($"Failed to initialize Steamworks network relay: " +
119 $"{Steamworks.SteamNetworkingUtils.Status.Avail}, " +
120 $"{Steamworks.SteamNetworkingUtils.Status.AvailNetConfig}, " +
121 $"{Steamworks.SteamNetworkingUtils.Status.Avail}, " +
122 $"{Steamworks.SteamNetworkingUtils.Status.Msg}");
123 break;
124 }
125 await Task.Delay(25);
126 status = Steamworks.SteamNetworkingUtils.Status;
127 }
128 //SetSteamworksNetworkingDebugLog(false);
