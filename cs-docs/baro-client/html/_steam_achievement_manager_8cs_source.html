54 public static void Update(float deltaTime)
55 {
56 if (GameMain.GameSession == null) { return; }
57#if CLIENT
58 if (GameMain.Client != null) { return; }
59#endif
60
61 updateTimer -= deltaTime;
62 if (updateTimer > 0.0f) { return; }
63 updateTimer = UpdateInterval;
64
65 if (Level.Loaded != null && roundData != null && Screen.Selected == GameMain.GameScreen)
66 {
67 if (GameMain.GameSession.EventManager.CurrentIntensity > 0.99f)
68 {
69 UnlockAchievement("maxintensity".ToIdentifier(), true, c => c != null && !c.IsDead && !c.IsUnconscious);
70 }
71
72 foreach (Character c in Character.CharacterList)
73 {
74 if (c.IsDead) { continue; }
75 //achievement for descending below crush depth and coming back
76 if (Timing.TotalTime > GameMain.GameSession.RoundStartTime + 30.0f)
77 {
78 if (c.Submarine != null && c.Submarine.AtDamageDepth || Level.Loaded.GetRealWorldDepth(c.WorldPosition.Y) > Level.Loaded.RealWorldCrushDepth)
79 {
80 roundData.EnteredCrushDepth.Add(c);
81 }
82 else if (Level.Loaded.GetRealWorldDepth(c.WorldPosition.Y) < Level.Loaded.RealWorldCrushDepth - 500.0f)
83 {
84 //all characters that have entered crush depth and are still alive get an achievement
85 if (roundData.EnteredCrushDepth.Contains(c)) { UnlockAchievement(c, "survivecrushdepth".ToIdentifier()); }
86 }
87 }
88 }
89
90 foreach (Submarine sub in Submarine.Loaded)
91 {
92 foreach (Reactor reactor in roundData.Reactors)
93 {
94 if (reactor.Item.Condition <= 0.0f && reactor.Item.Submarine == sub)
95 {
96 //characters that were inside the sub during a reactor meltdown
97 //get an achievement if they're still alive at the end of the round
98 foreach (Character c in Character.CharacterList)
99 {
100 if (!c.IsDead && c.Submarine == sub) roundData.ReactorMeltdown.Add(c);
101 }
102 }
103 }
104
105 //convert submarine velocity to km/h
106 Vector2 submarineVel = Physics.DisplayToRealWorldRatio * ConvertUnits.ToDisplayUnits(sub.Velocity) * 3.6f;
107 //achievement for going > 100 km/h
108 if (Math.Abs(submarineVel.X) > 100.0f)
109 {
110 //all conscious characters inside the sub get an achievement
111 UnlockAchievement("subhighvelocity".ToIdentifier(), true, c => c != null && c.Submarine == sub && !c.IsDead && !c.IsUnconscious);
112 }
113
114 //achievement for descending ridiculously deep
115 float realWorldDepth = sub.RealWorldDepth;
116 if (realWorldDepth > 5000.0f && Timing.TotalTime > GameMain.GameSession.RoundStartTime + 30.0f)
117 {
118 //all conscious characters inside the sub get an achievement
119 UnlockAchievement("subdeep".ToIdentifier(), true, c => c != null && c.Submarine == sub && !c.IsDead && !c.IsUnconscious);
120 }
121 }
122
123 if (!roundData.SubWasDamaged)
124 {
125 roundData.SubWasDamaged = SubWallsDamaged(Submarine.MainSub);
126 }
127 }
128
129 if (GameMain.GameSession != null)
130 {
131#if CLIENT
132 if (Character.Controlled != null && !(GameMain.GameSession.GameMode is TestGameMode))
133 {
134 CheckMidRoundAchievements(Character.Controlled);
135 }
136#else
137 foreach (Client client in GameMain.Server.ConnectedClients)
138 {
139 if (client.Character != null)
140 {
141 CheckMidRoundAchievements(client.Character);
142 }
143 }
144#endif
145 }

148 private static void CheckMidRoundAchievements(Character c)
149 {
150 if (c == null || c.Removed) { return; }
151
152 if (c.HasEquippedItem("clownmask".ToIdentifier()) &&
153 c.HasEquippedItem("clowncostume".ToIdentifier()))
154 {
155 UnlockAchievement(c, "clowncostume".ToIdentifier());
156 }
157
158 if (Submarine.MainSub != null && c.Submarine == null && c.SpeciesName == CharacterPrefab.HumanSpeciesName)
159 {
160 float requiredDist = 500 / Physics.DisplayToRealWorldRatio;
161 float distSquared = Vector2.DistanceSquared(c.WorldPosition, Submarine.MainSub.WorldPosition);
162 if (cachedDistances.TryGetValue(c, out var cachedDistance))
163 {
164 if (cachedDistance.ShouldUpdateDistance(c.WorldPosition, Submarine.MainSub.WorldPosition))
165 {
166 cachedDistances.Remove(c);
167 cachedDistance = CalculateNewCachedDistance(c);
168 if (cachedDistance != null)
169 {
170 cachedDistances.Add(c, cachedDistance);
171 }
172 }
173 }
174 else
175 {
176 cachedDistance = CalculateNewCachedDistance(c);
177 if (cachedDistance != null)
178 {
179 cachedDistances.Add(c, cachedDistance);
180 }
181 }
182 if (cachedDistance != null)
183 {
184 distSquared = Math.Max(distSquared, cachedDistance.Distance * cachedDistance.Distance);
185 }
186 if (distSquared > requiredDist * requiredDist)
187 {
188 UnlockAchievement(c, "crewaway".ToIdentifier());
189 }

191 static CachedDistance CalculateNewCachedDistance(Character c)
192 {
193 pathFinder ??= new PathFinder(WayPoint.WayPointList, false);
194 var path = pathFinder.FindPath(ConvertUnits.ToSimUnits(c.WorldPosition), ConvertUnits.ToSimUnits(Submarine.MainSub.WorldPosition));
195 if (path.Unreachable) { return null; }
196 return new CachedDistance(c.WorldPosition, Submarine.MainSub.WorldPosition, path.TotalLength, Timing.TotalTime + Rand.Range(1.0f, 5.0f));
197 }
198 }

201 private static bool SubWallsDamaged(Submarine sub)
202 {
203 foreach (Structure structure in Structure.WallList)
204 {
205 if (structure.Submarine != sub || structure.HasBody) { continue; }
206 for (int i = 0; i < structure.SectionCount; i++)
207 {
208 if (structure.SectionIsLeaking(i))
209 {
210 return true;
211 }
212 }
213 }
214 return false;

343 public static void OnRoundEnded(GameSession gameSession)
344 {
345 if (CheatsEnabled) { return; }
346
347 //made it to the destination
348 if (gameSession?.Submarine != null && Level.Loaded != null && gameSession.Submarine.AtEndExit)
349 {
350 float levelLengthMeters = Physics.DisplayToRealWorldRatio * Level.Loaded.Size.X;
351 float levelLengthKilometers = levelLengthMeters / 1000.0f;
352 //in multiplayer the client's/host's character must be inside the sub (or end outpost) and alive
353 if (GameMain.NetworkMember != null)
354 {
355#if CLIENT
356 Character myCharacter = Character.Controlled;
357 if (myCharacter != null &&
358 !myCharacter.IsDead &&
359 (myCharacter.Submarine == gameSession.Submarine || (Level.Loaded?.EndOutpost != null && myCharacter.Submarine == Level.Loaded.EndOutpost)))
360 {
361 IncrementStat("kmstraveled".ToIdentifier(), levelLengthKilometers);
362 }
363#endif
364 }
365 else
366 {
367 //in sp making it to the end is enough
368 IncrementStat("kmstraveled".ToIdentifier(), levelLengthKilometers);
369 }
370 }
371
372 //make sure changed stats (kill count, kms traveled) get stored
373 SteamManager.StoreStats();
374
375 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
376
377 foreach (Mission mission in gameSession.Missions)
378 {
379 if (mission is CombatMission combatMission && GameMain.GameSession.WinningTeam.HasValue)
380 {
381 //all characters that are alive and in the winning team get an achievement
382 var achvIdentifier =
383 $"{mission.Prefab.AchievementIdentifier}{(int) GameMain.GameSession.WinningTeam}"
384 .ToIdentifier();
385 UnlockAchievement(achvIdentifier, true,
386 c => c != null && !c.IsDead && !c.IsUnconscious && combatMission.IsInWinningTeam(c));
387 }
388 else if (mission.Completed)
389 {
390 //all characters get an achievement
391 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsServer)
392 {
393 UnlockAchievement(mission.Prefab.AchievementIdentifier, true, c => c != null);
394 }
395 else
396 {
397 UnlockAchievement(mission.Prefab.AchievementIdentifier);
398 }
399 }
400 }
401
402 //made it to the destination
403 if (gameSession.Submarine.AtEndExit)
404 {
405 bool noDamageRun = !roundData.SubWasDamaged && !gameSession.Casualties.Any();
406
407#if SERVER
408 if (GameMain.Server != null)
409 {
410 //in MP all characters that were inside the sub during reactor meltdown and still alive at the end of the round get an achievement
411 UnlockAchievement("survivereactormeltdown".ToIdentifier(), true, c => c != null && !c.IsDead && roundData.ReactorMeltdown.Contains(c));
412 if (noDamageRun)
413 {
414 UnlockAchievement("nodamagerun".ToIdentifier(), true, c => c != null && !c.IsDead);
415 }
416 }
417#endif
418#if CLIENT
419 if (noDamageRun) { UnlockAchievement("nodamagerun".ToIdentifier()); }
420 if (roundData.ReactorMeltdown.Any()) //in SP getting to the destination after a meltdown is enough
421 {
422 UnlockAchievement("survivereactormeltdown".ToIdentifier());
423 }
424#endif
425 var charactersInSub = Character.CharacterList.FindAll(c =>
426 !c.IsDead &&
427 c.TeamID != CharacterTeamType.FriendlyNPC &&
428 !(c.AIController is EnemyAIController) &&
429 (c.Submarine == gameSession.Submarine || gameSession.Submarine.GetConnectedSubs().Contains(c.Submarine) || (Level.Loaded?.EndOutpost != null && c.Submarine == Level.Loaded.EndOutpost)));
430
431 if (charactersInSub.Count == 1)
432 {
433 //there must be some casualties to get the last mant standing achievement
434 if (gameSession.Casualties.Any())
435 {
436 UnlockAchievement(charactersInSub[0], "lastmanstanding".ToIdentifier());
437 }
438#if CLIENT
439 else if (GameMain.GameSession.CrewManager.GetCharacters().Count() == 1)
440 {
441 UnlockAchievement(charactersInSub[0], "lonesailor".ToIdentifier());
442 }
443#else
444 //lone sailor achievement if alone in the sub and there are no other characters with the same team ID
445 else if (!Character.CharacterList.Any(c =>
446 c != charactersInSub[0] &&
447 c.TeamID == charactersInSub[0].TeamID &&
448 !(c.AIController is EnemyAIController)))
449 {
450 UnlockAchievement(charactersInSub[0], "lonesailor".ToIdentifier());
451 }
452#endif
453
454 }
455 foreach (Character character in charactersInSub)
456 {
457 if (roundData.EnteredCrushDepth.Contains(character))
458 {
459 UnlockAchievement(character, "survivecrushdepth".ToIdentifier());
460 }
461 if (character.Info.Job == null) { continue; }
462 UnlockAchievement(character, $"{character.Info.Job.Prefab.Identifier}round".ToIdentifier());
463 }
464 }

507 public static void UnlockAchievement(Identifier identifier, bool unlockClients = false, Func<Character, bool> conditions = null)
508 {
509 if (CheatsEnabled) { return; }
510
511#if SERVER
512 if (unlockClients && GameMain.Server != null)
513 {
514 foreach (Client c in GameMain.Server.ConnectedClients)
515 {
516 if (conditions != null && !conditions(c.Character)) { continue; }
517 GameMain.Server.GiveAchievement(c, identifier);
518 }
519 }
520#endif
521 //already unlocked, no need to do anything
522 if (unlockedAchievements.Contains(identifier)) { return; }
523 unlockedAchievements.Add(identifier);
524
525#if CLIENT
526 if (conditions != null && !conditions(Character.Controlled)) { return; }
527#endif
528
529 SteamManager.UnlockAchievement(identifier);
