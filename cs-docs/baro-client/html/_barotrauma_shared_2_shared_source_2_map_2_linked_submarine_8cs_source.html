167  private void GenerateWallVertices(XElement rootElement)
168  {
169  List<Vector2> points = new List<Vector2>();
170 
171  foreach (XElement element in rootElement.Elements())
172  {
173  if (element.Name != "Structure") { continue; }
174 
175  string name = element.GetAttributeString("name", "");
176  Identifier identifier = element.GetAttributeIdentifier("identifier", "");
177 
178  StructurePrefab prefab = Structure.FindPrefab(name, identifier);
179  if (prefab == null) { continue; }
180 
181  float scale = element.GetAttributeFloat("scale", prefab.Scale);
182 
183  var rect = element.GetAttributeVector4("rect", Vector4.Zero);
184  rect.Z *= scale / prefab.Scale;
185  rect.W *= scale / prefab.Scale;
186 
187  points.Add(new Vector2(rect.X, rect.Y));
188  points.Add(new Vector2(rect.X + rect.Z, rect.Y));
189  points.Add(new Vector2(rect.X, rect.Y - rect.W));
190  points.Add(new Vector2(rect.X + rect.Z, rect.Y - rect.W));
191  }
192 
193  wallVertices = MathUtils.GiftWrap(points);
194  }

212  linkedSub = new LinkedSubmarine(submarine, id)
213  {
214  purchasedLostShuttles = GameMain.GameSession?.GameMode is CampaignMode campaign && campaign.PurchasedLostShuttles,
215  saveElement = element
216  };
217 
218  if (!string.IsNullOrWhiteSpace(levelSeed) && levelData != null &&
219  levelData.Seed != levelSeed && !linkedSub.purchasedLostShuttles)
220  {
221  linkedSub.loadSub = false;
222  }
223  else
224  {
225  linkedSub.loadSub = true;
226  linkedSub.rect.Location = MathUtils.ToPoint(pos);
227  }
228  }
229 
230  #warning TODO: revise
231  linkedSub.filePath = element.GetAttributeContentPath("filepath")?.Value ?? string.Empty;
232  int[] linkedToIds = element.GetAttributeIntArray("linkedto", Array.Empty<int>());
233  for (int i = 0; i < linkedToIds.Length; i++)

235  linkedSub.linkedToID.Add(idRemap.GetOffsetId(linkedToIds[i]));

238  linkedSub.originalMyPortID = (ushort)element.GetAttributeInt("originalmyport", 0);

256  public override void OnMapLoaded()
257  {
258  if (!loadSub) { return; }
259 
260  SubmarineInfo info = new SubmarineInfo(Submarine.Info.FilePath, "", saveElement);
261  if (!info.SubmarineElement.HasElements)
262  {
263  DebugConsole.ThrowError("Failed to load a linked submarine (empty XML element). The save file may be corrupted.");
264  return;
265  }
266  if (!info.SubmarineElement.Elements().Any(e => e.Name.ToString().Equals("hull", StringComparison.OrdinalIgnoreCase)))
267  {
268  DebugConsole.ThrowError("Failed to load a linked submarine (the submarine contains no hulls).");
269  return;
270  }
271 
272  IdRemap parentRemap = new IdRemap(Submarine.Info.SubmarineElement, Submarine.IdOffset);
273  sub = Submarine.Load(info, false, parentRemap);
274  sub.Info.SubmarineClass = Submarine.Info.SubmarineClass;
275 
276  IdRemap childRemap = new IdRemap(saveElement, sub.IdOffset);
277 
278  Vector2 worldPos = saveElement.GetAttributeVector2("worldpos", Vector2.Zero);
279  if (worldPos != Vector2.Zero)
280  {
281  if (GameMain.GameSession != null && GameMain.GameSession.MirrorLevel)
282  {
283  worldPos.X = GameMain.GameSession.LevelData.Size.X - worldPos.X;
284  }
285  sub.SetPosition(worldPos);
286  }
287  else
288  {
289  sub.SetPosition(WorldPosition);
290  }
291 
292  DockingPort linkedPort = null;
293  DockingPort myPort = null;
294 
295  MapEntity linkedItem = linkedTo.FirstOrDefault(lt => (lt as Item)?.GetComponent<DockingPort>() != null);
296  if (linkedItem == null)
297  {
298  linkedPort = DockingPort.List.FirstOrDefault(dp => dp.DockingTarget != null && dp.DockingTarget.Item.Submarine == sub);
299  }
300  else
301  {
302  linkedPort = ((Item)linkedItem).GetComponent<DockingPort>();
303  }
304 
305  if (linkedPort == null)
306  {
307  if (purchasedLostShuttles)
308  {
309  linkedPort = (FindEntityByID(originalLinkedToID) as Item)?.GetComponent<DockingPort>();
310  }
311  if (linkedPort == null) { return; }
312  }
313  originalLinkedPort = linkedPort;
314 
315  ushort originalMyId = childRemap.GetOffsetId(originalMyPortID);
316  myPort = (FindEntityByID(originalMyId) as Item)?.GetComponent<DockingPort>();
317  if (myPort == null)
318  {
319  float closestDistance = 0.0f;
320  foreach (DockingPort port in DockingPort.List)
321  {
322  if (port.Item.Submarine != sub || port.IsHorizontal != linkedPort.IsHorizontal) { continue; }
323  float dist = Vector2.Distance(port.Item.WorldPosition, linkedPort.Item.WorldPosition);
324  if (myPort == null || dist < closestDistance)
325  {
326  myPort = port;
327  closestDistance = dist;
328  }
329  }
330  }
331 
332  if (myPort != null)
333  {
334  originalMyPortID = myPort.Item.ID;
335 
336  myPort.Undock(applyEffects: false);
337  myPort.DockingDir = 0;
338 
339  //something else is already docked to the port this sub should be docked to
340  //may happen if a shuttle is lost, another vehicle docked to where the shuttle used to be,
341  //and the shuttle is then restored in the campaign mode
342  //or if the user connects multiple subs to the same docking ports in the sub editor
343  if (linkedPort.Docked && linkedPort.DockingTarget != null && linkedPort.DockingTarget != myPort)
344  {
345  //just spawn below the main sub
346  sub.SetPosition(
347  linkedPort.Item.Submarine.WorldPosition -
348  new Vector2(0, linkedPort.Item.Submarine.GetDockedBorders().Height / 2 + sub.GetDockedBorders().Height / 2));
349  }
350  else
351  {
352  Vector2 portDiff = myPort.Item.WorldPosition - sub.WorldPosition;
353  Vector2 offset = myPort.IsHorizontal ?
354  Vector2.UnitX * myPort.GetDir(linkedPort) :
355  Vector2.UnitY * myPort.GetDir(linkedPort);
356  offset *= myPort.DockedDistance;
357 
358  sub.SetPosition((linkedPort.Item.WorldPosition - portDiff) - offset);
359 
360  myPort.Dock(linkedPort);
361  myPort.Lock(isNetworkMessage: true, applyEffects: false);
362  }
363  }
364 
365  if (GameMain.GameSession?.GameMode is CampaignMode campaign && campaign.PurchasedLostShuttles)
366  {
367  foreach (Structure wall in Structure.WallList)
368  {
369  if (wall.Submarine != sub) { continue; }
370  for (int i = 0; i < wall.SectionCount; i++)
371  {
372  wall.SetDamage(i, 0, createNetworkEvent: false);
373  }
374  }
375  foreach (Hull hull in Hull.HullList)
376  {
377  if (hull.Submarine != sub) { continue; }
378  hull.WaterVolume = 0.0f;
379  hull.OxygenPercentage = 100.0f;
380  hull.BallastFlora?.Kill();
381  }
382  }
383 
384  sub.SetPosition(sub.WorldPosition - Submarine.WorldPosition, forceUndockFromStaticSubmarines: false);
385  sub.Submarine = Submarine;
386  }

410

412  saveElement.Add(new XAttribute("pos", XMLExtensions.Vector2ToString(Position - Submarine.HiddenSubPosition)));
413

415  if (linkedPort != null)

426 
427  saveElement.Attribute("originallinkedto")?.Remove();

429  saveElement.Attribute("originalmyport")?.Remove();
430  saveElement.Add(new XAttribute("originalmyport", originalMyPortID));
431 
432  if (sub != null)
433  {
434  bool leaveBehind = false;
435  if (!sub.DockedTo.Contains(Submarine.MainSub))
436  {
437  System.Diagnostics.Debug.Assert(Submarine.MainSub.AtEndExit || Submarine.MainSub.AtStartExit);
438  if (Submarine.MainSub.AtEndExit)
439  {
440  leaveBehind = sub.AtEndExit != Submarine.MainSub.AtEndExit;
441  }
442  else
443  {
444  leaveBehind = sub.AtStartExit != Submarine.MainSub.AtStartExit;
445  }
446  }
447 
448  if (leaveBehind)
449  {
450  saveElement.SetAttributeValue("location", Level.Loaded.Seed);
451  Vector2 position = sub.SubBody.Position;
452  if (Level.Loaded.Mirrored)
453  {
454  position.X = Level.Loaded.Size.X - position.X;
455  }
456  saveElement.SetAttributeValue("worldpos", XMLExtensions.Vector2ToString(position));
457  }
458  else
459  {
460  if (saveElement.Attribute("location") != null) saveElement.Attribute("location").Remove();
461  if (saveElement.Attribute("worldpos") != null) saveElement.Attribute("worldpos").Remove();
462  }
463  saveElement.SetAttributeValue("pos", XMLExtensions.Vector2ToString(Position - Submarine.HiddenSubPosition));
464  }
465 
466  parentElement.Add(saveElement);
467 
468  return saveElement;
469  }
