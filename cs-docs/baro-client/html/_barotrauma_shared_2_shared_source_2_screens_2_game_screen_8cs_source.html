103#endif

108 {
109 if (GameMain.GameSession.Level != null && GameMain.GameSession.Submarine != null)
110 {
111 Submarine closestSub = Submarine.FindClosest(cam.WorldViewCenter) ?? GameMain.GameSession.Submarine;
112
113 Vector2 targetMovement = Vector2.Zero;
114 if (PlayerInput.KeyDown(Keys.I)) { targetMovement.Y += 1.0f; }
115 if (PlayerInput.KeyDown(Keys.K)) { targetMovement.Y -= 1.0f; }
116 if (PlayerInput.KeyDown(Keys.J)) { targetMovement.X -= 1.0f; }
117 if (PlayerInput.KeyDown(Keys.L)) { targetMovement.X += 1.0f; }
118
119 if (targetMovement != Vector2.Zero)
120 {
121 closestSub.ApplyForce(targetMovement * closestSub.SubBody.Body.Mass * 100.0f);
122 }
123 }
124 }
125#endif
126
127#if CLIENT
128 GameMain.LightManager?.Update((float)deltaTime);
129#endif

131 GameTime += deltaTime;

133 foreach (PhysicsBody body in PhysicsBody.List)

231 Character.Controlled?.UpdateLocalCursor(cam);

310 private void ExecutePhysics()
311 {
312 while (true)
313 {
314 while (physicsTime >= Timing.Step)
315 {
316 lock (updateLock)
317 {
318 GameMain.World.Step((float)Timing.Step);
319 physicsTime -= Timing.Step;
320 }
321 }
322 }
323 }
