166  character.SelectedConstruction = null;

175 
176  if (behavior == BehaviorType.StayInHull)

185  {
186  PathSteering.SteeringSeek(character.GetRelativeSimPosition(currentTarget), weight: 1, nodeFilter: node => node.Waypoint.CurrentHull != null);
187  }
188  }
189  else
190  {
191  bool currentTargetIsInvalid = currentTarget == null || IsForbidden(currentTarget) ||
192  (PathSteering.CurrentPath != null && PathSteering.CurrentPath.Nodes.Any(n => HumanAIController.UnsafeHulls.Contains(n.CurrentHull)));
193 
194  if (currentTarget != null && !currentTargetIsInvalid)
195  {
196  if (character.TeamID == CharacterTeamType.FriendlyNPC && !character.IsEscorted)
197  {
198  if (currentTarget.Submarine.TeamID != character.TeamID)
199  {
200  currentTargetIsInvalid = true;
201  }
202  }
203  else
204  {
205  if (currentTarget.Submarine != character.Submarine)
206  {
207  currentTargetIsInvalid = true;
208  }
209  }
210  }
211 
212  if (currentTargetIsInvalid || currentTarget == null || IsForbidden(character.CurrentHull) && IsSteeringFinished())
213  {
214  if (newTargetTimer > timerMargin)
215  {
216  //don't reset to zero, otherwise the character will keep calling FindTargetHulls
217  //almost constantly when there's a small number of potential hulls to move to
218  SetTargetTimerLow();
219  }
220  }
221  else if (character.IsClimbing)
222  {
223  if (currentTarget == null)
224  {
225  SetTargetTimerLow();
226  }
227  else if (Math.Abs(character.AnimController.TargetMovement.Y) > 0.9f)

239  }

244  //find all available hulls first
245  searchingNewHull = true;
246  FindTargetHulls();
247  }
248  else if (targetHulls.Any())
249  {
250  //choose a random available hull
251  currentTarget = ToolBox.SelectWeightedRandom(targetHulls, hullWeights, Rand.RandSync.Unsynced);
252  bool isInWrongSub = (character.TeamID == CharacterTeamType.FriendlyNPC && !character.IsEscorted) && character.Submarine.TeamID != character.TeamID;
253  bool isCurrentHullAllowed = !isInWrongSub && !IsForbidden(character.CurrentHull);
254  var path = PathSteering.PathFinder.FindPath(character.SimPosition, currentTarget.SimPosition, character.Submarine, nodeFilter: node =>
255  {
256  if (node.Waypoint.CurrentHull == null) { return false; }
257  // Check that there is no unsafe hulls on the way to the target
258  if (node.Waypoint.CurrentHull != character.CurrentHull && HumanAIController.UnsafeHulls.Contains(node.Waypoint.CurrentHull)) { return false; }
259  return true;
260  }, endNodeFilter: node => !isCurrentHullAllowed | !IsForbidden(node.Waypoint.CurrentHull));
261  if (path.Unreachable)
262  {
263  //can't go to this room, remove it from the list and try another room
264  int index = targetHulls.IndexOf(currentTarget);
265  targetHulls.RemoveAt(index);
266  hullWeights.RemoveAt(index);

271  }

273  PathSteering.SetPath(path);
274  SetTargetTimerNormal();
275  searchingNewHull = false;
276  }
277  else

279  // Couldn't find a valid hull
280  SetTargetTimerHigh();
281  searchingNewHull = false;
282  }
283  }
284  newTargetTimer -= deltaTime;
285  if (!character.IsClimbing && (PathSteering == null || PathSteering.CurrentPath == null || IsSteeringFinished()))
286  {
287  Wander(deltaTime);
288  }
289  else if (currentTarget != null)
290  {
291  PathSteering.SteeringSeek(character.GetRelativeSimPosition(currentTarget), weight: 1, nodeFilter: node => node.Waypoint.CurrentHull != null);
292  }
293  }

296  public void Wander(float deltaTime)
297  {
298  if (character.IsClimbing) { return; }
299  var currentHull = character.CurrentHull;
300  if (!character.AnimController.InWater && currentHull != null)
301  {
302  standStillTimer -= deltaTime;
303  if (standStillTimer > 0.0f)
304  {
305  walkDuration = Rand.Range(walkDurationMin, walkDurationMax);
306  if (currentHull.Rect.Width > IndoorsSteeringManager.smallRoomSize / 2 && tooCloseCharacter == null)
307  {
308  foreach (Character c in Character.CharacterList)
309  {
310  if (c == character || !c.IsBot || c.CurrentHull != currentHull || !(c.AIController is HumanAIController humanAI)) { continue; }
311  if (Vector2.DistanceSquared(c.WorldPosition, character.WorldPosition) > 60.0f * 60.0f) { continue; }
312  if ((humanAI.ObjectiveManager.CurrentObjective is AIObjectiveIdle idleObjective && idleObjective.standStillTimer > 0.0f) ||
313  (humanAI.ObjectiveManager.CurrentObjective is AIObjectiveGoTo gotoObjective && gotoObjective.IsCloseEnough))
314  {
315  //if there are characters too close on both sides, don't try to steer away from them
316  //because it'll cause the character to spaz out trying to avoid both
317  if (tooCloseCharacter != null &&
318  Math.Sign(tooCloseCharacter.WorldPosition.X - character.WorldPosition.X) != Math.Sign(c.WorldPosition.X - character.WorldPosition.X))
319  {
320  tooCloseCharacter = null;
321  break;
322  }
323  tooCloseCharacter = c;
324  }
325  HumanAIController.FaceTarget(c);
326  }
327  }
328 
329  if (tooCloseCharacter != null && !tooCloseCharacter.Removed && Vector2.DistanceSquared(tooCloseCharacter.WorldPosition, character.WorldPosition) < 50.0f * 50.0f)
330  {
331  Vector2 diff = character.WorldPosition - tooCloseCharacter.WorldPosition;
332  if (diff.LengthSquared() < 0.0001f) { diff = Rand.Vector(1.0f); }
333  if (Math.Abs(diff.X) > 0 &&
334  (character.WorldPosition.X > currentHull.WorldRect.Right - 50 || character.WorldPosition.X < currentHull.WorldRect.Left + 50))
335  {
336  // Between a wall and a character -> move away
337  tooCloseCharacter = null;
338  PathSteering.Reset();
339  standStillTimer = 0;
340  walkDuration = Math.Min(walkDuration, walkDurationMin);
341  if (Behavior != BehaviorType.StayInHull && (currentHull.Size.X < IndoorsSteeringManager.smallRoomSize || currentHull.Size.X < (IndoorsSteeringManager.smallRoomSize / 2 * Character.CharacterList.Count(c => c.CurrentHull == currentHull))))
342  {
343  // Small room -> find another
344  newTargetTimer = Math.Min(newTargetTimer, 1);
345  }
346  }
347  else
348  {
349  PathSteering.SteeringManual(deltaTime, Vector2.Normalize(diff));
350  }
351  return;
352  }
353  else
354  {
355  PathSteering.Reset();
356  tooCloseCharacter = null;
357  }
358 
359  chairCheckTimer -= deltaTime;
360  if (chairCheckTimer <= 0.0f && character.SelectedConstruction == null)
361  {
362  foreach (Item item in Item.ItemList)
363  {
364  if (item.CurrentHull != currentHull || !item.HasTag("chair")) { continue; }
365  var controller = item.GetComponent<Controller>();
366  if (controller == null || controller.User != null) { continue; }
367  item.TryInteract(character, forceSelectKey: true);
368  }
369  chairCheckTimer = chairCheckInterval;
370  }
371  return;
372  }
373  if (standStillTimer < -walkDuration)
374  {
375  standStillTimer = Rand.Range(standStillMin, standStillMax);
376  }
377  }
378 
379  PathSteering.Wander(deltaTime);
380  }

390  private void FindTargetHulls()
391  {
392  targetHulls.Clear();
393  hullWeights.Clear();
394  foreach (var hull in Hull.HullList)
395  {
396  if (character.Submarine == null) { break; }
397  if (HumanAIController.UnsafeHulls.Contains(hull)) { continue; }
398  if (hull.Submarine == null) { continue; }
399  if (hull.Submarine.Info.IsRuin || hull.Submarine.Info.IsWreck) { continue; }
400  if (character.TeamID == CharacterTeamType.FriendlyNPC && !character.IsEscorted)
401  {
402  if (hull.Submarine.TeamID != character.TeamID)
403  {
404  // Don't allow npcs to idle in a sub that's not in their team (like the player sub)
405  continue;
406  }
407  }
408  else
409  {
410  if (hull.Submarine.TeamID != character.Submarine.TeamID)
411  {
412  // Don't allow to idle in the subs that are not in the same team as the current sub
413  // -> the crew ai bots can't change the sub from outpost to main sub or vice versa on their own
414  continue;
415  }
416  }
417  if (IsForbidden(hull)) { continue; }
418  // Check that the hull is linked
419  if (!character.Submarine.IsConnectedTo(hull.Submarine)) { continue; }
420  // Ignore very narrow hulls.
421  if (hull.RectWidth < 200) { continue; }
422  // Ignore hulls that are too low to stand inside.
423  if (character.AnimController is HumanoidAnimController animController)
424  {
425  if (hull.CeilingHeight < ConvertUnits.ToDisplayUnits(animController.HeadPosition.Value))
426  {
427  continue;
428  }
429  }
430  if (!targetHulls.Contains(hull))
431  {
432  targetHulls.Add(hull);
433  float weight = hull.RectWidth;
434  // Prefer rooms that are closer. Avoid rooms that are not in the same level.
435  // If the behavior is active, prefer rooms that are not close.
436  float yDist = Math.Abs(character.WorldPosition.Y - hull.WorldPosition.Y);
437  yDist = yDist > 100 ? yDist * 5 : 0;
438  float dist = Math.Abs(character.WorldPosition.X - hull.WorldPosition.X) + yDist;
439  float distanceFactor = behavior == BehaviorType.Patrol ? MathHelper.Lerp(1, 0, MathUtils.InverseLerp(2500, 0, dist)) : MathHelper.Lerp(1, 0, MathUtils.InverseLerp(0, 2500, dist));
440  float waterFactor = MathHelper.Lerp(1, 0, MathUtils.InverseLerp(0, 100, hull.WaterPercentage * 2));
441  weight *= distanceFactor * waterFactor;
442  hullWeights.Add(weight);
443  }
444  }
445 
446  if (PreferredOutpostModuleTypes.Any() && character.CurrentHull != null)
447  {
448  for (int i = 0; i < targetHulls.Count; i++)
449  {
450  if (targetHulls[i].OutpostModuleTags.Any(t => PreferredOutpostModuleTypes.Contains(t)))
451  {
452  hullWeights[i] *= Rand.Range(10.0f, 100.0f);
453  }
454  }
455  }
456  }
