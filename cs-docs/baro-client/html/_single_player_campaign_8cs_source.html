82 private SinglePlayerCampaign(XElement element) : base(GameModePreset.SinglePlayerCampaign)
83 {
84 IsFirstRound = false;
85
86 foreach (var subElement in element.Elements())
87 {
88 switch (subElement.Name.ToString().ToLowerInvariant())
89 {
90 case "campaignsettings":
91 Settings = new CampaignSettings(subElement);
92 break;
93 case "crew":
94 GameMain.GameSession.CrewManager = new CrewManager(subElement, true);
95 break;
96 case "map":
97 map = Map.Load(this, subElement, Settings);
98 break;
99 case "metadata":
100 CampaignMetadata = new CampaignMetadata(this, subElement);
101 break;
102 case "cargo":
103 CargoManager.LoadPurchasedItems(subElement);
104 break;
105 case "pendingupgrades": //backwards compatibility
106 case "upgrademanager":
107 UpgradeManager = new UpgradeManager(this, subElement, isSingleplayer: true);
108 break;
109 case "pets":
110 petsElement = subElement;
111 break;
112 case Wallet.LowerCaseSaveElementName:
113 Bank = new Wallet(Option<Character>.None(), subElement);
114 break;
115 case "stats":
116 LoadStats(subElement);
117 break;
118 }
119 }
120
121 CampaignMetadata ??= new CampaignMetadata(this);
122 UpgradeManager ??= new UpgradeManager(this);
123
124 InitCampaignData();
125
126 InitUI();
127
128 //backwards compatibility for saves made prior to the addition of personal wallets
129 int oldMoney = element.GetAttributeInt("money", 0);
130 if (oldMoney > 0)
131 {
132 Bank = new Wallet(Option<Character>.None())
133 {
134 Balance = oldMoney
135 };
136 }
137
138 PurchasedLostShuttles = element.GetAttributeBool("purchasedlostshuttles", false);
139 PurchasedHullRepairs = element.GetAttributeBool("purchasedhullrepairs", false);

154 if (map == null)
155 {
156 throw new System.Exception("Failed to load the campaign save file (saved with an older, incompatible version of Barotrauma).");

397 protected override IEnumerable<CoroutineStatus> DoLevelTransition(TransitionType transitionType, LevelData newLevel, Submarine leavingSub, bool mirror, List<TraitorMissionResult> traitorResults = null)
398 {
399 NextLevel = newLevel;
400 bool success = CrewManager.GetCharacters().Any(c => !c.IsDead);
401 SoundPlayer.OverrideMusicType = (success ? "endround" : "crewdead").ToIdentifier();
402 SoundPlayer.OverrideMusicDuration = 18.0f;
403 GUI.SetSavingIndicatorState(success);
404 crewDead = false;
405
406 GameMain.GameSession.EndRound("", traitorResults, transitionType);
407 var continueButton = GameMain.GameSession.RoundSummary?.ContinueButton;
408 RoundSummary roundSummary = null;
409 if (GUIMessageBox.VisibleBox?.UserData is RoundSummary)
410 {
411 roundSummary = GUIMessageBox.VisibleBox?.UserData as RoundSummary;
412 }
413 if (continueButton != null)
414 {
415 continueButton.Visible = false;
416 }
417
418 lastControlledCharacter = Character.Controlled;
419 Character.Controlled = null;
420
421 switch (transitionType)
422 {
423 case TransitionType.None:
424 throw new InvalidOperationException("Level transition failed (no transitions available).");
425 case TransitionType.ReturnToPreviousLocation:
426 //deselect destination on map
427 map.SelectLocation(-1);
428 break;
429 case TransitionType.ProgressToNextLocation:
430 Map.MoveToNextLocation();
431 TotalPassedLevels++;
432 break;
433 case TransitionType.ProgressToNextEmptyLocation:
434 TotalPassedLevels++;
435 break;
436 }
437
438 Map.ProgressWorld(transitionType, (float)(Timing.TotalTime - GameMain.GameSession.RoundStartTime));
439
440 var endTransition = new CameraTransition(Submarine.MainSub, GameMain.GameScreen.Cam, null,
441 transitionType == TransitionType.LeaveLocation ? Alignment.BottomCenter : Alignment.Center,
442 fadeOut: false,
443 panDuration: EndTransitionDuration);
444
445 GUI.ClearMessages();
446
447 Location portraitLocation = Map.SelectedLocation ?? Map.CurrentLocation;
448 overlaySprite = portraitLocation.Type.GetPortrait(portraitLocation.PortraitId);
449 float fadeOutDuration = endTransition.PanDuration;
450 float t = 0.0f;
451 while (t < fadeOutDuration || endTransition.Running)
452 {
453 t += CoroutineManager.UnscaledDeltaTime;
454 overlayColor = Color.Lerp(Color.Transparent, Color.White, t / fadeOutDuration);
455 yield return CoroutineStatus.Running;
456 }
457 overlayColor = Color.White;
458 yield return CoroutineStatus.Running;
459
460 //--------------------------------------
461
462 if (success)
463 {
464 if (leavingSub != Submarine.MainSub && !leavingSub.DockedTo.Contains(Submarine.MainSub))
465 {
466 Submarine.MainSub = leavingSub;
467 GameMain.GameSession.Submarine = leavingSub;
468 GameMain.GameSession.SubmarineInfo = leavingSub.Info;
469 leavingSub.Info.FilePath = System.IO.Path.Combine(SaveUtil.TempPath, leavingSub.Info.Name + ".sub");
470 var subsToLeaveBehind = GetSubsToLeaveBehind(leavingSub);
471 GameMain.GameSession.OwnedSubmarines.Add(leavingSub.Info);
472 foreach (Submarine sub in subsToLeaveBehind)
473 {
474 GameMain.GameSession.OwnedSubmarines.RemoveAll(s => s != leavingSub.Info && s.Name == sub.Info.Name);
475 MapEntity.mapEntityList.RemoveAll(e => e.Submarine == sub && e is LinkedSubmarine);
476 LinkedSubmarine.CreateDummy(leavingSub, sub);
477 }
478 }
479
480 GameMain.GameSession.SubmarineInfo = new SubmarineInfo(GameMain.GameSession.Submarine);
481
482 if (PendingSubmarineSwitch != null)
483 {
484 SubmarineInfo previousSub = GameMain.GameSession.SubmarineInfo;
485 GameMain.GameSession.SubmarineInfo = PendingSubmarineSwitch;
486
487 for (int i = 0; i < GameMain.GameSession.OwnedSubmarines.Count; i++)
488 {
489 if (GameMain.GameSession.OwnedSubmarines[i].Name == previousSub.Name)
490 {
491 GameMain.GameSession.OwnedSubmarines[i] = previousSub;
492 break;
493 }
494 }
495 }
496
497 SaveUtil.SaveGame(GameMain.GameSession.SavePath);
498 PendingSubmarineSwitch = null;
499 }
500 else
501 {
502 PendingSubmarineSwitch = null;
503 EnableRoundSummaryGameOverState();
504 }
505
506 CrewManager?.ClearCurrentOrders();
507
508 //--------------------------------------
509
510 SelectSummaryScreen(roundSummary, newLevel, mirror, () =>
511 {
512 GameMain.GameScreen.Select();
513 if (continueButton != null)
514 {
515 continueButton.Visible = true;
516 }
517
518 GUI.DisableHUD = false;
519 GUI.ClearCursorWait();
520 overlayColor = Color.Transparent;
521 });
522
523 GUI.SetSavingIndicatorState(false);
524 yield return CoroutineStatus.Success;
525 }

565 public override void Update(float deltaTime)
566 {
567 if (CoroutineManager.IsCoroutineRunning("LevelTransition") || CoroutineManager.IsCoroutineRunning("SubmarineTransition") || gameOver) { return; }
568
569 base.Update(deltaTime);
570
571 Map?.Radiation?.UpdateRadiation(deltaTime);
572
573 if (PlayerInput.SecondaryMouseButtonClicked() ||
574 PlayerInput.KeyHit(Microsoft.Xna.Framework.Input.Keys.Escape))
575 {
576 ShowCampaignUI = false;
577 if (GUIMessageBox.VisibleBox?.UserData is RoundSummary roundSummary &&
578 roundSummary.ContinueButton != null &&
579 roundSummary.ContinueButton.Visible)
580 {
581 GUIMessageBox.MessageBoxes.Remove(GUIMessageBox.VisibleBox);
582 }
583 }
584
585#if DEBUG
586 if (GUI.KeyboardDispatcher.Subscriber == null && PlayerInput.KeyHit(Microsoft.Xna.Framework.Input.Keys.M))
587 {
588 if (GUIMessageBox.MessageBoxes.Any()) { GUIMessageBox.MessageBoxes.Remove(GUIMessageBox.MessageBoxes.Last()); }
589
590 GUIFrame summaryFrame = GameMain.GameSession.RoundSummary.CreateSummaryFrame(GameMain.GameSession, "", null);
591 GUIMessageBox.MessageBoxes.Add(summaryFrame);
592 GameMain.GameSession.RoundSummary.ContinueButton.OnClicked = (_, __) => { GUIMessageBox.MessageBoxes.Remove(summaryFrame); return true; };
593 }
594#endif
595
596 if (ShowCampaignUI || ForceMapUI)
597 {
598 Character.DisableControls = true;
599 }
600
601 if (!GUI.DisableHUD && !GUI.DisableUpperHUD)
602 {
603 endRoundButton.UpdateManually(deltaTime);
604 if (CoroutineManager.IsCoroutineRunning("LevelTransition") || ForceMapUI) { return; }
605 }
606
607 if (Level.Loaded.Type == LevelData.LevelType.Outpost)
608 {
609 KeepCharactersCloseToOutpost(deltaTime);
610 if (wasDocked)
611 {
612 var connectedSubs = Submarine.MainSub.GetConnectedSubs();
613 bool isDocked = Level.Loaded.StartOutpost != null && connectedSubs.Contains(Level.Loaded.StartOutpost);
614 if (!isDocked)
615 {
616 //undocked from outpost, need to choose a destination
617 ForceMapUI = true;
618 CampaignUI.SelectTab(InteractionType.Map);
619 }
620 }
621 else
622 {
623 //wasn't initially docked (sub doesn't have a docking port?)
624 // -> choose a destination when the sub is far enough from the start outpost
625 if (!Submarine.MainSub.AtStartExit)
626 {
627 ForceMapUI = true;
628 CampaignUI.SelectTab(InteractionType.Map);
629 }
630 }
631 }
632 else
633 {
634 var transitionType = GetAvailableTransition(out _, out Submarine leavingSub);
635 if (transitionType == TransitionType.End)
636 {
637 EndCampaign();
638 }
639 if (transitionType == TransitionType.ProgressToNextLocation &&
640 Level.Loaded.EndOutpost != null && Level.Loaded.EndOutpost.DockedTo.Contains(leavingSub))
641 {
642 LoadNewLevel();
643 }
644 else if (transitionType == TransitionType.ReturnToPreviousLocation &&
645 Level.Loaded.StartOutpost != null && Level.Loaded.StartOutpost.DockedTo.Contains(leavingSub))
646 {
647 LoadNewLevel();
648 }
649 else if (transitionType == TransitionType.None && CampaignUI.SelectedTab == InteractionType.Map)
650 {
651 ShowCampaignUI = false;
652 }
653 HintManager.OnAvailableTransition(transitionType);
654 }
655
656 if (!crewDead)
657 {
658 if (!CrewManager.GetCharacters().Any(c => !c.IsDead)) { crewDead = true; }
659 }
660 else
661 {
662 endTimer -= deltaTime;
663 if (endTimer <= 0.0f) { GameOver(); }
664 }
665 }

667 private bool TryEndRound()
668 {
669 var transitionType = GetAvailableTransition(out LevelData nextLevel, out Submarine leavingSub);
670 if (leavingSub == null || transitionType == TransitionType.None) { return false; }
671
672 if (nextLevel == null)
673 {
674 //no level selected -> force the player to select one
675 ForceMapUI = true;
676 CampaignUI.SelectTab(InteractionType.Map);
677 map.SelectLocation(-1);
678 return false;
679 }
680 else if (transitionType == TransitionType.ProgressToNextEmptyLocation)
681 {
682 Map.SetLocation(Map.Locations.IndexOf(Level.Loaded.EndLocation ?? Map.CurrentLocation));
683 }
684
685 var subsToLeaveBehind = GetSubsToLeaveBehind(leavingSub);
686 if (subsToLeaveBehind.Any())
687 {
688 LocalizedString msg = TextManager.Get(subsToLeaveBehind.Count == 1 ? "LeaveSubBehind" : "LeaveSubsBehind");
689
690 var msgBox = new GUIMessageBox(TextManager.Get("Warning"), msg, new LocalizedString[] { TextManager.Get("Yes"), TextManager.Get("No") });
691 msgBox.Buttons[0].OnClicked += (btn, userdata) => { LoadNewLevel(); return true; } ;
692 msgBox.Buttons[0].OnClicked += msgBox.Close;
693 msgBox.Buttons[0].UserData = Submarine.Loaded.FindAll(s => !subsToLeaveBehind.Contains(s));
694 msgBox.Buttons[1].OnClicked += msgBox.Close;
695 }
696 else
697 {
698 LoadNewLevel();
699 }
700
701 return true;
702 }

742 public override void Save(XElement element)
743 {
744 XElement modeElement = new XElement("SinglePlayerCampaign",
745 new XAttribute("purchasedlostshuttles", PurchasedLostShuttles),
746 new XAttribute("purchasedhullrepairs", PurchasedHullRepairs),
747 new XAttribute("purchaseditemrepairs", PurchasedItemRepairs),
748 new XAttribute("cheatsenabled", CheatsEnabled));
749 modeElement.Add(Settings.Save());
750 modeElement.Add(SaveStats());
751
752 //save and remove all items that are in someone's inventory so they don't get included in the sub file as well
753 foreach (Character c in Character.CharacterList)
754 {
755 if (c.Info == null) { continue; }
756 if (c.IsDead) { CrewManager.RemoveCharacterInfo(c.Info); }
757 c.Info.LastControlled = c == lastControlledCharacter;
758 c.Info.HealthData = new XElement("health");
759 c.CharacterHealth.Save(c.Info.HealthData);
760 if (c.Inventory != null)
761 {
762 c.Info.InventoryData = new XElement("inventory");
763 c.SaveInventory();
764 c.Inventory?.DeleteAllItems();
765 }
766 c.Info.SaveOrderData();
767 }
