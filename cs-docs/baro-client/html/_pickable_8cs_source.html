40  public Pickable(Item item, ContentXElement element)
41  : base(item, element)
42  {
43  allowedSlots = new List<InvSlotType>();
44 
45  string slotString = element.GetAttributeString("slots", "Any");
46  string[] slotCombinations = slotString.Split(',');
47  foreach (string slotCombination in slotCombinations)
48  {
49  string[] slots = slotCombination.Split('+');
50  InvSlotType allowedSlot = InvSlotType.None;
51  foreach (string slot in slots)
52  {
53  switch (slot.ToLowerInvariant())
54  {
55  case "bothhands":
56  allowedSlot = InvSlotType.LeftHand | InvSlotType.RightHand;
57  break;
58  default:
59  allowedSlot = allowedSlot | (InvSlotType)Enum.Parse(typeof(InvSlotType), slot.Trim());
60  break;
61  }
62  }
63  allowedSlots.Add(allowedSlot);
64  }
65 
66  canBePicked = true;
67  }

69  public override bool Pick(Character picker)
70  {
71  //return if someone is already trying to pick the item
72  if (pickTimer > 0.0f) { return false; }
73  if (picker == null || picker.Inventory == null) { return false; }
74 
75  if (PickingTime > 0.0f)
76  {
77  var abilityPickingTime = new AbilityItemPickingTime(PickingTime, item.Prefab);
78  picker.CheckTalents(AbilityEffectType.OnItemPicked, abilityPickingTime);
79 
80  if (requiredItems.ContainsKey(RelatedItem.RelationType.Equipped))
81  {
82  foreach (RelatedItem ri in requiredItems[RelatedItem.RelationType.Equipped])
83  {
84  foreach (var heldItem in picker.HeldItems)
85  {
86  if (ri.MatchesItem(heldItem))
87  {
88  abilityPickingTime.Value /= 1 + heldItem.Prefab.AddedPickingSpeedMultiplier;
89  }
90  }
91  }
92  }
93 
94  if ((picker.PickingItem == null || picker.PickingItem == item) && PickingTime <= float.MaxValue)
95  {
96 #if SERVER
97  // Set active picker before creating the server event to make sure it's set correctly
98  activePicker = picker;
99  item.CreateServerEvent(this);
100 #endif
101  pickingCoroutine = CoroutineManager.StartCoroutine(WaitForPick(picker, abilityPickingTime.Value));
102  }
103  return false;
104  }
105  else
106  {
107  return OnPicked(picker);
108  }
109  }

111  public virtual bool OnPicked(Character picker)
112  {
113  //if the item has multiple Pickable components (e.g. Holdable and Wearable, check that we don't equip it in hands when the item is worn or vice versa)
114  if (item.GetComponents<Pickable>().Count() > 0)
115  {
116  bool alreadyEquipped = false;
117  for (int i = 0; i < picker.Inventory.Capacity; i++)
118  {
119  if (picker.Inventory.GetItemsAt(i).Contains(item))
120  {
121  if (picker.Inventory.SlotTypes[i] != InvSlotType.Any &&
122  !allowedSlots.Any(a => a.HasFlag(picker.Inventory.SlotTypes[i])))
123  {
124  alreadyEquipped = true;
125  break;
126  }
127  }
128  }
129  if (alreadyEquipped) { return false; }
130  }
131  if (picker.Inventory.TryPutItemWithAutoEquipCheck(item, picker, allowedSlots))
132  {
133  if (!picker.HeldItems.Contains(item) && item.body != null) { item.body.Enabled = false; }
134  this.picker = picker;
135 
136  for (int i = item.linkedTo.Count - 1; i >= 0; i--)
137  {
138  item.linkedTo[i].RemoveLinked(item);
139  }
140  item.linkedTo.Clear();
141 
142  DropConnectedWires(picker);
143 
144  ApplyStatusEffects(ActionType.OnPicked, 1.0f, picker);
145 #if CLIENT
146  if (!GameMain.Instance.LoadingScreenOpen && picker == Character.Controlled) SoundPlayer.PlayUISound(GUISoundType.PickItem);
147  PlaySound(ActionType.OnPicked, picker);
148 #endif
149  return true;
150  }
151 
152 #if CLIENT
153  if (!GameMain.Instance.LoadingScreenOpen && picker == Character.Controlled) SoundPlayer.PlayUISound(GUISoundType.PickItemFail);
154 #endif
155 
156  return false;
157  }

159  private IEnumerable<CoroutineStatus> WaitForPick(Character picker, float requiredTime)
160  {
161  activePicker = picker;
162  picker.PickingItem = item;
163  pickTimer = 0.0f;
164  while (pickTimer < requiredTime && Screen.Selected != GameMain.SubEditorScreen)
165  {
166  //cancel if the item is currently selected
167  //attempting to pick does not select the item, so if it is selected at this point, another ItemComponent
168  //must have been selected and we should not keep deattaching (happens when for example interacting with
169  //an electrical component while holding both a screwdriver and a wrench).
170  if (picker.SelectedConstruction == item ||
171  picker.IsKeyDown(InputType.Aim) ||
172  !picker.CanInteractWith(item) ||
173  item.Removed || item.ParentInventory != null)
174  {
175  StopPicking(picker);
176  yield return CoroutineStatus.Success;
177  }
178 
179 #if CLIENT
180  Character.Controlled?.UpdateHUDProgressBar(
181  this,
182  item.WorldPosition,
183  pickTimer / requiredTime,
184  GUIStyle.Red, GUIStyle.Green,
185  !string.IsNullOrWhiteSpace(PickingMsg) ? PickingMsg : this is Door ? "progressbar.opening" : "progressbar.deattaching");
186 #endif
187 
188  picker.AnimController.UpdateUseItem(true, item.WorldPosition + new Vector2(0.0f, 100.0f) * ((pickTimer / 10.0f) % 0.1f));
189  pickTimer += CoroutineManager.DeltaTime;
190 
191  yield return CoroutineStatus.Running;
192  }
193 
194  StopPicking(picker);
195 
196  bool isNotRemote = true;
197 #if CLIENT
198  isNotRemote = !picker.IsRemotePlayer;
199 #endif
200  if (isNotRemote) OnPicked(picker);
201 
202  yield return CoroutineStatus.Success;
203  }

265  picker = null;
266  }

274  + (item.Removed ? ", item has been removed)" : ")"));
275  }
276  else
