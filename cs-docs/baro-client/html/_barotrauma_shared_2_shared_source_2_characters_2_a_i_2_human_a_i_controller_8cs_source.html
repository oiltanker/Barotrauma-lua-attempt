1887  waterFactor = 1;

1895  fireFactor = MathHelper.Lerp(1, 0, MathHelper.Clamp(fire, 0, 1));
1896  }

1898  if (!ignoreEnemies)
1899  {

1902  Character.CharacterList.Count(e => isValidTarget(e) && e.CurrentHull == hull) :
1903  Character.CharacterList.Count(e => isValidTarget(e) && visibleHulls.Contains(e.CurrentHull));

1905  enemyFactor = MathHelper.Lerp(1, 0, MathHelper.Clamp(enemyCount * 0.9f, 0, 1));

1914  }
1915  }
