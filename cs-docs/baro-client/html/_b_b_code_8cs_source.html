48 protected static GUICustomComponent CreateBBCodeElement(string bbCode, GUIListBox container)
49 {
50 Point cachedContainerSize = Point.Zero;
51 List<BBWord> bbWords = new List<BBWord>();
52 Stack<BBWord.TagType> tagStack = new Stack<BBWord.TagType>();
53
54 void recalculate()
55 {
56 if (cachedContainerSize == container.Content.RectTransform.NonScaledSize) { return; }
57
58 bbWords.Clear();
59 cachedContainerSize = container.Content.RectTransform.NonScaledSize;
60
61 var matches = new Stack<Match>(bbTagRegex.Matches(bbCode).Reverse());
62 Match? nextTag = null;
63 matches.TryPop(out nextTag);
64 int wordStart = 0;
65 BBWord.TagType currTagType;
66 for (int i = 0; i < bbCode.Length; i++)
67 {
68 char currChar = bbCode[i];
69 currTagType = tagStack.TryPeek(out var t) ? t : BBWord.TagType.None;
70
71 bool charIsCJK = TextManager.IsCJK($"{currChar}");
72 bool wordEnd = char.IsWhiteSpace(currChar) || charIsCJK;
73 int reachedTagLength = 0;
74 if (nextTag is { Index: int tagIndex, Length: int tagLength }
75 && i == tagIndex)
76 {
77 reachedTagLength = tagLength;
78 string tagStr = nextTag.Value.Replace("[", "").Replace("]", "").Trim();
79 bool isClosing = tagStr.StartsWith("/");
80 tagStr = tagStr.Replace("/", "").Trim().ToLowerInvariant();
81 BBWord.TagType tagType = tagStr switch
82 {
83 "b" => BBWord.TagType.Bold,
84 "i" => BBWord.TagType.Italic,
85 "h1" => BBWord.TagType.Header,
86 _ => BBWord.TagType.None
87 };
88
89 if (tagType != BBWord.TagType.None)
90 {
91 if (isClosing)
92 {
93 if (currTagType == tagType)
94 {
95 tagStack.Pop();
96 }
97 }
98 else
99 {
100 tagStack.Push(tagType);
101 }
102 }
103 }
104
105 if (wordEnd || reachedTagLength > 0)
106 {
107 string word = bbCode[wordStart..i];
108 if (charIsCJK) { word = bbCode[wordStart..(i + 1)]; }
109 else if (char.IsWhiteSpace(currChar) && currChar != '\n') { word += " "; }
110
111 if (!word.IsNullOrEmpty())
112 {
113 bbWords.Add(new BBWord(word, currTagType));
114 }
115 else if (currChar == '\n')
116 {
117 bbWords.Add(new BBWord("", BBWord.TagType.NewLine));
118 }
119
120 if (reachedTagLength > 0)
121 {
122 i += reachedTagLength - 1;
123 nextTag = matches.TryPop(out var tag) ? tag : null;
124 }
125
126 wordStart = i + 1;
127 }
128 }
129
130 currTagType = tagStack.TryPeek(out var ft) ? ft : BBWord.TagType.None;
131 string finalWord = bbCode[wordStart..];
132 if (!finalWord.IsNullOrEmpty())
133 {
134 bbWords.Add(new BBWord(finalWord, currTagType));
135 }
136 }

138 void draw(SpriteBatch spriteBatch, GUICustomComponent component)
139 {
140 recalculate();
141 Vector2 currPos = Vector2.Zero;
142 Vector2 rectPos = component.Rect.Location.ToVector2();
143 for (int i = 0; i < bbWords.Count; i++)
144 {
145 var bbWord = bbWords[i];
146 if (currPos.X > 0.0f
147 && currPos.X + bbWord.Size.X >= component.Rect.Width)
148 {
149 //wrap because we went over width limit
150 currPos = (0.0f, currPos.Y + bbWord.Size.Y);
151 }
152
153 bbWord.Font.DrawString(
154 spriteBatch,
155 bbWord.Text,
156 (currPos + rectPos).ToPoint().ToVector2(),
157 GUIStyle.TextColorNormal,
158 forceUpperCase: ForceUpperCase.No,
159 italics: bbWord.TagTypes.HasFlag(BBWord.TagType.Italic));
160 bool breakLine
161 = bbWord.TagTypes.HasFlag(BBWord.TagType.NewLine)
162 || (i < bbWords.Count - 1 &&
163 bbWords[i + 1].TagTypes.HasFlag(BBWord.TagType.Header) !=
164 bbWord.TagTypes.HasFlag(BBWord.TagType.Header));
165 if (breakLine)
166 {
167 //break line because of a header change or newline was found
168 currPos = (0.0f, currPos.Y + bbWord.Size.Y);
169 }
170 else
171 {
172 currPos.X += bbWord.Size.X;
173 }
174 }
175
176 component.RectTransform.NonScaledSize
177 = (component.RectTransform.NonScaledSize.X,
178 (int)(currPos.Y + bbWords.LastOrDefault().Size.Y));
179 component.RectTransform.RelativeSize
180 = component.RectTransform.NonScaledSize.ToVector2() / component.Parent.Rect.Size.ToVector2();
181 }
182
183 return new GUICustomComponent(new RectTransform(Vector2.One, container.Content.RectTransform),
184 onDraw: draw);
185 }
