59 public GUIContextMenu(Vector2? position, LocalizedString header, string style, params ContextMenuOption[] options) : base(style, new RectTransform(Point.Zero, GUI.Canvas))
60 {
61 Vector2 pos = position ?? PlayerInput.MousePosition;
62 GUIFont headerFont = GUIStyle.SubHeadingFont;
63 GUIFont font = GUIStyle.SmallFont; // font the context menu options use
64 Vector4 padding = new Vector4(4), headerPadding = new Vector4(8);
65 int horizontalPadding = (int) (padding.X + padding.Z), verticalPadding = (int) (padding.Y + padding.W);
66 bool hasHeader = !header.IsNullOrWhiteSpace();
67
68 //----------------------------------------------------------------------------------
69 // Estimate the size of the context menu
70 //----------------------------------------------------------------------------------
71
72 Dictionary<ContextMenuOption, Vector2> optionsAndSizes = new Dictionary<ContextMenuOption, Vector2>();
73
74 // estimate how big the context menu needs to be
75 Point estimatedSize = new Point(horizontalPadding, verticalPadding);
76
77 if (hasHeader)
78 {
79 InflateSize(ref estimatedSize, header, headerFont);
80 }
81
82 foreach (ContextMenuOption option in options)
83 {
84 Vector2 optionSize = InflateSize(ref estimatedSize, option.Label, font);
85 optionsAndSizes.Add(option, optionSize);
86 }
87
88 // it's better to overestimate the size since it's going to be cropped anyways
89 estimatedSize = estimatedSize.Multiply(1.2f);
90
91 RectTransform.NonScaledSize = estimatedSize;
92 RectTransform.AbsoluteOffset = pos.ToPoint();
93
94 //----------------------------------------------------------------------------------
95 // Construct the GUI elements
96 //----------------------------------------------------------------------------------
97
98 GUILayoutGroup background = new GUILayoutGroup(new RectTransform(Vector2.One, RectTransform, Anchor.Center))
99 {
100 Stretch = true
101 };
102
103 Point listSize = estimatedSize;
104 if (hasHeader)
105 {
106 Point sz = Point.Zero;
107 InflateSize(ref sz, header, headerFont);
108 listSize.Y -= sz.Y;
109 HeaderLabel = new GUITextBlock(new RectTransform(sz, background.RectTransform), header, font: headerFont) { Padding = headerPadding };
110 }
111
112 GUIListBox optionList = new GUIListBox(new RectTransform(listSize, background.RectTransform), style: null)
113 {
114 AutoHideScrollBar = false,
115 ScrollBarVisible = false,
116 Padding = hasHeader ? new Vector4(4, 0, 4, 4) : padding
117 };
118
119 foreach (var (option, size) in optionsAndSizes)
120 {
121 GUITextBlock optionElement = new GUITextBlock(new RectTransform(size.ToPoint(), optionList.Content.RectTransform), option.Label, font: font)
122 {
123 UserData = option,
124 Enabled = option.IsEnabled
125 };
126 Options.Add(option, optionElement);
127
128 if (!option.Tooltip.IsNullOrWhiteSpace() && optionElement.Enabled)
129 {
130 optionElement.ToolTip = option.Tooltip;
131 }
132
133 if (!option.IsEnabled)
134 {
135 optionElement.TextColor *= 0.5f;
136 }
137 }
138
139 //----------------------------------------------------------------------------------
140 // Positioning and cropping the context menu
141 //----------------------------------------------------------------------------------
142
143 List<GUIComponent> children = optionList.Content.Children.ToList();
144
145 // Resize all children to the size of their text
146 foreach (GUITextBlock block in children.Where(c => c is GUITextBlock).Cast<GUITextBlock>())
147 {
148 block.RectTransform.NonScaledSize = new Point((int) (block.TextSize.X + (block.Padding.X + block.Padding.Z)), (int) (18 * GUI.Scale));
149 }
150
151 int largestWidth = children.Max(c => c.Rect.Width + horizontalPadding);
152
153 // if the header is bigger than any of the options then overwrite
154 if (HeaderLabel != null)
155 {
156 RectTransform headerTransform = HeaderLabel.RectTransform;
157 headerTransform.MinSize = new Point((int) (HeaderLabel.TextSize.X + (headerPadding.X + headerPadding.Z)), headerTransform.NonScaledSize.Y);
158 if (largestWidth < headerTransform.MinSize.X)
159 {
160 largestWidth = headerTransform.MinSize.X;
161 }
162 }
163
164 // resize all children to the size of the longest element
165 foreach (GUIComponent c in children)
166 {
167 c.RectTransform.MinSize = new Point(largestWidth, c.Rect.Height);
168 }
169
170 // the cropped size of the option list
171 Point newSize = new Point(largestWidth, children.Sum(c => c.Rect.Height) + verticalPadding);
172 // resize the menu itself taking into account the option menus relative Y size
173 RectTransform.NonScaledSize = new Point(newSize.X, (int) (newSize.Y / optionList.RectTransform.RelativeSize.Y));
174 optionList.RectTransform.NonScaledSize = newSize;
175
176 // move the context menu if it would go outside of screen
177 if (RectTransform.Rect.Bottom > GameMain.GraphicsHeight)
178 {
179 Rectangle rect = RectTransform.Rect;
180 RectTransform.AbsoluteOffset = new Point(rect.X, rect.Y - rect.Height);
181 }
182
183 if (RectTransform.Rect.Right > GameMain.GraphicsWidth)
184 {
185 Rectangle rect = RectTransform.Rect;
186 RectTransform.AbsoluteOffset = new Point(rect.X - rect.Width, rect.Y);
187 }
188
189 background.Recalculate();
190
191 optionList.OnSelected = OnSelected;
192 }

226 private Vector2 InflateSize(ref Point size, LocalizedString label, ScalableFont font)
227 {
228 Vector2 textSize = font.MeasureString(label);
229 size.X = Math.Max((int) Math.Ceiling(textSize.X), size.X);
230 size.Y += (int) Math.Ceiling(textSize.Y);
231 return textSize;

234 protected override void Update(float deltaTime)
235 {
236 base.Update(deltaTime);
237
238 // keep the parent highlighted
239 if (ParentOption != null)
240 {
241 ParentOption.State = ComponentState.Hover;
242 }
243
244 if (SubMenu != null && !SubMenu.IsMouseOver())
245 {
246 SubMenu = null;
247 return;
248 }
249
250 foreach (var (option, textBlock) in Options)
251 {
252 // Create a new sub context menu if hovering over an option with sub options
253 if (GUI.MouseOn != textBlock) { continue; }
254 if (option.IsEnabled && option.SubOptions is { } subOptions && subOptions.Any())
255 {
256 Vector2 subMenuPos = new Vector2(textBlock.MouseRect.Right + 4, textBlock.MouseRect.Y);
257 SubMenu = new GUIContextMenu(subMenuPos, "", "GUIToolTip", subOptions)
258 {
259 ParentOption = textBlock
260 };
261 }
262 }

269 private bool IsMouseOver()
270 {
271 Rectangle expandedRect = Rect;
272 expandedRect.Inflate(20, 20);
273
274 bool isMouseOn = expandedRect.Contains(PlayerInput.MousePosition);
275
276 if (ParentOption != null)
277 {
278 isMouseOn |= GUI.MouseOn == ParentOption;
279 }
280
281 // Recursively check sub context menus
282 if (!isMouseOn && SubMenu != null)
283 {
284 isMouseOn = SubMenu.IsMouseOver();
285 }
286
287 return isMouseOn;
288 }
