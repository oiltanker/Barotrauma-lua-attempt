296 private void DrawLanguageSelectionPrompt(SpriteBatch spriteBatch, GraphicsDevice graphicsDevice)
297 {
298 if (AvailableLanguages is null) { return; }
299
300 if (languageSelectionFont == null)
301 {
302 languageSelectionFont = new ScalableFont("Content/Fonts/NotoSans/NotoSans-Bold.ttf",
303 (uint)(30 * (GameMain.GraphicsHeight / 1080.0f)), graphicsDevice);
304 }
305 if (languageSelectionFontCJK == null)
306 {
307 languageSelectionFontCJK = new ScalableFont("Content/Fonts/NotoSans/NotoSansCJKsc-Bold.otf",
308 (uint)(30 * (GameMain.GraphicsHeight / 1080.0f)), graphicsDevice, dynamicLoading: true);
309 }
310 if (languageSelectionCursor == null)
311 {
312 languageSelectionCursor = new Sprite("Content/UI/cursor.png", Vector2.Zero);
313 }
314
315 Vector2 textPos = new Vector2(GameMain.GraphicsWidth / 2, GameMain.GraphicsHeight * 0.3f);
316 Vector2 textSpacing = new Vector2(0.0f, (GameMain.GraphicsHeight * 0.5f) / AvailableLanguages.Length);
317 foreach (LanguageIdentifier language in AvailableLanguages)
318 {
319 string localizedLanguageName = TextManager.GetTranslatedLanguageName(language);
320 var font = TextManager.IsCJK(localizedLanguageName) ? languageSelectionFontCJK : languageSelectionFont;
321
322 Vector2 textSize = font.MeasureString(localizedLanguageName);
323 bool hover =
324 Math.Abs(PlayerInput.MousePosition.X - textPos.X) < textSize.X / 2 &&
325 Math.Abs(PlayerInput.MousePosition.Y - textPos.Y) < textSpacing.Y / 2;
326
327 font.DrawString(spriteBatch, localizedLanguageName, textPos - textSize / 2,
328 hover ? Color.White : Color.White * 0.6f);
329 if (hover && PlayerInput.PrimaryMouseButtonClicked())
330 {
331 var config = GameSettings.CurrentConfig;
332 config.Language = language;
333 GameSettings.SetCurrentConfig(config);
334 //reload tip in the selected language
335 SetSelectedTip(TextManager.Get("LoadingScreenTip"));
336 WaitForLanguageSelection = false;
337 languageSelectionFont?.Dispose(); languageSelectionFont = null;
338 languageSelectionFontCJK?.Dispose(); languageSelectionFontCJK = null;
339 break;
340 }
341
342 textPos += textSpacing;
343 }
344
345 languageSelectionCursor.Draw(spriteBatch, PlayerInput.LatestMousePosition, scale: 0.5f);
346 }

348 private void DrawSplashScreen(SpriteBatch spriteBatch, GraphicsDevice graphics)
349 {
350 if (currSplashScreen == null && PendingSplashScreens.Count == 0) { return; }
351
352 if (currSplashScreen == null)
353 {
354 var newSplashScreen = PendingSplashScreens.Dequeue();
355 string fileName = newSplashScreen.Filename;
356 try
357 {
358 currSplashScreen = Video.Load(graphics, GameMain.SoundManager, fileName);
359 currSplashScreen.AudioGain = newSplashScreen.Gain;
360 videoStartTime = DateTime.Now;
361 }
362 catch (Exception e)
363 {
364 DisableSplashScreen();
365 DebugConsole.ThrowError("Playing the splash screen \"" + fileName + "\" failed.", e);
366 PendingSplashScreens.Clear();
367 currSplashScreen = null;
368 }
369
370 if (currSplashScreen == null) { return; }
371 }
372
373 if (currSplashScreen.IsPlaying)
374 {
375 graphics.Clear(Color.Black);
376 float videoAspectRatio = (float)currSplashScreen.Width / (float)currSplashScreen.Height;
377 int width; int height;
378 if (GameMain.GraphicsHeight * videoAspectRatio > GameMain.GraphicsWidth)
379 {
380 width = GameMain.GraphicsWidth;
381 height = (int)(GameMain.GraphicsWidth / videoAspectRatio);
382 }
383 else
384 {
385 width = (int)(GameMain.GraphicsHeight * videoAspectRatio);
386 height = GameMain.GraphicsHeight;
387 }
388
389 spriteBatch.Begin();
390 spriteBatch.Draw(
391 currSplashScreen.GetTexture(),
392 destinationRectangle: new Rectangle(
393 GameMain.GraphicsWidth / 2 - width / 2,
394 GameMain.GraphicsHeight / 2 - height / 2,
395 width,
396 height),
397 sourceRectangle: new Rectangle(0, 0, currSplashScreen.Width, currSplashScreen.Height),
398 Color.White,
399 rotation: 0.0f,
400 origin: Vector2.Zero,
401 SpriteEffects.None,
402 layerDepth: 0.0f);
403 spriteBatch.End();
404
405 if (DateTime.Now > videoStartTime + new TimeSpan(0, 0, 0, 0, milliseconds: 500)
406 && GameMain.WindowActive
407 && (PlayerInput.KeyHit(Keys.Escape)
408 || PlayerInput.KeyHit(Keys.Space)
409 || PlayerInput.KeyHit(Keys.Enter)
410 || PlayerInput.PrimaryMouseButtonDown()))
411 {
412 currSplashScreen.Dispose(); currSplashScreen = null;
413 }
414 }
415 else if (DateTime.Now > videoStartTime + new TimeSpan(0, 0, 0, 0, milliseconds: 1500))
416 {
417 currSplashScreen.Dispose(); currSplashScreen = null;
418 }
419 }

430 {
431 drawn = false;
432 LoadState = null;
433 SetSelectedTip(TextManager.Get("LoadingScreenTip"));
434 currentBackgroundTexture = LocationType.Prefabs.GetRandomUnsynced()?.GetPortrait(Rand.Int(int.MaxValue))?.Texture;

436 while (!drawn)

438 yield return CoroutineStatus.Running;

440
441 CoroutineManager.StartCoroutine(loader);
442
443 yield return CoroutineStatus.Running;
444
445 while (CoroutineManager.IsCoroutineRunning(loader.ToString()))
446 {
447 yield return CoroutineStatus.Running;
448 }
449
450 LoadState = 100.0f;
451
452 yield return CoroutineStatus.Success;
453 }
