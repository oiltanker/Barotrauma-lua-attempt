253  public static UInt64 SteamIDStringToUInt64(string str)
254  {
255  if (string.IsNullOrWhiteSpace(str)) { return 0; }
256  UInt64 retVal;
257  if (str.StartsWith("STEAM64_", StringComparison.InvariantCultureIgnoreCase)) { str = str.Substring(8); }
258  if (UInt64.TryParse(str, out retVal) && retVal > (1 << 52)) { return retVal; }
259  if (!str.StartsWith("STEAM_", StringComparison.InvariantCultureIgnoreCase)) { return 0; }
260  string[] split = str.Substring(6).Split(':');
261  if (split.Length != 3) { return 0; }
262 
263  if (!UInt64.TryParse(split[0], out UInt64 universe)) { return 0; }
264  if (!UInt64.TryParse(split[1], out UInt64 y)) { return 0; }
265  if (!UInt64.TryParse(split[2], out UInt64 accountNumber)) { return 0; }
266 
267  UInt64 accountInstance = 1; UInt64 accountType = 1;
268 
269  return (universe << 56) | (accountType << 52) | (accountInstance << 32) | (accountNumber << 1) | y;
270  }

272  public static string SteamIDUInt64ToString(UInt64 uint64)
273  {
274  UInt64 y = uint64 & 0x1;
275  UInt64 accountNumber = (uint64 >> 1) & 0x7fffffff;
276  UInt64 universe = (uint64 >> 56) & 0xff;
277 
278  string retVal = "STEAM_" + universe.ToString() + ":" + y.ToString() + ":" + accountNumber.ToString();
279 
280  if (SteamIDStringToUInt64(retVal) != uint64) { return "STEAM64_" + uint64.ToString(); }
