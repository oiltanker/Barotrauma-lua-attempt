132 public static void Update(float deltaTime)
133 {
134 lock (queuedMessages)
135 {
136 while (queuedMessages.Count > 0)
137 {
138 var newMsg = queuedMessages.Dequeue();
139 AddMessage(newMsg);
140
141 if (GameSettings.CurrentConfig.SaveDebugConsoleLogs || GameSettings.CurrentConfig.VerboseLogging)
142 {
143 unsavedMessages.Add(newMsg);
144 if (unsavedMessages.Count >= messagesPerFile)
145 {
146 SaveLogs();
147 unsavedMessages.Clear();
148 }
149 }
150 }
151 }
152
153 if (!IsOpen && GUI.KeyboardDispatcher.Subscriber == null)
154 {
155 foreach (var (key, command) in DebugConsoleMapping.Instance.Bindings)
156 {
157 if (key.IsHit())
158 {
159 ExecuteCommand(command);
160 }
161 }
162 }
163
164 activeQuestionText?.SetAsLastChild();
165
166 if (PlayerInput.KeyHit(Keys.F3))
167 {
168 Toggle();
169 }
170 else if (isOpen && PlayerInput.KeyHit(Keys.Escape))
171 {
172 isOpen = false;
173 GUI.ForceMouseOn(null);
174 textBox.Deselect();
175 }
176
177 if (isOpen)
178 {
179 frame.UpdateManually(deltaTime);
180
181 Character.DisableControls = true;
182
183 if (PlayerInput.KeyHit(Keys.Tab))
184 {
185 textBox.Text = AutoComplete(textBox.Text, increment: string.IsNullOrEmpty(currentAutoCompletedCommand) ? 0 : 1 );
186 }
187
188 if (PlayerInput.KeyDown(Keys.LeftControl) || PlayerInput.KeyDown(Keys.RightControl))
189 {
190 if ((PlayerInput.KeyDown(Keys.C) || PlayerInput.KeyDown(Keys.D) || PlayerInput.KeyDown(Keys.Z)) && activeQuestionCallback != null)
191 {
192 activeQuestionCallback = null;
193 activeQuestionText = null;
194 NewMessage(PlayerInput.KeyDown(Keys.C) ? "^C" : PlayerInput.KeyDown(Keys.D) ? "^D" : "^Z", Color.White, true);
195 }
196 }
197
198 if (PlayerInput.KeyHit(Keys.Enter))
199 {
200 chatManager.Store(textBox.Text);
201 ExecuteCommand(textBox.Text);
202 textBox.Text = "";
203 }
204 }
205 }
