106  private void OnP2PData(ulong steamId, byte[] data, int dataLength)
107  {
108  if (!isActive) { return; }
109  if (steamId != hostSteamId) { return; }
110 
111  timeout = Screen.Selected == GameMain.GameScreen ?
112  NetworkConnection.TimeoutThresholdInGame :
113  NetworkConnection.TimeoutThreshold;
114 
115  PacketHeader packetHeader = (PacketHeader)data[0];
116 
117  if (!packetHeader.IsServerMessage()) { return; }
118 
119  if (packetHeader.IsConnectionInitializationStep())
120  {
121  ulong low = Lidgren.Network.NetBitWriter.ReadUInt32(data, 32, 8);
122  ulong high = Lidgren.Network.NetBitWriter.ReadUInt32(data, 32, 8 + 32);
123  ulong lobbyId = low + (high << 32);
124 
125  Steam.SteamManager.JoinLobby(lobbyId, false);
126  IReadMessage inc = new ReadOnlyMessage(data, false, 1 + 8, dataLength - (1 + 8), ServerConnection);
127  if (initializationStep != ConnectionInitialization.Success)
128  {
129  incomingInitializationMessages.Add(inc);
130  }
131  }
132  else if (packetHeader.IsHeartbeatMessage())
133  {
134  return; //TODO: implement heartbeats
135  }
136  else if (packetHeader.IsDisconnectMessage())
137  {
138  IReadMessage inc = new ReadOnlyMessage(data, false, 1, dataLength - 1, ServerConnection);
139  string msg = inc.ReadString();
140  Close(msg);
141  OnDisconnectMessageReceived?.Invoke(msg);
142  }
143  else
144  {
145  UInt16 length = Lidgren.Network.NetBitWriter.ReadUInt16(data, 16, 8);
146 
147  IReadMessage inc = new ReadOnlyMessage(data, packetHeader.IsCompressed(), 3, length, ServerConnection);
148  incomingDataMessages.Add(inc);
149  }
150  }

152  public override void Update(float deltaTime)
153  {
154  if (!isActive) { return; }
155 
156  if (GameMain.Client == null || !GameMain.Client.RoundStarting)
157  {
158  timeout -= deltaTime;
159  }
160  heartbeatTimer -= deltaTime;
161 
162  if (initializationStep != ConnectionInitialization.Password &&
163  initializationStep != ConnectionInitialization.ContentPackageOrder &&
164  initializationStep != ConnectionInitialization.Success)
165  {
166  connectionStatusTimer -= deltaTime;
167  if (connectionStatusTimer <= 0.0)
168  {
169  var state = Steamworks.SteamNetworking.GetP2PSessionState(hostSteamId);
170  if (state == null)
171  {
172  Close("SteamP2P connection could not be established");
173  OnDisconnectMessageReceived?.Invoke(DisconnectReason.SteamP2PError.ToString());
174  }
175  else
176  {
177  if (state?.P2PSessionError != Steamworks.P2PSessionError.None)
178  {
179  Close($"SteamP2P error code: {state?.P2PSessionError}");
180  OnDisconnectMessageReceived?.Invoke($"{DisconnectReason.SteamP2PError}/SteamP2P error code: {state?.P2PSessionError}");
181  }
182  }
183  connectionStatusTimer = 1.0f;
184  }
185  }
186 
187  for (int i = 0; i < 100; i++)
188  {
189  if (!Steamworks.SteamNetworking.IsP2PPacketAvailable()) { break; }
190  var packet = Steamworks.SteamNetworking.ReadP2PPacket();
191  if (packet.HasValue)
192  {
193  OnP2PData(packet?.SteamId ?? 0, packet?.Data, packet?.Data.Length ?? 0);
194  receivedBytes += packet?.Data.Length ?? 0;
195  }
196  }
197 
198  GameMain.Client?.NetStats?.AddValue(NetStats.NetStatType.ReceivedBytes, receivedBytes);
199  GameMain.Client?.NetStats?.AddValue(NetStats.NetStatType.SentBytes, sentBytes);
200 
201  if (heartbeatTimer < 0.0)
202  {
203  IWriteMessage outMsg = new WriteOnlyMessage();
204  outMsg.Write((byte)DeliveryMethod.Unreliable);
205  outMsg.Write((byte)PacketHeader.IsHeartbeatMessage);
206 
207  Steamworks.SteamNetworking.SendP2PPacket(hostSteamId, outMsg.Buffer, outMsg.LengthBytes, 0, Steamworks.P2PSend.Unreliable);
208  sentBytes += outMsg.LengthBytes;
209 
210  heartbeatTimer = 5.0;
211  }
212 
213  if (timeout < 0.0)
214  {
215  Close("Timed out");
216  OnDisconnectMessageReceived?.Invoke(DisconnectReason.SteamP2PTimeOut.ToString());
217  return;
218  }
219 
220  if (initializationStep != ConnectionInitialization.Success)
221  {
222  if (incomingDataMessages.Count > 0)
223  {
224  OnInitializationComplete?.Invoke();
225  initializationStep = ConnectionInitialization.Success;
226  }
227  else
228  {
229  foreach (IReadMessage inc in incomingInitializationMessages)
230  {
231  ReadConnectionInitializationStep(inc);
232  }
233  }
234  }
235 
236  if (initializationStep == ConnectionInitialization.Success)
237  {
238  foreach (IReadMessage inc in incomingDataMessages)
239  {
240  OnMessageReceived?.Invoke(inc);
241  }
242  }
243 
244  incomingInitializationMessages.Clear();
245  incomingDataMessages.Clear();
246  }

248  public override void Send(IWriteMessage msg, DeliveryMethod deliveryMethod, bool compressPastThreshold = true)
249  {
250  if (!isActive) { return; }
251 
252  byte[] buf = new byte[msg.LengthBytes + 4];
253  buf[0] = (byte)deliveryMethod;
254 
255  byte[] bufAux = new byte[msg.LengthBytes];
256  msg.PrepareForSending(ref bufAux, compressPastThreshold, out bool isCompressed, out int length);
257 
258  buf[1] = (byte)(isCompressed ? PacketHeader.IsCompressed : PacketHeader.None);
259 
260  buf[2] = (byte)(length & 0xff);
261  buf[3] = (byte)((length >> 8) & 0xff);
262 
263  Array.Copy(bufAux, 0, buf, 4, length);
264 
265  Steamworks.P2PSend sendType;
266  switch (deliveryMethod)
267  {
268  case DeliveryMethod.Reliable:
269  case DeliveryMethod.ReliableOrdered:
270  //the documentation seems to suggest that the Reliable send type
271  //enforces packet order (TODO: verify)
272  sendType = Steamworks.P2PSend.Reliable;
273  break;
274  default:
275  sendType = Steamworks.P2PSend.Unreliable;
276  break;
277  }
278 
279  if (length + 8 >= MsgConstants.MTU)
280  {
281  DebugConsole.Log("WARNING: message length comes close to exceeding MTU, forcing reliable send (" + length.ToString() + " bytes)");
282  sendType = Steamworks.P2PSend.Reliable;
283  }
284 
285  heartbeatTimer = 5.0;
286 
287 #if DEBUG
288  CoroutineManager.Invoke(() =>
289  {
290  if (GameMain.Client == null) { return; }
291  if (Rand.Range(0.0f, 1.0f) < GameMain.Client.SimulatedLoss && sendType != Steamworks.P2PSend.Reliable) { return; }
292  int count = Rand.Range(0.0f, 1.0f) < GameMain.Client.SimulatedDuplicatesChance ? 2 : 1;
293  for (int i = 0; i < count; i++)
294  {
295  Send(buf, length + 4, sendType);
296  }
297  },
298  GameMain.Client.SimulatedMinimumLatency + Rand.Range(0.0f, GameMain.Client.SimulatedRandomLatency));
299 #else
300  Send(buf, length + 4, sendType);
301 #endif
302  }

304  private void Send(byte[] buf, int length, Steamworks.P2PSend sendType)
305  {
306  bool successSend = Steamworks.SteamNetworking.SendP2PPacket(hostSteamId, buf, length + 4, 0, sendType);
307  sentBytes += length + 4;
308  if (!successSend)
309  {
310  if (sendType != Steamworks.P2PSend.Reliable)
311  {
312  DebugConsole.Log("WARNING: message couldn't be sent unreliably, forcing reliable send (" + length.ToString() + " bytes)");
313  sendType = Steamworks.P2PSend.Reliable;
314  successSend = Steamworks.SteamNetworking.SendP2PPacket(hostSteamId, buf, length + 4, 0, sendType);
315  sentBytes += length + 4;
316  }
317  if (!successSend)
318  {
319  DebugConsole.AddWarning("Failed to send message to remote peer! (" + length.ToString() + " bytes)");
320  }
321  }
