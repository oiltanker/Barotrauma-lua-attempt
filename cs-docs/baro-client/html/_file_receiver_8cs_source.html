202  public void ReadMessage(IReadMessage inc)
203  {
204  System.Diagnostics.Debug.Assert(!activeTransfers.Any(t =>
205  t.Status == FileTransferStatus.Error ||
206  t.Status == FileTransferStatus.Canceled ||
207  t.Status == FileTransferStatus.Finished), "List of active file transfers contains entires that should have been removed");
208 
209  byte transferMessageType = inc.ReadByte();
210 
211  switch (transferMessageType)
212  {
213  case (byte)FileTransferMessageType.Initiate:
214  {
215  byte transferId = inc.ReadByte();
216  var existingTransfer = activeTransfers.Find(t => t.Connection.EndpointMatches(t.Connection.EndPointString) && t.ID == transferId);
217  finishedTransfers.RemoveAll(t => t.transferId == transferId);
218  byte fileType = inc.ReadByte();
219  //ushort chunkLen = inc.ReadUInt16();
220  int fileSize = inc.ReadInt32();
221  string fileName = inc.ReadString();
222 
223  if (existingTransfer != null)
224  {
225  if (fileType != (byte)existingTransfer.FileType ||
226  fileSize != existingTransfer.FileSize ||
227  fileName != existingTransfer.FileName)
228  {
229  GameMain.Client.CancelFileTransfer(transferId);
230  DebugConsole.AddWarning("File transfer error: file transfer initiated with an ID that's already in use");
231  }
232  else //resend acknowledgement packet
233  {
234  GameMain.Client.UpdateFileTransfer(existingTransfer, existingTransfer.Received, existingTransfer.LastSeen);
235  }
236  return;
237  }
238 
239  if (!ValidateInitialData(fileType, fileName, fileSize, out string errorMsg))
240  {
241  GameMain.Client.CancelFileTransfer(transferId);
242  DebugConsole.ThrowError("File transfer failed (" + errorMsg + ")");
243  return;
244  }
245 
246  if (GameSettings.CurrentConfig.VerboseLogging)
247  {
248  DebugConsole.Log("Received file transfer initiation message: ");
249  DebugConsole.Log(" File: " + fileName);
250  DebugConsole.Log(" Size: " + fileSize);
251  DebugConsole.Log(" ID: " + transferId);
252  }
253 
254  string downloadFolder = downloadFolders[(FileTransferType)fileType];
255  if (!Directory.Exists(downloadFolder))
256  {
257  try
258  {
259  Directory.CreateDirectory(downloadFolder);
260  }
261  catch (Exception e)
262  {
263  DebugConsole.ThrowError("Could not start a file transfer: failed to create the folder \"" + downloadFolder + "\".", e);
264  return;
265  }
266  }
267 
268  FileTransferIn newTransfer = new FileTransferIn(inc.Sender, Path.Combine(downloadFolder, fileName), (FileTransferType)fileType)
269  {
270  ID = transferId,
271  Status = FileTransferStatus.Receiving,
272  FileSize = fileSize
273  };
274 
275  int maxRetries = 4;
276  for (int i = 0; i <= maxRetries; i++)
277  {
278  try
279  {
280  newTransfer.OpenStream();
281  }
282  catch (System.IO.IOException e)
283  {
284  if (i < maxRetries)
285  {
286  DebugConsole.NewMessage("Failed to initiate a file transfer {" + e.Message + "}, retrying in 250 ms...", Color.Red);
287  Thread.Sleep(250);
288  }
289  else
290  {
291  DebugConsole.NewMessage("Failed to initiate a file transfer {" + e.Message + "}", Color.Red);
292  GameMain.Client.CancelFileTransfer(transferId);
293  newTransfer.Status = FileTransferStatus.Error;
294  OnTransferFailed(newTransfer);
295  return;
296  }
297  }
298  }
299  activeTransfers.Add(newTransfer);
300 
301  GameMain.Client.UpdateFileTransfer(newTransfer, 0, 0); //send acknowledgement packet
302  }
303  break;
304  case (byte)FileTransferMessageType.TransferOnSameMachine:
305  {
306  byte transferId = inc.ReadByte();
307  byte fileType = inc.ReadByte();
308  string filePath = inc.ReadString();
309 
310  if (GameSettings.CurrentConfig.VerboseLogging)
311  {
312  DebugConsole.Log("Received file transfer message on the same machine: ");
313  DebugConsole.Log(" File: " + filePath);
314  DebugConsole.Log(" ID: " + transferId);
315  }
316 
317  if (!File.Exists(filePath))
318  {
319  DebugConsole.ThrowError("File transfer on the same machine failed, file \"" + filePath + "\" not found.");
320  GameMain.Client.CancelFileTransfer(transferId);
321  return;
322  }
323 
324  FileTransferIn directTransfer = new FileTransferIn(inc.Sender, filePath, (FileTransferType)fileType)
325  {
326  ID = transferId,
327  Status = FileTransferStatus.Finished,
328  FileSize = 0
329  };
330 
331  Md5Hash.Cache.Remove(directTransfer.FilePath);
332  OnFinished(directTransfer);
333  }
334  break;
335  case (byte)FileTransferMessageType.Data:
336  {
337  byte transferId = inc.ReadByte();
338 
339  var activeTransfer = activeTransfers.Find(t => t.Connection.EndpointMatches(t.Connection.EndPointString) && t.ID == transferId);
340  if (activeTransfer == null)
341  {
342  //it's possible for the server to send some extra data
343  //before it acknowledges that the download is finished,
344  //so let's suppress the error message in that case
345  finishedTransfers.RemoveAll(t => t.finishedTime + 5.0 < Timing.TotalTime);
346  if (!finishedTransfers.Any(t => t.transferId == transferId))
347  {
348  GameMain.Client.CancelFileTransfer(transferId);
349  DebugConsole.AddWarning("File transfer error: received data without a transfer initiation message");
350  }
351  return;
352  }
353 
354  int offset = inc.ReadInt32();
355  int bytesToRead = inc.ReadUInt16();
356  if (offset != activeTransfer.Received)
357  {
358  activeTransfer.LastSeen = Math.Max(offset, activeTransfer.LastSeen);
359  DebugConsole.Log($"Received {bytesToRead} bytes of the file {activeTransfer.FileName} (ignoring: offset {offset}, waiting for {activeTransfer.Received})");
360  GameMain.Client.UpdateFileTransfer(activeTransfer, activeTransfer.Received, activeTransfer.LastSeen);
361  return;
362  }
363  activeTransfer.LastSeen = offset;
364 
365  if (activeTransfer.Received + bytesToRead > activeTransfer.FileSize)
366  {
367  GameMain.Client.CancelFileTransfer(transferId);
368  DebugConsole.ThrowError("File transfer error: Received more data than expected (total received: " + activeTransfer.Received +
369  ", msg received: " + (inc.LengthBytes - inc.BytePosition) +
370  ", msg length: " + inc.LengthBytes +
371  ", msg read: " + inc.BytePosition +
372  ", filesize: " + activeTransfer.FileSize);
373  activeTransfer.Status = FileTransferStatus.Error;
374  StopTransfer(activeTransfer);
375  return;
376  }
377 
378  try
379  {
380  activeTransfer.ReadBytes(inc, bytesToRead);
381  }
382  catch (Exception e)
383  {
384  GameMain.Client.CancelFileTransfer(transferId);
385  DebugConsole.ThrowError("File transfer error: " + e.Message);
386  activeTransfer.Status = FileTransferStatus.Error;
387  StopTransfer(activeTransfer, true);
388  return;
389  }
390 
391  GameMain.Client.UpdateFileTransfer(activeTransfer, activeTransfer.Received, activeTransfer.LastSeen, reliable: activeTransfer.Status == FileTransferStatus.Finished);
392  if (activeTransfer.Status == FileTransferStatus.Finished)
393  {
394  activeTransfer.Dispose();
395 
396  if (ValidateReceivedData(activeTransfer, out string errorMessage))
397  {
398  finishedTransfers.Add((transferId, Timing.TotalTime));
399  StopTransfer(activeTransfer);
400  Md5Hash.Cache.Remove(activeTransfer.FilePath);
401  OnFinished(activeTransfer);
402  }
403  else
404  {
405  new GUIMessageBox("File transfer aborted", errorMessage);
406 
407  activeTransfer.Status = FileTransferStatus.Error;
408  StopTransfer(activeTransfer, true);
409  }
410  }
411  }
412  break;
413  case (byte)FileTransferMessageType.Cancel:
414  {
415  byte transferId = inc.ReadByte();
416  var matchingTransfer = activeTransfers.Find(t => t.Connection.EndpointMatches(t.Connection.EndPointString) && t.ID == transferId);
417  if (matchingTransfer != null)
418  {
419  new GUIMessageBox("File transfer cancelled", "The server has cancelled the transfer of the file \"" + matchingTransfer.FileName + "\".");
420  StopTransfer(matchingTransfer);
421  }
422  break;
423  }
424  }

427  private bool ValidateInitialData(byte type, string fileName, int fileSize, out string errorMessage)
428  {
429  errorMessage = "";
430 
431  if (!Enum.IsDefined(typeof(FileTransferType), (int)type))
432  {
433  errorMessage = "Unknown file type";
434  return false;
435  }
436 
437  if (fileSize > GetMaxFileSizeInBytes((FileTransferType)type))
438  {
439  errorMessage = $"File too large ({MathUtils.GetBytesReadable(fileSize)} > {MathUtils.GetBytesReadable(GetMaxFileSizeInBytes((FileTransferType)type))})";
440  return false;
441  }
442 
443  if (string.IsNullOrEmpty(fileName) ||
444  fileName.IndexOfAny(Path.GetInvalidFileNameChars().ToArray()) > -1)
445  {
446  errorMessage = "Illegal characters in file name ''" + fileName + "''";
447  return false;
448  }
449 
450  switch (type)
451  {
452  case (byte)FileTransferType.Submarine:
453  if (Path.GetExtension(fileName) != ".sub")
454  {
455  errorMessage = "Wrong file extension ''" + Path.GetExtension(fileName) + "''! (Expected .sub)";
456  return false;
457  }
458  break;
459  case (byte)FileTransferType.CampaignSave:
460  if (Path.GetExtension(fileName) != ".save")
461  {
462  errorMessage = "Wrong file extension ''" + Path.GetExtension(fileName) + "''! (Expected .save)";
463  return false;
464  }
465  break;
466  }
467 
468  return true;

471  private bool ValidateReceivedData(FileTransferIn fileTransfer, out string ErrorMessage)
472  {
473  ErrorMessage = "";
474  switch (fileTransfer.FileType)
475  {
476  case FileTransferType.Submarine:
477  System.IO.Stream stream;
478  try
479  {
480  stream = SaveUtil.DecompressFiletoStream(fileTransfer.FilePath);
481  }
482  catch (Exception e)
483  {
484  ErrorMessage = "Loading received submarine \"" + fileTransfer.FileName + "\" failed! {" + e.Message + "}";
485  return false;
486  }
487 
488  if (stream == null)
489  {
490  ErrorMessage = "Decompressing received submarine file \"" + fileTransfer.FilePath + "\" failed!";
491  return false;
492  }
493 
494  try
495  {
496  stream.Position = 0;
497 
498  XmlReaderSettings settings = new XmlReaderSettings
499  {
500  DtdProcessing = DtdProcessing.Prohibit,
501  IgnoreProcessingInstructions = true
502  };
503 
504  using (var reader = XmlReader.Create(stream, settings))
505  {
506  while (reader.Read());
507  }
508  }
509  catch
510  {
511  stream?.Close();
512  ErrorMessage = "Parsing file \"" + fileTransfer.FilePath + "\" failed! The file may not be a valid submarine file.";
513  return false;
514  }
515 
516  stream?.Close();
517  break;
518  case FileTransferType.CampaignSave:
519  try
520  {
521  var files = SaveUtil.EnumerateContainedFiles(fileTransfer.FilePath);
522  foreach (var file in files)
523  {
524  string extension = Path.GetExtension(file);
525  if ((!extension.Equals(".sub", StringComparison.OrdinalIgnoreCase)
526  && !file.Equals("gamesession.xml"))
527  || file.CleanUpPathCrossPlatform(correctFilenameCase: false).Contains('/'))
528  {
529  ErrorMessage = $"Found unexpected file in \"{fileTransfer.FileName}\"! ({file})";
530  return false;
531  }
532  }
533  }
534  catch (Exception e)
535  {
536  ErrorMessage = $"Loading received campaign save \"{fileTransfer.FileName}\" failed! {{{e.Message}}}";
537  return false;
538  }
539  break;
540  }
541 
542  return true;
