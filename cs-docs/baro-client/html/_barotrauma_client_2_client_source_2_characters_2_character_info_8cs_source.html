514 public static CharacterInfo ClientRead(Identifier speciesName, IReadMessage inc)
515 {
516 ushort infoID = inc.ReadUInt16();
517 string newName = inc.ReadString();
518 string originalName = inc.ReadString();
519 int tagCount = inc.ReadByte();
520 HashSet<Identifier> tagSet = new HashSet<Identifier>();
521 for (int i = 0; i < tagCount; i++)
522 {
523 tagSet.Add(inc.ReadIdentifier());
524 }
525 int hairIndex = inc.ReadByte();
526 int beardIndex = inc.ReadByte();
527 int moustacheIndex = inc.ReadByte();
528 int faceAttachmentIndex = inc.ReadByte();
529 Color skinColor = inc.ReadColorR8G8B8();
530 Color hairColor = inc.ReadColorR8G8B8();
531 Color facialHairColor = inc.ReadColorR8G8B8();
532 string ragdollFile = inc.ReadString();
533
534 string jobIdentifier = inc.ReadString();
535 int variant = inc.ReadByte();
536
537 JobPrefab jobPrefab = null;
538 Dictionary<Identifier, float> skillLevels = new Dictionary<Identifier, float>();
539 if (!string.IsNullOrEmpty(jobIdentifier))
540 {
541 jobPrefab = JobPrefab.Get(jobIdentifier);
542 byte skillCount = inc.ReadByte();
543 for (int i = 0; i < skillCount; i++)
544 {
545 Identifier skillIdentifier = inc.ReadIdentifier();
546 float skillLevel = inc.ReadSingle();
547 skillLevels.Add(skillIdentifier, skillLevel);
548 }
549 }
550
551 // TODO: animations
552 CharacterInfo ch = new CharacterInfo(speciesName, newName, originalName, jobPrefab, ragdollFile, variant)
553 {
554 ID = infoID,
555 };
556 ch.RecreateHead(tagSet.ToImmutableHashSet(), hairIndex, beardIndex, moustacheIndex, faceAttachmentIndex);
557 ch.Head.SkinColor = skinColor;
558 ch.Head.HairColor = hairColor;
559 ch.Head.FacialHairColor = facialHairColor;
560 ch.SetPersonalityTrait();
561 if (ch.Job != null)
562 {
563 ch.Job.OverrideSkills(skillLevels);
564 }
565
566 ch.ExperiencePoints = inc.ReadUInt16();
567 ch.AdditionalTalentPoints = inc.ReadRangedInteger(0, MaxAdditionalTalentPoints);
568 return ch;
569 }
