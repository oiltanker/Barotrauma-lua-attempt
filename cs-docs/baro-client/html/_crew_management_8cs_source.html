210 private void UpdateLocationView(Location location, bool removePending, Location prevLocation = null)
211 {
212 if (prevLocation != null && prevLocation == location && GameMain.NetworkMember != null) { return; }
213
214 if (characterPreviewFrame != null)
215 {
216 characterPreviewFrame.Parent?.RemoveChild(characterPreviewFrame);
217 characterPreviewFrame = null;
218 }
219 UpdateHireables(location);
220 if (pendingList != null)
221 {
222 if (removePending)
223 {
224 PendingHires?.Clear();
225 pendingList.Content.ClearChildren();
226 }
227 else
228 {
229 PendingHires?.ForEach(ci => AddPendingHire(ci));
230 }
231 SetTotalHireCost();
232 }
233 UpdateCrew();
234 }

249 else

252 {
253 if (c == null) { continue; }

255 }
256 }
257 sortingDropDown.SelectItem(SortingMethod.JobAsc);
258 hireableList.UpdateScrollBarSize();
259 }

262 public void SetHireables(Location location, List<CharacterInfo> availableHires)
263 {
264 HireManager hireManager = location.HireManager;
265 if (hireManager == null) { return; }
266 int hireVal = hireManager.AvailableCharacters.Aggregate(0, (curr, hire) => curr + hire.GetIdentifier());
267 int newVal = availableHires.Aggregate(0, (curr, hire) => curr + hire.GetIdentifier());
268 if (hireVal != newVal)
269 {
270 location.HireManager.AvailableCharacters = availableHires;
271 UpdateHireables(location);
272 }
273 }

287 private void SortCharacters(GUIListBox list, SortingMethod sortingMethod)
288 {
289 if (sortingMethod == SortingMethod.AlphabeticalAsc)
290 {
291 list.Content.RectTransform.SortChildren((x, y) =>
292 ((InfoSkill)x.GUIComponent.UserData).CharacterInfo.Name.CompareTo(((InfoSkill)y.GUIComponent.UserData).CharacterInfo.Name));
293 }
294 else if (sortingMethod == SortingMethod.JobAsc)
295 {
296 SortCharacters(list, SortingMethod.AlphabeticalAsc);
297 list.Content.RectTransform.SortChildren((x, y) =>
298 String.Compare(((InfoSkill)x.GUIComponent.UserData).CharacterInfo.Job.Name.Value, ((InfoSkill)y.GUIComponent.UserData).CharacterInfo.Job.Name.Value, StringComparison.Ordinal));
299 }
300 else if (sortingMethod == SortingMethod.PriceAsc || sortingMethod == SortingMethod.PriceDesc)
301 {
302 SortCharacters(list, SortingMethod.AlphabeticalAsc);
303 list.Content.RectTransform.SortChildren((x, y) =>
304 ((InfoSkill)x.GUIComponent.UserData).CharacterInfo.Salary.CompareTo(((InfoSkill)y.GUIComponent.UserData).CharacterInfo.Salary));
305 if (sortingMethod == SortingMethod.PriceDesc) { list.Content.RectTransform.ReverseChildren(); }
306 }
307 else if (sortingMethod == SortingMethod.SkillAsc || sortingMethod == SortingMethod.SkillDesc)
308 {
309 SortCharacters(list, SortingMethod.AlphabeticalAsc);
310 list.Content.RectTransform.SortChildren((x, y) =>
311 ((InfoSkill)x.GUIComponent.UserData).SkillLevel.CompareTo(((InfoSkill)y.GUIComponent.UserData).SkillLevel));
312 if (sortingMethod == SortingMethod.SkillDesc) { list.Content.RectTransform.ReverseChildren(); }
313 }

626 {

881 {

883 {

886

894 }

897 msg.Write(validateHires);

900 msg.Write(validRenaming);
901 if (validRenaming)

903 int identifier = renameCharacter.info.GetIdentifierUsingOriginalName();
904 msg.Write(identifier);
905 msg.Write(renameCharacter.newName);
906 bool existingCrewMember = campaign.CrewManager?.GetCharacterInfos().Any(ci => ci.GetIdentifierUsingOriginalName() == identifier) ?? false;
907 msg.Write(existingCrewMember);
908 }
909
910 msg.Write(firedCharacter != null);
911 if (firedCharacter != null)
912 {
913 msg.Write(firedCharacter.GetIdentifier());
914 }
915
916 GameMain.Client.ClientPeer?.Send(msg, DeliveryMethod.Reliable);
917 }
