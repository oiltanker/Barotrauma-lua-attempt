110 internal class AddOrDeleteCommand : Command
111 {
112 private readonly Dictionary<InventorySlotItem, Inventory> PreviousInventories = new Dictionary<InventorySlotItem, Inventory>();
113 private readonly List<MapEntity> Receivers;
114 private readonly List<MapEntity> CloneList;
115 private readonly bool WasDeleted;
116 private readonly List<AddOrDeleteCommand> ContainedItemsCommand = new List<AddOrDeleteCommand>();
117
124 public AddOrDeleteCommand(List<MapEntity> receivers, bool wasDeleted, bool handleInventoryBehavior = true)
125 {
126 Debug.Assert(receivers.Count > 0, "Command has 0 receivers");
127 WasDeleted = wasDeleted;
128 Receivers = new List<MapEntity>(receivers);
129
130 try
131 {
132 foreach (MapEntity receiver in receivers)
133 {
134 if (receiver is Item it && it.ParentInventory != null)
135 {
136 PreviousInventories.Add(new InventorySlotItem(it.ParentInventory.FindIndex(it), it), it.ParentInventory);
137 }
138 }
139
140 List<MapEntity> clonedTargets = MapEntity.Clone(receivers);
141
142 List<MapEntity> itemsToDelete = new List<MapEntity>();
143 foreach (MapEntity receiver in Receivers)
144 {
145 if (receiver is Item it)
146 {
147 foreach (ItemContainer component in it.GetComponents<ItemContainer>())
148 {
149 if (component.Inventory == null) { continue; }
150 itemsToDelete.AddRange(component.Inventory.AllItems.Where(item => !item.Removed));
151 }
152 }
153 }
154
155 if (itemsToDelete.Any() && handleInventoryBehavior)
156 {
157 ContainedItemsCommand.Add(new AddOrDeleteCommand(itemsToDelete, wasDeleted));
158 if (wasDeleted)
159 {
160 foreach (MapEntity item in itemsToDelete)
161 {
162 if (item != null && !item.Removed)
163 {
164 item.Remove();
165 }
166 }
167 }
168 }
169
170 foreach (MapEntity clone in clonedTargets)
171 {
172 clone.ShallowRemove();
173 if (clone is Item it)
174 {
175 foreach (ItemContainer container in it.GetComponents<ItemContainer>())
176 {
177 container.Inventory?.DeleteAllItems();
178 }
179 }
180 }
181
182 CloneList = clonedTargets;
183 }
184 // This should never happen except if we decide to make a new type of MapEntity that isn't finished yet
185 catch (Exception e)
186 {
187 Receivers = new List<MapEntity>();
188 CloneList = new List<MapEntity>();
189 DebugConsole.ThrowError("Could not store object", e);
190 }
191 }
192
193 public override void Execute()
194 {
195 DeleteUndelete(true);
196 ContainedItemsCommand?.ForEach(cmd => cmd.Execute());
197 }
198
199 public override void UnExecute()
200 {
201 DeleteUndelete(false);
202 ContainedItemsCommand?.ForEach(cmd => cmd.UnExecute());
203 }
204
205 public override void Cleanup()
206 {
207 foreach (MapEntity entity in CloneList)
208 {
209 if (!entity.Removed)
210 {
211 entity.Remove();
212 }
213 }
214
215 CloneList?.Clear();
216 Receivers.Clear();
217 PreviousInventories?.Clear();
218 ContainedItemsCommand?.ForEach(cmd => cmd.Cleanup());
219 }
220
221 private void DeleteUndelete(bool redo)
222 {
223 bool wasDeleted = WasDeleted;
224
225 // We are redoing instead of undoing, flip the behavior
226 if (redo) { wasDeleted = !wasDeleted; }
227
228 if (wasDeleted)
229 {
230 Debug.Assert(Receivers.All(entity => entity.GetReplacementOrThis().Removed), "Tried to redo a deletion but some items were not deleted");
231
232 List<MapEntity> clones = MapEntity.Clone(CloneList);
233 int length = Math.Min(Receivers.Count, clones.Count);
234 for (int i = 0; i < length; i++)
235 {
236 MapEntity clone = clones[i], receiver = Receivers[i];
237
238 if (receiver.GetReplacementOrThis() is Item item && clone is Item cloneItem)
239 {
240 foreach (ItemComponent ic in item.Components)
241 {
242 int index = item.GetComponentIndex(ic);
243 ItemComponent component = cloneItem.Components.ElementAtOrDefault(index);
244 switch (component)
245 {
246 case null:
247 continue;
248 case ItemContainer newContainer when newContainer.Inventory != null && ic is ItemContainer itemContainer && itemContainer.Inventory != null:
249 itemContainer.Inventory.GetReplacementOrThiS().ReplacedBy = newContainer.Inventory;
250 goto default;
251 default:
252 ic.GetReplacementOrThis().ReplacedBy = component;
253 break;
254 }
255 }
256 }
257
258 receiver.GetReplacementOrThis().ReplacedBy = clone;
259 }
260
261 for (int i = 0; i < length; i++)
262 {
263 MapEntity clone = clones[i], receiver = Receivers[i];
264
265 if (clone is Item it)
266 {
267 foreach (var (slotRef, inventory) in PreviousInventories)
268 {
269 if (slotRef.Item == receiver)
270 {
271 inventory.GetReplacementOrThiS().TryPutItem(it, slotRef.Slot, false, false, null, createNetworkEvent: false);
272 }
273 }
274 }
275 }
276
277 foreach (MapEntity clone in clones)
278 {
279 clone.Submarine = Submarine.MainSub;
280 }
281 }
282 else
283 {
284 foreach (MapEntity t in Receivers)
285 {
286 MapEntity receiver = t.GetReplacementOrThis();
287 if (!receiver.Removed)
288 {
289 receiver.Remove();
290 }
291 }
292 }
293 }

297 master.Receivers.AddRange(Receivers);

347 private void ContainUncontain(bool drop)
348 {
349 // flip the behavior if the item was dropped instead of inserted
350 if (wasDropped) { drop = !drop; }
351
352 foreach (var (slot, receiver) in Receivers)
353 {
354 Item item = (Item) receiver.GetReplacementOrThis();
355
356 if (drop)
357 {
358 item.Drop(null, createNetworkEvent: false);
359 }
360 else
361 {
362 Inventory.GetReplacementOrThiS().TryPutItem(item, slot, false, false, null, createNetworkEvent: false);
363 }
364 }
365 }
366
367 public override LocalizedString GetDescription()
368 {
369 if (wasDropped)
370 {
371 return TextManager.GetWithVariable("Undo.DroppedItem", "[item]", Receivers.FirstOrDefault().Item.Name);
372 }
373
374 string container = "[ERROR]";
375
376 if (Inventory.Owner is Item item)
377 {
378 container = item.Name;
379 }
380
381 return Receivers.Count > 1
382 ? TextManager.GetWithVariables("Undo.ContainedItemsMultiple", ("[count]", Receivers.Count.ToString()), ("[container]", container))
383 : TextManager.GetWithVariables("Undo.ContainedItem", ("[item]", Receivers.FirstOrDefault().Item.Name), ("[container]", container));

434 private void SanitizeProperty()
435 {
436 sanitizedProperty = NewProperties switch
437 {
438 float f => f.FormatSingleDecimal(),
439 Point point => XMLExtensions.PointToString(point),
440 Vector2 vector2 => vector2.FormatZeroDecimal(),
441 Vector3 vector3 => vector3.FormatSingleDecimal(),
442 Vector4 vector4 => vector4.FormatSingleDecimal(),
443 Color color => XMLExtensions.ColorToString(color),
444 Rectangle rectangle => XMLExtensions.RectToString(rectangle),
445 _ => NewProperties.ToString()
446 };
447 }
448
449 public override void Execute() => SetProperties(false);
450 public override void UnExecute() => SetProperties(true);
451
452 public override void Cleanup()
453 {
454 Receivers.Clear();
455 OldProperties.Clear();
456 }

458 private void SetProperties(bool undo)
459 {
460 foreach (ISerializableEntity t in Receivers)
461 {
462 ISerializableEntity receiver;
463 switch (t)
464 {
465 case MapEntity me when me.GetReplacementOrThis() is ISerializableEntity sEntity:
466 receiver = sEntity;
467 break;
468 case ItemComponent ic when ic.GetReplacementOrThis() is ISerializableEntity sItemComponent:
469 receiver = sItemComponent;
470 break;
471 default:
472 receiver = t;
473 break;
474 }
475
476 object data = NewProperties;
477
478 if (undo)
479 {
480 foreach (var (key, value) in OldProperties)
481 {
482 if (value.Contains(t)) { data = key; }
483 }
484 }
485
486 if (receiver.SerializableProperties != null)
487 {
488 Dictionary<Identifier, SerializableProperty> props = receiver.SerializableProperties;
489
490 if (props.TryGetValue(PropertyName, out SerializableProperty prop))
491 {
492 prop.TrySetValue(receiver, data);
493 // Update the editing hud
494 if (MapEntity.EditingHUD == null || (MapEntity.EditingHUD.UserData != receiver && (receiver is ItemComponent ic && MapEntity.EditingHUD.UserData != ic.Item))) { continue; }
495
496 GUIListBox list = MapEntity.EditingHUD.GetChild<GUIListBox>();
497 if (list == null) { continue; }
498
499 IEnumerable<SerializableEntityEditor> editors = list.Content.FindChildren(comp => comp is SerializableEntityEditor).Cast<SerializableEntityEditor>();
500 SerializableEntityEditor.LockEditing = true;
501 foreach (SerializableEntityEditor editor in editors)
502 {
503 if (editor.UserData == receiver && editor.Fields.TryGetValue(PropertyName, out GUIComponent[] _))
504 {
505 editor.UpdateValue(prop, data);
506 }
507 }
508
509 SerializableEntityEditor.LockEditing = false;
510 }
511 }
512 }
513 }

520 ("[count]", Receivers.Count.ToString()),
521 ("[value]", sanitizedProperty))
522 : TextManager.GetWithVariables("Undo.ChangedProperty",
523 ("[property]", PropertyName.Value),
524 ("[item]", Receivers.FirstOrDefault()?.Name),
525 ("[value]", sanitizedProperty));
