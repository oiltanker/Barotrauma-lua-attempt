133 public static XElement CreateVariantXml(XElement variantXML, XElement baseXML)
134 {
135 XElement newXml = variantXML.CreateVariantXML(baseXML);
136 XElement variantAi = variantXML.GetChildElement("ai");
137 XElement baseAi = baseXML.GetChildElement("ai");
138 if (baseAi is null || baseAi.Elements().None()
139 || variantAi is null || variantAi.Elements().None())
140 {
141 return newXml;
142 }
143 // CreateVariantXML seems to merge the ai targets so that in the new xml we have both the old and the new target definitions.
144 var finalAiElement = newXml.GetChildElement("ai");
145 var processedTags = new HashSet<string>();
146 foreach (var aiTarget in finalAiElement.Elements().ToArray())
147 {
148 string tag = aiTarget.GetAttributeString("tag", null);
149 if (tag == null) { continue; }
150 if (processedTags.Contains(tag))
151 {
152 aiTarget.Remove();
153 continue;
154 }
155 processedTags.Add(tag);
156 var matchInSelf = variantAi.Elements().FirstOrDefault(e => e.GetAttributeString("tag", null) == tag);
157 var matchInParent = baseAi.Elements().FirstOrDefault(e => e.GetAttributeString("tag", null) == tag);
158 if (matchInSelf != null && matchInParent != null)
159 {
160 aiTarget.ReplaceWith(new XElement(matchInSelf));
161 }
162 }
163 return newXml;
164 }

167 {
168 UpdatePath(File.Path);
169 doc = XMLExtensions.TryLoadXml(Path);

172 {

178 }
179 IsLoaded = Deserialize(MainElement);

181 if (SpeciesName.IsEmpty && MainElement != null)

344 protected bool RemoveSubParam<T>(T subParam, IList<T> collection = null) where T : SubParam
345 {
346 if (subParam == null || subParam.Element == null || subParam.Element.Parent == null) { return false; }
347 if (collection != null && !collection.Contains(subParam)) { return false; }
348 if (!SubParams.Contains(subParam)) { return false; }
349 collection?.Remove(subParam);
350 SubParams.Remove(subParam);
351 subParam.Element.Remove();
352 return true;
353 }
354
355 protected bool TryAddSubParam<T>(ContentXElement element, Func<ContentXElement, CharacterParams, T> constructor, out T subParam, IList<T> collection = null, Func<IList<T>, bool> filter = null) where T : SubParam
356 {
357 subParam = constructor(element, this);
358 if (collection != null && filter != null)
359 {
360 if (filter(collection)) { return false; }
361 }
362 MainElement.Add(element);
363 SubParams.Add(subParam);
364 collection?.Add(subParam);
365 return subParam != null;
366 }
