142  public override void ReceiveSignal(Signal signal, Connection connection)
143  {
144  Item source = signal.source;
145  if (source == null || source.CurrentHull == null) { return; }
146 
147  Hull sourceHull = source.CurrentHull;
148  if (!hullDatas.TryGetValue(sourceHull, out HullData hullData))
149  {
150  hullData = new HullData();
151  hullDatas.Add(sourceHull, hullData);
152  }
153 
154  if (hullData.Distort) { return; }
155 
156  switch (connection.Name)
157  {
158  case "water_data_in":
159  //cheating a bit because water detectors don't actually send the water level
160  bool fromWaterDetector = source.GetComponent<WaterDetector>() != null;
161  hullData.ReceivedWaterAmount = null;
162  if (fromWaterDetector)
163  {
164  hullData.ReceivedWaterAmount = Math.Min(sourceHull.WaterVolume / sourceHull.Volume, 1.0f);
165  }
166  foreach (var linked in sourceHull.linkedTo)
167  {
168  if (!(linked is Hull linkedHull)) { continue; }
169  if (!hullDatas.TryGetValue(linkedHull, out HullData linkedHullData))
170  {
171  linkedHullData = new HullData();
172  hullDatas.Add(linkedHull, linkedHullData);
173  }
174  linkedHullData.ReceivedWaterAmount = null;
175  if (fromWaterDetector)
176  {
177  linkedHullData.ReceivedWaterAmount = Math.Min(linkedHull.WaterVolume / linkedHull.Volume, 1.0f);
178  }
179  }
180  break;
181  case "oxygen_data_in":
182  if (!float.TryParse(signal.value, System.Globalization.NumberStyles.Float, System.Globalization.CultureInfo.InvariantCulture, out float oxy))
183  {
184  oxy = Rand.Range(0.0f, 100.0f);
185  }
186  hullData.ReceivedOxygenAmount = oxy;
187  foreach (var linked in sourceHull.linkedTo)
188  {
189  if (!(linked is Hull linkedHull)) { continue; }
190  if (!hullDatas.TryGetValue(linkedHull, out HullData linkedHullData))
191  {
192  linkedHullData = new HullData();
193  hullDatas.Add(linkedHull, linkedHullData);
194  }
195  linkedHullData.ReceivedOxygenAmount = oxy;
196  }
197  break;
198  }
199  }
