117  public static string IncrementModVersion(string modVersion)
118  {
119  //look for an integer at the end of the string and increment it
120  int startIndex = modVersion.Length - 1;
121  while (char.IsDigit(modVersion[startIndex])) { startIndex--; }
122  startIndex++;
123 
124  if (startIndex >= modVersion.Length
125  || !char.IsDigit(modVersion[startIndex])
126  || !int.TryParse(
127  modVersion[startIndex..],
128  NumberStyles.Any,
129  CultureInfo.InvariantCulture,
130  out int theFinalInteger))
131  {
132  return modVersion;
133  }
134 
135  return $"{modVersion[..startIndex]}{(theFinalInteger + 1).ToString(CultureInfo.InvariantCulture)}";
136  }
