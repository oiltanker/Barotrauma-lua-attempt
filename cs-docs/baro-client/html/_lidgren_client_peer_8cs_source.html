154  public override void SendPassword(string password)
155  {

157 
158  if (initializationStep != ConnectionInitialization.Password) { return; }
159  NetOutgoingMessage outMsg = netClient.CreateMessage();
160  outMsg.Write((byte)PacketHeader.IsConnectionInitializationStep);
161  outMsg.Write((byte)ConnectionInitialization.Password);
162  byte[] saltedPw = ServerSettings.SaltPassword(Encoding.UTF8.GetBytes(password), passwordSalt);
163  outMsg.Write((byte)saltedPw.Length);
164  outMsg.Write(saltedPw, 0, saltedPw.Length);
165  NetSendResult result = netClient.SendMessage(outMsg, NetDeliveryMethod.ReliableUnordered);
