361  private void CreateNodes(ContentXElement element, ref bool hadNodes, EditorNode? parent = null, int ident = 0)
362  {
363  EditorNode? lastNode = null;
364  foreach (var subElement in element.Elements())
365  {
366  bool skip = true;
367  switch (subElement.Name.ToString().ToLowerInvariant())
368  {
369  case "failure":
370  case "success":
371  case "option":
372  CreateNodes(subElement, ref hadNodes, parent, ident);
373  break;
374  default:
375  skip = false;
376  break;
377  }
378 
379  if (!skip)
380  {
381  Vector2 defaultNodePos = new Vector2(-16000, -16000);
382  EditorNode newNode;
383  Type? t = Type.GetType($"Barotrauma.{subElement.Name}");
384  if (t != null && EditorNode.IsInstanceOf(t, typeof(EventAction)))
385  {
386  newNode = new EventNode(t, subElement.Name.ToString()) { Position = new Vector2(ident, 0), ID = CreateID() };
387  }
388  else
389  {
390  newNode = new CustomNode(subElement.Name.ToString()) { Position = new Vector2(ident, 0), ID = CreateID() };
391  foreach (XAttribute attribute in subElement.Attributes().Where(attribute => !attribute.ToString().StartsWith("_")))
392  {
393  newNode.Connections.Add(new NodeConnection(newNode, NodeConnectionType.Value, attribute.Name.ToString(), typeof(string)));
394  }
395  }
396 
397  Vector2 npos = subElement.GetAttributeVector2("_npos", defaultNodePos);
398  if (npos != defaultNodePos)
399  {
400  newNode.Position = npos;
401  }
402  else
403  {
404  hadNodes = false;
405  }
406 
407  var parentElement = subElement.Parent;
408 
409  foreach (var xElement in subElement.Elements())
410  {
411  if (xElement.Name.ToString().ToLowerInvariant() == "option")
412  {
413  NodeConnection optionConnection = new NodeConnection(newNode, NodeConnectionType.Option)
414  {
415  OptionText = xElement.GetAttributeString("text", string.Empty),
416  EndConversation = xElement.GetAttributeBool("endconversation", false)
417  };
418  newNode.Connections.Add(optionConnection);
419  }
420  }
421 
422  foreach (NodeConnection connection in newNode.Connections)
423  {
424  if (connection.Type == NodeConnectionType.Value)
425  {
426  foreach (XAttribute attribute in subElement.Attributes())
427  {
428  if (string.Equals(connection.Attribute, attribute.Name.ToString(), StringComparison.InvariantCultureIgnoreCase) && connection.ValueType != null)
429  {
430  if (connection.ValueType.IsEnum)
431  {
432  Array values = Enum.GetValues(connection.ValueType);
433  foreach (object? @enum in values)
434  {
435  if (string.Equals(@enum?.ToString(), attribute.Value, StringComparison.InvariantCultureIgnoreCase))
436  {
437  connection.OverrideValue = @enum;
438  }
439  }
440  }
441  else
442  {
443  connection.OverrideValue = ChangeType(attribute.Value, connection.ValueType);
444  }
445  }
446  }
447  }
448  }
449 
450  if (npos == defaultNodePos)
451  {
452  hadNodes = false;
453  bool Predicate(EditorNode node) => Rectangle.Union(node.GetDrawRectangle(), node.HeaderRectangle).Intersects(Rectangle.Union(newNode.GetDrawRectangle(), newNode.HeaderRectangle));
454 
455  while (nodeList.Any(Predicate))
456  {
457  EditorNode? otherNode = nodeList.Find(Predicate);
458  if (otherNode != null)
459  {
460  newNode.Position += new Vector2(128, otherNode.GetDrawRectangle().Height + otherNode.HeaderRectangle.Height + new Random().Next(128, 256));
461  }
462  }
463  }
464 
465  if (parentElement?.FirstElement() == subElement)
466  {
467  switch (parentElement?.Name.ToString().ToLowerInvariant())
468  {
469  case "failure":
470  parent?.Connect(newNode, NodeConnectionType.Failure);
471  break;
472  case "success":
473  parent?.Connect(newNode, NodeConnectionType.Success);
474  break;
475  case "option":
476  if (parent != null)
477  {
478  NodeConnection? activateConnection = newNode.Connections.Find(connection => connection.Type == NodeConnectionType.Activate);
479  NodeConnection? optionConnection = parent.Connections.FirstOrDefault(connection =>
480  connection.Type == NodeConnectionType.Option && string.Equals(connection.OptionText, parentElement.GetAttributeString("text", string.Empty), StringComparison.Ordinal));
481 
482  if (activateConnection != null)
483  {
484  optionConnection?.ConnectedTo.Add(activateConnection);
485  }
486  }
487  break;
488  default:
489  parent?.Connect(newNode, NodeConnectionType.Add);
490  break;
491  }
492  }
493  else
494  {
495  lastNode?.Connect(newNode, NodeConnectionType.Next);
496  }
497 
498  lastNode = newNode;
499  nodeList.Add(newNode);
500  ident += 600;
501  CreateNodes(subElement, ref hadNodes, newNode, ident);
502  }
503  else
504  {
505 
506  }
507  }
508  }

539  private XElement? ExportXML()
540  {
541  XElement mainElement = new XElement("ScriptedEvent", new XAttribute("identifier", projectName.RemoveWhitespace().ToLowerInvariant()));
542  EditorNode? startNode = null;
543  foreach (EditorNode eventNode in nodeList.Where(node => node is EventNode || node is SpecialNode))
544  {
545  if (eventNode.GetParent() == null)
546  {
547  if (startNode != null)
548  {
549  DebugConsole.ThrowError("You have more than one start node, only one will be picked while the others will get ignored.");
550  }
551  startNode ??= eventNode;
552  }
553  }
554 
555  if (startNode == null) { return null; }
556 
557  ExportChildNodes(startNode, mainElement);
558 
559  return mainElement;
560  }

562  private void ExportChildNodes(EditorNode startNode, XElement parent)
563  {
564  XElement? newElement = startNode.ToXML();
565  if (newElement == null) { return; }
566  parent.Add(newElement);
567 
568  EditorNode? success = startNode.GetNext(NodeConnectionType.Success);
569  EditorNode? failure = startNode.GetNext(NodeConnectionType.Failure);
570  EditorNode? add = startNode.GetNext(NodeConnectionType.Add);
571  Tuple<EditorNode?, string?, bool>[] options = startNode is EventNode eNode ? eNode.GetOptions() : new Tuple<EditorNode?, string?, bool>[0];
572 
573  if (success != null)
574  {
575  XElement successElement = new XElement("Success");
576  ExportChildNodes(success, successElement);
577  newElement.Add(successElement);
578  }
579 
580  if (failure != null)
581  {
582  XElement failureElement = new XElement("Failure");
583  ExportChildNodes(failure, failureElement);
584  newElement.Add(failureElement);
585  }
586 
587  if (add is CustomNode custom)
588  {
589  ExportChildNodes(custom, newElement);
590  }
591 
592  foreach (var (node, text, end) in options)
593  {
594  XElement optionElement = new XElement("Option");
595  optionElement.Add(new XAttribute("text", text ?? ""));
596  if (end) { optionElement.Add(new XAttribute("endconversation", true)); }
597 
598  if (node is EventNode eventNode)
599  {
600  ExportChildNodes(eventNode, optionElement);
601  }
602 
603  newElement.Add(optionElement);
604  }
605 
606  EditorNode? next = startNode.GetNext();
607  if (next != null)
608  {
609  ExportChildNodes(next, parent);
610  }
611  }

634  private void Load(XElement saveElement)
635  {
636  nodeList.Clear();
637  projectName = saveElement.GetAttributeString("name", TextManager.Get("EventEditor.Unnamed").Value);
638  foreach (XElement element in saveElement.Elements())
639  {
640  switch (element.Name.ToString().ToLowerInvariant())
641  {
642  case "nodes":
643  {
644  foreach (var subElement in element.Elements())
645  {
646  EditorNode? node = EditorNode.Load(subElement);
647  if (node != null)
648  {
649  nodeList.Add(node);
650  }
651  }
652 
653  break;
654  }
655  case "allconnections":
656  {
657  foreach (var subElement in element.Elements())
