178 private void FilterEmitters(string text)
179 {
180 if (string.IsNullOrWhiteSpace(text))
181 {
182 filterLabel.Visible = true;
183 prefabList.Content.Children.ForEach(c => c.Visible = true);
184 return;
185 }
186
187 text = text.ToLower();
188 filterLabel.Visible = false;
189 foreach (GUIComponent child in prefabList.Content.Children)
190 {
191 if (!(child is GUITextBlock textBlock)) { continue; }
192 textBlock.Visible = textBlock.Text.ToLower().Contains(text);
193 }

196 private void SerializeAll()
197 {
198 Barotrauma.IO.Validation.SkipValidationInDebugBuilds = true;
199 foreach (var configFile in ContentPackageManager.AllPackages.SelectMany(p => p.GetFiles<ParticlesFile>()))
200 {
201 XDocument doc = XMLExtensions.TryLoadXml(configFile.Path);
202 if (doc == null) { continue; }
203
204 var prefabList = GameMain.ParticleManager.GetPrefabList();
205 foreach (ParticlePrefab prefab in prefabList)
206 {
207 foreach (XElement element in doc.Root.Elements())
208 {
209 if (!element.Name.ToString().Equals(prefab.Name, StringComparison.OrdinalIgnoreCase)) { continue; }
210 SerializableProperty.SerializeProperties(prefab, element, true);
211 }
212 }
213
214 System.Xml.XmlWriterSettings settings = new System.Xml.XmlWriterSettings
215 {
216 Indent = true,
217 OmitXmlDeclaration = true,
218 NewLineOnAttributes = true
219 };
220
221 using (var writer = XmlWriter.Create(configFile.Path.Value, settings))
222 {
223 doc.WriteTo(writer);
224 writer.Flush();
225 }
226 }
227 Barotrauma.IO.Validation.SkipValidationInDebugBuilds = false;
228 }

256 private void SerializeToClipboard(ParticlePrefab prefab)
257 {
258 if (prefab == null) { return; }
259
260 System.Xml.XmlWriterSettings settings = new System.Xml.XmlWriterSettings
261 {
262 Indent = true,
263 OmitXmlDeclaration = true,
264 NewLineOnAttributes = true
265 };
266
267 XElement originalElement = null;
268 foreach (var configFile in ContentPackageManager.AllPackages.SelectMany(p => p.GetFiles<ParticlesFile>()))
269 {
270 XDocument doc = XMLExtensions.TryLoadXml(configFile.Path);
271 if (doc == null) { continue; }
272
273 var prefabList = GameMain.ParticleManager.GetPrefabList();
274 foreach (ParticlePrefab otherPrefab in prefabList)
275 {
276 foreach (var subElement in doc.Root.Elements())
277 {
278 if (!subElement.Name.ToString().Equals(prefab.Name, StringComparison.OrdinalIgnoreCase)) { continue; }
279 SerializableProperty.SerializeProperties(prefab, subElement, true);
280 originalElement = subElement;
281 break;
282 }
283 }
284 }
285
286 if (originalElement == null)
287 {
288 originalElement = new XElement(prefab.Name);
289 SerializableProperty.SerializeProperties(prefab, originalElement, true);
290 }
291
292 StringBuilder sb = new StringBuilder();
293 using (var writer = System.Xml.XmlWriter.Create(sb, settings))
294 {
295 originalElement.WriteTo(writer);
296 writer.Flush();
297 }
298
299 Clipboard.SetText(sb.ToString());
300 }
