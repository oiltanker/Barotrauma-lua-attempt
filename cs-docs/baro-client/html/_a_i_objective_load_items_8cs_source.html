53  public static bool IsValidTarget(Item item, Character character, ImmutableArray<Identifier>? targetContainerTags = null, ItemCondition? targetCondition = null)
54  {
55  if (item == null) { return false; }
56  if (item.Removed) { return false; }
57  if (targetContainerTags.HasValue && !OrderPrefab.TargetItemsMatchItem(targetContainerTags.Value, item)) { return false; }
58  if (!(item.GetComponent<ItemContainer>() is ItemContainer container)) { return false; }
59  if (container.Inventory == null) { return false; }
60  if (targetCondition.HasValue && container.Inventory.IsFull() && container.Inventory.AllItems.None(i => ItemMatchesTargetCondition(i, targetCondition.Value))) { return false; }
61  if (!AIObjectiveCleanupItems.IsItemInsideValidSubmarine(item, character)) { return false; }
62  if (item.GetRootInventoryOwner() is Character owner && owner != character) { return false; }
63  if (item.IsClaimedByBallastFlora) { return false; }
64  if (!item.HasAccess(character)) { return false; }
65  // Ignore items that require power but don't have it
66  if (item.GetComponent<Powered>() is Powered powered && powered.PowerConsumption > 0 && powered.Voltage < powered.MinVoltage) { return false; }
67  return true;
68  }
