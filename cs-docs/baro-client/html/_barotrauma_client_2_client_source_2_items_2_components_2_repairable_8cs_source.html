60 public override bool ShouldDrawHUD(Character character)
61 {
62 if (item.HiddenInGame) { return false; }
63 if (!HasRequiredItems(character, false) || character.SelectedConstruction != item) { return false; }
64 if (character.IsTraitor && item.ConditionPercentage > MinSabotageCondition) { return true; }
65
66 float defaultMaxCondition = item.MaxCondition / item.MaxRepairConditionMultiplier;
67
68 if (MathUtils.Percentage(item.Condition, defaultMaxCondition) < RepairThreshold) { return true; }
69
70 if (CurrentFixer == character)
71 {
72 if (item.Condition < item.MaxCondition)
73 {
74 return true;
75 }
76 }
77 if (IsTinkerable(character)) { return true; }
78
79 return false;
80 }

82 partial void InitProjSpecific(ContentXElement element)
83 {
84 CreateGUI();
85 foreach (var subElement in element.Elements())
86 {
87 switch (subElement.Name.ToString().ToLowerInvariant())
88 {
89 case "emitter":
90 case "particleemitter":
91 particleEmitters.Add(new ParticleEmitter(subElement));
92 float minCondition = subElement.GetAttributeFloat("mincondition", 0.0f);
93 float maxCondition = subElement.GetAttributeFloat("maxcondition", 100.0f);
94
95 if (maxCondition < minCondition)
96 {
97 DebugConsole.ThrowError("Invalid damage particle configuration in the Repairable component of " + item.Name + ". MaxCondition needs to be larger than MinCondition.");
98 float temp = maxCondition;
99 maxCondition = minCondition;
100 minCondition = temp;
101 }
102 particleEmitterConditionRanges.Add(new Vector2(minCondition, maxCondition));
103
104 break;
105 }
106 }

224 partial void UpdateProjSpecific(float deltaTime)
225 {
226 if (item.HiddenInGame) { return; }
227 if (FakeBrokenTimer > 0.0f)
228 {
229 item.FakeBroken = true;
230 if (Character.Controlled == null || (Character.Controlled.CharacterHealth.GetAffliction("psychosis")?.Strength ?? 0.0f) <= 0.0f)
231 {
232 FakeBrokenTimer = 0.0f;
233 }
234 else
235 {
236 FakeBrokenTimer -= deltaTime;
237 }
238 }
239 else
240 {
241 item.FakeBroken = false;
242 }
243
244
245 if (!GameMain.IsMultiplayer)
246 {
247 switch (requestStartFixAction)
248 {
249 case FixActions.Repair:
250 case FixActions.Sabotage:
251 case FixActions.Tinker:
252 StartRepairing(Character.Controlled, requestStartFixAction);
253 requestStartFixAction = FixActions.None;
254 break;
255 default:
256 requestStartFixAction = FixActions.None;
257 break;
258 }
259 }
260
261 for (int i = 0; i < particleEmitters.Count; i++)
262 {
263 if ((item.ConditionPercentage >= particleEmitterConditionRanges[i].X && item.ConditionPercentage <= particleEmitterConditionRanges[i].Y) || FakeBrokenTimer > 0.0f)
264 {
265 particleEmitters[i].Emit(deltaTime, item.WorldPosition, item.CurrentHull);
266 }
267 }
268
269 if (CurrentFixer != null && CurrentFixer.SelectedConstruction == item)
270 {
271 if (repairSoundChannel == null || !repairSoundChannel.IsPlaying)
272 {
273 repairSoundChannel = SoundPlayer.PlaySound("repair", item.WorldPosition, hullGuess: item.CurrentHull);
274 }
275
276 if (qteCooldown > 0.0f)
277 {
278 qteCooldown -= deltaTime;
279 if (qteCooldown <= 0.0f)
280 {
281 qteTimer = QteDuration;
282 }
283 }
284 else
285 {
286 qteTimer -= deltaTime * (qteTimer / QteDuration);
287 if (qteTimer < 0.0f) { qteTimer = QteDuration; }
288 }
289 }
290 else
291 {
292 repairSoundChannel?.FadeOutAndDispose();
293 repairSoundChannel = null;
294 }
295 }

297 public override void DrawHUD(SpriteBatch spriteBatch, Character character)
298 {
299 IsActive = true;
300
301 float defaultMaxCondition = (item.MaxCondition / item.MaxRepairConditionMultiplier);
302
303 progressBar.BarSize = item.Condition / defaultMaxCondition;
304 progressBar.Color = ToolBox.GradientLerp(progressBar.BarSize, GUIStyle.Red, GUIStyle.Orange, GUIStyle.Green);
305
306 Rectangle sliderRect = progressBar.GetSliderRect(1.0f);
307 Color qteSliderColor = Color.White;
308 if (qteCooldown > 0.0f)
309 {
310 qteSliderColor = qteSuccess ? GUIStyle.Green : GUIStyle.Red * 0.5f;
311 progressBar.Color = ToolBox.GradientLerp(qteCooldown / QteCooldownDuration, progressBar.Color, qteSliderColor, Color.White);
312 }
313 else
314 {
315 if (qteTimer / QteDuration <= item.Condition / item.MaxCondition)
316 {
317 qteSliderColor = Color.Lerp(qteSliderColor, GUIStyle.Green, 0.5f);
318 }
319 }
320
321 progressBar.Parent.Parent.Parent.DrawManually(spriteBatch, true);
322 GUI.DrawRectangle(spriteBatch,
323 new Rectangle(sliderRect.X + (int)((qteTimer / QteDuration) * sliderRect.Width), sliderRect.Y - 5, 2, sliderRect.Height + 10),
324 qteSliderColor, true);
325
326 if (item.Condition > defaultMaxCondition)
327 {
328 float extraCondition = item.MaxCondition * (item.MaxRepairConditionMultiplier - 1.0f);
329 progressBar.Color = ToolBox.GradientLerp((item.Condition - defaultMaxCondition) / extraCondition, GUIStyle.ColorReputationHigh, GUIStyle.ColorReputationVeryHigh);
330 progressBarOverlayText.Visible = true;
331 progressBarOverlayText.Text = $"{(int)Math.Round((item.Condition / defaultMaxCondition) * 100)}%";
332 }
333 else
334 {
335 progressBarOverlayText.Visible = false;
336 }
337
338 RepairButton.Enabled = (currentFixerAction == FixActions.None || CurrentFixer == character) && !item.IsFullCondition;
339 RepairButton.Text = (currentFixerAction == FixActions.None || CurrentFixer != character || currentFixerAction != FixActions.Repair) ?
340 repairButtonText :
341 repairingText + new string('.', ((int)(Timing.TotalTime * 2.0f) % 3) + 1);
342
343 SabotageButton.Visible = character.IsTraitor;
344 SabotageButton.IgnoreLayoutGroups = !SabotageButton.Visible;
345 SabotageButton.Enabled = (currentFixerAction == FixActions.None || (CurrentFixer == character && currentFixerAction != FixActions.Sabotage)) && character.IsTraitor && item.ConditionPercentage > MinSabotageCondition;
346 SabotageButton.Text = (currentFixerAction == FixActions.None || CurrentFixer != character || currentFixerAction != FixActions.Sabotage || !character.IsTraitor) ?
347 sabotageButtonText :
348 sabotagingText + new string('.', ((int)(Timing.TotalTime * 2.0f) % 3) + 1);
349
350 TinkerButton.Visible = IsTinkerable(character);
351 TinkerButton.IgnoreLayoutGroups = !TinkerButton.Visible;
352 TinkerButton.Enabled = (currentFixerAction == FixActions.None || (CurrentFixer == character && currentFixerAction != FixActions.Tinker)) && CanTinker(character);
353 TinkerButton.Text = (currentFixerAction == FixActions.None || CurrentFixer != character || currentFixerAction != FixActions.Tinker) ?
354 tinkerButtonText :
355 tinkeringText + new string('.', ((int)(Timing.TotalTime * 2.0f) % 3) + 1);
356
357 System.Diagnostics.Debug.Assert(GuiFrame.GetChild(0) is GUILayoutGroup, "Repair UI hierarchy has changed, could not find skill texts");
358
359 extraButtonContainer.Visible = SabotageButton.Visible || TinkerButton.Visible;
360 extraButtonContainer.IgnoreLayoutGroups = !extraButtonContainer.Visible;
361
362 foreach (GUIComponent c in GuiFrame.GetChild(0).Children)
363 {
364 if (!(c.UserData is Skill skill)) continue;
365
366 GUITextBlock textBlock = (GUITextBlock)c;
367 if (character.GetSkillLevel(skill.Identifier) < (skill.Level * SkillRequirementMultiplier))
368 {
369 textBlock.TextColor = GUIStyle.Red;
370 }
371 else
372 {
373 textBlock.TextColor = Color.White;
374 }
375 }

408 private void QTEAction()
409 {
410 if (currentFixerAction == FixActions.Repair)
411 {
412 float defaultMaxCondition = item.MaxCondition / item.MaxRepairConditionMultiplier;
413 qteSuccess = qteCooldown <= 0.0f && qteTimer / QteDuration <= item.Condition / defaultMaxCondition;
414 }
415 else
416 {
417 return;
418 }
419
420 if (!GameMain.IsMultiplayer) { RepairBoost(qteSuccess); }
421
422 SoundPlayer.PlayUISound(qteSuccess ? GUISoundType.IncreaseQuantity : GUISoundType.DecreaseQuantity);
423
424 //on failure during cooldown reset cursor to beginning
425 if (!qteSuccess && qteCooldown > 0.0f) { qteTimer = QteDuration; }
426 qteCooldown = QteCooldownDuration;
427 //this will be set on button down so we can reset it here
428 requestStartFixAction = FixActions.None;
429 item.CreateClientEvent(this);
430 }

432 public void ClientEventRead(IReadMessage msg, float sendingTime)
433 {
434 deteriorationTimer = msg.ReadSingle();
435 deteriorateAlwaysResetTimer = msg.ReadSingle();
436 DeteriorateAlways = msg.ReadBoolean();
437 tinkeringDuration = msg.ReadSingle();
438 tinkeringStrength = msg.ReadSingle();
439 tinkeringPowersDevices = msg.ReadBoolean();
440 ushort currentFixerID = msg.ReadUInt16();
441 currentFixerAction = (FixActions)msg.ReadRangedInteger(0, 2);
442 CurrentFixer = currentFixerID != 0 ? Entity.FindEntityByID(currentFixerID) as Character : null;
443 item.MaxRepairConditionMultiplier = GetMaxRepairConditionMultiplier(CurrentFixer);
444 if (CurrentFixer == null)
445 {
446 qteTimer = QteDuration;
447 qteCooldown = 0.0f;
448 }
