224  partial void UpdateProjSpecific(float deltaTime)
225  {
226  if (item.HiddenInGame) { return; }
227  if (FakeBrokenTimer > 0.0f)
228  {
229  item.FakeBroken = true;
230  if (Character.Controlled == null || (Character.Controlled.CharacterHealth.GetAffliction("psychosis")?.Strength ?? 0.0f) <= 0.0f)
231  {
232  FakeBrokenTimer = 0.0f;
233  }
234  else
235  {
236  FakeBrokenTimer -= deltaTime;
237  }
238  }
239  else
240  {
241  item.FakeBroken = false;
242  }
243 
244 
245  if (!GameMain.IsMultiplayer)
246  {
247  switch (requestStartFixAction)
248  {
249  case FixActions.Repair:
250  case FixActions.Sabotage:
251  case FixActions.Tinker:
252  StartRepairing(Character.Controlled, requestStartFixAction);
253  requestStartFixAction = FixActions.None;
254  break;
255  default:
256  requestStartFixAction = FixActions.None;
257  break;
258  }
259  }
260 
261  for (int i = 0; i < particleEmitters.Count; i++)
262  {
263  if ((item.ConditionPercentage >= particleEmitterConditionRanges[i].X && item.ConditionPercentage <= particleEmitterConditionRanges[i].Y) || FakeBrokenTimer > 0.0f)
264  {
265  particleEmitters[i].Emit(deltaTime, item.WorldPosition, item.CurrentHull);
266  }
267  }
268 
269  if (CurrentFixer != null && CurrentFixer.SelectedConstruction == item)
270  {
271  if (repairSoundChannel == null || !repairSoundChannel.IsPlaying)
272  {
273  repairSoundChannel = SoundPlayer.PlaySound("repair", item.WorldPosition, hullGuess: item.CurrentHull);
274  }
275 
276  if (qteCooldown > 0.0f)
277  {
278  qteCooldown -= deltaTime;
279  if (qteCooldown <= 0.0f)
280  {
281  qteTimer = QteDuration;
282  }
283  }
284  else
285  {
286  qteTimer -= deltaTime * (qteTimer / QteDuration);
287  if (qteTimer < 0.0f) { qteTimer = QteDuration; }
288  }
289  }
290  else
291  {
292  repairSoundChannel?.FadeOutAndDispose();
293  repairSoundChannel = null;
294  }
295  }
