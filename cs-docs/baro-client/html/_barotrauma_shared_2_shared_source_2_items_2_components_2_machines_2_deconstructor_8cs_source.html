155  private void ProcessItem(Item targetItem, IEnumerable<Item> inputItems, List<DeconstructItem> validDeconstructItems, bool allowRemove = true)
156  {
157  // In multiplayer, the server handles the deconstruction into new items
158  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
159 
160  float amountMultiplier = 1f;
161 
162  if (user != null && !user.Removed)
163  {

165  user.CheckTalents(AbilityEffectType.OnItemDeconstructed, abilityTargetItem);
166

168  {
169  character.CheckTalents(AbilityEffectType.OnItemDeconstructedByAlly, abilityTargetItem);
170  }
171 
172  var itemCreationMultiplier = new AbilityItemCreationMultiplier(targetItem.Prefab, amountMultiplier);
173  user.CheckTalents(AbilityEffectType.OnItemDeconstructedMaterial, itemCreationMultiplier);
174  amountMultiplier = (int)itemCreationMultiplier.Value;
175  }
176 
177  if (targetItem.Prefab.RandomDeconstructionOutput)
178  {
179  int amount = targetItem.Prefab.RandomDeconstructionOutputAmount;
180  List<int> deconstructItemIndexes = new List<int>();
181  for (int i = 0; i < validDeconstructItems.Count; i++)
182  {
183  deconstructItemIndexes.Add(i);
184  }
185  List<float> commonness = validDeconstructItems.Select(i => i.Commonness).ToList();
186  List<DeconstructItem> products = new List<DeconstructItem>();
187 
188  for (int i = 0; i < amount; i++)
189  {
190  if (deconstructItemIndexes.Count < 1) { break; }
191  var itemIndex = ToolBox.SelectWeightedRandom(deconstructItemIndexes, commonness, Rand.RandSync.Unsynced);
192  products.Add(validDeconstructItems[itemIndex]);
193  var removeIndex = deconstructItemIndexes.IndexOf(itemIndex);
194  deconstructItemIndexes.RemoveAt(removeIndex);
195  commonness.RemoveAt(removeIndex);
196  }
197 
198  foreach (DeconstructItem deconstructProduct in products)
199  {
200  CreateDeconstructProduct(deconstructProduct, inputItems, amountMultiplier);
201  }
202  }
203  else
204  {
205  foreach (DeconstructItem deconstructProduct in validDeconstructItems)
206  {
207  CreateDeconstructProduct(deconstructProduct, inputItems, amountMultiplier);

211  void CreateDeconstructProduct(DeconstructItem deconstructProduct, IEnumerable<Item> inputItems, float amountMultiplier)
212  {
213  float percentageHealth = targetItem.Condition / targetItem.MaxCondition;
214 
215  if (percentageHealth < deconstructProduct.MinCondition || percentageHealth > deconstructProduct.MaxCondition) { return; }
216 
217  if (!(MapEntityPrefab.Find(null, deconstructProduct.ItemIdentifier) is ItemPrefab itemPrefab))
218  {
219  DebugConsole.ThrowError("Tried to deconstruct item \"" + targetItem.Name + "\" but couldn't find item prefab \"" + deconstructProduct.ItemIdentifier + "\"!");
220  return;
221  }
222 
223  float condition = deconstructProduct.CopyCondition ?
224  percentageHealth * itemPrefab.Health * deconstructProduct.OutConditionMax :
225  itemPrefab.Health * Rand.Range(deconstructProduct.OutConditionMin, deconstructProduct.OutConditionMax);
226 
227  if (DeconstructItemsSimultaneously && deconstructProduct.RequiredOtherItem.Length > 0)
228  {
229  foreach (Item otherItem in inputItems)
230  {
231  if (targetItem == otherItem) { continue; }
232  if (deconstructProduct.RequiredOtherItem.Any(r => otherItem.HasTag(r) || r == otherItem.Prefab.Identifier))
233  {
234  user?.CheckTalents(AbilityEffectType.OnGeneticMaterialCombinedOrRefined);
235  foreach (Character character in Character.GetFriendlyCrew(user))
236  {
237  character.CheckTalents(AbilityEffectType.OnCrewGeneticMaterialCombinedOrRefined);
238  }
239 
240  var geneticMaterial1 = targetItem.GetComponent<GeneticMaterial>();
241  var geneticMaterial2 = otherItem.GetComponent<GeneticMaterial>();
242  if (geneticMaterial1 != null && geneticMaterial2 != null)
243  {
244  if (geneticMaterial1.Combine(geneticMaterial2, user))
245  {
246  inputContainer.Inventory.RemoveItem(otherItem);
247  OutputContainer.Inventory.RemoveItem(otherItem);
248  Entity.Spawner.AddItemToRemoveQueue(otherItem);
249  }
250  allowRemove = false;
251  return;
252  }
253  inputContainer.Inventory.RemoveItem(otherItem);
254  OutputContainer.Inventory.RemoveItem(otherItem);
255  Entity.Spawner.AddItemToRemoveQueue(otherItem);
256  }
257  }
258  }
259 
260  if (user != null && !user.Removed)
261  {
262  // used to spawn items directly into the deconstructor
263  var itemDeconstructedInventory = new AbilityItemDeconstructedInventory(targetItem.Prefab, item);
264  user.CheckTalents(AbilityEffectType.OnItemDeconstructedInventory, itemDeconstructedInventory);
265  }
266 
267  int amount = (int)amountMultiplier;
268  for (int i = 0; i < amount; i++)
269  {
270  Entity.Spawner.AddItemToSpawnQueue(itemPrefab, outputContainer.Inventory, condition, onSpawned: (Item spawnedItem) =>
271  {
272  spawnedItem.StolenDuringRound = targetItem.StolenDuringRound;
273  spawnedItem.AllowStealing = targetItem.AllowStealing;
274  for (int i = 0; i < outputContainer.Capacity; i++)
275  {
276  var containedItem = outputContainer.Inventory.GetItemAt(i);
277  bool combined = false;
278  if (containedItem?.OwnInventory != null)
279  {
280  foreach (Item subItem in containedItem.ContainedItems.ToList())
281  {
282  if (subItem.Combine(spawnedItem, null))
283  {
284  combined = true;
285  break;
286  }
287  }
288  }
289  if (!combined)
290  {
291  if (containedItem?.Combine(spawnedItem, null) ?? false)
292  {
293  break;
294  }
295  }
296  }
297  PutItemsToLinkedContainer();
298  });
299  }
300  }
301 
302  GameAnalyticsManager.AddDesignEvent("ItemDeconstructed:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "none") + ":" + targetItem.Prefab.Identifier);
303 
304  bool? result = GameMain.LuaCs.Hook.Call<bool?>("item.deconstructed", targetItem, this, user, allowRemove);

387  private IEnumerable<(Item item, DeconstructItem output)> GetAvailableOutputs(bool checkRequiredOtherItems = true)
388  {
389  var items = inputContainer.Inventory.AllItems;
390  foreach (Item inputItem in items)
391  {
392  if (!inputItem.AllowDeconstruct) { continue; }
393  foreach (var deconstructItem in inputItem.Prefab.DeconstructItems)
394  {
395  if (deconstructItem.RequiredDeconstructor.Length > 0)
396  {
397  if (!deconstructItem.RequiredDeconstructor.Any(r => item.HasTag(r) || item.Prefab.Identifier == r)) { continue; }
398  }
399  if (deconstructItem.RequiredOtherItem.Length > 0 && checkRequiredOtherItems)
400  {
401  if (!deconstructItem.RequiredOtherItem.Any(r => items.Any(it => it.HasTag(r) || it.Prefab.Identifier == r))) { continue; }
402  bool validOtherItemFound = false;
403  foreach (Item otherInputItem in items)
404  {
405  if (otherInputItem == inputItem) { continue; }
406  if (!deconstructItem.RequiredOtherItem.Any(r => otherInputItem.HasTag(r) || otherInputItem.Prefab.Identifier == r)) { continue; }
407 
408  var geneticMaterial1 = inputItem.GetComponent<GeneticMaterial>();
409  var geneticMaterial2 = otherInputItem.GetComponent<GeneticMaterial>();
410  if (geneticMaterial1 != null && geneticMaterial2 != null)
411  {
412  if (!geneticMaterial1.CanBeCombinedWith(geneticMaterial2)) { continue; }
413  }
414  validOtherItemFound = true;
415  }
416  if (!validOtherItemFound) { continue; }
417  }
418  yield return (inputItem, deconstructItem);
419  }
