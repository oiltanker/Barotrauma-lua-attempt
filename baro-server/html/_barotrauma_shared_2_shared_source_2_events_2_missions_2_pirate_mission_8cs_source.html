131 private float GetDifficultyModifiedValue(float preferredDifficulty, float levelDifficulty, float randomnessModifier, Random rand)
132 {
133 return Math.Abs(levelDifficulty - preferredDifficulty + MathHelper.Lerp(-randomnessModifier, randomnessModifier, (float)rand.NextDouble()));
134 }
135 private int GetDifficultyModifiedAmount(int minAmount, int maxAmount, float levelDifficulty, Random rand)
136 {
137 return Math.Max((int)Math.Round(minAmount + (maxAmount - minAmount) * (levelDifficulty + MathHelper.Lerp(-RandomnessModifier, RandomnessModifier, (float)rand.NextDouble())) / MaxDifficulty), minAmount);
138 }

140 private XElement GetRandomDifficultyModifiedElement(XElement parentElement, float levelDifficulty, float randomnessModifier)
141 {
142 Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
143 // look for the element that is closest to our difficulty, with some randomness
144 XElement bestElement = null;
145 float bestValue = float.MaxValue;
146 foreach (XElement element in parentElement.Elements())
147 {
148 float applicabilityValue = GetDifficultyModifiedValue(element.GetAttributeFloat(0f, "preferreddifficulty"), levelDifficulty, randomnessModifier, rand);
149 if (applicabilityValue < bestValue)
150 {
151 bestElement = element;
152 bestValue = applicabilityValue;
153 }
154 }
155 return bestElement;
156 }

158 private void CreateMissionPositions(out Vector2 preferredSpawnPos)
159 {
160 Vector2 patrolPos = enemySub.WorldPosition;
161 Point subSize = enemySub.GetDockedBorders().Size;
162
163 if (!Level.Loaded.TryGetInterestingPosition(true, Level.PositionType.MainPath, Level.Loaded.Size.X * 0.3f, out preferredSpawnPos))
164 {
165 DebugConsole.ThrowError("Could not spawn pirate submarine in an interesting location! " + this);
166 }
167 if (!Level.Loaded.TryGetInterestingPositionAwayFromPoint(true, Level.PositionType.MainPath, Level.Loaded.Size.X * 0.3f, out patrolPos, preferredSpawnPos, minDistFromPoint: 10000f))
168 {
169 DebugConsole.ThrowError("Could not give pirate submarine an interesting location to patrol to! " + this);
170 }
171
172 patrolPos = enemySub.FindSpawnPos(patrolPos, subSize);
173
174 patrolPositions.Add(patrolPos);
175 patrolPositions.Add(preferredSpawnPos);
176
177 if (!IsClient)
178 {
179 PathFinder pathFinder = new PathFinder(WayPoint.WayPointList, false);
180 var path = pathFinder.FindPath(ConvertUnits.ToSimUnits(patrolPos), ConvertUnits.ToSimUnits(preferredSpawnPos));
181 if (!path.Unreachable)
182 {
183 var validNodes = path.Nodes.FindAll(n => !Level.Loaded.ExtraWalls.Any(w => w.Cells.Any(c => c.IsPointInside(n.WorldPosition))));
184 if (validNodes.Any())
185 {
186 preferredSpawnPos = validNodes.GetRandomUnsynced().WorldPosition; // spawn the sub in a random point in the path if possible
187 }
188 }
189
190 int graceDistance = 500; // the sub still spawns awkwardly close to walls, so this helps. could also be given as a parameter instead
191 preferredSpawnPos = enemySub.FindSpawnPos(preferredSpawnPos, new Point(subSize.X + graceDistance, subSize.Y + graceDistance));
192 }
193 }

195 private void InitPirateShip()
196 {
197 enemySub.NeutralizeBallast();
198 if (enemySub.GetItems(alsoFromConnectedSubs: false).Find(i => i.HasTag("reactor") && !i.NonInteractable)?.GetComponent<Reactor>() is Reactor reactor)
199 {
200 reactor.PowerUpImmediately();
201 }
202 enemySub.EnableMaintainPosition();
203 enemySub.TeamID = CharacterTeamType.None;
204 //make the enemy sub withstand atleast the same depth as the player sub
205 enemySub.RealWorldCrushDepth = Math.Max(enemySub.RealWorldCrushDepth, Submarine.MainSub.RealWorldCrushDepth);
206 if (Level.Loaded != null)
207 {
208 //...and the depth of the patrol positions + 1000 m
209 foreach (var patrolPos in patrolPositions)
210 {
211 enemySub.RealWorldCrushDepth = Math.Max(enemySub.RealWorldCrushDepth, Level.Loaded.GetRealWorldDepth(patrolPos.Y) + 1000);
212 }
213 }
214 enemySub.ImmuneToBallastFlora = true;

217 private void InitPirates()
218 {
219 characters.Clear();
220 characterItems.Clear();
221
222 if (characterConfig == null)
223 {
224 DebugConsole.ThrowError("Failed to initialize characters for escort mission (characterConfig == null)");
225 return;
226 }
227
228 int playerCount = 1;
229
230#if SERVER
231 playerCount = GameMain.Server.ConnectedClients.Where(c => !c.SpectateOnly || !GameMain.Server.ServerSettings.AllowSpectating).Count();
232#endif
233
234 float enemyCreationDifficulty = missionDifficulty + playerCount * addedMissionDifficultyPerPlayer;
235
236 Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
237
238 bool commanderAssigned = false;
239 foreach (XElement element in characterConfig.Elements())
240 {
241 // it is possible to get more than the "max" amount of characters if the modified difficulty is high enough; this is intentional
242 // if necessary, another "hard max" value could be used to clamp the value for performance/gameplay concerns
243 int amountCreated = GetDifficultyModifiedAmount(element.GetAttributeInt("minamount", 0), element.GetAttributeInt("maxamount", 0), enemyCreationDifficulty, rand);
244 for (int i = 0; i < amountCreated; i++)
245 {
246 XElement characterType = characterTypeConfig.Elements().Where(e => e.GetAttributeString("typeidentifier", string.Empty) == element.GetAttributeString("typeidentifier", string.Empty)).FirstOrDefault();
247
248 if (characterType == null)
249 {
250 DebugConsole.ThrowError($"No character types defined in CharacterTypes for a declared type identifier in mission \"{Prefab.Identifier}\".");
251 return;
252 }
253
254 XElement variantElement = GetRandomDifficultyModifiedElement(characterType, enemyCreationDifficulty, RandomnessModifier);
255
256 Character spawnedCharacter = CreateHuman(GetHumanPrefabFromElement(variantElement), characters, characterItems, enemySub, CharacterTeamType.None, null);
257 if (!commanderAssigned)
258 {
259 bool isCommander = variantElement.GetAttributeBool("iscommander", false);
260 if (isCommander && spawnedCharacter.AIController is HumanAIController humanAIController)
261 {
262 humanAIController.InitShipCommandManager();
263 foreach (var patrolPos in patrolPositions)
264 {
265 humanAIController.ShipCommandManager.patrolPositions.Add(patrolPos);
266 }
267 commanderAssigned = true;
268 }
269 }
270
271 foreach (Item item in spawnedCharacter.Inventory.AllItems)
272 {
273 if (item?.Prefab.Identifier == "idcard")
274 {
275 item.AddTag("id_pirate");
276 }
277 }
278 }
279 }

283 {
284 if (characters.Count > 0)
285 {
286#if DEBUG

294 if (patrolPositions.Count > 0)
295 {
296#if DEBUG
297 throw new Exception($"patrolPositions.Count > 0 ({patrolPositions.Count})");
298#else
299 DebugConsole.AddWarning("Patrol point list was not empty at the start of a pirate mission. The mission instance may not have been ended correctly on previous rounds.");
300 patrolPositions.Clear();
301#endif
302 }
303
304 enemySub = Submarine.MainSubs[1];
305
306 if (enemySub == null)
307 {
308 DebugConsole.ThrowError($"Enemy Submarine was not created. SubmarineInfo is likely not defined.");
309 // TODO: should we set the state to something here?
310 return;
311 }
312
313 Vector2 spawnPos = Level.Loaded.EndPosition; // in case TryGetInterestingPosition fails, though this should not happen
314 CreateMissionPositions(out spawnPos); // patrol positions are not explicitly replicated, instead they are acquired the same way the server acquires them
315#if DEBUG
316 if (IsClient)
317 {
318 DebugConsole.NewMessage("The patrol positions set by client were: ");
319 }
320 else
321 {
322 DebugConsole.NewMessage("The patrol positions set by server were: ");
323 }
324 foreach (var patrolPos in patrolPositions)
325 {
326 DebugConsole.NewMessage("Patrol pos: " + patrolPos);
327 }
328#endif
329 if (!IsClient)
330 {
331 InitPirateShip();
332 }
333 enemySub.SetPosition(spawnPos);
334
335 // flipping the sub on the frame it is moved into place must be done after it's been moved, or it breaks item connections in the submarine
336 // creating the pirates has to be done after the sub has been flipped, or it seems to break the AI pathing
337 enemySub.FlipX();
338 enemySub.ShowSonarMarker = false;
339
340 if (!IsClient)
341 {
342 InitPirates();
343 }
344 }

346 protected override void UpdateMissionSpecific(float deltaTime)
347 {
348 if (state >= 2) { return; }
349
350 float sqrSonarRange = MathUtils.Pow2(Sonar.DefaultSonarRange);
351 outsideOfSonarRange = Vector2.DistanceSquared(enemySub.WorldPosition, Submarine.MainSub.WorldPosition) > sqrSonarRange;
352 if (CheckWinState())
353 {
354 State = 2;
355 }
356 else
357 {
358 switch (State)
359 {
360 case 0:
361 for (int i = patrolPositions.Count - 1; i >= 0; i--)
362 {
363 if (Vector2.DistanceSquared(patrolPositions[i], Submarine.MainSub.WorldPosition) < sqrSonarRange)
364 {
365 patrolPositions.RemoveAt(i);
366 }
367 }
368 if (!outsideOfSonarRange || patrolPositions.None())
369 {
370 State = 1;
371 }
372 break;
373 case 1:
374 if (outsideOfSonarRange)
375 {
376 if (lastSighting.HasValue && Vector2.DistanceSquared(lastSighting.Value, Submarine.MainSub.WorldPosition) < sqrSonarRange)
377 {
378 lastSighting = null;
379 }
380 pirateSightingUpdateTimer -= deltaTime;
381 if (pirateSightingUpdateTimer < 0)
382 {
383 pirateSightingUpdateTimer = pirateSightingUpdateFrequency;
384 lastSighting = enemySub.WorldPosition;
385 }
386 }
387 else
388 {
389 lastSighting = enemySub.WorldPosition;
390 pirateSightingUpdateTimer = 0;
391 }
392 break;
393 }
394 }
395 }
