29 protected override float GetPriority()
30 {
31 if (!IsAllowed)
32 {
33 Priority = 0;
34 Abandon = true;
35 return Priority;
36 }
37 else
38 {
39 float distanceFactor = 0.9f;
40 if (!IsPriority && item.CurrentHull != character.CurrentHull)
41 {
42 float yDist = Math.Abs(character.WorldPosition.Y - item.WorldPosition.Y);
43 yDist = yDist > 100 ? yDist * 5 : 0;
44 float dist = Math.Abs(character.WorldPosition.X - item.WorldPosition.X) + yDist;
45 distanceFactor = MathHelper.Lerp(0.9f, 0, MathUtils.InverseLerp(0, 5000, dist));
46 }
47 bool isSelected = character.HasItem(item);
48 float selectedBonus = isSelected ? 100 - MaxDevotion : 0;
49 float devotion = (CumulatedDevotion + selectedBonus) / 100;
50 float reduction = IsPriority ? 1 : isSelected ? 2 : 3;
51 float max = AIObjectiveManager.LowestOrderPriority - reduction;
52 Priority = MathHelper.Lerp(0, max, MathHelper.Clamp(devotion + (distanceFactor * PriorityModifier), 0, 1));
53 if (decontainObjective == null)
54 {
55 // Halve the priority until there's a decontain objective (a valid container was found).
56 Priority /= 2;
57 }
58 }
59 return Priority;
60 }
61
62 protected override void Act(float deltaTime)
63 {
64 if (subObjectives.Any()) { return; }
65 if (HumanAIController.FindSuitableContainer(character, item, ignoredContainers, ref itemIndex, out Item suitableContainer))
66 {
67 itemIndex = 0;
68 if (suitableContainer != null)
69 {
70 bool equip = item.GetComponent<Holdable>() != null ||
71 item.AllowedSlots.Any(s => s != InvSlotType.Any) &&
72 item.AllowedSlots.None(s =>
73 s == InvSlotType.Card ||
74 s == InvSlotType.Head ||
75 s == InvSlotType.Headset ||
76 s == InvSlotType.InnerClothes ||
77 s == InvSlotType.OuterClothes ||
78 s == InvSlotType.HealthInterface);
79
80 TryAddSubObjective(ref decontainObjective, () => new AIObjectiveDecontainItem(character, item, objectiveManager, targetContainer: suitableContainer.GetComponent<ItemContainer>())
81 {
82 Equip = equip,
83 TakeWholeStack = true,
84 DropIfFails = true
85 },
86 onCompleted: () =>
87 {
88 if (equip)
89 {
90 HumanAIController.ReequipUnequipped();
91 }
92 IsCompleted = true;
93 },
94 onAbandon: () =>
95 {
96 if (equip)
97 {
98 HumanAIController.ReequipUnequipped();
99 }
100 if (decontainObjective != null && decontainObjective.ContainObjective != null && decontainObjective.ContainObjective.CanBeCompleted)
101 {
102 ignoredContainers.Add(suitableContainer);
103 }
104 else
105 {
106 Abandon = true;
107 }
108 });
109 }
110 else
111 {
112 Abandon = true;
113 }
114 }
115 else
116 {
117 objectiveManager.GetObjective<AIObjectiveIdle>().Wander(deltaTime);
118 }
119 }

121 protected override bool CheckObjectiveSpecific()
122 {
123 if (item.IgnoreByAI(character))
124 {
125 Abandon = true;
126 return false;
127 }
128 if (item.ParentInventory != null)
129 {
130 if (item.Container != null && !AIObjectiveCleanupItems.IsValidContainer(item.Container, character, allowUnloading: objectiveManager.HasOrder<AIObjectiveCleanupItems>()))
131 {
132 // Target was picked up or moved by someone.
133 Abandon = true;
134 return false;
135 }
136 }
137 return IsCompleted;
138 }
