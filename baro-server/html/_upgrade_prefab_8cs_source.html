46 public int GetBuyprice(int level, Location? location = null)
47 {
48 int price = BasePrice;
49 for (int i = 1; i <= level; i++)
50 {
51 price += (int)(price * MathHelper.Lerp(IncreaseLow, IncreaseHigh, i / (float)Prefab.MaxLevel) / 100);

151 {
152 if ("Structure" == element.NameAsIdentifier()) { return IsWallUpgrade; }

155 if (identifier.IsEmpty) { return false; }

158 if (item == null) { return false; }

354 public static int ParsePercentage(string value, Identifier attribute = default, XElement? sourceElement = null, bool suppressWarnings = false)
355 {
356 string? line = sourceElement?.ToString().Split('\n')[0].Trim();
357 bool doWarnings = !suppressWarnings && !attribute.IsEmpty && sourceElement != null && line != null;
358
359 if (string.IsNullOrWhiteSpace(value))
360 {
361 if (doWarnings)
362 {
363 DebugConsole.AddWarning($"Attribute \"{attribute}\" not found at {sourceElement!.Document?.ParseContentPathFromUri()} @ '{line}'.\n " +
364 "Value has been assumed to be '0'.");
365 }
366
367 return 1;
368 }
369
370 if (!int.TryParse(value, NumberStyles.Number, CultureInfo.InvariantCulture, out var price))
371 {
372 string str = value;
373
374 if (str.Length > 1 && str[0] == '+') { str = str.Substring(1); }
375
376 if (str.Length > 1 && str[^1] == '%') { str = str.Substring(0, str.Length - 1); }
377
378 if (int.TryParse(str, out price))
379 {
380 return price;
381 }
382 }
383 else
384 {
385 return price;
386 }
387
388 if (doWarnings)
389 {
390 DebugConsole.AddWarning($"Value in attribute \"{attribute}\" is not formatted correctly\n " +
391 $"at {sourceElement!.Document?.ParseContentPathFromUri()} @ '{line}'.\n " +
392 "It should be an integer with optionally a '+' or '-' at the front and/or '%' at the end.\n" +
393 "The value has been assumed to be '0'.");
394 }
395
396 return 1;
397 }
