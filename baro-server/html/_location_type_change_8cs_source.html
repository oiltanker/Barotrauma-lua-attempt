58 public Requirement(XElement element, LocationTypeChange change)
59 {
60 RequiredLocations = element.GetAttributeIdentifierArray("requiredlocations", element.GetAttributeIdentifierArray("requiredadjacentlocations", Array.Empty<Identifier>())).ToImmutableArray();
61 RequiredProximity = Math.Max(element.GetAttributeInt("requiredproximity", 1), 1);
62 ProximityProbabilityIncrease = element.GetAttributeFloat("proximityprobabilityincrease", 0.0f);
63 RequiredProximityForProbabilityIncrease = element.GetAttributeInt("requiredproximityforprobabilityincrease", -1);
64 RequireBeaconStation = element.GetAttributeBool("requirebeaconstation", false);
65 RequireHuntingGrounds = element.GetAttributeBool("requirehuntinggrounds", false);
66
67 string functionStr = element.GetAttributeString("function", "Add");
68 if (!Enum.TryParse(functionStr, ignoreCase: true, out Function))
69 {
70 DebugConsole.ThrowError(
71 $"Invalid location type change in location type \"{change.CurrentType}\". " +
72 $"\"{functionStr}\" is not a valid function.");
73 }
74
75 Probability = element.GetAttributeFloat("probability", 1.0f);
76
77 if (RequiredProximityForProbabilityIncrease > 0 || ProximityProbabilityIncrease > 0.0f)
78 {
79 if (!RequiredLocations.Any() && !RequireBeaconStation && !RequireHuntingGrounds)
80 {
81 DebugConsole.AddWarning(
82 $"Invalid location type change in location type \"{change.CurrentType}\". " +
83 "Probability is configured to increase when near some other type of location, but the RequiredLocations attribute is not set.");
84 }
85 if (Probability >= 1.0f)
86 {
87 DebugConsole.AddWarning(
88 $"Invalid location type change in location type \"{change.CurrentType}\". " +
89 "Probability is configured to increase when near some other type of location, but the base probability is already 100%");
90 }
91 }
92 }

99 public bool AnyWithinDistance(Location location, int maxDistance, int currentDistance = 0, HashSet<Location> checkedLocations = null)
100 {
101 if (currentDistance > maxDistance) { return false; }
102 if (currentDistance > 0 && MatchesLocation(location)) { return true; }
103
104 checkedLocations ??= new HashSet<Location>();
105 checkedLocations.Add(location);
106
107 foreach (var connection in location.Connections)
108 {
109 if (RequireBeaconStation && connection.LevelData.HasBeaconStation && connection.LevelData.IsBeaconActive)
110 {
111 return true;
112 }
113 if (RequireHuntingGrounds && connection.LevelData.HasHuntingGrounds)
114 {
115 return true;
116 }
117
118 var otherLocation = connection.OtherLocation(location);
119 if (!checkedLocations.Contains(otherLocation))
120 {
121 if (AnyWithinDistance(otherLocation, maxDistance, currentDistance + 1, checkedLocations)) { return true; }
122 }
123 }
124
125 return false;
126 }

222 public float DetermineProbability(Location location)
223 {
224 if (RequireDiscovered && !location.Discovered) { return 0.0f; }
225 if (location.IsCriticallyRadiated()) { return 0.0f; }
226 if (location.LocationTypeChangeCooldown > 0) { return 0.0f; }
227 if (location.IsGateBetweenBiomes) { return 0.0f; }
228
229 if (DisallowedAdjacentLocations.Any() &&
230 AnyWithinDistance(location, DisallowedProximity, (otherLocation) => { return DisallowedAdjacentLocations.Contains(otherLocation.Type.Identifier); }))
231 {
232 return 0.0f;
233 }
234
235 float probability = Probability;
236 foreach (Requirement requirement in Requirements)
237 {
238 if (requirement.AnyWithinDistance(location, requirement.RequiredProximity))
239 {
240 if (requirement.Function == Requirement.FunctionType.Add)
241 {
242 probability += requirement.Probability;
243 }
244 else
245 {
246 probability *= requirement.Probability;
247 }
248 }
249
250 if (location.ProximityTimer.ContainsKey(requirement))
251 {
252 if (requirement.AnyWithinDistance(location, requirement.RequiredProximityForProbabilityIncrease))
253 {
254 if (requirement.Function == Requirement.FunctionType.Add)
255 {
256 probability += requirement.ProximityProbabilityIncrease * location.ProximityTimer[requirement];
257 }
258 else
259 {
260 probability *= requirement.ProximityProbabilityIncrease * location.ProximityTimer[requirement];
261 }
262 }
263 }
264 }
265
266 return probability;
267 }

272 if (currentDistance > 0 && predicate(location)) { return true; }
273

283 }
