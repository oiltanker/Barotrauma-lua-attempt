148 public void Save()
149 {
150 if (!Directory.Exists(SavePath))
151 {
152 try
153 {
154 Directory.CreateDirectory(SavePath);
155 }
156 catch (Exception e)
157 {
158 DebugConsole.ThrowError("Failed to create a folder for server logs", e);
159 return;
160 }
161 }
162
163 string fileName = ServerName + "_" + DateTime.Now.ToString("yyyy-MM-dd_HH:mm");
164 fileName = ToolBox.RemoveInvalidFileNameChars(fileName);
165
166 string filePath = Path.Combine(SavePath, fileName + ".txt");
167 int i = 2;
168 while (File.Exists(filePath))
169 {
170 filePath = Path.Combine(SavePath, fileName + " (" + i + ").txt");
171 i++;
172 }
173
174 try
175 {
176 File.WriteAllLines(filePath, unsavedLines.Select(l => l.Text.SanitizedValue));
177 }
178 catch (Exception e)
179 {
180 DebugConsole.ThrowError("Saving the server log to " + filePath + " failed", e);
181 return;
182 }
183 }
