75 public AfflictionSeverity AfflictionSeverity
76 {
77 get
78 {
79 if (Prefab is null) { return AfflictionSeverity.Low; }
80
81 float normalizedStrength = Strength / Prefab.MaxStrength;
82
83 // lesser than 0.1
84 if (normalizedStrength <= 0.1)
85 {
86 return AfflictionSeverity.Low;
87 }
88
89 // between 0.1 and 0.5
90 if (normalizedStrength > 0.1f && normalizedStrength < 0.5f)
91 {
92 return AfflictionSeverity.Medium;
93 }
94
95 // greater than 0.5
96 return AfflictionSeverity.High;
97 }
98 }

210 private HealRequestResult HealAllPending(bool force = false, Client? client = null)
211 {
212 int totalCost = GetTotalCost();
213 if (!force)
214 {
215 if (IsOutpostInCombat()) { return HealRequestResult.Refused; }
216 if (!(campaign?.TryPurchase(client, totalCost) ?? false)) { return HealRequestResult.InsufficientFunds; }
217 }
218
219 ImmutableArray<CharacterInfo> crew = GetCrewCharacters();
220 foreach (NetCrewMember crewMember in PendingHeals)
221 {
222 CharacterInfo? targetCharacter = crewMember.FindCharacterInfo(crew);
223 if (!(targetCharacter?.Character is { CharacterHealth: { } health })) { continue; }
224
225 foreach (NetAffliction affliction in crewMember.Afflictions)
226 {
227 health.ReduceAfflictionOnAllLimbs(affliction.Identifier, affliction.Prefab?.MaxStrength ?? affliction.Strength);
228 }
229 }
230
231 ClearPendingHeals();
232
233 return HealRequestResult.Success;
234 }

241 private void RemovePendingAffliction(NetCrewMember crewMember, NetAffliction affliction)
242 {
243 foreach (NetCrewMember listMember in PendingHeals.ToList())
244 {
245 PendingHeals.Remove(listMember);
246 NetCrewMember pendingMember = listMember;
247
248 if (pendingMember.CharacterEquals(crewMember))
249 {
250 List<NetAffliction> newAfflictions = new List<NetAffliction>();
251 foreach (NetAffliction pendingAffliction in pendingMember.Afflictions)
252 {
253 if (pendingAffliction.AfflictionEquals(affliction)) { continue; }
254
255 newAfflictions.Add(pendingAffliction);
256 }
257
258 pendingMember.Afflictions = newAfflictions.ToArray();
259 }
260
261 if (!pendingMember.Afflictions.Any()) { continue; }
262
263 PendingHeals.Add(pendingMember);
264 }
265 }

277 public static bool IsHealable(Affliction affliction)
278 {
279 return affliction.Prefab.HealableInMedicalClinic && affliction.Strength > GetShowTreshold(affliction);
280 static float GetShowTreshold(Affliction affliction) => Math.Max(0, Math.Min(affliction.Prefab.ShowIconToOthersThreshold, affliction.Prefab.ShowInHealthScannerThreshold));
281 }

283 private NetAffliction[] GetAllAfflictions(CharacterHealth health)
284 {
285 IEnumerable<Affliction> rawAfflictions = health.GetAllAfflictions().Where(a => IsHealable(a));
286
287 List<NetAffliction> afflictions = new List<NetAffliction>();
288
289 foreach (Affliction affliction in rawAfflictions)
290 {
291 NetAffliction newAffliction;
292 if (afflictions.FirstOrNull(netAffliction => netAffliction.AfflictionEquals(affliction.Prefab)) is { } foundAffliction)
293 {
294 afflictions.Remove(foundAffliction);
295 foundAffliction.Strength += (ushort)affliction.Strength;
296 foundAffliction.Price += (ushort)GetAdjustedPrice(GetHealPrice(affliction));
297 newAffliction = foundAffliction;
298 }
299 else
300 {
301 newAffliction = new NetAffliction { Affliction = affliction };
302 newAffliction.Price = (ushort)GetAdjustedPrice(newAffliction.Price);
303 }
304
305 afflictions.Add(newAffliction);
306 }
307
308 return afflictions.ToArray();
309
310 static int GetHealPrice(Affliction affliction) => (int)(affliction.Prefab.BaseHealCost + (affliction.Prefab.HealCostMultiplier * affliction.Strength));
311 }
