30 class DelayedEffect : StatusEffect
31 {
32 public static readonly List<DelayedListElement> DelayList = new List<DelayedListElement>();
33
34 private enum DelayTypes { Timer = 0, ReachCursor = 1 }
35
36 private readonly DelayTypes delayType;
37 private readonly float delay;
38
39 public DelayedEffect(ContentXElement element, string parentDebugName)
40 : base(element, parentDebugName)
41 {
42 string delayTypeStr = element.GetAttributeString("delaytype", "timer");
43 if (!Enum.TryParse(typeof(DelayTypes), delayTypeStr, ignoreCase: true, out var delayType))
44 {
45 DebugConsole.ThrowError("Invalid delay type \"" + delayTypeStr + "\" in StatusEffect (" + parentDebugName + ")");
46 }
47 switch (delayType)
48 {
49 case DelayTypes.Timer:
50 delay = element.GetAttributeFloat("delay", 1.0f);
51 break;
52 }
53 }
54
55 public override void Apply(ActionType type, float deltaTime, Entity entity, ISerializableEntity target, Vector2? worldPosition = null)
56 {
57 if (this.type != type || !HasRequiredItems(entity)) { return; }
58 if (!Stackable)
59 {
60 foreach (var existingEffect in DelayList)
61 {
62 if (existingEffect.Parent == this && existingEffect.Targets.FirstOrDefault() == target) { return; }
63 }
64 }
65 if (!IsValidTarget(target)) { return; }
66
67 currentTargets.Clear();
68 currentTargets.Add(target);
69 if (!HasRequiredConditions(currentTargets)) { return; }
70
71 switch (delayType)
72 {
73 case DelayTypes.Timer:
74 DelayList.Add(new DelayedListElement(this, entity, currentTargets, delay, worldPosition, null));
75 break;
76 case DelayTypes.ReachCursor:
77 Projectile projectile = (entity as Item)?.GetComponent<Projectile>();
78 if (projectile == null)
79 {
80 DebugConsole.ShowError("Non-projectile using a delaytype of reachcursor");
81 return;
82 }
83
84 if (projectile.User == null)
85 {
86 DebugConsole.ShowError("Projectile: '" + projectile.Name + "' missing user to determine distance");
87 return;
88 }
89
90 DelayList.Add(new DelayedListElement(this, entity, currentTargets, Vector2.Distance(entity.WorldPosition, projectile.User.CursorWorldPosition), worldPosition, entity.WorldPosition));
91 break;
92 }
93 }
94
95 public override void Apply(ActionType type, float deltaTime, Entity entity, IReadOnlyList<ISerializableEntity> targets, Vector2? worldPosition = null)
96 {
97 if (this.type != type || !HasRequiredItems(entity)) { return; }
98 if (delayType == DelayTypes.ReachCursor && Character.Controlled == null) { return; }
99 if (!Stackable)
100 {
101 foreach (var existingEffect in DelayList)
102 {
103 if (existingEffect.Parent == this && existingEffect.Targets.SequenceEqual(targets)) { return; }
104 }
105 }
106
107 currentTargets.Clear();
108 foreach (ISerializableEntity target in targets)
109 {
110 if (!IsValidTarget(target)) { continue; }
111 currentTargets.Add(target);
112 }
113
114 if (!HasRequiredConditions(currentTargets)) { return; }
115
116 switch (delayType)
117 {
118 case DelayTypes.Timer:
119 DelayList.Add(new DelayedListElement(this, entity, targets, delay, worldPosition, null));
120 break;
121 case DelayTypes.ReachCursor:
122 Projectile projectile = (entity as Item)?.GetComponent<Projectile>();
123 if (projectile == null)
124 {
125#if DEBUG
126 DebugConsole.ShowError("Non-projectile using a delaytype of reachcursor");
127#endif
128 return;
129 }
130
131 if (projectile.User == null)
132 {
133#if DEBUG
134 DebugConsole.ShowError("Projectile " + projectile.Name + "missing user");
135#endif
136 return;
137 }
138
139 DelayList.Add(new DelayedListElement(this, entity, targets, Vector2.Distance(entity.WorldPosition, projectile.User.CursorWorldPosition), worldPosition, entity.WorldPosition));
140 break;
141 }
142 }
143
144 public static void Update(float deltaTime)
145 {
146 for (int i = DelayList.Count - 1; i >= 0; i--)
147 {
148 DelayedListElement element = DelayList[i];
149 if (element.Parent.CheckConditionalAlways && !element.Parent.HasRequiredConditions(element.Targets))
150 {
151 DelayList.Remove(element);
152 continue;
153 }
154
155 switch (element.Parent.delayType)
156 {
157 case DelayTypes.Timer:
158 element.Delay -= deltaTime;
159 if (element.Delay > 0.0f) { continue; }
160 break;
161 case DelayTypes.ReachCursor:
162 if (Vector2.Distance(element.Entity.WorldPosition, element.StartPosition.Value) < element.Delay) { continue; }
163 break;
164 }
165
166 element.Parent.Apply(deltaTime, element.Entity, element.Targets, element.WorldPosition);
167 DelayList.Remove(element);
168 }
169 }
170 }
