50 public static void PerformOperation(CampaignMetadata metadata, Identifier identifier, object value, OperationType operation)
51 {
52 if (metadata == null) { return; }
53
54 object currentValue = metadata.GetValue(identifier);
55
56 float? originalValue = ConvertValueToFloat(currentValue ?? 0);
57 float? newValue = ConvertValueToFloat(value);
58
59 if ((originalValue == null || newValue == null) && operation != OperationType.Set)
60 {
61 DebugConsole.ThrowError($"Tried to perform numeric operations to a non number via SetDataAction (Existing: {currentValue?.GetType()}, New: {value.GetType()})");
62 return;
63 }
64
65 switch (operation)
66 {
67 case OperationType.Set:
68 metadata.SetValue(identifier, value);
69 break;
70 case OperationType.Add:
71 metadata.SetValue(identifier, originalValue + newValue ?? 0);
72 break;
73 case OperationType.Multiply:
74 metadata.SetValue(identifier, originalValue * newValue ?? 0);
75 break;
76 }
77 }
