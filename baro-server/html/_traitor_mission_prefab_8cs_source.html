56 public void Dispose()
57 {
58 foreach (var requiredName in required)

60 if (element.Attributes().All(attribute => attribute.Name != requiredName))

62 GameServer.Log($"Required attribute \"{requiredName}\" is missing in \"{element.Name}\"", ServerLog.MessageType.Error);
63 }
64 }
65 foreach (var attribute in element.Attributes())
66 {
67 var attributeName = attribute.Name.ToString();
68 if (!required.Contains(attributeName) && !optional.Contains(attributeName))

114 checker.Optional("targetcount");
115 checker.Optional("targetpercentage");

418 public class Role
419 {
420 public readonly Traitor.TraitorMission.RoleFilter Filter;
421
422 public Role(IEnumerable<Traitor.TraitorMission.RoleFilter> filters)
423 {
424 Filter = character => filters.All(filter => filter(character));
425 }

427 public Role()
428 {
429 Filter = character => true;
430 }

432 public readonly Dictionary<string, Role> Roles = new Dictionary<string, Role>();
433
434 public readonly Identifier Identifier;
435 public readonly string StartText;
436 public readonly string EndMessageSuccessText;
437 public readonly string EndMessageSuccessDeadText;
438 public readonly string EndMessageSuccessDetainedText;
439 public readonly string EndMessageFailureText;
440 public readonly string EndMessageFailureDeadText;
441 public readonly string EndMessageFailureDetainedText;
442
443 public readonly List<ObjectiveBase> Objectives = new List<ObjectiveBase>();
444
445 public Traitor.TraitorMission Instantiate()
446 {
447 var objectivesWithSync = new List<ObjectiveBase>();
448 var objectivesCount = Objectives.Count;
449 if (objectivesCount > 0)
450 {
451 var pendingRoles = new HashSet<string>();
452 var pendingCount = 1;
453 objectivesWithSync.Add(Objectives[0]);
454 pendingRoles.UnionWith(Objectives[0].Roles);
455 for (var i = 1; i < objectivesCount; ++i)
456 {
457 var objective = Objectives[i];
458 if (pendingRoles.IsSupersetOf(objective.Roles))
459 {
460 if (pendingCount > 1)
461 {
462 objectivesWithSync.Add(new WaitObjective(objective.Roles));
463 }
464 pendingRoles.Clear();
465 pendingCount = 0;
466 }
467 objectivesWithSync.Add(objective);
468 pendingRoles.UnionWith(objective.Roles);
469 ++pendingCount;
470 }
471 if (pendingCount > 1 && pendingRoles.IsSubsetOf(Roles.Keys))
472 {
473 // TODO: If last objective includes only one traitor, other traitors will get the wrong end message.
474 objectivesWithSync.Add(new WaitObjective(Roles.Keys));
475 }
476 }
477
478 return new Traitor.TraitorMission(
479 Identifier,
480 StartText ?? "TraitorMissionStartMessage",
481 EndMessageSuccessText ?? "TraitorObjectiveEndMessageSuccess",
482 EndMessageSuccessDeadText ?? "TraitorObjectiveEndMessageSuccessDead",
483 EndMessageSuccessDetainedText ?? "TraitorObjectiveEndMessageSuccessDetained",
484 EndMessageFailureText ?? "TraitorObjectiveEndMessageFailure",
485 EndMessageFailureDeadText ?? "TraitorObjectiveEndMessageFailureDead",
486 EndMessageFailureDetainedText ?? "TraitorObjectiveEndMessageFailureDetained",
487 Roles.ToDictionary(kv => kv.Key, kv => kv.Value.Filter),
488 objectivesWithSync.SelectMany(objective =>
489 {
490 objective.InstantiateGoals();
491 return objective.Roles.Select(role => objective.Instantiate(new[] { role }));
492 }).ToArray());
493 }
