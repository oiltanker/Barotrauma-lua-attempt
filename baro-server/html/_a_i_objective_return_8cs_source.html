53 protected override void Act(float deltaTime)
54 {
55 if (ReturnTarget == null)
56 {
57 Abandon = true;
58 return;
59 }
60 bool shouldUseEscapeBehavior = false;
61 if (character.CurrentHull != null || isSteeringThroughGap)
62 {
63 if (character.Submarine == null || !character.Submarine.IsConnectedTo(ReturnTarget))
64 {
65 // Character is on another sub that is not connected to the target sub, use the escape behavior to get them out
66 shouldUseEscapeBehavior = true;
67 if (!usingEscapeBehavior)
68 {
69 HumanAIController.ResetEscape();
70 }
71 isSteeringThroughGap = HumanAIController.Escape(deltaTime);
72 if (!isSteeringThroughGap && (HumanAIController.EscapeTarget == null || HumanAIController.IsCurrentPathUnreachable))
73 {
74 Abandon = true;
75 }
76 }
77 else if (character.Submarine != ReturnTarget)
78 {
79 // Character is on another sub that is connected to the target sub, create a Go To objective to reach the target sub
80 if (moveInsideObjective == null)
81 {
82 Hull targetHull = null;
83 foreach (var d in ReturnTarget.ConnectedDockingPorts.Values)
84 {
85 if (!d.Docked) { continue; }
86 if (d.DockingTarget == null) { continue; }
87 if (d.DockingTarget.Item.Submarine != character.Submarine) { continue; }
88 targetHull = d.Item.CurrentHull;
89 break;
90 }
91 if (targetHull != null && !targetHull.IsTaggedAirlock())
92 {
93 // Target the closest airlock
94 float closestDist = 0;
95 Hull airlock = null;
96 foreach (Hull hull in Hull.HullList)
97 {
98 if (hull.Submarine != targetHull.Submarine) { continue; }
99 if (!hull.IsTaggedAirlock()) { continue; }
100 float dist = Vector2.DistanceSquared(targetHull.Position, hull.Position);
101 if (airlock == null || closestDist <= 0 || dist < closestDist)
102 {
103 airlock = hull;
104 closestDist = dist;
105 }
106
107 }
108 if (airlock != null)
109 {
110 targetHull = airlock;
111 }
112 }
113 if (targetHull != null)
114 {
115 RemoveSubObjective(ref moveOutsideObjective);
116 TryAddSubObjective(ref moveInsideObjective,
117 constructor: () => new AIObjectiveGoTo(targetHull, character, objectiveManager)
118 {
119 AllowGoingOutside = true,
120 endNodeFilter = n => n.Waypoint.Submarine == targetHull.Submarine
121 },
122 onCompleted: () => RemoveSubObjective(ref moveInsideObjective),
123 onAbandon: () => Abandon = true);
124 }
125 else
126 {
127#if DEBUG
128 DebugConsole.ThrowError("Error with a Return objective: no suitable target for 'moveInsideObjective'");
129#endif
130 }
131 }
132 }
133 else
134 {
135 // Character is on the target sub, the objective is completed
136 IsCompleted = true;
137 }
138 }
139 else if (!isSteeringThroughGap && moveOutsideObjective == null)
140 {
141 Hull targetHull = null;
142 float targetDistanceSquared = float.MaxValue;
143 bool targetIsAirlock = false;
144 foreach (var hull in ReturnTarget.GetHulls(false))
145 {
146 bool hullIsAirlock = hull.IsTaggedAirlock();
147 if(hullIsAirlock || (!targetIsAirlock && hull.LeadsOutside(character)))
148 {
149 float distanceSquared = Vector2.DistanceSquared(character.WorldPosition, hull.WorldPosition);
150 if (targetHull == null || distanceSquared < targetDistanceSquared)
151 {
152 targetHull = hull;
153 targetDistanceSquared = distanceSquared;
154 targetIsAirlock = hullIsAirlock;
155 }
156 }
157 }
158 if (targetHull != null)
159 {
160 RemoveSubObjective(ref moveInsideObjective);
161 TryAddSubObjective(ref moveOutsideObjective,
162 constructor: () => new AIObjectiveGoTo(targetHull, character, objectiveManager)
163 {
164 AllowGoingOutside = true
165 },
166 onCompleted: () => RemoveSubObjective(ref moveOutsideObjective),
167 onAbandon: () => Abandon = true);
168 }
169 else
170 {
171#if DEBUG
172 DebugConsole.ThrowError("Error with a Return objective: no suitable target for 'moveOutsideObjective'");
173#endif
174 }
175 }
176 usingEscapeBehavior = shouldUseEscapeBehavior;
177 }
