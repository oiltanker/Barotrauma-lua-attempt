182 public virtual float GetSpeed(AnimationType type)
183 {
184 GroundedMovementParams movementParams;
185 switch (type)
186 {
187 case AnimationType.Walk:
188 if (!CanWalk)
189 {
190 DebugConsole.ThrowError($"{character.SpeciesName} cannot walk!");
191 return 0;
192 }
193 movementParams = WalkParams;
194 break;
195 case AnimationType.Run:
196 if (!CanWalk)
197 {
198 DebugConsole.ThrowError($"{character.SpeciesName} cannot run!");
199 return 0;
200 }
201 movementParams = RunParams;
202 break;
203 case AnimationType.SwimSlow:
204 return SwimSlowParams.MovementSpeed;
205 case AnimationType.SwimFast:
206 return SwimFastParams.MovementSpeed;
207 default:
208 throw new NotImplementedException(type.ToString());
209 }
210 return IsMovingBackwards ? movementParams.MovementSpeed * movementParams.BackwardsMovementMultiplier : movementParams.MovementSpeed;
211 }

336 public void HoldItem(float deltaTime, Item item, Vector2[] handlePos, Vector2 holdPos, Vector2 aimPos, bool aim, float holdAngle, float itemAngleRelativeToHoldAngle = 0.0f, bool aimMelee = false)
337 {
338 aimingMelee = aimMelee;
339 if (character.Stun > 0.0f || character.IsIncapacitated)
340 {
341 aim = false;
342 }
343
344 //calculate the handle positions
345 Matrix itemTransfrom = Matrix.CreateRotationZ(item.body.Rotation);
346 float horizontalOffset = ConvertUnits.ToSimUnits((item.Sprite.size.X / 2 - item.Sprite.Origin.X) * item.Scale);
347
348 //handlePos[0] = ConvertUnits.ToSimUnits(new Vector2(-45,25) * 0.5f);
349 //handlePos[1] = ConvertUnits.ToSimUnits(new Vector2(-65,30) * 0.5f);
350
351 transformedHandlePos[0] = Vector2.Transform(new Vector2(handlePos[0].X + horizontalOffset, handlePos[0].Y), itemTransfrom);
352 transformedHandlePos[1] = Vector2.Transform(new Vector2(handlePos[1].X + horizontalOffset, handlePos[1].Y), itemTransfrom);
353
354 Limb torso = GetLimb(LimbType.Torso) ?? MainLimb;
355 Limb leftHand = GetLimb(LimbType.LeftHand);
356 Limb rightHand = GetLimb(LimbType.RightHand);
357
358 Vector2 itemPos = aim ? aimPos : holdPos;
359
360 var controller = character.SelectedConstruction?.GetComponent<Controller>();
361 bool usingController = controller != null && !controller.AllowAiming;
362 bool isClimbing = character.IsClimbing && Math.Abs(character.AnimController.TargetMovement.Y) > 0.01f;
363 float itemAngle;
364 Holdable holdable = item.GetComponent<Holdable>();
365 float torsoRotation = torso.Rotation;
366
367 Item rightHandItem = character.Inventory?.GetItemInLimbSlot(InvSlotType.RightHand);
368 bool equippedInRightHand = rightHandItem == item && rightHand != null && !rightHand.IsSevered;
369 Item leftHandItem = character.Inventory?.GetItemInLimbSlot(InvSlotType.LeftHand);
370 bool equippedInLefthand = leftHandItem == item && leftHand != null && !leftHand.IsSevered;
371 if (aim && !isClimbing && !usingController && character.Stun <= 0.0f && itemPos != Vector2.Zero && !character.IsIncapacitated)
372 {
373 Vector2 mousePos = ConvertUnits.ToSimUnits(character.SmoothedCursorPosition);
374 Vector2 diff = holdable.Aimable ? (mousePos - AimSourceSimPos) * Dir : Vector2.UnitX;
375 holdAngle = MathUtils.VectorToAngle(new Vector2(diff.X, diff.Y * Dir)) - torsoRotation * Dir;
376 holdAngle += GetAimWobble(rightHand, leftHand, item);
377 itemAngle = torsoRotation + holdAngle * Dir;
378
379 if (holdable.ControlPose)
380 {
381 //if holding two items that should control the characters' pose, let the item in the right hand do it
382 bool anotherItemControlsPose = equippedInLefthand && rightHandItem != item && (rightHandItem?.GetComponent<Holdable>()?.ControlPose ?? false);
383 if (!anotherItemControlsPose)
384 {
385 var head = GetLimb(LimbType.Head);
386 if (head != null)
387 {
388 head.body.SmoothRotate(itemAngle, force: 30 * head.Mass);
389 }
390 if (TargetMovement == Vector2.Zero && inWater)
391 {
392 torso.body.AngularVelocity -= torso.body.AngularVelocity * 0.1f;
393 torso.body.ApplyForce(torso.body.LinearVelocity * -0.5f);
394 }
395 }
396 aiming = true;
397 }
398 }
399 else
400 {
401 if (holdable.UseHandRotationForHoldAngle)
402 {
403 if (equippedInRightHand)
404 {
405 itemAngle = rightHand.Rotation + holdAngle * Dir;
406 }
407 else if (equippedInLefthand)
408 {
409 itemAngle = leftHand.Rotation + holdAngle * Dir;
410 }
411 else
412 {
413 itemAngle = torsoRotation + holdAngle * Dir;
414 }
415 }
416 else
417 {
418 itemAngle = torsoRotation + holdAngle * Dir;
419 }
420 }
421
422 if (rightShoulder == null) { return; }
423 Vector2 transformedHoldPos = rightShoulder.WorldAnchorA;
424 if (itemPos == Vector2.Zero || isClimbing || usingController)
425 {
426 if (equippedInRightHand)
427 {
428 transformedHoldPos = rightHand.PullJointWorldAnchorA - transformedHandlePos[0];
429 itemAngle = rightHand.Rotation + (holdAngle - rightHand.Params.GetSpriteOrientation() + MathHelper.PiOver2) * Dir;
430 }
431 else if (equippedInLefthand)
432 {
433 transformedHoldPos = leftHand.PullJointWorldAnchorA - transformedHandlePos[1];
434 itemAngle = leftHand.Rotation + (holdAngle - leftHand.Params.GetSpriteOrientation() + MathHelper.PiOver2) * Dir;
435 }
436 }
437 else
438 {
439 if (equippedInRightHand)
440 {
441 transformedHoldPos = rightShoulder.WorldAnchorA;
442 rightHand.Disabled = true;
443 }
444 if (equippedInLefthand)
445 {
446 if (leftShoulder == null) { return; }
447 transformedHoldPos = leftShoulder.WorldAnchorA;
448 leftHand.Disabled = true;
449 }
450 itemPos.X *= Dir;
451 transformedHoldPos += Vector2.Transform(itemPos, Matrix.CreateRotationZ(itemAngle));
452 }
453
454 item.body.ResetDynamics();
455
456 Vector2 currItemPos = equippedInRightHand ?
457 rightHand.PullJointWorldAnchorA - transformedHandlePos[0] :
458 leftHand.PullJointWorldAnchorA - transformedHandlePos[1];
459
460 if (!MathUtils.IsValid(currItemPos))
461 {
462 string errorMsg = "Attempted to move the item \"" + item + "\" to an invalid position in HumanidAnimController.HoldItem: " +
463 currItemPos + ", rightHandPos: " + rightHand.PullJointWorldAnchorA + ", leftHandPos: " + leftHand.PullJointWorldAnchorA +
464 ", handlePos[0]: " + handlePos[0] + ", handlePos[1]: " + handlePos[1] +
465 ", transformedHandlePos[0]: " + transformedHandlePos[0] + ", transformedHandlePos[1]:" + transformedHandlePos[1] +
466 ", item pos: " + item.SimPosition + ", itemAngle: " + itemAngle +
467 ", collider pos: " + character.SimPosition;
468 DebugConsole.Log(errorMsg);
469 GameAnalyticsManager.AddErrorEventOnce(
470 "HumanoidAnimController.HoldItem:InvalidPos:" + character.Name + item.Name,
471 GameAnalyticsManager.ErrorSeverity.Error,
472 errorMsg);
473
474 return;
475 }
476
477 if (holdable.Pusher != null)
478 {
479 if (character.Stun > 0.0f || character.IsIncapacitated)
480 {
481 holdable.Pusher.Enabled = false;
482 }
483 else
484 {
485 if (!holdable.Pusher.Enabled)
486 {
487 holdable.Pusher.Enabled = true;
488 holdable.Pusher.ResetDynamics();
489 holdable.Pusher.SetTransform(currItemPos, itemAngle);
490 }
491 else
492 {
493 holdable.Pusher.TargetPosition = currItemPos;
494 holdable.Pusher.TargetRotation = holdAngle * Dir;
495
496 holdable.Pusher.MoveToTargetPosition(true);
497
498 currItemPos = holdable.Pusher.SimPosition;
499 itemAngle = holdable.Pusher.Rotation;
500 }
501 }
502 }
503 float targetAngle = MathUtils.WrapAngleTwoPi(itemAngle + itemAngleRelativeToHoldAngle * Dir);
504 float currentRotation = MathUtils.WrapAngleTwoPi(item.body.Rotation);
505 float itemRotation = MathHelper.SmoothStep(currentRotation, targetAngle, deltaTime * 25);
506 if (previousDirection != dir || Math.Abs(targetAngle - currentRotation) > MathHelper.Pi)
507 {
508 itemRotation = targetAngle;
509 }
510 item.SetTransform(currItemPos, itemRotation, setPrevTransform: false);
511 previousDirection = dir;
512
513 if (!isClimbing && !character.IsIncapacitated && itemPos != Vector2.Zero && (aim || !holdable.UseHandRotationForHoldAngle))
514 {
515 for (int i = 0; i < 2; i++)
516 {
517 if (!character.Inventory.IsInLimbSlot(item, i == 0 ? InvSlotType.RightHand : InvSlotType.LeftHand)) { continue; }
518#if DEBUG
519 if (handlePos[i].LengthSquared() > ArmLength)
520 {
521 DebugConsole.AddWarning($"Aim position for the item {item.Name} may be incorrect (further than the length of the character's arm)");
522 }
523#endif
524 HandIK(
525 i == 0 ? rightHand : leftHand, transformedHoldPos + transformedHandlePos[i],
526 CurrentAnimationParams.ArmIKStrength,
527 CurrentAnimationParams.HandIKStrength,
528 maxAngularVelocity: 15.0f);
529 }
530 }

533 private float GetAimWobble(Limb rightHand, Limb leftHand, Item heldItem)
534 {
535 float wobbleStrength = 0.0f;
536 if (character.Inventory?.GetItemInLimbSlot(InvSlotType.RightHand) == heldItem)
537 {
538 wobbleStrength += Character.CharacterHealth.GetLimbDamage(rightHand, afflictionType: "damage");
539 }
540 if (character.Inventory?.GetItemInLimbSlot(InvSlotType.LeftHand) == heldItem)
541 {
542 wobbleStrength += Character.CharacterHealth.GetLimbDamage(leftHand, afflictionType: "damage");
543 }
544 if (wobbleStrength <= 0.1f) { return 0.0f; }
545 wobbleStrength = (float)Math.Min(wobbleStrength, 1.0f);
546
547 float lowFreqNoise = PerlinNoise.GetPerlin((float)Timing.TotalTime / 320.0f, (float)Timing.TotalTime / 240.0f) - 0.5f;
548 float highFreqNoise = PerlinNoise.GetPerlin((float)Timing.TotalTime / 40.0f, (float)Timing.TotalTime / 50.0f) - 0.5f;
549
550 return (lowFreqNoise * 1.0f + highFreqNoise * 0.1f) * wobbleStrength;
551 }

553 public void HandIK(Limb hand, Vector2 pos, float armTorque = 1.0f, float handTorque = 1.0f, float maxAngularVelocity = float.PositiveInfinity)
554 {
555 Vector2 shoulderPos;
556
557 Limb arm, forearm;
558 if (hand.type == LimbType.LeftHand)
559 {
560 if (leftShoulder == null) { return; }
561 shoulderPos = leftShoulder.WorldAnchorA;
562 arm = GetLimb(LimbType.LeftArm);
563 forearm = GetLimb(LimbType.LeftForearm);
564 LeftHandIKPos = pos;
565 }
566 else
567 {
568 if (rightShoulder == null) { return; }
569 shoulderPos = rightShoulder.WorldAnchorA;
570 arm = GetLimb(LimbType.RightArm);
571 forearm = GetLimb(LimbType.RightForearm);
572 RightHandIKPos = pos;
573 }
574 if (arm == null) { return; }
575
576 //distance from shoulder to holdpos
577 float c = Vector2.Distance(pos, shoulderPos);
578 c = MathHelper.Clamp(c, Math.Abs(upperArmLength - forearmLength), forearmLength + upperArmLength - 0.01f);
579
580 float armAngle = MathUtils.VectorToAngle(pos - shoulderPos) + arm.Params.GetSpriteOrientation() - MathHelper.PiOver2;
581 float upperArmAngle = MathUtils.SolveTriangleSSS(forearmLength, upperArmLength, c) * Dir;
582 float lowerArmAngle = MathUtils.SolveTriangleSSS(upperArmLength, forearmLength, c) * Dir;
583
584 //make sure the arm angle "has the same number of revolutions" as the arm
585 while (arm.Rotation - armAngle > MathHelper.Pi)
586 {
587 armAngle += MathHelper.TwoPi;
588 }
589 while (arm.Rotation - armAngle < -MathHelper.Pi)
590 {
591 armAngle -= MathHelper.TwoPi;
592 }
593
594 if (arm?.body != null && Math.Abs(arm.body.AngularVelocity) < maxAngularVelocity)
595 {
596 arm.body.SmoothRotate(armAngle - upperArmAngle, 100.0f * armTorque * arm.Mass, wrapAngle: false);
597 }
598 float forearmAngle = armAngle + lowerArmAngle;
599 if (forearm?.body != null && Math.Abs(forearm.body.AngularVelocity) < maxAngularVelocity)
600 {
601 forearm.body.SmoothRotate(forearmAngle, 100.0f * handTorque * forearm.Mass, wrapAngle: false);
602 }
603 if (hand?.body != null && Math.Abs(hand.body.AngularVelocity) < maxAngularVelocity)
604 {
605 float handAngle = forearm != null ? forearmAngle : armAngle;
606 hand.body.SmoothRotate(handAngle, 10.0f * handTorque * hand.Mass, wrapAngle: false);
607 }
608 }

610 public void ApplyPose(Vector2 leftHandPos, Vector2 rightHandPos, Vector2 leftFootPos, Vector2 rightFootPos, float footMoveForce = 10)
611 {
612 var leftHand = GetLimb(LimbType.LeftHand);
613 var rightHand = GetLimb(LimbType.RightHand);
614 var waist = GetLimb(LimbType.Waist) ?? GetLimb(LimbType.Torso);
615 if (waist == null) { return; }
616 Vector2 midPos = waist.SimPosition;
617 if (leftHand != null)
618 {
619 leftHand.Disabled = true;
620 leftHandPos.X *= Dir;
621 leftHandPos += midPos;
622 HandIK(leftHand, leftHandPos);
623 }
624 if (rightHand != null)
625 {
626 rightHand.Disabled = true;
627 rightHandPos.X *= Dir;
628 rightHandPos += midPos;
629 HandIK(rightHand, rightHandPos);
630 }
631 var leftFoot = GetLimb(LimbType.LeftFoot);
632 if (leftFoot != null)
633 {
634 leftFoot.Disabled = true;
635 leftFootPos = new Vector2(waist.SimPosition.X + leftFootPos.X * Dir, GetColliderBottom().Y + leftFootPos.Y);
636 MoveLimb(leftFoot, leftFootPos, Math.Abs(leftFoot.SimPosition.X - leftFootPos.X) * footMoveForce * leftFoot.Mass, true);
637 }
638 var rightFoot = GetLimb(LimbType.RightFoot);
639 if (rightFoot != null)
640 {
641 rightFoot.Disabled = true;
642 rightFootPos = new Vector2(waist.SimPosition.X + rightFootPos.X * Dir, GetColliderBottom().Y + rightFootPos.Y);
643 MoveLimb(rightFoot, rightFootPos, Math.Abs(rightFoot.SimPosition.X - rightFootPos.X) * footMoveForce * rightFoot.Mass, true);
644 }

660 protected void CalculateArmLengths()
661 {
662 //calculate arm and forearm length (atm this assumes that both arms are the same size)
663 Limb rightForearm = GetLimb(LimbType.RightForearm);
664 Limb rightHand = GetLimb(LimbType.RightHand);
665 if (rightHand == null) { return; }
666
667 rightShoulder = GetJointBetweenLimbs(LimbType.Torso, LimbType.RightArm) ?? GetJointBetweenLimbs(LimbType.Head, LimbType.RightArm) ?? GetJoint(LimbType.RightArm, new LimbType[] { LimbType.RightHand, LimbType.RightForearm });
668 leftShoulder = GetJointBetweenLimbs(LimbType.Torso, LimbType.LeftArm) ?? GetJointBetweenLimbs(LimbType.Head, LimbType.LeftArm) ?? GetJoint(LimbType.LeftArm, new LimbType[] { LimbType.LeftHand, LimbType.LeftForearm });
669
670 Vector2 localAnchorShoulder = Vector2.Zero;
671 Vector2 localAnchorElbow = Vector2.Zero;
672 if (rightShoulder != null)
673 {
674 localAnchorShoulder = rightShoulder.LimbA.type == LimbType.RightArm ? rightShoulder.LocalAnchorA : rightShoulder.LocalAnchorB;
675 }
676 LimbJoint rightElbow = rightForearm == null ?
677 GetJointBetweenLimbs(LimbType.RightArm, LimbType.RightHand) :
678 GetJointBetweenLimbs(LimbType.RightArm, LimbType.RightForearm);
679 if (rightElbow != null)
680 {
