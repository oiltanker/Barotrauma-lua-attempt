25 public void UpdateAbilityGroup(float deltaTime)
26 {
27 if (!IsActive) { return; }
28 TimeSinceLastUpdate += deltaTime;
29 if (TimeSinceLastUpdate >= interval)
30 {
31 bool conditionsMatched = IsApplicable();
32 effectDelayTimer = conditionsMatched ? effectDelayTimer + TimeSinceLastUpdate : 0f;
33 conditionsMatched &= effectDelayTimer >= effectDelay;
34
35 foreach (var characterAbility in characterAbilities)
36 {
37 if (characterAbility.IsViable())
38 {
39 characterAbility.UpdateCharacterAbility(conditionsMatched, TimeSinceLastUpdate);
40 }
41 }
42 if (conditionsMatched)
43 {
44 timesTriggered++;
45 }
46 TimeSinceLastUpdate = 0;
47 }
48 }
49 private bool IsApplicable()
