314 public override void DragCharacter(Character target, float deltaTime)
315 {
316 if (target == null) { return; }
317 Limb mouthLimb = GetLimb(LimbType.Head);
318 if (mouthLimb == null) { return; }
319
320 if (GameMain.NetworkMember == null || !GameMain.NetworkMember.IsClient)
321 {
322 //stop dragging if there's something between the pull limb and the target
323 Vector2 sourceSimPos = SimplePhysicsEnabled ? character.SimPosition : mouthLimb.SimPosition;
324 Vector2 targetSimPos = target.SimPosition;
325 if (character.Submarine != null && character.SelectedCharacter.Submarine == null)
326 {
327 targetSimPos -= character.Submarine.SimPosition;
328 }
329 else if (character.Submarine == null && character.SelectedCharacter.Submarine != null)
330 {
331 sourceSimPos -= character.SelectedCharacter.Submarine.SimPosition;
332 }
333 var body = Submarine.CheckVisibility(sourceSimPos, targetSimPos, ignoreSubs: true);
334 if (body != null)
335 {
336 character.DeselectCharacter();
337 return;
338 }
339 }
340
341 float dmg = character.Params.EatingSpeed;
342 float eatSpeed = dmg / ((float)Math.Sqrt(Math.Max(target.Mass, 1)) * 10);
343 eatTimer += deltaTime * eatSpeed;
344
345 Vector2 mouthPos = SimplePhysicsEnabled ? character.SimPosition : GetMouthPosition().Value;
346 Vector2 attackSimPosition = character.Submarine == null ? ConvertUnits.ToSimUnits(target.WorldPosition) : target.SimPosition;
347
348 Vector2 limbDiff = attackSimPosition - mouthPos;
349 float extent = Math.Max(mouthLimb.body.GetMaxExtent(), 1);
350 bool tooFar = character.InWater ? limbDiff.LengthSquared() > extent * extent : limbDiff.X > extent;
351 if (tooFar)
352 {
353 character.SelectedCharacter = null;
354 }
355 else
356 {
357 //pull the target character to the position of the mouth
358 //(+ make the force fluctuate to waggle the character a bit)
359 float dragForce = MathHelper.Clamp(eatSpeed * 10, 0, 40);
360 if (dragForce > 0.1f)
361 {
362 Vector2 targetPos = mouthPos;
363 if (target.Submarine != null && character.Submarine == null)
364 {
365 targetPos -= target.Submarine.SimPosition;
366 }
367 else if (target.Submarine == null && character.Submarine != null)
368 {
369 targetPos += character.Submarine.SimPosition;
370 }
371 target.AnimController.MainLimb.body.SmoothRotate(mouthLimb.Rotation, dragForce * 2);
372 if (!target.AnimController.SimplePhysicsEnabled)
373 {
374 target.AnimController.MainLimb.MoveToPos(targetPos, (float)(Math.Sin(eatTimer) + dragForce));
375 }
376 target.AnimController.Collider.MoveToPos(targetPos, (float)(Math.Sin(eatTimer) + dragForce));
377 }
378
379 if (InWater)
380 {
381 //pull the character's mouth to the target character (again with a fluctuating force)
382 float pullStrength = (float)(Math.Sin(eatTimer) * Math.Max(Math.Sin(eatTimer * 0.5f), 0.0f));
383 mouthLimb.body.ApplyForce(limbDiff * mouthLimb.Mass * 50.0f * pullStrength);
384 }
385 else
386 {
387 float force = (float)Math.Sin(eatTimer * 100) * mouthLimb.Mass;
388 mouthLimb.body.ApplyLinearImpulse(Vector2.UnitY * force * 2, maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
389 mouthLimb.body.ApplyTorque(-force * 50);
390 }
391
392 if (Character.CanEat && target.IsDead)
393 {
394 var jaw = GetLimb(LimbType.Jaw);
395 if (jaw != null)
396 {
397 jaw.body.ApplyTorque(-(float)Math.Sin(eatTimer * 150) * jaw.Mass * 25);
398 }
399
400 character.ApplyStatusEffects(ActionType.OnEating, deltaTime);
401
402 float particleFrequency = MathHelper.Clamp(eatSpeed / 2, 0.02f, 0.5f);
403 if (Rand.Value() < particleFrequency / 6)
404 {
405 target.AnimController.MainLimb.AddDamage(target.SimPosition, dmg, 0, 0, false);
406 }
407 if (Rand.Value() < particleFrequency)
408 {
409 target.AnimController.MainLimb.AddDamage(target.SimPosition, 0, dmg, 0, false);
410 }
411 if (eatTimer % 1.0f < 0.5f && (eatTimer - deltaTime * eatSpeed) % 1.0f > 0.5f)
412 {
413 static bool CanBeSevered(LimbJoint j) => !j.IsSevered && j.CanBeSevered && j.LimbA != null && !j.LimbA.IsSevered && j.LimbB != null && !j.LimbB.IsSevered;
414 //keep severing joints until there is only one limb left
415 var nonSeveredJoints = target.AnimController.LimbJoints.Where(CanBeSevered);
416 if (nonSeveredJoints.None())
417 {
418 //small monsters don't eat the contents of the character's inventory
419 if (Mass < target.AnimController.Mass)
420 {
421 target.Inventory?.AllItemsMod.ForEach(it => it?.Drop(dropper: null));
422 }
423
424 //only one limb left, the character is now full eaten
425 Entity.Spawner?.AddEntityToRemoveQueue(target);
426
427 if (Character.AIController is EnemyAIController enemyAi)
428 {
429 enemyAi.PetBehavior?.OnEat(target);
430 }
431
432 character.SelectedCharacter = null;
433 }
434 else //sever a random joint
435 {
436 target.AnimController.SeverLimbJoint(nonSeveredJoints.GetRandomUnsynced());
437 }
438 }
439 }

445 void UpdateSineAnim(float deltaTime)
446 {
447 if (CurrentSwimParams == null) { return; }
448 movement = TargetMovement;
449 bool isMoving = movement.LengthSquared() > 0.00001f;
450 var mainLimb = MainLimb;
451 float t = 0.5f;
452 if (isMoving && !SimplePhysicsEnabled && CurrentSwimParams.RotateTowardsMovement)
453 {
454 Vector2 forward = VectorExtensions.Forward(Collider.Rotation + MathHelper.PiOver2);
455 float dot = Vector2.Dot(forward, Vector2.Normalize(movement));
456 if (dot < 0)
457 {
458 // Reduce the linear movement speed when not facing the movement direction
459 t = MathHelper.Clamp((1 + dot) / 10, 0.01f, 0.1f);
460 }
461 }
462 Collider.LinearVelocity = Vector2.Lerp(Collider.LinearVelocity, movement, t);
463 //limbs are disabled when simple physics is enabled, no need to move them
464 if (SimplePhysicsEnabled) { return; }
465 mainLimb.PullJointEnabled = true;
466
467 if (aiming && movement.Length() <= 0.1f)
468 {
469 Vector2 mousePos = ConvertUnits.ToSimUnits(character.CursorPosition);
470 Vector2 diff = (mousePos - (GetLimb(LimbType.Torso) ?? MainLimb).SimPosition) * Dir;
471 TargetMovement = new Vector2(0.0f, -0.1f);
472 float newRotation = MathUtils.VectorToAngle(diff);
473 Collider.SmoothRotate(newRotation, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
474 }
475
476 if (!isMoving)
477 {
478 WalkPos = MathHelper.SmoothStep(WalkPos, MathHelper.PiOver2, deltaTime * 5);
479 mainLimb.PullJointWorldAnchorB = Collider.SimPosition;
480 }
481 else
482 {
483 Vector2 transformedMovement = reverse ? -movement : movement;
484 float movementAngle = MathUtils.VectorToAngle(transformedMovement) - MathHelper.PiOver2;
485 float mainLimbAngle = 0;
486 if (mainLimb.type == LimbType.Torso && TorsoAngle.HasValue)
487 {
488 mainLimbAngle = TorsoAngle.Value;
489 }
490 else if (mainLimb.type == LimbType.Head && HeadAngle.HasValue)
491 {
492 mainLimbAngle = HeadAngle.Value;
493 }
494 mainLimbAngle *= Dir;
495 while (mainLimb.Rotation - (movementAngle + mainLimbAngle) > MathHelper.Pi)
496 {
497 movementAngle += MathHelper.TwoPi;
498 }
499 while (mainLimb.Rotation - (movementAngle + mainLimbAngle) < -MathHelper.Pi)
500 {
501 movementAngle -= MathHelper.TwoPi;
502 }
503 if (CurrentSwimParams.RotateTowardsMovement)
504 {
505 Collider.SmoothRotate(movementAngle, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
506 if (TorsoAngle.HasValue)
507 {
508 Limb torso = GetLimb(LimbType.Torso);
509 if (torso != null)
510 {
511 SmoothRotateWithoutWrapping(torso, movementAngle + TorsoAngle.Value * Dir, mainLimb, TorsoTorque);
512 }
513 }
514 if (HeadAngle.HasValue)
515 {
516 Limb head = GetLimb(LimbType.Head);
517 if (head != null)
518 {
519 SmoothRotateWithoutWrapping(head, movementAngle + HeadAngle.Value * Dir, mainLimb, HeadTorque);
520 }
521 }
522 if (TailAngle.HasValue)
523 {
524 bool isAngleApplied = false;
525 foreach (var limb in Limbs)
526 {
527 if (limb.IsSevered) { continue; }
528 if (limb.type != LimbType.Tail) { continue; }
529 if (!limb.Params.ApplyTailAngle) { continue; }
530 RotateTail(limb);
531 isAngleApplied = true;
532 }
533 if (!isAngleApplied)
534 {
535 RotateTail(GetLimb(LimbType.Tail));
536 }

538 void RotateTail(Limb tail)
539 {
540 if (tail == null) { return; }
541 float? mainLimbTargetAngle = null;
542 if (mainLimb.type == LimbType.Torso)
543 {
544 mainLimbTargetAngle = TorsoAngle;
545 }
546 else if (mainLimb.type == LimbType.Head)
547 {
548 mainLimbTargetAngle = HeadAngle;
549 }
550 float torque = TailTorque;
551 float maxMultiplier = CurrentSwimParams.TailTorqueMultiplier;
552 if (mainLimbTargetAngle.HasValue && maxMultiplier > 1)
553 {
554 float diff = Math.Abs(mainLimb.Rotation - tail.Rotation);
555 float offset = Math.Abs(mainLimbTargetAngle.Value - TailAngle.Value);
556 torque *= MathHelper.Lerp(1, maxMultiplier, MathUtils.InverseLerp(0, MathHelper.PiOver2, diff - offset));
557 }
558 SmoothRotateWithoutWrapping(tail, movementAngle + TailAngle.Value * Dir, mainLimb, torque);
559 }
560 }
561 }
562 else
563 {
564 movementAngle = Dir > 0 ? -MathHelper.PiOver2 : MathHelper.PiOver2;
565 if (reverse)
566 {
567 movementAngle = MathUtils.WrapAngleTwoPi(movementAngle - MathHelper.Pi);
568 }
569 if (mainLimb.type == LimbType.Head && HeadAngle.HasValue)
570 {
571 Collider.SmoothRotate(HeadAngle.Value * Dir, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
572 }
573 else if (mainLimb.type == LimbType.Torso && TorsoAngle.HasValue)
574 {
575 Collider.SmoothRotate(TorsoAngle.Value * Dir, CurrentSwimParams.SteerTorque * character.SpeedMultiplier);
576 }
577 if (TorsoAngle.HasValue)
578 {
579 Limb torso = GetLimb(LimbType.Torso);
580 torso?.body.SmoothRotate(TorsoAngle.Value * Dir, TorsoTorque);
581 }
582 if (HeadAngle.HasValue)
583 {
584 Limb head = GetLimb(LimbType.Head);
585 head?.body.SmoothRotate(HeadAngle.Value * Dir, HeadTorque);
586 }
587 if (TailAngle.HasValue)
588 {
589 bool isAngleApplied = false;
590 foreach (var limb in Limbs)
591 {
592 if (limb.IsSevered) { continue; }
593 if (limb.type != LimbType.Tail) { continue; }
594 if (!limb.Params.ApplyTailAngle) { continue; }
595 RotateTail(limb);
596 isAngleApplied = true;
597 }
598 if (!isAngleApplied)
599 {
600 RotateTail(GetLimb(LimbType.Tail));
601 }

603 void RotateTail(Limb tail)
604 {
605 if (tail != null)
606 {
607 tail.body.SmoothRotate(TailAngle.Value * Dir, TailTorque);
608 }
609 }
610 }
611 }
612
613 var waveLength = Math.Abs(CurrentSwimParams.WaveLength * RagdollParams.JointScale);
614 var waveAmplitude = Math.Abs(CurrentSwimParams.WaveAmplitude * character.SpeedMultiplier);
615 if (waveLength > 0 && waveAmplitude > 0)
616 {
617 WalkPos -= transformedMovement.Length() / Math.Abs(waveLength);
618 WalkPos = MathUtils.WrapAngleTwoPi(WalkPos);
619 }
620
621 foreach (var limb in Limbs)
622 {
623 if (limb.IsSevered) { continue; }
624 switch (limb.type)
625 {
626 case LimbType.LeftFoot:
627 case LimbType.RightFoot:
628 if (CurrentSwimParams.FootAnglesInRadians.ContainsKey(limb.Params.ID))
629 {
630 SmoothRotateWithoutWrapping(limb, movementAngle + CurrentSwimParams.FootAnglesInRadians[limb.Params.ID] * Dir, mainLimb, FootTorque);
631 }
632 break;
633 case LimbType.Tail:
634 if (waveLength > 0 && waveAmplitude > 0)
635 {
636 float waveRotation = (float)Math.Sin(WalkPos * limb.Params.SineFrequencyMultiplier);
637 limb.body.ApplyTorque(waveRotation * limb.Mass * waveAmplitude * limb.Params.SineAmplitudeMultiplier);
638 }
639 break;
640 }
641 }
642
643 for (int i = 0; i < Limbs.Length; i++)
644 {
645 var limb = Limbs[i];
646 if (limb.IsSevered) { continue; }
647 if (limb.SteerForce <= 0.0f) { continue; }
648 if (!Collider.PhysEnabled) { continue; }
649 Vector2 pullPos = limb.PullJointWorldAnchorA;
650 limb.body.ApplyForce(movement * limb.SteerForce * limb.Mass * Math.Max(character.SpeedMultiplier, 1), pullPos);
651 }
652
653 Vector2 mainLimbDiff = mainLimb.PullJointWorldAnchorB - mainLimb.SimPosition;
654 if (CurrentSwimParams.UseSineMovement)
655 {
656 mainLimb.PullJointWorldAnchorB = Vector2.SmoothStep(
657 mainLimb.PullJointWorldAnchorB,
658 Collider.SimPosition,
659 mainLimbDiff.LengthSquared() > 10.0f ? 1.0f : (float)Math.Abs(Math.Sin(WalkPos)));
660 }
661 else
662 {
663 //mainLimb.PullJointWorldAnchorB = Collider.SimPosition;
664 mainLimb.PullJointWorldAnchorB = Vector2.Lerp(
665 mainLimb.PullJointWorldAnchorB,
666 Collider.SimPosition,
667 mainLimbDiff.LengthSquared() > 10.0f ? 1.0f : 0.5f);
668 }
669 }
670
671 foreach (var limb in Limbs)
672 {
673 if (limb.IsSevered) { continue; }
674 if (Math.Abs(limb.Params.ConstantTorque) > 0)
675 {
676 float movementFactor = Math.Max(character.AnimController.Collider.LinearVelocity.Length() * 0.5f, 1);
677 limb.body.SmoothRotate(MainLimb.Rotation + MathHelper.ToRadians(limb.Params.ConstantAngle) * Dir, limb.Mass * limb.Params.ConstantTorque * movementFactor, wrapAngle: true);
678 }
679 if (limb.Params.BlinkFrequency > 0)
680 {
681 limb.UpdateBlink(deltaTime, MainLimb.Rotation);
682 }
683 }
684
685 floorY = Limbs[0].SimPosition.Y;
686 }

688 void UpdateWalkAnim(float deltaTime)
689 {
690 movement = MathUtils.SmoothStep(movement, TargetMovement, 0.2f);
691
692 Collider.LinearVelocity = new Vector2(
693 movement.X,
694 Collider.LinearVelocity.Y > 0.0f ? Collider.LinearVelocity.Y * 0.5f : Collider.LinearVelocity.Y);
695
696 //limbs are disabled when simple physics is enabled, no need to move them
697 if (SimplePhysicsEnabled) { return; }
698
699 Vector2 colliderBottom = GetColliderBottom();
700
701 float movementAngle = 0.0f;

898 void UpdateDying(float deltaTime)
899 {

901
902 float noise = (PerlinNoise.GetPerlin(WalkPos * 0.002f, WalkPos * 0.003f) - 0.5f) * 5.0f;

904
905 Limb baseLimb = GetLimb(LimbType.Head);
906 //if head is the main limb, it technically can't be severed - the rest of the limbs are considered severed if the head gets cut off

910 //if there's nothing connected to the head, don't make it wiggle by itself

914 {

917 if (connectedToTorsoCount > connectedToHeadCount)
918 {
919 baseLimb = torso;
920 }
921 }
922 }
923 else if (baseLimb == null)
924 {
925 baseLimb = GetLimb(LimbType.Torso, excludeSevered: true);

930
931 Limb tail = GetLimb(LimbType.Tail);
932 if (baseLimb != null) { baseLimb.body.ApplyTorque((float)(Math.Sqrt(baseLimb.Mass) * Dir * (Math.Sin(WalkPos) + noise)) * 30.0f * animStrength); }

935 WalkPos += deltaTime * 10.0f * animStrength;
936
937 Vector2 centerOfMass = GetCenterOfMass();
938
939 foreach (Limb limb in Limbs)
940 {
941 if (!connectedToBaseLimb.Contains(limb)) { continue; }
942#if CLIENT

973 }
