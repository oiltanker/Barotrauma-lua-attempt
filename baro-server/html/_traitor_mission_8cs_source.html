90 protected List<Tuple<Client, Character>> FindTraitorCandidates(GameServer server, CharacterTeamType team, RoleFilter traitorRoleFilter)
91 {
92 var traitorCandidates = new List<Tuple<Client, Character>>();
93 foreach (Client c in server.ConnectedClients)
94 {
95 var result = GameMain.LuaCs.Hook.Call<bool?>("traitor.findTraitorCandidate", c, team);
96 if (result != null && result.Value)
97 {
98 traitorCandidates.Add(Tuple.Create(c, c.Character));
99 continue;
100 }
101
102 if (c.Character == null || c.Character.IsDead || c.Character.Removed || !traitorRoleFilter(c.Character) ||
103 (team != CharacterTeamType.None && c.Character.TeamID != team))
104 {
105 continue;
106 }
107#if !ALLOW_NONHUMANOID_TRAITOR
108 if (!c.Character.IsHumanoid) { continue; }
109#endif
110 traitorCandidates.Add(Tuple.Create(c, c.Character));
111 }
112 return traitorCandidates;
113 }

125 protected List<Tuple<string, Tuple<Client, Character>>> AssignTraitors(GameServer server, TraitorManager traitorManager, CharacterTeamType team)
126 {
127 List<Character> characters = FindCharacters();
128#if !ALLOW_SOLO_TRAITOR
129 if (characters.Count < 2)
130 {
131 return null;
132 }
133#endif
134 var roleCandidates = new Dictionary<string, HashSet<Tuple<Client, Character>>>();
135 foreach (var role in Roles)
136 {
137 roleCandidates.Add(role.Key, new HashSet<Tuple<Client, Character>>(FindTraitorCandidates(server, team, role.Value)));
138 if (roleCandidates[role.Key].Count <= 0)
139 {
140 return null;
141 }
142 }
143 var candidateRoleCounts = new Dictionary<Tuple<Client, Character>, int>();
144 foreach (var candidateEntry in roleCandidates)
145 {
146 foreach (var candidate in candidateEntry.Value)
147 {
148 candidateRoleCounts[candidate] = candidateRoleCounts.TryGetValue(candidate, out var count) ? count + 1 : 1;
149 }
150 }
151 var unassignedRoles = new List<string>(roleCandidates.Keys);
152 unassignedRoles.Sort((a, b) => roleCandidates[a].Count - roleCandidates[b].Count);
153 var assignedCandidates = new List<Tuple<string, Tuple<Client, Character>>>();
154 while (unassignedRoles.Count > 0)
155 {
156 var currentRole = unassignedRoles[0];
157 var availableCandidates = roleCandidates[currentRole].ToList();
158 if (availableCandidates.Count <= 0)
159 {
160 break;
161 }
162 unassignedRoles.RemoveAt(0);
163 availableCandidates.Sort((a, b) => candidateRoleCounts[b] - candidateRoleCounts[a]);
164 unassignedRoles.Sort((a, b) => roleCandidates[a].Count - roleCandidates[b].Count);
165
166 int numCandidates = 1;
167 for (int i = 1; i < availableCandidates.Count && candidateRoleCounts[availableCandidates[i]] == candidateRoleCounts[availableCandidates[0]]; ++i)
168 {
169 ++numCandidates;
170 }
171
172 var selected = ToolBox.SelectWeightedRandom(availableCandidates, availableCandidates.Select(c => Math.Max(c.Item1.RoundsSincePlayedAsTraitor, 0.1f)).ToList(), TraitorManager.Random);
173 assignedCandidates.Add(Tuple.Create(currentRole, selected));
174 foreach (var candidate in roleCandidates.Values)
175 {
176 candidate.Remove(selected);
177 }
178 }
179 if (unassignedRoles.Count > 0)
180 {
181 return null;
182 }
183 return assignedCandidates;
184 }

199 public bool Start(GameServer server, TraitorManager traitorManager, CharacterTeamType team)
200 {
201 var assignedCandidates = AssignTraitors(server, traitorManager, team);
202 if (assignedCandidates == null)
203 {
204 return false;
205 }
206
207 foreach (Client client in server.ConnectedClients)
208 {
209 client.RoundsSincePlayedAsTraitor++;
210 }
211
212 Traitors.Clear();
213 foreach (var candidate in assignedCandidates)
214 {
215 var traitor = new Traitor(this, candidate.Item1, candidate.Item2.Item1.Character);
216 Traitors.Add(candidate.Item1, traitor);
217 candidate.Item2.Item1.RoundsSincePlayedAsTraitor = 0;
218 }
219 CodeWords = ToolBox.GetRandomLine(wordsTxt) + ", " + ToolBox.GetRandomLine(wordsTxt);
220 CodeResponse = ToolBox.GetRandomLine(wordsTxt) + ", " + ToolBox.GetRandomLine(wordsTxt);
221
222 if (pendingObjectives.Count <= 0 || !pendingObjectives[0].CanBeStarted(Traitors.Values))
223 {
224 Traitors.Clear();
225 return false;
226 }
227
228 var pendingMessages = new Dictionary<Traitor, List<string>>();
229 pendingMessages.Clear();
230 foreach (var traitor in Traitors.Values)
231 {
232 pendingMessages.Add(traitor, new List<string>());
233 }
234 foreach (var traitor in Traitors.Values)
235 {
236 traitor.Greet(server, CodeWords, CodeResponse, message => pendingMessages[traitor].Add(message));
237 GameMain.LuaCs.Hook.Call("traitor.traitorAssigned", new object[] { traitor });
238 }
239 pendingMessages.ForEach(traitor => traitor.Value.ForEach(message => traitor.Key.SendChatMessage(message, Identifier)));
240 pendingMessages.ForEach(traitor => traitor.Value.ForEach(message => traitor.Key.SendChatMessageBox(message, Identifier)));
241
242 Update(0.0f, () => { GameMain.Server.TraitorManager.ShouldEndRound = true; });
243#if SERVER
244 foreach (var traitor in Traitors.Values)
245 {
246 GameServer.Log($"{GameServer.CharacterLogName(traitor.Character)} is a traitor and the current goals are:\n{(traitor.CurrentObjective?.GoalInfos != null ? TextManager.GetServerMessage(traitor.CurrentObjective?.GoalInfos) : "(empty)")}", ServerLog.MessageType.ServerMessage);
247 }
248#endif
249 return true;

254 public void Update(float deltaTime, TraitorWinHandler winHandler)
255 {
256 if (pendingObjectives.Count <= 0 || Traitors.Count <= 0)
257 {
258 return;
259 }
260 if (Traitors.Values.Any(traitor => traitor.Character == null || traitor.Character.IsDead || traitor.Character.Removed))
261 {
262 Traitors.Values.ForEach(traitor => traitor.UpdateCurrentObjective("", Identifier));
263 pendingObjectives.Clear();
264 Traitors.Clear();
265 return;
266 }
267 var startedObjectives = new List<Objective>();
268 foreach (var traitor in Traitors.Values)
269 {
270 startedObjectives.Clear();
271 while (pendingObjectives.Count > 0)
272 {
273 var objective = GetCurrentObjective(traitor);
274 if (objective == null)
275 {
276 // No more objectives left for traitor or waiting for another traitor's objective.
277 break;
278 }
279 if (!objective.IsStarted)
280 {
281 if (!objective.Start(traitor))
282 {
283 //the mission fails if an objective cannot be started
284 if (completedObjectives.Count > 0)
285 {
286 objective.EndMessage();
287 }
288 pendingObjectives.Clear();
289 break;
290 }
291 startedObjectives.Add(objective);
292 }
293 objective.Update(deltaTime);
294 if (objective.IsCompleted)
295 {
296 pendingObjectives.Remove(objective);
297 completedObjectives.Add(objective);
298 objective.EndMessage();
299 continue;
300 }
301 if (objective.IsStarted && !objective.CanBeCompleted)
302 {
303 objective.EndMessage();
304 pendingObjectives.Clear();
305 }
306 break;
307 }
308 if (pendingObjectives.Count > 0)
309 {
310 startedObjectives.ForEach(objective => objective.StartMessage());
311 }
312 }
313 if (completedObjectives.Count >= allObjectives.Count)
314 {
315 foreach (var traitor in Traitors)
316 {
317 SteamAchievementManager.OnTraitorWin(traitor.Value.Character);
318 }
319 winHandler();
320 }
321 }

324 public List<Character> FindKillTarget(Character traitor, CharacterFilter filter, int count = -1, float percentage = -1f)
325 {
326 if (traitor == null) { return null; }
327
328 List<Character> validCharacters = Character.CharacterList.FindAll(c => c.TeamID == traitor.TeamID &&
329 c != traitor && !c.IsDead &&
330 (filter == null || filter(c)));
331
332 int targetCount = 1;
333 if (count > 0)
334 {
335 targetCount = count;
336 }
337 else if (percentage > 0f)
338 {
339 targetCount = (int)Math.Max(1, Math.Floor(validCharacters.Count * percentage));
340 }
341
342 List<Character> targetCharacters = new List<Character>();
343
344 if (validCharacters.Count > 0)
345 {
346 for (int i = 0; i < targetCount; i++)
347 {
348 if (validCharacters.Count == 0) break;
349 Character character = validCharacters[TraitorManager.RandomInt(validCharacters.Count)];
350 targetCharacters.Add(character);
351 validCharacters.Remove(character);
352 }
353 return targetCharacters;
354 }
355
356#if ALLOW_SOLO_TRAITOR
357 targetCharacters.Add(traitor);
358 return targetCharacters;
359#else
360 return null;
361#endif
362 }
