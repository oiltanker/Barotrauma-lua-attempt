249 {
250 this.item = item;
251 originalElement = element;
252 name = element.Name.ToString();
253 SerializableProperties = SerializableProperty.GetProperties(this);
254 requiredItems = new Dictionary<RelatedItem.RelationType, List<RelatedItem>>();
255 requiredSkills = new List<Skill>();
256
257#if CLIENT
258 hasSoundsOfType = new bool[Enum.GetValues(typeof(ActionType)).Length];
259 sounds = new Dictionary<ActionType, List<ItemSound>>();
260#endif
261
262 SelectKey = InputType.Select;
263
264 try
265 {
266 string selectKeyStr = element.GetAttributeString("selectkey", "Select");
267 selectKeyStr = ToolBox.ConvertInputType(selectKeyStr);
268 SelectKey = (InputType)Enum.Parse(typeof(InputType), selectKeyStr, true);
269 }
270 catch (Exception e)
271 {
272 DebugConsole.ThrowError("Invalid select key in " + element + "!", e);
273 }
274
275 PickKey = InputType.Select;
276
277 try
278 {
279 string pickKeyStr = element.GetAttributeString("pickkey", "Select");
280 pickKeyStr = ToolBox.ConvertInputType(pickKeyStr);
281 PickKey = (InputType)Enum.Parse(typeof(InputType), pickKeyStr, true);
282 }
283 catch (Exception e)
284 {
285 DebugConsole.ThrowError("Invalid pick key in " + element + "!", e);
286 }
287
288 SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
289 ParseMsg();
290
291 string inheritRequiredSkillsFrom = element.GetAttributeString("inheritrequiredskillsfrom", "");
292 if (!string.IsNullOrEmpty(inheritRequiredSkillsFrom))
293 {
294 var component = item.Components.Find(ic => ic.Name.Equals(inheritRequiredSkillsFrom, StringComparison.OrdinalIgnoreCase));
295 if (component == null)
296 {
297 DebugConsole.ThrowError($"Error in item \"{item.Name}\" - component \"{name}\" is set to inherit its required skills from \"{inheritRequiredSkillsFrom}\", but a component of that type couldn't be found.");
298 }
299 else
300 {
301 requiredSkills = component.requiredSkills;
302 }
303 }
304
305 string inheritStatusEffectsFrom = element.GetAttributeString("inheritstatuseffectsfrom", "");
306 if (!string.IsNullOrEmpty(inheritStatusEffectsFrom))
307 {
308 var component = item.Components.Find(ic => ic.Name.Equals(inheritStatusEffectsFrom, StringComparison.OrdinalIgnoreCase));
309 if (component == null)
310 {
311 DebugConsole.ThrowError($"Error in item \"{item.Name}\" - component \"{name}\" is set to inherit its StatusEffects from \"{inheritStatusEffectsFrom}\", but a component of that type couldn't be found.");
312 }
313 else if (component.statusEffectLists != null)
314 {
315 statusEffectLists ??= new Dictionary<ActionType, List<StatusEffect>>();
316 foreach (KeyValuePair<ActionType, List<StatusEffect>> kvp in component.statusEffectLists)
317 {
318 if (!statusEffectLists.TryGetValue(kvp.Key, out List<StatusEffect> effectList))
319 {
320 effectList = new List<StatusEffect>();
321 statusEffectLists.Add(kvp.Key, effectList);
322 }
323 effectList.AddRange(kvp.Value);
324 }
325 }
326 }
327
328 foreach (var subElement in element.Elements())
329 {
330 switch (subElement.Name.ToString().ToLowerInvariant())
331 {
332 case "activeconditional":
333 case "isactive":
334 IsActiveConditionals = IsActiveConditionals ?? new List<PropertyConditional>();
335 foreach (XAttribute attribute in subElement.Attributes())
336 {
337 if (PropertyConditional.IsValid(attribute))
338 {
339 IsActiveConditionals.Add(new PropertyConditional(attribute));
340 }
341 }
342 break;
343 case "requireditem":
344 case "requireditems":
345 SetRequiredItems(subElement);
346 break;
347 case "requiredskill":
348 case "requiredskills":
349 if (subElement.GetAttribute("name") != null)
350 {
351 DebugConsole.ThrowError("Error in item config \"" + item.ConfigFilePath + "\" - skill requirement in component " + GetType().ToString() + " should use a skill identifier instead of the name of the skill.");
352 continue;
353 }
354
355 Identifier skillIdentifier = subElement.GetAttributeIdentifier("identifier", "");
356 requiredSkills.Add(new Skill(skillIdentifier, subElement.GetAttributeInt("level", 0)));
357 break;
358 case "statuseffect":
359 statusEffectLists ??= new Dictionary<ActionType, List<StatusEffect>>();
360 LoadStatusEffect(subElement);
361 break;
362 default:
363 if (LoadElemProjSpecific(subElement)) { break; }
364 ItemComponent ic = Load(subElement, item, false);
365 if (ic == null) { break; }
366
367 ic.Parent = this;
368 ic.IsActive = isActive;
369 OnActiveStateChanged += ic.SetActiveState;
370
371 item.AddComponent(ic);
372 break;
373 }
374 }

503 public virtual bool Combine(Item item, Character user)
504 {
505 if (canBeCombined && this.item.Prefab == item.Prefab &&
506 item.Condition > 0.0f && this.item.Condition > 0.0f &&
507 !item.IsFullCondition && !this.item.IsFullCondition)
508 {
509 float transferAmount = Math.Min(item.Condition, this.item.MaxCondition - this.item.Condition);
510
511 if (MathUtils.NearlyEqual(transferAmount, 0.0f)) { return false; }
512 if (removeOnCombined)
513 {
514 if (item.Condition - transferAmount <= 0.0f)
515 {
516 if (item.ParentInventory != null)
517 {
518 if (item.ParentInventory.Owner is Character owner && owner.HeldItems.Contains(item))
519 {
520 item.Unequip(owner);
521 }
522 item.ParentInventory.RemoveItem(item);
523 }
524 Entity.Spawner.AddItemToRemoveQueue(item);
525 }
526 else
527 {
528 item.Condition -= transferAmount;
529 }
530 if (this.Item.Condition + transferAmount <= 0.0f)
531 {
532 if (this.Item.ParentInventory != null)
533 {
534 if (this.Item.ParentInventory.Owner is Character owner && owner.HeldItems.Contains(this.Item))
535 {
536 this.Item.Unequip(owner);
537 }
538 this.Item.ParentInventory.RemoveItem(this.Item);
539 }
540 Entity.Spawner.AddItemToRemoveQueue(this.Item);
541 }
542 else
543 {
544 this.Item.Condition += transferAmount;
545 }
546 }
547 else
548 {
549 this.Item.Condition += transferAmount;
550 item.Condition -= transferAmount;
551 }
552 return true;
553 }
554 return false;
555 }

625 public bool HasRequiredSkills(Character character, out Skill insufficientSkill)
626 {
627 foreach (Skill skill in requiredSkills)
628 {
629 float characterLevel = character.GetSkillLevel(skill.Identifier);
630 if (characterLevel < skill.Level * GetSkillMultiplier())
631 {
632 insufficientSkill = skill;
633 return false;
634 }
635 }
636 insufficientSkill = null;
637 return true;

655 public float DegreeOfSuccess(Character character, List<Skill> requiredSkills)
656 {
657 if (requiredSkills.Count == 0) return 1.0f;
658
659 if (character == null)
660 {
661 string errorMsg = "ItemComponent.DegreeOfSuccess failed (character was null).\n" + Environment.StackTrace.CleanupStackTrace();
662 DebugConsole.ThrowError(errorMsg);
663 GameAnalyticsManager.AddErrorEventOnce("ItemComponent.DegreeOfSuccess:CharacterNull", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
664 return 0.0f;
665 }
666
667 float skillSuccessSum = 0.0f;
668 for (int i = 0; i < requiredSkills.Count; i++)
669 {
670 float characterLevel = character.GetSkillLevel(requiredSkills[i].Identifier);
671 skillSuccessSum += (characterLevel - requiredSkills[i].Level);
672 }
673 float average = skillSuccessSum / requiredSkills.Count;
674
675 return ((average + 100.0f) / 2.0f) / 100.0f;
676 }

730 public virtual bool HasRequiredItems(Character character, bool addMessage, LocalizedString msg = null)
731 {
732 if (requiredItems.None()) { return true; }
733 if (character.Inventory == null) { return false; }
734 bool hasRequiredItems = false;
735 bool canContinue = true;
736 if (requiredItems.ContainsKey(RelatedItem.RelationType.Equipped))
737 {
738 foreach (RelatedItem ri in requiredItems[RelatedItem.RelationType.Equipped])
739 {
740 canContinue = CheckItems(ri, character.HeldItems);
741 if (!canContinue) { break; }
742 }
743 }
744 if (canContinue)
745 {
746 if (requiredItems.ContainsKey(RelatedItem.RelationType.Picked))
747 {
748 foreach (RelatedItem ri in requiredItems[RelatedItem.RelationType.Picked])
749 {
750 if (!CheckItems(ri, character.Inventory.AllItems)) { break; }
751 }
752 }
753 }
754
755#if CLIENT
756 if (!hasRequiredItems && addMessage && !msg.IsNullOrEmpty())
757 {
758 GUI.AddMessage(msg, Color.Red);
759 }
760#endif
761 return hasRequiredItems;

763 bool CheckItems(RelatedItem relatedItem, IEnumerable<Item> itemList)
764 {
765 bool Predicate(Item it)
766 {
767 if (it == null || it.Condition <= 0.0f || !relatedItem.MatchesItem(it)) { return false; }
768 if (item.Submarine != null)
769 {
770 var idCard = it.GetComponent<IdCard>();
771 if (idCard != null)
772 {
773 //id cards don't work in enemy subs (except on items that only require the default "idcard" tag)
774 if (idCard.TeamID != CharacterTeamType.None && idCard.TeamID != item.Submarine.TeamID && relatedItem.Identifiers.Any(id => id != "idcard"))
775 {
776 return false;
777 }
778 else if (idCard.SubmarineSpecificID != 0 && item.Submarine.SubmarineSpecificIDTag != idCard.SubmarineSpecificID)
779 {
780 return false;
781 }
782 }
783 }
784 return true;
785 };
786 bool shouldBreak = false;
787 bool inEditor = false;
788#if CLIENT
789 inEditor = Screen.Selected == GameMain.SubEditorScreen;
790#endif
791 if (relatedItem.IgnoreInEditor && inEditor)
792 {
793 hasRequiredItems = true;
794 }
795 else if (relatedItem.IsOptional)
796 {
797 if (!hasRequiredItems)
798 {
799 hasRequiredItems = itemList.Any(Predicate);
800 }
801 }
802 else
803 {
804 hasRequiredItems = itemList.Any(Predicate);
805 if (!hasRequiredItems)
806 {
807 shouldBreak = true;
808 }
809 }
810 if (!hasRequiredItems)
811 {
812 if (msg == null && !relatedItem.Msg.IsNullOrEmpty())
813 {
814 msg = relatedItem.Msg;
815 }
816 }
817 return !shouldBreak;
818 }

821 public void ApplyStatusEffects(ActionType type, float deltaTime, Character character = null, Limb targetLimb = null, Entity useTarget = null, Character user = null, Vector2? worldPosition = null, float applyOnUserFraction = 0.0f)
822 {
823 if (statusEffectLists == null) { return; }
824
825 if (!statusEffectLists.TryGetValue(type, out List<StatusEffect> statusEffects)) { return; }
826
827 bool broken = item.Condition <= 0.0f;
828 bool reducesCondition = false;
829 foreach (StatusEffect effect in statusEffects)
830 {
831 if (broken && !effect.AllowWhenBroken && effect.type != ActionType.OnBroken) { continue; }
832 if (user != null) { effect.SetUser(user); }
833 item.ApplyStatusEffect(effect, type, deltaTime, character, targetLimb, useTarget, isNetworkEvent: false, checkCondition: false, worldPosition);
834 if (user != null && applyOnUserFraction > 0.0f && effect.HasTargetType(StatusEffect.TargetType.Character))
835 {
836 effect.AfflictionMultiplier = applyOnUserFraction;
837 item.ApplyStatusEffect(effect, type, deltaTime, user, targetLimb == null ? null : user.AnimController.GetLimb(targetLimb.type), useTarget, false, false, worldPosition);
838 effect.AfflictionMultiplier = 1.0f;
839 }
840 reducesCondition |= effect.ReducesItemCondition();
841 }
842 //if any of the effects reduce the item's condition, set the user for OnBroken effects as well
843 if (reducesCondition && user != null && type != ActionType.OnBroken)
844 {
845 foreach (ItemComponent ic in item.Components)
846 {
847 if (ic.statusEffectLists == null || !ic.statusEffectLists.TryGetValue(ActionType.OnBroken, out List<StatusEffect> brokenEffects)) { continue; }
848 foreach (var brokenEffect in brokenEffects)
849 {
850 brokenEffect.SetUser(user);
851 }
852 }

891 public static ItemComponent Load(ContentXElement element, Item item, bool errorMessages = true)
892 {
893 Type type;
894 Identifier typeName = element.NameAsIdentifier();
895 try
896 {
897 // Get the type of a specified class.
898 type = ReflectionUtils.GetDerivedNonAbstract<ItemComponent>().Append(typeof(ItemComponent)).FirstOrDefault(t => t.Name == typeName);
899 if (type == null)
900 {
901 if (errorMessages)
902 {
903 DebugConsole.ThrowError($"Could not find the component \"{typeName}\" ({item.Prefab.ContentFile.Path})");
904 }
905 return null;
906 }
907 }
908 catch (Exception e)
909 {
910 if (errorMessages)
911 {
912 DebugConsole.ThrowError($"Could not find the component \"{typeName}\" ({item.Prefab.ContentFile.Path})", e);
913 }
914 return null;
915 }
916
917 ConstructorInfo constructor;
918 try
919 {
920 if (type != typeof(ItemComponent) && !type.IsSubclassOf(typeof(ItemComponent))) { return null; }
921 constructor = type.GetConstructor(new Type[] { typeof(Item), typeof(ContentXElement) });
922 if (constructor == null)
923 {
924 DebugConsole.ThrowError(
925 $"Could not find the constructor of the component \"{typeName}\" ({item.Prefab.ContentFile.Path})");
926 return null;
927 }
928 }
929 catch (Exception e)
930 {
931 DebugConsole.ThrowError(
932 $"Could not find the constructor of the component \"{typeName}\" ({item.Prefab.ContentFile.Path})", e);
933 return null;
934 }
935 ItemComponent ic = null;
936 try
937 {
938 object[] lobject = new object[] { item, element };
939 object component = constructor.Invoke(lobject);
940 ic = (ItemComponent)component;
941 ic.name = element.Name.ToString();
942 }
943 catch (TargetInvocationException e)
944 {
945 DebugConsole.ThrowError($"Error while loading component of the type {type}.", e.InnerException);
946 GameAnalyticsManager.AddErrorEventOnce(
947 $"ItemComponent.Load:TargetInvocationException{item.Name}{element.Name}",
948 GameAnalyticsManager.ErrorSeverity.Error,
949 $"Error while loading entity of the type {type} ({e.InnerException})\n{Environment.StackTrace.CleanupStackTrace()}");
950 }
951
952 return ic;
953 }

990 private void OverrideRequiredItems(ContentXElement element)
991 {
992 var prevRequiredItems = new Dictionary<RelatedItem.RelationType, List<RelatedItem>>(requiredItems);
993 requiredItems.Clear();
994
995 bool returnEmptyRequirements = false;
996#if CLIENT
997 returnEmptyRequirements = Screen.Selected == GameMain.SubEditorScreen;
998#endif
999 foreach (var subElement in element.Elements())
1000 {
1001 switch (subElement.Name.ToString().ToLowerInvariant())
1002 {
1003 case "requireditem":
1004 case "requireditems":
1005 RelatedItem newRequiredItem = RelatedItem.Load(subElement, returnEmptyRequirements, item.Name);
1006 if (newRequiredItem == null) continue;
1007
1008 var prevRequiredItem = prevRequiredItems.ContainsKey(newRequiredItem.Type) ?
1009 prevRequiredItems[newRequiredItem.Type].Find(ri => ri.JoinedIdentifiers == newRequiredItem.JoinedIdentifiers) : null;
1010 if (prevRequiredItem != null)
1011 {
1012 newRequiredItem.statusEffects = prevRequiredItem.statusEffects;
1013 newRequiredItem.Msg = prevRequiredItem.Msg;
1014 newRequiredItem.IsOptional = prevRequiredItem.IsOptional;
1015 newRequiredItem.IgnoreInEditor = prevRequiredItem.IgnoreInEditor;
1016 }
1017
1018 if (!requiredItems.ContainsKey(newRequiredItem.Type))
1019 {
1020 requiredItems[newRequiredItem.Type] = new List<RelatedItem>();
1021 }
1022 requiredItems[newRequiredItem.Type].Add(newRequiredItem);
1023 break;
1024 }
1025 }
