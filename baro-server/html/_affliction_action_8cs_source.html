36 public override void Update(float deltaTime)
37 {
38 if (isFinished) { return; }
39 var afflictionPrefab = AfflictionPrefab.List.FirstOrDefault(p => p.Identifier == Affliction);
40 if (afflictionPrefab != null)
41 {
42 var targets = ParentEvent.GetTargets(TargetTag);
43 foreach (var target in targets)
44 {
45 if (target != null && target is Character character)
46 {
47 if (LimbType != LimbType.None)
48 {
49 var limb = character.AnimController.GetLimb(LimbType);
50 if (Strength > 0.0f)
51 {
52 character.CharacterHealth.ApplyAffliction(limb, afflictionPrefab.Instantiate(Strength));
53 }
54 else if (Strength < 0.0f)
55 {
56 character.CharacterHealth.ReduceAfflictionOnLimb(limb, Affliction, -Strength);
57 }
58 }
59 else
60 {
61 if (Strength > 0.0f)
62 {
63 character.CharacterHealth.ApplyAffliction(null, afflictionPrefab.Instantiate(Strength));
64 }
65 else if (Strength < 0.0f)
66 {
67 character.CharacterHealth.ReduceAfflictionOnAllLimbs(Affliction, -Strength);
68 }
69 }
70 }
71 }
72 }
73 isFinished = true;
74 }
