76 public FileTransferOut(NetworkConnection recipient, FileTransferType fileType, string filePath)
77 {
78 Connection = recipient;
79
80 FileType = fileType;
81 FilePath = filePath;
82 FileName = Path.GetFileName(filePath);
83
84 Acknowledged = false;
85 SentOffset = 0;
86 KnownReceivedOffset = 0;
87
88 Status = FileTransferStatus.NotStarted;
89
90 StartingTime = DateTime.Now;
91
92 int maxRetries = 4;
93 for (int i = 0; i <= maxRetries; i++)
94 {
95 try
96 {
97 Data = File.ReadAllBytes(filePath);
98 }
99 catch (System.IO.IOException e)
100 {
101 if (i >= maxRetries) { throw; }
102 DebugConsole.NewMessage("Failed to initiate a file transfer {" + e.Message + "}, retrying in 250 ms...", Color.Red);
103 Thread.Sleep(250);
104 }
105 }
106 }

138 public FileTransferOut StartTransfer(NetworkConnection recipient, FileTransferType fileType, string filePath)
139 {
140 if (activeTransfers.Count >= MaxTransferCount)
141 {
142 return null;
143 }
144
145 if (activeTransfers.Count(t => t.Connection == recipient) > MaxTransferCountPerRecipient)
146 {
147 return null;
148 }
149
150 if (!File.Exists(filePath))
151 {
152 DebugConsole.ThrowError("Failed to initiate file transfer (file \"" + filePath + "\" not found).\n" + Environment.StackTrace);
153 return null;
154 }
155
156 FileTransferOut transfer = null;
157 try
158 {
159 transfer = new FileTransferOut(recipient, fileType, filePath)
160 {
161 ID = 1
162 };
163 while (activeTransfers.Any(t => t.Connection == recipient && t.ID == transfer.ID))
164 {
165 transfer.ID++;
166 }
167 activeTransfers.Add(transfer);
168 }
169 catch (Exception e)
170 {
171 DebugConsole.ThrowError("Failed to initiate file transfer", e);
172 return null;
173 }
174
175 OnStarted(transfer);
176 GameMain.Server.LastClientListUpdateID++;
177
178 return transfer;
179 }

181 public void Update(float deltaTime)
182 {
183 int numRemoved = activeTransfers.RemoveAll(t => t.Connection.Status != NetworkConnectionStatus.Connected);
184
185 var endedTransfers = activeTransfers.FindAll(t =>
186 t.Connection.Status != NetworkConnectionStatus.Connected ||
187 t.Status == FileTransferStatus.Finished ||
188 t.Status == FileTransferStatus.Canceled ||
189 t.Status == FileTransferStatus.Error);
190
191 foreach (FileTransferOut transfer in endedTransfers)
192 {
193 activeTransfers.Remove(transfer);
194 OnEnded(transfer);
195 }
196
197 foreach (FileTransferOut transfer in activeTransfers)
198 {
199 transfer.WaitTimer -= deltaTime;
200 if (transfer.WaitTimer > 0.0f) { continue; }
201 Send(transfer);
202 }
203
204 if (numRemoved > 0 || endedTransfers.Count > 0)
205 {
206 GameMain.Server.LastClientListUpdateID++;
207 }
208 }

210 private void Send(FileTransferOut transfer)
211 {
212 // send another part of the file
213 IWriteMessage message;
214
215 try
216 {
217 //first message; send length, file name etc
218 //wait for acknowledgement before sending data
219 if (!transfer.Acknowledged)
220 {
221 message = new WriteOnlyMessage();
222 message.Write((byte)ServerPacketHeader.FILE_TRANSFER);
223
224 //if the recipient is the owner of the server (= a client running the server from the main exe)
225 //we don't need to send anything, the client can just read the file directly
226 if (transfer.Connection == GameMain.Server.OwnerConnection)
227 {
228 message.Write((byte)FileTransferMessageType.TransferOnSameMachine);
229 message.Write((byte)transfer.ID);
230 message.Write((byte)transfer.FileType);
231 message.Write(transfer.FilePath);
232 peer.Send(message, transfer.Connection, DeliveryMethod.Unreliable);
233 transfer.Status = FileTransferStatus.Finished;
234 }
235 else
236 {
237 message.Write((byte)FileTransferMessageType.Initiate);
238 message.Write((byte)transfer.ID);
239 message.Write((byte)transfer.FileType);
240 //message.Write((ushort)chunkLen);
241 message.Write(transfer.Data.Length);
242 message.Write(transfer.FileName);
243 peer.Send(message, transfer.Connection, DeliveryMethod.Unreliable);
244
245 transfer.Status = FileTransferStatus.Sending;
246
247 if (GameSettings.CurrentConfig.VerboseLogging)
248 {
249 DebugConsole.Log("Sending file transfer initiation message: ");
250 DebugConsole.Log(" File: " + transfer.FileName);
251 DebugConsole.Log(" Size: " + transfer.Data.Length);
252 DebugConsole.Log(" ID: " + transfer.ID);
253 }
254 }
255 transfer.WaitTimer = 0.1f;
256 return;
257 }
258
259 for (int i = 0; i < Math.Floor(transfer.PacketsPerUpdate); i++)
260 {
261 long remaining = transfer.Data.Length - transfer.SentOffset;
262 int sendByteCount = (remaining > chunkLen ? chunkLen : (int)remaining);
263
264 message = new WriteOnlyMessage();
265 message.Write((byte)ServerPacketHeader.FILE_TRANSFER);
266 message.Write((byte)FileTransferMessageType.Data);
267
268 message.Write((byte)transfer.ID);
269 message.Write(transfer.SentOffset);
270
271 message.Write((ushort)sendByteCount);
272 int chunkDestPos = message.BytePosition;
273 message.BitPosition += sendByteCount * 8;
274 message.LengthBits = Math.Max(message.LengthBits, message.BitPosition);
275 Array.Copy(transfer.Data, transfer.SentOffset, message.Buffer, chunkDestPos, sendByteCount);
276
277 transfer.SentOffset += sendByteCount;
278
279 peer.Send(message, transfer.Connection, DeliveryMethod.Unreliable, compressPastThreshold: false);
280
281 if (GameSettings.CurrentConfig.VerboseLogging)
282 {
283 DebugConsole.Log($"Sending {sendByteCount} bytes of the file {transfer.FileName} ({transfer.SentOffset / 1000}/{transfer.Data.Length / 1000} kB sent)");
284 }
285
286 if (transfer.SentOffset >= transfer.Data.Length)
287 {
288 transfer.SentOffset = transfer.KnownReceivedOffset;
289 transfer.WaitTimer = 1.0f;
290 }
291
292 //try to increase the packet rate so large files get sent faster,
293 //this gets reset when packet loss or disorder sets in
294 transfer.PacketsPerUpdate = Math.Min(FileTransferOut.MaxPacketsPerUpdate,
295 transfer.PacketsPerUpdate + 0.05f);
296 }
297
298#if DEBUG
299 transfer.WaitTimer = Math.Max(transfer.WaitTimer, StallPacketsTime);
300#endif
301 }
302
303 catch (Exception e)
304 {
305 DebugConsole.ThrowError("FileSender threw an exception when trying to send data", e);
306 GameAnalyticsManager.AddErrorEventOnce(
307 "FileSender.Update:Exception",
308 GameAnalyticsManager.ErrorSeverity.Error,
309 "FileSender threw an exception when trying to send data:\n" + e.Message + "\n" + e.StackTrace.CleanupStackTrace());
310 transfer.Status = FileTransferStatus.Error;
311 return;
312 }

325 public void ReadFileRequest(IReadMessage inc, Client client)
326 {
327 FileTransferMessageType messageType = (FileTransferMessageType)inc.ReadByte();
328
329 if (messageType == FileTransferMessageType.Cancel)
330 {
331 byte transferId = inc.ReadByte();
332 var matchingTransfer = activeTransfers.Find(t => t.Connection == inc.Sender && t.ID == transferId);
333 if (matchingTransfer != null) CancelTransfer(matchingTransfer);
334 return;
335 }
336 else if (messageType == FileTransferMessageType.Data)
337 {
338 byte transferId = inc.ReadByte();
339 var matchingTransfer = activeTransfers.Find(t => t.Connection == inc.Sender && t.ID == transferId);
340 if (matchingTransfer != null)
341 {
342 matchingTransfer.Acknowledged = true;
343 int expecting = inc.ReadInt32(); //the offset the client is waiting for
344 int lastSeen = Math.Min(matchingTransfer.SentOffset, inc.ReadInt32()); //the last offset the client got from us
345 matchingTransfer.KnownReceivedOffset = Math.Max(expecting, matchingTransfer.KnownReceivedOffset);
346 if (matchingTransfer.SentOffset < matchingTransfer.KnownReceivedOffset)
347 {
348 matchingTransfer.WaitTimer = 0.0f;
349 matchingTransfer.SentOffset = matchingTransfer.KnownReceivedOffset;
350 }
351
352 if (lastSeen - matchingTransfer.KnownReceivedOffset >= chunkLen * 10 ||
353 matchingTransfer.SentOffset >= matchingTransfer.Data.Length)
354 {
355 matchingTransfer.SentOffset = matchingTransfer.KnownReceivedOffset;
356 matchingTransfer.WaitTimer = 1.0f;
357 }
358
359 if (matchingTransfer.KnownReceivedOffset >= matchingTransfer.Data.Length)
360 {
361 matchingTransfer.Status = FileTransferStatus.Finished;
362 }
363 }
364 return;
365 }
366
367 FileTransferType fileType = (FileTransferType)inc.ReadByte();
368 switch (fileType)
369 {
370 case FileTransferType.Submarine:
371 string fileName = inc.ReadString();
372 string fileHash = inc.ReadString();
373 var requestedSubmarine = SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Name == fileName && s.MD5Hash.StringRepresentation == fileHash);
374
375 DebugConsole.Log($"Received a submarine file request from \"{client.Name}\" ({fileName}).");
376
377 if (requestedSubmarine != null)
378 {
379 if (activeTransfers.Any(t => t.Connection == inc.Sender && t.FilePath == requestedSubmarine.FilePath))
380 {
381 DebugConsole.Log($"Ignoring a submarine file request from \"{client.Name}\" ({fileName}) - already transferring.");
382 return;
383 }
384 StartTransfer(inc.Sender, FileTransferType.Submarine, requestedSubmarine.FilePath);
385 }
386 break;
387 case FileTransferType.CampaignSave:
388 if (GameMain.GameSession != null &&
389 !ActiveTransfers.Any(t => t.Connection == inc.Sender && t.FileType == FileTransferType.CampaignSave))
390 {
391 StartTransfer(inc.Sender, FileTransferType.CampaignSave, GameMain.GameSession.SavePath);
392 if (GameMain.GameSession?.GameMode is MultiPlayerCampaign campaign)
393 {
394 client.LastCampaignSaveSendTime = new Pair<ushort, float>(campaign.LastSaveID, (float)Lidgren.Network.NetTime.Now);
395 }
396 }
397 break;
398 case FileTransferType.Mod:
399 string modName = inc.ReadString();
400 Md5Hash modHash = Md5Hash.StringAsHash(inc.ReadString());
401
402 DebugConsole.Log($"Received a mod file request from \"{client.Name}\" ({modName}).");
403
404 if (!GameMain.Server.ServerSettings.AllowModDownloads) { return; }
405 if (!(GameMain.Server.ModSender is { Ready: true })) { return; }
406
407 ContentPackage mod = ContentPackageManager.AllPackages.FirstOrDefault(p => p.Hash.Equals(modHash));
408
409 if (mod is null) { return; }
410
411 string modCompressedPath = ModSender.GetCompressedModPath(mod);
412 if (!File.Exists(modCompressedPath)) { return; }
413
414 if (activeTransfers.Any(t => t.Connection == inc.Sender && t.FilePath == modCompressedPath))
415 {
416 DebugConsole.Log($"Ignoring a mod file request from \"{client.Name}\" ({modName}) - already transferring.");
417 return;
418 }
419
420 StartTransfer(inc.Sender, FileTransferType.Mod, modCompressedPath);
421
422 break;
423 }
424 }
