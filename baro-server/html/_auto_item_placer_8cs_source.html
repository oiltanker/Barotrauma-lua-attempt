22 public static void PlaceIfNeeded()
23 {
24 if (GameMain.NetworkMember != null && !GameMain.NetworkMember.IsServer) { return; }
25
26 for (int i = 0; i < Submarine.MainSubs.Length; i++)
27 {
28 if (Submarine.MainSubs[i] == null || Submarine.MainSubs[i].Info.InitialSuppliesSpawned) { continue; }

30 subs.AddRange(Submarine.MainSubs[i].DockedTo.Where(d => !d.Info.IsOutpost));
31 Place(subs);
32 subs.ForEach(s => s.Info.InitialSuppliesSpawned = true);
33 }

36 foreach (var sub in Submarine.Loaded)

38 if (sub.Info.Type == SubmarineType.Player ||
39 sub.Info.Type == SubmarineType.Outpost ||
40 sub.Info.Type == SubmarineType.OutpostModule ||
41 sub.Info.Type == SubmarineType.EnemySubmarine)

43 continue;

71 private static void Place(IEnumerable<Submarine> subs, ItemContainer regeneratedContainer = null, float difficultyModifier = DefaultDifficultyModifier)
72 {
73 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
74 {
75 DebugConsole.ThrowError("Clients are not allowed to use AutoItemPlacer.\n" + Environment.StackTrace.CleanupStackTrace());
76 return;
77 }
78
79 List<Item> spawnedItems = new List<Item>(100);
80
81 int itemCountApprox = MapEntityPrefab.List.Count() / 3;
82 var containers = new List<ItemContainer>(70 + 30 * subs.Count());
83 var prefabsItemsCanSpawnIn = new List<ItemPrefab>(itemCountApprox / 3);
84 var singlePrefabs = new List<ItemPrefab>(itemCountApprox);
85 var removals = new List<ItemPrefab>();
86
87 // generate loot only for a specific container if defined
88 if (regeneratedContainer != null)
89 {
90 containers.Add(regeneratedContainer);
91 }
92 else
93 {
94 foreach (Item item in Item.ItemList)
95 {
96 if (!subs.Contains(item.Submarine)) { continue; }
97 if (item.GetRootInventoryOwner() is Character) { continue; }
98 containers.AddRange(item.GetComponents<ItemContainer>());
99 }
100 containers.Shuffle(Rand.RandSync.ServerAndClient);
101 }
102
103 foreach (ItemPrefab ip in ItemPrefab.Prefabs)
104 {
105 if (!ip.PreferredContainers.Any()) { continue; }
106 if (ip.ConfigElement.Elements().Any(e => string.Equals(e.Name.ToString(), typeof(ItemContainer).Name.ToString(), StringComparison.OrdinalIgnoreCase)) &&
107 ItemPrefab.Prefabs.Any(ip2 => CanSpawnIn(ip2, ip)))
108 {
109 prefabsItemsCanSpawnIn.Add(ip);
110 }
111 else
112 {
113 singlePrefabs.Add(ip);
114 }
115 }

117 bool CanSpawnIn(ItemPrefab item, ItemPrefab container)
118 {
119 foreach (var preferredContainer in item.PreferredContainers)
120 {
121 if (ItemPrefab.IsContainerPreferred(preferredContainer.Primary, container.Identifier.ToEnumerable().Union(container.Tags))) { return true; }
122 }
123 return false;
124 }
125
126 var validContainers = new Dictionary<ItemContainer, PreferredContainer>();
127 prefabsItemsCanSpawnIn.Shuffle(Rand.RandSync.ServerAndClient);
128 // Spawn items that other items can spawn in first so we can fill them up with items if needed (oxygen tanks inside the spawned diving masks, etc)
129 for (int i = 0; i < prefabsItemsCanSpawnIn.Count; i++)
130 {
131 var itemPrefab = prefabsItemsCanSpawnIn[i];
132 if (itemPrefab == null) { continue; }
133 SpawnItems(itemPrefab);
134 }
135
136 // Spawn items that nothing can spawn in last
137 singlePrefabs.Shuffle(Rand.RandSync.ServerAndClient);
138 singlePrefabs.ForEach(i => SpawnItems(i));
139
140 if (OutputDebugInfo)
141 {
142 var subNames = subs.Select(s => s.Info.Name).ToList();
143 DebugConsole.NewMessage($"Automatically placed items in { string.Join(", ", subNames) }:");
144 foreach (string itemName in spawnedItems.Select(it => it.Name).Distinct())
145 {
146 DebugConsole.NewMessage(" - " + itemName + " x" + spawnedItems.Count(it => it.Name == itemName));
147 }
148 }
149
150 if (GameMain.GameSession?.Level != null &&
151 GameMain.GameSession.Level.Type == LevelData.LevelType.Outpost &&
152 GameMain.GameSession.StartLocation?.TakenItems != null)
153 {
154 foreach (Location.TakenItem takenItem in GameMain.GameSession.StartLocation.TakenItems)
155 {
156 var matchingItem = spawnedItems.Find(it => takenItem.Matches(it));
157 if (matchingItem == null) { continue; }
158 var containedItems = spawnedItems.FindAll(it => it.ParentInventory?.Owner == matchingItem);
159 matchingItem.Remove();
160 spawnedItems.Remove(matchingItem);
161 foreach (Item containedItem in containedItems)
162 {
163 containedItem.Remove();
164 spawnedItems.Remove(containedItem);
165 }
166 }
167 }
168 foreach (Item spawnedItem in spawnedItems)
169 {
170#if SERVER
171 Entity.Spawner.CreateNetworkEvent(new EntitySpawner.SpawnEntity(spawnedItem));
172#endif
173 foreach (ItemComponent ic in spawnedItem.Components)
174 {
175 ic.OnItemLoaded();
176 }
177 }

179 bool SpawnItems(ItemPrefab itemPrefab)
180 {
181 if (itemPrefab == null)
182 {
183 string errorMsg = "Error in AutoItemPlacer.SpawnItems - itemPrefab was null.\n" + Environment.StackTrace.CleanupStackTrace();
184 DebugConsole.ThrowError(errorMsg);
185 GameAnalyticsManager.AddErrorEventOnce("AutoItemPlacer.SpawnItems:ItemNull", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
186 return false;
187 }
188 bool success = false;
189 foreach (PreferredContainer preferredContainer in itemPrefab.PreferredContainers)
190 {
191 if (preferredContainer.SpawnProbability <= 0.0f || preferredContainer.MaxAmount <= 0) { continue; }
192 validContainers = GetValidContainers(preferredContainer, containers, validContainers, primary: true);
193 if (validContainers.None())
194 {
195 validContainers = GetValidContainers(preferredContainer, containers, validContainers, primary: false);
196 }
197 foreach (var validContainer in validContainers)
198 {
199 var newItems = SpawnItem(itemPrefab, containers, validContainer, difficultyModifier);
200 if (newItems.Any())
201 {
202 spawnedItems.AddRange(newItems);
203 success = true;
204 }
205 }
206 }
207 return success;
208 }

211 private static Dictionary<ItemContainer, PreferredContainer> GetValidContainers(PreferredContainer preferredContainer, IEnumerable<ItemContainer> allContainers, Dictionary<ItemContainer, PreferredContainer> validContainers, bool primary)
212 {
213 validContainers.Clear();
214 foreach (ItemContainer container in allContainers)
215 {
216 if (!container.AutoFill) { continue; }
217 if (primary)
218 {
219 if (!ItemPrefab.IsContainerPreferred(preferredContainer.Primary, container)) { continue; }
220 }
221 else
222 {
223 if (!ItemPrefab.IsContainerPreferred(preferredContainer.Secondary, container)) { continue; }
224 }
225 if (!validContainers.ContainsKey(container))
226 {
227 validContainers.Add(container, preferredContainer);
228 }
229 }
230 return validContainers;
231 }

241 private static List<Item> SpawnItem(ItemPrefab itemPrefab, List<ItemContainer> containers, KeyValuePair<ItemContainer, PreferredContainer> validContainer, float difficultyModifier)
242 {
243 List<Item> spawnedItems = new List<Item>();
244 if (Rand.Value(Rand.RandSync.ServerAndClient) > validContainer.Value.SpawnProbability * (1f + difficultyModifier)) { return spawnedItems; }
245 // Don't add dangerously reactive materials in thalamus wrecks
246 if (validContainer.Key.Item.Submarine.WreckAI != null && itemPrefab.Tags.Contains("explodesinwater"))
247 {
248 return spawnedItems;
249 }
250 int amount = Rand.Range(validContainer.Value.MinAmount, validContainer.Value.MaxAmount + 1, Rand.RandSync.ServerAndClient);
251 for (int i = 0; i < amount; i++)
252 {
253 if (validContainer.Key.Inventory.IsFull(takeStacksIntoAccount: true))
254 {
255 containers.Remove(validContainer.Key);
256 break;
257 }
258
259 var existingItem = validContainer.Key.Inventory.AllItems.FirstOrDefault(it => it.Prefab == itemPrefab);
260 int quality =
261 existingItem?.Quality ??
262 ToolBox.SelectWeightedRandom(
263 qualityCommonnesses.Select(q => q.quality).ToList(),
264 qualityCommonnesses.Select(q => q.commonness).ToList(),
265 Rand.RandSync.ServerAndClient);
266 if (!validContainer.Key.Inventory.CanBePut(itemPrefab, quality: quality)) { break; }
267 var item = new Item(itemPrefab, validContainer.Key.Item.Position, validContainer.Key.Item.Submarine, callOnItemLoaded: false)
268 {
269 SpawnedInCurrentOutpost = validContainer.Key.Item.SpawnedInCurrentOutpost,
270 AllowStealing = validContainer.Key.Item.AllowStealing,
271 Quality = quality,
272 OriginalModuleIndex = validContainer.Key.Item.OriginalModuleIndex,
273 OriginalContainerIndex =
274 Item.ItemList.Where(it => it.Submarine == validContainer.Key.Item.Submarine && it.OriginalModuleIndex == validContainer.Key.Item.OriginalModuleIndex).ToList().IndexOf(validContainer.Key.Item)
275 };
276 foreach (WifiComponent wifiComponent in item.GetComponents<WifiComponent>())
277 {
278 wifiComponent.TeamID = validContainer.Key.Item.Submarine.TeamID;
279 }
280 spawnedItems.Add(item);
281 validContainer.Key.Inventory.TryPutItem(item, null, createNetworkEvent: false);
282 containers.AddRange(item.GetComponents<ItemContainer>());
283 }
284 return spawnedItems;
285 }
