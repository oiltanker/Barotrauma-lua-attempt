162 public OutpostGenerationParams(ContentXElement element, ContentFile file) : base(file, element.GetAttributeIdentifier("identifier", ""))
163 {
164 Name = element.GetAttributeString("name", Identifier.Value);
165 allowedLocationTypes = element.GetAttributeIdentifierArray("allowedlocationtypes", Array.Empty<Identifier>()).ToHashSet();
166 SerializableProperties = SerializableProperty.DeserializeProperties(this, element);
167
168 var humanPrefabCollections = new List<IReadOnlyList<HumanPrefab>>();
169 foreach (var subElement in element.Elements())
170 {
171 switch (subElement.Name.ToString().ToLowerInvariant())
172 {
173 case "modulecount":
174 Identifier moduleFlag = subElement.GetAttributeIdentifier("flag", subElement.GetAttributeIdentifier("moduletype", ""));
175 moduleCounts[moduleFlag] = subElement.GetAttributeInt("count", 0);
176 break;
177 case "npcs":
178 var newCollection = new NpcCollection();
179 foreach (var npcElement in subElement.Elements())
180 {
181 Identifier from = npcElement.GetAttributeIdentifier("from", Identifier.Empty);
182
183 if (from != Identifier.Empty)
184 {
185 newCollection.Add(from, npcElement.GetAttributeIdentifier("identifier", Identifier.Empty));
186 }
187 else
188 {
189 newCollection.Add(new HumanPrefab(npcElement, file));
190 }
191 }
192 humanPrefabCollections.Add(newCollection);
193 break;
194 }
195 }
196
197 this.humanPrefabCollections = humanPrefabCollections.ToImmutableArray();
198 }

235 public ImmutableHashSet<Identifier> GetStoreIdentifiers()

239 var storeIdentifiers = new HashSet<Identifier>();

242 foreach (var prefab in collection)

244 if (prefab?.CampaignInteractionType == CampaignMode.InteractionType.Store)

246 storeIdentifiers.Add(prefab.Identifier);
247 }
248 }
249 }

251 }
