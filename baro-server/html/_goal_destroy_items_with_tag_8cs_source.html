29 private int CountMatchingItems()
30 {
31 int result = 0;
32 foreach (var item in Item.ItemList)
33 {
34 if (!matchInventory && Traitors.All(traitor => item.FindParentInventory(inventory => inventory.Owner is Character && inventory.Owner != traitor.Character) != null))
35 {
36 continue;
37 }
38
39 if (item.Submarine == null)
40 {
41 //items outside the sub don't count as destroyed if they're still in the traitor's inventory
42 bool carriedByTraitor = Traitors.Any(traitor => item.IsOwnedBy(traitor.Character));
43 if (!carriedByTraitor) { continue; }
44 }
45 else
46 {
47 if (Traitors.All(traitor => item.Submarine.TeamID != traitor.Character.TeamID)) { continue; }
48 }
49
50 if (item.Condition <= 0.0f)
51 {
52 continue;
53 }
54 var identifierMatches = matchIdentifier && ((MapEntity)item).Prefab.Identifier == tag;
55 if (identifierMatches && tagPrefabName == null)
56 {
57 var textId = item.Prefab.GetItemNameTextId();
58 tagPrefabName = textId != null ? TextManager.FormatServerMessage(textId) : item.Prefab.Name.Value;
59 }
60 if (identifierMatches || (matchTag && item.HasTag(tag)))
61 {
62 ++result;
63 }
64 }
65
66 // Quick fix
67 if (tagPrefabName == null && matchIdentifier)
68 {
69 tagPrefabName = TextManager.FormatServerMessage($"entityname.{tag}");
70 }
71
72 return result;
73 }
