166 public bool Connect(Connection newConnection, bool addNode = true, bool sendNetworkEvent = false)
167 {
168 for (int i = 0; i < 2; i++)
169 {
170 if (connections[i] == newConnection) { return false; }
171 }
172
173 if (!connections.Any(c => c == null)) { return false; }
174
175 for (int i = 0; i < 2; i++)
176 {
177 if (connections[i] != null && connections[i].Item == newConnection.Item)
178 {
179 addNode = false;
180 break;
181 }
182 }
183 if (item.body != null) { item.Submarine = newConnection.Item.Submarine; }
184
185 newConnection.ConnectionPanel.DisconnectedWires.Remove(this);
186
187 for (int i = 0; i < 2; i++)
188 {
189 if (connections[i] != null) { continue; }
190
191 connections[i] = newConnection;
192 FixNodeEnds();
193
194 if (!addNode) { break; }
195
196 Submarine refSub = newConnection.Item.Submarine;
197 if (refSub == null)
198 {
199 Structure attachTarget = Structure.GetAttachTarget(newConnection.Item.WorldPosition);
200 if (attachTarget == null && !(newConnection.Item.GetComponent<Holdable>()?.Attached ?? false))
201 {
202 connections[i] = null;
203 continue;
204 }
205 refSub = attachTarget?.Submarine;
206 }
207
208 Vector2 nodePos = refSub == null ?
209 newConnection.Item.Position :
210 newConnection.Item.Position - refSub.HiddenSubPosition;
211
212 if (nodes.Count > 0 && nodes[0] == nodePos) { break; }
213 if (nodes.Count > 1 && nodes[nodes.Count - 1] == nodePos) { break; }
214
215 //make sure we place the node at the correct end of the wire (the end that's closest to the new node pos)
216 int newNodeIndex = 0;
217 if (nodes.Count > 1)
218 {
219 if (connections[0] != null && connections[0] != newConnection)
220 {
221 if (Vector2.DistanceSquared(nodes[0], connections[0].Item.Position - (refSub?.HiddenSubPosition ?? Vector2.Zero)) <
222 Vector2.DistanceSquared(nodes[nodes.Count - 1], connections[0].Item.Position - (refSub?.HiddenSubPosition ?? Vector2.Zero)))
223 {
224 newNodeIndex = nodes.Count;
225 }
226 }
227 else if (connections[1] != null && connections[1] != newConnection)
228 {
229 if (Vector2.DistanceSquared(nodes[0], connections[1].Item.Position - (refSub?.HiddenSubPosition ?? Vector2.Zero)) <
230 Vector2.DistanceSquared(nodes[nodes.Count - 1], connections[1].Item.Position - (refSub?.HiddenSubPosition ?? Vector2.Zero)))
231 {
232 newNodeIndex = nodes.Count;
233 }
234 }
235 else if (Vector2.DistanceSquared(nodes[nodes.Count - 1], nodePos) < Vector2.DistanceSquared(nodes[0], nodePos))
236 {
237 newNodeIndex = nodes.Count;
238 }
239 }
240
241 if (newNodeIndex == 0 && nodes.Count > 1)
242 {
243 nodes.Insert(0, nodePos);
244 }
245 else
246 {
247 nodes.Add(nodePos);
248 }
249
250 break;
251 }
252
253 SetConnectedDirty();
254
255 if (connections[0] != null && connections[1] != null)
256 {
257 foreach (ItemComponent ic in item.Components)
258 {
259 if (ic == this) { continue; }
260 ic.Drop(null);
261 }
262 if (item.Container != null) { item.Container.RemoveContained(this.item); }
263 if (item.body != null) { item.body.Enabled = false; }
264
265 IsActive = false;
266
267 CleanNodes();
268 }
269
270 if (item.body != null) { item.Submarine = newConnection.Item.Submarine; }
271
272 if (sendNetworkEvent)
273 {
274#if SERVER
275 if (GameMain.Server != null)
276 {
277 CreateNetworkEvent();
278 }
279#endif
280 //the wire is active if only one end has been connected
281 IsActive = connections[0] == null ^ connections[1] == null;
282 }
283
284 Drawable = IsActive || nodes.Any();
285
286 UpdateSections();
287 return true;
288 }

308 public override void Update(float deltaTime, Camera cam)
309 {
310 if (nodes.Count == 0) { return; }
311
312 Character user = item.ParentInventory?.Owner as Character;
313 editNodeDelay = (user?.SelectedConstruction == null) ? editNodeDelay - deltaTime : 0.5f;
314
315 Submarine sub = item.Submarine;
316 if (connections[0] != null && connections[0].Item.Submarine != null) { sub = connections[0].Item.Submarine; }
317 if (connections[1] != null && connections[1].Item.Submarine != null) { sub = connections[1].Item.Submarine; }
318
319 if (Screen.Selected != GameMain.SubEditorScreen)
320 {
321 if (user != null) { NoAutoLock = true; }
322
323 //cannot run wires from sub to another
324 if (item.Submarine != sub && sub != null && item.Submarine != null)
325 {
326 ClearConnections();
327 return;
328 }
329
330 if (item.CurrentHull == null)
331 {
332 Structure attachTarget = Structure.GetAttachTarget(item.WorldPosition);
333 canPlaceNode = attachTarget != null;
334
335 sub ??= attachTarget?.Submarine;
336 Vector2 attachPos = GetAttachPosition(user);
337 newNodePos = sub == null ?
338 attachPos :
339 attachPos - sub.Position - sub.HiddenSubPosition;
340 }
341 else
342 {
343 newNodePos = GetAttachPosition(user);
344 if (sub != null) { newNodePos -= sub.HiddenSubPosition; }
345 canPlaceNode = true;
346 }
347
348 //prevent the wire from extending too far when rewiring
349 if (nodes.Count > 0)
350 {
351 if (user == null) { return; }
352
353 Vector2 prevNodePos = nodes[nodes.Count - 1];
354 if (sub != null) { prevNodePos += sub.HiddenSubPosition; }
355
356 currLength = 0.0f;
357 for (int i = 0; i < nodes.Count - 1; i++)
358 {
359 currLength += Vector2.Distance(nodes[i], nodes[i + 1]);
360 }
361 Vector2 itemPos = item.Position;
362 if (sub != null && user.Submarine == null) { prevNodePos += sub.Position; }
363 currLength += Vector2.Distance(prevNodePos, itemPos);
364 if (currLength > MaxLength)
365 {
366 Vector2 diff = prevNodePos - user.Position;
367 Vector2 pullBackDir = diff == Vector2.Zero ? Vector2.Zero : Vector2.Normalize(diff);
368 Vector2 forceDir = pullBackDir;
369 if (!user.AnimController.InWater) { forceDir.Y = 0.0f; }
370 user.AnimController.Collider.ApplyForce(forceDir * user.Mass * 50.0f, maxVelocity: NetConfig.MaxPhysicsBodyVelocity * 0.5f);
371 if (diff.LengthSquared() > 50.0f * 50.0f)
372 {
373 user.AnimController.UpdateUseItem(true, user.WorldPosition + pullBackDir * Math.Min(150.0f, diff.Length()));
374 }
375
376 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
377 {
378 if (currLength > MaxLength * 1.5f)
379 {
380 ClearConnections();
381#if SERVER
382 CreateNetworkEvent();
383#endif
384 return;
385 }
386 }
387 }
388 }
389 }
390 else
391 {
392#if CLIENT
393 bool disableGrid = SubEditorScreen.IsSubEditor() && PlayerInput.IsShiftDown();
394 newNodePos = disableGrid ? item.Position : RoundNode(item.Position);
395#else
396 newNodePos = RoundNode(item.Position);
397#endif
398 if (sub != null) { newNodePos -= sub.HiddenSubPosition; }
399 canPlaceNode = true;
400 }
401
402 if (item != null)
403 {
404 Vector2 relativeNodePos = newNodePos - item.Position;
405
406 if (sub != null)
407 {
408 relativeNodePos += sub.HiddenSubPosition;
409 }
410
411 sectionExtents = new Vector2(
412 Math.Max(Math.Abs(relativeNodePos.X), sectionExtents.X),
413 Math.Max(Math.Abs(relativeNodePos.Y), sectionExtents.Y));
414 }
415 }

417 private Vector2 GetAttachPosition(Character user)
418 {
419 if (user == null) { return item.Position; }
420
421 Vector2 mouseDiff = user.CursorWorldPosition - user.WorldPosition;
422 mouseDiff = mouseDiff.ClampLength(MaxAttachDistance);
423
424 return new Vector2(
425 MathUtils.RoundTowardsClosest(user.Position.X + mouseDiff.X, Submarine.GridSize.X),
426 MathUtils.RoundTowardsClosest(user.Position.Y + mouseDiff.Y, Submarine.GridSize.Y));

429 public override bool Use(float deltaTime, Character character = null)
430 {
431 if (character == null || character != Character.Controlled) { return false; }
432 if (character.SelectedConstruction != null) { return false; }
433#if CLIENT
434 if (Screen.Selected == GameMain.SubEditorScreen && !PlayerInput.PrimaryMouseButtonClicked())
435 {
436 return false;
437 }
438#endif
439 //clients communicate node addition/removal with network events
440 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsServer) { return false; }
441
442 if (newNodePos != Vector2.Zero && canPlaceNode && editNodeDelay <= 0.0f && nodes.Count > 0 &&
443 Vector2.DistanceSquared(newNodePos, nodes[nodes.Count - 1]) > MinNodeDistance * MinNodeDistance)
444 {
445 if (nodes.Count >= MaxNodeCount)
446 {
447 nodes.RemoveAt(nodes.Count - 1);
448 }
449
450 nodes.Add(newNodePos);
451 CleanNodes();
452 UpdateSections();
453 Drawable = true;
454 newNodePos = Vector2.Zero;
455#if CLIENT
456 if (GameMain.NetworkMember != null)
457 {
458 item.CreateClientEvent(this, new ClientEventData(nodes.Count));
459 }
460#endif
461 }
462 editNodeDelay = 0.1f;
463 return true;
464 }

634 private void CleanNodes()
635 {
636 bool removed;
637 do
638 {
639 removed = false;
640 for (int i = nodes.Count - 2; i > 0; i--)
641 {
642 if (Math.Abs(nodes[i - 1].X - nodes[i].X) < 1.0f && Math.Abs(nodes[i + 1].X - nodes[i].X) < 1.0f &&
643 Math.Sign(nodes[i - 1].Y - nodes[i].Y) != Math.Sign(nodes[i + 1].Y - nodes[i].Y))
644 {
645 nodes.RemoveAt(i);
646 removed = true;
647 }
648 else if (Math.Abs(nodes[i - 1].Y - nodes[i].Y) < 1.0f && Math.Abs(nodes[i + 1].Y - nodes[i].Y) < 1.0f &&
649 Math.Sign(nodes[i - 1].X - nodes[i].X) != Math.Sign(nodes[i + 1].X - nodes[i].X))
650 {
651 nodes.RemoveAt(i);
652 removed = true;
653 }
654 }
655
656 } while (removed);
657 }

659 public void FixNodeEnds()
660 {
661 Item item0 = connections[0]?.Item;
662 Item item1 = connections[1]?.Item;
663
664 if (item0 == null && item1 != null)
665 {
666 item0 = Item.ItemList.Find(it => it.GetComponent<ConnectionPanel>()?.DisconnectedWires.Contains(this) ?? false);
667 }
668 else if (item0 != null && item1 == null)
669 {
670 item1 = Item.ItemList.Find(it => it.GetComponent<ConnectionPanel>()?.DisconnectedWires.Contains(this) ?? false);
671 }
672
673 if (item0 == null || item1 == null || nodes.Count == 0) { return; }
674
675 Vector2 nodePos = nodes[0];
676
677 Submarine refSub = item0.Submarine ?? item1.Submarine;
678 if (refSub != null) { nodePos += refSub.HiddenSubPosition; }
679
680 float dist1 = Vector2.DistanceSquared(item0.Position, nodePos);
681 float dist2 = Vector2.DistanceSquared(item1.Position, nodePos);
682
683 //first node is closer to the second item
684 //= the nodes are "backwards", need to reverse them
685 if (dist1 > dist2)
686 {
687 nodes.Reverse();
688 UpdateSections();
689 }
690 }

692 private int GetClosestNodeIndex(Vector2 pos, float maxDist, out float closestDist)
693 {
694 closestDist = 0.0f;
695 int closestIndex = -1;
696
697 for (int i = 0; i < nodes.Count; i++)
698 {
699 float dist = Vector2.Distance(nodes[i], pos);
700 if (dist > maxDist) continue;
701
702 if (closestIndex == -1 || dist < closestDist)
703 {
704 closestIndex = i;
705 closestDist = dist;
706 }
707 }
708
709 return closestIndex;
710 }
711
712 private int GetClosestSectionIndex(Vector2 mousePos, float maxDist, out float closestDist)
713 {
714 closestDist = 0.0f;
715 int closestIndex = -1;
716
717 maxDist *= maxDist;
718 for (int i = 0; i < nodes.Count-1; i++)
719 {
720 if ((Math.Abs(nodes[i].X - nodes[i + 1].X)<5 || Math.Sign(mousePos.X - nodes[i].X) != Math.Sign(mousePos.X - nodes[i + 1].X)) &&
721 (Math.Abs(nodes[i].Y - nodes[i + 1].Y)<5 || Math.Sign(mousePos.Y - nodes[i].Y) != Math.Sign(mousePos.Y - nodes[i + 1].Y)))
722 {
723 float dist = MathUtils.LineToPointDistanceSquared(nodes[i], nodes[i + 1], mousePos);
724 if (dist > maxDist) continue;
725
726 if (closestIndex == -1 || dist < closestDist)
727 {
728 closestIndex = i;
729 closestDist = dist;
730 }
731 }
732 }
733 closestDist = (float)Math.Sqrt(closestDist);
734
735 return closestIndex;
736 }

778 public override void Load(ContentXElement componentElement, bool usePrefabValues, IdRemap idRemap)
779 {
780 base.Load(componentElement, usePrefabValues, idRemap);
781
782 string nodeString = componentElement.GetAttributeString("nodes", "");
783 if (nodeString == "") return;
784
785 string[] nodeCoords = nodeString.Split(';');
786 for (int i = 0; i < nodeCoords.Length / 2; i++)
787 {
788 float.TryParse(nodeCoords[i * 2], NumberStyles.Float, CultureInfo.InvariantCulture, out float x);
789 float.TryParse(nodeCoords[i * 2 + 1], NumberStyles.Float, CultureInfo.InvariantCulture, out float y);
790 nodes.Add(new Vector2(x, y));
791 }
792
793 Drawable = nodes.Any();
794 }
