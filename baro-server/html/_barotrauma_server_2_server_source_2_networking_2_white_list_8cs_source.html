99 public bool IsWhiteListed(string name, IPAddress address)
100 {
101 if (!Enabled) return true;
102 WhiteListedPlayer wlp = whitelistedPlayers.Find(p => p.Name == name);
103 if (wlp == null) return false;
104 if (!string.IsNullOrWhiteSpace(wlp.IP))
105 {
106 if (address.IsIPv4MappedToIPv6 && wlp.IP == address.MapToIPv4NoThrow().ToString())
107 {
108 return true;
109 }
110 else
111 {
112 return wlp.IP == address.ToString();
113 }
114 }
115 return true;

158 public bool ServerAdminRead(IReadMessage incMsg, Client c)
159 {
160 if (!c.HasPermission(ClientPermissions.ManageSettings))
161 {
162 bool enabled = incMsg.ReadBoolean(); incMsg.ReadPadBits();
163 UInt16 removeCount = incMsg.ReadUInt16();
164 incMsg.BitPosition += removeCount * 4 * 8;
165 UInt16 addCount = incMsg.ReadUInt16();
166 for (int i = 0; i < addCount; i++)
167 {
168 incMsg.ReadString(); //skip name
169 incMsg.ReadString(); //skip ip
170 }
171 return false;
172 }
173 else
174 {
175 bool prevEnabled = Enabled;
176 bool enabled = incMsg.ReadBoolean(); incMsg.ReadPadBits();
177 Enabled = enabled;
178
179 UInt16 removeCount = incMsg.ReadUInt16();
180 for (int i = 0; i < removeCount; i++)
181 {
182 UInt16 id = incMsg.ReadUInt16();
183 WhiteListedPlayer whitelistedPlayer = whitelistedPlayers.Find(p => p.UniqueIdentifier == id);
184 if (whitelistedPlayer != null)
185 {
186 GameServer.Log(GameServer.ClientLogName(c) + " removed " + whitelistedPlayer.Name + " from whitelist (" + whitelistedPlayer.IP + ")", ServerLog.MessageType.ConsoleUsage);
187 RemoveFromWhiteList(whitelistedPlayer);
188 }
189 }
190
191 UInt16 addCount = incMsg.ReadUInt16();
192 for (int i = 0; i < addCount; i++)
193 {
194 string name = incMsg.ReadString();
195 string ip = incMsg.ReadString();
196
197 GameServer.Log(GameServer.ClientLogName(c) + " added " + name + " to whitelist (" + ip + ")", ServerLog.MessageType.ConsoleUsage);
198 AddToWhiteList(name, ip);
199 }
200
201 bool changed = removeCount > 0 || addCount > 0 || prevEnabled != enabled;
202 if (changed) { Save(); }
203 return changed;
204 }
205 }
