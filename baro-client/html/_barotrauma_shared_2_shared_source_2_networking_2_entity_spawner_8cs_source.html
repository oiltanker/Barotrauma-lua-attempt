66 public Entity Spawn()
67 {
68 if (Prefab == null)
69 {
70 return null;
71 }
72 Item spawnedItem;
73 if (Inventory?.Owner != null)
74 {
75 if (!SpawnIfInventoryFull && !Inventory.CanBePut(Prefab))
76 {
77 return null;
78 }
79 spawnedItem = new Item(Prefab, Vector2.Zero, null)
80 {
81 Condition = Condition,
82 Quality = Quality
83 };
84 var slot = Slot != InvSlotType.None ? Slot.ToEnumerable() : spawnedItem.AllowedSlots;
85 if (!Inventory.Owner.Removed && !Inventory.TryPutItem(spawnedItem, null, slot))
86 {
87 if (IgnoreLimbSlots)
88 {
89 for (int i = 0; i < Inventory.Capacity; i++)
90 {
91 if (Inventory.GetItemAt(i) == null)
92 {
93 Inventory.ForceToSlot(spawnedItem, i);
94 break;
95 }
96 }
97 }
98 spawnedItem.SetTransform(FarseerPhysics.ConvertUnits.ToSimUnits(Inventory.Owner?.WorldPosition ?? Vector2.Zero), spawnedItem.body?.Rotation ?? 0.0f, findNewHull: false);
99 }
100 }
101 else
102 {
103 spawnedItem = new Item(Prefab, Position, Submarine)
104 {
105 Condition = Condition,
106 Quality = Quality
107 };
108 }
109 return spawnedItem;
110 }

200 public abstract class SpawnOrRemove : NetEntityEvent.IData
201 {
202 public readonly Entity Entity;
203 public UInt16 ID => Entity.ID;
204
205 public readonly UInt16 InventoryID;
206
207 public readonly byte ItemContainerIndex;
208 public readonly int SlotIndex;
209
210 public override string ToString()
211 {
212 return
213 "(" +
214 ((Entity as MapEntity)?.Name ?? "[NULL]") +
215 $", {ID}, {InventoryID}, {SlotIndex})";
216 }
217
218 protected SpawnOrRemove(Entity entity)
219 {
220 Entity = entity;
221 if (!(entity is Item { ParentInventory: { Owner: { } } } item)) { return; }
222
223 InventoryID = item.ParentInventory.Owner.ID;
224 SlotIndex = item.ParentInventory.FindIndex(item);
225 //find the index of the ItemContainer this item is inside to get the item to
226 //spawn in the correct inventory in multi-inventory items like fabricators
227 if (item.Container == null) { return; }
228
229 foreach (ItemComponent component in item.Container.Components)
230 {
231 if (component is ItemContainer container &&
232 container.Inventory == item.ParentInventory)
233 {
234 ItemContainerIndex = (byte)item.Container.GetComponentIndex(component);
235 break;
236 }
237 }
238 }

350 public void AddEntityToRemoveQueue(Entity entity)
351 {
352 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
353 if (removeQueue.Contains(entity) || entity.Removed || entity == null || entity.IdFreed) { return; }
354 if (entity is Item item) { AddItemToRemoveQueue(item); return; }
355 if (entity is Character)
356 {
357 Character character = entity as Character;
358#if SERVER
359 if (GameMain.Server != null)
360 {
361 Client client = GameMain.Server.ConnectedClients.Find(c => c.Character == character);
362 if (client != null) GameMain.Server.SetClientCharacter(client, null);
363 }
364#endif
365 }
366
367 removeQueue.Enqueue(entity);
368 }
369
370 public void AddItemToRemoveQueue(Item item)
371 {
372 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
373 if (removeQueue.Contains(item) || item.Removed) { return; }
374
375 removeQueue.Enqueue(item);
376 var containedItems = item.OwnInventory?.AllItems;
377 if (containedItems == null) { return; }
378 foreach (Item containedItem in containedItems)
379 {
380 if (containedItem != null)
381 {
382 AddItemToRemoveQueue(containedItem);
383 }
384 }
385 }

408 public void Update(bool createNetworkEvents = true)
409 {
410 if (GameMain.NetworkMember is { IsClient: true }) { return; }
411 while (spawnQueue.Count > 0)
412 {
413 var entitySpawnInfo = spawnQueue.Dequeue();
414
415 var spawnedEntity = entitySpawnInfo.Spawn();
416 if (spawnedEntity == null) { continue; }
417
418 if (createNetworkEvents)
419 {
420 CreateNetworkEventProjSpecific(new SpawnEntity(spawnedEntity));
421 }
422 entitySpawnInfo.OnSpawned(spawnedEntity);
423 }
424
425 while (removeQueue.Count > 0)
426 {
427 var removedEntity = removeQueue.Dequeue();
428 if (removedEntity is Item item)
429 {
430 item.SendPendingNetworkUpdates();
431 }
432 if (createNetworkEvents)
433 {
434 CreateNetworkEventProjSpecific(new RemoveEntity(removedEntity));
435 }
436 removedEntity.Remove();
437 }
438 }
