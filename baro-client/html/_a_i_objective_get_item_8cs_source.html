100 public static IEnumerable<Identifier> ParseGearTags(IEnumerable<Identifier> identifiersOrTags)
101 {
102 var tags = new List<Identifier>();

151 protected override void Act(float deltaTime)
152 {
153 if (character.LockHands)
154 {
155 Abandon = true;
156 return;
157 }
158 if (character.Submarine == null)
159 {
160 Abandon = true;
161 return;
162 }
163 if (IdentifiersOrTags != null && !isDoneSeeking)
164 {
165 if (checkInventory)
166 {
167 if (CheckInventory())
168 {
169 isDoneSeeking = true;
170 }
171 }
172 if (!isDoneSeeking)
173 {
174 if (!AllowDangerousPressure)
175 {
176 bool dangerousPressure = character.CurrentHull == null || character.CurrentHull.LethalPressure > 0 && character.PressureProtection <= 0;
177 if (dangerousPressure)
178 {
179#if DEBUG
180 string itemName = targetItem != null ? targetItem.Name : IdentifiersOrTags.FirstOrDefault().Value;
181 DebugConsole.NewMessage($"{character.Name}: Seeking item ({itemName}) aborted, because the pressure is dangerous.", Color.Yellow);
182#endif
183 Abandon = true;
184 return;
185 }
186 }
187 FindTargetItem();
188 if (!objectiveManager.IsCurrentOrder<AIObjectiveGoTo>())
189 {
190 objectiveManager.GetObjective<AIObjectiveIdle>().Wander(deltaTime);
191 }
192 return;
193 }
194 }
195 if (targetItem == null || targetItem.Removed)
196 {
197#if DEBUG
198 DebugConsole.NewMessage($"{character.Name}: Target null or removed. Aborting.", Color.Red);
199#endif
200 Abandon = true;
201 return;
202 }
203 else if (isDoneSeeking && moveToTarget == null)
204 {
205#if DEBUG
206 DebugConsole.NewMessage($"{character.Name}: Move target null. Aborting.", Color.Red);
207#endif
208 Abandon = true;
209 return;
210 }
211 if (character.IsItemTakenBySomeoneElse(targetItem))
212 {
213#if DEBUG
214 DebugConsole.NewMessage($"{character.Name}: Found an item, but it's already equipped by someone else.", Color.Yellow);
215#endif
216 if (originalTarget == null)
217 {
218 // Try again
219 ignoredItems.Add(targetItem);
220 ResetInternal();
221 }
222 else
223 {
224 Abandon = true;
225 }
226 return;
227 }
228 bool canInteract = false;
229 if (moveToTarget is Character c)
230 {
231 if (character == c)
232 {
233 canInteract = true;
234 moveToTarget = null;
235 }
236 else
237 {
238 character.SelectCharacter(c);
239 canInteract = character.CanInteractWith(c, maxDist: DefaultReach);
240 character.DeselectCharacter();
241 }
242 }
243 else if (moveToTarget is Item parentItem)
244 {
245 canInteract = character.CanInteractWith(parentItem, checkLinked: false);
246 }
247 if (canInteract)
248 {
249 var pickable = targetItem.GetComponent<Pickable>();
250 if (pickable == null)
251 {
252#if DEBUG
253 DebugConsole.NewMessage($"{character.Name}: Target not pickable. Aborting.", Color.Yellow);
254#endif
255 Abandon = true;
256 return;
257 }
258
259 Inventory itemInventory = targetItem.ParentInventory;
260 var slots = itemInventory?.FindIndices(targetItem);
261 if (HumanAIController.TakeItem(targetItem, character.Inventory, Equip, Wear, storeUnequipped: true))
262 {
263 if (TakeWholeStack && slots != null)
264 {
265 foreach (int slot in slots)
266 {
267 foreach (Item item in itemInventory.GetItemsAt(slot).ToList())
268 {
269 HumanAIController.TakeItem(item, character.Inventory, equip: false, storeUnequipped: true);
270 }
271 }
272 }
273 if (IdentifiersOrTags == null)
274 {
275 IsCompleted = true;
276 }
277 else
278 {
279 IsCompleted = CountItems();
280 if (!IsCompleted)
281 {
282 ResetInternal();
283 }
284 }
285 }
286 else
287 {
288 if (!Equip)
289 {
290 // Try equipping and wearing the item
291 Wear = true;
292 Equip = true;
293 return;
294 }
295#if DEBUG
296 DebugConsole.NewMessage($"{character.Name}: Failed to equip/move the item '{targetItem.Name}' into the character inventory. Aborting.", Color.Red);
297#endif
298 Abandon = true;
299 }
300 }
301 else if (moveToTarget != null)
302 {
303 TryAddSubObjective(ref goToObjective,
304 constructor: () =>
305 {
306 return new AIObjectiveGoTo(moveToTarget, character, objectiveManager, repeat: false, getDivingGearIfNeeded: AllowToFindDivingGear, closeEnough: DefaultReach)
307 {
308 // If the root container changes, the item is no longer where it was (taken by someone -> need to find another item)
309 AbortCondition = obj => targetItem == null || targetItem.GetRootInventoryOwner() != moveToTarget,
310 SpeakIfFails = false
311 };
312 },
313 onAbandon: () =>
314 {
315 if (originalTarget == null)
316 {
317 // Try again
318 ignoredItems.Add(targetItem);
319 if (targetItem != moveToTarget && moveToTarget is Item item)
320 {
321 ignoredItems.Add(item);
322 }
323 ResetInternal();
324 }
325 else
326 {
327 Abandon = true;
328 }
329 },

334 private void FindTargetItem()
335 {
336 if (IdentifiersOrTags == null)
337 {
338 if (targetItem == null)
339 {
340#if DEBUG
341 DebugConsole.NewMessage($"{character.Name}: Cannot find an item, because neither identifiers nor item was defined.", Color.Red);
342#endif
343 Abandon = true;
344 }
345 return;
346 }
347
348 float priority = Math.Clamp(objectiveManager.GetCurrentPriority(), 10, 100);
349 if (!CheckPathForEachItem)
350 {
351 // While following the player, let's ensure that there's a valid path to the target before accepting it.
352 // Otherwise it will take some time for us to find a valid item when there are multiple items that we can't reach and some that we can.
353 // This is relatively expensive, so let's do this only when it significantly improves the behavior.
354 // Only allow one path find call per frame.
355 CheckPathForEachItem = priority >= AIObjectiveManager.LowestOrderPriority && (objectiveManager.IsCurrentOrder<AIObjectiveFixLeaks>() || objectiveManager.CurrentOrder is AIObjectiveGoTo gotoOrder && gotoOrder.IsFollowOrderObjective);
356 }
357 bool checkPath = CheckPathForEachItem;
358 bool hasCalledPathFinder = false;
359 int itemsPerFrame = (int)priority;
360 for (int i = 0; i < itemsPerFrame && currSearchIndex < Item.ItemList.Count - 1; i++)
361 {
362 currSearchIndex++;
363 var item = Item.ItemList[currSearchIndex];
364 Submarine itemSub = item.Submarine ?? item.ParentInventory?.Owner?.Submarine;
365 if (itemSub == null) { continue; }
366 Submarine mySub = character.Submarine;
367 if (mySub == null) { continue; }
368 if (!checkInventory)
369 {
370 // Ignore items in the inventory when defined not to check it.
371 if (item.IsOwnedBy(character)) { continue; }
372 }
373 if (!AllowStealing)
374 {
375 if (character.TeamID == CharacterTeamType.FriendlyNPC != item.SpawnedInCurrentOutpost) { continue; }
376 }
377 if (!CheckItem(item)) { continue; }
378 if (item.Container != null)
379 {
380 if (item.Container.HasTag("donttakeitems")) { continue; }
381 if (ignoredItems.Contains(item.Container)) { continue; }
382 if (ignoredContainerIdentifiers != null)
383 {
384 if (ignoredContainerIdentifiers.Contains(item.ContainerIdentifier)) { continue; }
385 }
386 }
387 // Don't allow going into another sub, unless it's connected and of the same team and type.
388 if (!character.Submarine.IsEntityFoundOnThisSub(item, includingConnectedSubs: true)) { continue; }
389 if (character.IsItemTakenBySomeoneElse(item)) { continue; }
390 if (item.ParentInventory is ItemInventory itemInventory)
391 {
392 if (!itemInventory.Container.HasRequiredItems(character, addMessage: false)) { continue; }
393 }
394 float itemPriority = 1;
395 if (GetItemPriority != null)
396 {
397 itemPriority = GetItemPriority(item);
398 }
399 Entity rootInventoryOwner = item.GetRootInventoryOwner();
400 if (rootInventoryOwner is Item ownerItem)
401 {
402 if (!ownerItem.IsInteractable(character)) { continue; }
403 if (!(ownerItem.GetComponent<ItemContainer>()?.HasRequiredItems(character, addMessage: false) ?? true)) { continue; }
404 //the item is inside an item inside an item (e.g. fuel tank in a welding tool in a cabinet -> reduce priority to prefer items that aren't inside a tool)
405 if (ownerItem != item.Container)
406 {
407 itemPriority *= 0.1f;
408 }
409 }
410 Vector2 itemPos = (rootInventoryOwner ?? item).WorldPosition;
411 float yDist = Math.Abs(character.WorldPosition.Y - itemPos.Y);
412 yDist = yDist > 100 ? yDist * 5 : 0;
413 float dist = Math.Abs(character.WorldPosition.X - itemPos.X) + yDist;
414 float minDistFactor = EvaluateCombatPriority ? 0.1f : 0;
415 float distanceFactor = MathHelper.Lerp(1, minDistFactor, MathUtils.InverseLerp(100, 10000, dist));
416 itemPriority *= distanceFactor;
417 if (EvaluateCombatPriority)
418 {
419 var mw = item.GetComponent<MeleeWeapon>();
420 var rw = item.GetComponent<RangedWeapon>();
421 float combatFactor = 0;
422 if (mw != null)
423 {
424 if (mw.CombatPriority > 0)
425 {
426 combatFactor = mw.CombatPriority / 100;
427 }
428 else
429 {
430 // The combat factor of items with zero combat priority is not allowed to be greater than 0.1f
431 combatFactor = Math.Min(AIObjectiveCombat.GetLethalDamage(mw) / 1000, 0.1f);
432 }
433 }
434 else if (rw != null)
435 {
436 if (rw.CombatPriority > 0)
437 {
438 combatFactor = rw.CombatPriority / 100;
439 }
440 else
441 {
442 combatFactor = Math.Min(AIObjectiveCombat.GetLethalDamage(rw) / 1000, 0.1f);
443 }
444 }
445 else
446 {
447 combatFactor = Math.Min(item.Components.Sum(ic => AIObjectiveCombat.GetLethalDamage(ic)) / 1000, 0.1f);
448 }
449 itemPriority *= combatFactor;
450 }
451 else
452 {
453 itemPriority *= item.Condition / item.MaxCondition;
454 }
455 // Ignore if the item has a lower priority than the currently selected one
456 if (itemPriority < currItemPriority) { continue; }
457 if (!hasCalledPathFinder && PathSteering != null && checkPath)
458 {
459 hasCalledPathFinder = true;
460 var path = PathSteering.PathFinder.FindPath(character.SimPosition, item.SimPosition, character.Submarine, errorMsgStr: $"AIObjectiveGetItem {character.DisplayName}", nodeFilter: node => node.Waypoint.CurrentHull != null);
461 if (path.Unreachable) { continue; }
462 }
463 currItemPriority = itemPriority;
464 targetItem = item;
465 moveToTarget = rootInventoryOwner ?? item;
466 }
467 if (currSearchIndex >= Item.ItemList.Count - 1)
468 {
469 isDoneSeeking = true;
470 if (targetItem == null)
471 {
472 if (spawnItemIfNotFound)
473 {
474 if (!(MapEntityPrefab.List.FirstOrDefault(me => me is ItemPrefab ip && IdentifiersOrTags.Any(id => id == ip.Identifier || ip.Tags.Contains(id))) is ItemPrefab prefab))
475 {
476#if DEBUG
477 DebugConsole.NewMessage($"{character.Name}: Cannot find an item with the following identifier(s) or tag(s): {string.Join(", ", IdentifiersOrTags)}, tried to spawn the item but no matching item prefabs were found.", Color.Yellow);
478#endif
479 Abandon = true;
480 }
481 else
482 {
483 Entity.Spawner.AddItemToSpawnQueue(prefab, character.Inventory, onSpawned: (Item spawnedItem) =>
484 {
485 targetItem = spawnedItem;
486 if (character.TeamID == CharacterTeamType.FriendlyNPC && (character.Submarine?.Info.IsOutpost ?? false))
487 {
488 spawnedItem.SpawnedInCurrentOutpost = true;
489 }
490 });
491 }
492 }
493 else
494 {
495#if DEBUG
496 DebugConsole.NewMessage($"{character.Name}: Cannot find an item with the following identifier(s) or tag(s): {string.Join(", ", IdentifiersOrTags)}", Color.Yellow);
497#endif
498 Abandon = true;
499 }
500 }
501 }

529 private bool CheckItem(Item item)
530 {
531 if (!item.HasAccess(character)) { return false; }
532 if (ignoredItems.Contains(item)) { return false; };
533 if (ignoredIdentifiersOrTags != null && ignoredIdentifiersOrTags.Any(id => item.Prefab.Identifier == id || item.HasTag(id))) { return false; }
534 if (item.Condition < TargetCondition) { return false; }
535 if (ItemFilter != null && !ItemFilter(item)) { return false; }
536 if (RequireLoaded && item.Components.Any(i => !i.IsLoaded(character))) { return false; }
537 return IdentifiersOrTags.Any(id => id == item.Prefab.Identifier || item.HasTag(id) || (AllowVariants && !item.Prefab.VariantOf.IsEmpty && item.Prefab.VariantOf == id));
538 }
