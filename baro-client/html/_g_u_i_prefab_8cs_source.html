68 public void LoadFont()
69 {
70 string fontPath = GetFontFilePath(element);
71 uint size = GetFontSize(element);
72 bool dynamicLoading = GetFontDynamicLoading(element);
73 bool isCJK = GetIsCJK(element);
74 font?.Dispose();
75 cjkFont?.Dispose();
76 font = new ScalableFont(fontPath, size, GameMain.Instance.GraphicsDevice, dynamicLoading, isCJK)
77 {
78 ForceUpperCase = element.GetAttributeBool("forceuppercase", false)
79 };
80 if (!isCJK)
81 {
82 cjkFont = ExtractCjkFont(element)
83 ?? new ScalableFont("Content/Fonts/NotoSans/NotoSansCJKsc-Bold.otf",
84 font.Size, GameMain.Instance.GraphicsDevice, dynamicLoading: true, isCJK: true);
85 cjkFont.ForceUpperCase = font.ForceUpperCase;
86 }
87 Language = GameSettings.CurrentConfig.Language;
88 }
89
90 public override void Dispose()
91 {
92 font?.Dispose(); font = null;
93 cjkFont?.Dispose(); cjkFont = null;
94 }
95
96 private ScalableFont ExtractCjkFont(ContentXElement element)
97 {
98 foreach (var subElement in element.Elements().Reverse())
99 {
100 if (subElement.NameAsIdentifier() != "override") { continue; }
101
102 if (subElement.GetAttributeBool("iscjk", false))
103 {
104 return new ScalableFont(subElement, GameMain.Instance.GraphicsDevice);
105 }
106 }
107 return null;
108 }
109
110 private string GetFontFilePath(ContentXElement element)
111 {
112 foreach (var subElement in element.Elements())
113 {
114 if (!subElement.Name.ToString().Equals("override", StringComparison.OrdinalIgnoreCase)) { continue; }
115 if (GameSettings.CurrentConfig.Language == subElement.GetAttributeIdentifier("language", "").ToLanguageIdentifier())
116 {
117 return subElement.GetAttributeContentPath("file")?.Value;
118 }
119 }
120 return element.GetAttributeContentPath("file")?.Value;
121 }
122
123 private uint GetFontSize(XElement element, uint defaultSize = 14)
124 {
125 //check if any of the language override fonts want to override the font size as well
126 foreach (var subElement in element.Elements())
127 {
128 if (!subElement.Name.ToString().Equals("override", StringComparison.OrdinalIgnoreCase)) { continue; }
129 if (GameSettings.CurrentConfig.Language == subElement.GetAttributeIdentifier("language", "").ToLanguageIdentifier())
130 {
131 uint overrideFontSize = GetFontSize(subElement, 0);
132 if (overrideFontSize > 0) { return (uint)Math.Round(overrideFontSize * GameSettings.CurrentConfig.Graphics.TextScale); }
133 }
134 }
135
136 foreach (var subElement in element.Elements())
137 {
138 if (!subElement.Name.ToString().Equals("size", StringComparison.OrdinalIgnoreCase)) { continue; }
139 Point maxResolution = subElement.GetAttributePoint("maxresolution", new Point(int.MaxValue, int.MaxValue));
140 if (GameMain.GraphicsWidth <= maxResolution.X && GameMain.GraphicsHeight <= maxResolution.Y)
141 {
142 return (uint)Math.Round(subElement.GetAttributeInt("size", 14) * GameSettings.CurrentConfig.Graphics.TextScale);
143 }
144 }
145 return (uint)Math.Round(defaultSize * GameSettings.CurrentConfig.Graphics.TextScale);
146 }
147
148 private bool GetFontDynamicLoading(XElement element)
149 {
150 foreach (var subElement in element.Elements())
151 {
152 if (!subElement.Name.ToString().Equals("override", StringComparison.OrdinalIgnoreCase)) { continue; }
153 if (GameSettings.CurrentConfig.Language == subElement.GetAttributeIdentifier("language", "").ToLanguageIdentifier())
154 {
155 return subElement.GetAttributeBool("dynamicloading", false);
156 }
157 }
158 return element.GetAttributeBool("dynamicloading", false);
159 }
160
161 private bool GetIsCJK(XElement element)
162 {
163 foreach (var subElement in element.Elements())
164 {
165 if (!subElement.Name.ToString().Equals("override", StringComparison.OrdinalIgnoreCase)) { continue; }
166 if (GameSettings.CurrentConfig.Language == subElement.GetAttributeIdentifier("language", "").ToLanguageIdentifier())
167 {
168 return subElement.GetAttributeBool("iscjk", false);
169 }
170 }
171 return element.GetAttributeBool("iscjk", false);
172 }
