24 public AIObjectiveExtinguishFire(Character character, Hull targetHull, AIObjectiveManager objectiveManager, float priorityModifier = 1)
25 : base(character, objectiveManager, priorityModifier)
26 {
27 this.targetHull = targetHull;

30 protected override float GetPriority()
31 {
32 if (!IsAllowed)
33 {
34 Priority = 0;
35 Abandon = true;
36 return Priority;
37 }
38 bool isOrder = objectiveManager.HasOrder<AIObjectiveExtinguishFires>();
39 if (!isOrder && Character.CharacterList.Any(c => c.CurrentHull == targetHull && !HumanAIController.IsFriendly(c) && HumanAIController.IsActive(c)))
40 {
41 // Don't go into rooms with any enemies, unless it's an order
42 Priority = 0;
43 Abandon = true;
44 }
45 else
46 {
47 float yDist = Math.Abs(character.WorldPosition.Y - targetHull.WorldPosition.Y);
48 yDist = yDist > 100 ? yDist * 3 : 0;
49 float dist = Math.Abs(character.WorldPosition.X - targetHull.WorldPosition.X) + yDist;
50 float distanceFactor = MathHelper.Lerp(1, 0.1f, MathUtils.InverseLerp(0, 5000, dist));
51 if (targetHull == character.CurrentHull || HumanAIController.VisibleHulls.Contains(targetHull))
52 {
53 distanceFactor = 1;
54 }
55 float severity = AIObjectiveExtinguishFires.GetFireSeverity(targetHull);
56 if (severity > 0.75f && !isOrder &&
57 targetHull.RoomName != null &&
58 !targetHull.RoomName.Contains("reactor", StringComparison.OrdinalIgnoreCase) &&
59 !targetHull.RoomName.Contains("engine", StringComparison.OrdinalIgnoreCase) &&
60 !targetHull.RoomName.Contains("command", StringComparison.OrdinalIgnoreCase))
61 {
62 // Ignore severe fires to prevent casualities unless ordered to extinguish.
63 Priority = 0;
64 Abandon = true;
65 }
66 else
67 {
68 float devotion = CumulatedDevotion / 100;
69 Priority = MathHelper.Lerp(0, 100, MathHelper.Clamp(devotion + (severity * distanceFactor * PriorityModifier), 0, 1));
70 }
71 }
72 return Priority;
73 }
74
75 protected override bool CheckObjectiveSpecific() => targetHull.FireSources.None();
76
77 private float sinTime;
78 protected override void Act(float deltaTime)
79 {
80 var extinguisherItem = character.Inventory.FindItemByTag("fireextinguisher".ToIdentifier());
81 if (extinguisherItem == null || extinguisherItem.Condition <= 0.0f || !character.HasEquippedItem(extinguisherItem))
82 {
83 TryAddSubObjective(ref getExtinguisherObjective, () =>
84 {
85 if (character.IsOnPlayerTeam && !character.HasEquippedItem("fireextinguisher".ToIdentifier(), allowBroken: false))
86 {
87 character.Speak(TextManager.Get("DialogFindExtinguisher").Value, null, 2.0f, "findextinguisher".ToIdentifier(), 30.0f);
88 }
89 var getItemObjective = new AIObjectiveGetItem(character, "fireextinguisher".ToIdentifier(), objectiveManager, equip: true)
90 {
91 AllowStealing = true,
92 // If the item is inside an unsafe hull, decrease the priority
93 GetItemPriority = i => HumanAIController.UnsafeHulls.Contains(i.CurrentHull) ? 0.1f : 1
94 };
95 if (objectiveManager.HasOrder<AIObjectiveExtinguishFires>())
96 {
97 getItemObjective.Abandoned += () => character.Speak(TextManager.Get("dialogcannotfindfireextinguisher").Value, null, 0.0f, "dialogcannotfindfireextinguisher".ToIdentifier(), 10.0f);
98 };
99 return getItemObjective;
100 });
101 }
102 else
103 {
104 var extinguisher = extinguisherItem.GetComponent<RepairTool>();
105 if (extinguisher == null)
106 {
107#if DEBUG
108 DebugConsole.ThrowError($"{character.Name}: AIObjectiveExtinguishFire failed - the item \"" + extinguisherItem + "\" has no RepairTool component but is tagged as an extinguisher");
109#endif
110 Abandon = true;
111 return;
112 }
113 foreach (FireSource fs in targetHull.FireSources)
114 {
115 if (fs == null) { continue; }
116 if (fs.Removed) { continue; }
117 if (character.CurrentHull == null)
118 {
119 Abandon = true;
120 break;
121 }
122 float xDist = Math.Abs(character.WorldPosition.X - fs.WorldPosition.X) - fs.DamageRange;
123 float yDist = Math.Abs(character.WorldPosition.Y - fs.WorldPosition.Y);
124 bool inRange = xDist + yDist < extinguisher.Range;
125 // Use the hull position, because the fire x pos is sometimes inside a wall -> the bot can't ever see it and continues running towards the wall.
126 ISpatialEntity lookTarget = character.CurrentHull == targetHull || character.CurrentHull.linkedTo.Contains(targetHull) ? targetHull : fs as ISpatialEntity;
127 bool move = !inRange || !character.CanSeeTarget(lookTarget);
128 if ((inRange && character.CanSeeTarget(lookTarget)) || useExtinquisherTimer > 0)
129 {
130 useExtinquisherTimer += deltaTime;
131 if (useExtinquisherTimer > 2.0f)
132 {
133 useExtinquisherTimer = 0.0f;
134 }
135 // Aim
136 character.CursorPosition = fs.Position;
137 Vector2 fromCharacterToFireSource = fs.WorldPosition - character.WorldPosition;
138 float dist = fromCharacterToFireSource.Length();
139 character.CursorPosition += VectorExtensions.Forward(extinguisherItem.body.TransformedRotation + (float)Math.Sin(sinTime) / 2, dist / 2);
140 if (extinguisherItem.RequireAimToUse)
141 {
142 character.SetInput(InputType.Aim, false, true);
143 sinTime += deltaTime * 10;
144 }
145 character.SetInput(extinguisherItem.IsShootable ? InputType.Shoot : InputType.Use, false, true);
146 extinguisher.Use(deltaTime, character);
147 if (!targetHull.FireSources.Contains(fs))
148 {
149 character.Speak(TextManager.GetWithVariable("DialogPutOutFire", "[roomname]", targetHull.DisplayName, FormatCapitals.Yes).Value, null, 0, "putoutfire".ToIdentifier(), 10.0f);
150 }
151 }
152 if (move)
153 {
154 //go to the first firesource
155 if (TryAddSubObjective(ref gotoObjective, () => new AIObjectiveGoTo(fs, character, objectiveManager, closeEnough: Math.Max(fs.DamageRange, extinguisher.Range * 0.7f))
156 {
157 DialogueIdentifier = "dialogcannotreachfire".ToIdentifier(),
158 TargetName = fs.Hull.DisplayName
159 },
160 onAbandon: () => Abandon = true,
161 onCompleted: () => RemoveSubObjective(ref gotoObjective)))
162 {
163 gotoObjective.requiredCondition = () => character.CanSeeTarget(targetHull);
164 }
165 }
166 else
167 {
168 character.AIController.SteeringManager.Reset();
169 }
170 break;
171 }
172 }
