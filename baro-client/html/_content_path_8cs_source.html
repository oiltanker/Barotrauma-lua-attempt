27 public string Value
28 {
29 get
30 {
31 if (RawValue.IsNullOrEmpty()) { return ""; }
32 if (!cachedValue.IsNullOrEmpty()) { return cachedValue!; }
33
34 string? modName = ContentPackage?.Name;
35
36 var otherMods = OtherModDirRegex.Matches(RawValue ?? throw new NullReferenceException($"{nameof(RawValue)} is null."))
37 .Select(m => m.Groups[1].Value.Trim().ToIdentifier())
38 .Distinct().Where(id => !id.IsEmpty && id != modName).ToHashSet();
39 cachedValue = RawValue!;
40 if (!(ContentPackage is null))
41 {
42 string modPath = Path.GetDirectoryName(ContentPackage.Path)!;
43 cachedValue = cachedValue
44 .Replace(ModDirStr, modPath, StringComparison.OrdinalIgnoreCase)
45 .Replace(string.Format(OtherModDirFmt, ContentPackage.Name), modPath, StringComparison.OrdinalIgnoreCase);
46 if (ContentPackage.SteamWorkshopId != 0)
47 {
48 cachedValue = cachedValue
49 .Replace(string.Format(OtherModDirFmt, ContentPackage.SteamWorkshopId.ToString(CultureInfo.InvariantCulture)), modPath, StringComparison.OrdinalIgnoreCase);
50 }
51 }
52 var allPackages = ContentPackageManager.EnabledPackages.All;
53 foreach (Identifier otherModName in otherMods)
54 {
55 if (!UInt64.TryParse(otherModName.Value, out UInt64 workshopId)) { workshopId = 0; }
56 ContentPackage? otherMod =
57 allPackages.FirstOrDefault(p => workshopId != 0 && p.SteamWorkshopId != 0 && workshopId == p.SteamWorkshopId)
58 ?? allPackages.FirstOrDefault(p => p.Name == otherModName)
59 ?? allPackages.FirstOrDefault(p => p.NameMatches(otherModName))
60 ?? throw new MissingContentPackageException(ContentPackage, otherModName.Value);
61 cachedValue = cachedValue.Replace(string.Format(OtherModDirFmt, otherModName.Value), Path.GetDirectoryName(otherMod.Path));
62 }
63 cachedValue = cachedValue.CleanUpPath();
64 return cachedValue;
65 }
