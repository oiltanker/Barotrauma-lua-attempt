20 public bool ToggleTabMenu()
21 {
22 if (GameMain.NetworkMember != null && GameMain.NetLobbyScreen != null)
23 {
24 GameMain.NetLobbyScreen.CharacterAppearanceCustomizationMenu?.Dispose();
25 GameMain.NetLobbyScreen.CharacterAppearanceCustomizationMenu = null;
26 if (GameMain.NetLobbyScreen.JobSelectionFrame != null) { GameMain.NetLobbyScreen.JobSelectionFrame.Visible = false; }
27 }
28 if (tabMenu == null && !(GameMode is TutorialMode) && !ConversationAction.IsDialogOpen)
29 {
30 tabMenu = new TabMenu();
31 HintManager.OnShowTabMenu();
32 }
33 else
34 {
35 tabMenu?.OnClose();
36 tabMenu = null;
37 NetLobbyScreen.JobInfoFrame = null;
38 }
39 return true;
40 }
