108 public float GetVitalityDecrease(CharacterHealth characterHealth, float strength)
109 {
110 if (strength < Prefab.ActivationThreshold) { return 0.0f; }
111 strength = MathHelper.Clamp(strength, 0.0f, Prefab.MaxStrength);
112 AfflictionPrefab.Effect currentEffect = Prefab.GetActiveEffect(strength);
113 if (currentEffect == null) { return 0.0f; }
114 if (currentEffect.MaxStrength - currentEffect.MinStrength <= 0.0f) { return 0.0f; }
115
116 float currVitalityDecrease = MathHelper.Lerp(
117 currentEffect.MinVitalityDecrease,
118 currentEffect.MaxVitalityDecrease,
119 (strength - currentEffect.MinStrength) / (currentEffect.MaxStrength - currentEffect.MinStrength));
120
121 if (currentEffect.MultiplyByMaxVitality)
122 {
123 currVitalityDecrease *= characterHealth == null ? 100.0f : characterHealth.MaxVitality;
124 }
125
126 return currVitalityDecrease;
127 }

131 {
132 if (Strength < Prefab.ActivationThreshold) { return 0.0f; }
133 AfflictionPrefab.Effect currentEffect = GetActiveEffect();
134 if (currentEffect == null) { return 0.0f; }
135 if (MathUtils.NearlyEqual(currentEffect.MaxGrainStrength, 0f)) { return 0.0f; }
136
137 float amount = MathHelper.Lerp(
138 currentEffect.MinGrainStrength,

143 {
144 return Math.Min(AdditionStrength, 1.0f);

328 public virtual void Update(CharacterHealth characterHealth, Limb targetLimb, float deltaTime)
329 {
330 foreach (AfflictionPrefab.PeriodicEffect periodicEffect in Prefab.PeriodicEffects)
331 {
332 PeriodicEffectTimers[periodicEffect] -= deltaTime;
333 if (PeriodicEffectTimers[periodicEffect] <= 0.0f)
334 {
335 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
336 {
337 PeriodicEffectTimers[periodicEffect] = 0.0f;
338 }
339 else
340 {
341 foreach (StatusEffect statusEffect in periodicEffect.StatusEffects)
342 {
343 ApplyStatusEffect(ActionType.OnActive, statusEffect, 1.0f, characterHealth, targetLimb);
344 PeriodicEffectTimers[periodicEffect] = Rand.Range(periodicEffect.MinInterval, periodicEffect.MaxInterval);
345 }
346 }
347 }
348 }
349
350 AfflictionPrefab.Effect currentEffect = GetActiveEffect();
351 if (currentEffect == null) { return; }
352
353 fluctuationTimer += deltaTime * currentEffect.ScreenEffectFluctuationFrequency;
354 fluctuationTimer %= 1.0f;
355
356 if (currentEffect.StrengthChange < 0) // Reduce diminishing of buffs if boosted
357 {
358 float durationMultiplier = 1 / (1 + (Prefab.IsBuff ? characterHealth.Character.GetStatValue(StatTypes.BuffDurationMultiplier)
359 : characterHealth.Character.GetStatValue(StatTypes.DebuffDurationMultiplier)));
360
361 _strength += currentEffect.StrengthChange * deltaTime * StrengthDiminishMultiplier * durationMultiplier;
362
363 }
364 else if (currentEffect.StrengthChange > 0) // Reduce strengthening of afflictions if resistant
365 {
366 _strength += currentEffect.StrengthChange * deltaTime * (1f - characterHealth.GetResistance(Prefab));
367 }
368 // Don't use the property, because it's virtual and some afflictions like husk overload it for external use.
369 _strength = MathHelper.Clamp(_strength, 0.0f, Prefab.MaxStrength);
370
371 foreach (StatusEffect statusEffect in currentEffect.StatusEffects)
372 {
373 ApplyStatusEffect(ActionType.OnActive, statusEffect, deltaTime, characterHealth, targetLimb);
374 }
375
376 float amount = deltaTime;
377 if (Prefab.GrainBurst > 0)
378 {
379 amount /= Prefab.GrainBurst;
380 }
381 if (PendingAdditionStrength >= 0)
382 {
383 AdditionStrength += amount;
384 PendingAdditionStrength -= deltaTime;
385 }
386 else if (AdditionStrength > 0)
387 {
388 AdditionStrength -= amount;
389 }
390 #if SERVER
391 GameMain.LuaCs.Hook.Call("afflictionUpdate", new object[] { this, characterHealth, targetLimb, deltaTime });
392 #endif
393 }

404 private readonly List<ISerializableEntity> targets = new List<ISerializableEntity>();
405 public void ApplyStatusEffect(ActionType type, StatusEffect statusEffect, float deltaTime, CharacterHealth characterHealth, Limb targetLimb)
406 {
407 if (type == ActionType.OnDamaged && !statusEffect.HasRequiredAfflictions(characterHealth.Character.LastDamage)) { return; }
408
409 statusEffect.SetUser(Source);
410 if (statusEffect.HasTargetType(StatusEffect.TargetType.Character))
411 {
412 statusEffect.Apply(type, deltaTime, characterHealth.Character, characterHealth.Character);
413 }
414 if (targetLimb != null && statusEffect.HasTargetType(StatusEffect.TargetType.Limb))
415 {
416 statusEffect.Apply(type, deltaTime, characterHealth.Character, targetLimb);
417 }
418 if (targetLimb != null && statusEffect.HasTargetType(StatusEffect.TargetType.AllLimbs))
419 {
420 statusEffect.Apply(type, deltaTime, targetLimb.character, targets: targetLimb.character.AnimController.Limbs);
421 }
422 if (statusEffect.HasTargetType(StatusEffect.TargetType.NearbyItems) ||
423 statusEffect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
424 {
425 targets.Clear();
426 targets.AddRange(statusEffect.GetNearbyTargets(characterHealth.Character.WorldPosition, targets));
427 statusEffect.Apply(type, deltaTime, characterHealth.Character, targets);
428 }
