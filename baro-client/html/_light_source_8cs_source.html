487 public void Update(float deltaTime)
488 {
489 float brightness = 1.0f;
490 if (lightSourceParams.BlinkFrequency > 0.0f)
491 {
492 blinkTimer = (blinkTimer + deltaTime * lightSourceParams.BlinkFrequency) % 1.0f;
493 if (blinkTimer > 0.5f)
494 {
495 CurrentBrightness = 0.0f;
496 return;
497 }
498 }
499 if (lightSourceParams.PulseFrequency > 0.0f && lightSourceParams.PulseAmount > 0.0f)
500 {
501 pulseState = (pulseState + deltaTime * lightSourceParams.PulseFrequency) % 1.0f;
502 //oscillate between 0-1
503 brightness *= 1.0f - (float)(Math.Sin(pulseState * MathHelper.TwoPi) + 1.0f) / 2.0f * lightSourceParams.PulseAmount;
504 }
505 if (lightSourceParams.Flicker > 0.0f)
506 {
507 flickerState += deltaTime * lightSourceParams.FlickerSpeed;
508 flickerState %= 255;
509 brightness *= 1.0f - PerlinNoise.GetPerlin(flickerState, flickerState * 0.5f) * lightSourceParams.Flicker;
510 }
511 CurrentBrightness = brightness;
512 }

541 private void CheckHullsInRange(Submarine sub)
542 {
543 //find the list of convexhulls that belong to the sub
544 ConvexHullList chList = null;
545 foreach (var ch in hullsInRange)
546 {
547 if (ch.Submarine == sub)
548 {
549 chList = ch;
550 break;
551 }
552 }
553
554 //not found -> create one
555 if (chList == null)
556 {
557 chList = new ConvexHullList(sub);
558 hullsInRange.Add(chList);
559 NeedsRecalculation = true;
560 }
561
562 foreach (var ch in chList.List)
563 {
564 if (ch.LastVertexChangeTime > lastRecalculationTime && !chList.IsHidden.Contains(ch))
565 {
566 NeedsRecalculation = true;
567 break;
568 }
569 }
570
571 Vector2 lightPos = position;
572 if (ParentSub == null)
573 {
574 //light and the convexhulls are both outside
575 if (sub == null)
576 {
577 if (NeedsHullCheck)
578 {
579 RefreshConvexHullList(chList, lightPos, null);
580 }
581 }
582 //light is outside, convexhulls inside a sub
583 else
584 {
585 lightPos -= sub.Position;
586
587 Rectangle subBorders = sub.Borders;
588 subBorders.Location += sub.HiddenSubPosition.ToPoint() - new Point(0, sub.Borders.Height);
589
590 //only draw if the light overlaps with the sub
591 if (!MathUtils.CircleIntersectsRectangle(lightPos, TextureRange, subBorders))
592 {
593 if (chList.List.Count > 0) { NeedsRecalculation = true; }
594 chList.List.Clear();
595 return;
596 }
597
598 RefreshConvexHullList(chList, lightPos, sub);
599 }
600 }
601 else
602 {
603 //light is inside, convexhull outside
604 if (sub == null) { return; }
605
606 //light and convexhull are both inside the same sub
607 if (sub == ParentSub)
608 {
609 if (NeedsHullCheck)
610 {
611 RefreshConvexHullList(chList, lightPos, sub);
612 }
613 }
614 //light and convexhull are inside different subs
615 else
616 {
617 if (sub.DockedTo.Contains(ParentSub) && !NeedsHullCheck) { return; }
618
619 lightPos -= (sub.Position - ParentSub.Position);
620
621 Rectangle subBorders = sub.Borders;
622 subBorders.Location += sub.HiddenSubPosition.ToPoint() - new Point(0, sub.Borders.Height);
623
624 //don't draw any shadows if the light doesn't overlap with the borders of the sub
625 if (!MathUtils.CircleIntersectsRectangle(lightPos, TextureRange, subBorders))
626 {
627 if (chList.List.Count > 0) { NeedsRecalculation = true; }
628 chList.List.Clear();
629 return;
630 }
631
632 //recalculate vertices if the subs have moved > 5 px relative to each other
633 Vector2 diff = ParentSub.WorldPosition - sub.WorldPosition;
634 if (!diffToSub.TryGetValue(sub, out Vector2 prevDiff))
635 {
636 diffToSub.Add(sub, diff);
637 NeedsRecalculation = true;
638 }
639 else if (Vector2.DistanceSquared(diff, prevDiff) > 5.0f * 5.0f)
640 {
641 diffToSub[sub] = diff;
642 NeedsRecalculation = true;
643 }
644
645 RefreshConvexHullList(chList, lightPos, sub);
646 }
647 }

650 private List<Vector2> FindRaycastHits()
651 {
652 if (!CastShadows || Range < 1.0f || Color.A < 1) { return null; }
653
654 Vector2 drawPos = position;
655 if (ParentSub != null) { drawPos += ParentSub.DrawPosition; }
656
657 var hulls = new List<ConvexHull>();
658 foreach (ConvexHullList chList in hullsInRange)
659 {
660 foreach (ConvexHull hull in chList.List)
661 {
662 if (!chList.IsHidden.Contains(hull)) { hulls.Add(hull); }
663 }
664 foreach (ConvexHull hull in chList.List)
665 {
666 chList.IsHidden.Add(hull);
667 }
668 }
669
670 float bounds = TextureRange;
671 //find convexhull segments that are close enough and facing towards the light source
672 List<Segment> visibleSegments = new List<Segment>();
673 List<SegmentPoint> points = new List<SegmentPoint>();
674 foreach (ConvexHull hull in hulls)
675 {
676 hull.RefreshWorldPositions();
677 hull.GetVisibleSegments(drawPos, visibleSegments, ignoreEdges: false);
678 }
679
680 //add a square-shaped boundary to make sure we've got something to construct the triangles from
681 //even if there aren't enough hull segments around the light source
682
683 //(might be more effective to calculate if we actually need these extra points)
684
685 Vector2 drawOffset = Vector2.Zero;
686 float boundsExtended = bounds;
687 if (OverrideLightTexture != null)
688 {
689 float cosAngle = (float)Math.Cos(Rotation);
690 float sinAngle = -(float)Math.Sin(Rotation);
691
692 var overrideTextureDims = new Vector2(OverrideLightTexture.SourceRect.Width, OverrideLightTexture.SourceRect.Height);
693
694 Vector2 origin = OverrideLightTextureOrigin;
695
696 origin /= Math.Max(overrideTextureDims.X, overrideTextureDims.Y);
697 origin -= Vector2.One * 0.5f;
698
699 if (Math.Abs(origin.X) >= 0.45f || Math.Abs(origin.Y) >= 0.45f)
700 {
701 boundsExtended += 5.0f;
702 }
703
704 origin *= TextureRange;
705
706 drawOffset.X = -origin.X * cosAngle - origin.Y * sinAngle;
707 drawOffset.Y = origin.X * sinAngle + origin.Y * cosAngle;
708 }
709
710 var boundaryCorners = new SegmentPoint[] {
711 new SegmentPoint(new Vector2(drawPos.X + drawOffset.X + boundsExtended, drawPos.Y + drawOffset.Y + boundsExtended), null),
712 new SegmentPoint(new Vector2(drawPos.X + drawOffset.X + boundsExtended, drawPos.Y + drawOffset.Y - boundsExtended), null),
713 new SegmentPoint(new Vector2(drawPos.X + drawOffset.X - boundsExtended, drawPos.Y + drawOffset.Y - boundsExtended), null),
714 new SegmentPoint(new Vector2(drawPos.X + drawOffset.X - boundsExtended, drawPos.Y + drawOffset.Y + boundsExtended), null)
715 };
716
717 for (int i = 0; i < 4; i++)
718 {
719 var s = new Segment(boundaryCorners[i], boundaryCorners[(i + 1) % 4], null);
720 visibleSegments.Add(s);
721 }
722
723 //Generate new points at the intersections between segments
724 //This is necessary for the light volume to generate properly on some subs
725 for (int i = 0; i < visibleSegments.Count; i++)
726 {
727 Vector2 p1a = visibleSegments[i].Start.WorldPos;
728 Vector2 p1b = visibleSegments[i].End.WorldPos;
729
730 for (int j = i + 1; j < visibleSegments.Count; j++)
731 {
732 //ignore intersections between parallel axis-aligned segments
733 if (visibleSegments[i].IsAxisAligned && visibleSegments[j].IsAxisAligned &&
734 visibleSegments[i].IsHorizontal == visibleSegments[j].IsHorizontal)
735 {
736 continue;
737 }
738
739 Vector2 p2a = visibleSegments[j].Start.WorldPos;
740 Vector2 p2b = visibleSegments[j].End.WorldPos;
741
742 if (Vector2.DistanceSquared(p1a, p2a) < 5.0f ||
743 Vector2.DistanceSquared(p1a, p2b) < 5.0f ||
744 Vector2.DistanceSquared(p1b, p2a) < 5.0f ||
745 Vector2.DistanceSquared(p1b, p2b) < 5.0f)
746 {
747 continue;
748 }
749
750 bool intersects;
751 Vector2 intersection = Vector2.Zero;
752 if (visibleSegments[i].IsAxisAligned)
753 {
754 intersects = MathUtils.GetAxisAlignedLineIntersection(p2a, p2b, p1a, p1b, visibleSegments[i].IsHorizontal, out intersection);
755 }
756 else if (visibleSegments[j].IsAxisAligned)
757 {
758 intersects = MathUtils.GetAxisAlignedLineIntersection(p1a, p1b, p2a, p2b, visibleSegments[j].IsHorizontal, out intersection);
759 }
760 else
761 {
762 intersects = MathUtils.GetLineIntersection(p1a, p1b, p2a, p2b, out intersection);
763 }
764
765 if (intersects)
766 {
767 SegmentPoint start = visibleSegments[i].Start;
768 SegmentPoint end = visibleSegments[i].End;
769 SegmentPoint mid = new SegmentPoint(intersection, null);
770 if (visibleSegments[i].ConvexHull?.ParentEntity?.Submarine != null)
771 {
772 mid.Pos -= visibleSegments[i].ConvexHull.ParentEntity.Submarine.DrawPosition;
773 }
774
775 if (Vector2.DistanceSquared(start.WorldPos, mid.WorldPos) < 5.0f ||
776 Vector2.DistanceSquared(end.WorldPos, mid.WorldPos) < 5.0f)
777 {
778 continue;
779 }
780
781 Segment seg1 = new Segment(start, mid, visibleSegments[i].ConvexHull)
782 {
783 IsHorizontal = visibleSegments[i].IsHorizontal,
784 };
785
786 Segment seg2 = new Segment(mid, end, visibleSegments[i].ConvexHull)
787 {
788 IsHorizontal = visibleSegments[i].IsHorizontal
789 };
790
791 visibleSegments[i] = seg1;
792 visibleSegments.Insert(i + 1, seg2);
793 i--;
794 break;
795 }
796 }
797 }
798
799 //remove segments that fall out of bounds
800 for (int i = 0; i < visibleSegments.Count; i++)
801 {
802 Segment s = visibleSegments[i];
803 if (Math.Abs(s.Start.WorldPos.X - drawPos.X - drawOffset.X) > boundsExtended + 1.0f ||
804 Math.Abs(s.Start.WorldPos.Y - drawPos.Y - drawOffset.Y) > boundsExtended + 1.0f ||
805 Math.Abs(s.End.WorldPos.X - drawPos.X - drawOffset.X) > boundsExtended + 1.0f ||
806 Math.Abs(s.End.WorldPos.Y - drawPos.Y - drawOffset.Y) > boundsExtended + 1.0f)
807 {
808 visibleSegments.RemoveAt(i);
809 i--;
810 }
811 else
812 {
813 points.Add(s.Start);
814 points.Add(s.End);
815 }
816 }
817
818 visibleSegments = visibleSegments.OrderBy(s => MathUtils.LineToPointDistanceSquared(s.Start.WorldPos, s.End.WorldPos, drawPos)).ToList();
819
820 var compareCCW = new CompareSegmentPointCW(drawPos);
821 try
822 {
823 points.Sort(compareCCW);
824 }
825 catch (Exception e)
826 {
827 StringBuilder sb = new StringBuilder("Constructing light volumes failed! Light pos: " + drawPos + ", Hull verts:\n");
828 foreach (SegmentPoint sp in points)
829 {
830 sb.AppendLine(sp.Pos.ToString());
831 }
832 DebugConsole.ThrowError(sb.ToString(), e);
833 }
834
835 List<Vector2> output = new List<Vector2>();
836 //List<Pair<int, Vector2>> preOutput = new List<Pair<int, Vector2>>();
837
838 //remove points that are very close to each other
839 for (int i = 0; i < points.Count; i++)
840 {
841 for (int j = Math.Min(i + 4, points.Count-1); j > i; j--)
842 {
843 if (Math.Abs(points[i].WorldPos.X - points[j].WorldPos.X) < 6 &&
844 Math.Abs(points[i].WorldPos.Y - points[j].WorldPos.Y) < 6)
845 {
846 points.RemoveAt(j);
847 }
848 }
849 }
850
851 foreach (SegmentPoint p in points)
852 {
853 Vector2 dir = Vector2.Normalize(p.WorldPos - drawPos);
854 Vector2 dirNormal = new Vector2(-dir.Y, dir.X) * 3;
855
856 //do two slightly offset raycasts to hit the segment itself and whatever's behind it
857 var intersection1 = RayCast(drawPos, drawPos + dir * boundsExtended * 2 - dirNormal, visibleSegments);
858 var intersection2 = RayCast(drawPos, drawPos + dir * boundsExtended * 2 + dirNormal, visibleSegments);
859
860 if (intersection1.index < 0) return null;
861 if (intersection2.index < 0) return null;
862 Segment seg1 = visibleSegments[intersection1.index];
863 Segment seg2 = visibleSegments[intersection2.index];
864
865 bool isPoint1 = MathUtils.LineToPointDistanceSquared(seg1.Start.WorldPos, seg1.End.WorldPos, p.WorldPos) < 25.0f;
866 bool isPoint2 = MathUtils.LineToPointDistanceSquared(seg2.Start.WorldPos, seg2.End.WorldPos, p.WorldPos) < 25.0f;
867
868 if (isPoint1 && isPoint2)
869 {
870 //hit at the current segmentpoint -> place the segmentpoint into the list
871 output.Add(p.WorldPos);
872
873 foreach (ConvexHullList hullList in hullsInRange)
874 {
875 hullList.IsHidden.Remove(p.ConvexHull);
876 hullList.IsHidden.Remove(seg1.ConvexHull);
877 hullList.IsHidden.Remove(seg2.ConvexHull);
878 }
879 }
880 else if (intersection1.index != intersection2.index)
881 {
882 //the raycasts landed on different segments
883 //we definitely want to generate new geometry here
884 output.Add(isPoint1 ? p.WorldPos : intersection1.pos);
885 output.Add(isPoint2 ? p.WorldPos : intersection2.pos);
886
887 foreach (ConvexHullList hullList in hullsInRange)
888 {
889 hullList.IsHidden.Remove(p.ConvexHull);
890 hullList.IsHidden.Remove(seg1.ConvexHull);
891 hullList.IsHidden.Remove(seg2.ConvexHull);
892 }
893 }
894 //if neither of the conditions above are met, we just assume
895 //that the raycasts both resulted on the same segment
896 //and creating geometry here would be wasteful
897 }
898
899 //remove points that are very close to each other
900 for (int i = 0; i < output.Count - 1; i++)
901 {
902 for (int j = Math.Min(i + 4, output.Count - 1); j > i; j--)
903 {
904 if (Math.Abs(output[i].X - output[j].X) < 6 &&
905 Math.Abs(output[i].Y - output[j].Y) < 6)
906 {
907 output.RemoveAt(j);
908 }
909 }
910 }
911
912 return output;

915 private (int index, Vector2 pos) RayCast(Vector2 rayStart, Vector2 rayEnd, List<Segment> segments)
916 {
917 Vector2? closestIntersection = null;
918 int segment = -1;
919
920 float minX = Math.Min(rayStart.X, rayEnd.X);
921 float maxX = Math.Max(rayStart.X, rayEnd.X);
922 float minY = Math.Min(rayStart.Y, rayEnd.Y);
923 float maxY = Math.Max(rayStart.Y, rayEnd.Y);
924
925 for (int i = 0; i < segments.Count; i++)
926 {
927 Segment s = segments[i];
928
929 //segment's end position always has a higher or equal y coordinate than the start position
930 //so we can do this comparison and skip segments that are at the wrong side of the ray
931 /*if (s.End.WorldPos.Y < s.Start.WorldPos.Y)
932 {
933 System.Diagnostics.Debug.Assert(s.End.WorldPos.Y >= s.Start.WorldPos.Y,
934 "LightSource raycast failed. Segment's end positions should never be below the start position. Parent entity: " + (s.ConvexHull?.ParentEntity == null ? "null" : s.ConvexHull.ParentEntity.ToString()));
935 }*/
936 if (s.Start.WorldPos.Y > maxY || s.End.WorldPos.Y < minY) { continue; }
937 //same for the x-axis
938 if (s.Start.WorldPos.X > s.End.WorldPos.X)
939 {
940 if (s.Start.WorldPos.X < minX) continue;
941 if (s.End.WorldPos.X > maxX) continue;
942 }
943 else
944 {
945 if (s.End.WorldPos.X < minX) continue;
946 if (s.Start.WorldPos.X > maxX) continue;
947 }
948
949 bool intersects;
950 Vector2 intersection;
951 if (s.IsAxisAligned)
952 {
953 intersects = MathUtils.GetAxisAlignedLineIntersection(rayStart, rayEnd, s.Start.WorldPos, s.End.WorldPos, s.IsHorizontal, out intersection);
954 }
955 else
956 {
957 intersects = MathUtils.GetLineIntersection(rayStart, rayEnd, s.Start.WorldPos, s.End.WorldPos, out intersection);
958 }
959
960 if (intersects)
961 {
962 closestIntersection = intersection;
963
964 rayEnd = intersection;
965 minX = Math.Min(rayStart.X, rayEnd.X);
966 maxX = Math.Max(rayStart.X, rayEnd.X);
967 minY = Math.Min(rayStart.Y, rayEnd.Y);
968 maxY = Math.Max(rayStart.Y, rayEnd.Y);
969
970 segment = i;
971 }
972 }
973
974 return (segment, closestIntersection == null ? rayEnd : (Vector2)closestIntersection);
