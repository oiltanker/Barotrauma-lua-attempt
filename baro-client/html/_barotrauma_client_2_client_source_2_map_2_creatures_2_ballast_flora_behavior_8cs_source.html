288 public void ClientRead(IReadMessage msg, NetworkHeader header)
289 {
290 switch (header)
291 {
292 case NetworkHeader.Infect:
293 int infectBranch = -1;
294 ushort itemId = msg.ReadUInt16();
295 bool infect = msg.ReadBoolean();
296 if (infect)
297 {
298 infectBranch = msg.ReadInt32();
299 }
300
301 Entity? entity = Entity.FindEntityByID(itemId);
302 if (entity is Item item)
303 {
304 if (infect)
305 {
306 ClaimTarget(item, Branches.FirstOrDefault(b => b.ID == infectBranch));
307 }
308 else
309 {
310 RemoveClaim(item);
311 }
312 }
313 else
314 {
315 DebugConsole.AddWarning($"Received Infect.{infect} Network Header with invalid item ID: {itemId}, which belongs to {entity?.ToString() ?? "null!"}");
316 }
317 break;
318 case NetworkHeader.BranchCreate:
319 int parentId = msg.ReadInt32();
320 BallastFloraBranch branch = ReadBranch(msg);
321 BallastFloraBranch? parent = Branches.FirstOrDefault(b => b.ID == parentId);
322
323 if (parent == null)
324 {
325 DebugConsole.AddWarning($"Received BranchCreate with an invalid parent ID: {parentId}, Maximum ID is {Branches.Max(b => b.ID)}");
326 }
327
328 UpdateConnections(branch, parent);
329 Branches.Add(branch);
330 OnBranchGrowthSuccess(branch);
331 break;
332 case NetworkHeader.BranchRemove:
333
334 int removedBranchId = msg.ReadInt32();
335 BallastFloraBranch? removedBranch = Branches.FirstOrDefault(b => b.ID == removedBranchId);
336 if (removedBranch != null)
337 {
338 RemoveBranch(removedBranch);
339 }
340 else
341 {
342 DebugConsole.AddWarning($"Received BranchRemove for a branch that doesn't exist. ID: {removedBranchId}, Maximum ID is {Branches.Max(b => b.ID)}");
343 }
344
345 break;
346 case NetworkHeader.BranchDamage:
347 int damageBranchId = msg.ReadInt32();
348 float health = msg.ReadSingle();
349 BallastFloraBranch? damagedBranch = Branches.FirstOrDefault(b => b.ID == damageBranchId);
350 if (damagedBranch != null)
351 {
352 damagedBranch.Health = health;
353 }
354 else
355 {
356 DebugConsole.AddWarning($"Received BranchDamage for a branch that doesn't exist. ID: {damageBranchId}, Maximum ID is {Branches.Max(b => b.ID)}");
357 }
358 break;
359 case NetworkHeader.Kill:
360 Kill();
361 break;
362 case NetworkHeader.Remove:
363 Remove();
364 break;
365 }
366
367 PowerConsumptionTimer = msg.ReadSingle();
368 }

370 private BallastFloraBranch ReadBranch(IReadMessage msg)
371 {
372 int id = msg.ReadInt32();
373 bool isRootGrowth = msg.ReadBoolean();
374 byte type = (byte)msg.ReadRangedInteger(0b0000, 0b1111);
375 byte sides = (byte)msg.ReadRangedInteger(0b0000, 0b1111);
376 int flowerConfig = msg.ReadRangedInteger(0, 0xFFF);
377 int leafConfig = msg.ReadRangedInteger(0, 0xFFF);
378 int maxHealth = msg.ReadUInt16();
379 int posX = msg.ReadInt32(), posY = msg.ReadInt32();
380 int parentBranchIndex = msg.ReadInt32();
381 Vector2 pos = new Vector2(posX * VineTile.Size, posY * VineTile.Size);
382
383 BallastFloraBranch? parentBranch = parentBranchIndex < 0 || parentBranchIndex >= Branches.Count ? null : Branches[parentBranchIndex];
384
385 return new BallastFloraBranch(this, parentBranch, pos, (VineTileType)type, FoliageConfig.Deserialize(flowerConfig), FoliageConfig.Deserialize(leafConfig))
386 {
387 ID = id,
388 MaxHealth = maxHealth,
389 Sides = (TileSide) sides,
390 IsRootGrowth = isRootGrowth
391 };
392 }
