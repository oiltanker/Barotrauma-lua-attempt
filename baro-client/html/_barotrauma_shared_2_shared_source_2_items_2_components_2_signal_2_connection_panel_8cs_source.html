98 public void InitializeLinks()
99 {
100 foreach (Connection c in Connections)
101 {
102 c.ConnectLinked();
103 }
104
105 if (disconnectedWireIds != null)
106 {
107 foreach (ushort disconnectedWireId in disconnectedWireIds)
108 {
109 if (!(Entity.FindEntityByID(disconnectedWireId) is Item wireItem)) { continue; }
110 Wire wire = wireItem.GetComponent<Wire>();
111 if (wire != null)
112 {
113 if (Item.ItemList.Any(it => it != item && (it.GetComponent<ConnectionPanel>()?.DisconnectedWires.Contains(wire) ?? false)))
114 {
115 if (wire.Item.body != null) { wire.Item.body.Enabled = false; }
116 wire.IsActive = false;
117 wire.UpdateSections();
118 }
119 DisconnectedWires.Add(wire);
120 base.IsActive = true;
121 }
122 }
123 }

125 linksInitialized = true;

142 public void MoveConnectedWires(Vector2 amount)
143 {
144 Vector2 wireNodeOffset = item.Submarine == null ? Vector2.Zero : item.Submarine.HiddenSubPosition + amount;
145 foreach (Connection c in Connections)
146 {
147 foreach (Wire wire in c.Wires)
148 {

158
159 void TryMoveWire(Wire wire)
160 {
161#if CLIENT
162 if (wire.Item.IsSelected) { return; }
163#endif
164 var wireNodes = wire.GetNodes();
165 if (wireNodes.Count == 0) { return; }
166
167 if (Submarine.RectContains(item.Rect, wireNodes[0] + wireNodeOffset))
168 {
169 wire.MoveNode(0, amount);
170 }
171 else if (Submarine.RectContains(item.Rect, wireNodes[wireNodes.Count - 1] + wireNodeOffset))
172 {
173 wire.MoveNode(wireNodes.Count - 1, amount);
174 }
175 }

235 public bool CheckCharacterSuccess(Character character)
236 {
237 if (character == null) { return false; }
238 //no electrocution in sub editor
239 if (Screen.Selected == GameMain.SubEditorScreen) { return true; }
240
241 var reactor = item.GetComponent<Reactor>();
242 if (reactor != null)
243 {
244 //reactors that arent generating power atm can be rewired without the risk of electrical shock
245 if (MathUtils.NearlyEqual(reactor.CurrPowerConsumption, 0.0f)) { return true; }
246 }
247 var powerContainer = item.GetComponent<PowerContainer>();
248 if (powerContainer != null)
249 {
250 //empty batteries/supercapacitors can be rewired without the risk of electrical shock
251 //non-empty ones always have a chance of zapping the user
252 if (powerContainer.Charge <= 0.0f) { return true; }
253 }
254 var powered = item.GetComponent<Powered>();
255 if (powered != null && powerContainer == null)
256 {
257 //unpowered panels can be rewired without the risk of electrical shock
258 if (powered.Voltage < 0.1f) { return true; }
259 }
260
261 float degreeOfSuccess = DegreeOfSuccess(character);
262 if (Rand.Range(0.0f, 0.5f) < degreeOfSuccess) { return true; }
263
264 ApplyStatusEffects(ActionType.OnFailure, 1.0f, character);
265 return false;
266 }

268 public override void Load(ContentXElement element, bool usePrefabValues, IdRemap idRemap)
269 {
270 base.Load(element, usePrefabValues, idRemap);
271
272 List<Connection> loadedConnections = new List<Connection>();
273
274 foreach (var subElement in element.Elements())
275 {
276 switch (subElement.Name.ToString())
277 {
278 case "input":
279 loadedConnections.Add(new Connection(subElement, this, idRemap));
280 break;
281 case "output":
282 loadedConnections.Add(new Connection(subElement, this, idRemap));
283 break;
284 }
285 }
286
287 for (int i = 0; i < loadedConnections.Count && i < Connections.Count; i++)
288 {
289 if (loadedConnections[i].wireId.Length == Connections[i].wireId.Length)
290 {
291 loadedConnections[i].wireId.CopyTo(Connections[i].wireId, 0);
292 }
293 else
294 {
295 //backwards compatibility when maximum number of wires has changed
296 foreach (ushort id in loadedConnections[i].wireId)
297 {
298 for (int j = 0; j < Connections[i].wireId.Length; j++)
299 {
300 if (Connections[i].wireId[j] == 0)
301 {
302 Connections[i].wireId[j] = id;
303 break;
304 }
305 }
306 }
307 }
308 }
309
310 disconnectedWireIds = element.GetAttributeUshortArray("disconnectedwires", Array.Empty<ushort>()).ToList();
311 for (int i = 0; i < disconnectedWireIds.Count; i++)
312 {
313 disconnectedWireIds[i] = idRemap.GetOffsetId(disconnectedWireIds[i]);
314 }
315 }
