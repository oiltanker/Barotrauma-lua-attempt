165 Static = element.GetAttributeBool("static", Static);
166 StaticSight = element.GetAttributeBool("staticsight", StaticSight);
167 StaticSound = element.GetAttributeBool("staticsound", StaticSound);
168 if (Static)

171 StaticSight = true;
172 }
173 SonarDisruption = element.GetAttributeFloat("sonardisruption", 0.0f);
174 string label = element.GetAttributeString("sonarlabel", "");
175 SonarLabel = TextManager.Get(label).Fallback(label);
176 SonarIconIdentifier = element.GetAttributeIdentifier("sonaricon", Identifier.Empty);
177 Type = element.GetAttributeEnum("type", TargetType.Any);
178 Reset();

187 public void Update(float deltaTime)
188 {
189 InDetectable = false;
190 if (!Static && FadeOutTime > 0)
191 {
192 // The aitarget goes silent/invisible if the components don't keep it active
193 if (!StaticSight && SightRange > 0)
194 {
195 DecreaseSightRange(deltaTime);
196 }
197 if (!StaticSound && SoundRange > 0)
198 {
199 DecreaseSoundRange(deltaTime);
200 }
201 }
202 }

204 public void IncreaseSoundRange(float deltaTime, float speed = 1)
205 {
206 SoundRange += speed * deltaTime * (MaxSoundRange / FadeOutTime);
207 }

212 }

215 {
216 SoundRange -= speed * deltaTime * (MaxSoundRange / FadeOutTime);
217 }

224 public bool IsWithinSector(Vector2 worldPosition)
225 {
226 if (sectorRad >= MathHelper.TwoPi) { return true; }
227 Vector2 diff = worldPosition - WorldPosition;
228 return Math.Abs(MathUtils.GetShortestAngle(MathUtils.VectorToAngle(diff), MathUtils.VectorToAngle(sectorDir))) <= sectorRad * 0.5f;
229 }

238 {
239 if (Static)

242 SoundRange = MaxSoundRange;
243 }
244 else
245 {
246 // Non-static ai targets must be kept alive by a custom logic (e.g. item components)
247 SightRange = StaticSight ? MaxSightRange : MinSightRange;
248 SoundRange = StaticSound ? MaxSoundRange : MinSoundRange;
249 }
