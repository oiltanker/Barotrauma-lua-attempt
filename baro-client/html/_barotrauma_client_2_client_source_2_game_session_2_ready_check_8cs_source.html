139 public static void ClientRead(IReadMessage inc)
140 {
141 ReadyCheckState state = (ReadyCheckState) inc.ReadByte();
142 CrewManager? crewManager = GameMain.GameSession?.CrewManager;
143 List<Client> otherClients = GameMain.Client.ConnectedClients;
144 if (crewManager == null || otherClients == null)
145 {
146 if (state == ReadyCheckState.Start)
147 {
148 SendState(ReadyStatus.No);
149 }
150 return;
151 }
152
153 switch (state)
154 {
155 case ReadyCheckState.Start:
156 bool isOwn = false;
157 byte authorId = 0;
158
159 float duration = inc.ReadSingle();
160 string author = inc.ReadString();
161 bool hasAuthor = inc.ReadBoolean();
162
163 if (hasAuthor)
164 {
165 authorId = inc.ReadByte();
166 isOwn = authorId == GameMain.Client.ID;
167 }
168
169 ushort clientCount = inc.ReadUInt16();
170 List<byte> clients = new List<byte>();
171 for (int i = 0; i < clientCount; i++)
172 {
173 clients.Add(inc.ReadByte());
174 }
175
176 ReadyCheck rCheck = new ReadyCheck(clients, duration);
177 crewManager.ActiveReadyCheck = rCheck;
178
179 if (isOwn)
180 {
181 SendState(ReadyStatus.Yes);
182 rCheck.CreateResultsMessage();
183 }
184 else
185 {
186 rCheck.CreateMessageBox(author);
187 }
188
189 if (hasAuthor && rCheck.Clients.ContainsKey(authorId))
190 {
191 rCheck.Clients[authorId] = ReadyStatus.Yes;
192 }
193 break;
194 case ReadyCheckState.Update:
195 float time = inc.ReadSingle();
196 ReadyStatus newState = (ReadyStatus) inc.ReadByte();
197 byte targetId = inc.ReadByte();
198 if (crewManager.ActiveReadyCheck != null)
199 {
200 crewManager.ActiveReadyCheck.time = time;
201 crewManager.ActiveReadyCheck?.UpdateState(targetId, newState);
202 }
203 break;
204 case ReadyCheckState.End:
205 ushort count = inc.ReadUInt16();
206 for (int i = 0; i < count; i++)
207 {
208 byte id = inc.ReadByte();
209 ReadyStatus status = (ReadyStatus) inc.ReadByte();
210 crewManager.ActiveReadyCheck?.UpdateState(id, status);
211 }
212
213 crewManager.ActiveReadyCheck?.EndReadyCheck();
214 crewManager.ActiveReadyCheck?.msgBox?.Close();
215 crewManager.ActiveReadyCheck = null;
216 break;
217 }
218 }
