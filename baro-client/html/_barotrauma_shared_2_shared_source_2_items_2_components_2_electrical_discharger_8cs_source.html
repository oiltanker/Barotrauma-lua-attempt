134 public override void Update(float deltaTime, Camera cam)
135 {
136#if CLIENT
137 frameOffset = Rand.Int(electricitySprite.FrameCount);
138#endif
139 if (waitForVoltageRecalculation)
140 {
141 waitForVoltageRecalculation = false;
142 return;
143 }
144
145 if (timer <= 0.0f)
146 {
147 IsActive = false;
148 return;
149 }
150
151 timer -= deltaTime;
152 if (charging)
153 {
154 if (GetAvailableInstantaneousBatteryPower() >= PowerConsumption)
155 {
156 List<PowerContainer> batteries = GetConnectedBatteries();
157 float neededPower = PowerConsumption;
158 while (neededPower > 0.0001f && batteries.Count > 0)
159 {
160 batteries.RemoveAll(b => b.Charge <= 0.0001f || b.MaxOutPut <= 0.0001f);
161 float takePower = neededPower / batteries.Count;
162 takePower = Math.Min(takePower, batteries.Min(b => Math.Min(b.Charge * 3600.0f, b.MaxOutPut)));
163 foreach (PowerContainer battery in batteries)
164 {
165 neededPower -= takePower;
166 battery.Charge -= takePower / 3600.0f;
167 #if SERVER
168 if (GameMain.Server != null) { battery.Item.CreateServerEvent(battery); }
169 #endif
170 }
171 }
172 Discharge();
173
174 }
175 else if (Voltage > MinVoltage)
176 {
177 Discharge();
178 }
179 }

215 public void FindNodes(Vector2 worldPosition, float range)
216 {
217 //see which submarines are within range so we can skip structures that are in far-away subs
218 List<Submarine> submarinesInRange = new List<Submarine>();
219 foreach (Submarine sub in Submarine.Loaded)
220 {
221 if (item.Submarine == sub)
222 {
223 submarinesInRange.Add(sub);
224 }
225 else
226 {
227 Rectangle subBorders = new Rectangle(
228 sub.Borders.X - (int)range, sub.Borders.Y + (int)range,
229 sub.Borders.Width + (int)(range * 2), sub.Borders.Height + (int)(range * 2));
230 subBorders.Location += MathUtils.ToPoint(sub.SubBody.Position);
231 if (Submarine.RectContains(subBorders, worldPosition))
232 {
233 submarinesInRange.Add(sub);
234 }
235 }
236 }
237
238 //get all walls within range
239 List<Entity> entitiesInRange = new List<Entity>(100);
240 foreach (Structure structure in Structure.WallList)
241 {
242 if (!structure.HasBody || structure.IsPlatform) { continue; }
243 if (structure.Submarine != null&& !submarinesInRange.Contains(structure.Submarine)) { continue; }
244
245 var structureWorldRect = structure.WorldRect;
246 if (worldPosition.X < structureWorldRect.X - range) continue;
247 if (worldPosition.X > structureWorldRect.Right + range) continue;
248 if (worldPosition.Y > structureWorldRect.Y + range) continue;
249 if (worldPosition.Y < structureWorldRect.Y -structureWorldRect.Height - range) continue;
250
251 if (structure.Submarine != null)
252 {
253 if (!submarinesInRange.Contains(structure.Submarine)) { continue; }
254 if (OutdoorsOnly)
255 {
256 //check if the structure is within a hull
257 //add a small offset away from the sub's center so structures right at the edge of a hull are still valid
258 Vector2 offset = Vector2.Normalize(structure.WorldPosition - structure.Submarine.WorldPosition);
259 if (Hull.FindHull(structure.Position + offset * Submarine.GridSize, useWorldCoordinates: false) != null) { continue; }
260 }
261 }
262
263 entitiesInRange.Add(structure);
264 }
265
266 foreach (Character character in Character.CharacterList)
267 {
268 if (!character.Enabled) continue;
269 if (OutdoorsOnly && character.Submarine != null) continue;
270 if (character.Submarine != null && !submarinesInRange.Contains(character.Submarine)) continue;
271
272 if (Vector2.DistanceSquared(character.WorldPosition, worldPosition) < range * range * RangeMultiplierInWalls)
273 {
274 entitiesInRange.Add(character);
275 }
276 }
277
278 nodes.Clear();
279 nodes.Add(new Node(worldPosition, -1));
280 FindNodes(entitiesInRange, worldPosition, 0, range);
281
282 //construct final nodes (w/ lengths and angles so they don't have to be recalculated when rendering the discharge)
283 for (int i = 0; i < nodes.Count; i++)
284 {
285 if (nodes[i].ParentIndex < 0) continue;
286 Node parentNode = nodes[nodes[i].ParentIndex];
287 float length = Vector2.Distance(nodes[i].WorldPosition, parentNode.WorldPosition) * Rand.Range(1.0f, 1.25f);
288 float angle = MathUtils.VectorToAngle(parentNode.WorldPosition - nodes[i].WorldPosition);
289 nodes[i] = new Node(nodes[i].WorldPosition, nodes[i].ParentIndex, length, angle);
290 }
291 }

293 private void FindNodes(List<Entity> entitiesInRange, Vector2 currPos, int parentNodeIndex, float currentRange)
294 {
295 if (currentRange <= 0.0f || nodes.Count >= MaxNodes) return;
296
297 //find the closest structure
298 int closestIndex = -1;
299 float closestDist = float.MaxValue;
300 for (int i = 0; i < entitiesInRange.Count; i++)
301 {
302 float dist = float.MaxValue;
303
304 if (entitiesInRange[i] is Structure structure)
305 {
306 if (structure.IsHorizontal)
307 {
308 dist = Math.Abs(structure.WorldPosition.Y - currPos.Y);
309 if (currPos.X < structure.WorldRect.X)
310 dist += structure.WorldRect.X - currPos.X;
311 else if (currPos.X > structure.WorldRect.Right)
312 dist += currPos.X - structure.WorldRect.Right;
313 }
314 else
315 {
316 dist = Math.Abs(structure.WorldPosition.X - currPos.X);
317 if (currPos.Y < structure.WorldRect.Y - structure.Rect.Height)
318 dist += (structure.WorldRect.Y - structure.Rect.Height) - currPos.Y;
319 else if (currPos.Y > structure.WorldRect.Y)
320 dist += currPos.Y - structure.WorldRect.Y;
321 }
322 }
323 else if (entitiesInRange[i] is Character character)
324 {
325 dist = Vector2.Distance(character.WorldPosition, currPos);
326 }
327
328 if (dist < closestDist)
329 {
330 closestIndex = i;
331 closestDist = dist;
332 }
333 }
334
335 if (closestIndex == -1 || closestDist > currentRange)
336 {
337 //nothing in range, create some arcs to random directions
338 for (int i = 0; i < Rand.Int(4); i++)
339 {
340 Vector2 targetPos = currPos + Rand.Vector(MaxNodeDistance * Rand.Range(0.5f, 1.5f));
341 nodes.Add(new Node(targetPos, parentNodeIndex));
342 }
343 return;
344 }
345 currentRange -= closestDist;
346
347 if (entitiesInRange[closestIndex] is Structure targetStructure)
348 {
349 if (targetStructure.IsHorizontal)
350 {
351 //which side of the structure to add the nodes to
352 //if outside the sub, use the sides that's furthers from the sub's center position
353 //otherwise the side that's closer to the previous node
354 int yDir = OutdoorsOnly && targetStructure.Submarine != null ?
355 Math.Sign(targetStructure.WorldPosition.Y - targetStructure.Submarine.WorldPosition.Y) :
356 Math.Sign(currPos.Y - targetStructure.WorldPosition.Y);
357
358 int sectionIndex = targetStructure.FindSectionIndex(currPos, world: true, clamp: true);
359 if (sectionIndex == -1) { return; }
360 Vector2 sectionPos = targetStructure.SectionPosition(sectionIndex, world: true);
361 Vector2 targetPos =
362 new Vector2(
363 MathHelper.Clamp(sectionPos.X, targetStructure.WorldRect.X, targetStructure.WorldRect.Right),
364 sectionPos.Y + targetStructure.BodyHeight / 2 * yDir);
365
366 //create nodes from the current position to the closest point on the structure
367 AddNodesBetweenPoints(currPos, targetPos, 0.25f, ref parentNodeIndex);
368
369 //add a node at the closest point
370 nodes.Add(new Node(targetPos, parentNodeIndex));
371 int nodeIndex = nodes.Count - 1;
372 entitiesInRange.RemoveAt(closestIndex);
373
374 float newRange = currentRange - (targetStructure.Rect.Width / 2) * (1.0f / RangeMultiplierInWalls);
375
376 //continue the discharge to the left edge of the structure and extend from there
377 int leftNodeIndex = nodeIndex;
378 Vector2 leftPos = targetStructure.SectionPosition(0, world: true);
379 leftPos.Y += targetStructure.BodyHeight / 2 * yDir;
380 AddNodesBetweenPoints(targetPos, leftPos, 0.05f, ref leftNodeIndex);
381 nodes.Add(new Node(leftPos, leftNodeIndex));
382 FindNodes(entitiesInRange, leftPos, nodes.Count - 1, newRange);
383
384 //continue the discharge to the right edge of the structure and extend from there
385 int rightNodeIndex = nodeIndex;
386 Vector2 rightPos = targetStructure.SectionPosition(targetStructure.SectionCount - 1, world: true);
387 leftPos.Y += targetStructure.BodyHeight / 2 * yDir;
388 AddNodesBetweenPoints(targetPos, rightPos, 0.05f, ref rightNodeIndex);
389 nodes.Add(new Node(rightPos, rightNodeIndex));
390 FindNodes(entitiesInRange, rightPos, nodes.Count - 1, newRange);
391 }
392 else
393 {
394 int xDir = OutdoorsOnly && targetStructure.Submarine != null ?
395 Math.Sign(targetStructure.WorldPosition.X - targetStructure.Submarine.WorldPosition.X) :
396 Math.Sign(currPos.X - targetStructure.WorldPosition.X);
397
398 int sectionIndex = targetStructure.FindSectionIndex(currPos, world: true, clamp: true);
399 if (sectionIndex == -1) { return; }
400 Vector2 sectionPos = targetStructure.SectionPosition(sectionIndex, world: true);
401
402 Vector2 targetPos = new Vector2(
403 sectionPos.X + targetStructure.BodyWidth / 2 * xDir,
404 MathHelper.Clamp(sectionPos.Y, targetStructure.WorldRect.Y - targetStructure.Rect.Height, targetStructure.WorldRect.Y));
405
406 //create nodes from the current position to the closest point on the structure
407 AddNodesBetweenPoints(currPos, targetPos, 0.25f, ref parentNodeIndex);
408
409 //add a node at the closest point
410 nodes.Add(new Node(targetPos, parentNodeIndex));
411 int nodeIndex = nodes.Count - 1;
412 entitiesInRange.RemoveAt(closestIndex);
413
414 float newRange = currentRange - (targetStructure.Rect.Height / 2) * (1.0f / RangeMultiplierInWalls);
415
416 //continue the discharge to the top edge of the structure and extend from there
417 int topNodeIndex = nodeIndex;
418 Vector2 topPos = targetStructure.SectionPosition(0, world: true);
419 topPos.X += targetStructure.BodyWidth / 2 * xDir;
420 AddNodesBetweenPoints(targetPos, topPos, 0.05f, ref topNodeIndex);
421 nodes.Add(new Node(topPos, topNodeIndex));
422 FindNodes(entitiesInRange, topPos, nodes.Count - 1, newRange);
423
424 //continue the discharge to the bottom edge of the structure and extend from there
425 int bottomNodeIndex = nodeIndex;
426 Vector2 bottomBos = targetStructure.SectionPosition(targetStructure.SectionCount - 1, world: true);
427 bottomBos.X += targetStructure.BodyWidth / 2 * xDir;
428 AddNodesBetweenPoints(targetPos, bottomBos, 0.05f, ref bottomNodeIndex);
429 nodes.Add(new Node(bottomBos, bottomNodeIndex));
430 FindNodes(entitiesInRange, bottomBos, nodes.Count - 1, newRange);
431 }
432
433 //check if any character is close to this structure
434 for (int j = 0; j < entitiesInRange.Count; j++)
435 {
436 var otherEntity = entitiesInRange[j];
437 if (!(otherEntity is Character character)) continue;
438 if (OutdoorsOnly && character.Submarine != null) continue;
439
440 if (targetStructure.IsHorizontal)
441 {
442 if (otherEntity.WorldPosition.X < targetStructure.WorldRect.X) continue;
443 if (otherEntity.WorldPosition.X > targetStructure.WorldRect.Right) continue;
444 if (Math.Abs(otherEntity.WorldPosition.Y - targetStructure.WorldPosition.Y) > currentRange) continue;
445 }
446 else
447 {
448 if (otherEntity.WorldPosition.Y < targetStructure.WorldRect.Y - targetStructure.Rect.Height) continue;
449 if (otherEntity.WorldPosition.Y > targetStructure.WorldRect.Y) continue;
450 if (Math.Abs(otherEntity.WorldPosition.X - targetStructure.WorldPosition.X) > currentRange) continue;
451 }
452 float closestNodeDistSqr = float.MaxValue;
453 int closestNodeIndex = -1;
454 for (int i = 0; i < nodes.Count; i++)
455 {
456 float distSqr = Vector2.DistanceSquared(character.WorldPosition, nodes[i].WorldPosition);
457 if (distSqr < closestNodeDistSqr)
458 {
459 closestNodeDistSqr = distSqr;
460 closestNodeIndex = i;
461 }
462 }
463 if (closestNodeIndex > -1)
464 {
465 FindNodes(entitiesInRange, nodes[closestNodeIndex].WorldPosition, closestNodeIndex, currentRange - (float)Math.Sqrt(closestNodeDistSqr));
466 }
467 }
468 }
469 else if (entitiesInRange[closestIndex] is Character character)
470 {
471 Vector2 targetPos = character.WorldPosition;
472 //create nodes from the current position to the closest point on the character
473 AddNodesBetweenPoints(currPos, targetPos, 0.25f, ref parentNodeIndex);
474 nodes.Add(new Node(targetPos, parentNodeIndex));
475 entitiesInRange.RemoveAt(closestIndex);
476 charactersInRange.Add((character, nodes[parentNodeIndex]));
477 FindNodes(entitiesInRange, targetPos, nodes.Count - 1, currentRange);
478 }
479 }

481 private void AddNodesBetweenPoints(Vector2 currPos, Vector2 targetPos, float variance, ref int parentNodeIndex)
482 {
483 Vector2 diff = targetPos - currPos;
484 float dist = diff.Length();
485 Vector2 normal = new Vector2(-diff.Y, diff.X) / dist;
486 for (float x = MaxNodeDistance; x < dist - MaxNodeDistance; x += MaxNodeDistance * Rand.Range(0.5f, 1.5f))
487 {
488 //0 at the edges, 1 at the center
489 float normalOffset = (0.5f - Math.Abs(x / dist - 0.5f)) * 2.0f;
490 normalOffset *= variance * dist * Rand.Range(-1.0f, 1.0f);
491
492 nodes.Add(new Node(currPos + (diff / dist) * x + normal * normalOffset, parentNodeIndex));
493 parentNodeIndex = nodes.Count - 1;
494 }
