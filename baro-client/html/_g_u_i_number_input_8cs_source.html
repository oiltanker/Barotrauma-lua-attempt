334 private void ReduceValue()
335 {
336 if (inputType == NumberType.Int)
337 {
338 IntValue -= valueStep > 0 ? (int)valueStep : 1;
339 ClampIntValue();
340 }
341 else if (maxValueFloat.HasValue && minValueFloat.HasValue)
342 {
343 FloatValue -= valueStep > 0 ? valueStep : Round();
344 ClampFloatValue();
345 }
346 }
347
348 private void IncreaseValue()
349 {
350 if (inputType == NumberType.Int)
351 {
352 IntValue += valueStep > 0 ? (int)valueStep : 1;
353 ClampIntValue();
354 }
355 else if (inputType == NumberType.Float)
356 {
357 FloatValue += valueStep > 0 ? valueStep : Round();
358 ClampFloatValue();
359 }
360 }
361
367 private float Round()
368 {
369 if (!maxValueFloat.HasValue || !minValueFloat.HasValue) { return 0; }
370 float onePercent = MathHelper.Lerp(minValueFloat.Value, maxValueFloat.Value, 0.01f);
371 float diff = maxValueFloat.Value - minValueFloat.Value;
372 int decimals = (int)MathHelper.Lerp(3, 0, MathUtils.InverseLerp(10, 1000, diff));
373 return MathHelper.Clamp((float)Math.Round(onePercent, decimals), 0.1f, 1000);
374 }

376 private bool TextChanged(GUITextBox textBox, string text)
377 {
378 switch (InputType)
379 {
380 case NumberType.Int:
381 if (string.IsNullOrWhiteSpace(text) || text == "-")
382 {
383 intValue = 0;
384 }
385 else if (int.TryParse(text, out int newIntValue))
386 {
387 intValue = newIntValue;
388 }
389 break;
390 case NumberType.Float:
391 if (string.IsNullOrWhiteSpace(text) || text == "-")
392 {
393 floatValue = 0;
394 }
395 else if (float.TryParse(text, NumberStyles.Any, CultureInfo.InvariantCulture, out float newFloatValue))
396 {
397 floatValue = newFloatValue;
398 }
399 break;
400 }
401 OnValueChanged?.Invoke(this);
402 return true;
403 }

405 private void ClampFloatValue()
406 {
407 if (MinValueFloat != null)
408 {
409 floatValue = Math.Max(floatValue, MinValueFloat.Value);
410 MinusButton.Enabled = floatValue > MinValueFloat;
411 }
412 if (MaxValueFloat != null)
413 {
414 floatValue = Math.Min(floatValue, MaxValueFloat.Value);
415 PlusButton.Enabled = floatValue < MaxValueFloat;
416 }
417 }
418
419 private void ClampIntValue()
420 {
421 if (MinValueInt != null && intValue < MinValueInt.Value)
422 {
423 intValue = Math.Max(intValue, MinValueInt.Value);
424 UpdateText();
425 }
426 if (MaxValueInt != null && intValue > MaxValueInt.Value)
427 {
428 intValue = Math.Min(intValue, MaxValueInt.Value);
429 UpdateText();
430 }
431 PlusButton.Enabled = intValue < MaxValueInt;
432 MinusButton.Enabled = intValue > MinValueInt;
433 }

435 private void UpdateText()
436 {
437 switch (InputType)
438 {
439 case NumberType.Float:
440 TextBox.Text = FloatValue.Format(decimalsToDisplay);
441 break;
442 case NumberType.Int:
443 TextBox.Text = IntValue.ToString();
444 break;
445 }
