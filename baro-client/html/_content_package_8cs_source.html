61 protected ContentPackage(XDocument doc, string path)
62 {
63 Path = path.CleanUpPathCrossPlatform();
64 XElement rootElement = doc.Root ?? throw new NullReferenceException("XML document is invalid: root element is null.");
65
66 Name = rootElement.GetAttributeString("name", "").Trim();
67 AltNames = rootElement.GetAttributeStringArray("altnames", Array.Empty<string>())
68 .Select(n => n.Trim()).ToImmutableArray();
69 AssertCondition(!string.IsNullOrEmpty(Name), "Name is null or empty");
70 SteamWorkshopId = rootElement.GetAttributeUInt64("steamworkshopid", 0);
71
72 GameVersion = rootElement.GetAttributeVersion("gameversion", GameMain.Version);
73 ModVersion = rootElement.GetAttributeString("modversion", DefaultModVersion);
74 if (rootElement.Attribute("installtime") != null)
75 {
76 InstallTime = ToolBox.Epoch.ToDateTime(rootElement.GetAttributeUInt("installtime", 0));
77 }
78 else
79 {
80 InstallTime = null;
81 }
82
83 var fileResults = rootElement.Elements()
84 .Select(e => ContentFile.CreateFromXElement(this, e))
85 .ToArray();
86
87 Files = fileResults
88 .OfType<Success<ContentFile, string>>()
89 .Select(f => f.Value)
90 .ToImmutableArray();
91
92 Errors = fileResults
93 .OfType<Failure<ContentFile, string>>()
94 .Select(f => (f.Error, f.StackTrace))
95 .ToImmutableArray();
96
97 HasMultiplayerSyncedContent = Files.Any(f => !f.NotSyncedInMultiplayer);
98
99 Hash = CalculateHash();
100 var expectedHash = rootElement.GetAttributeString("expectedhash", "");
101 if (HashMismatches(expectedHash))
102 {
103 DebugConsole.ThrowError($"Hash calculation for content package \"{Name}\" didn't match expected hash ({Hash.StringRepresentation} != {expectedHash})");
104 }
105 }

126 {

148 public Md5Hash CalculateHash(bool logging = false)
149 {
150 using IncrementalHash incrementalHash = IncrementalHash.CreateHash(HashAlgorithmName.MD5);
151
152 if (logging)
153 {
154 DebugConsole.NewMessage("****************************** Calculating content package hash " + Name);
155 }
156
157 foreach (ContentFile file in Files)
158 {
159 try
160 {
161 var hash = file.Hash;
162 if (logging)
163 {
164 DebugConsole.NewMessage(" " + file.Path + ": " + hash.StringRepresentation);
165 }
166 incrementalHash.AppendData(hash.ByteRepresentation);
167 }
168
169 catch (Exception e)
170 {
171 DebugConsole.ThrowError($"Error while calculating the MD5 hash of the content package \"{Name}\" (file path: {Path}). The content package may be corrupted. You may want to delete or reinstall the package.", e);
172 break;
173 }
174 }
175
176 var md5Hash = Md5Hash.BytesAsHash(incrementalHash.GetHashAndReset());
177 if (logging)
178 {
179 DebugConsole.NewMessage("****************************** Package hash: " + md5Hash.StringRepresentation);
180 }
181
182 return md5Hash;
183 }

222#if DEBUG

230 IEnumerable<ContentPackageManager.LoadProgress> loadFiles(ContentFile[] filesToLoad, int indexOffset)
231 {
232 for (int i = 0; i < filesToLoad.Length; i++)
233 {
234 Exception? exception = null;
235 try
236 {
237 //do not allow exceptions thrown here to crash the game
238 filesToLoad[i].LoadFile();
239 }
240 catch (Exception e)
241 {
242 exception = e;
243 }
244 if (exception != null)
245 {
246 yield return ContentPackageManager.LoadProgress.Failure(exception);
247 break;
248 }
249 yield return new ContentPackageManager.LoadProgress((i + indexOffset) / (float)Files.Length);
250 }
251 }
252
253 //Load the UI and text files first. This is to allow the game
254 //to render the text in the loading screen as soon as possible.
255 var priorityFiles = getFilesToLoad(f => f is UIStyleFile || f is TextFile);
256
257 var remainder = getFilesToLoad(f => !priorityFiles.Contains(f));
258
259 var loadEnumerable =
260 loadFiles(priorityFiles, 0)
261 .Concat(loadFiles(remainder, priorityFiles.Length));
262
263 foreach (var p in loadEnumerable)
264 {
265 if (p.Exception != null)
266 {
267 HandleLoadException(p.Exception);
268 yield return p;
269 break;
270 }
271 yield return p;
272 }
273 }
