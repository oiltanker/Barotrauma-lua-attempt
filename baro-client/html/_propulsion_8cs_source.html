41 public override bool Use(float deltaTime, Character character = null)
42 {
43 if (character == null || character.Removed) return false;
44 if (!character.IsKeyDown(InputType.Aim) || character.Stun > 0.0f) { return false; }
45
46 IsActive = true;
47 useState = 0.1f;
48
49 if (character.AnimController.InWater)
50 {
51 if (UsableIn == UseEnvironment.Air) { return true; }
52 }
53 else
54 {
55 if (UsableIn == UseEnvironment.Water) { return true; }
56 }
57
58 Vector2 dir = Vector2.Normalize(character.CursorPosition - character.Position);
59 //move upwards if the cursor is at the position of the character
60 if (!MathUtils.IsValid(dir)) dir = Vector2.UnitY;
61
62 Vector2 propulsion = dir * Force * character.PropulsionSpeedMultiplier;
63
64 if (character.AnimController.InWater && Force > 0.0f) { character.AnimController.TargetMovement = dir; }
65
66 foreach (Limb limb in character.AnimController.Limbs)
67 {
68 if (limb.WearingItems.Find(w => w.WearableComponent.Item == item) == null) { continue; }
69 limb.body.ApplyForce(propulsion);
70 }
71
72 character.AnimController.Collider.ApplyForce(propulsion);
73
74 if (character.Inventory.IsInLimbSlot(item, InvSlotType.RightHand))
75 {
76 character.AnimController.GetLimb(LimbType.RightHand)?.body.ApplyForce(propulsion);
77 }
78 if (character.Inventory.IsInLimbSlot(item, InvSlotType.LeftHand))
79 {
80 character.AnimController.GetLimb(LimbType.LeftHand)?.body.ApplyForce(propulsion);
81 }
82
83#if CLIENT
84 if (!string.IsNullOrWhiteSpace(particles))
85 {
86 GameMain.ParticleManager.CreateParticle(particles, item.WorldPosition,
87 item.body.Rotation + ((item.body.Dir > 0.0f) ? 0.0f : MathHelper.Pi), 0.0f, item.CurrentHull);
88 }
89#endif
90
91 return true;
92 }
