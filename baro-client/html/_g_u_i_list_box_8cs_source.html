436 private void CalculateChildrenOffsets(Action<int, Point> callback)
437 {
438 Vector2 topOffset = CalculateTopOffset();
439 int x = (int)topOffset.X;
440 int y = (int)topOffset.Y;
441
442 for (int i = 0; i < Content.CountChildren; i++)
443 {
444 GUIComponent child = Content.GetChild(i);
445 if (child == null || !child.Visible) { continue; }
446 if (RectTransform != null)
447 {
448 callback(i, new Point(x, y));
449 }
450
451 if (useGridLayout)
452 {
453 void advanceGridLayout(
454 ref int primaryCoord,
455 ref int secondaryCoord,
456 int primaryChildDimension,
457 int secondaryChildDimension,
458 int primaryParentDimension)
459 {
460 if (primaryCoord + primaryChildDimension + Spacing > primaryParentDimension)
461 {
462 primaryCoord = 0;
463 secondaryCoord += secondaryChildDimension + Spacing;
464 callback(i, new Point(x, y));
465 }
466 primaryCoord += primaryChildDimension + Spacing;
467 }
468
469 if (ScrollBar.IsHorizontal)
470 {
471 advanceGridLayout(
472 primaryCoord: ref y,
473 secondaryCoord: ref x,
474 primaryChildDimension: child.Rect.Height,
475 secondaryChildDimension: child.Rect.Width,
476 primaryParentDimension: Content.Rect.Height);
477 }
478 else
479 {
480 advanceGridLayout(
481 primaryCoord: ref x,
482 secondaryCoord: ref y,
483 primaryChildDimension: child.Rect.Width,
484 secondaryChildDimension: child.Rect.Height,
485 primaryParentDimension: Content.Rect.Width);
486 }
487 }
488 else
489 {
490 if (ScrollBar.IsHorizontal)
491 {
492 x += child.Rect.Width + Spacing;
493 }
494 else
495 {
496 y += child.Rect.Height + Spacing;
497 }
498 }
499 }

548 public void ScrollToEnd(float duration)
549 {
550 CoroutineManager.StartCoroutine(ScrollCoroutine());
551
552 IEnumerable<CoroutineStatus> ScrollCoroutine()
553 {
554 if (BarSize >= 1.0f)
555 {
556 yield return CoroutineStatus.Success;
557 }
558 float t = 0.0f;
559 float startScroll = BarScroll * BarSize;
560 float distanceToTravel = ScrollBar.MaxValue - startScroll;
561 float progress = startScroll;
562 float speed = distanceToTravel / duration;
563
564 while (t < duration && !MathUtils.NearlyEqual(ScrollBar.MaxValue, progress))
565 {
566 t += CoroutineManager.DeltaTime;
567 progress += speed * CoroutineManager.DeltaTime;
568 BarScroll = progress;
569 yield return CoroutineStatus.Running;
570 }
571
572 yield return CoroutineStatus.Success;
573 }
574 }

581 private bool UpdateDragging()
582 {
583 if (CurrentDragMode == DragMode.NoDragging || !isDraggingElement) { return false; }
584 if (!PlayerInput.PrimaryMouseButtonHeld())
585 {
586 var draggedElem = draggedElement;
587 OnRearranged?.Invoke(this, draggedElem.UserData);
588 DraggedElement = null;
589 RepositionChildren();
590 if (AllSelected.Contains(draggedElem)) { return true; }
591 }
592 else
593 {
594 Vector2 topOffset = CalculateTopOffset();
595 var mousePos = PlayerInput.MousePosition.ToPoint();
596 draggedElement.RectTransform.AbsoluteOffset = mousePos - Content.Rect.Location - dragMousePosRelativeToTopLeftCorner;
597 if (CurrentDragMode != DragMode.DragOutsideBox)
598 {
599 var offset = draggedElement.RectTransform.AbsoluteOffset;
600 draggedElement.RectTransform.AbsoluteOffset =
601 isHorizontal ? new Point(offset.X, 0) : new Point(0, offset.Y);
602 }
603
604 int index = Content.RectTransform.GetChildIndex(draggedElement.RectTransform);
605 int newIndex = index;
606
607 Point draggedOffsetWhenReleased = Point.Zero;
608 CalculateChildrenOffsets((i, offset) =>
609 {
610 if (index != i) { return; }
611 draggedOffsetWhenReleased = offset;
612 });
613 Rectangle draggedRectWhenReleased = new Rectangle(Content.Rect.Location + draggedOffsetWhenReleased, draggedElement.Rect.Size);
614
615 void shiftIndices(
616 float mousePos,
617 ref int draggedRectWhenReleasedLocation,
618 int draggedRectWhenReleasedSize)
619 {
620 while (mousePos > (draggedRectWhenReleasedLocation + draggedRectWhenReleasedSize) && newIndex < Content.CountChildren-1)
621 {
622 newIndex++;
623 draggedRectWhenReleasedLocation += draggedRectWhenReleasedSize;
624 }
625 while (mousePos < draggedRectWhenReleasedLocation && newIndex > 0)
626 {
627 newIndex--;
628 draggedRectWhenReleasedLocation -= draggedRectWhenReleasedSize;
629 }
630
631 if (newIndex != index && AllSelected.Count > 1)
632 {
633 this.selected.Sort((a, b) => Content.GetChildIndex(a) - Content.GetChildIndex(b));
634 int draggedPos = AllSelected.IndexOf(draggedElement);
635 if (newIndex < draggedPos)
636 {
637 newIndex = draggedPos;
638 }
639 if (newIndex >= Content.CountChildren - (AllSelected.Count - draggedPos))
640 {
641 int max = Content.CountChildren - (AllSelected.Count - draggedPos);
642 newIndex = max;
643 }
644 }
645 }
646
647 if (isHorizontal)
648 {
649 shiftIndices(
650 mousePos.X,
651 ref draggedRectWhenReleased.X,
652 draggedRectWhenReleased.Width);
653 }
654 else
655 {
656 shiftIndices(
657 mousePos.Y,
658 ref draggedRectWhenReleased.Y,
659 draggedRectWhenReleased.Height);
660 }
661
662 if (newIndex != index)
663 {
664 if (AllSelected.Count > 1)
665 {
666 this.selected.Sort((a, b) => Content.GetChildIndex(a) - Content.GetChildIndex(b));
667 int indexOfDraggedElem = AllSelected.IndexOf(draggedElement);
668 IEnumerable<GUIComponent> allSelected = AllSelected;
669 if (newIndex > index) { allSelected = allSelected.Reverse(); }
670 foreach (var elem in allSelected)
671 {
672 elem.RectTransform.RepositionChildInHierarchy(newIndex + AllSelected.IndexOf(elem) - indexOfDraggedElem);
673 }
674 }
675 else
676 {
677 draggedElement.RectTransform.RepositionChildInHierarchy(newIndex);
678 }
679 HasDraggedElementIndexChanged = true;
680 }
681
682 return true;
683 }
684
685 return false;
686 }

688 private void UpdateChildrenRect()
689 {
690 if (UpdateDragging()) { return; }
691
692 if (SelectTop)
693 {
694 foreach (GUIComponent child in Content.Children)
695 {
696 child.CanBeFocused = !selected.Contains(child);
697 if (!child.CanBeFocused)
698 {
699 child.State = ComponentState.None;
700 }
701 }
702 }
703
704 if (SelectTop && Content.Children.Any() && scrollToElement == null)
705 {
706 GUIComponent component = Content.Children.FirstOrDefault(c => (c.Rect.Y - Content.Rect.Y) / (float)c.Rect.Height > -0.1f);
707
708 if (component != null && !selected.Contains(component))
709 {
710 int index = Content.Children.ToList().IndexOf(component);
711 if (index >= 0)
712 {
713 Select(index, false, false, takeKeyBoardFocus: true);
714 }
715 }
716 }
717
718 for (int i = 0; i < Content.CountChildren; i++)
719 {
720 var child = Content.RectTransform.GetChild(i)?.GUIComponent;
721 if (!(child is { Visible: true })) { continue; }
722
723 // selecting
724 if (Enabled && (CanBeFocused || CanInteractWhenUnfocusable) && child.CanBeFocused && child.Rect.Contains(PlayerInput.MousePosition) && GUI.IsMouseOn(child))
725 {
726 child.State = ComponentState.Hover;
727
728 var mouseDown = useMouseDownToSelect ? PlayerInput.PrimaryMouseButtonDown() : PlayerInput.PrimaryMouseButtonClicked();
729
730 if (mouseDown)
731 {
732 if (SelectTop)
733 {
734 ScrollToElement(child);
735 }
736 Select(i, autoScroll: false, takeKeyBoardFocus: true);
737 }
738
739 if (CurrentDragMode != DragMode.NoDragging
740 && (CurrentSelectMode != SelectMode.RequireShiftToSelectMultiple || (!PlayerInput.IsShiftDown() && !PlayerInput.IsCtrlDown()))
741 && PlayerInput.PrimaryMouseButtonDown() && GUI.MouseOn == child)
742 {
743 StartDraggingElement(child);
744 }
745 }
746 else if (selected.Contains(child))
747 {
748 child.State = ComponentState.Selected;
749
750 if (CheckSelected != null)
751 {
752 if (CheckSelected() != child.UserData) selected.Remove(child);
753 }
754 }
755 else
756 {
757 child.State = !child.ExternalHighlight ? ComponentState.None : ComponentState.Hover;
758 }
759 }

762 public override void AddToGUIUpdateList(bool ignoreChildren = false, int order = 0)
763 {
764 if (!Visible) { return; }
765
766 if (!ignoreChildren)
767 {
768 foreach (GUIComponent child in Children)
769 {
770 if (child == Content || child == ScrollBar || child == ContentBackground) { continue; }
771 child.AddToGUIUpdateList(ignoreChildren, order);
772 }
773 }
774
775 foreach (GUIComponent child in Content.Children)
776 {
777 if (!childVisible.ContainsKey(child)) { childVisible[child] = child.Visible; }
778 if (childVisible[child] != child.Visible)
779 {
780 childVisible[child] = child.Visible;
781 childrenNeedsRecalculation = true;
782 scrollBarNeedsRecalculation = true;
783 break;
784 }
785 }
786
787 if (childrenNeedsRecalculation)
788 {
789 RecalculateChildren();
790 childVisible.Clear();
791 }
792
793 UpdateOrder = order;
794 GUI.AddToUpdateList(this);
795
796 if (ignoreChildren)
797 {
798 OnAddedToGUIUpdateList?.Invoke(this);
799 return;
800 }
801
802 int lastVisible = 0;
803 for (int i = 0; i < Content.CountChildren; i++)
804 {
805 var child = Content.GetChild(i);
806 if (!child.Visible) continue;
807 if (!IsChildInsideFrame(child))
808 {
809 if (lastVisible > 0) break;
810 continue;
811 }
812 lastVisible = i;
813 child.AddToGUIUpdateList(false, order);
814 }
815 if (ScrollBar.Enabled)
816 {
817 ScrollBar.AddToGUIUpdateList(false, order);
818 }
819 OnAddedToGUIUpdateList?.Invoke(this);
820 }

852 protected override void Update(float deltaTime)
853 {
854 if (!Visible) { return; }
855
856 UpdateChildrenRect();
857 RepositionChildren();
858
859 if (scrollBarNeedsRecalculation)
860 {
861 UpdateScrollBarSize();
862 }
863
864 if (FadeElements)
865 {
866 foreach (var (component, _) in childVisible)
867 {
868 float lerp = 0;
869 float y = component.Rect.Y;
870 float contentY = Content.Rect.Y;
871 float height = component.Rect.Height;
872 if (y < Content.Rect.Y)
873 {
874 float distance = (contentY - y) / height;
875 lerp = distance;
876 }
877
878 float centerY = Content.Rect.Y + Content.Rect.Height / 2.0f;
879 if (y > centerY)
880 {
881 float distance = (y - centerY) / (centerY - height);
882 lerp = distance;
883 }
884
885 component.Color = component.HoverColor = ToolBox.GradientLerp(lerp, component.DefaultColor, Color.Transparent);
886 component.DisabledColor = ToolBox.GradientLerp(lerp, component.Style.DisabledColor, Color.Transparent);
887 component.HoverColor = ToolBox.GradientLerp(lerp, component.Style.HoverColor, Color.Transparent);
888
889 foreach (var child in component.GetAllChildren())
890 {
891 Color gradient = ToolBox.GradientLerp(lerp, child.DefaultColor, Color.Transparent);
892 child.Color = child.HoverColor = gradient;
893 if (child is GUITextBlock block)
894 {
895 block.TextColor = block.HoverTextColor = gradient;
896 }
897 }
898 }
899 }
900
901 if (scrollToElement != null)
902 {
903 if (!scrollToElement.Visible || !Content.Children.Contains(scrollToElement))
904 {
905 scrollToElement = null;
906 }
907 else
908 {

1005 public void Select(int childIndex, bool force = false, bool autoScroll = true, bool takeKeyBoardFocus = false)
1006 {
1007 if (childIndex >= Content.CountChildren || childIndex < 0) { return; }
1008
1009 GUIComponent child = Content.GetChild(childIndex);
1010 if (child is null) { return; }
1011
1012 bool wasSelected = true;
1013 if (OnSelected != null)
1014 {
1015 // TODO: The callback is called twice, fix this!
1016 wasSelected = force || OnSelected(child, child.UserData);
1017 }
1018
1019 if (!wasSelected) { return; }
1020
1021 if (CurrentSelectMode == SelectMode.SelectMultiple ||
1022 (CurrentSelectMode == SelectMode.RequireShiftToSelectMultiple && PlayerInput.IsCtrlDown()))
1023 {
1024 if (selected.Contains(child))
1025 {
1026 selected.Remove(child);
1027 }
1028 else
1029 {
1030 selected.Add(child);
1031 }
1032 }
1033 else if (CurrentSelectMode == SelectMode.RequireShiftToSelectMultiple && PlayerInput.IsShiftDown())
1034 {
1035 var first = SelectedComponent ?? child;
1036 var last = child;
1037 int firstIndex = Content.GetChildIndex(first);
1038 int lastIndex = Content.GetChildIndex(last);
1039 int sgn = Math.Sign(lastIndex - firstIndex);
1040 selected.Clear(); selected.Add(first);
1041 for (int i = firstIndex + sgn; i != lastIndex; i += sgn)
1042 {
1043 if (Content.GetChild(i) is { Visible: true } interChild)
1044 {
1045 selected.Add(interChild);
1046 }
1047 }
1048 if (first != last) { selected.Add(last); }
1049 }
1050 else
1051 {
1052 selected.Clear();
1053 selected.Add(child);
1054 }
1055
1056 // Ensure that the selected element is visible. This may not be the case, if the selection is run from code. (e.g. if we have two list boxes that are synced)
1057 // TODO: This method only works when moving one item up/down (e.g. when using the up and down arrows)
1058 if (autoScroll)
1059 {
1060 if (ScrollBar.IsHorizontal)
1061 {
1062 if (child.Rect.X < MouseRect.X)
1063 {
1064 //child outside the left edge of the frame -> move left

1066 }
1067 else if (child.Rect.Right > MouseRect.Right)
1068 {
1069 //child outside the right edge of the frame -> move right
1070 ScrollBar.BarScroll += (float)(child.Rect.Right - MouseRect.Right) / (totalSize - Content.Rect.Width);
1071 }
1072 }
1073 else
1074 {
1075 if (child.Rect.Y < MouseRect.Y)
1076 {
1077 //child above the top of the frame -> move up
1078 ScrollBar.BarScroll -= (float)(MouseRect.Y - child.Rect.Y) / (totalSize - Content.Rect.Height);
1079 }
1080 else if (child.Rect.Bottom > MouseRect.Bottom)
1081 {
1082 //child below the bottom of the frame -> move down
1083 ScrollBar.BarScroll += (float)(child.Rect.Bottom - MouseRect.Bottom) / (totalSize - Content.Rect.Height);
1084 }
1085 }
1086 }

1089 if (takeKeyBoardFocus && CanTakeKeyBoardFocus && RectTransform.GetAllChildren().None(rt => rt.GUIComponent == GUI.KeyboardDispatcher.Subscriber))
