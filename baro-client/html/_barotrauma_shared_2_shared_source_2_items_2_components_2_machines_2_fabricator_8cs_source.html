250 public override void Update(float deltaTime, Camera cam)
251 {
252 if (refreshIngredientsTimer <= 0.0f)
253 {
254 RefreshAvailableIngredients();
255 refreshIngredientsTimer = RefreshIngredientsInterval;
256 }
257 refreshIngredientsTimer -= deltaTime;
258
259 bool isClient = GameMain.NetworkMember?.IsClient ?? false;
260
261 if (!isClient)
262 {
263 if (fabricatedItem == null || !CanBeFabricated(fabricatedItem, availableIngredients, user))
264 {
265 CancelFabricating();
266 return;
267 }
268 }
269
270 progressState = fabricatedItem == null ? 0.0f : (requiredTime - timeUntilReady) / requiredTime;
271
272 if (isClient)
273 {
274 hasPower = State != FabricatorState.Paused;
275 if (!hasPower)
276 {
277 return;
278 }
279 }
280 else
281 {
282 hasPower = Voltage >= MinVoltage;
283
284 if (!hasPower)
285 {
286 State = FabricatorState.Paused;
287 return;
288 }
289 State = FabricatorState.Active;
290 }
291
292 float tinkeringStrength = 0f;
293 var repairable = item.GetComponent<Repairable>();
294 if (repairable != null)
295 {
296 repairable.LastActiveTime = (float)Timing.TotalTime + 10.0f;
297 if (repairable.IsTinkering)
298 {
299 tinkeringStrength = repairable.TinkeringStrength;
300 }
301 }
302
303 ApplyStatusEffects(ActionType.OnActive, deltaTime, null);
304
305
306 float fabricationSpeedIncrease = 1f + tinkeringStrength * TinkeringSpeedIncrease;
307
308 timeUntilReady -= deltaTime * fabricationSpeedIncrease * Math.Min(powerConsumption <= 0 ? 1 : Voltage, 1.0f);
309
310 UpdateRequiredTimeProjSpecific();
311
312 if (timeUntilReady <= 0.0f)
313 {
314 Fabricate();
315 }
316 }

327 private void Fabricate()
328 {
329 RefreshAvailableIngredients();
330 if (fabricatedItem == null || !CanBeFabricated(fabricatedItem, availableIngredients, user))
331 {
332 CancelFabricating();
333 return;
334 }
335
336 if (fabricatedItem.RequiredMoney > 0)
337 {
338 if (user == null) { return; }
339 if (GameMain.GameSession?.GameMode is MultiPlayerCampaign mpCampaign)
340 {
341#if CLIENT
342 mpCampaign.TryPurchase(null, fabricatedItem.RequiredMoney);
343#elif SERVER
344 if (GetUsingClient() is { } client)
345 {
346 mpCampaign.TryPurchase(client, fabricatedItem.RequiredMoney);
347 }
348 else
349 {
350 user.Wallet.Deduct(fabricatedItem.RequiredMoney);
351 }
352#endif
353 }
354 else if (GameMain.GameSession?.GameMode is CampaignMode campaign)
355 {
356 campaign.Bank.Deduct(fabricatedItem.RequiredMoney);
357 }
358 }
359
360 bool ingredientsStolen = false;
361 bool ingredientsAllowStealing = true;
362
363 if (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer)
364 {
365 fabricatedItem.RequiredItems.ForEach(requiredItem =>
366 {
367 for (int usedPrefabsAmount = 0; usedPrefabsAmount < requiredItem.Amount; usedPrefabsAmount++)
368 {
369 foreach (ItemPrefab requiredPrefab in requiredItem.ItemPrefabs)
370 {
371 if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
372
373 var availableItems = availableIngredients[requiredPrefab.Identifier];
374 var availableItem = availableItems.FirstOrDefault(potentialPrefab =>
375 {
376 return potentialPrefab.ConditionPercentage >= requiredItem.MinCondition * 100.0f &&
377 potentialPrefab.ConditionPercentage <= requiredItem.MaxCondition * 100.0f;
378 });
379
380 if (availableItem == null) { continue; }
381
382 ingredientsStolen |= availableItem.StolenDuringRound;
383 if (!availableItem.AllowStealing)
384 {
385 ingredientsAllowStealing = false;
386 }
387
388 //Leave it behind with reduced condition if it has enough to stay above 0
389 if (requiredItem.UseCondition && availableItem.ConditionPercentage - requiredItem.MinCondition * 100 > 0.0f)
390 {
391 availableItem.Condition -= availableItem.Prefab.Health * requiredItem.MinCondition;
392 continue;
393 }
394 if (availableItem.OwnInventory != null)
395 {
396 foreach (Item containedItem in availableItem.OwnInventory.AllItemsMod)
397 {
398 if (availableItem.GetComponent<ItemContainer>()?.RemoveContainedItemsOnDeconstruct ?? false)
399 {
400 Entity.Spawner.AddItemToRemoveQueue(containedItem);
401 }
402 else
403 {
404 containedItem.Drop(dropper: null);
405 }
406 }
407 }
408
409 availableItems.Remove(availableItem);
410 Entity.Spawner.AddItemToRemoveQueue(availableItem);
411 inputContainer.Inventory.RemoveItem(availableItem);
412 break;
413 }
414 }
415 });
416
417 int amountFittingContainer = outputContainer.Inventory.HowManyCanBePut(fabricatedItem.TargetItem, fabricatedItem.OutCondition * fabricatedItem.TargetItem.Health);
418
419 var fabricationitemAmount = new AbilityFabricationItemAmount(fabricatedItem.TargetItem, fabricatedItem.Amount);
420
421 int quality = 0;
422 if (fabricatedItem.Quality.HasValue)
423 {
424 quality = fabricatedItem.Quality.Value;
425 }
426 else if (user?.Info != null)
427 {
428 foreach (Character character in Character.GetFriendlyCrew(user))
429 {
430 character.CheckTalents(AbilityEffectType.OnAllyItemFabricatedAmount, fabricationitemAmount);
431 }
432 user.CheckTalents(AbilityEffectType.OnItemFabricatedAmount, fabricationitemAmount);
433 quality = GetFabricatedItemQuality(fabricatedItem, user);
434 }
435
436 int amount = (int)fabricationitemAmount.Value;
437 if (fabricationLimits.ContainsKey(fabricatedItem.RecipeHash))
438 {
439 if (amount > fabricationLimits[fabricatedItem.RecipeHash])
440 {
441 amount = fabricationLimits[fabricatedItem.RecipeHash];
442 fabricationLimits[fabricatedItem.RecipeHash] = 0;
443 }
444 else
445 {
446 fabricationLimits[fabricatedItem.RecipeHash] -= amount;
447 }
448 }
449
450 var tempUser = user;
451 for (int i = 0; i < amount; i++)
452 {
453 float outCondition = fabricatedItem.OutCondition;
454 GameAnalyticsManager.AddDesignEvent("ItemFabricated:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "none") + ":" + fabricatedItem.TargetItem.Identifier);
455 if (i < amountFittingContainer)
456 {
457 Entity.Spawner.AddItemToSpawnQueue(fabricatedItem.TargetItem, outputContainer.Inventory, fabricatedItem.TargetItem.Health * outCondition, quality,
458 onSpawned: (Item spawnedItem) =>
459 {
460 onItemSpawned(spawnedItem, tempUser);
461 spawnedItem.Quality = quality;
462 spawnedItem.StolenDuringRound = ingredientsStolen;
463 spawnedItem.AllowStealing = ingredientsAllowStealing;
464 //reset the condition in case the max condition is higher than the prefab's due to e.g. quality modifiers
465 spawnedItem.Condition = spawnedItem.MaxCondition * outCondition;
466 });
467 }
468 else
469 {
470 Entity.Spawner.AddItemToSpawnQueue(fabricatedItem.TargetItem, item.Position, item.Submarine, fabricatedItem.TargetItem.Health * outCondition, quality,
471 onSpawned: (Item spawnedItem) =>
472 {
473 onItemSpawned(spawnedItem, tempUser);
474 spawnedItem.Quality = quality;
475 spawnedItem.StolenDuringRound = ingredientsStolen;
476 spawnedItem.AllowStealing = ingredientsAllowStealing;
477 //reset the condition in case the max condition is higher than the prefab's due to e.g. quality modifiers
478 spawnedItem.Condition = spawnedItem.MaxCondition * outCondition;
479 });
480 }

543 private int GetFabricatedItemQuality(FabricationRecipe fabricatedItem, Character user)
544 {
545 if (user == null) { return 0; }
546 if (fabricatedItem.TargetItem.ConfigElement.GetChildElement("Quality") == null) { return 0; }
547 int quality = 0;
548 float floatQuality = 0.0f;
549 foreach (var tag in fabricatedItem.TargetItem.Tags)
550 {
551 floatQuality += user.Info.GetSavedStatValue(StatTypes.IncreaseFabricationQuality, tag);
552 }
553 if (!fabricatedItem.TargetItem.Tags.Contains(fabricatedItem.TargetItem.Identifier))
554 {
555 floatQuality += user.Info.GetSavedStatValue(StatTypes.IncreaseFabricationQuality, fabricatedItem.TargetItem.Identifier);
556 }
557 quality = (int)floatQuality;
558
559 const int MaxCraftingSkill = 100;
560
561 quality += fabricatedItem.RequiredSkills.All(s => user.GetSkillLevel(s.Identifier) >= MaxCraftingSkill) ? 1 : 0;
562 quality += FabricationDegreeOfSuccess(user, fabricatedItem.RequiredSkills) >= 0.5f ? 1 : 0;
563 return quality;
564 }

568 private bool CanBeFabricated(FabricationRecipe fabricableItem, IReadOnlyDictionary<Identifier, List<Item>> availableIngredients, Character character)
569 {
570 if (fabricableItem == null) { return false; }
571 if (fabricableItem.RequiresRecipe && (character == null || !character.HasRecipeForItem(fabricableItem.TargetItem.Identifier))) { return false; }
572
573 if (fabricableItem.RequiredMoney > 0)
574 {
575 switch (GameMain.GameSession?.GameMode)
576 {
577 case MultiPlayerCampaign mpCampaign:
578 {
579 if (!mpCampaign.CanAfford(fabricableItem.RequiredMoney, GetUsingClient())) { return false; }
580
581 break;
582 }
583 case CampaignMode campaign:
584 {
585 if (campaign.Bank.Balance < fabricableItem.RequiredMoney) { return false; }
586
587 break;
588 }
589 default:
590 return false;
591 }
592 }
593
594 if (fabricationLimits.TryGetValue(fabricableItem.RecipeHash, out int amount) && amount <= 0)
595 {
596 return false;
597 }
598
599 return fabricableItem.RequiredItems.All(requiredItem =>
600 {
601 int availablePrefabsAmount = 0;
602 foreach (ItemPrefab requiredPrefab in requiredItem.ItemPrefabs)
603 {
604 if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
605
606 var availablePrefabs = availableIngredients[requiredPrefab.Identifier];
607 foreach (Item availablePrefab in availablePrefabs)
608 {
609 if (availablePrefab.ConditionPercentage / 100.0f >= requiredItem.MinCondition &&
610 availablePrefab.ConditionPercentage / 100.0f <= requiredItem.MaxCondition)
611 {
612 availablePrefabsAmount++;
613 }
614
615 if (availablePrefabsAmount >= requiredItem.Amount)
616 {
617 return true;
618 }
619 }
620 }
621
622 return false;
623 });
624 }

626 private float GetRequiredTime(FabricationRecipe fabricableItem, Character user)
627 {
628 float degreeOfSuccess = FabricationDegreeOfSuccess(user, fabricableItem.RequiredSkills);
629
630 float t = degreeOfSuccess < 0.5f ? degreeOfSuccess * degreeOfSuccess : degreeOfSuccess * 2;
631
632 //fabricating takes 100 times longer if degree of success is close to 0
633 //characters with a higher skill than required can fabricate up to 100% faster
634 return fabricableItem.RequiredTime / FabricationSpeed / MathHelper.Clamp(t, 0.01f, 2.0f);
635 }
636
637 public float FabricationDegreeOfSuccess(Character character, ImmutableArray<Skill> skills)
638 {
639 if (skills.Length == 0) { return 1.0f; }
640 if (character == null) { return 0.0f; }
641
642 float skillSum = (from t in skills let characterLevel = character.GetSkillLevel(t.Identifier) select (characterLevel - (t.Level * SkillRequirementMultiplier))).Sum();
643 float average = skillSum / skills.Length;
644
645 return (average + 100.0f) / 2.0f / 100.0f;
646 }

653 private void RefreshAvailableIngredients()
654 {
655 Character user = this.user;
656#if CLIENT
657 user ??= Character.Controlled;
658#endif
659
660 List<Item> itemList = new List<Item>();
661 itemList.AddRange(inputContainer.Inventory.AllItems);
662 foreach (MapEntity linkedTo in item.linkedTo)
663 {
664 if (linkedTo is Item linkedItem)
665 {
666 var itemContainer = linkedItem.GetComponent<ItemContainer>();
667 if (itemContainer == null) { continue; }
668 if (user != null)
669 {
670 if (!itemContainer.HasRequiredItems(user, addMessage: false)) { continue; }
671 }
672
673 var deconstructor = linkedItem.GetComponent<Deconstructor>();
674 if (deconstructor != null)
675 {
676 itemContainer = deconstructor.OutputContainer;
677 }
678
679 itemList.AddRange(itemContainer.Inventory.AllItems);
680 }
681 }
682 for (int i = 0; i < itemList.Count; i++)
683 {
684 var container = itemList[i].GetComponent<ItemContainer>();
685 if (container != null)
686 {
687 itemList.AddRange(container.Inventory.AllItems);
688 }
689 }
690 if (user?.Inventory != null)
691 {
692 itemList.AddRange(user.Inventory.AllItems);
693 }
694 availableIngredients.Clear();
695 foreach (Item item in itemList)
696 {
697 var itemIdentifier = item.Prefab.Identifier;
698 if (!availableIngredients.ContainsKey(itemIdentifier))
699 {
700 availableIngredients[itemIdentifier] = new List<Item>(itemList.Count);
701 }
702 availableIngredients[itemIdentifier].Add(item);
703 }
704 }

710 private void MoveIngredientsToInputContainer(FabricationRecipe targetItem)
711 {
712 //required ingredients that are already present in the input container
713 List<Item> usedItems = new List<Item>();
714
715 targetItem.RequiredItems.ForEach(requiredItem => {
716 for (int i = 0; i < requiredItem.Amount; i++)
717 {
718 foreach (ItemPrefab requiredPrefab in requiredItem.ItemPrefabs)
719 {
720 if (!availableIngredients.ContainsKey(requiredPrefab.Identifier)) { continue; }
721
722 var availablePrefabs = availableIngredients[requiredPrefab.Identifier];
723 var availablePrefab = availablePrefabs.FirstOrDefault(potentialPrefab =>
724 {
725 return !usedItems.Contains(potentialPrefab) &&
726 potentialPrefab.ConditionPercentage >= requiredItem.MinCondition * 100.0f &&
727 potentialPrefab.ConditionPercentage <= requiredItem.MaxCondition * 100.0f;
728 });
729 if (availablePrefab == null) { continue; }
730
731 availablePrefabs.Remove(availablePrefab);
732
733 if (availablePrefab.ParentInventory == inputContainer.Inventory)
734 {
735 //already in input container, all good
736 usedItems.Add(availablePrefab);
737 }
738 else //in another inventory, we need to move the item
739 {
740 if (!inputContainer.Inventory.CanBePut(availablePrefab))
741 {
742 var unneededItem = inputContainer.Inventory.AllItems.FirstOrDefault(it => !usedItems.Contains(it));
743 unneededItem?.Drop(null);
744 }
745 inputContainer.Inventory.TryPutItem(availablePrefab, user: null);
746 }
747 break;
748 }
749 }
750 });
751 RefreshAvailableIngredients();
752 }
