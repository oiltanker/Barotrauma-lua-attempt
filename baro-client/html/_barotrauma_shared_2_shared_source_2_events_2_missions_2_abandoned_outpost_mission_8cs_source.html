116 private void InitItems(Submarine submarine)
117 {
118 if (!string.IsNullOrEmpty(itemTag))
119 {
120 var itemsToDestroy = Item.ItemList.FindAll(it => it.Submarine?.Info.Type != SubmarineType.Player && it.HasTag(itemTag));
121 if (!itemsToDestroy.Any())
122 {
123 DebugConsole.ThrowError($"Error in mission \"{Prefab.Identifier}\". Could not find an item with the tag \"{itemTag}\".");
124 }
125 else
126 {
127 items.AddRange(itemsToDestroy);
128 }
129 }
130
131 if (itemConfig != null && !IsClient)
132 {
133 foreach (XElement element in itemConfig.Elements())
134 {
135 string itemIdentifier = element.GetAttributeString("identifier", "");
136 if (!(MapEntityPrefab.Find(null, itemIdentifier) is ItemPrefab itemPrefab))
137 {
138 DebugConsole.ThrowError("Couldn't spawn item for outpost destroy mission: item prefab \"" + itemIdentifier + "\" not found");
139 continue;
140 }
141
142 Identifier[] moduleFlags = element.GetAttributeIdentifierArray("moduleflags", null);
143 Identifier[] spawnPointTags = element.GetAttributeIdentifierArray("spawnpointtags", null);
144 ISpatialEntity spawnPoint = SpawnAction.GetSpawnPos(
145 SpawnAction.SpawnLocationType.Outpost, SpawnType.Human | SpawnType.Enemy,
146 moduleFlags, spawnPointTags, element.GetAttributeBool("asfaraspossible", false));
147 if (spawnPoint == null)
148 {
149 spawnPoint = submarine.GetHulls(alsoFromConnectedSubs: false).GetRandomUnsynced();
150 }
151 Vector2 spawnPos = spawnPoint.WorldPosition;
152 if (spawnPoint is WayPoint wp && wp.CurrentHull != null && wp.CurrentHull.Rect.Width > 100)
153 {
154 spawnPos = new Vector2(
155 MathHelper.Clamp(wp.WorldPosition.X + Rand.Range(-200, 201), wp.CurrentHull.WorldRect.X + 50, wp.CurrentHull.WorldRect.Right - 50),
156 wp.CurrentHull.WorldRect.Y - wp.CurrentHull.Rect.Height + 16.0f);
157 }
158 var item = new Item(itemPrefab, spawnPos, null);
159 items.Add(item);
160#if SERVER
161 spawnedItems.Add(item);
162#endif
163 }
164 }
165 }

167 private void InitCharacters(Submarine submarine)
168 {
169 characters.Clear();
170 characterItems.Clear();
171
172 if (characterConfig != null)
173 {
174 foreach (XElement element in characterConfig.Elements())
175 {
176 if (GameMain.NetworkMember == null && element.GetAttributeBool("multiplayeronly", false)) { continue; }
177
178 int defaultCount = element.GetAttributeInt("count", -1);
179 if (defaultCount < 0)
180 {
181 defaultCount = element.GetAttributeInt("amount", 1);
182 }
183 int min = Math.Min(element.GetAttributeInt("min", defaultCount), 255);
184 int max = Math.Min(Math.Max(min, element.GetAttributeInt("max", defaultCount)), 255);
185 int count = Rand.Range(min, max + 1);
186
187 if (element.Attribute("identifier") != null && element.Attribute("from") != null)
188 {
189 HumanPrefab humanPrefab = GetHumanPrefabFromElement(element);
190 for (int i = 0; i < count; i++)
191 {
192 LoadHuman(humanPrefab, element, submarine);
193 }
194 }
195 else
196 {
197 Identifier speciesName = element.GetAttributeIdentifier("character", element.GetAttributeIdentifier("identifier", Identifier.Empty));
198 var characterPrefab = CharacterPrefab.FindBySpeciesName(speciesName);
199 if (characterPrefab == null)
200 {
201 DebugConsole.ThrowError("Couldn't spawn a character for abandoned outpost mission: character prefab \"" + speciesName + "\" not found");
202 continue;
203 }
204 for (int i = 0; i < count; i++)
205 {
206 LoadMonster(characterPrefab, element, submarine);
207 }
208 }
209 }
210 }

213 private void LoadHuman(HumanPrefab humanPrefab, XElement element, Submarine submarine)
214 {
215 Identifier[] moduleFlags = element.GetAttributeIdentifierArray("moduleflags", null);
216 Identifier[] spawnPointTags = element.GetAttributeIdentifierArray("spawnpointtags", null);
217 ISpatialEntity spawnPos = SpawnAction.GetSpawnPos(
218 SpawnAction.SpawnLocationType.Outpost, SpawnType.Human,
219 moduleFlags ?? humanPrefab.GetModuleFlags(),
220 spawnPointTags ?? humanPrefab.GetSpawnPointTags(),
221 element.GetAttributeBool("asfaraspossible", false));
222 if (spawnPos == null)
223 {
224 spawnPos = submarine.GetHulls(alsoFromConnectedSubs: false).GetRandomUnsynced();
225 }
226
227 bool requiresRescue = element.GetAttributeBool("requirerescue", false);
228
229 Character spawnedCharacter = CreateHuman(humanPrefab, characters, characterItems, submarine, requiresRescue ? CharacterTeamType.FriendlyNPC : CharacterTeamType.None, spawnPos, giveTags: true);
230
231 if (spawnPos is WayPoint wp)
232 {
233 spawnedCharacter.GiveIdCardTags(wp);
234 }
235
236 if (requiresRescue)
237 {
238 requireRescue.Add(spawnedCharacter);
239#if CLIENT
240 GameMain.GameSession.CrewManager.AddCharacterToCrewList(spawnedCharacter);
241#endif
242 }
243
244 if (element.GetAttributeBool("requirekill", false))
245 {
246 requireKill.Add(spawnedCharacter);
247 }

250 private void LoadMonster(CharacterPrefab monsterPrefab, XElement element, Submarine submarine)
251 {
252 Identifier[] moduleFlags = element.GetAttributeIdentifierArray("moduleflags", null);
253 Identifier[] spawnPointTags = element.GetAttributeIdentifierArray("spawnpointtags", null);
254 ISpatialEntity spawnPos = SpawnAction.GetSpawnPos(SpawnAction.SpawnLocationType.Outpost, SpawnType.Enemy, moduleFlags, spawnPointTags, element.GetAttributeBool("asfaraspossible", false));
255 if (spawnPos == null)
256 {
257 spawnPos = submarine.GetHulls(alsoFromConnectedSubs: false).GetRandomUnsynced();
258 }
259 Character spawnedCharacter = Character.Create(monsterPrefab.Identifier, spawnPos.WorldPosition, ToolBox.RandomSeed(8), createNetworkEvent: false);
260 characters.Add(spawnedCharacter);
261 if (element.GetAttributeBool("requirekill", false))
262 {
263 requireKill.Add(spawnedCharacter);
264 }
265 if (spawnedCharacter.Inventory != null)
266 {
267 characterItems.Add(spawnedCharacter, spawnedCharacter.Inventory.FindAllItems(recursive: true));
268 }
269 if (submarine != null && spawnedCharacter.AIController is EnemyAIController enemyAi)
270 {
271 enemyAi.UnattackableSubmarines.Add(submarine);
272 enemyAi.UnattackableSubmarines.Add(Submarine.MainSub);
273 foreach (Submarine sub in Submarine.MainSub.DockedTo)
274 {
275 enemyAi.UnattackableSubmarines.Add(sub);
276 }

281 protected override void UpdateMissionSpecific(float deltaTime)
282 {
283 if (State != HostagesKilledState)
284 {
285 if (requireRescue.Any(r => r.Removed || r.IsDead))
286 {
287 State = HostagesKilledState;
288 return;
289 }
290 }
291 else
292 {
293 endTimer += deltaTime;
294 if (endTimer > EndDelay)
295 {
296#if SERVER
297 if (!(GameMain.GameSession.GameMode is CampaignMode) && GameMain.Server != null)
298 {
299 GameMain.Server.EndGame();
300 }
301#endif
302 }
303 }
304
305 switch (state)
306 {
307 case 0:
308
309 if (items.All(it => it.Removed || it.Condition <= 0.0f) &&
310 requireKill.All(c => c.Removed || c.IsDead || (c.LockHands && c.Submarine == Submarine.MainSub)) &&
311 requireRescue.All(c => c.Submarine?.Info.Type == SubmarineType.Player))
312 {
313 State = 1;
314 }
315 break;
316#if SERVER
317 case 1:
318 if (!(GameMain.GameSession.GameMode is CampaignMode) && GameMain.Server != null)
319 {
320 if (!Submarine.MainSub.AtStartExit || (wasDocked && !Submarine.MainSub.DockedTo.Contains(Level.Loaded.StartOutpost)))
321 {
322 GameMain.Server.EndGame();
323 State = 2;
324 }
325 }
326 break;
327#endif
328 }
