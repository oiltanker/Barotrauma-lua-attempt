67 public override void Update(float deltaTime, Camera cam)
68 {
69 if (stateSwitchDelay > 0.0f)
70 {
71 stateSwitchDelay -= deltaTime;
72 }
73 else
74 {
75 bool prevState = isInWater;
76
77 isInWater = false;
78 if (item.InWater)
79 {
80 //item in water -> we definitely want to send the True output
81 isInWater = true;
82 }
83 else if (item.CurrentHull != null && item.CurrentHull.WaterPercentage > 0.0f && item.CurrentHull.WaterVolume > 1.0f)
84 {
85 //(center of the) item in not water -> check if the water surface is below the bottom of the item's rect
86 if (item.CurrentHull.Surface > item.Rect.Y - item.Rect.Height)
87 {
88 isInWater = true;
89 }
90 }
91
92 if (prevState != isInWater)
93 {
94 stateSwitchDelay = StateSwitchInterval;
95 }
96 }
97
98 string signalOut = isInWater ? Output : FalseOutput;
99 if (!string.IsNullOrEmpty(signalOut))
100 {
101 item.SendSignal(signalOut, "signal_out");
102 }
103
104 if (item.CurrentHull != null)
105 {
106 int waterPercentage = 0;
107 //ignore minuscule amounts of water
108 if (item.CurrentHull.WaterVolume > 1.0f)
109 {
110 waterPercentage = MathHelper.Clamp((int)Math.Ceiling(item.CurrentHull.WaterPercentage), 0, 100);
111 }
112 if (prevSentWaterPercentageValue != waterPercentage || waterPercentageSignal == null)
113 {
114 prevSentWaterPercentageValue = waterPercentage;
115 waterPercentageSignal = prevSentWaterPercentageValue.ToString();
116 }
117 item.SendSignal(waterPercentageSignal, "water_%");
118 }
119 string highPressureOut = (item.CurrentHull == null || item.CurrentHull.LethalPressure > 5.0f) ? "1" : "0";
120 item.SendSignal(highPressureOut, "high_pressure");
121 }
122 }
