40 protected override float GetPriority()
41 {
42 if (!IsAllowed)
43 {
44 Priority = 0;
45 return Priority;
46 }
47 if (character.CurrentHull == null)
48 {
49 if (!character.NeedsAir)
50 {
51 Priority = 0;
52 }
53 else
54 {
55 Priority = (
56 objectiveManager.HasOrder<AIObjectiveGoTo>(o => o.Priority > 0) ||
57 objectiveManager.HasOrder<AIObjectiveReturn>(o => o.Priority > 0) ||
58 objectiveManager.HasActiveObjective<AIObjectiveRescue>() ||
59 objectiveManager.Objectives.Any(o => o is AIObjectiveCombat && o.Priority > 0))
60 && HumanAIController.HasDivingSuit(character) ? 0 : 100;
61 }
62 }
63 else
64 {
65 if (HumanAIController.NeedsDivingGear(character.CurrentHull, out bool needsSuit) &&
66 (needsSuit ?
67 !HumanAIController.HasDivingSuit(character, conditionPercentage: AIObjectiveFindDivingGear.GetMinOxygen(character)) :
68 !HumanAIController.HasDivingGear(character, conditionPercentage: AIObjectiveFindDivingGear.GetMinOxygen(character))))
69 {
70 Priority = 100;
71 }
72 else if ((objectiveManager.IsCurrentOrder<AIObjectiveGoTo>() || objectiveManager.IsCurrentOrder<AIObjectiveReturn>()) &&
73 character.Submarine != null && !HumanAIController.IsOnFriendlyTeam(character.TeamID, character.Submarine.TeamID))
74 {
75 // Ordered to follow, hold position, or return back to main sub inside a hostile sub
76 // -> ignore find safety unless we need to find a diving gear
77 Priority = 0;
78 }
79 else if (objectiveManager.Objectives.Any(o => o is AIObjectiveCombat && o.Priority > 0))
80 {
81 Priority = 0;
82 }
83 Priority = MathHelper.Clamp(Priority, 0, 100);
84 if (divingGearObjective != null && !divingGearObjective.IsCompleted && divingGearObjective.CanBeCompleted)
85 {
86 // Boost the priority while seeking the diving gear
87 Priority = Math.Max(Priority, Math.Min(AIObjectiveManager.HighestOrderPriority + 20, 100));
88 }
89 }
90 return Priority;
91 }

93 public override void Update(float deltaTime)
94 {
95 if (retryTimer > 0)
96 {
97 retryTimer -= deltaTime;
98 if (retryTimer <= 0)
99 {
100 retryCounter = 0;
101 }
102 }
103 if (resetPriority)
104 {
105 Priority = 0;
106 resetPriority = false;
107 return;
108 }
109 if (character.CurrentHull == null)
110 {
111 currenthullSafety = 0;
112 }
113 else
114 {
115 currenthullSafety = HumanAIController.CurrentHullSafety;
116 if (currenthullSafety > HumanAIController.HULL_SAFETY_THRESHOLD)
117 {
118 Priority -= priorityDecrease * deltaTime;
119 }
120 else
121 {
122 float dangerFactor = (100 - currenthullSafety) / 100;
123 Priority += dangerFactor * priorityIncrease * deltaTime;
124 }
125 Priority = MathHelper.Clamp(Priority, 0, 100);
126 }

129 private Hull currentSafeHull;
130 private Hull previousSafeHull;
131 private bool cannotFindSafeHull;
132 private bool cannotFindDivingGear;
133 private readonly int findDivingGearAttempts = 2;
134 private int retryCounter;
135 private readonly float retryResetTime = 5;
136 private float retryTimer;
137 protected override void Act(float deltaTime)
138 {
139 var currentHull = character.CurrentHull;
140 bool dangerousPressure = currentHull == null || currentHull.LethalPressure > 0 && character.PressureProtection <= 0;
141 if (!character.LockHands && (!dangerousPressure || cannotFindSafeHull))
142 {
143 bool needsDivingGear = HumanAIController.NeedsDivingGear(currentHull, out bool needsDivingSuit);
144 bool needsEquipment = false;
145 if (needsDivingSuit)
146 {
147 needsEquipment = !HumanAIController.HasDivingSuit(character, AIObjectiveFindDivingGear.GetMinOxygen(character));
148 }
149 else if (needsDivingGear)
150 {
151 needsEquipment = !HumanAIController.HasDivingGear(character, AIObjectiveFindDivingGear.GetMinOxygen(character));
152 }
153 if (needsEquipment)
154 {
155 if (cannotFindDivingGear && retryCounter < findDivingGearAttempts)
156 {
157 retryTimer = retryResetTime;
158 retryCounter++;
159 needsDivingSuit = !needsDivingSuit;
160 RemoveSubObjective(ref divingGearObjective);
161 }
162 if (divingGearObjective == null)
163 {
164 cannotFindDivingGear = false;
165 RemoveSubObjective(ref goToObjective);
166 TryAddSubObjective(ref divingGearObjective,
167 constructor: () => new AIObjectiveFindDivingGear(character, needsDivingSuit, objectiveManager),
168 onAbandon: () =>
169 {
170 searchHullTimer = Math.Min(1, searchHullTimer);
171 cannotFindDivingGear = true;
172 // Don't reset the diving gear objective, because it's possible that there is no diving gear -> seek a safe hull and then reset so that we can check again.
173 },
174 onCompleted: () =>
175 {
176 resetPriority = true;
177 searchHullTimer = Math.Min(1, searchHullTimer);
178 RemoveSubObjective(ref divingGearObjective);
179 });
180 }
181 }
182 }
183 if (divingGearObjective == null || !divingGearObjective.CanBeCompleted)
184 {
185 if (currenthullSafety < HumanAIController.HULL_SAFETY_THRESHOLD)
186 {
187 searchHullTimer = Math.Min(1, searchHullTimer);
188 }
189 if (searchHullTimer > 0.0f)
190 {
191 searchHullTimer -= deltaTime;
192 }
193 else
194 {
195 searchHullTimer = SearchHullInterval * Rand.Range(0.9f, 1.1f);
196 previousSafeHull = currentSafeHull;
197 currentSafeHull = FindBestHull(allowChangingTheSubmarine: character.TeamID != CharacterTeamType.FriendlyNPC);
198 cannotFindSafeHull = currentSafeHull == null || HumanAIController.NeedsDivingGear(currentSafeHull, out _);
199 if (currentSafeHull == null)
200 {
201 currentSafeHull = previousSafeHull;
202 }
203 if (currentSafeHull != null && currentSafeHull != currentHull)
204 {
205 if (goToObjective?.Target != currentSafeHull)
206 {
207 RemoveSubObjective(ref goToObjective);
208 }
209 TryAddSubObjective(ref goToObjective,
210 constructor: () => new AIObjectiveGoTo(currentSafeHull, character, objectiveManager, getDivingGearIfNeeded: true)
211 {
212 AllowGoingOutside = HumanAIController.HasDivingSuit(character, conditionPercentage: 50)
213 },
214 onCompleted: () =>
215 {
216 if (currenthullSafety > HumanAIController.HULL_SAFETY_THRESHOLD ||
217 HumanAIController.NeedsDivingGear(currentHull, out bool needsSuit) && (needsSuit ? HumanAIController.HasDivingSuit(character) : HumanAIController.HasDivingMask(character)))
218 {
219 resetPriority = true;
220 searchHullTimer = Math.Min(1, searchHullTimer);
221 }
222 RemoveSubObjective(ref goToObjective);
223 if (cannotFindDivingGear)
224 {
225 // If diving gear objective failed, let's reset it here.
226 RemoveSubObjective(ref divingGearObjective);
227 }
228 },
229 onAbandon: () =>
230 {
231 // Don't ignore any hulls if outside, because apparently it happens that we can't find a path, in which case we just want to try again.
232 // If we ignore the hull, it might be the only airlock in the target sub, which ignores the whole sub.
233 // If the target hull is inside a submarine that is not our main sub, just ignore it normally when it cannot be reached. This happens with outposts.
234 if (goToObjective != null)
235 {
236 if (goToObjective.Target is Hull hull)
237 {
238 if (currentHull != null || !Submarine.MainSubs.Contains(hull.Submarine))
239 {
240 HumanAIController.UnreachableHulls.Add(hull);
241 }
242 }
243 }
244 RemoveSubObjective(ref goToObjective);
245 });
246 }
247 else
248 {
249 RemoveSubObjective(ref goToObjective);
250 }
251 }
252 if (subObjectives.Any(so => so.CanBeCompleted)) { return; }
253 if (currentHull != null)
254 {
255 //goto objective doesn't exist (a safe hull not found, or a path to a safe hull not found)
256 // -> attempt to manually steer away from hazards
257 Vector2 escapeVel = Vector2.Zero;
258 foreach (Hull hull in HumanAIController.VisibleHulls)
259 {
260 foreach (FireSource fireSource in hull.FireSources)
261 {
262 Vector2 dir = character.Position - fireSource.Position;
263 float distMultiplier = MathHelper.Clamp(100.0f / Vector2.Distance(fireSource.Position, character.Position), 0.1f, 10.0f);
264 escapeVel += new Vector2(Math.Sign(dir.X) * distMultiplier, !character.IsClimbing ? 0 : Math.Sign(dir.Y) * distMultiplier);
265 }
266 }
267 foreach (Character enemy in Character.CharacterList)
268 {
269 if (!HumanAIController.IsActive(enemy) || HumanAIController.IsFriendly(enemy) || enemy.IsArrested) { continue; }
270 if (HumanAIController.VisibleHulls.Contains(enemy.CurrentHull))
271 {
272 Vector2 dir = character.Position - enemy.Position;
273 float distMultiplier = MathHelper.Clamp(100.0f / Vector2.Distance(enemy.Position, character.Position), 0.1f, 10.0f);
274 escapeVel += new Vector2(Math.Sign(dir.X) * distMultiplier, !character.IsClimbing ? 0 : Math.Sign(dir.Y) * distMultiplier);
275 }
276 }
277 if (escapeVel != Vector2.Zero)
278 {
279 float left = currentHull.Rect.X + 50;
280 float right = currentHull.Rect.Right - 50;
281 //only move if we haven't reached the edge of the room
282 if (escapeVel.X < 0 && character.Position.X > left || escapeVel.X > 0 && character.Position.X < right)
283 {
284 character.AIController.SteeringManager.SteeringManual(deltaTime, escapeVel);
285 }
286 else
287 {
288 character.AnimController.TargetDir = escapeVel.X < 0.0f ? Direction.Right : Direction.Left;
289 character.AIController.SteeringManager.Reset();
290 }
291 return;
292 }
293 }
294 objectiveManager.GetObjective<AIObjectiveIdle>().Wander(deltaTime);
295 }

298 public Hull FindBestHull(IEnumerable<Hull> ignoredHulls = null, bool allowChangingTheSubmarine = true)
299 {
300 //sort the hulls based on distance and which sub they're in
301 //tends to make the method much faster, because we find a potential hull earlier and can discard further-away hulls more easily
302 //(for instance, an NPC in an outpost might otherwise go through all the hulls in the main sub first and do tons of expensive
303 //path calculations, only to discard all of them when going through the hulls in the outpost)
304 float EstimateHullSuitability(Hull hull)
305 {
306 float dist =
307 Math.Abs(hull.WorldPosition.X - character.WorldPosition.X) +
308 Math.Abs(hull.WorldPosition.Y - character.WorldPosition.Y) * 3;
309 float suitability = -dist;
310 if (hull.Submarine != character.Submarine)
311 {
312 suitability -= 10000.0f;
313 }
314 return suitability;
315 }
316
317 Hull bestHull = null;
318 float bestValue = 0;
319 bool bestIsAirlock = false;
320 foreach (Hull hull in Hull.HullList.OrderByDescending(h => EstimateHullSuitability(h)))
321 {
322 if (hull.Submarine == null) { continue; }
323 // Ruins are mazes filled with water. There's no safe hulls and we don't want to use the resources on it.
324 if (hull.Submarine.Info.IsRuin) { continue; }
325 if (!allowChangingTheSubmarine && hull.Submarine != character.Submarine) { continue; }
326 if (hull.Rect.Height < ConvertUnits.ToDisplayUnits(character.AnimController.ColliderHeightFromFloor) * 2) { continue; }
327 if (ignoredHulls != null && ignoredHulls.Contains(hull)) { continue; }
328 if (HumanAIController.UnreachableHulls.Contains(hull)) { continue; }
329 float hullSafety = 0;
330 bool hullIsAirlock = false;
331 bool isCharacterInside = character.CurrentHull != null && character.Submarine != null;
332 if (isCharacterInside)
333 {
334 if (!character.Submarine.IsConnectedTo(hull.Submarine)) { continue; }
335 hullSafety = HumanAIController.GetHullSafety(hull, hull.GetConnectedHulls(true, 1), character);
336 float yDist = Math.Abs(character.WorldPosition.Y - hull.WorldPosition.Y);
337 yDist = yDist > 100 ? yDist * 3 : 0;
338 float dist = Math.Abs(character.WorldPosition.X - hull.WorldPosition.X) + yDist;
339 float distanceFactor = MathHelper.Lerp(1, 0.9f, MathUtils.InverseLerp(0, 10000, dist));
340 hullSafety *= distanceFactor;
341 //skip the hull if the safety is already less than the best hull
342 //(no need to do the expensive pathfinding if we already know we're not going to choose this hull)
343 if (hullSafety < bestValue) { continue; }
344 //avoid airlock modules if not allowed to change the sub
345 if (!allowChangingTheSubmarine && hull.OutpostModuleTags.Any(t => t == "airlock"))
346 {
347 continue;
348 }
349 // Don't allow to go outside if not already outside.
350 var path = PathSteering.PathFinder.FindPath(character.SimPosition, hull.SimPosition, character.Submarine, nodeFilter: node => node.Waypoint.CurrentHull != null);
351 if (path.Unreachable)
352 {
353 HumanAIController.UnreachableHulls.Add(hull);
354 continue;
355 }
356 // Each unsafe node reduces the hull safety value.
357 // Ignore the current hull, because otherwise we couldn't find a path out.
358 int unsafeNodes = path.Nodes.Count(n => n.CurrentHull != character.CurrentHull && HumanAIController.UnsafeHulls.Contains(n.CurrentHull));
359 hullSafety /= 1 + unsafeNodes;
360 // If the target is not inside a friendly submarine, considerably reduce the hull safety.
361 if (!character.Submarine.IsEntityFoundOnThisSub(hull, true))
362 {
363 hullSafety /= 10;
364 }
365 }
366 else
367 {
368 // TODO: could also target gaps that get us inside?
369 if (hull.IsTaggedAirlock())
370 {
371 hullSafety = 100;
372 hullIsAirlock = true;
373 }
374 else if(!bestIsAirlock && hull.LeadsOutside(character))
375 {
376 hullSafety = 100;
377 }
378 // Huge preference for closer targets
379 float distance = Vector2.DistanceSquared(character.WorldPosition, hull.WorldPosition);
380 float distanceFactor = MathHelper.Lerp(1, 0.2f, MathUtils.InverseLerp(0, MathUtils.Pow(100000, 2), distance));
381 hullSafety *= distanceFactor;
382 // If the target is not inside a friendly submarine, considerably reduce the hull safety.
383 // Intentionally exclude wrecks from this check
384 if (hull.Submarine.TeamID != character.TeamID && hull.Submarine.TeamID != CharacterTeamType.FriendlyNPC)
385 {
386 hullSafety /= 10;
387 }
388 }
389 if (hullSafety > bestValue || (!isCharacterInside && hullIsAirlock && !bestIsAirlock))
390 {
391 bestHull = hull;
392 bestValue = hullSafety;
393 bestIsAirlock = hullIsAirlock;
394 }
395 }
396 return bestHull;
397 }
