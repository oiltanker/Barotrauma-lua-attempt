250 protected Rectangle ClampRect(Rectangle r)
251 {
252 if (Parent is null) { return r; }
253 Rectangle parentRect = !Parent.ClampMouseRectToParent ? Parent.Rect : Parent.ClampRect(Parent.Rect);
254 if (parentRect.Width <= 0 || parentRect.Height <= 0) { return Rectangle.Empty; }
255 if (parentRect.X > r.X)
256 {
257 int diff = parentRect.X - r.X;
258 r.X = parentRect.X;
259 r.Width -= diff;
260 }
261 if (parentRect.Y > r.Y)
262 {
263 int diff = parentRect.Y - r.Y;
264 r.Y = parentRect.Y;
265 r.Height -= diff;
266 }
267 if (parentRect.X + parentRect.Width < r.X + r.Width)
268 {
269 int diff = (r.X + r.Width) - (parentRect.X + parentRect.Width);
270 r.Width -= diff;
271 }
272 if (parentRect.Y + parentRect.Height < r.Y + r.Height)
273 {
274 int diff = (r.Y + r.Height) - (parentRect.Y + parentRect.Height);
275 r.Height -= diff;
276 }
277 if (r.Width <= 0 || r.Height <= 0) { return Rectangle.Empty; }
278 return r;
279 }

471 protected virtual void Update(float deltaTime)
472 {
473 if (!Visible) { return; }
474
475 if (CanBeFocused && OnSecondaryClicked != null)
476 {
477 if (GUI.IsMouseOn(this) && PlayerInput.SecondaryMouseButtonClicked())
478 {
479 OnSecondaryClicked?.Invoke(this, UserData);
480 }
481 }
482
483 if (Bounce)
484 {
485 if (bounceTimer > 3.0f || bounceDown)
486 {
487 RectTransform.ScreenSpaceOffset = new Point(RectTransform.ScreenSpaceOffset.X, (int) -(bounceJump * 10f));
488 if (!bounceDown)
489 {
490 bounceJump += deltaTime * 4;
491 if (bounceJump > 0.5f)
492 {
493 bounceDown = true;
494 }
495 }
496 else
497 {
498 bounceJump -= deltaTime * 4;
499 if (bounceJump <= 0.0f)
500 {
501 bounceJump = 0.0f;
502 bounceTimer = 0.0f;
503 bounceDown = false;
504 }
505 }
506 }
507 else
508 {
509 bounceTimer += deltaTime;
510 }
511 }
512
513 if (flashTimer > 0.0f)
514 {
515 flashTimer -= deltaTime;
516 }
517 if (spriteFadeTimer > 0)
518 {
519 spriteFadeTimer -= deltaTime;
520 }
521 if (colorFadeTimer > 0)
522 {
523 colorFadeTimer -= deltaTime;
524 }
525 }

614 protected virtual void Draw(SpriteBatch spriteBatch)
615 {
616 if (!Visible) { return; }
617 var rect = Rect;
618
619 GetBlendedColor(GetColor(State), ref _currentColor);
620
621 if (_currentColor.A > 0.0f && (sprites == null || !sprites.Any()))
622 {
623 GUI.DrawRectangle(spriteBatch, rect, _currentColor * (_currentColor.A / 255.0f), true);
624 }
625
626 if (sprites != null && _currentColor.A > 0)
627 {
628 if (!sprites.TryGetValue(_previousState, out List<UISprite> previousSprites) || previousSprites.None())
629 {
630 switch (FallBackState)
631 {
632 case SpriteFallBackState.Toggle:
633 sprites.TryGetValue(Selected ? ComponentState.Selected : ComponentState.None, out previousSprites);
634 break;
635 default:
636 if (Enum.TryParse(FallBackState.ToString(), ignoreCase: true, out ComponentState fallBackState))
637 {
638 sprites.TryGetValue(fallBackState, out previousSprites);
639 }
640 break;
641 }
642 }
643 // Handle fallbacks when some of the sprites are not defined
644 if (!sprites.TryGetValue(State, out List<UISprite> currentSprites) || currentSprites.None())
645 {
646 switch (FallBackState)
647 {
648 case SpriteFallBackState.Toggle:
649 sprites.TryGetValue(Selected ? ComponentState.Selected : ComponentState.None, out currentSprites);
650 break;
651 default:
652 if (Enum.TryParse(FallBackState.ToString(), ignoreCase: true, out ComponentState fallBackState))
653 {
654 sprites.TryGetValue(fallBackState, out currentSprites);
655 }
656 break;
657 }
658 }
659 if (_previousState != State && currentSprites != previousSprites)
660 {
661 if (previousSprites != null && previousSprites.Any())
662 {
663 // Draw the previous sprites(s) only while cross fading out
664 Color previousColor = GetColor(_previousState);
665 foreach (UISprite uiSprite in previousSprites)
666 {
667 float alphaMultiplier = SpriteCrossFadeTime > 0 && (uiSprite.CrossFadeOut || currentSprites != null && currentSprites.Any(s => s.CrossFadeIn))
668 ? MathUtils.InverseLerp(0, SpriteCrossFadeTime, ToolBox.GetEasing(uiSprite.TransitionMode, spriteFadeTimer)) : 0;
669 if (alphaMultiplier > 0)
670 {
671 uiSprite.Draw(spriteBatch, rect, previousColor * alphaMultiplier, SpriteEffects, uvOffset: UVOffset);
672 }
673 }
674 }
675 }
676 if (currentSprites != null && currentSprites.Any())
677 {
678 // Draw the current sprite(s)
679 foreach (UISprite uiSprite in currentSprites)
680 {
681 float alphaMultiplier = SpriteCrossFadeTime > 0 && (uiSprite.CrossFadeIn || previousSprites != null && previousSprites.Any(s => s.CrossFadeOut))
682 ? MathUtils.InverseLerp(SpriteCrossFadeTime, 0, ToolBox.GetEasing(uiSprite.TransitionMode, spriteFadeTimer)) : (_currentColor.A / 255.0f);
683 if (alphaMultiplier > 0)
684 {
685 // * (rect.Location.Y - PlayerInput.MousePosition.Y) / rect.Height
686 Vector2 offset = new Vector2(
687 MathUtils.PositiveModulo((int)-UVOffset.X, uiSprite.Sprite.SourceRect.Width),
688 MathUtils.PositiveModulo((int)-UVOffset.Y, uiSprite.Sprite.SourceRect.Height));
689 uiSprite.Draw(spriteBatch, rect, _currentColor * alphaMultiplier, SpriteEffects, uvOffset: offset);
690 }
691 }
692 }
693 }
694
695 if (GlowOnSelect && State == ComponentState.Selected)
696 {
697 GUIStyle.UIGlow.Draw(spriteBatch, Rect, SelectedColor);
698 }
699
700 if (flashTimer > 0.0f)
701 {
702 //the number of flashes depends on the duration, 1 flash per 1 full second
703 int flashCycleCount = (int)Math.Max(flashDuration, 1);
704 float flashCycleDuration = flashDuration / flashCycleCount;
705
706 Rectangle flashRect = Rect;
707 flashRect.Inflate(flashRectInflate.X, flashRectInflate.Y);
708
709 //MathHelper.Pi * 0.8f -> the curve goes from 144 deg to 0,
710 //i.e. quickly bumps up from almost full brightness to full and then fades out
711 if (useRectangleFlash)
712 {
713 GUI.DrawRectangle(spriteBatch, flashRect, flashColor * (float)Math.Sin(flashTimer % flashCycleDuration / flashCycleDuration * MathHelper.Pi * 0.8f), true);
714 }
715 else
716 {
717 var glow = useCircularFlash ? GUIStyle.UIGlowCircular : GUIStyle.UIGlow;
718 glow.Draw(spriteBatch,
719 flashRect,
720 flashColor * (float)Math.Sin(flashTimer % flashCycleDuration / flashCycleDuration * MathHelper.Pi * 0.8f));
721 }
722 }
723 }
