107 {
108 this.prefab = prefab;
109#if DEBUG
110 debugName = $"Particle ({prefab.Name})";
111#else
112 //don't instantiate new string objects in release builds
113 debugName = prefab.Name;
114#endif
115 spriteIndex = Rand.Int(prefab.Sprites.Count);
116
117 animState = 0;
118 animFrame = 0;
119
120 currentHull = Hull.FindHull(position, hullGuess);
121
122 size = prefab.StartSizeMin + (prefab.StartSizeMax - prefab.StartSizeMin) * Rand.Range(0.0f, 1.0f);
123
124 if (tracerPoints != null)
125 {
126 size = new Vector2(Vector2.Distance(tracerPoints.Item1, tracerPoints.Item2), size.Y);
127 position = (tracerPoints.Item1 + tracerPoints.Item2) / 2;

129

132 this.position = position;

141 velocity += ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);

143
144 this.rotation = rotation + Rand.Range(prefab.StartRotationMinRad, prefab.StartRotationMaxRad);

149

207 public UpdateResult Update(float deltaTime)
208 {
209 if (startDelay > 0.0f)
210 {
211 startDelay -= deltaTime;
212 return UpdateResult.Normal;
213 }
214
215 prevPosition = position;
216 prevRotation = rotation;
217
218 //over 3 times faster than position += velocity * deltatime
219 position.X += velocity.X * deltaTime;
220 position.Y += velocity.Y * deltaTime;
221
222 if (prefab.RotateToDirection)
223 {
224 if (velocityChange != Vector2.Zero || angularVelocity != 0.0f)
225 {
226 Vector2 relativeVel = velocity;
227 if (currentHull?.Submarine != null)
228 {
229 relativeVel -= ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);
230 }
231 rotation = MathUtils.VectorToAngle(new Vector2(relativeVel.X, -relativeVel.Y));
232 }
233 }
234 else
235 {
236 rotation += angularVelocity * deltaTime;
237 }
238
239 bool inWater = (currentHull == null || (currentHull.Submarine != null && position.Y - currentHull.Submarine.DrawPosition.Y < currentHull.Surface));
240 if (inWater)
241 {
242 velocity.X += velocityChangeWater.X * deltaTime;
243 velocity.Y += velocityChangeWater.Y * deltaTime;
244 if (prefab.WaterDrag > 0.0f)
245 {
246 ApplyDrag(prefab.WaterDrag, deltaTime);
247 }
248 }
249 else
250 {
251 velocity.X += velocityChange.X * deltaTime;
252 velocity.Y += velocityChange.Y * deltaTime;
253 if (prefab.Drag > 0.0f)
254 {
255 ApplyDrag(prefab.Drag, deltaTime);
256 }
257 }
258
259 size.X += sizeChange.X * deltaTime;
260 size.Y += sizeChange.Y * deltaTime;
261
262 if (UseMiddleColor)
263 {
264 if (lifeTime > totalLifeTime * 0.5f)
265 {
266 color = Color.Lerp(prefab.MiddleColor, prefab.StartColor, (lifeTime / totalLifeTime - 0.5f) * 2.0f);
267 }
268 else
269 {
270 color = Color.Lerp(prefab.EndColor, prefab.MiddleColor, lifeTime / totalLifeTime * 2.0f);
271 }
272 }
273 else
274 {
275 if (changeColor)
276 {
277 color = Color.Lerp(prefab.EndColor, prefab.StartColor, lifeTime / totalLifeTime);
278 }
279 }
280
281 if (prefab.Sprites[spriteIndex] is SpriteSheet)
282 {
283 animState += deltaTime;
284 int frameCount = ((SpriteSheet)prefab.Sprites[spriteIndex]).FrameCount;
285 if (prefab.LoopAnim)
286 {
287 animFrame = (int)(Math.Floor(animState / prefab.AnimDuration * frameCount) % frameCount);
288 }
289 else
290 {
291 animFrame = (int)Math.Min(Math.Floor(animState / prefab.AnimDuration * frameCount), frameCount - 1);
292 }
293 }
294
295 lifeTime -= deltaTime;
296 if (lifeTime <= 0.0f || color.A <= 0 || size.X <= 0.0f || size.Y <= 0.0f) { return UpdateResult.Delete; }
297
298 if (hasSubEmitters)
299 {
300 foreach (ParticleEmitter emitter in subEmitters)
301 {
302 emitter.Emit(deltaTime, position, currentHull);
303 }
304 }
305
306 if (collisionIgnoreTimer > 0f)
307 {
308 collisionIgnoreTimer -= deltaTime;
309 if (collisionIgnoreTimer <= 0f) { currentHull ??= Hull.FindHull(position); }
310 return UpdateResult.Normal;
311 }
312 if (!prefab.UseCollision) { return UpdateResult.Normal; }
313
314 if (HighQualityCollisionDetection)
315 {
316 return CollisionUpdate();
317 }
318 else
319 {
320 collisionUpdateTimer -= deltaTime;
321 if (collisionUpdateTimer <= 0.0f)
322 {
323 //more frequent collision updates if the particle is moving fast
324 collisionUpdateTimer = 0.5f - Math.Min((Math.Abs(velocity.X) + Math.Abs(velocity.Y)) * 0.01f, 0.45f);
325 return CollisionUpdate();
326 }
327 }
328
329 return UpdateResult.Normal;
330 }

332 private UpdateResult CollisionUpdate()
333 {
334 if (currentHull == null)
335 {
336 Hull collidedHull = Hull.FindHull(position);
337 if (collidedHull != null)
338 {
339 if (prefab.DeleteOnCollision) { return UpdateResult.Delete; }
340 OnWallCollisionOutside(collidedHull);
341 }
342 }
343 else
344 {
345 Rectangle hullRect = currentHull.WorldRect;
346 Vector2 collisionNormal = Vector2.Zero;
347 if (velocity.Y < 0.0f && position.Y - prefab.CollisionRadius * size.Y < hullRect.Y - hullRect.Height)
348 {
349 collisionNormal = new Vector2(0.0f, 1.0f);
350 }
351 else if (velocity.Y > 0.0f && position.Y + prefab.CollisionRadius * size.Y > hullRect.Y)
352 {
353 collisionNormal = new Vector2(0.0f, -1.0f);
354 }
355
356 if (collisionNormal != Vector2.Zero)
357 {
358 bool gapFound = false;
359 foreach (Gap gap in hullGaps)
360 {
361 if (gap.Open <= 0.9f || gap.IsHorizontal) { continue; }
362
363 if (gap.WorldRect.X > position.X || gap.WorldRect.Right < position.X) { continue; }
364 float hullCenterY = currentHull.WorldRect.Y - currentHull.WorldRect.Height / 2;
365 int gapDir = Math.Sign(gap.WorldRect.Y - hullCenterY);
366 if (Math.Sign(velocity.Y) != gapDir || Math.Sign(position.Y - hullCenterY) != gapDir) { continue; }
367
368 gapFound = true;
369 break;
370 }
371
372 if (prefab.DeleteOnCollision && !gapFound)
373 {
374 OnCollision?.Invoke(position, currentHull);
375 return UpdateResult.Delete;
376 }
377 handleCollision(gapFound, collisionNormal);
378 }
379
380 collisionNormal = Vector2.Zero;
381 if (velocity.X < 0.0f && position.X - prefab.CollisionRadius * size.X < hullRect.X)
382 {
383 collisionNormal = new Vector2(1.0f, 0.0f);
384 }
385 else if (velocity.X > 0.0f && position.X + prefab.CollisionRadius * size.X > hullRect.Right)
386 {
387 collisionNormal = new Vector2(-1.0f, 0.0f);
388 }
389
390 if (collisionNormal != Vector2.Zero)
391 {
392 bool gapFound = false;
393 foreach (Gap gap in hullGaps)
394 {
395 if (gap.Open <= 0.9f || !gap.IsHorizontal) { continue; }
396
397 if (gap.WorldRect.Y < position.Y || gap.WorldRect.Y - gap.WorldRect.Height > position.Y) { continue; }
398 int gapDir = Math.Sign(gap.WorldRect.Center.X - currentHull.WorldRect.Center.X);
399 if (Math.Sign(velocity.X) != gapDir || Math.Sign(position.X - currentHull.WorldRect.Center.X) != gapDir) { continue; }
400
401 gapFound = true;
402 break;
403 }
404 if (prefab.DeleteOnCollision && !gapFound)
405 {
406 OnCollision?.Invoke(position, currentHull);
407 return UpdateResult.Delete;
408 }
409 handleCollision(gapFound, collisionNormal);
410 }
411
412 void handleCollision(bool gapFound, Vector2 collisionNormal)
413 {
414 if (!gapFound)
415 {
416 OnWallCollisionInside(currentHull, collisionNormal);
417 }
418 else
419 {
420 Hull newHull = Hull.FindHull(position, currentHull);
421 if (newHull != currentHull)
422 {
423 currentHull = newHull;
424 hullGaps = currentHull == null ? new List<Gap>() : currentHull.ConnectedGaps;
425 OnChangeHull?.Invoke(position, currentHull);
426 }
427 }
428 }
429 }
430
431 return UpdateResult.Normal;

434 private void ApplyDrag(float dragCoefficient, float deltaTime)
435 {
436 Vector2 relativeVel = velocity;
437 if (currentHull?.Submarine != null)
438 {
439 relativeVel = velocity - ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);
440 }
441
442 float speed = relativeVel.Length();
443
444 relativeVel /= speed;
445
446 float drag = speed * speed * dragCoefficient * 0.01f * deltaTime;
447 if (drag > speed)
448 {
449 relativeVel = Vector2.Zero;
450 }
451 else
452 {
453 speed -= drag;
454 relativeVel *= speed;
455 }
456
457 velocity = relativeVel;
458 if (currentHull?.Submarine != null)
459 {
460 velocity += ConvertUnits.ToDisplayUnits(currentHull.Submarine.Velocity);
461 }
462 }

465 private void OnWallCollisionInside(Hull prevHull, Vector2 collisionNormal)
466 {
467 if (prevHull == null) { return; }
468
469 Rectangle prevHullRect = prevHull.WorldRect;
470
471 Vector2 subVel = prevHull?.Submarine != null ? ConvertUnits.ToDisplayUnits(prevHull.Submarine.Velocity) : Vector2.Zero;
472 velocity -= subVel;
473
474 if (Math.Abs(collisionNormal.X) > Math.Abs(collisionNormal.Y))
475 {
476 if (collisionNormal.X > 0.0f)
477 {
478 position.X = Math.Max(position.X, prevHullRect.X + prefab.CollisionRadius * size.X);
479 }
480 else
481 {
482 position.X = Math.Min(position.X, prevHullRect.Right - prefab.CollisionRadius * size.X);
483 }
484 velocity.X = Math.Sign(collisionNormal.X) * Math.Abs(velocity.X) * prefab.Restitution;
485 velocity.Y *= (1.0f - prefab.Friction);
486 }
487 else
488 {
489 if (collisionNormal.Y > 0.0f)
490 {
491 position.Y = Math.Max(position.Y, prevHullRect.Y - prevHullRect.Height + prefab.CollisionRadius * size.Y);
492 }
493 else
494 {
495 position.Y = Math.Min(position.Y, prevHullRect.Y - prefab.CollisionRadius * size.Y);
496
497 }
498 velocity.X *= (1.0f - prefab.Friction);
499 velocity.Y = Math.Sign(collisionNormal.Y) * Math.Abs(velocity.Y) * prefab.Restitution;
500 }
501
502 OnCollision?.Invoke(position, currentHull);
503
504 velocity += subVel;
505 }

508 private void OnWallCollisionOutside(Hull collisionHull)
509 {
510 Rectangle hullRect = collisionHull.WorldRect;
511
512 Vector2 center = new Vector2(hullRect.X + hullRect.Width / 2, hullRect.Y - hullRect.Height / 2);
513
514 if (position.Y < center.Y)
515 {
516 position.Y = hullRect.Y - hullRect.Height - prefab.CollisionRadius;
517 velocity.X *= (1.0f - prefab.Friction);
518 velocity.Y = -velocity.Y * prefab.Restitution;
519 }
520 else if (position.Y > center.Y)
521 {
522 position.Y = hullRect.Y + prefab.CollisionRadius;
523 velocity.X *= (1.0f - prefab.Friction);
524 velocity.Y = -velocity.Y * prefab.Restitution;
525 }
526
527 if (position.X < center.X)
528 {
529 position.X = hullRect.X - prefab.CollisionRadius;
530 velocity.X = -velocity.X * prefab.Restitution;
531 velocity.Y *= (1.0f - prefab.Friction);
532 }
533 else if (position.X > center.X)
534 {
535 position.X = hullRect.X + hullRect.Width + prefab.CollisionRadius;
536 velocity.X = -velocity.X * prefab.Restitution;
537 velocity.Y *= (1.0f - prefab.Friction);
538 }
539
540 OnCollision?.Invoke(position, currentHull);
541
542 velocity *= prefab.Restitution;
543 }

555 if (prefab.GrowTime > 0.0f && totalLifeTime - lifeTime < prefab.GrowTime)

564 ((SpriteSheet)prefab.Sprites[spriteIndex]).Draw(
565 spriteBatch, animFrame,
566 new Vector2(drawPosition.X, -drawPosition.Y),
567 currColor * (currColor.A / 255.0f),
568 prefab.Sprites[spriteIndex].Origin, drawRotation,
569 drawSize, SpriteEffects.None, prefab.Sprites[spriteIndex].Depth);
570 }
571 else
572 {
573 prefab.Sprites[spriteIndex].Draw(spriteBatch,
574 new Vector2(drawPosition.X, -drawPosition.Y),
