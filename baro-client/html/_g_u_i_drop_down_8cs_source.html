164 public GUIDropDown(RectTransform rectT, LocalizedString text = null, int elementCount = 4, string style = "", bool selectMultiple = false, bool dropAbove = false, Alignment textAlignment = Alignment.CenterLeft) : base(style, rectT)
165 {
166 text ??= new RawLString("");
167
168 HoverCursor = CursorState.Hand;
169 CanBeFocused = true;
170
171 this.selectMultiple = selectMultiple;
172
173 button = new GUIButton(new RectTransform(Vector2.One, rectT), text, textAlignment, style: "GUIDropDown")
174 {
175 OnClicked = OnClicked,
176 TextBlock = { OverflowClip = true }
177 };
178 GUIStyle.Apply(button, "", this);
179 button.TextBlock.SetTextPos();
180
181 Anchor listAnchor = dropAbove ? Anchor.TopCenter : Anchor.BottomCenter;
182 Pivot listPivot = dropAbove ? Pivot.BottomCenter : Pivot.TopCenter;
183 listBox = new GUIListBox(new RectTransform(new Point(Rect.Width, Rect.Height * MathHelper.Clamp(elementCount, 2, 10)), rectT, listAnchor, listPivot)
184 { IsFixedSize = false }, style: null)
185 {
186 Enabled = !selectMultiple
187 };
188 if (!selectMultiple) { listBox.OnSelected = SelectItem; }
189 GUIStyle.Apply(listBox, "GUIListBox", this);
190 GUIStyle.Apply(listBox.ContentBackground, "GUIListBox", this);
191
192 if (button.Style.ChildStyles.ContainsKey("dropdownicon".ToIdentifier()))
193 {
194 icon = new GUIImage(new RectTransform(new Vector2(0.6f, 0.6f), button.RectTransform, Anchor.CenterRight, scaleBasis: ScaleBasis.BothHeight) { AbsoluteOffset = new Point(5, 0) }, null, scaleToFit: true);
195 icon.ApplyStyle(button.Style.ChildStyles["dropdownicon".ToIdentifier()]);
196 }
197
198 currentHighestParent = FindHighestParent();
199 currentHighestParent.GUIComponent.OnAddedToGUIUpdateList += AddListBoxToGUIUpdateList;
200 rectT.ParentChanged += (RectTransform newParent) =>
201 {
202 currentHighestParent.GUIComponent.OnAddedToGUIUpdateList -= AddListBoxToGUIUpdateList;
203 if (newParent != null)
204 {
205 currentHighestParent = FindHighestParent();
206 currentHighestParent.GUIComponent.OnAddedToGUIUpdateList += AddListBoxToGUIUpdateList;
207 }
208 };
209 }

217 private RectTransform FindHighestParent()
218 {
219 parentHierarchy.Clear();
220
221 //collect entire parent hierarchy to a list
222 parentHierarchy = new List<RectTransform>() { RectTransform.Parent };
223 RectTransform parent = parentHierarchy.Last();
224 while (parent?.Parent != null)
225 {
226 parentHierarchy.Add(parent.Parent);
227 parent = parent.Parent;
228 }
229
230 //find the highest parent that has a guicomponent with a style
231 //(and so should be rendered and not just some empty parent/root element used for constructing a layout)
232 for (int i = parentHierarchy.Count - 1; i > 0; i--)
233 {
234 if (parentHierarchy[i] is GUICanvas ||
235 parentHierarchy[i].GUIComponent == null ||
236 parentHierarchy[i].GUIComponent.Style == null ||
237 parentHierarchy[i].GUIComponent == Screen.Selected?.Frame)
238 {
239 parentHierarchy.RemoveAt(i);
240 }
241 else
242 {
243 break;
244 }
245 }
246 return parentHierarchy.Last();
247 }

249 public void AddItem(LocalizedString text, object userData = null, LocalizedString toolTip = null)
250 {
251 toolTip ??= "";
252 if (selectMultiple)
253 {
254 var frame = new GUIFrame(new RectTransform(new Point(button.Rect.Width, button.Rect.Height), listBox.Content.RectTransform)
255 { IsFixedSize = false }, style: "ListBoxElement")
256 {
257 UserData = userData,
258 ToolTip = toolTip
259 };
260
261 new GUITickBox(new RectTransform(new Vector2(1.0f, 0.8f), frame.RectTransform, anchor: Anchor.CenterLeft) { MaxSize = new Point(int.MaxValue, (int)(button.Rect.Height * 0.8f)) }, text)
262 {
263 UserData = userData,
264 ToolTip = toolTip,
265 OnSelected = (GUITickBox tb) =>
266 {
267 List<LocalizedString> texts = new List<LocalizedString>();
268 selectedDataMultiple.Clear();
269 selectedIndexMultiple.Clear();
270 int i = 0;
271 foreach (GUIComponent child in ListBox.Content.Children)
272 {
273 var tickBox = child.GetChild<GUITickBox>();
274 if (tickBox.Selected)
275 {
276 selectedDataMultiple.Add(child.UserData);
277 selectedIndexMultiple.Add(i);
278 texts.Add(tickBox.Text);
279 }
280 i++;
281 }
282 button.Text = LocalizedString.Join(", ", texts);
283 // TODO: The callback is called at least twice, remove this?
284 OnSelected?.Invoke(tb.Parent, tb.Parent.UserData);
285 return true;
286 }
287 };
288 }
289 else
290 {
291 new GUITextBlock(new RectTransform(new Point(button.Rect.Width, button.Rect.Height), listBox.Content.RectTransform)
292 { IsFixedSize = false }, text, style: "ListBoxElement")
293 {
294 UserData = userData,
295 ToolTip = toolTip
296 };
297 }

310 private bool SelectItem(GUIComponent component, object obj)
311 {
312 if (selectMultiple)
313 {
314 foreach (GUIComponent child in ListBox.Content.Children)
315 {
316 var tickBox = child.GetChild<GUITickBox>();
317 if (Equals(obj, child.UserData)) { tickBox.Selected = true; }
318 }
319 }
320 else
321 {
322 if (!(component is GUITextBlock textBlock))
323 {
324 textBlock = component.GetChild<GUITextBlock>();
325 if (textBlock is null && !AllowNonText) { return false; }
326 }
327 button.Text = textBlock?.Text ?? "";
328 }
329 Dropped = false;
330 // TODO: OnSelected can be called multiple times and when it shouldn't be called -> turn into an event so that nobody else can call it.
331 OnSelected?.Invoke(component, component.UserData);
332 return true;
333 }

386 private void AddListBoxToGUIUpdateList(GUIComponent parent)
387 {
388 //the parent is not our parent anymore :(
389 //can happen when subscribed to a parent higher in the hierarchy (instead of the direct parent),
390 //and somewhere between this component and the higher parent a component was removed
391 for (int i = 1; i < parentHierarchy.Count; i++)
392 {
393 if (!parentHierarchy[i].IsParentOf(parentHierarchy[i - 1], recursive: false))
394 {
395 parent.OnAddedToGUIUpdateList -= AddListBoxToGUIUpdateList;
396 return;
397 }
398 }
399
400 if (Dropped)
401 {
402 listBox.AddToGUIUpdateList(false, 1);
403 }
