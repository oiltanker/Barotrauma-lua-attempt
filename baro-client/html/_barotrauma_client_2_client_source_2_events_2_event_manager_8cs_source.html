342 if (Screen.Selected is GameScreen screen)

414 private Rectangle DrawScriptedEvent(SpriteBatch spriteBatch, ScriptedEvent scriptedEvent, Rectangle? parentRect = null)
415 {
416 EventAction? currentEvent = !scriptedEvent.IsFinished ? scriptedEvent.Actions[scriptedEvent.CurrentActionIndex] : null;
417
418 List<DebugLine> positions = new List<DebugLine>();
419
420 string text = $"Finished: {scriptedEvent.IsFinished.ColorizeObject()}\n" +
421 $"Action index: {scriptedEvent.CurrentActionIndex.ColorizeObject()}\n" +
422 $"Current action: {currentEvent?.ToDebugString() ?? ToolBox.ColorizeObject(null)}\n";
423
424 text += "All actions:\n";
425 text += FindActions(scriptedEvent).Aggregate(string.Empty, (current, action) => current + $"{new string(' ', action.Item1 * 6)}{action.Item2.ToDebugString()}\n");
426
427 text += "Targets:\n";
428 foreach (var (key, value) in scriptedEvent.Targets)
429 {
430 text += $" {key.ColorizeObject()}: {value.Aggregate(string.Empty, (current, entity) => current + $"{entity.ColorizeObject()} ")}\n";
431 }
432
433 if (scriptedEvent.Targets != null)
434 {
435 foreach ((_, List<Entity> entities) in scriptedEvent.Targets)
436 {
437 if (entities == null || !entities.Any()) { continue; }
438
439 foreach (var entity in entities)
440 {
441 positions.Add(new DebugLine(entity.WorldPosition, Color.White));
442 }
443 }
444 }
445
446 return DrawInfoRectangle(spriteBatch, scriptedEvent, text, parentRect, positions);

571 public void ClientRead(IReadMessage msg)
572 {
573 NetworkEventType eventType = (NetworkEventType)msg.ReadByte();
574 switch (eventType)
575 {
576 case NetworkEventType.STATUSEFFECT:
577 Identifier eventIdentifier = msg.ReadIdentifier();
578 UInt16 actionIndex = msg.ReadUInt16();
579 UInt16 targetCount = msg.ReadUInt16();
580 List<Entity> targets = new List<Entity>();
581 for (int i = 0; i < targetCount; i++)
582 {
583 UInt16 targetID = msg.ReadUInt16();
584 Entity target = Entity.FindEntityByID(targetID);
585 if (target != null) { targets.Add(target); }
586 }
587
588 var eventPrefab = EventSet.GetEventPrefab(eventIdentifier);
589 if (eventPrefab == null) { return; }
590 int j = 0;
591 foreach (var element in eventPrefab.ConfigElement.Descendants())
592 {
593 if (j != actionIndex)
594 {
595 j++;
596 continue;
597 }
598 foreach (var subElement in element.Elements())
599 {

621 {
622 options.Add(msg.ReadString());
623 }
624

628 {

637 {
638 (mb as GUIMessageBox)?.Close();
639 }
640 });
641 }
642 else
643 {

645 }
646 if (Entity.FindEntityByID(speakerId) is Character speaker)
647 {
648 speaker.CampaignInteractionType = CampaignMode.InteractionType.None;
649 speaker.SetCustomInteract(null, null);
650 }
651 break;
652 case NetworkEventType.MISSION:
653 Identifier missionIdentifier = msg.ReadIdentifier();
654
655 MissionPrefab? prefab = MissionPrefab.Prefabs.Find(mp => mp.Identifier == missionIdentifier);
656 if (prefab != null)
657 {
658 new GUIMessageBox(string.Empty, TextManager.GetWithVariable("missionunlocked", "[missionname]", prefab.Name),
659 Array.Empty<LocalizedString>(), type: GUIMessageBox.Type.InGame, icon: prefab.Icon, relativeSize: new Vector2(0.3f, 0.15f), minSize: new Point(512, 128))
660 {
661 IconColor = prefab.IconColor
662 };
663 }
664 break;
665 case NetworkEventType.UNLOCKPATH:
666 UInt16 connectionIndex = msg.ReadUInt16();
667 if (GameMain.GameSession?.Map?.Connections != null)
668 {
669 if (connectionIndex >= GameMain.GameSession.Map.Connections.Count)
670 {
671 DebugConsole.ThrowError($"Failed to unlock a path on the campaign map. Connection index out of bounds (index: {connectionIndex}, number of connections: {GameMain.GameSession.Map.Connections.Count})");
672 }
673 else
674 {
675 GameMain.GameSession.Map.Connections[connectionIndex].Locked = false;
676 new GUIMessageBox(string.Empty, TextManager.Get("pathunlockedgeneric"),
677 Array.Empty<LocalizedString>(), type: GUIMessageBox.Type.InGame, iconStyle: "UnlockPathIcon", relativeSize: new Vector2(0.3f, 0.15f), minSize: new Point(512, 128));
678 }
679 }
680 break;
681 }
