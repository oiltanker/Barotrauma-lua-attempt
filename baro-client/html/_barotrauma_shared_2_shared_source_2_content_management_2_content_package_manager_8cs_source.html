78 public static IEnumerable<LoadProgress> SetRegularEnumerable(IReadOnlyList<RegularPackage> inNewRegular)
79 {
80 if (ReferenceEquals(inNewRegular, regular)) { yield break; }
81 if (inNewRegular.SequenceEqual(regular)) { yield break; }
82 ThrowIfDuplicates(inNewRegular);
83 var newRegular = inNewRegular.ToList();
84 IEnumerable<RegularPackage> toUnload = regular.Where(r => !newRegular.Contains(r));
85 RegularPackage[] toLoad = newRegular.Where(r => !regular.Contains(r)).ToArray();
86 toUnload.ForEach(r => r.UnloadPackage());
87
88 Range<float> loadingRange = new Range<float>(0.0f, 1.0f);
89
90 for (int i = 0; i < toLoad.Length; i++)
91 {
92 var package = toLoad[i];
93 loadingRange = new Range<float>(i / (float)toLoad.Length, (i + 1) / (float)toLoad.Length);
94 foreach (var progress in package.LoadPackageEnumerable())
95 {
96 if (progress.Exception != null)
97 {
98 //If an exception was thrown while loading this package, refuse to add it to the list of enabled packages
99 newRegular.Remove(package);
100 break;
101 }
102 yield return progress.Transform(loadingRange);
103 }
104 }
105 regular.Clear(); regular.AddRange(newRegular);
106 SortContent();
107 yield return new LoadProgress(1.0f);
108 }

270 public void Refresh()
271 {
272 //remove packages that have been deleted from the directory
273 corePackages.RemoveWhere(p => !File.Exists(p.Path));
274 regularPackages.RemoveWhere(p => !File.Exists(p.Path));
275
276 //load packages that have been added to the directory
277 var subDirs = Directory.GetDirectories(directory);
278 foreach (string subDir in subDirs)
279 {
280 var fileListPath = Path.Combine(subDir, ContentPackage.FileListFileName).CleanUpPathCrossPlatform();
281 if (this.Any(p => p.Path.Equals(fileListPath, StringComparison.OrdinalIgnoreCase))) { continue; }
282 if (File.Exists(fileListPath))
283 {
284 if (skipPredicate?.Invoke(fileListPath) is true) { continue; }
285
286 ContentPackage? newPackage = ContentPackage.TryLoad(fileListPath);
287 if (newPackage is CorePackage corePackage)
288 {
289 corePackages.Add(corePackage);
290 }
291 else if (newPackage is RegularPackage regularPackage)
292 {
293 regularPackages.Add(regularPackage);
294 }
295
296 if (!(newPackage is null))
297 {
298 Debug.WriteLine($"Loaded \"{newPackage.Name}\"");
299 }
300 }
301 }

377 public static ContentPackage? ReloadContentPackage(ContentPackage p)
378 {
379 ContentPackage? newPackage = ContentPackage.TryLoad(p.Path);
380 if (newPackage is CorePackage core)
381 {
382 if (EnabledPackages.Core == p) { EnabledPackages.SetCore(core); }
383 }
384 else if (newPackage is RegularPackage regular)
385 {
386 int index = EnabledPackages.Regular.IndexOf(p);
387 if (index >= 0)
388 {
389 var newRegular = EnabledPackages.Regular.ToArray();
390 newRegular[index] = regular;
391 EnabledPackages.SetRegular(newRegular);
392 }
393 }
394
395 if (newPackage != null)
396 {
397 LocalPackages.SwapPackage(p, newPackage);
398 WorkshopPackages.SwapPackage(p, newPackage);
399 }
400 EnabledPackages.DisableRemovedMods();
401 return newPackage;

439 public static IEnumerable<LoadProgress> Init()
440 {
441 Range<float> loadingRange = new Range<float>(0.0f, 1.0f);
442
443 SteamManager.Workshop.DeleteFailedCopies();
444 UpdateContentPackageList();
445
446 if (VanillaCorePackage is null) { LoadVanillaFileList(); }
447
448 CorePackage enabledCorePackage = VanillaCorePackage!;
449 List<RegularPackage> enabledRegularPackages = new List<RegularPackage>();
450
451#if CLIENT
452 TaskPool.Add("EnqueueWorkshopUpdates", EnqueueWorkshopUpdates(), t => { });
453#else
454 #warning TODO: implement Workshop updates for servers at some point
455#endif
456
457 var contentPackagesElement = XMLExtensions.TryLoadXml(GameSettings.PlayerConfigPath)?.Root
458 ?.GetChildElement("ContentPackages");
459 if (contentPackagesElement != null)
460 {
461 T? findPackage<T>(IEnumerable<T> packages, XElement? elem) where T : ContentPackage
462 {
463 if (elem is null) { return null; }
464 string name = elem.GetAttributeString("name", "");
465 string path = elem.GetAttributeStringUnrestricted("path", "").CleanUpPathCrossPlatform(correctFilenameCase: false);
466 return
467 packages.FirstOrDefault(p => p.Path.Equals(path, StringComparison.OrdinalIgnoreCase))
468 ?? packages.FirstOrDefault(p => p.NameMatches(name));
469 }
470
471 var corePackageElement = contentPackagesElement.GetChildElement(CorePackageElementName);
472 if (corePackageElement == null)
473 {
474 DebugConsole.AddWarning($"No core package selected. Switching to the \"{enabledCorePackage.Name}\" package.");
475 }
476 else
477 {
478 var configEnabledCorePackage = findPackage(CorePackages, corePackageElement);
479 if (configEnabledCorePackage == null)
480 {
481 string packageStr = corePackageElement.GetAttributeString("name", null) ?? corePackageElement.GetAttributeStringUnrestricted("path", "UNKNOWN");
482 DebugConsole.ThrowError($"Could not find the selected core package \"{packageStr}\". Switching to the \"{enabledCorePackage.Name}\" package.");
483 }
484 else
485 {
486 enabledCorePackage = configEnabledCorePackage;
487 }
488 }
489
490 var regularPackagesElement = contentPackagesElement.GetChildElement(RegularPackagesElementName);
491 if (regularPackagesElement != null)
492 {
493 XElement[] regularPackageElements = regularPackagesElement.GetChildElements(RegularPackagesSubElementName).ToArray();
494 for (int i = 0; i < regularPackageElements.Length; i++)
495 {
496 var regularPackage = findPackage(RegularPackages, regularPackageElements[i]);
497 if (regularPackage != null) { enabledRegularPackages.Add(regularPackage); }
498 }
499 }
500 }
501
502 int pkgCount = 1 + enabledRegularPackages.Count; //core + regular
503
504 loadingRange = new Range<float>(0.01f, 0.01f + (0.99f / pkgCount));
505 foreach (var p in EnabledPackages.SetCoreEnumerable(enabledCorePackage))
506 {
507 yield return p.Transform(loadingRange);
508 }
509
510 loadingRange = new Range<float>(0.01f + (0.99f / pkgCount), 1.0f);
511 foreach (var p in EnabledPackages.SetRegularEnumerable(enabledRegularPackages))
512 {
513 yield return p.Transform(loadingRange);
514 }
515
516 yield return new LoadProgress(1.0f);
517 }
