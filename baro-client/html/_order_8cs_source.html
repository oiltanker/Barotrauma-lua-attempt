151 public OrderPrefab(ContentXElement orderElement, OrdersFile file) : base(file, orderElement.GetAttributeIdentifier("identifier", ""))
152 {
153 Name = TextManager.Get($"OrderName.{Identifier}");
154 ContextualName = TextManager.Get($"OrderNameContextual.{Identifier}");
155
156 string targetItemType = orderElement.GetAttributeString("targetitemtype", "");
157 if (!string.IsNullOrWhiteSpace(targetItemType))
158 {
159 try
160 {
161 ItemComponentType = Type.GetType("Barotrauma.Items.Components." + targetItemType, true, true);
162 }
163
164 catch (Exception e)
165 {
166 DebugConsole.ThrowError("Error in the order definitions: item component type " + targetItemType + " not found", e);
167 }
168 }
169 CanTypeBeSubclass = orderElement.GetAttributeBool("cantypebesubclass", false);
170 color = orderElement.GetAttributeColor("color");
171 FadeOutTime = orderElement.GetAttributeFloat("fadeouttime", 0.0f);
172 UseController = orderElement.GetAttributeBool("usecontroller", false);
173 ControllerTags = orderElement.GetAttributeIdentifierArray("controllertags", Array.Empty<Identifier>()).ToImmutableArray();
174 TargetAllCharacters = orderElement.GetAttributeBool("targetallcharacters", false);
175 AppropriateJobs = orderElement.GetAttributeIdentifierArray("appropriatejobs", Array.Empty<Identifier>()).ToImmutableArray();
176 PreferredJobs = orderElement.GetAttributeIdentifierArray("preferredjobs", Array.Empty<Identifier>()).ToImmutableArray();
177 Options = orderElement.GetAttributeIdentifierArray("options", Array.Empty<Identifier>()).ToImmutableArray();
178 HiddenOptions = orderElement.GetAttributeIdentifierArray("hiddenoptions", Array.Empty<Identifier>()).ToImmutableArray();
179 AllOptions = Options.Concat(HiddenOptions).ToImmutableArray();
180
181 var optionTargetItems = new Dictionary<Identifier, ImmutableArray<Identifier>>();
182 if (orderElement.GetAttributeString("targetitems", "") is string targetItems && targetItems.Contains(';'))
183 {
184 string[] splitTargetItems = targetItems.Split(';');
185#if DEBUG
186 if (splitTargetItems.Length != AllOptions.Length)
187 {
188 DebugConsole.ThrowError($"Order \"{Identifier}\" has option-specific target items, but the option count doesn't match the target item count");
189 }
190#endif
191 var allTargetItems = new List<Identifier>();
192 for (int i = 0; i < AllOptions.Length; i++)
193 {
194 Identifier[] optionTargetItemsSplit = i < splitTargetItems.Length ? splitTargetItems[i].Split(',', '，').ToIdentifiers() : Array.Empty<Identifier>();
195 for (int j = 0; j < optionTargetItemsSplit.Length; j++)
196 {
197 optionTargetItemsSplit[j] = optionTargetItemsSplit[j].Value.Trim().ToIdentifier();
198 allTargetItems.Add(optionTargetItemsSplit[j]);
199 }
200 optionTargetItems.Add(AllOptions[i], optionTargetItemsSplit.ToImmutableArray());
201 }
202 TargetItems = allTargetItems.ToImmutableArray();
203 }
204 else
205 {
206 TargetItems = orderElement.GetAttributeIdentifierArray("targetitems", Array.Empty<Identifier>(), trim: true).ToImmutableArray();
207 }
208 RequireItems = orderElement.GetAttributeIdentifierArray("requireitems", Array.Empty<Identifier>(), trim: true).ToImmutableArray();
209 OptionTargetItems = optionTargetItems.ToImmutableDictionary();
210
211 var category = orderElement.GetAttributeString("category", null);
212 this.Category = !string.IsNullOrWhiteSpace(category) ? Enum.Parse<OrderCategory>(category, true) : (OrderCategory?)null;
213 this.CategoryIdentifier = (this.Category?.ToString() ?? string.Empty).ToIdentifier();
214 MustSetTarget = orderElement.GetAttributeBool("mustsettarget", false);
215 CanBeGeneralized = !MustSetTarget && orderElement.GetAttributeBool("canbegeneralized", true);
216 AppropriateSkill = orderElement.GetAttributeIdentifier("appropriateskill", Identifier.Empty);
217 Hidden = orderElement.GetAttributeBool("hidden", false);
218 IgnoreAtOutpost = orderElement.GetAttributeBool("ignoreatoutpost", false);
219
220 OptionNames =
221 new ListDictionary<Identifier, LocalizedString>(
222 TextManager.Get("OrderOptions." + Identifier).Split(',', '，'), Options.Length, i => Options[i]);
223
224 var spriteElement = orderElement.GetChildElement("sprite");
225 if (spriteElement != null)
226 {
227 SymbolSprite = new Sprite(spriteElement, lazyLoad: true);
228 }
229
230 var optionSprites = new Dictionary<Identifier, Sprite>();
231 if (Options != null && Options.Length > 0)
232 {
233 var optionSpriteElements = orderElement.GetChildElement("optionsprites")?.GetChildElements("sprite");
234 if (optionSpriteElements != null && optionSpriteElements.Any())
235 {
236 for (int i = 0; i < Options.Length; i++)
237 {
238 if (i >= optionSpriteElements.Count()) { break; };
239 var sprite = new Sprite(optionSpriteElements.ElementAt(i), lazyLoad: true);
240 optionSprites.Add(Options[i], sprite);
241 }
242 }
243 }
244 OptionSprites = optionSprites.ToImmutableDictionary();
245
246 MustManuallyAssign = orderElement.GetAttributeBool("mustmanuallyassign", false);
247 DrawIconWhenContained = orderElement.GetAttributeBool("displayiconwhencontained", false);
248 AutoDismiss = orderElement.GetAttributeBool("autodismiss", Category == OrderCategory.Operate || Category == OrderCategory.Movement);
249 AssignmentPriority = Math.Clamp(orderElement.GetAttributeInt("assignmentpriority", 100), 0, 100);
250 ColoredWhenControllingGiver = orderElement.GetAttributeBool("coloredwhencontrollinggiver", false);
251 DisplayGiverInTooltip = orderElement.GetAttributeBool("displaygiverintooltip", false);
252 }

254 private bool HasSpecifiedJob(Character character, IReadOnlyList<Identifier> jobs)
255 {
256 if (jobs == null || jobs.Count == 0) { return false; }
257 Identifier jobIdentifier = character?.Info?.Job?.Prefab?.Identifier ?? Identifier.Empty;
258 if (jobIdentifier.IsEmpty) { return false; }
259 for (int i = 0; i < jobs.Count; i++)
260 {
261 if (jobIdentifier == jobs[i]) { return true; }
262 }
263 return false;
264 }

270 public string GetChatMessage(string targetCharacterName, string targetRoomName, bool givingOrderToSelf, Identifier orderOption = default, bool isNewOrder = true)
271 {
272 if (!TargetAllCharacters && !isNewOrder && Identifier != "dismissed")
273 {
274 // Use special dialogue when we're rearranging character orders
275 if (!givingOrderToSelf)
276 {
277 return TextManager.GetWithVariable("rearrangedorders", "[name]", targetCharacterName ?? string.Empty).Value;
278 }
279 else
280 {
281 // Say nothing when rearranging the orders of the character you're controlling
282 return string.Empty;
283 }
284 }
285 string messageTag = $"{(givingOrderToSelf && !TargetAllCharacters ? "OrderDialogSelf" : "OrderDialog")}.{Identifier}";
286 if (!orderOption.IsEmpty)
287 {
288 if (Identifier != "dismissed")
289 {
290 messageTag += $".{orderOption}";
291 }
292 else
293 {
294 string[] splitOption = orderOption.Value.Split('.');
295 if (splitOption.Length > 0)
296 {
297 messageTag += $".{splitOption[0]}";
298 }
299 }
300 }
301 return TextManager.GetWithVariables(messageTag,
302 ("[name]", targetCharacterName ?? string.Empty, FormatCapitals.No),
303 ("[roomname]", targetRoomName ?? string.Empty, FormatCapitals.Yes)).Fallback("").Value;
304 }

309 public ItemComponent GetTargetItemComponent(Item item)
310 {
311 if (item?.Components == null || ItemComponentType == null) { return null; }
312 foreach (ItemComponent component in item.Components)
313 {
314 if (component?.GetType() is Type componentType)
315 {
316 if (componentType == ItemComponentType) { return component; }
317 if (CanTypeBeSubclass && componentType.IsSubclassOf(ItemComponentType)) { return component; }
318 }
319 }
320 return null;
321 }

330 public List<Item> GetMatchingItems(Submarine submarine, bool mustBelongToPlayerSub, CharacterTeamType? requiredTeam = null, Character interactableFor = null, Identifier orderOption = default)
331 {
332 List<Item> matchingItems = new List<Item>();
333 if (submarine == null) { return matchingItems; }
334 if (ItemComponentType != null || TargetItems.Any() || RequireItems.Any())
335 {
336 foreach (var item in Item.ItemList)
337 {
338 if (RequireItems.Any() && !TargetItemsMatchItem(RequireItems, item)) { continue; }
339 if (TargetItems.Any() && !TargetItemsMatchItem(item, orderOption)) { continue; }
340 if (RequireItems.None() && TargetItems.None() && !TryGetTargetItemComponent(item, out _)) { continue; }
341 if (mustBelongToPlayerSub && item.Submarine?.Info != null && item.Submarine.Info.Type != SubmarineType.Player) { continue; }
342 if (item.Submarine != submarine && !submarine.DockedTo.Contains(item.Submarine)) { continue; }
343 if (requiredTeam.HasValue && (item.Submarine == null || item.Submarine.TeamID != requiredTeam.Value)) { continue; }
344 if (item.NonInteractable) { continue; }
345 if (ItemComponentType != null && item.Components.None(c => c.GetType() == ItemComponentType)) { continue; }
346 Controller controller = null;
347 if (UseController && !item.TryFindController(out controller, tags: ControllerTags)) { continue; }
348 if (interactableFor != null && (!item.IsInteractable(interactableFor) || (UseController && !controller.Item.IsInteractable(interactableFor)))) { continue; }
349 matchingItems.Add(item);
350 }
351 }
352 return matchingItems;
353 }
