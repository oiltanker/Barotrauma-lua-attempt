71 int CalculateScalingEscortedCharacterCount(bool inMission = false)
72 {
73 if (missionSub == null || missionSub.Info == null) // UI logic failing to get the correct value is not important, but the mission logic must succeed
74 {
75 if (inMission)
76 {
77 DebugConsole.ThrowError("MainSub was null when trying to retrieve submarine size for determining escorted character count!");
78 }
79 return 1;
80 }
81 return (int)Math.Round(baseEscortedCharacters + scalingEscortedCharacters * (missionSub.Info.RecommendedCrewSizeMin + missionSub.Info.RecommendedCrewSizeMax) / 2);
82 }

84 private void InitEscort()
85 {
86 characters.Clear();
87 characterItems.Clear();
88
89 WayPoint explicitStayInHullPos = WayPoint.GetRandom(SpawnType.Human, null, Submarine.MainSub);
90 Rand.RandSync randSync = Rand.RandSync.ServerAndClient;
91
92 if (terroristChance > 0f)
93 {
94 // in terrorist missions, reroll characters each retry to avoid confusion as to who the terrorists are
95 randSync = Rand.RandSync.Unsynced;
96 }
97
98 //if any of the escortees have a job defined, try to use a spawnpoint designated for that job
99 foreach (XElement element in characterConfig.Elements())
100 {
101 var humanPrefab = GetHumanPrefabFromElement(element);
102 if (humanPrefab == null || string.IsNullOrEmpty(humanPrefab.Job) || humanPrefab.Job.Equals("any", StringComparison.OrdinalIgnoreCase)) { continue; }
103
104 var jobPrefab = humanPrefab.GetJobPrefab();
105 if (jobPrefab != null)
106 {
107 var jobSpecificSpawnPos = WayPoint.GetRandom(SpawnType.Human, jobPrefab, Submarine.MainSub);
108 if (jobSpecificSpawnPos != null)
109 {
110 explicitStayInHullPos = jobSpecificSpawnPos;
111 break;
112 }
113 }
114 }
115
116 foreach (XElement element in characterConfig.Elements())
117 {
118 int count = CalculateScalingEscortedCharacterCount(inMission: true);
119 for (int i = 0; i < count; i++)
120 {
121 Character spawnedCharacter = CreateHuman(GetHumanPrefabFromElement(element), characters, characterItems, Submarine.MainSub, CharacterTeamType.FriendlyNPC, explicitStayInHullPos, humanPrefabRandSync: randSync);
122 if (spawnedCharacter.AIController is HumanAIController humanAI)
123 {
124 humanAI.InitMentalStateManager();
125 }
126 }
127 }
128
129 if (terroristChance > 0f)
130 {
131 int terroristCount = (int)Math.Ceiling(terroristChance * Rand.Range(0.8f, 1.2f) * characters.Count);
132 terroristCount = Math.Clamp(terroristCount, 1, characters.Count);
133
134 terroristCharacters.Clear();
135 characters.GetRange(0, terroristCount).ForEach(c => terroristCharacters.Add(c));
136
137 terroristDistanceSquared = Vector2.DistanceSquared(Level.Loaded.StartPosition, Level.Loaded.EndPosition) * Rand.Range(0.35f, 0.65f);
138
139#if DEBUG
140 DebugConsole.AddWarning("Terrorists will trigger at range " + Math.Sqrt(terroristDistanceSquared));
141 foreach (Character character in terroristCharacters)
142 {
143 DebugConsole.AddWarning(character.Name + " is a terrorist.");
144 }
145#endif
146 }
147 }

149 private void InitCharacters()
150 {
151 int scalingCharacterCount = CalculateScalingEscortedCharacterCount(inMission: true);
152
153 if (scalingCharacterCount * characterConfig.Elements().Count() != characters.Count)
154 {
155 DebugConsole.AddWarning("Character count did not match expected character count in InitCharacters of EscortMission");
156 return;
157 }
158 int i = 0;
159
160 foreach (XElement element in characterConfig.Elements())
161 {
162 string escortIdentifier = element.GetAttributeString("escortidentifier", string.Empty);
163 string colorIdentifier = element.GetAttributeString("color", string.Empty);
164 for (int k = 0; k < scalingCharacterCount; k++)
165 {
166 // for each element defined, we need to initialize that type of character equal to the scaling escorted character count
167 characters[k + i].IsEscorted = true;
168 if (escortIdentifier != string.Empty)
169 {
170 if (escortIdentifier == "vip")
171 {
172 vipCharacter = characters[k + i];
173 }
174 }
175 characters[k + i].UniqueNameColor = element.GetAttributeColor("color", Color.LightGreen);
176 }
177 i++;
178 }

213 void TryToTriggerTerrorists()
214 {
215 if (terroristsShouldAct)
216 {
217 // decoupled from range check to prevent from weirdness if players handcuff a terrorist and move backwards
218 foreach (Character character in terroristCharacters)
219 {
220 if (character.HasTeamChange(TerroristTeamChangeIdentifier))
221 {
222 // already triggered
223 continue;
224 }
225
226 if (IsAlive(character) && !character.IsIncapacitated && !character.LockHands)
227 {
228 character.TryAddNewTeamChange(TerroristTeamChangeIdentifier, new ActiveTeamChange(CharacterTeamType.None, ActiveTeamChange.TeamChangePriorities.Willful, aggressiveBehavior: true));
229 character.Speak(TextManager.Get("dialogterroristannounce").Value, null, Rand.Range(0.5f, 3f));
230 XElement randomElement = itemConfig.Elements().GetRandomUnsynced(e => e.GetAttributeFloat(0f, "mindifficulty") <= Level.Loaded.Difficulty);
231 if (randomElement != null)
232 {
233 HumanPrefab.InitializeItem(character, randomElement, character.Submarine, humanPrefab: null, createNetworkEvents: true);
234 }
235 }
236 }
237 }
238 else if (Vector2.DistanceSquared(Submarine.MainSub.WorldPosition, Level.Loaded.EndPosition) < terroristDistanceSquared)
239 {
240 foreach (Character character in terroristCharacters)
241 {
242 if (character.AIController is HumanAIController humanAI)
243 {
244 humanAI.ObjectiveManager.AddObjective(new AIObjectiveEscapeHandcuffs(character, humanAI.ObjectiveManager, shouldSwitchTeams: false, beginInstantly: true));
245 }
246 }
247 terroristsShouldAct = true;
248 }

256 protected override void UpdateMissionSpecific(float deltaTime)
257 {
258 if (!IsClient)
259 {
260 int newState = State;
261 TryToTriggerTerrorists();
262 switch (State)
263 {
264 case 0: // base
265 if (!NonTerroristsStillAlive(characters))
266 {
267 newState = 1;
268 }
269 if (terroristCharacters.Any() && terroristCharacters.All(c => !IsAlive(c)))
270 {
271 newState = 2;
272 }
273 break;
274 case 1: // failure
275 break;
276 case 2: // terrorists killed
277 if (!NonTerroristsStillAlive(characters))
278 {
279 newState = 1;
280 }
281 break;
282 }
283 State = newState;
284 }

303 public override void End()
304 {
305 if (Submarine.MainSub != null && Submarine.MainSub.AtEndExit)
306 {
307 bool terroristsSurvived = terroristCharacters.Any(c => Survived(c) && !IsCaptured(c));
308 bool friendliesSurvived = characters.Except(terroristCharacters).All(c => Survived(c));
309 bool vipDied = false;
310
311 // this logic is currently irrelevant, as the mission is failed regardless of who dies
312 if (vipCharacter != null)
313 {
314 vipDied = !Survived(vipCharacter);
315 }
316
317 if (friendliesSurvived && !terroristsSurvived && !vipDied)
318 {
319 GiveReward();
320 completed = true;
321 }
322 }
323
324 if (!IsClient)
325 {
326 foreach (Character character in characters)
327 {
328 if (character.Inventory == null) { continue; }
329 foreach (Item item in character.Inventory.AllItemsMod)
330 {
331 //item didn't spawn with the characters -> drop it
332 if (!characterItems.Any(c => c.Value.Contains(item)))
333 {
334 item.Drop(character);
335 }
336 }
337 }
338
339 // characters that survived will take their items with them, in case players tried to be crafty and steal them
340 // this needs to run here in case players abort the mission by going back home
341 foreach (var characterItem in characterItems)
342 {
343 if (Survived(characterItem.Key) || !completed)
344 {
345 foreach (Item item in characterItem.Value)
346 {
347 if (!item.Removed)
348 {
349 item.Remove();
350 }
351 }
352 }
353 }
354 }
355
356 characters.Clear();
357 characterItems.Clear();
358 failed = !completed;
359 }
