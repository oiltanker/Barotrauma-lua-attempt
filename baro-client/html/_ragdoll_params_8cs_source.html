130 public static T GetRagdollParams<T>(Identifier speciesName, string fileName = null) where T : RagdollParams, new()
131 {
132 if (speciesName.IsEmpty)
133 {
134 throw new Exception($"Species name null or empty!");
135 }
136 if (!allRagdolls.TryGetValue(speciesName, out Dictionary<string, RagdollParams> ragdolls))
137 {
138 ragdolls = new Dictionary<string, RagdollParams>();
139 allRagdolls.Add(speciesName, ragdolls);
140 }
141
142 if (!string.IsNullOrEmpty(fileName) && ragdolls.TryGetValue(fileName, out RagdollParams ragdoll))
143 {
144 return (T)ragdoll;

149 void tryFolderForSpecies(Identifier species, out string err)
150 {
151 err = null;
152 string folder = GetFolder(species);
153 if (!Directory.Exists(folder))
154 {
155 err = $"[RagdollParams] Invalid directory: {folder}. Using the default ragdoll.";
156 selectedFile = GetDefaultFile(species);
157 return;
158 }
159
160 string[] files = Directory.GetFiles(folder);
161 if (files.None())
162 {
163 err = $"[RagdollParams] Could not find any ragdoll files from the folder: {folder}. Using the default ragdoll.";
164 selectedFile = GetDefaultFile(species);
165 }
166 else if (string.IsNullOrEmpty(fileName))
167 {
168 // Files found, but none specified
169 selectedFile = GetDefaultFile(species);
170 }
171 else
172 {
173 selectedFile = files.FirstOrDefault(f => IO.Path.GetFileNameWithoutExtension(f).Equals(fileName, StringComparison.OrdinalIgnoreCase));
174 if (selectedFile == null)
175 {
176 err = $"[RagdollParams] Could not find a ragdoll file that matches the name {fileName}. Using the default ragdoll.";
177 selectedFile = GetDefaultFile(species);
178 }
179 }
180 }
181
182 tryFolderForSpecies(speciesName, out var error);
183 Identifier parentSpeciesName = CharacterPrefab.Prefabs.TryGet(speciesName, out var prefab)
184 ? prefab.VariantOf
185 : Identifier.Empty;
186 if (!error.IsNullOrEmpty() && !parentSpeciesName.IsEmpty)
187 {
188 tryFolderForSpecies(parentSpeciesName, out error);
189 }
190
191 if (!error.IsNullOrEmpty())
192 {
193 DebugConsole.ThrowError(error);
194 }
195
196 if (selectedFile == null)
197 {
198 throw new Exception("[RagdollParams] Selected file null!");
199 }
200 DebugConsole.Log($"[RagdollParams] Loading ragdoll from {selectedFile}.");
201 var characterPrefab = CharacterPrefab.Prefabs[speciesName];
202 T r = new T();
203 if (r.Load(ContentPath.FromRaw(characterPrefab.ContentPackage, selectedFile), speciesName))
204 {
205 if (!ragdolls.ContainsKey(r.Name))
206 {
207 ragdolls.Add(r.Name, r);
208 }
209 return r;
210 }
211 else
212 {
213 // Failing to create a ragdoll causes so many issues that cannot be handled. Dummy ragdoll just seems to make things harded to debug. It's better to fail early.
214 throw new Exception($"[RagdollParams] Failed to load ragdoll {r.Name} from {selectedFile} for the character {speciesName}.");
215 }

222 public static T CreateDefault<T>(string fullPath, Identifier speciesName, XElement mainElement) where T : RagdollParams, new()
223 {
224 // Remove the old ragdolls, if found.
225 if (allRagdolls.ContainsKey(speciesName))
226 {
227 DebugConsole.NewMessage($"[RagdollParams] Removing the old ragdolls from {speciesName}.", Color.Red);
228 allRagdolls.Remove(speciesName);
229 }
230 var ragdolls = new Dictionary<string, RagdollParams>();
231 allRagdolls.Add(speciesName, ragdolls);
232 var instance = new T
233 {
234 doc = new XDocument(mainElement)
235 };
236 var characterPrefab = CharacterPrefab.Prefabs[speciesName];
237 var contentPath = ContentPath.FromRaw(characterPrefab.ContentPackage, fullPath);
238 instance.UpdatePath(contentPath);
239 instance.IsLoaded = instance.Deserialize(mainElement);
240 instance.Save();
241 instance.Load(contentPath, speciesName);
242 ragdolls.Add(instance.Name, instance);
243 DebugConsole.NewMessage("[RagdollParams] New default ragdoll params successfully created at " + fullPath, Color.NavajoWhite);
244 return instance as T;
245 }

444 private void RevertTo(RagdollParams source)
445 {
446 if (source.MainElement == null)
447 {
448 DebugConsole.ThrowError("[RagdollParams] The source XML Element of the given RagdollParams is null!");
449 return;
450 }
451 Deserialize(source.MainElement, alsoChildren: false);
452 var sourceSubParams = source.GetAllSubParams().ToList();
453 var subParams = GetAllSubParams().ToList();
454 // TODO: cannot currently undo joint/limb deletion.
455 if (sourceSubParams.Count != subParams.Count)
456 {
457 DebugConsole.ThrowError("[RagdollParams] The count of the sub params differs! Failed to revert to the previous snapshot! Please reset the ragdoll to undo the changes.");
458 return;
459 }
460 for (int i = 0; i < subParams.Count; i++)
461 {
462 var subSubParams = subParams[i].SubParams;
463 if (subSubParams.Count != sourceSubParams[i].SubParams.Count)
464 {
465 DebugConsole.ThrowError("[RagdollParams] The count of the sub sub params differs! Failed to revert to the previous snapshot! Please reset the ragdoll to undo the changes.");
466 return;
467 }
468 subParams[i].Deserialize(sourceSubParams[i].Element, recursive: false);
469 for (int j = 0; j < subSubParams.Count; j++)
470 {
471 subSubParams[j].Deserialize(sourceSubParams[i].SubParams[j].Element, recursive: false);
472 // Since we cannot use recursion here, we have to go deeper manually, if necessary.
473 }
474 }
