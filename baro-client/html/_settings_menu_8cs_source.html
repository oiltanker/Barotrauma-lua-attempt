278 private static string TrimAudioDeviceName(string name)
279 {
280 if (string.IsNullOrWhiteSpace(name)) { return string.Empty; }
281 string[] prefixes = { "OpenAL Soft on " };
282 foreach (string prefix in prefixes)
283 {
284 if (name.StartsWith(prefix, StringComparison.OrdinalIgnoreCase))
285 {
286 return name.Remove(0, prefix.Length);
287 }
288 }
289 return name;
290 }

292 private static int HandleAlErrors(string message)
293 {
294 int alcError = Alc.GetError(IntPtr.Zero);
295 if (alcError != Alc.NoError)
296 {
297 DebugConsole.ThrowError($"{message}: ALC error {Alc.GetErrorString(alcError)}");
298 return alcError;
299 }
300
301 int alError = Al.GetError();
302 if (alError != Al.NoError)
303 {
304 DebugConsole.ThrowError($"{message}: AL error {Al.GetErrorString(alError)}");
305 return alError;
306 }
307
308 return Al.NoError;
309 }

311 private static void GetAudioDevices(int listSpecifier, int defaultSpecifier, out IReadOnlyList<string> list, ref string current)
312 {
313 list = Array.Empty<string>();
314
315 var retVal = Alc.GetStringList(IntPtr.Zero, listSpecifier).ToList();
316 if (HandleAlErrors("Alc.GetStringList failed") != Al.NoError) { return; }
317
318 list = retVal;
319 if (string.IsNullOrEmpty(current))
320 {
321 current = Alc.GetString(IntPtr.Zero, defaultSpecifier);
322 if (HandleAlErrors("Alc.GetString failed") != Al.NoError) { return; }
323 }
324
325 string currentVal = current;
326 if (list.Any() && !list.Any(n => n.Equals(currentVal, StringComparison.OrdinalIgnoreCase)))
327 {
328 current = list[0];
329 }
330 }
