128 partial void EmitParticles(float deltaTime)
129 {
130 if (flowTargetHull == null) { return; }
131
132 if (linkedTo.Count == 2 && linkedTo[0] is Hull hull1 && linkedTo[1] is Hull hull2)
133 {
134 //no flow particles between linked hulls (= rooms consisting of multiple hulls)
135 if (hull1.linkedTo.Contains(hull2)) { return; }
136 foreach (var linkedEntity in hull1.linkedTo)
137 {
138 if (linkedEntity is Hull h && h.linkedTo.Contains(hull1) && h.linkedTo.Contains(hull2)) { return; }
139 }
140 foreach (var linkedEntity in hull2.linkedTo)
141 {
142 if (linkedEntity is Hull h && h.linkedTo.Contains(hull1) && h.linkedTo.Contains(hull2)) { return; }
143 }
144 }
145
146 Vector2 pos = Position;
147 if (IsHorizontal)
148 {
149 pos.X += Math.Sign(flowForce.X);
150 pos.Y = MathHelper.Clamp(Rand.Range(higherSurface, lowerSurface), rect.Y - rect.Height, rect.Y);
151 }
152 else
153 {
154 pos.Y += Math.Sign(flowForce.Y) * rect.Height / 2.0f;
155 }
156
157 //spawn less particles when there's already a large number of them
158 float particleAmountMultiplier = 1.0f - GameMain.ParticleManager.ParticleCount / (float)GameMain.ParticleManager.MaxParticles;
159 particleAmountMultiplier *= particleAmountMultiplier;
160
161 //light dripping
162 if (open < 0.2f && LerpedFlowForce.LengthSquared() > 100.0f)
163 {
164 particleTimer += deltaTime;
165 float particlesPerSec = open * 100.0f * particleAmountMultiplier;
166 float emitInterval = 1.0f / particlesPerSec;
167 while (particleTimer > emitInterval)
168 {
169 Vector2 velocity = flowForce;
170 if (!IsHorizontal)
171 {
172 velocity.X = Rand.Range(-500.0f, 500.0f) * open;
173 }
174 else
175 {
176 velocity.X *= Rand.Range(1.0f, 3.0f);
177 }
178
179 if (flowTargetHull.WaterVolume < flowTargetHull.Volume)
180 {
181 GameMain.ParticleManager.CreateParticle(
182 Rand.Range(0.0f, open) < 0.05f ? "waterdrop" : "watersplash",
183 (Submarine == null ? pos : pos + Submarine.Position),
184 velocity, 0, flowTargetHull);
185 }
186
187 GameMain.ParticleManager.CreateParticle(
188 "bubbles",
189 (Submarine == null ? pos : pos + Submarine.Position),
190 velocity, 0, flowTargetHull);
191
192 particleTimer -= emitInterval;
193 }
194 }
195 //heavy flow -> strong waterfall type of particles
196 else if (LerpedFlowForce.LengthSquared() > 20000.0f)
197 {
198 particleTimer += deltaTime;
199 if (IsHorizontal)
200 {
201 float particlesPerSec = open * rect.Height * 0.1f * particleAmountMultiplier;
202 if (openedTimer > 0.0f) { particlesPerSec *= 1.0f + openedTimer * 10.0f; }
203 float emitInterval = 1.0f / particlesPerSec;
204 while (particleTimer > emitInterval)
205 {
206 Vector2 velocity = new Vector2(
207 MathHelper.Clamp(flowForce.X, -5000.0f, 5000.0f) * Rand.Range(0.5f, 0.7f),
208 flowForce.Y * Rand.Range(0.5f, 0.7f));
209
210 if (flowTargetHull.WaterVolume < flowTargetHull.Volume * 0.95f)
211 {
212 var particle = GameMain.ParticleManager.CreateParticle(
213 "watersplash",
214 (Submarine == null ? pos : pos + Submarine.Position) - Vector2.UnitY * Rand.Range(0.0f, 10.0f),
215 velocity, 0, flowTargetHull);
216 if (particle != null)
217 {
218 if (particle.CurrentHull == null) { GameMain.ParticleManager.RemoveParticle(particle); }
219 particle.Size *= Math.Min(Math.Abs(flowForce.X / 500.0f), 5.0f);
220 }
221 }
222
223 if (Math.Abs(flowForce.X) > 300.0f && flowTargetHull.WaterVolume > flowTargetHull.Volume * 0.1f)
224 {
225 pos.X += Math.Sign(flowForce.X) * 10.0f;
226 if (rect.Height < 32)
227 {
228 pos.Y = rect.Y - rect.Height / 2;
229 }
230 else
231 {
232 float bottomY = rect.Y - rect.Height + 16;
233 float topY = MathHelper.Clamp(lowerSurface, bottomY, rect.Y - 16);
234 pos.Y = Rand.Range(bottomY, topY);
235 }
236 GameMain.ParticleManager.CreateParticle(
237 "bubbles",
238 Submarine == null ? pos : pos + Submarine.Position,
239 velocity, 0, flowTargetHull);
240 }
241 particleTimer -= emitInterval;
242 }
243 }
244 else
245 {
246 if (Math.Sign(flowTargetHull.Rect.Y - rect.Y) != Math.Sign(lerpedFlowForce.Y)) { return; }
247
248 float particlesPerSec = Math.Max(open * rect.Width * 0.3f * particleAmountMultiplier, 20.0f);
249 float emitInterval = 1.0f / particlesPerSec;
250 while (particleTimer > emitInterval)
251 {
252 pos.X = Rand.Range(rect.X, rect.X + rect.Width + 1);
253 Vector2 velocity = new Vector2(
254 lerpedFlowForce.X * Rand.Range(0.5f, 0.7f),
255 MathHelper.Clamp(lerpedFlowForce.Y, -500.0f, 1000.0f) * Rand.Range(0.5f, 0.7f));
256
257 if (flowTargetHull.WaterVolume < flowTargetHull.Volume * 0.95f)
258 {
259 var splash = GameMain.ParticleManager.CreateParticle(
260 "watersplash",
261 Submarine == null ? pos : pos + Submarine.Position,
262 velocity, 0, FlowTargetHull);
263 if (splash != null)
264 {
265 if (splash.CurrentHull == null) { GameMain.ParticleManager.RemoveParticle(splash); }
266 splash.Size *= MathHelper.Clamp(rect.Width / 50.0f, 1.5f, 4.0f);
267 }
268 }
269 if (Math.Abs(flowForce.Y) > 190.0f && Rand.Range(0.0f, 1.0f) < 0.3f && flowTargetHull.WaterVolume > flowTargetHull.Volume * 0.1f)
270 {
271 GameMain.ParticleManager.CreateParticle(
272 "bubbles",
273 Submarine == null ? pos : pos + Submarine.Position,
274 flowForce / 2.0f, 0, FlowTargetHull);
275 }
276 particleTimer -= emitInterval;
277 }
278 }
279 }
280 else
281 {
282 particleTimer = 0.0f;
283 }
