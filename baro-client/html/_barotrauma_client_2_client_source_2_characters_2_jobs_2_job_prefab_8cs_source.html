79 public List<OutfitPreview> GetJobOutfitSprites(CharacterInfoPrefab charInfoPrefab, bool useInventoryIcon, out Vector2 maxDimensions)
80 {
81 List<OutfitPreview> outfitPreviews = new List<OutfitPreview>();
82 maxDimensions = Vector2.One;
83
84 var equipIdentifiers = Element.GetChildElements("ItemSet").Elements().Where(e => e.GetAttributeBool("outfit", false)).Select(e => e.GetAttributeIdentifier("identifier", ""));
85
86 List<ItemPrefab> outfitPrefabs = new List<ItemPrefab>();
87 foreach (var equipIdentifier in equipIdentifiers)
88 {
89 var itemPrefab = ItemPrefab.Prefabs.Find(ip => ip.Identifier == equipIdentifier);
90 if (itemPrefab != null) { outfitPrefabs.Add(itemPrefab); }
91 }
92
93 if (!outfitPrefabs.Any()) { return null; }
94
95 for (int i = 0; i < outfitPrefabs.Count; i++)
96 {
97 var outfitPreview = new OutfitPreview();
98
99 if (!ItemSets.TryGetValue(i, out var itemSetElement)) { continue; }
100 var previewElement = itemSetElement.GetChildElement("PreviewSprites");
101 if (previewElement == null || useInventoryIcon)
102 {
103 if (outfitPrefabs[i] is ItemPrefab prefab && prefab.InventoryIcon != null)
104 {
105 outfitPreview.AddSprite(prefab.InventoryIcon, Vector2.Zero);
106 outfitPreview.Dimensions = prefab.InventoryIcon.SourceRect.Size.ToVector2();
107 maxDimensions.X = MathHelper.Max(maxDimensions.X, outfitPreview.Dimensions.X);
108 maxDimensions.Y = MathHelper.Max(maxDimensions.Y, outfitPreview.Dimensions.Y);
109 }
110 outfitPreviews.Add(outfitPreview);
111 continue;
112 }
113
114 var children = previewElement.Elements().ToList();
115 for (int n = 0; n < children.Count; n++)
116 {
117 var spriteElement = children[n];
118 string spriteTexture = charInfoPrefab.ReplaceVars(spriteElement.GetAttributeString("texture", ""), charInfoPrefab.Heads.First());
119 var sprite = new Sprite(spriteElement, file: spriteTexture);
120 sprite.size = new Vector2(sprite.SourceRect.Width, sprite.SourceRect.Height);
121 outfitPreview.AddSprite(sprite, children[n].GetAttributeVector2("offset", Vector2.Zero));
122 }
123
124 outfitPreview.Dimensions = previewElement.GetAttributeVector2("dims", Vector2.One);
125 maxDimensions.X = MathHelper.Max(maxDimensions.X, outfitPreview.Dimensions.X);
126 maxDimensions.Y = MathHelper.Max(maxDimensions.Y, outfitPreview.Dimensions.Y);
127
128 outfitPreviews.Add(outfitPreview);
129 }
130
131 return outfitPreviews;
132 }
133 }
