29 public static VotingInterface CreateSubmarineVotingInterface(Client starter, SubmarineInfo info, VoteType type, float votingTime)
30 {
31 if (starter == null || info == null) { return null; }
32
33 var subVoting = new VotingInterface()
34 {
35 votingTime = votingTime,
36 getYesVotes = () => GameMain.NetworkMember?.Voting?.GetVoteCountYes(type) ?? 0,
37 getNoVotes = () => GameMain.NetworkMember?.Voting?.GetVoteCountNo(type) ?? 0,
38 getMaxVotes = () => GameMain.NetworkMember?.Voting?.GetVoteCountMax(type) ?? 0,
39 };
40 subVoting.onVoteEnd = () => subVoting.SendSubmarineVoteEndMessage(info, type);
41 subVoting.SetSubmarineVotingText(starter, info, type);

141 public void Update(float deltaTime)
142 {
143 if (!VoteRunning) { return; }
144 if (GameMain.GraphicsWidth != createdForResolution.X || GameMain.GraphicsHeight != createdForResolution.Y) { CreateVotingGUI(); }
145 yesVotes = getYesVotes();
146 noVotes = getNoVotes();
147 maxVotes = getMaxVotes();
148 voteCounter.Text = $"({yesVotes + noVotes}/{maxVotes})";
149 timer += deltaTime;
150 votingTimer.BarSize = timer / votingTime;
151 }

163 private void SetSubmarineVotingText(Client starter, SubmarineInfo info, VoteType type)
164 {
165 string name = starter.Name;
166 JobPrefab prefab = starter?.Character?.Info?.Job?.Prefab;
167 Color nameColor = prefab != null ? prefab.UIColor : Color.White;
168 string characterRichString = $"‖color:{nameColor.R},{nameColor.G},{nameColor.B}‖{name}‖color:end‖";
169 string submarineRichString = $"‖color:{SubmarineColor.R},{SubmarineColor.G},{SubmarineColor.B}‖{info.DisplayName}‖color:end‖";
170
171 LocalizedString text = string.Empty;
172 switch (type)
173 {
174 case VoteType.PurchaseAndSwitchSub:
175 text = TextManager.GetWithVariables("submarinepurchaseandswitchvote",
176 ("[playername]", characterRichString),
177 ("[submarinename]", submarineRichString),
178 ("[amount]", info.Price.ToString()),
179 ("[currencyname]", TextManager.Get("credit").ToLower()));
180 break;
181 case VoteType.PurchaseSub:
182 text = TextManager.GetWithVariables("submarinepurchasevote",
183 ("[playername]", characterRichString),
184 ("[submarinename]", submarineRichString),
185 ("[amount]", info.Price.ToString()),
186 ("[currencyname]", TextManager.Get("credit").ToLower()));
187 break;
188 case VoteType.SwitchSub:
189 int deliveryFee = SubmarineSelection.DeliveryFeePerDistanceTravelled * GameMain.GameSession.Map.DistanceToClosestLocationWithOutpost(GameMain.GameSession.Map.CurrentLocation, out Location endLocation);
190 if (deliveryFee > 0)
191 {
192 text = TextManager.GetWithVariables("submarineswitchfeevote",
193 ("[playername]", characterRichString),
194 ("[submarinename]", submarineRichString),
195 ("[locationname]", endLocation.Name),
196 ("[amount]", deliveryFee.ToString()),
197 ("[currencyname]", TextManager.Get("credit").ToLower()));
198 }
199 else
200 {
201 text = TextManager.GetWithVariables("submarineswitchnofeevote",
202 ("[playername]", characterRichString),
203 ("[submarinename]", submarineRichString));
204 }
205 break;
206 }
207
208 votingOnText = RichString.Rich(text);
209 }

234 ("[currencyname]", TextManager.Get("credit").ToLower()),
235 ("[yesvotecount]", yesVoteCount.ToString()),
236 ("[novotecount]", noVoteCount.ToString()));

247 ("[currencyname]", TextManager.Get("credit").ToLower()),

254 ("[submarinename]", info.DisplayName),
255 ("[yesvotecount]", yesVoteCount.ToString()),
