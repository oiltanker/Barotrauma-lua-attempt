100 if (GameMain.Client == null || !lastReceivedState.HasValue) { return; }
101 //reset to last known server state after the state hasn't changed in 1.0 seconds client-side

103 if (resetPredictionCoroutine == null || !CoroutineManager.IsCoroutineRunning(resetPredictionCoroutine))
104 {

112 private IEnumerable<CoroutineStatus> ResetPredictionAfterDelay()
113 {
114 while (resetPredictionTimer > 0.0f)
115 {
116 resetPredictionTimer -= CoroutineManager.DeltaTime;
117 yield return CoroutineStatus.Running;
118 }
119 if (lastReceivedState.HasValue) { IsActive = lastReceivedState.Value; }
120 resetPredictionCoroutine = null;
121 yield return CoroutineStatus.Success;
122 }
