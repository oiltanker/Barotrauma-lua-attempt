205 public override void Update(float deltaTime, Camera cam)
206 {
207#if SERVER

216 (autoTemp ? ", Autotemp ON" : ", Autotemp OFF"),
217 ServerLog.MessageType.ItemInteraction);
218
219 nextServerLogWriteTime = null;
220 lastServerLogWriteTime = (float)Timing.TotalTime;
221 }
222 }
223#endif
224
225 //if an AI character was using the item on the previous frame but not anymore, turn autotemp on
226 // (= bots turn autotemp back on when leaving the reactor)
227 if (LastAIUser != null)
228 {
229 if (LastAIUser.SelectedConstruction != item && LastAIUser.CanInteractWith(item))

237#if CLIENT

244 prevAvailableFuel = AvailableFuel;
245 ApplyStatusEffects(ActionType.OnActive, deltaTime, null);
246
247 //use a smoothed "correct output" instead of the actual correct output based on the load
248 //so the player doesn't have to keep adjusting the rate impossibly fast when the load fluctuates heavily
249 if (!MathUtils.NearlyEqual(MaxPowerOutput, 0.0f))
250 {
251 CorrectTurbineOutput += MathHelper.Clamp((Load / MaxPowerOutput * 100.0f) - CorrectTurbineOutput, -20.0f, 20.0f) * deltaTime;
252 }
253
254 //calculate tolerances of the meters based on the skills of the user
255 //more skilled characters have larger "sweet spots", making it easier to keep the power output at a suitable level
256 float tolerance = MathHelper.Lerp(2.5f, 10.0f, degreeOfSuccess);
257 optimalTurbineOutput = new Vector2(CorrectTurbineOutput - tolerance, CorrectTurbineOutput + tolerance);
258 tolerance = MathHelper.Lerp(5.0f, 20.0f, degreeOfSuccess);
259 allowedTurbineOutput = new Vector2(CorrectTurbineOutput - tolerance, CorrectTurbineOutput + tolerance);
260
261 optimalTemperature = Vector2.Lerp(new Vector2(40.0f, 60.0f), new Vector2(30.0f, 70.0f), degreeOfSuccess);
262 allowedTemperature = Vector2.Lerp(new Vector2(30.0f, 70.0f), new Vector2(10.0f, 90.0f), degreeOfSuccess);
263
264 optimalFissionRate = Vector2.Lerp(new Vector2(30, AvailableFuel - 20), new Vector2(20, AvailableFuel - 10), degreeOfSuccess);
265 optimalFissionRate.X = Math.Min(optimalFissionRate.X, optimalFissionRate.Y - 10);
266 allowedFissionRate = Vector2.Lerp(new Vector2(20, AvailableFuel), new Vector2(10, AvailableFuel), degreeOfSuccess);
267 allowedFissionRate.X = Math.Min(allowedFissionRate.X, allowedFissionRate.Y - 10);
268
269 float heatAmount = GetGeneratedHeat(fissionRate);
270
271 float temperatureDiff = (heatAmount - turbineOutput) - Temperature;
272 Temperature += MathHelper.Clamp(Math.Sign(temperatureDiff) * 10.0f * deltaTime, -Math.Abs(temperatureDiff), Math.Abs(temperatureDiff));
273 //if (item.InWater && AvailableFuel < 100.0f) Temperature -= 12.0f * deltaTime;
274
275 FissionRate = MathHelper.Lerp(fissionRate, Math.Min(TargetFissionRate, AvailableFuel), deltaTime);
276
277 TurbineOutput = MathHelper.Lerp(turbineOutput, TargetTurbineOutput, deltaTime);
278
279 float temperatureFactor = Math.Min(temperature / 50.0f, 1.0f);
280
281 if (!PowerOn)
282 {
283 TargetFissionRate = 0.0f;
284 TargetTurbineOutput = 0.0f;
285 }
286 else if (autoTemp)

311 }
312 }

379 public override PowerRange MinMaxPowerOut(Connection conn, float load)
380 {
381 float tolerance = 1f;
382
383 //If within the optimal output allow for slight output adjustments
384 if (turbineOutput > optimalTurbineOutput.X && turbineOutput < optimalTurbineOutput.Y &&
385 temperature > optimalTemperature.X && temperature < optimalTemperature.Y)
386 {
387 tolerance = 3f;
388 }
389
390 float temperatureFactor = Math.Min(temperature / 50.0f, 1.0f);
391 float minOutput = MaxPowerOutput * Math.Clamp(Math.Min((turbineOutput - tolerance) / 100.0f, temperatureFactor), 0, 1);
392 float maxOutput = MaxPowerOutput * Math.Min((turbineOutput + tolerance) / 100.0f, temperatureFactor);
393
394 minUpdatePowerOut = minOutput;
395 maxUpdatePowerOut = maxOutput;
396
397 float reactorMax = PowerOn ? MaxPowerOutput : maxUpdatePowerOut;
398
399 return new PowerRange(minOutput, maxOutput, reactorMax);
400 }

405 public override float GetConnectionPowerOut(Connection conn, float power, PowerRange minMaxPower, float load)
406 {
407 //Load must be calculated at this stage instead of at gridResolved to remove influence of lower priority devices
408 float loadLeft = MathHelper.Max(load - power,0);
409 float expectedPower = MathHelper.Clamp(loadLeft, minMaxPower.Min, minMaxPower.Max);
410
411 //Delta ratio of Min and Max power output capability of the grid
412 float ratio = MathHelper.Max((loadLeft - minMaxPower.Min) / (minMaxPower.Max - minMaxPower.Min), 0);
413 if (float.IsInfinity(ratio))
414 {
415 ratio = 0;
416 }
417
418 float output = MathHelper.Clamp(ratio * (maxUpdatePowerOut - minUpdatePowerOut) + minUpdatePowerOut, minUpdatePowerOut, maxUpdatePowerOut);
419 float newLoad = loadLeft;
420
421 //Adjust behaviour for multi reactor setup
422 if (MaxPowerOutput != minMaxPower.ReactorMaxOutput)
423 {
424 float idealLoad = MaxPowerOutput / minMaxPower.ReactorMaxOutput * loadLeft;
425 float loadAdjust = MathHelper.Clamp((ratio - 0.5f) * 25 + idealLoad - (turbineOutput / 100 * MaxPowerOutput), -MaxPowerOutput / 100, MaxPowerOutput / 100);
426 newLoad = MathHelper.Clamp(loadLeft - (expectedPower - output) + loadAdjust, 0, loadLeft);
427 }
428
429 if (float.IsNegative(newLoad))
430 {
431 newLoad = 0.0f;
432 }
433
434 Load = newLoad;
435 currPowerConsumption = -output;
436 return output;
437 }

439 private float GetGeneratedHeat(float fissionRate)
440 {
441 return fissionRate * (prevAvailableFuel / 100.0f) * 2.0f;
442 }

449 private bool NeedMoreFuel(float minimumOutputRatio, float minCondition = 0)
450 {
451 float remainingFuel = item.ContainedItems.Sum(i => i.Condition);
452 if (remainingFuel <= minCondition && Load > 0.0f)
453 {
454 return true;
455 }
456
457 //fission rate is clamped to the amount of available fuel
458 float maxFissionRate = Math.Min(prevAvailableFuel, 100.0f);
459 if (maxFissionRate >= 100.0f) { return false; }
460
461 float maxTurbineOutput = 100.0f;
462
463 //calculate the maximum output if the fission rate is cranked as high as it goes and turbine output is at max
464 float theoreticalMaxHeat = GetGeneratedHeat(fissionRate: maxFissionRate);
465 float temperatureFactor = Math.Min(theoreticalMaxHeat / 50.0f, 1.0f);
466 float theoreticalMaxOutput = Math.Min(maxTurbineOutput / 100.0f, temperatureFactor) * MaxPowerOutput;
467
468 //maximum output not enough, we need more fuel
469 return theoreticalMaxOutput < Load * minimumOutputRatio;
470 }

472 private bool TooMuchFuel()
473 {
474 var containedItems = item.OwnInventory?.AllItems;
475 if (containedItems != null && containedItems.Count() <= 1) { return false; }
476
477 //get the amount of heat we'd generate if the fission rate was at the low end of the optimal range
478 float minimumHeat = GetGeneratedHeat(optimalFissionRate.X);
479
480 //if we need a very high turbine output to keep the engine from overheating, there's too much fuel
481 return minimumHeat > Math.Min(CorrectTurbineOutput * 1.5f, 90);
482 }

484 private void UpdateFailures(float deltaTime)
485 {
486 if (temperature > allowedTemperature.Y)
487 {
488 item.SendSignal("1", "meltdown_warning");
489 //faster meltdown if the item is in a bad condition
490 meltDownTimer += MathHelper.Lerp(deltaTime * 2.0f, deltaTime, item.Condition / item.MaxCondition);
491
492 if (meltDownTimer > MeltdownDelay)
493 {
494 MeltDown();
495 return;
496 }
497 }
498 else
499 {
500 item.SendSignal("0", "meltdown_warning");
501 meltDownTimer = Math.Max(0.0f, meltDownTimer - deltaTime);
502 }
503
504 if (temperature > optimalTemperature.Y)
505 {
506 fireTimer += MathHelper.Lerp(deltaTime * 2.0f, deltaTime, item.Condition / item.MaxCondition);
507#if SERVER
508 if (fireTimer > Math.Min(5.0f, FireDelay / 2) && blameOnBroken?.Character?.SelectedConstruction == item)
509 {
510 GameMain.Server.KarmaManager.OnReactorOverHeating(item, blameOnBroken.Character, deltaTime);
511 }
512#endif
513 if (fireTimer >= FireDelay)
514 {
515 new FireSource(item.WorldPosition);
516 fireTimer = 0.0f;
517 }
518 }
519 else
520 {
521 fireTimer = Math.Max(0.0f, fireTimer - deltaTime);
522 }
523 }

525 public void UpdateAutoTemp(float speed, float deltaTime)
526 {
527 float desiredTurbineOutput = (optimalTurbineOutput.X + optimalTurbineOutput.Y) / 2.0f;
528 TargetTurbineOutput += MathHelper.Clamp(desiredTurbineOutput - TargetTurbineOutput, -speed, speed) * deltaTime;
529 TargetTurbineOutput = MathHelper.Clamp(TargetTurbineOutput, 0.0f, 100.0f);
530
531 float desiredFissionRate = (optimalFissionRate.X + optimalFissionRate.Y) / 2.0f;
532 TargetFissionRate += MathHelper.Clamp(desiredFissionRate - TargetFissionRate, -speed, speed) * deltaTime;
533
534 if (temperature > (optimalTemperature.X + optimalTemperature.Y) / 2.0f)
535 {
536 TargetFissionRate = Math.Min(TargetFissionRate - speed * 2 * deltaTime, allowedFissionRate.Y);
537 }
538 else if (-currPowerConsumption < Load)
539 {
540 TargetFissionRate = Math.Min(TargetFissionRate + speed * 2 * deltaTime, 100.0f);
541 }
542 TargetFissionRate = MathHelper.Clamp(TargetFissionRate, 0.0f, 100.0f);
543
544 //don't push the target too far from the current fission rate
545 //otherwise we may "overshoot", cranking the target fission rate all the way up because it takes a while
546 //for the actual fission rate and temperature to follow
547 TargetFissionRate = MathHelper.Clamp(TargetFissionRate, FissionRate - 5, FissionRate + 5);
548 }

580 private void MeltDown()
581 {
582 if (item.Condition <= 0.0f) { return; }
583 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return; }
584
585 if (!ExplosionDamagesOtherSubs && (statusEffectLists?.ContainsKey(ActionType.OnBroken) ?? false))
586 {
587 foreach (var statusEffect in statusEffectLists[ActionType.OnBroken])
588 {
589 foreach (Explosion explosion in statusEffect.Explosions)
590 {
591 foreach (Submarine sub in Submarine.Loaded)
592 {
593 if (sub != item.Submarine) { explosion.IgnoredSubmarines.Add(sub); }
594 }
595 }
596 }
597 }
598
599 item.Condition = 0.0f;
600 fireTimer = 0.0f;
601 meltDownTimer = 0.0f;
602
603 var containedItems = item.OwnInventory?.AllItems;
604 if (containedItems != null)
605 {
606 foreach (Item containedItem in containedItems)
607 {
608 containedItem.Condition = 0.0f;
609 }
610 }
611#if SERVER
612 GameServer.Log("Reactor meltdown!", ServerLog.MessageType.ItemInteraction);
613 if (GameMain.Server != null)
614 {
615 GameMain.Server.KarmaManager.OnReactorMeltdown(item, blameOnBroken?.Character);
616 }
617#endif

625 public override bool AIOperate(float deltaTime, Character character, AIObjectiveOperateItem objective)
626 {
627 if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient) { return false; }
628 character.AIController.SteeringManager.Reset();
629 bool shutDown = objective.Option == "shutdown";
630
631 IsActive = true;
632
633 if (!shutDown)
634 {
635 float degreeOfSuccess = Math.Min(DegreeOfSuccess(character), 1.0f);
636 float refuelLimit = 0.3f;
637 //characters with insufficient skill levels don't refuel the reactor
638 if (degreeOfSuccess > refuelLimit)
639 {
640 if (aiUpdateTimer > 0.0f)
641 {
642 aiUpdateTimer -= deltaTime;
643 return false;
644 }
645 aiUpdateTimer = AIUpdateInterval;
646 // load more fuel if the current maximum output is only 50% of the current load
647 // or if the fuel rod is (almost) deplenished
648 float minCondition = fuelConsumptionRate * MathUtils.Pow2((degreeOfSuccess - refuelLimit) * 2);
649 if (NeedMoreFuel(minimumOutputRatio: 0.5f, minCondition: minCondition))
650 {
651 bool outOfFuel = false;
652 var container = item.GetComponent<ItemContainer>();
653 if (objective.SubObjectives.None())
654 {
655 var containObjective = AIContainItems<Reactor>(container, character, objective, itemCount: 1, equip: true, removeEmpty: true, spawnItemIfNotFound: !character.IsOnPlayerTeam, dropItemOnDeselected: true);
656 containObjective.Completed += ReportFuelRodCount;
657 containObjective.Abandoned += ReportFuelRodCount;
658 character.Speak(TextManager.Get("DialogReactorFuel").Value, null, 0.0f, "reactorfuel".ToIdentifier(), 30.0f);
659
660 void ReportFuelRodCount()
661 {
662 if (!character.IsOnPlayerTeam) { return; }
663 if (character.Submarine != Submarine.MainSub) { return; }
664 int remainingFuelRods = Submarine.MainSub.GetItems(false).Count(i => i.HasTag("reactorfuel") && i.Condition > 1);
665 if (remainingFuelRods == 0)
666 {
667 character.Speak(TextManager.Get("DialogOutOfFuelRods").Value, null, 0.0f, "outoffuelrods".ToIdentifier(), 30.0f);
668 outOfFuel = true;
669 }
670 else if (remainingFuelRods < 3)
671 {
672 character.Speak(TextManager.Get("DialogLowOnFuelRods").Value, null, 0.0f, "lowonfuelrods".ToIdentifier(), 30.0f);
673 }
674 }
675 }
676 return outOfFuel;
677 }
678 else
679 {
680 if (Item.ConditionPercentage <= 0 && AIObjectiveRepairItems.IsValidTarget(Item, character))
681 {
682 if (Item.Repairables.Average(r => r.DegreeOfSuccess(character)) > 0.4f)
683 {
684 objective.AddSubObjective(new AIObjectiveRepairItem(character, Item, objective.objectiveManager, isPriority: true));
685 return false;
686 }
687 else
688 {
689 character.Speak(TextManager.Get("DialogReactorIsBroken").Value, identifier: "reactorisbroken".ToIdentifier(), minDurationBetweenSimilar: 30.0f);
690 }
691 }
692 if (TooMuchFuel())
693 {
694 DropFuel(minCondition: 0.1f, maxCondition: 100);
695 }
696 else
697 {
698 DropFuel(minCondition: 0, maxCondition: 0);
699 }
700 }
701 }
702 }
703
704 if (objective.Override)
705 {
706 if (lastUser != null && lastUser != character && lastUser != LastAIUser)
707 {
708 if (lastUser.SelectedConstruction == item && character.IsOnPlayerTeam)
709 {
710 character.Speak(TextManager.Get("DialogReactorTaken").Value, null, 0.0f, "reactortaken".ToIdentifier(), 10.0f);
711 }
712 }
713 }
714 else if (LastUserWasPlayer && lastUser != null && lastUser.TeamID == character.TeamID)
715 {
716 return true;
717 }
718
719 LastUser = LastAIUser = character;
720
721 bool prevAutoTemp = autoTemp;
722 bool prevPowerOn = _powerOn;
723 float prevFissionRate = TargetFissionRate;
724 float prevTurbineOutput = TargetTurbineOutput;
725
726 if (shutDown)
727 {
728 PowerOn = false;
729 AutoTemp = false;
730 TargetFissionRate = 0.0f;
731 TargetTurbineOutput = 0.0f;
732 unsentChanges = true;
733 return true;
734 }
735 else
736 {
737 PowerOn = true;
738 if (objective.Override || !autoTemp)
739 {
740 //characters with insufficient skill levels simply set the autotemp on instead of trying to adjust the temperature manually
741 if (degreeOfSuccess < 0.5f)
742 {
743 AutoTemp = true;
744 }
745 else
746 {
747 AutoTemp = false;
748 UpdateAutoTemp(MathHelper.Lerp(0.5f, 2.0f, degreeOfSuccess), 1.0f);
749 }
750 }
751#if CLIENT
752 FissionRateScrollBar.BarScroll = FissionRate / 100.0f;
753 TurbineOutputScrollBar.BarScroll = TurbineOutput / 100.0f;
754#endif
755 if (autoTemp != prevAutoTemp ||
756 prevPowerOn != _powerOn ||
757 Math.Abs(prevFissionRate - TargetFissionRate) > 1.0f ||
758 Math.Abs(prevTurbineOutput - TargetTurbineOutput) > 1.0f)
759 {
760 unsentChanges = true;
761 }
762 aiUpdateTimer = AIUpdateInterval;
763 return false;
764 }
