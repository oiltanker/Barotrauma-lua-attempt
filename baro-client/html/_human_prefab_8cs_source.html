113 public void InitializeCharacter(Character npc, ISpatialEntity positionToStayIn = null)
114 {
115 npc.AddStaticHealthMultiplier(HealthMultiplier);
116 if (GameMain.NetworkMember != null)
117 {
118 npc.AddStaticHealthMultiplier(HealthMultiplierInMultiplayer);
119 }
120
121 var humanAI = npc.AIController as HumanAIController;
122 if (humanAI != null)
123 {
124 var idleObjective = humanAI.ObjectiveManager.GetObjective<AIObjectiveIdle>();
125 if (positionToStayIn != null && Behavior == AIObjectiveIdle.BehaviorType.StayInHull)
126 {
127 idleObjective.TargetHull = AIObjectiveGoTo.GetTargetHull(positionToStayIn);
128 idleObjective.Behavior = AIObjectiveIdle.BehaviorType.StayInHull;
129 }
130 else
131 {
132 idleObjective.Behavior = Behavior;
133 foreach (Identifier moduleType in PreferredOutpostModuleTypes)
134 {
135 idleObjective.PreferredOutpostModuleTypes.Add(moduleType);
136 }
137 }
138 humanAI.ReportRange = ReportRange;
139 humanAI.AimSpeed = AimSpeed;
140 humanAI.AimAccuracy = AimAccuracy;
141 }
142 if (CampaignInteractionType != CampaignMode.InteractionType.None)
143 {
144 (GameMain.GameSession.GameMode as CampaignMode)?.AssignNPCMenuInteraction(npc, CampaignInteractionType);
145 if (positionToStayIn != null && humanAI != null)
146 {
147 humanAI.ObjectiveManager.SetForcedOrder(new AIObjectiveGoTo(positionToStayIn, npc, humanAI.ObjectiveManager, repeat: true, getDivingGearIfNeeded: false, closeEnough: 200)
148 {
149 DebugLogWhenFails = false
150 });
151 }
152 }
153 }

174 public static void InitializeItem(Character character, XElement itemElement, Submarine submarine, HumanPrefab humanPrefab, Item parentItem = null, bool createNetworkEvents = true)
175 {
176 ItemPrefab itemPrefab;
177 string itemIdentifier = itemElement.GetAttributeString("identifier", "");
178 itemPrefab = MapEntityPrefab.FindByIdentifier(itemIdentifier.ToIdentifier()) as ItemPrefab;
179 if (itemPrefab == null)
180 {
181 DebugConsole.ThrowError("Tried to spawn \"" + humanPrefab?.Identifier + "\" with the item \"" + itemIdentifier + "\". Matching item prefab not found.");
182 return;
183 }
184 Item item = new Item(itemPrefab, character.Position, null);
185#if SERVER
186 if (GameMain.Server != null && Entity.Spawner != null && createNetworkEvents)
187 {
188 if (GameMain.Server.EntityEventManager.UniqueEvents.Any(ev => ev.Entity == item))
189 {
190 string errorMsg = $"Error while spawning job items. Item {item.Name} created network events before the spawn event had been created.";
191 DebugConsole.ThrowError(errorMsg);
192 GameAnalyticsManager.AddErrorEventOnce("Job.InitializeJobItem:EventsBeforeSpawning", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
193 GameMain.Server.EntityEventManager.UniqueEvents.RemoveAll(ev => ev.Entity == item);
194 GameMain.Server.EntityEventManager.Events.RemoveAll(ev => ev.Entity == item);
195 }
196
197 Entity.Spawner.CreateNetworkEvent(new EntitySpawner.SpawnEntity(item));
198 }
199#endif
200 if (itemElement.GetAttributeBool("equip", false))
201 {
202 //if the item is both pickable and wearable, try to wear it instead of picking it up
203 List<InvSlotType> allowedSlots =
204 item.GetComponents<Pickable>().Count() > 1 ?
205 new List<InvSlotType>(item.GetComponent<Wearable>()?.AllowedSlots ?? item.GetComponent<Pickable>().AllowedSlots) :
206 new List<InvSlotType>(item.AllowedSlots);
207 allowedSlots.Remove(InvSlotType.Any);
208
209 character.Inventory.TryPutItem(item, null, allowedSlots);
210 }
211 else
212 {
213 character.Inventory.TryPutItem(item, null, item.AllowedSlots);
214 }
215 IdCard idCardComponent = item.GetComponent<IdCard>();
216 if (idCardComponent != null)
217 {
218 idCardComponent.Initialize(null, character);
219 if (submarine != null && (submarine.Info.IsWreck || submarine.Info.IsOutpost))
220 {
221 idCardComponent.SubmarineSpecificID = submarine.SubmarineSpecificIDTag;
222 }
223
224 var idCardTags = itemElement.GetAttributeStringArray("tags", Array.Empty<string>());
225 foreach (string tag in idCardTags)
226 {
227 item.AddTag(tag);
228 }
229 }
230
231 foreach (WifiComponent wifiComponent in item.GetComponents<WifiComponent>())
232 {
233 wifiComponent.TeamID = character.TeamID;
234 }
235 parentItem?.Combine(item, user: null);
236 foreach (XElement childItemElement in itemElement.Elements())
237 {
238 InitializeItem(character, childItemElement, submarine, humanPrefab, item, createNetworkEvents);
239 }
240 }
