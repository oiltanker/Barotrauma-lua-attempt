34 protected override float GetPriority()
35 {
36 if (!IsAllowed)
37 {
38 Priority = 0;
39 Abandon = true;
40 }
41 else if (HumanAIController.IsTrueForAnyCrewMember(
42 other => other != HumanAIController &&
43 other.Character.IsBot &&
44 other.ObjectiveManager.GetActiveObjective<AIObjectiveFixLeaks>() is AIObjectiveFixLeaks fixLeaks &&
45 fixLeaks.SubObjectives.Any(so => so is AIObjectiveFixLeak fixObjective && fixObjective.Leak == Leak)))
46 {
47 Priority = 0;
48 }
49 else
50 {
51 float xDist = Math.Abs(character.WorldPosition.X - Leak.WorldPosition.X);
52 float yDist = Math.Abs(character.WorldPosition.Y - Leak.WorldPosition.Y);
53 // Vertical distance matters more than horizontal (climbing up/down is harder than moving horizontally).
54 // If the target is close, ignore the distance factor alltogether so that we keep fixing the leaks that are nearby.
55 float distanceFactor = isPriority || xDist < 200 && yDist < 100 ? 1 : MathHelper.Lerp(1, 0.1f, MathUtils.InverseLerp(0, 3000, xDist + yDist * 3.0f));
56 float severity = isPriority ? 1 : AIObjectiveFixLeaks.GetLeakSeverity(Leak) / 100;
57 float reduction = isPriority ? 1 : 2;
58 float max = AIObjectiveManager.LowestOrderPriority - reduction;
59 float devotion = CumulatedDevotion / 100;
60 Priority = MathHelper.Lerp(0, max, MathHelper.Clamp(devotion + (severity * distanceFactor * PriorityModifier), 0, 1));
61 }
62 return Priority;
63 }

65 protected override void Act(float deltaTime)
66 {
67 var weldingTool = character.Inventory.FindItemByTag("weldingequipment".ToIdentifier(), true);
68 if (weldingTool == null)
69 {
70 TryAddSubObjective(ref getWeldingTool, () => new AIObjectiveGetItem(character, "weldingequipment".ToIdentifier(), objectiveManager, equip: true, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC),
71 onAbandon: () =>
72 {
73 if (character.IsOnPlayerTeam && objectiveManager.IsCurrentOrder<AIObjectiveFixLeaks>())
74 {
75 character.Speak(TextManager.Get("dialogcannotfindweldingequipment").Value, null, 0.0f, "dialogcannotfindweldingequipment".ToIdentifier(), 10.0f);
76 }
77 Abandon = true;
78 },
79 onCompleted: () => RemoveSubObjective(ref getWeldingTool));
80 return;
81 }
82 else
83 {
84 if (weldingTool.OwnInventory == null)
85 {
86#if DEBUG
87 DebugConsole.ThrowError($"{character.Name}: AIObjectiveFixLeak failed - the item \"" + weldingTool + "\" has no proper inventory");
88#endif
89 Abandon = true;
90 return;
91 }
92 if (weldingTool.OwnInventory != null && weldingTool.OwnInventory.AllItems.None(i => i.HasTag("weldingfuel") && i.Condition > 0.0f))
93 {
94 TryAddSubObjective(ref refuelObjective, () => new AIObjectiveContainItem(character, "weldingfuel".ToIdentifier(), weldingTool.GetComponent<ItemContainer>(), objectiveManager, spawnItemIfNotFound: character.TeamID == CharacterTeamType.FriendlyNPC)
95 {
96 RemoveExisting = true
97 },
98 onAbandon: () =>
99 {
100 Abandon = true;
101 ReportWeldingFuelTankCount();
102 },
103 onCompleted: () =>
104 {
105 RemoveSubObjective(ref refuelObjective);
106 ReportWeldingFuelTankCount();
107 });
108
109 void ReportWeldingFuelTankCount()
110 {
111 if (character.Submarine != Submarine.MainSub) { return; }
112 int remainingOxygenTanks = Submarine.MainSub.GetItems(false).Count(i => i.HasTag("weldingfuel") && i.Condition > 1);
113 if (remainingOxygenTanks == 0)
114 {
115 character.Speak(TextManager.Get("DialogOutOfWeldingFuel").Value, null, 0.0f, "outofweldingfuel".ToIdentifier(), 30.0f);
116 }
117 else if (remainingOxygenTanks < 4)
118 {
119 character.Speak(TextManager.Get("DialogLowOnWeldingFuel").Value, null, 0.0f, "lowonweldingfuel".ToIdentifier(), 30.0f);
120 }
121 }
122 return;
123 }
124 }
125 if (subObjectives.Any()) { return; }
126 var repairTool = weldingTool.GetComponent<RepairTool>();
127 if (repairTool == null)
128 {
129#if DEBUG
130 DebugConsole.ThrowError($"{character.Name}: AIObjectiveFixLeak failed - the item \"" + weldingTool + "\" has no RepairTool component but is tagged as a welding tool");
131#endif
132 Abandon = true;
133 return;
134 }
135 Vector2 toLeak = Leak.WorldPosition - character.AnimController.AimSourceWorldPos;
136 // TODO: use the collider size/reach?
137 if (!character.AnimController.InWater && Math.Abs(toLeak.X) < 100 && toLeak.Y < 0.0f && toLeak.Y > -150)
138 {
139 HumanAIController.AnimController.Crouching = true;
140 }
141 float reach = CalculateReach(repairTool, character);
142 bool canOperate = toLeak.LengthSquared() < reach * reach;
143 if (canOperate)
144 {
145 TryAddSubObjective(ref operateObjective, () => new AIObjectiveOperateItem(repairTool, character, objectiveManager, option: Identifier.Empty, requireEquip: true, operateTarget: Leak),
146 onAbandon: () => Abandon = true,
147 onCompleted: () =>
148 {
149 if (CheckObjectiveSpecific()) { IsCompleted = true; }
150 else
151 {
152 // Failed to operate. Probably too far.
153 Abandon = true;
154 }
155 });
156 }
157 else
158 {
159 TryAddSubObjective(ref gotoObjective, () => new AIObjectiveGoTo(Leak, character, objectiveManager)
160 {
161 UseDistanceRelativeToAimSourcePos = true,
162 CloseEnough = reach,
163 DialogueIdentifier = Leak.FlowTargetHull != null ? "dialogcannotreachleak".ToIdentifier() : Identifier.Empty,
164 TargetName = Leak.FlowTargetHull?.DisplayName,
165 requiredCondition = () =>
166 Leak.Submarine == character.Submarine &&
167 Leak.linkedTo.Any(e => e is Hull h && character.CurrentHull == h),
168 // The Go To objective can be abandoned if the leak is fixed (in which case we don't want to use the dialogue)
169 SpeakCannotReachCondition = () => !CheckObjectiveSpecific()
170 },
171 onAbandon: () =>
172 {
173 if (CheckObjectiveSpecific()) { IsCompleted = true; }
174 else if ((Leak.WorldPosition - character.AnimController.AimSourceWorldPos).LengthSquared() > MathUtils.Pow(reach * 2, 2))
175 {
176 // Too far
177 Abandon = true;
178 }
179 else
180 {
181 // We are close, try again.
182 RemoveSubObjective(ref gotoObjective);
183 }
184 },
185 onCompleted: () => RemoveSubObjective(ref gotoObjective));
186 }

198 public static float CalculateReach(RepairTool repairTool, Character character)
199 {
200 float armLength = ConvertUnits.ToDisplayUnits(((HumanoidAnimController)character.AnimController).ArmLength);
201 // This is an approximation, because we don't know the exact reach until the pose is taken.
202 // And even then the actual range depends on the direction we are aiming to.
203 // Found out that without any multiplier the value (209) is often too short.
204 return repairTool.Range + armLength * 1.3f;
205 }
