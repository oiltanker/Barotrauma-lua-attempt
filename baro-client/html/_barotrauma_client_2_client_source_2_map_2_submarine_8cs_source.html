38 public static void CullEntities(Camera cam)
39 {
40 visibleSubs.Clear();
41 foreach (Submarine sub in Loaded)
42 {
43 if (Level.Loaded != null && sub.WorldPosition.Y < Level.MaxEntityDepth) { continue; }
44
45 int margin = 500;
46 Rectangle worldBorders = new Rectangle(
47 sub.VisibleBorders.X + (int)sub.WorldPosition.X - margin,
48 sub.VisibleBorders.Y + (int)sub.WorldPosition.Y + margin,
49 sub.VisibleBorders.Width + margin * 2,
50 sub.VisibleBorders.Height + margin * 2);
51
52 if (RectsOverlap(worldBorders, cam.WorldView))
53 {
54 visibleSubs.Add(sub);
55 }
56 }
57
58 if (visibleEntities == null)
59 {
60 visibleEntities = new List<MapEntity>(MapEntity.mapEntityList.Count);
61 }
62 else
63 {
64 visibleEntities.Clear();
65 }
66
67 Rectangle worldView = cam.WorldView;
68 foreach (MapEntity entity in MapEntity.mapEntityList)
69 {
70 if (entity.Submarine != null)
71 {
72 if (!visibleSubs.Contains(entity.Submarine)) { continue; }
73 }
74
75 if (entity.IsVisible(worldView)) { visibleEntities.Add(entity); }
76 }

138 public static void DrawDamageable(SpriteBatch spriteBatch, Effect damageEffect, bool editing = false, Predicate<MapEntity> predicate = null)
139 {
140 var entitiesToRender = !editing && visibleEntities != null ? visibleEntities : MapEntity.mapEntityList;
141
142 depthSortedDamageable.Clear();
143
144 //insertion sort according to draw depth
145 foreach (MapEntity e in entitiesToRender)
146 {
147 if (e is Structure structure && structure.DrawDamageEffect)
148 {
149 if (predicate != null)
150 {
151 if (!predicate(e)) continue;
152 }
153 float drawDepth = structure.GetDrawDepth();
154 int i = 0;
155 while (i < depthSortedDamageable.Count)
156 {
157 float otherDrawDepth = depthSortedDamageable[i].GetDrawDepth();
158 if (otherDrawDepth < drawDepth) { break; }
159 i++;
160 }
161 depthSortedDamageable.Insert(i, structure);
162 }
163 }
164
165 foreach (Structure s in depthSortedDamageable)
166 {
167 s.DrawDamage(spriteBatch, damageEffect, editing);
168 }
169 if (damageEffect != null)
170 {
171 damageEffect.Parameters["aCutoff"].SetValue(0.0f);
172 damageEffect.Parameters["cCutoff"].SetValue(0.0f);
173 DamageEffectCutoff = 0.0f;
174 }

214 public static void DrawGrid(SpriteBatch spriteBatch, int gridCells, Vector2 gridCenter, Vector2 roundedGridCenter, float alpha = 1.0f)
215 {
216 Vector2 topLeft = roundedGridCenter - Vector2.One * GridSize * gridCells / 2;
217 Vector2 bottomRight = roundedGridCenter + Vector2.One * GridSize * gridCells / 2;
218
219 for (int i = 0; i < gridCells; i++)
220 {
221 float distFromGridX = (MathUtils.RoundTowardsClosest(gridCenter.X, GridSize.X) - gridCenter.X) / GridSize.X;
222 float distFromGridY = (MathUtils.RoundTowardsClosest(gridCenter.Y, GridSize.Y) - gridCenter.Y) / GridSize.Y;
223
224 float normalizedDistX = Math.Abs(i + distFromGridX - gridCells / 2) / (gridCells / 2);
225 float normalizedDistY = Math.Abs(i - distFromGridY - gridCells / 2) / (gridCells / 2);
226
227 float expandX = MathHelper.Lerp(30.0f, 0.0f, normalizedDistY);
228 float expandY = MathHelper.Lerp(30.0f, 0.0f, normalizedDistX);
229
230 GUI.DrawLine(spriteBatch,
231 new Vector2(topLeft.X - expandX, -bottomRight.Y + i * GridSize.Y),
232 new Vector2(bottomRight.X + expandX, -bottomRight.Y + i * GridSize.Y),
233 Color.White * (1.0f - normalizedDistY) * alpha, depth: 0.6f, width: 3);
234 GUI.DrawLine(spriteBatch,
235 new Vector2(topLeft.X + i * GridSize.X, -topLeft.Y + expandY),
236 new Vector2(topLeft.X + i * GridSize.X, -bottomRight.Y - expandY),
237 Color.White * (1.0f - normalizedDistX) * alpha, depth: 0.6f, width: 3);
238 }

244 {
245 Rectangle worldBorders = GetDockedBorders();
246 worldBorders.Location += WorldPosition.ToPoint();
247
248 //create a container that has the same "aspect ratio" as the sub
249 float aspectRatio = worldBorders.Width / (float)worldBorders.Height;

251
252 float scale = 0.9f;
253
254 GUIFrame hullContainer = new GUIFrame(new RectTransform(
255 (parentAspectRatio > aspectRatio ? new Vector2(aspectRatio / parentAspectRatio, 1.0f) : new Vector2(1.0f, parentAspectRatio / aspectRatio)) * scale,
256 parent.RectTransform, Anchor.Center),
257 style: null)
258 {
259 UserData = "hullcontainer"

261
262 var connectedSubs = GetConnectedSubs();
263
264 HashSet<Hull> hullList = Hull.HullList.Where(hull => hull.Submarine == this || connectedSubs.Contains(hull.Submarine)).Where(hull => !ignoreOutpost || IsEntityFoundOnThisSub(hull, true)).ToHashSet();
265
266 Dictionary<Hull, HashSet<Hull>> combinedHulls = new Dictionary<Hull, HashSet<Hull>>();
267
268 foreach (Hull hull in hullList)
269 {
270 if (combinedHulls.ContainsKey(hull) || combinedHulls.Values.Any(hh => hh.Contains(hull))) { continue; }
271
272 List<Hull> linkedHulls = new List<Hull>();
273 MiniMap.GetLinkedHulls(hull, linkedHulls);
274
275 linkedHulls.Remove(hull);
276
277 foreach (Hull linkedHull in linkedHulls)
278 {
279 if (!combinedHulls.ContainsKey(hull))
280 {
281 combinedHulls.Add(hull, new HashSet<Hull>());
282 }
283
284 combinedHulls[hull].Add(linkedHull);
285 }
286 }
287
288 foreach (Hull hull in hullList)
289 {
290 Vector2 relativeHullPos = new Vector2(
291 (hull.WorldRect.X - worldBorders.X) / (float)worldBorders.Width,
292 (worldBorders.Y - hull.WorldRect.Y) / (float)worldBorders.Height);
293 Vector2 relativeHullSize = new Vector2(hull.Rect.Width / (float)worldBorders.Width, hull.Rect.Height / (float)worldBorders.Height);
294
295 bool hideHull = combinedHulls.ContainsKey(hull) || combinedHulls.Values.Any(hh => hh.Contains(hull));
296
297 if (hideHull) { continue; }
298
299 Color color = Color.DarkCyan * 0.8f;
300
301 var hullFrame = new GUIFrame(new RectTransform(relativeHullSize, hullContainer.RectTransform) { RelativeOffset = relativeHullPos }, style: "MiniMapRoom", color: color)
302 {
303 UserData = hull
304 };
305
306 new GUIFrame(new RectTransform(Vector2.One, hullFrame.RectTransform), style: "ScanLines", color: color);

309 foreach (var (mainHull, linkedHulls) in combinedHulls)
310 {
311 MiniMapHullData data = ConstructLinkedHulls(mainHull, linkedHulls, hullContainer, worldBorders);
312
313 Vector2 relativeHullPos = new Vector2(
314 (data.Bounds.X - worldBorders.X) / worldBorders.Width,
315 (worldBorders.Y - data.Bounds.Y) / worldBorders.Height);

320

323

325 {

334

391 public static MiniMapHullData ConstructLinkedHulls(Hull mainHull, HashSet<Hull> linkedHulls, GUIComponent parent, Rectangle worldBorders)
392 {
393 Rectangle parentRect = parent.Rect;
394
395 Dictionary<Hull, Rectangle> rects = new Dictionary<Hull, Rectangle>();
396 Rectangle worldRect = mainHull.WorldRect;
397 worldRect.Y = -worldRect.Y;
398
399 rects.Add(mainHull, worldRect);
400
401 foreach (Hull hull in linkedHulls)
402 {
403 Rectangle rect = hull.WorldRect;
404 rect.Y = -rect.Y;
405
406 worldRect = Rectangle.Union(worldRect, rect);
407 rects.Add(hull, rect);
408 }
409
410 worldRect.Y = -worldRect.Y;
411
412 List<RectangleF> normalizedRects = new List<RectangleF>();
413 List<Hull> hullRefs = new List<Hull>();
414 foreach (var (hull, rect) in rects)
415 {
416 Rectangle wRect = rect;
417 wRect.Y = -wRect.Y;
418
419 var (posX, posY) = new Vector2(
420 (wRect.X - worldBorders.X) / (float)worldBorders.Width,
421 (worldBorders.Y - wRect.Y) / (float)worldBorders.Height);
422
423 var (scaleX, scaleY) = new Vector2(wRect.Width / (float)worldBorders.Width, wRect.Height / (float)worldBorders.Height);
424
425 RectangleF newRect = new RectangleF(posX * parentRect.Width, posY * parentRect.Height, scaleX * parentRect.Width, scaleY * parentRect.Height);
426
427 normalizedRects.Add(newRect);
428 hullRefs.Add(hull);
429 }
430
431 ImmutableArray<RectangleF> snappedRectangles = ToolBox.SnapRectangles(normalizedRects, treshold: 1);
432
433 List<List<Vector2>> polygon = ToolBox.CombineRectanglesIntoShape(snappedRectangles);
434
435 List<List<Vector2>> scaledPolygon = new List<List<Vector2>>();
436
437 foreach (List<Vector2> list in polygon)
438 {
439 var (polySizeX, polySizeY) = ToolBox.GetPolygonBoundingBoxSize(list);
440 float sizeX = polySizeX - 1f,
441 sizeY = polySizeY - 1f;
442
443 scaledPolygon.Add(ToolBox.ScalePolygon(list, new Vector2(sizeX / polySizeX, sizeY / polySizeY)));
444 }
445
446 return new MiniMapHullData(scaledPolygon, worldRect, parentRect.Size, snappedRectangles, hullRefs.ToImmutableArray());
447 }

450 {

452 List<SubEditorScreen.WarningType> warnings = new List<SubEditorScreen.WarningType>();
453
454 if (!Hull.HullList.Any())

461 }
462

513 warnings.Add(SubEditorScreen.WarningType.NoBallastTag);
514 }
515 }
516 }
517 else if (Info.Type == SubmarineType.OutpostModule)
518 {
519 foreach (Item item in Item.ItemList)
520 {
521 var junctionBox = item.GetComponent<PowerTransfer>();
522 if (junctionBox == null) { continue; }
523 int doorLinks =
524 item.linkedTo.Count(lt => lt is Gap || (lt is Item it2 && it2.GetComponent<Door>() != null)) +
525 Item.ItemList.Count(it2 => it2.linkedTo.Contains(item) && !item.linkedTo.Contains(it2));
526 for (int i = 0; i < item.Connections.Count; i++)
527 {
528 int wireCount = item.Connections[i].Wires.Count(w => w != null);
529 if (doorLinks + wireCount > item.Connections[i].MaxWires)
530 {

533 ("[freeconnectioncount]", (item.Connections[i].MaxWires - wireCount).ToString())).Value);

535 }
536 }
537 }
538 }
539
540 if (Gap.GapList.Any(g => g.linkedTo.Count == 0))
541 {
542 if (!IsWarningSuppressed(SubEditorScreen.WarningType.NonLinkedGaps))
543 {
544 errorMsgs.Add(TextManager.Get("NonLinkedGapsWarning").Value);
545 warnings.Add(SubEditorScreen.WarningType.NonLinkedGaps);
546 }
547 }
548

550
